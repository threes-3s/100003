%{对于出世间法，要精进求索，永不厌倦。}%

@{不壞威儀而能隨俗起神通惠引道於眾生。}@
%{不违背礼仪，能够随顺世俗，起神通，引导众生。}%

@{此義云何答曰。}@
%{这是什么意思呢？}%

@{此義者即是三世諸佛妙。}@
%{这就是三世诸佛的妙法。}%

@{即證者乃智非心所惻問曰。}@
%{证悟的人，是智慧，不是心的作用。}%

@{既常求無念實想者。}@
%{既然要经常寻求无念，实相，}%

@{緣何事。}@
%{为什么呢？}%

@{諸經即說布施持戒人天之福。}@
%{佛经上说，布施，持戒，可以得到人天福报。}%

@{豈不是有念法耶。}@
%{这难道不是有念的法吗？}%

@{因何得不同而有差別。}@
%{为什么会有不同的差别呢？}%

@{令一切學人狐疑不信答曰。}@
%{让很多修行人疑惑，不相信。}%

@{汝莫不信。}@
%{你不要不相信。}%

@{汝不解事。}@
%{你没有理解。}%

@{佛說布施持戒人天之福者。}@
%{佛陀说布施，持戒，可以得到人天福报，}%

@{為眾生多在妄想。}@
%{是因为众生大多处于妄想之中。}%

@{佛無量善巧方便。}@
%{佛陀有无量善巧方便。}%

@{隨眾生意說妄想法。}@
%{根据众生的根性，说妄想法。}%

@{誘引令趣大乘門。}@
%{引导他们趣入大乘法门。}%

@{汝今不信。}@
%{你现在不相信，}%

@{即為引經來證也。}@
%{我就用佛经来证明给你看。}%

@{法華經云。}@
%{《法华经》上说：}%

@{我此九部法隨順眾生說大乘為本。}@
%{我所说的九部法，都是随顺众生，以大乘为根本。}%

@{又說云。}@
%{又说：}%

@{十方佛土中准有一乘法。}@
%{十方佛土中，只有一乘法。}%

@{且以假名字引道於眾生。}@
%{我用假名字，引导众生。}%

@{終不以小乘濟渡於眾生。}@
%{我决不以小乘法，来渡化众生。}%

@{又說云。}@
%{又说：}%

@{勿親近小乘三藏學者。}@
%{不要亲近小乘三藏学者。}%

@{又說云。}@
%{又说：}%

@{唯此一事實餘二即非真。}@
%{只有这一乘法是真实的，其他的两种法，都是假的。}%

@{又諸法無行經云。}@
%{《诸法无行经》上说：}%

@{若人分別戒。}@
%{如果有人执着于戒律，}%

@{是即無有戒。}@
%{实际上就没有戒律。}%

@{若見有戒者。}@
%{如果你认为有戒律，}%

@{是即為失戒。}@
%{实际上就是犯戒。}%

@{據此而言。}@
%{根据这些经文，}%

@{明知佛說究竟門非說人無福。}@
%{我们明白，佛陀说的是究竟法门，并不是说人没有福报。}%

@{只是引眾生令入佛惠。}@
%{只是引导众生，进入佛的智慧。}%

@{過去現在聖人說心得自在心得解脫心得聖人者。}@
%{过去，现在，未来，所有圣人，都说过，得到自在，得到解脱，得到圣人果位，}%

@{此是三世諸佛大印破汝疑也問。}@
%{这就是三世诸佛的大印，来破除你的疑惑。}%

@{諸佛若說一乘。}@
%{佛陀如果只说一乘法，}%

@{教化諸眾生故。}@
%{为了教化众生，}%

@{如今總得悟。}@
%{现在大家都应该得到悟道。}%

@{何須更說惑亂眾生。}@
%{为什么还要说那些迷惑众生的法呢？}%

@{豈不得罪過也答曰。}@
%{难道不会犯过吗？}%

@{汝莫起此意。}@
%{你不要有这样的想法。}%

@{諸佛大慈悲故。}@
%{佛陀大慈悲。}%

@{所為眾生落三惡道者眾。}@
%{因为有很多众生堕落到三恶道。}%

@{所以聞方便門為說六波羅蜜。}@
%{所以佛陀就用方便法门，为他们说六波罗蜜。}%

@{波羅蜜者。}@
%{波罗蜜，}%

@{布施持戒忍褥。}@
%{就是布施，持戒，忍辱。}%

@{行此三事得離三惡道人天來往。}@
%{修持这三事，可以脱离三恶道，不再轮回人天。}%

@{精進禪定智惠。}@
%{精进，禅定，智慧。}%

@{行此三事得離生死苦。}@
%{修持这三事，可以脱离生死苦。}%

@{當來成佛問曰。}@
%{将来就能成佛。}%

@{過去諸佛說三乘。}@
%{过去佛陀说三乘法。}%

@{為當現在諸佛說三乘答曰。}@
%{为什么现在佛陀也要说三乘法呢？}%

@{過去現在未來說佛總說問曰。}@
%{过去，现在，未来，所有的佛陀，都说过三乘法。}%

@{以何義理而得知耶答曰。}@
%{凭什么道理知道呢？}%

@{法華經云。}@
%{《法华经》上说：}%

@{若且讚佛乘。}@
%{如果只赞叹佛乘法，}%

@{眾生沒在苦。}@
%{众生就会沉迷于苦海，}%

@{不能信此法。}@
%{不能相信这个法。}%

@{破法不信。}@
%{毁谤佛法，不相信佛法。}%

@{故墮於三惡道。}@
%{就会堕落到三恶道。}%

@{我寧不說法。}@
%{我宁愿不说法。}%

@{疾入於涅槃。}@
%{迅速进入涅槃。}%

@{尋念過去佛所行方便力。}@
%{我回忆过去佛陀的方便力，}%

@{我今所得道亦應說三乘。}@
%{我今天所得的佛道，也应该说三乘法。}%

@{明知過去諸佛並說三乘引道於眾生令入於一乘問曰。}@
%{我明白，过去佛陀都说过三乘法，引导众生，让他们进入一乘法。}%

@{何者是一乘答曰。}@
%{什么是 一乘法呢？}%

@{心是一乘問曰。}@
%{心就是 一乘法。}%

@{心云何知是一乘答曰。}@
%{心怎么知道是 一乘法呢？}%

@{可見心空無所有。}@
%{能够看到心是空，没有所有，}%

@{即是一乘問曰。}@
%{这就是 一乘法。}%

@{了見心空無所有是一乘者得聖耶答曰。}@
%{能够看到心是空，没有所有，这就是 一乘法，这个人就得到圣果了吗？}%

@{得聖又問。}@
%{得到了圣果。}%

@{有凡耶答曰。}@
%{还有凡夫吗？}%

@{亦有凡問曰。}@
%{也有凡夫。}%

@{凡與聖有異無異答曰。}@
%{凡夫和圣人，有区别吗？}%

@{並無異也。}@
%{没有区别。}%

@{若悟朝凡暮聖。}@
%{如果悟道了，早上是凡夫，晚上是圣人。}%

@{不悟即六道受生問曰。}@
%{如果不能悟道，就在六道轮回中受生。}%

@{今言悟時何物答曰。}@
%{现在你说悟道，悟的是什么东西呢？}%

@{悟心問曰。}@
%{悟的是心。}%

@{凡心聖心是一是二答曰。}@
%{凡夫心和圣人心，是一个还是两个呢？}%

@{是一云何是一答曰。}@
%{如果是一个，怎么是一个呢？}%

@{了見性清淨從本以來無染無著。}@
%{能够看到本性清净，从本来就无染无着，}%

@{即知是一問曰。}@
%{就知道是一个。}%

@{何誰知無染無著答曰。}@
%{是谁知道无染无着呢？}%

@{心知無染問曰。}@
%{心知道无染。}%

@{心云何知無染答曰。}@
%{心怎么知道无染呢？}%

@{三世諸佛說心無形體畢竟不可得。}@
%{三世诸佛都说，心没有形体，最终不可得。}%

@{是以得知無染問曰。}@
%{所以就知道无染。}%

@{既無形體。}@
%{既然没有形体，}%

@{如何得知無染答曰。}@
%{怎么知道无染呢？}%

@{只為無形體。}@
%{正是因为没有形体，}%

@{故知無染。}@
%{所以就知道无染。}%

@{若有相貌處所還有染問曰。}@
%{如果还有相貌，还有处所，那就还有染污。}%

@{前言說心。}@
%{你前面说心，}%

@{未審心總有幾心答曰。}@

%{什么是凡夫心呢？}%

%{什么是圣人心呢？}%

%{如果你执着于相，就是凡夫心。}%

%{离于相，就是圣人心。}%

@{一切修道者若見來去心即是長矩心。}@
%{一切修行者，如果看到有来有去的心，这就是长矩心。}%

%{看到好恶。}%

@{見憎愛。}@
%{看到憎爱。}%

@{見嗔喜。}@
%{看到嗔喜。}%

@{見邪正。}@
%{看到邪正。}%

%{看到凡圣。}%

%{看到自在不自在。}%

@{見涅槃。}@
%{看到涅槃。}%

@{見解脫。}@
%{看到解脱。}%

@{見不解脫。}@
%{看到不解脱。}%

%{看到佛菩萨。}%

@{見究竟不究竟。}@
%{看到究竟不究竟。}%

%{看到知解不知解。}%

@{見精進。}@
%{看到精进。}%

%{看到禅定。}%

@{不起一念亦不見一物。}@
%{不起任何念头，也不见任何事物，}%

%{既然没有得到，怎么会有知呢？}%

@{我今亦無得亦無知。}@
%{我现在也没有得到，也没有知。}%
