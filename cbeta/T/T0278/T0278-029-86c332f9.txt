@{爾時，心王菩薩白佛言：世尊！所謂：阿僧祇、不可量、無分齊、無周遍、不可數、不可稱量、不可思議、不可說、不可說不可說。世尊！云何阿僧祇乃至不可說不可說耶？}@
%{当时，心王菩萨对佛说：世尊！所谓：阿僧祇、不可量、无分齐、无周遍、不可数、不可称量、不可思议、不可说、不可说不可说。世尊！阿僧祇乃至不可说不可说，究竟是怎样的呢？}%

@{佛告心王菩薩言：善哉！善哉！善男子！饒益眾生故，乃能問此如來、應供、等正覺，佛智境界甚深之義。善男子！汝今諦聽，我當說之。}@
%{佛对心王菩萨说：善哉！善哉！善男子！你为了利益众生，才能问到如来、应供、等正觉，佛智境界如此深奥的道理。善男子！你如今仔细听，我将为你解释。}%

@{爾時，心王菩薩白佛言：世尊！唯然善聽。}@
%{当时，心王菩萨对佛说：世尊！我一定仔细聆听。}%

@{爾時，世尊為心王菩薩以偈頌曰：}@
%{当时，释迦牟尼佛为了心王菩萨，用偈颂说道：}%

@{不可言說不可說，充滿一切不可說，}@
%{不可言说不可说，充满一切不可说，}%

@{不可言說諸劫中，說不可說不可盡。}@
%{不可言说诸劫中，说不可说不可尽。}%

@{不可言說諸佛剎，皆悉盡末為微塵，}@
%{不可言说诸佛刹，都悉尽末为微尘，}%

@{悉於一一微塵中，演說一切不可說。}@
%{悉于一一微尘中，演说一切不可说。}%

@{悉能善於一念中，說不可說諸世界，}@
%{悉能善于一念中，说不可说诸世界，}%

@{不可稱說諸劫中，念念次第而演說。}@
%{不可称说诸劫中，念念次第而演说。}%

@{不可說劫猶可盡，而不可說不可盡，}@
%{不可说劫犹可尽，而不可说不可尽，}%

@{悉於一一微塵中，分別演說不可說。}@
%{悉于一一微尘中，分别演说不可说。}%

@{不可說劫猶可盡，而不可說不可盡，}@
%{不可说劫犹可尽，而不可说不可尽，}%

@{不可言說微塵中，悉有不可說眾生。}@
%{不可言说微尘中，悉有不可说众生。}%

@{皆共讚歎普賢德，猶尚不能令窮盡，}@
%{皆共赞叹普贤德，犹尚不能令穷尽，}%

@{設使一微毛端處，有不可說諸普賢。}@
%{设使一微毛端处，有不可说诸普贤。}%

@{彼諸一切普賢等，說不可說不能盡，}@
%{彼诸一切普贤等，说不可说不能尽，}%

@{如一微細毛端處，十方世界亦如是。}@
%{如一微细毛端处，十方世界亦如是。}%

@{於彼一一毛端處，置不可說諸佛剎，}@
%{于彼一一毛端处，置不可说诸佛刹，}%

@{毛端能量虛空盡，而說佛剎不可盡。}@
%{毛端能量虚空尽，而说佛刹不可尽。}%

@{於彼一一毛道中，種種無量諸佛剎，}@
%{于彼一一毛道中，种种无量诸佛刹，}%

@{有同類者不可說，亦有異類不可說，}@
%{有同类者不可说，亦有异类不可说，}%

@{於彼一一毛道中，有不可說淨佛剎，}@
%{于彼一一毛道中，有不可说净佛刹，}%

@{以不可說莊嚴具，莊嚴彼彼諸佛剎，}@
%{以不可说庄严具，庄严彼彼诸佛刹，}%

@{於彼一一毛道中，演出名身不可說，}@
%{于彼一一毛道中，演出名身不可说，}%

@{於彼一一諸名身，廣宣無量諸佛名，}@
%{于彼一一诸名身，广宣无量诸佛名，}%

@{一一如來自身中，變化毛孔不可說，}@
%{一一如来自身中，变化毛孔不可说，}%

@{於彼一一毛孔中，出生異色不可說，}@
%{于彼一一毛孔中，出生异色不可说，}%

@{於彼一一異色中，放妙光明不可說，}@
%{于彼一一异色中，放妙光明不可说，}%

@{於彼一一光明中，出寶蓮華不可說，}@
%{于彼一一光明中，出宝莲华不可说，}%

@{於彼一一寶蓮華，各有寶葉不可說，}@
%{于彼一一宝莲华，各有宝叶不可说，}%

@{於彼一一寶蓮葉，有微妙色不可說，}@
%{于彼一一宝莲叶，有微妙色不可说，}%

@{於彼一一妙色中，出生蓮華不可說，}@
%{于彼一一妙色中，出生莲华不可说，}%

@{於彼一一蓮華中，各放光明不可說，}@
%{于彼一一莲华中，各放光明不可说，}%

@{於彼一一光明中，出生淨月不可說，}@
%{于彼一一光明中，出生净月不可说，}%

@{於彼一一諸月中，復出淨月不可說，}@
%{于彼一一诸月中，复出净月不可说，}%

@{於彼一一淨月中，出淨光明不可說，}@
%{于彼一一净月中，出净光明不可说，}%

@{於彼一一光明中，出不可說明淨日，}@
%{于彼一一光明中，出不可说明净日，}%

@{於彼一一諸日中，出不可說淨妙色，}@
%{于彼一一诸日中，出不可说净妙色，}%

@{於彼一一妙色中，出不可說淨光明，}@
%{于彼一一妙色中，出不可说净光明，}%

@{於彼淨妙光明中，出不可說師子座，}@
%{于彼净妙光明中，出不可说狮子座，}%

@{不可言說莊嚴具，出不可說淨光明，}@
%{不可言说庄严具，出不可说净光明，}%

@{於彼一一光明中，出不可說異妙色，}@
%{于彼一一光明中，出不可说异妙色，}%

@{於彼一一妙色中，出不可說明淨寶，}@
%{于彼一一妙色中，出不可说明净宝，}%

@{於彼一一明淨寶，出不可說不可說，}@
%{于彼一一明净宝，出不可说不可说，}%

@{金剛寶藏如須彌，清淨具足而莊嚴。}@
%{金刚宝藏如须弥，清净具足而庄严。}%

@{於彼一寶須彌中，有不可說不可說，}@
%{于彼一宝须弥中，有不可说不可说，}%

@{微妙殊特諸佛剎，清淨具足而莊嚴。}@
%{微妙殊特诸佛刹，清净具足而庄严。}%

@{如此一寶須彌山，一切須彌亦如是，}@
%{如此一宝须弥山，一切须弥亦如是，}%

@{悉有無量不可說，具足清淨諸佛剎。}@
%{悉有無量不可说，具足清净诸佛刹。}%

@{是不可說不可說，皆悉分別不可說，}@
%{是不可说不可说，皆悉分别不可说，}%

@{攝取不可言說轉，出生光明不可說，}@
%{摄取不可言说转，出生光明不可说，}%

@{於彼一一光明中，出生諸佛不可說，}@
%{于彼一一光明中，出生诸佛不可说，}%

@{彼諸一切如來等，說不可說清淨偈，}@
%{彼诸一切如来等，说不可说清净偈，}%

@{彼偈悉於念念中，說不可說真實諦，}@
%{彼偈悉于念念中，说不可说真实谛，}%

@{示現一切未來際，如來智慧無窮盡。}@
%{示现一切未来际，如来智慧无穷尽。}%

@{彼諸一一如來等，出不可說梵音聲，}@
%{彼诸一一如来等，出不可说梵音声，}%

@{於彼一一梵音中，轉不可說淨法輪，}@
%{于彼一一梵音中，转不可说净法轮，}%

@{於彼一一法輪中，雨不可說修多羅，}@
%{于彼一一法轮中，雨不可说修多罗，}%

@{於彼一一修多羅，分別諸法不可說，}@
%{于彼一一修多罗，分别诸法不可说，}%

@{於彼一一諸法中，又說諸法不可說，}@
%{于彼一一诸法中，又说诸法不可说，}%

@{又復於彼諸法中，說眾生依不可說。}@
%{又复于彼诸法中，说众生依不可说。}%

@{又於一一毛道中，不可說劫說正法，}@
%{又于一一毛道中，不可说劫说正法，}%

@{如彼一微毛端處，一切十方亦如是。}@
%{如彼一微毛端处，一切十方亦如是。}%

@{不可稱說無礙心，變化諸佛不可說，}@
%{不可称说无碍心，变化诸佛不可说，}%

@{彼諸一一化如來，所出變化不可說，}@
%{彼诸一一化如来，所出变化不可说，}%

@{彼諸一切所變化，遍遊佛剎不可說，}@

@{若於一小微塵中，有諸佛剎不可說，}@
%{若于一小微尘中，有诸佛刹不可说，}%

@{於彼一一佛剎中，各有賢首不可說，}@
%{于彼一一佛刹中，各有贤首不可说，}%

@{賢首如來世界中，復有佛剎不可說。}@
%{贤首如来世界中，复有佛刹不可说。}%

@{一切法界無有餘，其中所有諸微塵，}@
%{一切法界无有余，其中所有诸微尘，}%

@{於彼一一微塵中，佛剎成敗不可說，}@
%{于彼一一微尘中，佛刹成败不可说，}%

@{於彼一一微塵中，安置佛剎不可說，}@
%{于彼一一微尘中，安置佛刹不可说，}%

@{復置剎海不可說，分別方類不可說，}@
%{复置刹海不可说，分别方类不可说，}%

@{於彼一一佛剎中，不可稱說諸如來，}@
%{于彼一一佛刹中，不可称说诸如来，}%

@{彼諸如來所壽命，不可稱說諸大劫。}@
%{彼诸如来所寿命，不可称说诸大劫。}%

@{諸佛所行不可說，微妙正法不可說，}@
%{诸佛所行不可说，微妙正法不可说，}%

@{威神道力不可說，離障礙智不可說，}@
%{威神道力不可说，离障碍智不可说，}%

@{微妙智慧不可說，境界甚深不可說，}@
%{微妙智慧不可说，境界甚深不可说，}%

@{十力功德不可說，佛覺菩提不可說，}@
%{十力功德不可说，佛觉菩提不可说，}%

@{清淨深入不可說，清淨法界不可說，}@
%{清净深入不可说，清净法界不可说，}%

@{彼智慧藏不可說，分別功德不可說。}@
%{彼智慧藏不可说，分别功德不可说。}%

@{菩薩究竟正隨順，具足迴向不可說，}@
%{菩萨究竟正随顺，具足回向不可说，}%

@{分別無量諸迴向，迴向一切諸導師。}@
%{分别无量诸回向，回向一切诸导师。}%

@{不可稱說諸大劫，一心正受諸三昧，}@
%{不可称说诸大劫，一心正受诸三昧，}%

@{於不可說諸佛所，所行清淨不可說。}@
%{于不可说诸佛所，所行清净不可说。}%

@{得不可說無礙心，悉遍遊行十方界，}@
%{得不可说无碍心，悉遍游行十方界，}%

@{所修行業不可說，神力應現不可說，}@
%{所修行业不可说，神力应现不可说，}%

@{分別佛剎不可說，諸佛現前不可說，}@
%{分别佛刹不可说，诸佛现前不可说，}%

@{勇猛精進不可說，究竟智慧不可說，}@
%{勇猛精进不可说，究竟智慧不可说，}%

@{未曾離於一坐處，而能遍遊十方界。}@
%{未曾离于一坐处，而能遍游十方界。}%

@{不可稱說諸大劫，遍遊十方諸佛剎，}@
%{不可称说诸大劫，遍游十方诸佛刹，}%

@{智慧方便不可說，如實智慧不可說，}@
%{智慧方便不可说，如实智慧不可说，}%

@{轉淨智慧不可說，念念示現不可說，}@
%{转净智慧不可说，念念示现不可说，}%

@{於諸一一語言中，解佛智慧不可說。}@
%{于诸一一语言中，解佛智慧不可说。}%

@{或於一時覺菩提，或種種時覺菩提，}@
%{或于一时觉菩提，或种种时觉菩提，}%

@{入諸毛端不可說，入諸微細不可說，}@
%{入诸毛端不可说，入诸微细不可说，}%

@{殊特勝性不可說，見諸如來不可說，}@
%{殊特胜性不可说，见诸如来不可说，}%

@{一一方便不可說，隨順善入諸佛性。}@
%{一一方便不可说，随顺}%@{爾時，心王菩薩摩訶薩告諸菩薩言：佛子！如此娑婆世界釋迦牟尼佛剎一劫，於安樂世界阿彌陀佛剎為一日一夜；安樂世界一劫，於聖服幢世界金剛佛剎為一日一夜；聖服幢世界一劫，於不退轉音聲輪世界善樂光明清淨開敷佛剎為一日一夜；不退轉音聲輪世界一劫，於離垢世界法幢佛剎為一日一夜；離垢世界一劫，於善燈世界師子佛剎為一日一夜；善燈世界一劫，於善光明世界盧舍那藏佛剎為一日一夜；善光明世界一劫，於超出世界法光明清淨開敷蓮華佛剎為一日一夜；超出世界一劫，於莊嚴慧世界一切明光明佛剎為一日一夜；莊嚴慧世界一劫，於鏡光明世界覺月佛剎為一日一夜。佛子！如是次第，乃至百萬阿僧祇世界，最後世界一劫，於勝蓮華世界賢首佛剎為一日一夜，普賢菩薩等諸大菩薩充滿其中。}@
%{那个时候，心王菩萨对诸位菩萨说：佛子！像我们这个娑婆世界释迦牟尼佛的佛国，一劫的时间，在安乐世界阿弥陀佛的佛国，只相当于一天一夜；安乐世界一劫的时间，在圣服幢世界金刚佛的佛国，只相当于一天一夜；圣服幢世界一劫的时间，在不退转音声轮世界善乐光明清净开敷佛的佛国，只相当于一天一夜；不退转音声轮世界一劫的时间，在离垢世界法幢佛的佛国，只相当于一天一夜；离垢世界一劫的时间，在善灯世界师子佛的佛国，只相当于一天一夜；善灯世界一劫的时间，在善光明世界卢舍那藏佛的佛国，只相当于一天一夜；善光明世界一劫的时间，在超出世界法光明清净开敷莲花佛的佛国，只相当于一天一夜；超出世界一劫的时间，在庄严慧世界一切明光明佛的佛国，只相当于一天一夜；庄严慧世界一劫的时间，在镜光明世界觉月佛的佛国，只相当于一天一夜。佛子！以此类推，直到一百万阿僧祇世界，最后一个世界的最后一劫，在胜莲花世界贤首佛的佛国，只相当于一天一夜，普贤菩萨等诸位大菩萨都充满其中。}%
