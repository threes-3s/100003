@{大方廣佛華嚴經卷第四}@
%{大方廣佛華嚴經卷第四}%

@{東晉天竺三藏佛馱跋陀羅譯}@
%{東晉天竺三藏佛馱跋陀羅譯}%

@{彼香河中間一切平正，諸妙寶樹以為莊嚴，雜種寶幔彌覆其上，一切菩薩願力所起、佛所護念三世莊嚴而莊嚴之。}@
%{在香河中间，一切都平坦而正，用各种珍宝树来装饰，各种宝幔相互覆盖，所有菩萨的愿力所产生，佛所护念，三世庄严而庄严。}%

@{爾時，普賢菩薩以偈頌曰：}@
%{这时，普贤菩萨以偈颂说：}%

@{盧舍那佛遍十方，出一切化莊嚴身，}@
%{卢舍那佛遍布十方，显现一切化身，庄严自身，}%

@{彼亦不來亦不去，佛願力故皆悉見。}@
%{他不来也不去，因为佛的愿力，所以一切众生都能见到他。}%

@{一切佛剎微塵中，無量佛子修諸行，}@
%{在一切佛刹的微尘中，无数佛子修持各种行，}%

@{悉受清淨國土記，見嚴淨剎稱本行。}@
%{都得到清净国土的记号，见到严净刹，称赞自己的行为。}%

@{佛子！當知此蓮華藏世界海中，一一境界有世界海微塵數清淨莊嚴。}@
%{佛子！要知道，在这个莲花藏世界海中，每一个境界都有世界海微尘数的清净庄严。}%

@{諸佛子！此香水海上有不可說佛剎微塵數世界性住，或有世界性蓮華上住、或在無量色蓮華上住、或依真珠寶住、或依諸寶網住、或依種種眾生身住、或依佛摩尼寶王住；}@
%{诸佛子！在这个香水海中，有不可说佛刹微尘数的世界性住，有的世界性住在莲花上，有的住在无量色莲花上，有的依附着珍珠宝住，有的依附着各种宝网住，有的依附着各种众生身住，有的依附着佛摩尼宝王住；}%

@{或須彌山形、或河形、或轉形、或旋流形、或輪形、或樹形、或樓觀形、或雲形、或網形。}@
%{有的形状像须弥山，有的像河流，有的像旋转的形状，有的像旋流的形状，有的像轮子，有的像树，有的像楼阁，有的像云，有的像网。}%

@{爾時，普賢菩薩以偈頌曰：}@
%{这时，普贤菩萨以偈颂说：}%

@{堅固清淨諸佛剎，離垢解脫光明藏，}@
%{坚固清净的诸佛刹，远离垢染，解脱光明藏，}%

@{依止摩尼寶海住，或有依止香海住，}@
%{依附着摩尼宝海住，或者依附着香海住，}%

@{或依種種方便住，或依莊嚴眾色住，}@
%{或者依附着各种方便住，或者依附着庄严的各种颜色住，}%

@{或復須彌樹圓形，種種方門佛剎住。}@
%{或者像须弥树圆形，各种方门佛刹都住。}%

@{或光明身諸華藏，寶雲普放淨光明，}@
%{或者光明身，诸莲花藏，宝云普遍放出清净光明，}%

@{光明充滿勝世界，寶地海藏不可壞。}@
%{光明充满胜妙世界，宝地海藏不可破坏。}%

@{或淨佛剎無量色，光明焰雲眾色等，}@
%{或者清净佛刹，无量颜色，光明焰云，各种颜色都一样，}%

@{或有妙音諸世界，自然常音不思議。}@
%{或者有妙音的诸世界，自然常音不可思议。}%

@{無數願樂種種身，自在行雲音聲身，}@
%{无数愿乐，各种身，自在行云，音声身，}%

@{眾生無量德音身，最勝一切德音身。}@
%{众生无量德音身，最胜一切德音身。}%

@{種種門入諸佛剎，漸至無盡不思議，}@
%{各种门进入诸佛刹，逐渐到达无尽不可思议，}%

@{無數一切滿十方，無盡無量普自在。}@
%{无数一切充满十方，无尽无量，普遍自在。}%

@{一切諸方如來剎，廣大方便入佛界，}@
%{一切诸方如来的刹，广大方便进入佛界，}%

@{見十方剎漸次至，國土不增亦不減。}@
%{见到十方刹，逐渐到达，国土不增也不减。}%

@{以一國土滿十方，十方入一亦無餘，}@
%{以一个国土充满十方，十方进入一个，也无剩余，}%

@{世界本相亦不壞，無比功德故能爾。}@
%{世界本相也不破坏，无比功德，所以能这样。}%

@{一切佛剎微塵中，見盧舍那自在力，}@
%{一切佛刹微尘中，见到卢舍那自在力，}%

@{弘誓願海震音聲，調伏一切眾生類。}@
%{弘誓愿海震动声音，调伏一切众生类。}%

@{佛身充滿一切剎，無數菩薩亦如是，}@
%{佛身充满一切刹，无数菩萨也一样，}%

@{教化眾生無有量，佛現自在無倫匹。}@
%{教化众生，没有数量，佛现自在，无与伦比。}%

@{爾時，普賢菩薩告諸菩薩言：佛子！彼眾香水海中有一香水海，名樂光明，有一切香摩尼寶王莊嚴蓮華；}@
%{这时，普贤菩萨告诉诸菩萨说：佛子！在那众香水海中，有一个香水海，名叫乐光明，有一个一切香摩尼宝王庄严莲花；}%

@{上有世界，名清淨寶網光明，佛號離垢淨眼廣入。}@
%{上面有一个世界，名叫清净宝网光明，佛号离垢净眼广入。}%

@{彼世界上，過佛剎塵數世界，有佛國名雜香蓮華勝妙莊嚴，依寶網住，形如師子座，佛號師子座光明勝照。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫杂香莲花胜妙庄严，依附着宝网住，形状像狮子座，佛号狮子座光明胜照。}%

@{彼世界上，過佛剎塵數世界，有佛國名寶莊嚴普光明，依諸華住，形如日輪雲，佛號廣大光明智勝。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫宝庄严普光明，依附着各种莲花住，形状像日轮云，佛号广大光明智胜。}%

@{彼世界上，過佛剎塵數世界，有佛國名雜光蓮華，佛號金剛光明普精進善起。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫杂光莲花，佛号金刚光明普精进善起。}%

@{彼世界上，過佛剎塵數世界，有佛國名無畏嚴淨，佛號平等莊嚴妙音幢王。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫无畏严净，佛号平等庄严妙音幢王。}%

@{彼世界上，過佛剎塵數世界，有佛國名華開淨焰，佛號愛海功德稱王。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫华开净焰，佛号爱海功德称王。}%

@{彼世界上，過佛剎塵數世界，有佛國名總持，佛號淨智慧海。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫总持，佛号净智慧海。}%

@{彼世界上，過佛剎塵數世界，有佛國名解脫聲，佛號善相幢。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫解脱声，佛号善相幢。}%

@{彼世界上，過佛剎塵數世界，有佛國名勝起，佛號蓮華藏光。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫胜起，佛号莲花藏光。}%

@{彼世界上，過佛剎塵數世界，有佛國名善住金剛不可破壞，佛號那羅延不可破壞。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫善住金刚不可破坏，佛号那罗延不可破坏。}%

@{彼世界上，過佛剎塵數世界，有佛國名華林赤蓮華，佛號雜寶華鬘智王。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫华林赤莲花，佛号杂宝华鬘智王。}%

@{彼世界上，過佛剎塵數世界，有佛國名淨光勝電如來藏，佛號能起一切所願功德。}@
%{在这个世界上面，经过佛刹尘数世界，有一个佛国名叫净光胜电如来藏，佛号能起一切所愿功德。}%

@{彼世界上，有香水海，名淨光焰起，中有世界性，名善住。}@
%{在这个世界上面，有一个香水海，名叫净光焰起，其中有一个世界性，名叫善住。}%

@{次上復有香水海，名金剛眼光明，中有世界性，名法界等起。}@
%{上面还有香水海，名叫金刚眼光明，其中有一个世界性，名叫法界等起。}%

@{次上復有香水海，名蓮華平正，中有世界性，名出十方化身。}@
%{上面还有香水海，名叫莲花平正，其中有一个世界性，名叫出十方化身。}%

@{次上復有香水海，名寶地莊嚴光明，中有世界性，名寶枝莊嚴。}@
%{上面还有香水海，名叫宝地庄严光明，其中有一个世界性，名叫宝枝庄严。}%

@{次上復有香水海，名化香焰，中有世界性，名清淨化。}@
%{上面还有香水海，名叫化香焰，其中有一个世界性，名叫清净化。}%

@{次上復有香水海，名寶幢，中有世界性，名佛護念。}@
%{上面还有香水海，名叫宝幢，其中有一个世界性，名叫佛护念。}%

@{次上復有世界性，名眾色普光。}@
%{上面还有世界性，名叫众色普光。}%

@{如是次上復有世界塵數香水海及世界性，如一方，十方亦如是盧舍那佛常轉法輪處。}@
%{像这样，上面还有世界尘数香水海以及世界性，像一方，十方也一样，是卢舍那佛常转法轮的地方。}%
@{爾時，普賢菩薩以偈頌曰：}@
%{当时，普贤菩萨用偈颂说道：}%

@{法界不可壞，蓮華世界海，}@
%{法界不可破灭，莲花世界大海，}%

@{離垢廣莊嚴，安住於虛空。}@
%{远离污垢，广阔庄严，安住在虚空中。}%

@{此世界海中，剎性難思議，}@
%{这世界大海中，刹土的性质难以思议，}%

@{善住不雜亂，各各悉自在。}@
%{善住不杂乱，个个都自在。}%

@{平正住莊嚴，依種種色住，}@
%{平正住庄严，依各种颜色而住，}%

@{如來世界海，佛剎相隨順。}@
%{如来的世界大海，佛刹互相随顺。}%

@{種種身音聲，一切佛自在，}@
%{种种身、音、声，一切佛都自在，}%

@{普見諸世界，種種業莊嚴。}@
%{普见诸世界，种种业力庄严。}%

@{須彌山城網，水旋輪圓形，}@
%{须弥山城网，水旋轮圆形，}%

@{清淨色蓮華，彼彼悉圍遶。}@
%{清净色莲花，彼彼都围绕着。}%

@{尸羅幢盆形，隨順轉色形，}@
%{尸罗幢盆形，随顺转动着颜色形状，}%

@{如是難思議，諸佛國土形。}@
%{如此难以思议，诸佛的国土形状。}%

@{不思議世界，依止蓮華住，}@
%{不可思议的世界，依着莲花而住，}%

@{放大光明網，普照於一切。}@
%{放大光明网，普照于一切。}%

@{一一如來剎，放諸光明網，}@
%{一一如来刹，放出光明网，}%

@{照一切佛國，充滿十方海。}@
%{照一切佛国，充满十方海。}%

@{一切諸佛剎，一切境界門，}@
%{一切诸佛刹，一切境界门，}%

@{一切方便入，皆悉見無量，}@
%{一切方便进入，都能够见到无量，}%

@{不思議佛剎，不壞不可盡，}@
%{不可思议的佛刹，不坏不可尽，}%

@{無量淨莊嚴，大仙威神力，}@
%{无量净庄严，大仙威神力，}%

@{彼如來剎頂，不思議世界，}@
%{彼如来刹顶，不可思议的世界，}%

@{或成或有敗，不生亦不滅，}@
%{或成或败，不生也不灭，}%

@{譬如諸樹林，華葉或生落，}@
%{譬如诸树林，花叶或生或落，}%

@{如是諸佛剎，成敗亦復然，}@
%{如此诸佛刹，成败也复如此，}%

@{如依種種樹，有種種果生，}@
%{如依各种树，有各种果实生长，}%

@{如是種種剎，有種種眾生，}@
%{如此种种刹，有种种众生，}%

@{種子差別故，果實生不同，}@
%{种子差别故，果实生长不同，}%

@{行業若干故，佛剎種種異，}@
%{行业若干故，佛刹种种不同，}%

@{譬如意寶珠，隨意現眾色，}@
%{譬如如意宝珠，随心意显现各种颜色，}%

@{除諸妄想故，悉見清淨剎，}@
%{除掉诸妄想故，悉见清净刹，}%

@{譬如空中雲，龍王力能現，}@
%{譬如空中云，龙王之力能显现，}%

@{如是佛願力，一切佛剎起，}@
%{如此佛愿力，一切佛刹都起来，}%

@{猶如工幻師，能現種種眾，}@
%{犹如工幻师，能显现种种众生，}%

@{如是眾生業，佛剎不思議，}@
%{如此众生业，佛刹不可思议，}%

@{如見彩畫像，知是畫師造，}@
%{如见彩画像，知道是画师造的，}%

@{如是見佛剎，心畫師所成。}@
%{如此见佛刹，心画师所成。}%

@{眾生心不同，隨起諸妄想，}@
%{众生心不同，随起诸妄想，}%

@{如是諸佛剎，一切皆如化。}@
%{如此诸佛刹，一切皆如幻化。}%

@{猶如見導師，種種無量色，}@
%{犹如见导师，种种无量颜色，}%

@{隨眾生心行，見佛剎亦然。}@
%{随众生心行，见佛刹也如此。}%

@{無量真珠華，悉覆諸佛剎，}@
%{无量真珠华，悉覆诸佛刹，}%

@{色現各不同，離垢莊嚴現。}@
%{颜色显现各不相同，远离污垢，庄严显现。}%

@{彼蓮華網中，佛剎網依住，}@
%{彼莲花网中，佛刹网依着而住，}%

@{種種妙莊嚴，眾生所依處。}@
%{种种妙庄严，众生所依之处。}%

@{或有佛剎地，垢穢不平正，}@
%{或有佛刹之地，垢秽不平正，}%

@{眾生煩惱故，起如是佛剎。}@
%{众生烦恼故，起如此佛刹。}%

@{清淨不清淨，佛剎不可議，}@
%{清净不清净，佛刹不可议论，}%

@{眾生希望起，菩薩之所持。}@
%{众生希望起，菩萨之所持。}%

@{清淨不清淨，無量諸佛剎，}@
%{清净不清净，无量诸佛刹，}%

@{業海因緣起，菩薩之所化。}@
%{业海因缘起，菩萨之所化。}%

@{或放清淨光，離垢眾寶體，}@
%{或放出清净光，远离污垢，众宝之体，}%

@{種種妙莊嚴，諸佛令清淨。}@
%{种种妙庄严，诸佛令清净。}%

@{一切國土中，火災不可議，}@
%{一切国土中，火灾不可议论，}%

@{示現不清淨，而剎常堅固。}@
%{示现不清净，而刹土常坚固。}%

@{或依風輪住，或復依水輪，}@
%{或依风轮住，或复依水轮，}%

@{無量剎成敗，眾生行業故。}@
%{无量刹成败，众生行业故。}%

@{見無量佛剎，或成或有敗，}@
%{见无量佛刹，或成或败，}%

@{彼亦無有成，亦復無有敗。}@
%{彼亦无有成，亦复无有败。}%

@{於一一念中，無量佛剎起，}@
%{于一一念中，无量佛刹起，}%

@{諸佛所持故，國清淨離垢。}@
%{诸佛所持故，国土清净远离污垢。}%

@{或有佛剎起，泥土不清淨，}@
%{或有佛刹起，泥土不清净，}%

@{離明常闇冥，罪眾生所住。}@
%{远离光明，常暗冥，罪众生所住。}%

@{或有泥土剎，煩惱大恐怖，}@
%{或有泥土刹，烦恼大恐怖，}%

@{樂少憂苦多，薄福之所處。}@
%{快乐少，忧苦多，薄福之所处。}%

@{或有鐵世界，或有赤銅國，}@
%{或有铁世界，或有赤铜国，}%

@{諸石山穢惡，眾生業故起。}@
%{诸石山秽恶，众生业故起。}%

@{或有泥土剎，眾生常苦惱，}@
%{或有泥土刹，众生常苦恼，}%

@{長冥離光明，光明海能照。}@
%{长冥远离光明，光明海能照。}%

@{諸畜生趣中，受無量種身，}@
%{诸畜生趣中，受无量种身，}%

@{隨宿行業故，長受無量苦。}@
%{随宿行业故，长受无量苦。}%

@{閻羅王界中，飢渴苦常逼，}@

@{爾時，如來教化一切諸眾生故，於彼大眾海中說經，名現三世一切諸佛集會，世界塵數修多羅以為眷屬，隨諸眾生所應解故。}@
%{当时，如来为了教化一切众生，在那一大群人中说法，名为“现三世一切诸佛集会”，以世界尘数的修多罗作为眷属，随顺众生应该理解的道理。}%

@{爾時，普莊嚴童子聞是經已，宿世功德因緣故，得一切法具足三昧、一切法來入安住菩提心三昧、法界師子光方便三昧、法眼清淨三昧。}@

%{在生死海中，舍弃自身无数次，}%

%{修菩萨胜行，庄严清净佛刹海。}%

%{普贤菩萨发愿，修习诸行海，}%

%{一切佛刹海中，令佛土清净。}%

%{如同太阳的光明清净，所有颜色都具足，}%

%{佛智的光照耀之后，看到我本来的修行。}%

%{成等正觉的声音，都充满法界。}%

%{说这偈颂的时候，如同须弥山尘数的众生都发起了无上道心。}%

%{不厌烦生死之难，能够成为伟大的导师。}%

@{無量無邊願，一切諸佛海，}@
%{无量无边的愿，一切诸佛海，}%

@{能度無上道，方便海具足。}@
%{能够度脱无上道，方便海具足。}%

@{恭敬供養我，普莊嚴大力，}@
%{恭敬供养我，普庄严大力，}%

@{勝須彌山佛，成汝無上道。}@
%{胜过须弥山佛，成就你的无上道。}%

@{普賢常勇猛，具足大名稱，}@
%{普贤常勇猛，具足大名称，}%

@{一切法界滿，淨諸佛剎海。}@
%{一切法界充满，清净诸佛刹海。}%

@{爾時，一切功德本勝須彌山雲如來壽五十億歲。}@
%{当时，一切功德本胜须弥山云如来寿五十亿岁。}%

@{佛在摩竭提國寂滅道場，初始得佛，普光法堂，坐蓮華藏師子座上，善覺智無二念，了達法性、住佛所住、等諸如來至無礙趣具不退法無壞境界，住不思議等達三世，與十佛國土微塵數等大菩薩俱，盡一生補處，悉從他方世界來集，了眾生性，深入法界，常善思量世間涅槃，明了業報眾生心行，悉能解知諸法義味，觀察世間、離世間法，究竟分別為無為性，去、來、現在靡不貫達。}@
%{佛在摩竭提国寂灭道场，最初成佛，普光法堂，坐在莲花藏狮子座上，善觉智无二念，了达法性、住佛所住、等诸如来至无碍趣具不退法无坏境界，住不可思议等达三世，与十佛国土微尘数等的大菩萨一起，尽一生补处，都从他方世界来集聚，了知众生性，深入法界，常善思量世间涅槃，明了业报众生心行，都能理解诸法义味，观察世间、离世间法，究竟分别为无为性，去、来、现在无不贯通。}%

%{这时，文殊师利菩萨承接佛陀的神力，观察大众，感叹说：太好了！现在菩萨的聚会是前所未有的。诸佛子！应当知道佛刹不可思议，佛陀住处、佛国、佛法、佛刹清净、佛陀说法、佛陀出世、佛刹兴起、诸佛阿耨多罗三藐三菩提都不可思议。为什么呢？十方诸佛说法，知道众生的心行，随缘度化众生，与虚空、法界等同。为什么呢？在这个娑婆世界中，四大部洲教化一切众生，种种身、种种名、处所、形色、长短、寿命、诸得、诸入、诸根生处、业报，种种不同，众生所见也各不相同。为什么呢？诸佛子！这四大部洲佛陀的称呼不同，有的称为悉达，有的称为满月，有的称为师子吼，有的称为释迦牟尼，有的称为神仙，有的称为卢舍那，有的称为瞿昙，有的称为大沙门，有的称为最胜，有的称为能度。像这样称呼佛陀的名号，总共有1万个。}%

@{爾時，文殊師利告眾菩薩言：佛子！所說苦諦者，於此娑婆世界，或言害、或言逼迫、或言變異、或言境界、或言聚、或言刺、或言依根、或言不實、或言癰、或言童蒙行。所說苦集諦者，或言火、或言能壞、或言受義、或言覺、或言方便、或言決定、或言網、或言念、或言順眾生、或言顛倒根。所說苦滅諦者，或言無障礙、或言離垢淨、或言寂靜、或言無相、或言不死、或言無所有、或言因緣斷、或言滅、或言真實、或言自然住。所說苦滅道諦者，或言一乘、或言趣寂靜、或言引導、或言究竟希望、或言常不離、或言能捨擔、或言至非趣、或言聖人隨行、或言仙人行、或言十藏。}@
%{当时，文殊师利告诉众菩萨说：佛子！所说的苦谛，在这个娑婆世界，或者说害、或者说逼迫、或者说变异、或者说境界、或者说聚、或者说刺、或者说依根、或者说不实、或者说痈、或者说童蒙行。所说的苦集谛，或者说火、或者说能坏、或者说受义、或者说觉、或者说方便、或者说决定、或者说网、或者说念、或者说顺众生、或者说颠倒根。所说的苦灭谛，或者说无障碍、或者说离垢净、或者说寂静、或者说无相、或者说不死、或者说无所有、或者说因缘断、或者说灭、或者说真实、或者说自然住。所说的苦灭道谛，或者说一乘、或者说趣寂静、或者说引导、或者说究竟希望、或者说常不离、或者说能捨担、或者说至非趣、或者说圣人随行、或者说仙人行、或者说十藏。}%

@{諸佛子！此娑婆世界中，如是等四諦名字有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{佛子们！在这个娑婆世界中，像这样等等四谛的名字，有四十亿百千那由他，随着诸众生所应该调伏的，就那样说。}%

@{諸佛子！如娑婆世界所稱苦諦，於密訓世界，或名求根、或名不可出、或名不縛根、或名作不應作、或名一切不實、或名分別羸、或名處所成就、或名第一害、或名動、或名身事。所名苦集諦者，或名受、或名枝、或名燒、或名堅固、或名壞根、或名相續、或名害行、或名喜忘、或名生元、或名分。所名苦滅諦者，或名正義、或名堅固、或名讚歎、或名安隱、或名善趣、或名調伏、或名一道、或名離煩惱、或名不亂、或名究竟。所名苦滅道諦者，或名猛將、或名不沒、或名超出、或名勤方便、或名普眼、或名離邊、或名覺悟、或名得妙、或名無上目、或名觀方。}@
%{佛子们！就像娑婆世界所称呼的苦谛，在密训世界，或者叫求根、或者叫不可出、或者叫不缚根、或者叫作不应作、或者叫一切不实、或者叫分别羸、或者叫处所成就、或者叫第一害、或者叫动、或者叫身事。所称呼的苦集谛，或者叫受、或者叫枝、或者叫烧、或者叫坚固、或者叫坏根、或者叫相续、或者叫害行、或者叫喜忘、或者叫生元、或者叫分。所称呼的苦灭谛，或者叫正义、或者叫坚固、或者叫赞叹、或者叫安隐、或者叫善趣、或者叫调伏、或者叫一道、或者叫离烦恼、或者叫不乱、或者叫究竟。所称呼的苦灭道谛，或者叫猛将、或者叫不没、或者叫超出、或者叫勤方便、或者叫普眼、或者叫离边、或者叫觉悟、或者叫得妙、或者叫无上目、或者叫观方。}%
@{諸佛子！彼密訓世界如是等四諦名字有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{诸佛子！在密训世界，这四谛的名字有四十亿百千那由他那么多，佛陀根据众生的根基，用不同的名字来解释四谛。}%

@{諸佛子！如娑婆世界所名苦諦，於最勇世界，或名恐怖、或名福斷、或名應訶責、或名常給、或名麁歰、或名常怨、或名離勝、或名奪利、或名難共事、或名虛妄、或名勢力。所名苦集諦者，或名因緣、或名癡元、或名怨林、或名刃枝、或名滅味、或名仇對、或名味著、或名導引、或名增闇、或名害利。所名苦滅諦者，或名大義、或名饒益分、或名義中義、或名無量、或名見、或名虛妄斷、或名最勝、或名常、或名住、或名無為。所名苦滅道諦者，或名滅火、或名勝枝、或名定分別、或名不退、或名深方便、或名出、或名最上、或名至非趣、或名解脫、或名能令解脫。}@
%{诸佛子！在娑婆世界，我们把痛苦叫做苦谛，而在最勇世界，它还有很多其他的名字，比如恐怖、福断、应诃责、常给、粗歰、常怨、离胜、夺利、难共事、虚妄、势力等等。苦集谛，也就是痛苦的来源，在最勇世界，它也有一些其他的名字，比如因缘、痴元、怨林、刃枝、灭味、仇对、味著、导引、增暗、害利等等。苦灭谛，也就是痛苦的消除，在最勇世界，它也有一些其他的名字，比如大义、饶益分、义中义、无量、见、虚妄断、最胜、常、住、无为等等。苦灭道谛，也就是通往痛苦消除的道路，在最勇世界，它也有一些其他的名字，比如灭火、胜枝、定分别、不退、深方便、出、最上、至非趣、解脱、能令解脱等等。}%

@{諸佛子！彼最勇世界如是等四諦名有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{诸佛子！在最勇世界，这四谛的名字有四十亿百千那由他那么多，佛陀根据众生的根基，用不同的名字来解释四谛。}%

@{諸佛子！如娑婆世界所說苦諦，於離垢世界，或名悔恨、或名資待、或名分別、或名輪迴、或名前行、或名一味、或名非法、或名現前地、或名最邪、或名邪見、或名不可忍。所名苦集諦者，或名虛器、或名分、或名甘忍、或名生地、或名取、或名棄、或名增、或名擔、或名能生、或名堅縛。所名苦滅諦者，或名等等、或名空、或名無垢、或名勝根、或名勝等、或名無作、或名滅使、或名最上、或名畢竟、或名破印。所名苦滅道諦者，或名真堅固、或名方便分別、或名義根、或名真性、或名離愛、或名勝淨、或名有邊、或名寄全、或名究竟、或名淨虛妄。}@
%{诸佛子！在娑婆世界，我们把痛苦叫做苦谛，而在离垢世界，它还有很多其他的名字，比如悔恨、资待、分别、轮回、前行、一味、非法、现前地、最邪、邪见、不可忍等等。苦集谛，也就是痛苦的来源，在离垢世界，它也有一些其他的名字，比如虚器、分、甘忍、生地、取、弃、增、担、能生、坚缚等等。苦灭谛，也就是痛苦的消除，在离垢世界，它也有一些其他的名字，比如等等、空、无垢、胜根、胜等、无作、灭使、最上、毕竟、破印等等。苦灭道谛，也就是通往痛苦消除的道路，在离垢世界，它也有一些其他的名字，比如真坚固、方便分别、义根、真性、离爱、胜净、有边、寄全、究竟、净虚妄等等。}%

@{諸佛子！離垢世界如是等四諦名有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{诸佛子！在离垢世界，这四谛的名字有四十亿百千那由他那么多，佛陀根据众生的根基，用不同的名字来解释四谛。}%

@{諸佛子！如娑婆世界所說苦諦，於真實境世界，或名愛欲、或名險根、或名海分、或名邪方便、或名分別根、或名流轉、或名生滅、或名障礙、或名倒根、或名有數。所名苦集諦者，或名愛、或名陷溺、或名不可盡、或名分、或名不正趣、或名津梁、或名事、或名障礙、或名器、或名動。所名苦滅諦者，或名相續斷、或名解散、或名無名、或名不作、或名不現、或名無作、或名無色、或名無燒、或名明、或名淨。所名苦滅道諦者，或名寂行、或名正行、或名修證、或名安隱道、或名無量壽、或名修究竟、或名常道、或名難得、或名彼岸、或名無敵。}@
%{诸佛子！在娑婆世界，我们把痛苦叫做苦谛，而在真实境世界，它还有很多其他的名字，比如爱欲、险根、海分、邪方便、分别根、流转、生灭、障碍、倒根、有数等等。苦集谛，也就是痛苦的来源，在真实境世界，它也有一些其他的名字，比如爱、陷溺、不可尽、分、不正趣、津梁、事、障碍、器、动等等。苦灭谛，也就是痛苦的消除，在真实境世界，它也有一些其他的名字，比如相续断、解散、无名、不作、不现、无作、无色、无烧、明、净等等。苦灭道谛，也就是通往痛苦消除的道路，在真实境世界，它也有一些其他的名字，比如寂行、正行、修证、安隐道、无量寿、修究竟、常道、难得、彼岸、无敌等等。}%

@{諸佛子！真實境世界如是等四諦名有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{诸佛子！在真实境世界，这四谛的名字有四十亿百千那由他那么多，佛陀根据众生的根基，用不同的名字来解释四谛。}%

@{諸佛子！如娑婆世界所名苦諦者，於訶尼世界，或名掠取、或名非善友、或名戰怖、或名多言、或名真地獄、或名非法調伏、或名重擔、或名壞根、或名虛妄、或名虛妄根。所名苦集諦者，或名貪、或名作、或名惡、或名生、或名絞縛、或名想、或名有果、或名不愛、或名不應說、或名迴轉。所名苦滅諦者，或名不轉、或名解脫、或名無作、或名離愛、或名堅固、或名真實、或名離癡、或名寂滅、或名賢聖、或名離怨敵。所名苦滅道諦者，或名正語、或名無諍、或名教導、或名迴向心、或名廣妙、或名分別方便、或名有數、或名趣寂靜、或名勝智、或名善解義。}@
%{诸佛子！在娑婆世界，我们把痛苦叫做苦谛，而在诃尼世界，它还有很多其他的名字，比如掠取、非善友、战怖、多言、真地狱、非法调伏、重担、坏根、虚妄、虚妄根等等。苦集谛，也就是痛苦的来源，在诃尼世界，它也有一些其他的名字，比如贪、作、恶、生、绞缚、想、有果、不爱、不应说、迴转等等。苦灭谛，也就是痛苦的消除，在诃尼世界，它也有一些其他的名字，比如不转、解脱、无作、离爱、坚固、真实、离痴、寂灭、贤圣、离怨敌等等。苦灭道谛，也就是通往痛苦消除的道路，在诃尼世界，它也有一些其他的名字，比如正语、无诤、教导、迴向心、广妙、分别方便、有数、趣寂静、胜智、善解义等等。}%

@{諸佛子！訶尼世界如是等四諦名有四十億百千那由他，隨諸眾生所應調伏，作如是說。}@
%{诸佛子！在诃尼世界，这四谛的名字有四十亿百千那由他那么多，佛陀根据众生的根基，用不同的名字来解释四谛。}%
@{諸佛子！如娑婆世界所言苦諦者，於饒益世界，或名重擔、或名危脆、或名賊等、或名生死、或名非歡喜、或名流轉、或名疲勞、或名醜貌、或名能生、或名利刃。}@
%{各位佛弟子！就像娑婆世界说的苦谛，在饶益世界，也叫重担、危脆、贼等、生死、不欢喜、流转、疲劳、丑貌、能生、利刃。}%
@{所言苦集諦者，或名流散、或名擾亂、或名煩惱、或名羸劣、或名漂淪、或名乖違、或名非解脫、或名所作、或名取、或名虛妄。}@
%{说的苦集谛，也叫流散、扰乱、烦恼、羸劣、漂沦、乖违、不解脱、所作、取、虚妄。}%
@{所言苦滅諦者，或名離獄、或名真實、或名離諸難、或名覆護、或名善因、或名隨至、或名根、或名離枝、或名無為、或名無次第。}@
%{说的苦灭谛，也叫离狱、真实、离诸难、覆护、善因、随至、根、离枝、无为、无次第。}%
@{所言苦滅道諦者，或名達無所有、或名一切因、或名善本、或名明至、或名不轉法、或名有盡、或名大道、或名能調伏、或名安隱、或名非流轉。}@
%{说的苦灭道谛，也叫达无所有、一切因、善本、明至、不转法、有尽、大道、能调伏、安隐、不流转。}%
