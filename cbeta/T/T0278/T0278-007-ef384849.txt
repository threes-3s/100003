@{一切世界中，發心求佛者，}@
%{一切世界中，发心求佛的人，}%
@{先立清淨願，修習菩薩行。}@
%{首先要立下清净的愿，修习菩萨行。}%

@{菩薩淨修行，無量無數劫，}@
%{菩萨清净地修行，无量无数劫，}%
@{於法界無礙，無能測量者。}@
%{在法界中无障碍，没有能测量的。}%

@{悉普照十方，滅除愚癡闇，}@
%{遍照十方，灭除愚痴的黑暗，}%
@{一切無與等，是故莫能知。}@
%{一切无与伦比，所以没有人能知道。}%


@{爾時，一切慧菩薩承佛神力，普觀十方，以偈頌曰：}@
%{当时，一切慧菩萨承佛神力，遍观十方，以偈颂说：}%


@{無量無數劫，雖常見如來，}@
%{无量无数劫，虽然经常见到如来，}%
@{於此正法中，猶未覩真實，}@
%{在这个正法中，还没有见到真实，}%
@{妄想取諸法，增長癡惑網，}@
%{妄想执着诸法，增长痴惑的网，}%
@{輪迴生死中，盲冥不見佛。}@
%{轮回生死中，盲目地看不见佛陀。}%

@{雖復觀諸法，猶未見實相，}@
%{虽然观察诸法，还没有见到实相，}%
@{一切法生滅，但著假名字，}@
%{一切法生灭，只执着假的名字，}%
@{一切法無生，一切法無滅，}@
%{一切法无生，一切法无灭，}%
@{若能如是解，諸佛常現前。}@
%{如果能这样理解，诸佛就常在眼前。}%

@{無取亦無見，空寂無真實，}@
%{不取也不见，空寂无真实，}%
@{諸佛本來空，不可得思量。}@
%{诸佛本来空，不可思量。}%

@{若解一切法，不可思量者，}@
%{如果理解一切法，不可思量，}%
@{彼於諸煩惱，其心無所染。}@
%{那么他对一切烦恼，内心就不会染着。}%

@{虛妄取法相，是則為癡冥，}@
%{虚妄执着法相，这就是痴冥，}%
@{是故不見佛，亦不得真實。}@
%{所以看不见佛陀，也无法得到真实。}%

@{牟尼離三世，相好悉具足，}@
%{牟尼远离三世，相好都具足，}%
@{於住無所住，法界悉清淨。}@
%{在住处无所住，法界都清净。}%

@{因緣故法生，因緣故法滅，}@
%{因为缘故法生，因为缘故法灭，}%
@{如是觀如來，究竟離癡惑。}@
%{这样观察如来，最终远离痴惑。}%

@{法慧先已說，清淨微妙法，}@
%{法慧菩萨已经说了，清净微妙的法，}%
@{我從彼勝聞，菩提難思議。}@
%{我从他那里听到，菩提难以思议。}%


@{爾時，勝慧菩薩承佛神力，普觀十方，以偈頌曰：}@
%{当时，胜慧菩萨承佛神力，遍观十方，以偈颂说：}%


@{如來智甚深，一切莫能測，}@
%{如来的智慧很深，一切人无法测度，}%
@{不知真實法，世間悉迷惑。}@
%{不知道真实法，世间都迷惑。}%

@{童蒙思惟是，虛妄取諸法，}@
%{孩童般思惟，虚妄执着诸法，}%
@{是故不見佛，具足清淨相。}@
%{所以看不见佛陀，具足清净相。}%

@{愚癡心迷惑，妄取五陰相，}@
%{愚痴的心迷惑，妄执着五蕴相，}%
@{不了真實性，是故不見佛。}@
%{不明白真实性，所以看不见佛陀。}%

@{分別一切法，皆悉無真實，}@
%{分别一切法，都无真实，}%
@{如是解諸法，則見盧舍那。}@
%{这样理解诸法，就能见到卢舍那佛。}%

@{因前五陰故，後陰相續生，}@
%{因为前五蕴，后五蕴相续而生，}%
@{次第解五陰，見佛難思議。}@
%{次第理解五蕴，见到佛陀难以思议。}%

@{如寶在闇處，無明故不見，}@
%{就像宝藏在黑暗处，因为无明所以看不见，}%
@{真諦無說者，雖慧莫能覩。}@
%{真谛没有说的人，即使有智慧也无法看到。}%

@{如目不明淨，不見微妙色，}@
%{就像眼睛不清净，看不见微妙的颜色，}%
@{如是不淨心，不見諸佛法。}@
%{就像这样不清净的心，看不见诸佛法。}%

@{猶如明淨日，無目者不見，}@
%{就像明亮的太阳，没有眼睛的人看不见，}%
@{若人心諂曲，終不覩諸佛。}@
%{如果人心谄曲，最终不会看到诸佛。}%

@{故當淨慧眼，觀察諸法相，}@
%{所以应该清净慧眼，观察诸法相，}%
@{見法相明了，猶如鏡中像。}@
%{见到法相明了，就像镜子中的影像。}%

@{一切慧先說，清淨微妙法，}@
%{一切慧菩萨已经说了，清净微妙的法，}%
@{我從彼勝聞，見佛盧舍那。}@
%{我从他那里听到，见到卢舍那佛。}%


@{爾時，功德慧菩薩承佛神力，普觀十方，以偈頌曰：}@
%{当时，功德慧菩萨承佛神力，遍观十方，以偈颂说：}%


@{諸法虛無實，妄取堅固相，}@
%{诸法虚无真实，妄执着坚固相，}%
@{是故童蒙者，常轉生死輪。}@
%{所以孩童般的人，常转生死轮。}%

@{不善非勝法，妄作勝法相，}@
%{不善不是胜法，妄认为胜法相，}%
@{是故生障礙，愚癡常輪轉。}@
%{所以产生障碍，愚痴常转轮回。}%

@{不知八正道，云何知自心？}@
%{不知道八正道，怎么知道自己的心？}%
@{彼因顛倒想，增長一切惡。}@
%{他因为颠倒的妄想，增长一切恶。}%

@{不見諸法空，常受無量苦，}@
%{看不见诸法空，常受无量苦，}%
@{彼人不成就，清淨法眼故。}@
%{他不能成就，因为没有清净的法眼。}%

@{欲知一切心，先當求法眼，}@
%{想要知道一切心，首先应该求法眼，}%
@{如我所說者，能見真實佛。}@
%{就像我所说，就能见到真实的佛陀。}%

@{若有見佛者，其心無所著，}@
%{如果有见到佛陀的人，他的心无所着，}%
@{彼則見真實，如佛所說法。}@
%{那么他就见到真实，就像佛陀所说。}%

@{若見大智慧，如來妙法身，}@
%{如果见到大智慧，如来的妙法身，}%
@{能見如來故，彼有清淨眼。}@
%{能够见到如来，他就有清净眼。}%

@{無見乃能見，一切真實法，}@
%{无见才能见到，一切真实法，}%
@{於法有所見，彼則無所見。}@
%{对法有所见，他就是无所有见。}%

@{妙哉真實法，佛以導眾生，}@
%{妙哉真实法，佛陀以它引导众生，}%
@{一切諸有中，無生亦無死。}@
%{一切诸有中，无生也无死。}%

@{勝慧先已說，清淨微妙法，}@
%{胜慧菩萨已经说了，清净微妙的法，}%
@{我從彼勝聞，深解諸佛道。}@
%{我从他那里听到，深深理解诸佛之道。}%
@{爾時，精進慧菩薩承佛神力，普觀十方，以偈頌曰：}@
%{那个时候，精进慧菩萨承蒙佛陀的神力，放眼观望十方世界，用偈颂说道：}%

@{以諸妄想行，慧眼不清淨，}@

@{愚癡邪見增，常不見諸佛。}@
%{愚痴和邪见就会增多，永远见不到诸佛。}%

@{若能見邪偽，及以真實法，}@
%{如果能够辨别邪伪和真实法，}%

@{諦了實不實，則見清淨佛。}@
%{彻底明白什么是真实，什么是虚假，就能见到清净的佛。}%

@{見者則是垢，彼則無所見，}@
%{执着于“见”本身就是一种污垢，这样就什么也见不到，}%

@{諸佛離所見，是故見清淨。}@
%{诸佛远离“见”，所以能见清净。}%

%{世间语言的法，都是虚妄的，没有真实的，}%

@{虛妄非虛妄，非是諸佛法，}@
%{虚妄并非虚妄，这不是诸佛的法，}%

@{真實無二相，法性清淨故。}@
%{真实没有两个相，因为法性本来就是清净的。}%

@{法性自清淨，無相如虛空，}@
%{法性本来就是清净的，没有相，就像虚空一样，}%

@{一切無能說，智者如是觀。}@

@{樂觀一切法，寂滅無所有，}@

@{亦知不可修，能見牟尼尊。}@

@{如是見佛者，功德不可量，}@

@{一切所有行，寂靜空無相。}@
%{一切所有行为，都寂静、空、无相。}%
