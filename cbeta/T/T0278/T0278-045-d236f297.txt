@{入法界品第三十四之二}@
%{入法界品第三十四之二}%

@{爾時，壞散一切眾魔智幢王菩薩承佛神力，觀察十方，以偈頌曰：}@
%{这时，坏散一切众魔智幢王菩萨承佛神力，观察十方，以偈颂曰：}%

@{大智無礙身，非身難思議，}@
%{大智无碍身，非身难思议，}%
@{如來淨法身，一切莫能測。}@
%{如来净法身，一切莫能测。}%

@{不思議行業，起此清淨身，}@
%{不可思议行业，起此清净身，}%
@{無量妙莊嚴，不染於三界。}@
%{无量妙庄严，不染于三界。}%

@{普明照一切，清淨諸法界，}@
%{普明照一切，清净诸法界，}%
@{開發菩提門，出生深定智。}@
%{开发菩提门，出生深定智。}%

@{永離諸垢染，除滅一切障，}@
%{永离诸垢染，除灭一切障，}%
@{世間明淨日，普放慧光明。}@
%{世间明净日，普放慧光明。}%

@{永絕生死流，悉令三界淨，}@
%{永绝生死流，悉令三界净，}%
@{具足菩薩德，成就佛菩提。}@
%{具足菩萨德，成就佛菩提。}%

@{顯現無量色，於彼無所染，}@
%{显现无量色，于彼无所染，}%
@{所可現眾色，一切莫能思。}@
%{所可现众色，一切莫能思。}%

@{人王勝智慧，能於念念中，}@
%{人王胜智慧，能于念念中，}%
@{具無量菩提，一切莫能知，}@
%{具无量菩提，一切莫能知，}%

@{具足無盡智，一切莫能壞，}@
%{具足无尽智，一切莫能坏，}%
@{彼於一念中，明達三世佛。}@
%{彼于一念中，明达三世佛。}%

@{分別一切業，正念思菩提，}@
%{分别一切业，正念思菩提，}%
@{於思而非思，思法寂滅故，}@
%{于思而非思，思法寂灭故，}%

@{甚深不可說，遠離語言道，}@
%{甚深不可说，远离语言道，}%
@{如來從此起，佛業難思議。}@
%{如来从此起，佛业难思议。}%

@{爾時，明淨願智幢王菩薩承佛神力，觀察十方，以偈頌曰：}@
%{尔时，明净愿智幢王菩萨承佛神力，观察十方，以偈颂曰：}%

@{離癡清淨念，聞持一切法，}@
%{离痴清净念，闻持一切法，}%
@{深慧能分別，諸佛無盡海。}@
%{深慧能分别，诸佛无尽海。}%

@{菩薩決定心，修習菩薩行，}@
%{菩萨决定心，修习菩萨行，}%
@{出生甚深智，除滅諸疑惑。}@
%{出生甚深智，除灭诸疑惑。}%

@{其心無疲厭，遠離於懈怠，}@
%{其心无疲厌，远离于懈怠，}%
@{常勤修精進，究竟諸佛法。}@
%{常勤修精进，究竟诸佛法。}%

@{具足信智慧，安住不可動，}@
%{具足信智慧，安住不可动，}%
@{常樂甚深智，觀察無所著。}@
%{常乐甚深智，观察无所著。}%

@{無量無邊劫，積集諸功德，}@
%{无量无边劫，积集诸功德，}%
@{專心常迴向，諸佛甚深法。}@
%{专心常迴向，诸佛甚深法。}%

@{雖在生死中，其心無染著，}@
%{虽在生死中，其心无染著，}%
@{安住諸佛法，常樂如來行。}@
%{安住诸佛法，常乐如来行。}%

@{世間諸所有，陰界等諸法，}@
%{世间诸所有，阴界等诸法，}%
@{無畏悉除斷，安住佛正法。}@
%{无畏悉除断，安住佛正法。}%

@{世間顛倒惑，生死輪常轉，}@
%{世间颠倒惑，生死轮常转，}%
@{修習無礙行，實利益眾生。}@
%{修习无碍行，实利益众生。}%

@{菩薩行難稱，一切莫能知，}@
%{菩萨行难称，一切莫能知，}%
@{除滅一切苦，安樂諸群生。}@
%{除灭一切苦，安乐诸群生。}%

@{善覺菩提智，普照諸世間，}@
%{善觉菩提智，普照诸世间，}%
@{除滅愚癡闇，度脫一切眾。}@
%{除灭愚痴闇，度脱一切众。}%

@{爾時，壞散一切障智慧勢王菩薩承佛神力，觀察十方，以偈頌曰：}@
%{尔时，坏散一切障智慧势王菩萨承佛神力，观察十方，以偈颂曰：}%

@{無量無數劫，佛音難得聞，}@
%{无量无数劫，佛音难得闻，}%
@{何況親奉覲，除滅諸疑惑。}@
%{何况亲奉觐，除灭诸疑惑。}%

@{如來世間燈，究竟一切法，}@
%{如来世间灯，究竟一切法，}%
@{無上勝福田，令眾悉清淨。}@
%{无上胜福田，令众悉清净。}%

@{如來妙色身，一切莫能思，}@
%{如来妙色身，一切莫能思，}%
@{無量劫諦觀，其心無厭足。}@
%{无量劫谛观，其心无厌足。}%

@{佛子善觀察，如來妙色身，}@
%{佛子善观察，如来妙色身，}%
@{除滅一切障，究竟成菩提。}@
%{除灭一切障，究竟成菩提。}%

@{如來妙色身，出生淨妙音，}@
%{如来妙色身，出生净妙音，}%
@{無礙諸辯才，廣開菩提門。}@
%{无碍诸辩才，广开菩提门。}%

@{普照一切眾，無量難思議，}@
%{普照一切众，无量难思议，}%
@{建立大乘智，授以菩提記。}@
%{建立大乘智，授以菩提记。}%

@{功德圓滿日，出興照世間，}@
%{功德圆满日，出兴照世间，}%
@{長養一切世，無量功德身。}@
%{长养一切世，无量功德身。}%

@{若有值如來，遠離諸惡道，}@
%{若有值如来，远离诸恶道，}%
@{除滅一切苦，具足智慧身。}@
%{除灭一切苦，具足智慧身。}%

@{若有見如來，能發無量心，}@
%{若有见如来，能发无量心，}%
@{長養無數智，值遇諸導師。}@
%{长养无数智，值遇诸导师。}%

@{若有見如來，得定菩提心，}@
%{若有见如来，得定菩提心，}%
@{能自決定知，我必成菩提。}@
%{能自决定知，我必成菩提。}%

@{爾時，分別法界智通王菩薩承佛神力，觀察十方，以偈頌曰：}@
%{尔时，分别法界智通王菩萨承佛神力，观察十方，以偈颂曰：}%

@{菩薩見如來，無量淨功德，}@
%{菩萨见如来，无量净功德，}%
@{皆悉善迴向，究竟一切智。}@
%{皆悉善迴向，究竟一切智。}%

@{饒益眾生故，如來出世間，}@
%{饶益众生故，如来出世间，}%
@{具足大悲心，為世轉法輪。}@
%{具足大悲心，为世转法轮。}%

@{一切無能報，大仙普慈恩，}@
%{一切无能报，大仙普慈恩，}%
@{不可思議劫，代眾受苦故。}@
%{不可思议劫，代众受苦故。}%

@{無量億劫中，受諸地獄苦，}@
%{无量亿劫中，受诸地狱苦，}%
@{不捨一切眾，悉令得見佛。}@
%{不舍一切众，悉令得见佛。}%

@{普能代眾生，具受無量苦，}@
%{普能代众生，具受无量苦，}%
@{其心無疲倦，為度一切故。}@
%{其心无疲倦，为度一切故。}%

@{一切諸世間，所有惡道苦，}@
%{一切诸世间，所有恶道苦，}%
@{如來常處中，悉令聞正法。}@
%{如来常处中，悉令闻正法。}%

@{一一地獄住，不可思議劫，}@
%{一一地狱住，不可思议劫，}%
@{具受無量苦，終不離諸佛。}@
%{具受无量苦，终不离诸佛。}%

@{所以無量劫，常在三惡道，}@
%{所以无量劫，常在三恶道，}%
@{欲令諸群生，長養智慧故。}@
%{欲令诸群生，长养智慧故。}%

@{眾生見如來，除滅諸苦惱，}@
%{众生见如来，除灭诸苦恼，}%
@{安立於大智，一切佛境界。}@
%{安立于大智，一切佛境界。}%

@{若有見佛者，除滅一切障，}@
%{若有见佛者，除灭一切障，}%
@{長養功德藏，究竟成菩提。}@
%{长养功德藏，究竟成菩提。}%

@{如來能除滅，世間諸疑惑，}@
%{如来能除灭，世间诸疑惑，}%
@{隨其所應化，悉滿彼大願。}@
%{随其所应化，悉满彼大愿。}%
@{爾時，普賢菩薩觀察一切大眾，欲重開發，顯現照明；以法界等方便，廣說師子奮迅三昧；法界等、虛空界等、三世等、一切眾生界等、一切劫等、一切業性等、眾生希望等、眾生欲等、法光明等、隨時教化等、一切眾生根等，為諸菩薩十種廣說師子奮迅三昧。}@
%{当时，普贤菩萨观察着所有大众，想要重新开发，显现光明；用法界等等方便，广说狮子奋迅三昧；法界等等、虚空界等等、三世等等、一切众生界等等、一切劫等等、一切业性等等、众生希望等等、众生欲等等、法光明等等、随时教化等等、一切众生根等等，为诸菩萨十种广说狮子奋迅三昧。}%

@{何等為十？所謂：廣說一切法界中，一切佛剎微塵等佛，次第興世，演說正法；廣說虛空界等，一切佛剎中，盡未來劫一切諸佛所說；廣說一切佛剎中，一切如來現成正覺；廣說虛空界等，一切佛剎中，佛坐道場，眷屬圍遶，菩薩大眾，皆悉往詣；廣說一念中，三世一切佛出變化身，充滿一切法界；廣說一身充滿一切世界海、一切佛剎海，平等照持；廣說一一境界中，顯現三世一切諸佛自在功德地；廣說一一微塵中，顯現三世一切佛剎微塵等，佛自在神力；廣說一一毛孔，出三世一切佛大願海音，開發化導，盡未來劫一切菩薩；廣說處法界等師子之座，大眾圍遶，莊嚴道場，各隨其處，轉妙法輪，盡未來劫，未曾斷絕。佛子！此師子奮迅三昧，有如是等不可說佛剎微塵等廣說，唯是如來智慧境界。}@
%{这十种是什么呢？就是说：广说一切法界中，一切佛刹微尘等等佛，次第兴世，演说正法；广说虚空界等等，一切佛刹中，尽未来劫一切诸佛所说；广说一切佛刹中，一切如来现成正觉；广说虚空界等等，一切佛刹中，佛坐道场，眷属围绕，菩萨大众，皆悉往诣；广说一念中，三世一切佛出变化身，充满一切法界；广说一身充满一切世界海、一切佛刹海，平等照持；广说一一境界中，显现三世一切诸佛自在功德地；广说一一微尘中，显现三世一切佛刹微尘等等，佛自在神力；广说一一毛孔，出三世一切佛大愿海音，开发化导，尽未来劫一切菩萨；广说处法界等等狮子之座，大众围绕，庄严道场，各随其处，转妙法轮，尽未来劫，未曾断绝。佛子！此狮子奋迅三昧，有如是等不可说佛刹微尘等等广说，唯是如来智慧境界。}%

@{爾時，普賢菩薩摩訶薩欲重明師子奮迅三昧，承佛神力觀察如來；觀察大眾；觀察如來不可思議境界；觀察諸佛三昧；觀察不可思議世界；觀察不可思議智慧；觀察一切法，皆悉如幻；觀察不可思議諸佛平等；觀察無量無邊，一切音聲語言道故；以偈頌曰：}@
%{当时，普贤菩萨摩诃萨想要重新明了狮子奋迅三昧，承佛神力观察如来；观察大众；观察如来不可思議境界；观察诸佛三昧；观察不可思議世界；观察不可思議智慧；观察一切法，皆悉如幻；观察不可思議诸佛平等；观察无量无边，一切音聲語言道故；以偈颂曰：}%

@{一一毛孔中，普現最勝海，

佛處如來座，菩薩眾圍遶。}@
%{一一毛孔中，普现最胜海，

佛处如来座，菩萨众围绕。}%

@{一一毛孔中，無量諸佛海，

道場處華座，轉淨妙法輪。}@
%{一一毛孔中，无量诸佛海，

道场处华座，转净妙法轮。}%

@{一一毛孔中，一切剎塵等，

最勝跏趺坐，演說普賢行。}@
%{一一毛孔中，一切刹尘等，

最胜跏趺坐，演说普贤行。}%

@{最勝坐一剎，充滿十方界，

無盡菩薩雲，來詣於佛所。}@
%{最胜坐一刹，充满十方界，

无尽菩萨云，来诣于佛所。}%

@{無量億佛剎，塵數菩薩集，

圍遶於如來，為說諸法界。}@
%{无量亿佛刹，尘数菩萨集，

围绕于如来，为说诸法界。}%

@{顯現諸佛剎，入法界智海，

安住普賢行，滿足諸佛行。}@
%{显现诸佛刹，入法界智海，

安住普贤行，满足诸佛行。}%

@{安住於如來，一切諸世界，

深入菩薩行，樂聞勝法雲。}@
%{安住于如来，一切诸世界，

深入菩萨行，乐闻胜法云。}%

@{一一剎無量，億劫修諸行，

修習彼行已，究竟深法海，

滿足大願海，安住如來地。}@
%{一一刹无量，亿劫修诸行，

修习彼行已，究竟深法海，

满足大愿海，安住如来地。}%

@{出生最勝法，具足普賢行，

成就功德海，得無量自在，

如來身雲覆，一切諸佛剎，

普雨甘露法，令眾住佛道。}@
%{出生最胜法，具足普贤行，

成就功德海，得无量自在，

如来身云覆，一切诸佛刹，

普雨甘露法，令众住佛道。}%

@{爾時，世尊欲令諸菩薩安住師子奮迅三昧故，放眉間白毫相光，名普照三世法界門；不可說世界微塵等光明，以為眷屬，普照十方一切世界海。時，祇洹林菩薩大眾，普雲集者，悉見一切法界、虛空界等一切佛剎種種色、種種清淨、種種安住、種種形；如是等一切世界諸大菩薩，現坐道場，菩薩圍遶，諸天供養，成等正覺。或見於不可說佛剎微塵等，諸眷屬中，出妙音聲，充滿法界，轉淨法輪；或見在天宮殿，在龍宮殿，夜叉、乾闥婆、阿修羅、迦樓羅、緊那羅、摩睺羅伽，及人、非人等諸宮殿中；或見在人聚落城邑，大王京都，現種種身、種種姓名、種種色、種種圓光、種種光網、種種辯、種種眷屬、種種持、種種教持、種種音聲，而為說法。如此間，如來為諸菩薩現甚深三昧神力變化；一切法界、虛空界等，十方一切世界海中，現國土身及眾生身，諸業所起；乃至一毛孔中，一切悉現，亦復如是；而不壞三世，不壞眾生；普照一切諸眾生心，色身清淨，隨所應化；普現一切眾生類前，開示一切諸佛妙法；調伏眾生，顯現如來自在神力。其有眾生見聞念知如來自在神通力者，皆佛宿世善知識也；皆悉修習四攝善根，一向專求無上菩提，攝諸善根，成就方便，逮得如來不可思議自在三昧，悉與法界、虛空界等；或得法身；或得色身；或得菩薩具足諸行；或得清淨諸波羅蜜；或得菩薩圓滿淨行；或得菩薩諸地；或得菩提自在；或得如來不壞三昧；或得如來諸行智力；或得如來無礙辯才。}@
%{当时，世尊想要令诸菩萨安住狮子奋迅三昧，放眉间白毫相光，名普照三世法界门；不可说世界微尘等等光明，以为眷属，普照十方一切世界海。时，祇洹林菩萨大众，普云集者，悉见一切法界、虚空界等等一切佛刹种种色、种种清净、种种安住、种种形；如是等一切世界诸大菩萨，现坐道场，菩萨围绕，诸天供养，成等正觉。或见于不可说佛刹微尘等等，诸眷属中，出妙音聲，充满法界，转净法轮；或见在天宫殿，在龙宫殿，夜叉、乾闼婆、阿修罗、迦楼罗、紧那罗、摩睺罗伽，及人、非人等等诸宫殿中；或见在人聚落城邑，大王京都，现种种身、种种姓名、种种色、种种圆光、种种光网、种种辩、种种眷属、种种持、种种教持、种种音聲，而为说法。如此间，如来为诸菩萨现甚深三昧神力变化；一切法界、虚空界等等，十方一切世界海中，现国土身及众生身，诸业所起；乃至一毛孔中，一切悉现，亦复如是；而不坏三世，不坏众生；普照一切诸众生心，色身清净，随所应化；普现一切众生类前，开示一切诸佛妙法；调伏众生，显现如来自在神力。其有众生见闻念知如来自在神通力者，皆佛宿世善知识也；皆悉修习四摄善根，一向专求无上菩提，摄诸善根，成就方便，逮得如来不可思議自在三昧，悉与法界、虚空界等等；或得法身；或得色身；或得菩萨具足诸行；或得清净诸波罗蜜；或得菩萨圆满净行；或得菩萨诸地；或得菩提自在；或得如来不坏三昧；或得如来诸行智力；或得如来无碍辩才。}%
@{此諸菩薩得如是等十不可說佛剎微塵等諸妙功德，所謂：種種道、種種門、種種入、種種度、種種方便、種種至、種種方、種種光明、種種功德、種種功德具、種種自在。}@
%{这些菩萨获得了如此之多的功德，多到无法用语言形容，就像十个不可说佛刹中的微尘那样多，这些功德包括：各种各样的道、各种各样的门、各种各样的入、各种各样的度、各种各样的方便、各种各样的至、各种各样的方、各种各样的光明、各种各样的功德、各种各样的功德具足、各种各样的自在。}%

%{他们深入到各种各样的三昧境界，这些三昧包括：普庄严法界菩萨三昧；普照三世无碍三昧；不坏法界智三昧；随时深入如来境界三昧；普照虚空三昧；行如来力三昧；如来无畏庄严狮子奋迅三昧；一切法界方便藏三昧；无碍法界净月三昧；清净庄严法云三昧；除灭痴障法王幢三昧；一一境界中，悉见一切佛海三昧；一切世间不可坏身清净光明智幢三昧；深入佛身无坏三昧；随顺一切世藏三昧；诸法无迹无依三昧；圆满普照寂灭三昧；无所有，善化、普化，遍照三昧；摄持一切佛刹三昧；庄严一切佛刹，现成菩提三昧；行一切王法三昧；行一切众生境界无碍三昧；生一切诸佛三昧；究竟一切佛德海三昧；一一境界，出生尽未来际功德三昧；解了一切如来本生海三昧；护持尽未来际，一切如来种性三昧；令现在十方一切佛刹海悉净三昧；于一念中，普照一切佛住三昧；远离障碍，深入一切境界三昧；令一切刹，入一佛刹三昧；出一切佛化身三昧；决定智慧金刚王，入一切根海三昧；住持一切佛身，皆一身藏，无差别三昧；于一念中，住一切佛法界，方便无尽三昧；于一切法界佛刹中，示现涅槃三昧；住无上地三昧；令一切世界众生，悉见其身无别异三昧；一切佛智现前三昧；知一切法实相三昧；于一念中，具分别知三世三昧；于一念中，知一切法界藏三昧；皆随顺知如来智狮子行三昧；于一切境界，慧眼圆满三昧；十力境界等三昧；于一切境界，以平等眼示现三昧；出生一切妙色，众生见无厌足三昧；无动藏三昧；一法摄一切法三昧；一言普说一切音声三昧；一切佛无二法三昧；离三世三昧；分别一切劫不坏智三昧；微细方便十力内三昧；一切劫出生菩萨行无断三昧；一切十方普云现前三昧；菩提自在，法界无碍三昧；分别一切觉正希望安隐幢三昧；一切庄严，庄严虚空三昧；于念念中，出生化云三昧；离垢如空，如来月光三昧；一切佛持如空三昧；一切法庄严，法光三昧；开一切法义灯三昧；十力圆满光三昧；三世一切佛幢三昧；一切佛同一藏三昧；于念念中，发起究竟一切事三昧；无尽功德藏三昧；示现无量无边诸佛境界三昧；住一切法，金刚狮子座三昧；出生显现一切如来变化，无不知见三昧；一切念如来日三昧；一日悉觉三世三昧；自然寂静解脱三昧；见一切佛三昧；鉢曇摩華莊嚴，一切法界决定智三昧；一切法无著，虚空净眼三昧；一方摄十方海三昧；深入无底法界三昧；一切法海三昧；放一切光寂静身三昧；一念出生一切通明愿三昧；一切时，一切处，成菩提三昧；一切法界入一庄严三昧；一切佛住持三昧；一切众生胜地智明三昧；一念中，一身充满法界三昧；一身中，显现清净法界三昧；普门入法界，显现大庄严三昧；一切佛法圆满，轮智住持三昧；一切法方便，一方便庄严三昧；因陀罗网摄众生界，诸愿精进住持三昧；分别一切世界轮三昧；莲花妙德自在三昧；分别一切众生身三昧；对现一切众生身三昧；分别一切音声海三昧；了知一切众生地三昧；不可坏大悲藏三昧；一切佛入如来际三昧；修习一切佛法门三昧；观察狮子奋迅菩萨三昧。}%

@{如是等不可說佛剎微塵等三昧門，入如來海，入一切佛自在三昧；於念念中，充滿法界。}@
%{像这样多到无法形容的三昧门，他们都进入了如来海，进入了所有佛的自在三昧，在每一念中，都充满法界。}%

@{彼諸菩薩，一一皆有妙師子座，悉與十佛世界等，現大自在甚深智慧，悉得諸地明淨智慧；普觀一切，從智性生，專求一切智，具足成就，離癡慧眼。}@
%{这些菩萨，每一个都有奇妙的狮子座，都与十佛世界一样，显现出大自在的深奥智慧，他们都获得了所有地的清净智慧；他们能普遍观察一切，从智慧中生起，专心追求一切智，圆满成就，拥有离痴的慧眼。}%

@{悉為眾生作調御師；修行諸佛平等正法，決定了知一切境界，分別了知一切世界，樂寂滅法，遠離世間，常好閑靜；遊諸佛剎，無所染著；於一切法，心無所依；安住莊嚴妙法宮殿，教化成熟一切眾生。}@
%{他们都为众生做调御师；修行所有佛的平等正法，决定了知一切境界，分别了知一切世界，乐于寂灭法，远离世间，常常安然静默；游历诸佛刹，没有染着；对一切法，心无所依；安住在庄严妙法宫殿，教化成熟一切众生。}%

%{他们为一切众生显现佛刹，圆满成就无上智门；顺应离欲的境界，获得智慧身，消竭所有有为法海；他们为一切众生显现真实法界，法海慧光，圆满具足，都安住在坚固的三昧中。}%

@{以大悲心，常念眾生；解一切眾生皆悉如夢；一切如來悉如電光；一切言音皆悉如響；了一切法皆悉如化。}@

%{他们的智慧功德像金刚山一样，超越三世一切法王，觉悟无畏智功德幢；灭除所有魔幢，建立精进圆满的智幢；他们圆满成就无上之身，获得一切法无碍智慧；觉悟无尽智的真实法界，安住在真实法界，修行决定无相三昧。}%

@{巧方便生諸菩薩行，無二智慧，諦見境界世間諸趣，普照佛剎，無所染著；於一切法除滅癡闇，究竟智慧，皆悉圓滿，放淨法光，照十方界。}@
%{他们巧妙方便地生起菩萨行，拥有无二智慧，谛观境界，世间诸趣，普照佛刹，无所染着；对一切法，他们都灭除愚痴黑暗，究竟智慧，圆满具足，放出清净法光，照耀十方世界。}%

@{為一切眾生，作不虛福田；若見聞者，所願成滿；為一切世間，功德須彌，遠離恐怖，伏諸外道。}@
%{他们为一切众生，成为不虚的福田；如果有人见到或听到，他们的愿望都会圆满；他们为一切世间，成为功德须弥山，远离恐怖，降伏所有外道。}%

@{以微妙音，遍一切剎；常見諸佛，心無厭足，成就如來自在法身，隨其所應而化度之；能以一身，滿一切剎；以少方便，具足清淨自在神力，普遊十方無所障礙，智慧圓滿，遍照法界。}@
%{他们以微妙音，遍布一切佛刹；常常见到诸佛，心无所厌足，成就如来的自在法身，随其所应而化度众生；他们能以一身，充满一切佛刹；以很少的方便，具足清净自在神力，普遍游历十方，无所障碍，智慧圆满，遍照法界。}%

@{為一切眾耀明淨日，隨其所應，讚歎功德；了一切眾生諸根希望；於一切法，得無諍境界；分別了知諸法自性，大小相攝。}@
%{他们为一切众生，成为光明净日，随其所应，赞叹功德；了知一切众生的根性与愿望；对一切法，他们获得无诤境界；分别了知诸法自性，大小相摄。}%

@{決了如來甚深之地，說句、味身，諸法深義，無有窮盡；於一言中，普說一切修多羅海，究竟一切諸陀羅尼；廣智慧身，究竟無量劫陀羅尼。}@
%{他们决定了知如来的深奥境界，以句、味身，阐述诸法深义，没有穷尽；在一句话中，他们普遍宣说所有修多罗海，究竟所有陀罗尼；他们广大的智慧身，究竟无量劫陀罗尼。}%

@{於一念中，決定了知不可說劫；於一念中，了達三世法陀羅尼，普照無量諸佛法海。}@
%{在一念中，他们决定了知不可说劫；在一念中，他们了达三世法陀罗尼，普照无量诸佛法海。}%

@{為一切眾生，起淨智慧，轉正法輪，無能壞者；成就如來智慧境界，常入善現三昧正受，遠離障礙，深入諸法。}@
%{他们为一切众生，生起清净智慧，转动正法轮，没有能破坏者；他们成就如来智慧境界，常常进入善现三昧正受，远离障碍，深入诸法。}%

@{於一切法，得勝智自在，清淨莊嚴一切境界；深入十方甚深法界，攝取十方一切法界。}@
%{对一切法，他们获得胜智自在，清净庄严一切境界；他们深入十方深奥法界，摄取十方一切法界。}%

@{於一一微塵中，現成正覺；於無色性，現一切色；能以一方，攝一切方。}@
%{在每一微尘中，他们都现成正觉；在无色性中，他们都现一切色；他们能以一方，摄取一切方。}%

@{彼諸菩薩具足成滿如是等無量功德智慧之藏，常為一切諸佛讚歎；以句、味身，說其功德，不能窮盡，皆悉雲集於祇洹林。}@
%{这些菩萨具足圆满如此之多的功德智慧宝藏，常常为一切诸佛赞叹；以句、味身，说他们的功德，不能穷尽，他们都聚集在祇洹林。}%
@{爾時，彼諸菩薩深入如來功德大海。}@
%{那时候，那些菩萨深入如来功德的大海。}%

@{入已，於菩薩身中、及樓閣中，莊嚴具中、師子座中、以樂法力故；不可思議力故；於念念中，各放無量光明雲，普照法界，覺悟眾生。}@
%{进入大海后，在菩萨身体里、在楼阁里、在庄严器具里、在狮子座上，由于快乐的法力，由于不可思议的力量，在念念之中，各自放出无量光明云，普照法界，觉悟众生。}%

@{所謂：出一切寶香光明雲，讚歎三世諸佛功德，微妙音聲，充滿十方；}@
%{比如：放出充满一切宝香的光明云，赞叹三世诸佛功德，微妙的声音，充满十方；}%

@{出一切眾生雲，境界光明，說一切眾生清淨業報，微妙音聲，充滿十方；}@
%{放出充满一切众生的云，境界光明，说一切众生清净的业报，微妙的声音，充满十方；}%

@{出一切菩薩願行莊嚴光雲，說一切菩薩願行功德；}@
%{放出充满一切菩萨愿行庄严的光云，说一切菩萨愿行的功德；}%

@{出一切佛變化身雲，一切如來微妙音聲，充滿十方；}@
%{放出充满一切佛变化身的云，一切如来的微妙声音，充满十方；}%

@{出一切菩薩身雲，相好莊嚴，於一切佛剎，以微妙音，讚歎諸佛，充滿十方；}@
%{放出充满一切菩萨身体的云，相好庄严，在一切佛刹，以微妙的声音，赞叹诸佛，充满十方；}%

@{出三世佛莊嚴道場雲，現一切佛成等正覺，充滿十方；}@
%{放出充满三世佛庄严道场的云，显现一切佛成等正觉，充满十方；}%

@{於一切境界中，出龍王雲，雨一切香，充滿十方；}@
%{在一切境界中，放出龙王云，降下各种香，充满十方；}%

@{出一切佛身雲，歎普賢行，充滿十方；}@
%{放出充满一切佛身的云，赞叹普贤行，充满十方；}%

@{出一切佛剎淨光明雲，一切如來轉法輪音，充滿十方。}@
%{放出充满一切佛刹净光明的云，一切如来转法轮的声音，充满十方。}%

@{時，諸菩薩威神力故，此法力故，出如是等不可說佛剎微塵等雲。}@
%{这时，由于诸菩萨的威神力，由于这种法力，放出像这样不可说的佛刹微尘等云。}%

@{

爾時，文殊師利菩薩摩訶薩承佛神力，觀察十方，欲讚歎祇洹林中無量莊嚴，以偈頌曰：}@

%{观察祇洹林中，如来自在的力量，}%

一切境界出，無量功德雲。}@
%{一切境界都放出，无量功德云。}%

無量淨妙色，種種而莊嚴，}@
%{无量净妙的颜色，种种而庄严，}%

皆悉普照現，十方諸佛剎。}@

佛子身毛孔，出佛音聲雲，}@

種種寶莊嚴，充滿十方剎。}@

其身如梵王，威儀常安靜，}@
%{他的身体像梵王，威仪常安静，}%

遍遊十方剎，演出妙音聲。}@

@{

如來毛孔出，不可思議身，}@

%{三世诸如来，无量自在力。}%

如來毛孔中，一切諸世界，}@

三世一切佛，莊嚴妙道場，}@
%{三世一切佛，庄严妙道场，}%

@{

於此祇洹林，一切悉顯現，}@
%{于此祇洹林，一切都显现，}%

@{

普賢等佛子，無量種莊嚴。}@
%{普贤等佛子，无量种庄严。}%

@{

眾生等劫中，所修嚴淨剎，}@
%{众生等劫中，所修严净刹，}%

如是諸世界，悉現祇洹林。}@

@{

時，彼一切諸菩薩眾，以如來三昧照故，一一皆得不可說佛剎微塵等大悲法門，饒益安樂，攝取眾生。}@

@{彼諸菩薩於一一毛孔，各出不可說佛剎微塵等光明；}@
%{那些菩萨在每个毛孔，各自放出不可说的佛刹微尘等光明；}%

%{教化众生，未度者度，未脱者脱；}%

@{現不可說佛剎微塵等諸天宮殿，無常死相，一切諸法，皆悉如夢；}@

@{讚歎道場，說一切菩薩諸大願門。}@
%{赞叹道场，说一切菩萨诸大愿门。}%

@{或於一切世界，示現受生，為一切眾生，廣現檀波羅蜜門；}@
%{或者在一切世界，示现受生，为一切众生，广现檀波罗蜜门；}%

@{或現一切諸佛，圓滿淨戒功德，尸羅波羅蜜門；}@

%{或者显现勤修精进，毘梨耶波罗蜜门；}%

@{或現一切菩薩禪定三昧，相續解脫法門。}@
%{或者显现一切菩萨禅定三昧，相续解脱法门。}%

@{如來圓滿智慧光明，專求一切佛法，為一一句身、味身義故，能捨無量無數之身；}@
%{如来圆满智慧光明，专求一切佛法，为了一个一个字身、味身义故，能舍无量无数之身；}%

@{詣諸佛所，問無量法門，善知時會，隨其所應而為現法。}@
%{诣诸佛所，问无量法门，善知时会，随其所应而为现法。}%

@{令一切眾生住一切智，逮得方便智光海門，悉能供養諸佛菩薩；}@
%{令一切众生住一切智，逮得方便智光海门，都能供养诸佛菩萨；}%

@{降伏眾魔，制諸外道，悉能顯現菩薩力門；}@
%{降伏众魔，制诸外道，都能显现菩萨力门；}%

@{知一切技術明淨智地，欲令眾生得勝妙法，悉能了知眾生諸根，煩惱習氣，種種業報及智慧地。}@
%{知一切技术明净智地，想要令众生得胜妙法，都能了知众生诸根，烦恼习气，种种业报及智慧地。}%

@{以如是等不可說佛剎微塵等法門，教化眾生：}@
%{以像这样不可说的佛刹微尘等法门，教化众生：}%

@{或現天宮，或現龍宮，夜叉、乾闥婆、阿脩羅、迦樓羅、緊那羅、摩睺羅伽等宮，或現梵宮，或現人宮，或現閻羅王宮，或現地獄、餓鬼、畜生處，大悲智慧，及諸大願，不可沮壞，攝取眾生，不捨方便；}@

%{或者显现声闻色像；或者显现梵天色像；或者显现一切苦行色像；或者显现良医色像；或者显现商人色像；或者显现正命色像；或者显现妓人色像；或者显现天色像；或者显现一切技术色像；或者显现一切城邑、聚落、京都色像；}%

%{为世间灯，普照众生业报庄严，分别诸方，都行圆满菩萨诸行，都显现一切城邑、聚落、京都，化度众生。}%


@{爾時，文殊師利童子從善安住樓閣出，與一切同行諸菩薩俱；金剛力士，常隨侍衛。本願足天；樂聞法地天；常習大悲泉池方天；除滅愚癡夜天；出生佛晝日天；莊嚴正法界虛空河天；度眾生生死海天；長養一切善根薩婆若山天；莊嚴一切眾生身，滿足諸願，供養一切佛身天，守護一切眾生城天。守護一切眾生夜叉王；令一切眾生歡喜乾闥婆王；除滅一切餓鬼趣鳩槃荼王；於生死海，拔濟眾生迦樓羅王；正求薩婆若阿脩羅王；見佛歡喜無厭足摩睺羅伽王；常厭生死諸天王；常敬禮佛諸梵天王等；俱詣佛所，頭面禮足，設諸供養已，辭遊南方。}@
%{当时，文殊师利童子从善安住楼阁出来，和所有同行的大菩萨在一起；金刚力士，常随侍卫。本愿足天；乐闻法地天；常习大悲泉池方天；除灭愚痴夜天；出生佛昼日天；庄严正法界虚空河天；度众生生死海天；长养一切善根萨婆若山天；庄严一切众生身，满足诸愿，供养一切佛身天，守护一切众生城天。守护一切众生夜叉王；令一切众生欢喜乾闼婆王；除灭一切饿鬼趣鸠槃荼王；在生死海中，拔济众生迦楼罗王；正求萨婆若阿修罗王；见佛欢喜无厌足摩睺罗伽王；常厌生死诸天王；常敬礼佛诸梵天王等；都来到佛陀那里，顶礼佛陀的双足，供养完毕后，辞别前往南方。}%

@{爾時，尊者舍利弗承佛神力，見文殊師利童子，以菩薩莊嚴而自莊嚴，出祇洹林，遊行南方。見已，作如是念：我今當與文殊師利菩薩俱行。}@
%{当时，尊者舍利弗承佛神力，看见文殊师利童子，以菩萨的庄严而自庄严，走出祇洹林，游行南方。看见后，心想：我今天要和文殊师利菩萨一起走。}%

@{爾時，尊者舍利弗與六千比丘，眷屬圍遶，從自房出，來詣佛所，禮足辭退，向文殊師利。此六千比丘是舍利弗共行弟子，皆新出家，其名曰：海智比丘、大善調伏比丘、功德光比丘、大童子比丘、電光興比丘、清淨行比丘、天妙德比丘、因陀羅慧比丘、梵天比丘、寂靜慧比丘；如是等六千比丘，已曾供養過去諸佛；於諸佛所，種諸善根，性樂清淨，信心明徹，行諸大願，觀佛境界，了法實相，饒益眾生，常樂專求諸佛功德；此等比丘，皆是文殊師利之所化度。}@
%{当时，尊者舍利弗带着六千比丘，眷属围绕，从自己的房间出来，来到佛陀那里，礼拜佛陀的双足，辞别佛陀，走向文殊师利。这六千比丘是舍利弗一起修行弟子，都是新出家，他们的名字分别是：海智比丘、大善调伏比丘、功德光比丘、大童子比丘、电光兴比丘、清净行比丘、天妙德比丘、因陀罗慧比丘、梵天比丘、寂静慧比丘；像这样等等六千比丘，曾经供养过过去诸佛；在诸佛那里，种下善根，天生喜欢清净，信心明澈，行持大愿，观佛境界，了悟法性，利益众生，常乐专求诸佛功德；这些比丘，都是文殊师利所度化。}%

@{爾時，尊者舍利弗觀察大眾，告海智比丘言：汝可觀察文殊師利菩薩清淨之身，相好莊嚴，一切天人，莫能思議，光明圓滿，令無量眾生，發歡喜心，放大莊嚴妙光明網，除滅眾生無量苦惱。觀其眷屬，成就善根；觀其遊步，威儀庠序，所遊行處，自然平正，十方無礙；觀其功德所行道路，其傍悉有眾妙寶藏，自然發出；觀其供養過去諸佛，善根依果，從眾林樹出莊嚴藏；觀彼一切諸天大王，恭敬禮拜，供養雲雨。海智！汝觀文殊師利、一切如來，眉間毫相，放無量光，說諸佛法，悉入其頂。}@
%{当时，尊者舍利弗观察大众，对海智比丘说：你可以观察文殊师利菩萨清净的身躯，相好庄严，一切天人，都无法思量，光明圆满，令无数众生，生起欢喜心，放大庄严妙光明网，除灭众生无数苦恼。观察他的眷属，成就善根；观察他的游步，威仪庠序，所游行处，自然平正，十方无碍；观察他的功德所行道路，其旁皆有众妙宝藏，自然发出；观察他供养过去诸佛，善根依果，从众多树林中发出庄严宝藏；观察那些一切诸天大王，恭敬礼拜，供养云雨。海智！你观察文殊师利、一切如来，眉间毫相，放出无量光，宣说诸佛法，都进入他们的头顶。}%

@{爾時，尊者舍利弗，為諸比丘讚說文殊師利無量功德，諸大莊嚴。彼諸比丘聞讚歎已，皆悉歡喜；其心清淨，離諸垢穢，身體柔軟，調伏諸根；遠離障礙；現見諸佛，正求菩提，逮得菩薩清淨諸根；具菩薩力，長養大悲，入諸波羅蜜，發弘誓願，悉見十方諸如來海。}@
%{当时，尊者舍利弗，为诸比丘赞叹文殊师利无量功德，诸大庄严。那些比丘听到赞叹后，都非常欢喜；他们的心清净，远离一切污垢，身体柔软，调伏诸根；远离障碍；现见诸佛，正求菩提，获得菩萨清净诸根；具足菩萨力，长养大悲，进入诸波罗蜜，发弘誓愿，都看见十方诸如来海。}%

@{時，諸比丘白尊者舍利弗言：唯然。大師！願俱往詣文殊師利。}@
%{这时，诸比丘对尊者舍利弗说：是的。大师！愿我们一起前往文殊师利那里。}%

@{爾時，尊者舍利弗，與諸比丘往詣其所；到已，謂文殊師利：此諸比丘，皆新出家，欲見仁者。}@
%{当时，尊者舍利弗，带着诸比丘前往那里；到达后，对文殊师利说：这些比丘，都是新出家，想要见您。}%

@{爾時，文殊師利童子即為顯現菩薩自在，如象王迴顧視比丘。時，諸比丘頭面禮足，却住一面，合掌而立，作如是念：我等以此禮拜功德，知法實相，如和上舍利弗、釋迦牟尼世尊；得清淨身，相好音聲，神力自在，如文殊師利。}@
%{当时，文殊师利童子立刻显现出菩萨自在，像象王回过头来看着比丘。这时，诸比丘顶礼他的双足，退到一边，合掌而立，心想：我们以这样的礼拜功德，知道法性，如同和上舍利弗、释迦牟尼世尊；获得清净身，相好音声，神力自在，如同文殊师利。}%

@{爾時，文殊師利告諸比丘：汝等當知，若善男子、善女人，成就十種大心，則得佛地，況菩薩地；何等為十？所謂：發廣大心，長養一切善根，究竟不退，心無厭足；見一切佛，恭敬供養，心無厭足；正求一切佛法，心無厭足；遍行菩薩諸波羅蜜，心無厭足；具足一切菩薩三昧，心無厭足；於一切三世流轉，心無厭足；嚴淨佛剎，充滿十方，心無厭足；教化成熟一切眾生，心無厭足；於一切剎、一切劫中，行菩薩行，心無厭足；發廣大心，修習一切佛剎微塵等諸波羅蜜，度脫一切眾生，具佛十力，心無厭足。若善男子、善女人，成就如是十種大法，則能長養一切善根，離生死趣一切世間性，超出聲聞、緣覺之地；生如來家，具足成就菩薩大願，行菩薩行，住菩薩地，成就如來功德之力，降伏眾魔，制諸外道。彼諸比丘聞此法已，皆得無礙淨眼三昧；悉見十方一切如來，及其眷屬，無量眾生；亦見種種世界形類，眾寶宮殿及諸微塵；乃至如來十眼境界，皆悉覩見。彼諸如來，以種種句身、味身，種種辯才，微妙音聲，所說法海，皆悉聞知。彼世界中一切眾生，心念諸根，皆悉了知；知彼眾生，過去、未來，諸趣受生；又能知彼過去、未來，各十劫事；知彼如來十種本生、十種成就菩提自在、十種轉法輪、十種神力、十種教誡、十種說法、十種辯才。得此三昧時，具足成就十種實際菩提之心；一萬三昧，一萬淨波羅蜜，得大智慧圓滿光明，菩薩十明，住菩提心。}@
%{当时，文殊师利对诸比丘说：你们应该知道，如果善男子、善女人，成就十种大心，就能获得佛地，何况菩萨地；这十种大心是什么呢？就是说：发广大心，长养一切善根，究竟不退，心无厌足；见一切佛，恭敬供养，心无厌足；正求一切佛法，心无厌足；遍行菩萨诸波罗蜜，心无厌足；具足一切菩萨三昧，心无厌足；在一切三世流转中，心无厌足；严净佛刹，充满十方，心无厌足；教化成熟一切众生，心无厌足；在一切刹土、一切劫中，行菩萨行，心无厌足；发广大心，修习一切佛刹微尘等诸波罗蜜，度脱一切众生，具足佛十力，心无厌足。如果善男子、善女人，成就这样十种大法，就能长养一切善根，远离生死趣一切世间性，超越声闻、缘觉之地；生于如来家，具足成就菩萨大愿，行菩萨行，住菩萨地，成就如来功德之力，降伏众魔，制伏诸外道。那些比丘听到这些法以后，都获得无碍净眼三昧；都看见十方一切如来，及其眷属，无数众生；也看见种种世界形类，众多宝殿以及诸微尘；乃至如来十眼境界，都一一看见。那些如来，以种种句身、味身，种种辩才，微妙音声，所说法海，都一一闻知。那些世界中一切众生，心念诸根，都一一了知；知道那些众生，过去、未来，诸趣受生；还能知道那些众生过去、未来，各自十劫之事；知道那些如来十种本生、十种成就菩提自在、十种转法轮、十种神力、十种教诫、十种说法、十种辩才。获得这种三昧的时候，具足成就十种实际菩提之心；一万三昧，一万净波罗蜜，获得大智慧圆满光明，菩萨十明，住菩提心。}%
@{爾時，文殊師利菩薩勸諸比丘，修普賢行，住普賢行。}@
%{當時，文殊師利菩薩勸導這些比丘，要修習普賢行，並且一直保持普賢行。}%
@{彼諸比丘，出生大願海；生大願海已，身心清淨，得不死通明；得是明已，不離此處，出生一切如來法身，充滿十方，具足一切佛法。}@
%{這些比丘們，從此生起了廣大的願海，有了這個廣大的願海之後，身心變得清淨，得到了不死的智慧之光。有了這個智慧之光之後，他們沒有離開這個地方，而是從此生出了所有如來的法身，遍布十方世界，具備了所有佛法。}%

@{爾時，文殊師利菩薩建立彼諸比丘菩提心已，與其眷屬，漸遊南方。}@
%{當時，文殊師利菩薩幫助這些比丘建立了菩提心之後，就帶著他的眷屬，慢慢地向南方走去。}%
@{至覺城東，住莊嚴幢娑羅林中，大塔廟處，過去諸佛所遊止處；亦是過去諸佛為菩薩時，修苦行處。}@
%{他們來到了覺城的東邊，住在了莊嚴幢娑羅林中，大塔廟的地方，這裡是過去諸佛曾經遊玩停留過的地方，也是過去諸佛在做菩薩時，修行苦行的地方。}%
@{此處常為一切天、龍、夜叉、乾闥婆、阿脩羅、人非人等之所供養。}@
%{這個地方，一直以來都是天龍八部，包括天人、龍、夜叉、乾闥婆、阿修羅、人非人等等，供養的地方。}%

@{時，文殊師利即於此處，說普照一切法界修多羅，有百萬億修多羅，以為眷屬；說此法時，於大海中，有無量千億龍王與眷屬俱，來詣文殊師利。}@
%{當時，文殊師利菩薩就在這個地方，講述了普照一切法界的佛經，這些佛經有百萬億那麼多，他把這些佛經當成自己的眷屬。當他在講述這些佛經的時候，在海中，有無量千億的龍王，帶著他們的眷屬，一起來到文殊師利菩薩面前。}%
@{聞此法已，厭離龍趣，正求佛道；捨龍身已，生天人中；一萬龍王，於阿耨多羅三藐三菩提，得不退轉。}@
%{這些龍王們聽了佛經之後，厭惡了龍的這種低級生命，一心想要追求佛道。他們捨棄了龍的身體，轉生到天人中，其中一萬個龍王，在阿耨多羅三藐三菩提上，得到了不退轉的果位。}%

@{時，覺城人聞文殊師利在莊嚴幢娑羅林中，大塔廟處；聞已，優婆塞、優婆夷、童男、童女，皆悉往詣文殊師利。}@
%{當時，覺城的人們聽說文殊師利菩薩在莊嚴幢娑羅林中，大塔廟的地方，就都紛紛前去拜訪文殊師利菩薩。其中有優婆塞、優婆夷、童男、童女，都去拜訪了文殊師利菩薩。}%

@{時，有優婆塞，名曰大智，與千優婆塞眷屬俱，其名曰：須達多優婆塞、婆須達多優婆塞、功德光優婆塞、名稱德優婆塞、寂靜德優婆塞、歡喜德優婆塞、善慧優婆塞、大慧優婆塞、賢優婆塞、賢妙德優婆塞；如是等千優婆塞俱，頭面禮足，退坐一面。}@
%{當時，有一個優婆塞，名叫大智，帶著一千個優婆塞的眷屬，這些優婆塞的名字分別是：須達多優婆塞、婆須達多優婆塞、功德光優婆塞、名稱德優婆塞、寂靜德優婆塞、歡喜德優婆塞、善慧優婆塞、大慧優婆塞、賢優婆塞、賢妙德優婆塞等等，他們都一起來到文殊師利菩薩面前，頭頂腳趾，表示敬意，然後退到一邊坐下。}%
@{復有五百優婆夷，其名曰：大慧光優婆夷、善光優婆夷、善身優婆夷、可樂身優婆夷、跋陀羅優婆夷、賢德優婆夷、賢光優婆夷、光明幢優婆夷、妙德光優婆夷、善眼優婆夷；如是等五百優婆夷俱，頭面禮足，退坐一面。}@
%{此外，還有五百個優婆夷，她們的名字分別是：大慧光優婆夷、善光優婆夷、善身優婆夷、可樂身優婆夷、跋陀羅優婆夷、賢德優婆夷、賢光優婆夷、光明幢優婆夷、妙德光優婆夷、善眼優婆夷等等，她們也一起來到文殊師利菩薩面前，頭頂腳趾，表示敬意，然後退到一邊坐下。}%

@{復有五百童女，其名曰：善行童女、跋陀羅童女、悅樂顏童女、堅固慧童女、妙功德童女、勝體童女、梵天與童女、功德光童女、善光明童女；如是等五百童女俱，頭面禮足，退坐一面。}@
%{還有五百個童女，她們的名字分別是：善行童女、跋陀羅童女、悅樂顏童女、堅固慧童女、妙功德童女、勝體童女、梵天與童女、功德光童女、善光明童女等等，她們也一起來到文殊師利菩薩面前，頭頂腳趾，表示敬意，然後退到一邊坐下。}%

@{爾時，文殊師利知覺城大眾集已，隨其所應，以大慈力，令彼清涼；大悲現前，將為說法；甚深智慧，分別其心，以大辯力而為說法。}@
%{當時，文殊師利菩薩知道覺城的大眾都聚集起來了，就根據他們的需要，用大慈悲的力量，讓他們感到清涼，用大悲心，準備為他們講法。他用深奧的智慧，分別眾人的心，用辯才無礙的力量，開始講法。}%
@{觀察善財童子，以何因緣，名曰善財：此童子者，初受胎時，於其宅內，有七大寶藏；其藏普出七寶樓閣，自然周備，金、銀、瑠璃、玻瓈、真珠、硨磲、碼碯，從此七寶，生七種芽。}@
%{文殊師利菩薩觀察了善財童子，想知道他為什麼會被叫做善財，就發現這個童子在剛開始懷孕的時候，在他的家裡，就有了七個巨大的寶藏，這些寶藏自然地生出了七寶樓閣，裡面應有盡有，有金、銀、瑠璃、玻瓈、真珠、硨磲、碼碯，從這七種寶物中，生出了七種芽。}%
@{時，此童子處胎十月，出生端正，肢體具足；其七種寶芽，高二尋，廣七尋。}@
%{這個童子在母親肚子裡待了十個月之後，出生了，長得非常端正，身體四肢都非常健全。他身上的七種寶芽，每一株都高兩尋，寬七尋。}%
@{又其家內，自然具有五百寶器，盛滿眾寶：金器盛銀；銀器盛金；金剛器盛眾香；眾香器盛寶衣；玉石器盛上味饌；摩尼器盛雜寶；種種寶器盛酥油蜜，及以醍醐資生之具。}@
%{他的家裡，自然地就有了五百個寶器，裡面裝滿了各種寶物：金器裡面盛著銀子，銀器裡面盛著金子，金剛器裡面盛著各種香料，香料器裡面盛著寶衣，玉石器裡面盛著美味的食物，摩尼器裡面盛著各種寶石，各種寶器裡面盛著酥油蜜，以及醍醐等資生之具。}%
@{瑠璃器盛眾寶；玻瓈器盛硨磲；硨磲器盛玻瓈；碼碯器盛赤珠；赤珠器盛碼碯；火珠器盛淨水珠；淨水珠器盛火珠；如是等五百寶器，自然行列。}@
%{瑠璃器裡面盛著各種寶石，玻瓈器裡面盛著硨磲，硨磲器裡面盛著玻瓈，碼碯器裡面盛著赤珠，赤珠器裡面盛著碼碯，火珠器裡面盛著淨水珠，淨水珠器裡面盛著火珠，這五百個寶器，自然地排列成行。}%
@{又雨眾寶，滿諸庫藏。以此事故，婆羅門中，善明相師，字曰善財。}@
%{而且，他的家裡還經常下雨，下的是各種寶物，把所有的倉庫都填滿了。因為這些原因，婆羅門中的相師，就給他取名為善財。}%
@{此童子者，已曾供養過去諸佛，深種善根，常樂清淨；近善知識，身、口、意淨；修菩薩道，求一切智；修諸佛法，心淨如空，具菩薩行。}@
%{這個童子曾經供養過過去的諸佛，種下了深厚的善根，他經常感到快樂和清淨，他親近善知識，身、口、意都保持清淨，他修習菩薩道，追求一切智慧，他修習佛法，心像天空一樣清淨，具備了菩薩的行為。}%

@{爾時，文殊師利菩薩如象王迴，觀察善財而告之曰：吾當為汝說微妙法。}@
%{當時，文殊師利菩薩就像象王回頭一樣，觀察著善財童子，對他說：“我要給你講述微妙的佛法。”}%
@{即為分別諸佛正法；分別諸佛次興世法；淨眷屬法；轉梵輪法；諸佛色身相好，清淨莊嚴之法；一切諸佛具法身法；諸佛音聲妙莊嚴法；說一切如來平等正法。}@
%{於是，文殊師利菩薩開始分別講解諸佛的正法，分別講解諸佛依次興世的佛法，講解淨化眷屬的佛法，講解轉動法輪的佛法，講解諸佛色身相好，清淨莊嚴的佛法，講解一切諸佛具備法身的佛法，講解諸佛音聲妙不可言的佛法，講解一切如來平等的正法。}%
@{爾時，文殊師利知善財等一切大眾，聞說此法，皆大歡喜，發菩提心；顯明過去諸善根已，不捨本座；如應化度覺城眾生已，遊行南方。}@
%{当时，文殊师利知道善财等一切大众，听到佛说这些法，都非常欢喜，发起了菩提心；显明了过去的善根之后，没有离开自己的座位；就像应化度化觉城众生之后，到南方去游行。}%
