@{佛所行讚卷第五(亦云佛本行經)}@
%{佛所行讚卷第五(亦云佛本行經)}%

@{馬鳴菩薩造}@
%{馬鳴菩薩造}%

@{爾時鞞舍離，諸離車長者，}@
%{当时舍卫城，离车长者，}%
@{聞世尊入國，住菴摩羅園。}@
%{听说世尊入城，住在菴摩罗园。}%

@{有乘素車輿，素蓋素衣服，}@
%{有人乘坐白色车子，白色车盖，白色衣服，}%
@{青赤黃綠色，其眾各異儀。}@
%{青色红色黄色绿色，他们各自仪容不同。}%

@{導從翼前後，爭塗競路前，}@
%{导从左右前后，争着赶路向前，}%
@{天冠袞花服，寶飾以莊嚴，}@
%{天冠衮花服饰，宝饰来装饰，}%
@{威容盛明曜，增暉彼園林。}@
%{威容盛大光明耀眼，增辉了那园林。}%

@{除捨五威儀，下車而步進，}@
%{放下五种威仪，下车步行进入，}%
@{息慢而形恭，頂禮於佛足，}@
%{息慢而恭敬，顶礼佛足，}%
@{大眾圍遶佛，如日重輪光。}@
%{大众围绕着佛，如同太阳重轮的光芒。}%

@{離車名師子，為諸離車長，}@
%{离车名叫师子，是离车长者，}%
@{德貌如師子，位居師子臣，}@
%{德貌如同师子，位居师子臣，}%
@{滅除師子慢，受誨釋師子。}@
%{灭除师子慢，接受教诲释师子。}%

@{汝等大威德，名族美色容，}@
%{你们拥有大威德，名族美色容貌，}%
@{能除世憍慢，受法以增明，}@
%{能除世间憍慢，接受佛法以增光明，}%
@{財色香花飾，不如戒莊嚴。}@
%{财富色香花饰，不如戒律庄严。}%

@{國土豐安樂，唯以汝等榮，}@
%{国土富足安乐，只因为你们荣耀，}%
@{榮身而安民，在於調御心。}@
%{荣耀自身而安抚百姓，在于调御人心。}%

@{加以樂法情，令德轉崇高，}@
%{加上热爱佛法的情怀，令德行更加崇高，}%
@{非薄土群鄙，而能集眾賢。}@
%{并非薄土群鄙，而是能聚集众多贤才。}%

@{當日新其德，撫養於萬民，}@
%{应当日新其德，抚养万民，}%
@{導眾以明正，如牛王涉津。}@
%{引导大众以明正道，如同牛王涉过渡口。}%

@{若人能自念，今世及後世，}@
%{如果人能自己反省，今世和后世，}%
@{唯當脩正戒，福利二世安。}@
%{唯应当修持正戒，福利二世安稳。}%

@{為眾所敬重，名稱普流聞，}@
%{被大众敬重，名声广泛流传，}%
@{仁者樂為友，德流永無疆。}@
%{仁者乐于为友，德行流传永无止境。}%

@{山林寶玉石，皆依地而生，}@
%{山林宝玉石，都依附大地而生，}%
@{戒德亦如地，眾善之所由。}@
%{戒德也如同大地，是所有善行的根源。}%

@{無翅欲騰虛，渡河無良舟，}@
%{没有翅膀想要飞上天空，渡河没有好的船，}%
@{人而無戒德，濟苦為實難。}@
%{人如果没有戒德，济度苦难就真的很难。}%

@{如樹美花果，針刺難可攀，}@
%{如同树木开花结果，针刺难以攀爬，}%
@{多聞美色力，破戒者亦然。}@
%{多闻美色力量，破戒者也是如此。}%

@{端坐勝堂閣，王心自莊嚴，}@
%{端坐在高高的楼阁里，国王内心自然庄严，}%
@{淨戒功德具，隨大仙而征。}@
%{净戒功德具足，跟随大仙征战。}%

@{染服衣毛羽，螺髻剃鬚髮，}@
%{染色的衣服，羽毛，螺髻，剃须头发，}%
@{不脩於戒德，方涉眾苦難。}@
%{不修持戒德，就将陷入众苦难中。}%

@{日夜三沐浴，奉火修苦行，}@
%{日夜三次沐浴，奉火修行苦行，}%
@{遺身穢野獸，赴水火投巖。}@
%{遗弃身体成为污秽的野兽，投身水火，投身悬崖。}%

@{食菓餌草根，吸風飲恒水，}@
%{吃果实，吃草根，吸风饮恒水，}%
@{服氣以絕糧，遠離於正戒。}@
%{服气以绝粮，远离正戒。}%

@{習斯禽獸道，非為正法器，}@
%{习此禽兽之道，并非正法器，}%
@{毀戒招誹謗，仁者所不親。}@
%{毁戒招来诽谤，仁者所不亲近。}%

@{心常懷恐怖，惡名如影隨，}@
%{心中常常怀着恐惧，恶名如同影子跟随，}%
@{現世無利益，後世豈獲安？}@
%{今世没有利益，后世又怎么能够安稳呢？}%

@{是故智慧士，當修於淨戒，}@
%{所以智慧之人，应当修持净戒，}%
@{於生死曠野，戒為善導師。}@
%{在生死旷野中，戒律是善的导师。}%

@{持戒由自力，此則不為難，}@
%{持戒靠自身努力，这就不难，}%
@{淨戒為梯隥，令人上昇天。}@
%{净戒是通向天上的梯子，让人上升天堂。}%

@{建立戒者，斯由煩惱微，}@
%{建立戒律的人，是因为烦恼微弱，}%
@{諸過壞其心，喪失善功德。}@
%{各种过失破坏了他们的心，丧失了善功德。}%

@{先當離我所，我所覆諸善，}@
%{首先应当远离我所，我所遮蔽了所有善行，}%
@{猶灰覆火上，足蹈而覺燒。}@
%{如同灰烬覆盖在火上，脚踩上去就会感觉燃烧。}%

@{憍慢覆其心，如日隱重雲，}@
%{憍慢遮蔽了他们的心，如同太阳隐藏在厚厚的云层中，}%
@{慢怠滅慚愧，憂悲弱強志，}@
%{怠慢消除了羞愧，忧愁悲伤削弱了意志，}%
@{老病壞壯容，我慢滅諸善。}@
%{老病破坏了壮年容貌，我慢消除了所有善行。}%

@{諸天阿修羅，貪嫉興諍訟，}@
%{诸天阿修罗，贪婪嫉妒引发争讼，}%
@{喪失諸功德，悉由我慢懷。}@
%{丧失了所有功德，都是因为我慢的心。}%

@{我於勝中勝，我德勝者同，}@
%{我在最胜中是最胜，我的德行与最胜者相同，}%
@{我於勝小劣，斯則為愚夫，}@
%{我在最胜中是最劣，这便是愚夫，}%
@{色族悉無常，動搖不暫停，}@
%{色身和家族都无常，不停地动摇，}%
@{終為磨滅法，何用憍慢為？}@
%{最终都会成为磨灭的法，何必用憍慢来装饰呢？}%

@{貪欲為巨患，詐親而密怨，}@
%{贪欲是巨大的祸患，欺骗亲近之人而暗中怨恨，}%
@{猛火從內發，貪火亦復然。}@
%{猛火从内部爆发，贪火也是如此。}%

@{貪欲之熾燃，甚於世界火，}@
%{贪欲的炽热燃烧，比世间之火更加猛烈，}%
@{火盛水能滅，貪愛難可消。}@
%{火势旺盛水能灭，贪爱却难以消除。}%

@{猛火焚曠野，草盡還復生，}@
%{猛火焚烧旷野，草木烧尽后还会再生，}%
@{貪欲火焚心，正法生則難。}@
%{贪欲之火焚烧内心，正法再生就难了。}%

@{貪欲求世樂，樂增不淨業，}@
%{贪欲追求世间快乐，快乐却增加了不净业，}%
@{惡業墮惡道，怨無過貪欲。}@
%{恶业堕落恶道，怨恨没有比贪欲更严重的。}%

@{貪則生於愛，愛則習諸欲，}@
%{贪婪就会产生爱，爱就会习染各种欲望，}%
@{習欲招眾苦，元惡無過貪，}@
%{习染欲望招致各种苦难，根本的恶行没有比贪婪更严重的，}%
@{貪則為大病，智藥愚夫止，}@
%{贪婪是大的病症，智慧的良药愚蠢之人却止步不前，}%
@{邪覺不正思，能令貪欲增。}@
%{邪觉不正思维，能令贪欲增长。}%

@{無常苦不淨，無我無我所，}@
%{无常苦不净，没有我，没有我所，}%
@{智慧真實觀，能滅彼邪貪。}@
%{智慧真实观，能够灭除那些邪恶的贪婪。}%

@{是故於境界，當修真實觀，}@
%{所以面对境界，应当修持真实观，}%
@{真實觀已生，貪欲得解脫。}@
%{真实观已经生起，贪欲就能得到解脱。}%

@{見德生貪欲，見過起瞋恚，}@
%{看到德行就生起贪欲，看到过失就生起瞋恚，}%
@{德過二俱忘，貪恚得除滅。}@
%{德行和过失两者都忘记，贪欲和瞋恚就能消除。}%

@{瞋恚改素容，能壞端正色，}@
%{瞋恚改变了本来的容貌，能够破坏端正的脸色，}%
@{瞋恚翳明目，害法義欲聞，}@
%{瞋恚遮蔽了明亮的眼睛，妨碍了想要听闻佛法义理，}%
@{斷絕親愛義，為世所輕賤。}@
%{断绝了亲爱之义，被世人轻贱。}%

@{是故當捨恚，勿隨於瞋心，}@
%{所以应当舍弃瞋恚，不要跟随瞋恚之心，}%
@{能制狂恚心，是名善御者，}@
%{能够控制狂暴的瞋恚之心，这叫做善于驾驭者，}%
@{世稱善調駟，是為攝繩容。}@
%{世人称赞善于驾驭四匹马，这就是收敛绳索的容貌。}%

@{縱恚不自禁，憂悔火隨燒，}@
%{纵容瞋恚而不加控制，忧悔之火就会随之燃烧，}%
@{若人起瞋恚，先自燒其心，}@
%{如果人起了瞋恚，首先会烧毁自己的心，}%
@{然後加於彼，或燒或不燒。}@
%{然后加于他人，或者烧毁，或者不烧毁。}%

@{生老病死苦，逼迫於眾生，}@
%{生老病死苦，逼迫着众生，}%
@{復加於恚害，多怨復增怨。}@
%{再加上瞋恚的伤害，怨恨越来越多，怨恨又增加了怨恨。}%

@{見世眾苦迫，應起慈悲心，}@
%{看到世间大众受苦迫，应当生起慈悲心，}%
@{眾生起煩惱，增微無量差。}@
%{众生起了烦恼，增加了无量无边的痛苦。}%

@{如來善方便，隨病而略說，}@
%{如来善于方便，根据病症而略加开示，}%
@{譬如世良醫，隨病而投藥。}@
%{譬如世间良医，根据病症而投药。}%

@{爾時諸離車，聞佛所說法，}@
%{当时诸离车，听到佛所说法，}%
@{即起禮佛足，歡喜而頂受。}@
%{就起身礼拜佛足，欢喜地接受。}%

@{請佛及大眾，明日設薄供。}@
%{邀请佛陀和大众，明天设薄供。}%

@{佛告諸離車：菴摩羅已請。}@
%{佛告诉诸离车：菴摩罗已经邀请了。}%

@{離車懷感愧，彼何奪我利？}@
%{离车怀着惭愧，他怎么夺走了我的利益呢？}%
@{知佛心平等，而起隨喜心。}@
%{知道佛陀心怀平等，就生起了随喜之心。}%

@{如來善隨宜，安慰令心悅，}@
%{如来善于随顺，安慰令他们心悦，}%
@{伏化純熟歸，如蛇被嚴呪。}@
%{伏化纯熟归依，如同蛇被严咒。}%

@{夜過明相生，佛與大眾俱，}@
%{夜晚过去，光明诞生，佛陀和大众一起，}%
@{詣菴摩羅舍，受彼供養畢，}@
%{前往菴摩罗的住所，接受他们的供养完毕，}%
@{往詣毘紐村，於彼夏安居。}@
%{前往毘纽村，在那里夏安居。}%

@{三月安居竟，復還鞞舍離，}@
%{三个月安居完毕，又回到舍卫城，}%
@{住獼猴池側，坐於林樹間。}@
%{住在猕猴池旁边，坐在树林中。}%

@{普放大光明，以感魔波旬，}@
%{普放大光明，以感化魔波旬，}%
@{來詣於佛所，合掌勸請言：}@
%{前来佛陀面前，合掌劝请说：}%

@{昔尼連禪側，已發真實要，}@
%{从前在尼连禅河边，已经发誓了真实的誓言，}%
@{我所作事畢，當入於涅槃。}@
%{我所做的事情已经做完了，应当进入涅槃。}%

@{今所作已作，當遂於本心。}@
%{现在所做的事情已经做完了，应当顺从本心。}%

@{時佛告波旬：滅度時不遠，}@
%{这时佛陀告诉波旬：灭度的时候不远了，}%
@{却後三月滿，當入於涅槃。}@
%{再过三个月满，就应当进入涅槃。}%

@{時魔知如來，滅度已有期，}@
%{这时魔知道如来，灭度已经有了期限，}%
@{情願既已滿，歡喜還天宮。}@
%{情愿已经满足，欢喜地回到天宫。}%

@{如來坐樹下，正受三摩提，}@
%{如来坐在树下，正在受三摩地，}%
@{放捨業報壽，神力住命存。}@
%{放下业报寿数，神力住命存活。}%

@{以如來捨壽，大地普震動，}@
%{因为如来舍弃了寿数，大地普遍震动，}%
@{十方虛空境，周遍大火然。}@
%{十方虚空境界，周遍大火燃烧。}%

@{須彌頂崩頹，天雨飛礫石，}@
%{须弥山顶崩塌，天降飞石，}%
@{狂風四激起，樹木悉摧折，}@
%{狂风四面激起，树木都摧折了，}%
@{天樂發哀聲，天人心忘歡。}@
%{天乐发出哀伤的声音，天人的心忘记了欢喜。}%

@{佛從三昧起，普告諸眾生：}@
%{佛陀从三昧中醒来，普告所有众生：}%
@{我今已捨壽，三昧力存身，}@
%{我今天已经舍弃了寿数，三昧力支撑着身体，}%
@{身如朽敗車，無復往來因，}@
%{身体如同腐朽的车，不再有来去的缘由，}%
@{已脫於三有，如鳥破卵生。}@
%{已经脱离了三有，如同小鸟破卵而出。}%
@{佛所行讚離車辭別品第二十四}@
%{佛所行讚，离车辞别品第二十四}%

@{尊者阿難陀，見地普大動，}@
%{尊者阿难陀，看到大地剧烈震动，}%

@{心驚身毛竪，問佛何因緣？}@
%{心中害怕，毛发直竖，就问佛陀这是为什么？}%

@{佛告阿難陀：我住三月壽，}@
%{佛陀告诉阿难陀：我住世的时间只有三个月了，}%

@{餘命行悉捨，是故地大動。}@
%{剩下的时间我都要用来修行，所以大地才会震动。}%

@{阿難聞佛教，悲感淚交流，}@
%{阿难陀听到佛陀的教诲，悲痛得眼泪止不住地流，}%

@{猶如大力象，搖彼栴檀樹，}@
%{就像一只大力象，摇动着檀香树，}%

@{擾動理迫迮，香汁淚流下。}@
%{香树被摇动，香汁也流出来了，就像阿难陀流下的眼泪一样。}%

@{親重大師尊，恩深未離欲，}@
%{亲近伟大的师尊，恩情深厚，还没能摆脱欲望，}%

@{惟此四事故，悲苦不自勝。}@
%{只因为这四件事，阿难陀悲痛得无法控制自己。}%

@{今我聞世尊，涅槃決定教，}@
%{现在我听到世尊决定要涅槃的教诲，}%

@{舉體悉萎消，迷方失常音，}@
%{我的整个身体都变得虚弱，好像迷失了方向，失去了正常的音调，}%

@{所聞法悉忘，荒悸亡天地。}@
%{以前听到的法也全部忘记了，感到迷茫，好像天地都消失了。}%

@{怪哉救世主，滅度一何駃，}@
%{真是奇怪啊，救世主竟然要涅槃了，速度如此之快，}%

@{遭寒水垂死，遇火忽復滅。}@
%{就像掉进冰冷的水里快要死了，遇到火又突然灭了一样。}%

@{於煩惱曠野，迷亂失其方，}@
%{在烦恼的旷野里，迷失了方向，}%

@{忽遇善導師，未度忽復失。}@
%{突然遇到了善知识，还没来得及渡过，就又失去了。}%

@{如人涉長漠，熱渴久乏水，}@
%{就像一个人走在沙漠里，又热又渴，很久没有喝到水，}%

@{忽遇清涼池，奔趣悉枯竭。}@
%{突然遇到了一片清凉的池水，却发现它已经干涸了。}%

@{紺睫瞪睛目，明鑒於三世，}@
%{黑色的眼睫毛，瞪着明亮的眼睛，能明察三世，}%

@{智慧照幽冥，昏冥一何速？}@
%{智慧照亮幽冥，怎么会如此迅速地消失呢？}%

@{知佛有遺骼，神龍所供養，}@
%{但他知道佛陀的遗骨被神龙供养着，}%
@{增其信敬心。雖王領國土，}@
%{更加坚定了他的信仰和敬意。虽然国王统治着国家，}%
@{逮得初聖果，能令普天下，}@
%{却已经获得了初圣果位，能够让普天下的百姓，}%
@{供養如來塔，去來今現在，}@
%{供养如来的塔，无论过去、现在、未来，}%
@{悉皆得解脫。如來現在世，}@
%{都能获得解脱。如来现在还在世，}%
@{涅槃及舍利，恭敬供養者，}@
%{对他的涅槃和舍利恭敬供养的人，}%
@{其福等無異。明慧增上心，}@
%{他们的福报是一样的。如果拥有明慧增上的心，}%
@{深察如來德，懷道興供養，}@
%{深入了解如来的功德，怀着对佛法的敬畏之心兴建供养，}%
@{其福亦俱勝。佛得尊勝法，}@
%{他们的福报会更加殊胜。佛陀获得了最尊贵的佛法，}%
@{應受一切供，已到不死處，}@
%{应该受到一切供养，已经到达了不死之地，}%
@{信者亦隨安，是故諸天人，}@
%{信佛的人也会随之安稳。所以，诸天人}%
@{悉應常供養。}@
%{都应该经常供养他。}%

@{第一大慈悲，通達第一義，}@
%{佛陀拥有无与伦比的大慈悲，通达宇宙人生的终极真理，}%
@{度一切眾生，孰聞而不感。}@
%{普度一切众生，谁听了不会感动呢？}%

@{生老病死苦，世間苦無過，}@
%{生老病死，世间苦难无过于此，}%
@{死苦苦之大，諸天之所畏。}@
%{死亡的痛苦最大，连天人都感到害怕。}%

@{永離二種苦，云何不供養？}@
%{能够永远摆脱这两种痛苦，怎么能够不供养呢？}%

@{不受後有樂，世間樂無上，}@
%{不受后世轮回之苦，世间快乐无过于此，}%
@{增生苦之大，世間苦無比；}@
%{但轮回的痛苦最大，世间痛苦无与伦比。}%

@{佛得離生苦，不受後有樂，}@
%{佛陀能够摆脱轮回的痛苦，不受后世轮回之苦，}%
@{為世廣顯示，如何不供養？}@
%{为了世人广为宣示，怎么能够不供养呢？}%

@{讚諸牟尼尊，始終之所行，}@
%{赞叹所有佛陀，他们一直以来所做的一切，}%
@{不自顯知見，亦不求名利，}@
%{不炫耀自己的智慧和见解，也不追求名利，}%
@{隨順佛經說，以濟諸世間。}@
%{只是顺应佛经所说，来救济世间众生。}%
