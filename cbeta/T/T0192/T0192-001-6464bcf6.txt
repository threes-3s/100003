@{甘蔗之苗裔，釋迦無勝王，}@
%{甘蔗的苗裔，释迦无胜王，}%

@{淨財德純備，故名曰淨飯，}@
%{净财德纯备，所以叫净饭，}%

@{群生樂瞻仰，猶如初生月。}@
%{众生都乐于瞻仰，就像初生的月亮一样。}%

@{王如天帝釋，夫人猶舍脂，}@
%{国王就像天帝释，夫人就像舍脂，}%

@{執志安如地，心淨若蓮花，}@
%{执志安稳如大地，内心纯净像莲花，}%

@{假譬名摩耶，其實無倫比。}@
%{假使说她叫摩耶，其实无与伦比。}%

@{於彼象天后，降神而處胎。}@
%{在那个像天后一样的女人身上，神灵降临，入胎修行。}%

@{母悉離憂患，不生幻偽心，}@
%{母亲完全远离忧患，不生虚假的心，}%

@{厭惡彼諠俗，樂處空閑林。}@
%{厌恶那个喧嚣俗世，喜欢待在清闲的森林里。}%

@{藍毘尼勝園，流泉花果茂，}@
%{蓝毗尼胜园，泉水流淌，花果茂盛，}%

@{寂靜順禪思，啟王請遊彼。}@
%{寂静顺应禅思，启请国王去那里游玩。}%

@{王知其志願，而生奇特想，}@
%{国王知道她的心愿，就产生了奇特的念头，}%

@{勅內外眷屬，俱詣彼園林。}@
%{命令内外眷属，都到那个园林里去。}%

@{爾時摩耶后，自知產時至，}@
%{那时摩耶后，自己知道生产的时候到了，}%

@{偃寢安勝床，百千婇女侍。}@
%{就安卧在舒适的床上，上千个侍女侍候着。}%

@{時四月八日，清和氣調適，}@
%{那是四月初八，天气清和，气候宜人，}%

@{齋戒修淨德，菩薩右脇生，}@
%{斋戒修行净德，菩萨从右胁出生，}%

@{大悲救世間，不令母苦惱。}@
%{大悲救度世间，不让母亲痛苦。}%

@{優留王股生，卑偷王手生，}@
%{优留王从大腿出生，卑偷王从手出生，}%

@{曼陀王頂生，伽叉王腋生。}@
%{曼陀王从头顶出生，伽叉王从腋下出生。}%

@{菩薩亦如是，誕從右脇生，}@
%{菩萨也是这样，从右胁出生，}%

@{漸漸從胎出，光明普照耀，}@
%{慢慢地从胎里出来，光明普照四方，}%

@{如從虛空墮，不由於生門。}@
%{就像从虚空坠落，不是从生门出来的。}%

@{修德無量劫，自知生不死，}@
%{修行无量劫，自己知道生而不死，}%

@{安諦不傾動，明顯妙端嚴。}@
%{安然自在，不为外物所动，显露着奇妙的庄严。}%

@{晃然後胎現，猶如日初昇，}@
%{晃眼之后胎体显现，就像太阳初升一样，}%

@{觀察極明耀，而不害眼根。}@
%{观察起来极其明亮，却不会伤害眼睛。}%

@{縱視而不耀，如觀空中月，}@
%{纵目观看也不耀眼，就像观看空中的月亮一样，}%

@{自身光照耀，如日奪燈明，}@
%{自身的光芒照耀，就像太阳夺去了灯火的光明一样，}%

@{菩薩真金身，普照亦如是。}@
%{菩萨的真金身，普照四方也是这样。}%

@{正真心不亂，安庠行七步，}@
%{内心安然无乱，安详地走了七步，}%

@{足下安平趾，炳徹猶七星。}@
%{脚下的脚趾安平，明亮如同七颗星星一样。}%

@{獸王師子步，觀察於四方，}@
%{兽王狮子的步伐，观察着四方，}%

@{通達真實義，堪能如是說。}@
%{通达真实的道理，能够这样说。}%

@{此生為佛生，則為後邊生，}@
%{这一生是佛生，那么就是后边生，}%

@{我唯此一生，當度於一切。}@
%{我只这一生，就要度化一切众生。}%

@{應時虛空中，淨水雙流下，}@
%{应时虚空之中，清净的水从两边流下来，}%

@{一溫一清涼，灌頂令身樂。}@
%{一边温暖一边清凉，灌顶让身体感到快乐。}%

@{安處寶宮殿，臥於琉璃床，}@
%{安卧在宝宫殿里，躺在琉璃床上，}%

@{天王金華手，奉持床四足。}@
%{天王用金色的花手，捧着床的四只脚。}%

@{諸天於空中，執持寶蓋侍，}@
%{诸天在空中，拿着宝盖侍候着，}%

@{承威神讚歎，勸發成佛道。}@
%{承接威神赞叹，劝发成佛道。}%

@{諸龍王歡喜，渴仰殊勝法，}@
%{诸龙王欢喜，渴望着殊胜的法，}%

@{曾奉過去佛，今得值菩薩。}@
%{曾经供奉过过去的佛，现在有幸遇到了菩萨。}%

@{散曼陀羅花，專心樂供養，}@
%{散播曼陀罗花，专心快乐地供养，}%

@{如來出興世，淨居天歡喜。}@
%{如来出兴于世，净居天欢喜。}%

@{已除愛欲歡，為法而欣悅，}@
%{已经除去了爱欲的欢喜，为了法而欣悦，}%

@{眾生沒苦海，令得解脫故。}@
%{众生沉没在苦海里，为了让他们得到解脱。}%

@{須彌寶山王，堅持此大地，}@
%{须弥宝山王，支撑着这大地，}%

@{菩薩出興世，功德風所飄，}@
%{菩萨出兴于世，功德之风吹拂，}%

@{普皆大震動，如風鼓浪舟。}@
%{普天之下都震动，就像风鼓动着浪船一样。}%

@{栴檀細末香，眾寶蓮花藏，}@
%{檀香细末香，众宝莲花藏，}%

@{風吹隨空流，繽紛而亂墜，}@
%{风吹着随空飘散，缤纷乱坠，}%

@{天衣從空下，觸身生妙樂。}@
%{天衣从空中飘落下来，触碰到身体就产生奇妙的快乐。}%

@{日月如常度，光耀倍增明，}@
%{日月如常运转，光耀倍增光明，}%

@{世界諸火光，無薪自炎熾，}@
%{世界上的火光，没有柴薪却自行燃烧，}%

@{淨水清涼井，前後自然生。}@
%{清净清凉的井，前后自然地涌现出来。}%

@{中宮婇女眾，怪歎未曾有，}@
%{宫中的侍女们，惊叹从来没有见过这样的景象，}%

@{競赴而飲浴，皆起安樂想。}@
%{争先恐后地去饮用和沐浴，都产生了安乐的念头。}%

@{無量部多天，樂法悉雲集，}@
%{无量部多天，乐于法都云集，}%

@{於藍毘尼園，遍滿林樹間。}@
%{在蓝毗尼园，遍布林木之间。}%

@{奇特眾妙花，非時而敷榮，}@
%{奇特的众妙花，非时节而开放繁荣，}%

@{凶暴眾生類，一時生慈心，}@
%{凶暴的众生，一时都生起了慈悲心，}%

@{世間諸疾病，不療自然除。}@
%{世间上的疾病，不用治疗自然就消失了。}%

@{亂鳴諸禽獸，恬默寂無聲，}@
%{乱鸣的禽兽，都安静默然，没有声音，}%

@{萬川皆停流，濁水悉澄清，}@
%{万川都停止了流动，浑浊的水都变得清澈，}%

@{空中無雲翳，天鼓自然鳴。}@
%{空中没有云翳，天鼓自然地敲响。}%

@{一切諸世間，悉得安隱樂，}@
%{一切世间，都得到了安宁快乐，}%

@{猶如荒難國，忽得賢明主。}@
%{就像荒乱的国家，突然得到了贤明的君主。}%

@{菩薩所以生，為濟世眾苦，}@
%{菩萨所以出生，是为了救济世间众生的苦难，}%

@{唯彼魔天王，震動大憂惱。}@
%{只有那个魔王，震动不安，非常忧愁。}%

@{父王見生子，奇特未曾有，}@
%{父王看到生下儿子，奇特无比，}%

@{素性雖安重，驚駭改常容，}@
%{本来性格沉稳，却惊恐万分，改变了常态，}%

@{二息交胸起，一喜復一懼。}@
%{呼吸急促，一会儿高兴，一会儿又害怕。}%

@{夫人見其子，不由常道生，}@
%{夫人看到儿子，不是从常道出生，}%

@{女人性怯弱，怵惕懷氷炭，}@
%{女人天性胆小，心惊胆战，就像怀着冰炭一样，}%

@{不別吉凶相，反更生憂怖。}@
%{分不清吉凶，反而更加忧愁害怕。}%

@{長宿諸母人，互亂祈神明，}@
%{长辈们，互相混乱地祈求神明，}%

@{各請常所事，願令太子安。}@
%{各自请求自己常做的事情，希望太子平安。}%

@{時彼林中有，知相婆羅門，}@
%{那时，在那个森林里，有一个懂得相术的婆罗门，}%

@{威儀具多聞，才辯高名稱，}@
%{仪容举止端庄，博学多闻，才辩高超，名声远扬，}%

@{見相心歡喜，踊躍未曾有。}@
%{看到相貌，心里欢喜，高兴得前所未有。}%

@{知王心驚怖，白王以真實：}@
%{知道国王心里害怕，就对国王说真话：}%

@{人生於世間，唯求殊勝子。}@
%{人生在世，只求得到优秀的儿子。}%

@{王今如滿月，應生大歡喜，}@
%{你现在就像满月一样，应该生出大欢喜，}%

@{今生奇特子，必光顯宗族。}@
%{现在生下奇特的儿子，一定会光耀宗族。}%

@{安心自欣慶，莫生餘疑慮，}@
%{安心自豪，不要再有其他的疑虑，}%

@{靈祥集家國，從今轉休盛。}@
%{吉祥降临家国，从今以后国运昌盛。}%

@{所生殊勝子，必為世間救，}@
%{生下的优秀儿子，一定会成为世间的救世主，}%

@{惟此上士身，金色妙光明，}@
%{只有这上士之身，金色的妙光照耀，}%

@{如是殊勝相，必成等正覺；}@
%{这样的殊胜相貌，一定会成等正觉；}%

@{若習樂世間，必作轉輪王，}@
%{如果喜欢世间，一定会成为转轮王，}%

@{普為大地主，勇猛正法治，}@
%{普天之下都是他的领土，以勇猛的正法治理，}%

@{王領四天下，統御一切王。}@
%{统治着四大部洲，统御着所有的国王。}%

@{猶如世光明，日光為最勝，}@
%{就像世间的光明，日光是最胜的，}%

@{若處於山林，專心求解脫。}@
%{如果住在山林里，专心求解脱。}%

@{成就實智慧，普照於世間，}@
%{成就了真实的智慧，普照世间，}%

@{譬如須彌山，普為諸山王。}@
%{比如须弥山，是所有山王的君主。}%

@{眾寶金為最，眾流海為最，}@
%{众宝中金子是最珍贵的，众流中大海是最广阔的，}%

@{諸宿月為最，諸明日為最，}@
%{诸星宿中月亮是最明亮的，诸星宿中太阳是最耀眼的，}%

@{如來處世間，兩足中為最。}@
%{如来处在世间，是两足中最高贵的。}%

@{淨目脩且廣，上下瞬長睫，}@
%{眼睛清净修长，上下眼睫毛很长，}%

@{瞪矚紺青色，明煥半月形，}@
%{瞪眼注视，眼珠是深青色的，明亮如同半月形，}%

@{此相云何非，平等殊勝目？}@
%{这样的相貌，怎么不是平等殊胜的眼睛呢？}%

@{時王告二生：若如汝所說，}@
%{这时国王对婆罗门说：如果像你所说，}%

@{如此奇特相，以何因緣故，}@
%{如此奇特的相貌，是因为什么原因，}%

@{不應於先王，乃現於我世？}@
%{不应该出现在先王身上，而出现在我这一代呢？}%

@{婆羅門白王：不應如是說。}@
%{婆罗门对国王说：不应该这样说。}%

@{多聞與智慧，名稱及事業，}@
%{博学多闻和智慧，名声和事业，}%

@{如是四事者，不應顧先後。}@
%{这四件事，不应该顾虑先后。}%

@{物性之所生，各從因緣起，}@
%{事物的本性，都是从因缘而生，}%

@{今當說諸譬，王今且諦聽。}@
%{现在就说一些比喻，你仔细听。}%

@{毘求央耆羅，此二仙人族，}@
%{毘求央耆罗，这两个仙人家族，}%

%{经历了漫长的岁月，各自生下了不同的儿子。}%

@{毘利訶鉢低，及與儵迦羅，}@
%{毘利訶鉢低，以及儵迦羅，}%

%{萨罗萨仙人，经论很久就断绝了，}%

@{而生婆羅婆，續復明經論。}@
%{却生下了婆罗婆，继续阐明经论。}%

@{現在知見生，不必由先胄，}@
%{现在有智慧的人出现，不必从祖先那里继承，}%

@{毘耶娑仙人，多造諸經論，}@
%{毘耶娑仙人，创作了很多经论，}%

@{末後胤跋彌，廣集偈章句。}@
%{最后他的后代跋弥，广泛收集了偈颂和章句。}%

@{阿低利仙人，不解醫方論，}@
%{阿低利仙人，不懂医方论，}%

@{後生阿低離，善能治百病。}@
%{后来生下了阿低离，精通医术，能够治疗百病。}%

@{二生駒尸仙，不閑外道論，}@
%{二生駒尸仙，不熟悉外道论，}%

@{後伽提那王，悉解外道法。}@
%{后来伽提那王，精通外道法。}%

@{甘蔗王始族，不能制海潮，}@
%{甘蔗王最初的家族，不能控制海潮，}%

@{至娑伽羅王，生育千王子，}@
%{到了娑伽羅王，生下了上千个王子，}%

@{能制大海潮，使不越常限。}@
%{能够控制大海潮，使它不越过常限。}%

@{闍那駒仙人，無師得禪道。}@
%{闍那駒仙人，没有老师就得到了禅道。}%

@{凡得名稱者，皆生於自力，}@
%{凡是得到名声的人，都是靠自己的努力，}%

@{或先勝後劣，或先劣後勝。}@
%{有的先优秀后衰败，有的先衰败后优秀。}%

@{帝王諸神仙，不必承本族，}@
%{帝王和神仙，不必继承祖先，}%

@{是故諸世間，不應顧先後。}@
%{所以世间上，不应该顾虑先后。}%

@{大王今如是，應生歡喜心，}@
%{大王现在就应该生出欢喜心，}%

@{以心歡喜故，永離於疑惑。}@
%{因为内心欢喜，所以永远远离疑惑。}%

@{王聞仙人說，歡喜增供養。}@
%{国王听到仙人所说，欢喜地增加了供养。}%

@{我今生勝子，當紹轉輪位，}@
%{我生下了优秀的儿子，应该继承转轮王的位子，}%

@{我年已朽邁，出家修梵行，}@
%{我年老体衰，出家修行梵行，}%

@{無令聖王子，捨世遊山林。}@
%{不要让圣王子舍弃世间去游山玩水。}%

%{那时，在园林附近，有一个苦行仙人，}%

@{名曰阿私陀，善解於相法，}@
%{名叫阿私陀，精通相术，}%

@{來詣王宮門，王謂梵天應。}@
%{来到王宫门口，国王认为是梵天降临。}%

%{苦行和乐于正法，这两个相貌都出现了，}%

%{就请他进入宫内，恭敬地设供养，}%

@{將入內宮中，唯樂見王子。}@
%{将要进入内宫，只希望见到王子。}%

@{雖有婇女眾，如在空閑林，}@
%{虽然有侍女们，却像在清闲的森林里一样，}%

@{安處正法座，加敬尊奉事，}@
%{安然坐在正法座上，更加恭敬地侍奉，}%

@{如安低牒王，奉事波尸吒。}@
%{就像安低牒王侍奉波尸吒一样。}%

@{時王白仙人：我今得大利。}@
%{这时国王对仙人说：我现在得到了大好处。}%

@{勞屈大仙人，辱來攝受我，}@
%{劳烦您，大仙人，屈尊来到这里，摄受我，}%

@{諸有所應為，唯願時教勅。}@
%{所有应该做的事情，只希望您随时教诲。}%

@{如是勸請已，仙人大歡喜：}@
