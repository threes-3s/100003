@{馬鳴菩薩造}@
%{馬鳴菩薩造}%

@{太子辭王師，及正法大臣，}@
%{太子辞别老师和正法大臣，}%
@{冒浪濟恒河，路由靈鷲巖。}@
%{冒着风浪渡过恒河，一路经过灵鹫山。}%
@{藏根於五山，特秀峙中亭，}@
%{他隐居在五座山里，特别是在中间一座山顶上，}%
@{林木花果茂，流泉溫涼分，}@
%{那里树木繁茂，果实累累，泉水清凉甘甜。}%
@{入彼五山城，寂靜猶昇天。}@
%{走进五座山城，静谧得如同升入天堂。}%

@{國人見太子，容德深且明，}@
%{城里的人看见太子，容貌端庄，德行高尚，}%
@{少年身光澤，無比丈夫形，}@
%{虽然是少年的身躯，却有着非凡的男子气概，}%
@{悉起奇特想，如見自在幢。}@
%{大家心里都产生奇特的念头，仿佛看到了自在幢。}%
@{橫行為止足，隨後者速馳，}@
%{走在路上的人纷纷停下来，后面的人则快步追赶，}%
@{先進悉迴顧，瞻目視無厭。}@
%{走在前面的人也都回过头来，目不转睛地注视着太子。}%

@{四體諸相好，隨見目不移，}@
%{太子身上的一切相貌都非常美好，人们见了就移不开眼，}%
@{恭敬來奉迎，合掌禮問訊。}@
%{恭敬地上前迎接，合掌礼拜问候。}%
@{咸皆大歡喜，隨宜而供養，}@
%{大家都很高兴，根据自己的能力供养太子，}%
@{瞻仰尊勝顏，俯愧種種形。}@
%{仰望着太子尊贵的面容，低头惭愧自己的种种相貌。}%

@{政素輕躁儀，寂默加肅敬，}@
%{原本轻浮的人，变得沉默而恭敬，}%
@{結恨心永解，慈和情頓增。}@
%{心中积怨也消散了，慈祥和善的情感油然而生。}%
@{士女公私業，一時悉休廢，}@
%{无论是官府还是百姓，都暂时停止了工作，}%
@{敬形宗其德，隨觀盡忘歸。}@
%{敬佩太子的外形，崇拜他的德行，专注地观看着，忘记了回家。}%

@{眉間白毫相，脩廣紺青目，}@
%{太子眉间有白毫相，眼睛细长，深邃而清澈，}%
@{舉體金光曜，清淨網縵手，}@
%{全身散发着金光，双手洁净，如同网状的丝带，}%
@{雖為出家形，有應聖王相。}@
%{虽然是出家人的形象，却有着应运而生的圣王之相。}%

@{王舍城士女，長幼悉不安，}@
%{王舍城的男女老少，都感到不安，}%
@{此人尚出家，我等何俗歡？}@
%{这样的人尚且出家修行，我们这些俗人还能有什么快乐呢？}%

@{爾時瓶沙王，處於高觀上，}@
%{这时瓶沙王，站在高高的观台上，}%
@{見彼諸士女，惶惶異常儀，}@
%{看到那些男女慌慌张张，行为异常，}%
@{勅召一外人，備問何因緣？}@
%{就召来一个外地人，询问缘由。}%

@{恭跪王樓下，具白所見聞：}@
%{那人恭敬地跪在王宫楼下，详细地讲述自己所见所闻：}%
@{昔聞釋氏種，殊特殊勝子，}@
%{我以前听说释迦族，出了一个非常特殊、非常优秀的儿子，}%
@{神慧超世表，應王領八方，}@
%{他的智慧超凡脱俗，应该成为王中之王，统领四方，}%
@{今出家在此，眾人悉奉迎。}@
%{现在他出家修行，大家都在这里迎接他。}%

@{王聞心驚喜，形留神已馳，}@
%{瓶沙王听了，心里既惊且喜，身体还留在这里，心神却已经飞奔出去，}%
@{勅使者速還，伺候進趣宜，}@
%{就命令使者赶紧回去，伺候太子的行动，}%
@{奉教密隨從，瞻察所施為。}@
%{并暗中跟随太子，观察他的行为。}%

@{澄靜端目視，庠步顯真儀，}@
%{太子目光清澈，步履从容，显示出真人的仪态，}%
@{入里行乞食，為諸乞士光，}@
%{他走进村庄，向乞丐们乞讨食物，让那些乞丐都感到光明，}%
@{歛形心不亂，好惡靡不安。}@
%{他收敛形体，心不慌乱，对好恶没有一丝不安。}%

@{精麤隨所得，持鉢歸閑林，}@
%{无论食物粗糙还是精细，他都接受，拿着钵盂回到静谧的树林，}%
@{食訖漱清流，樂靜安白山。}@
%{吃完饭后，在清澈的溪流里漱口，享受着宁静安详的白山。}%

@{青林別高崖，丹華殖其間，}@
%{青翠的树林掩映着高耸的悬崖，红色的丹霞山生长在其中，}%
@{孔雀等眾鳥，翻飛而亂鳴。}@
%{孔雀等各种鸟儿，在空中翻飞，乱叫着。}%

@{法服助鮮明，如日照扶桑，}@
%{太子的法衣衬托着他的容颜更加鲜明，如同太阳照耀着扶桑树，}%
@{使見安住彼，次第具上聞。}@
%{使者看到太子安住在那里，就详细地把这些情况汇报给瓶沙王。}%

@{王聞心馳敬，即勅嚴駕行，}@
%{瓶沙王听了，心里敬佩不已，立刻命令整顿车驾，准备出行，}%
@{天冠佩花服，師子王遊步。}@
%{他戴着天冠，穿着鲜艳的衣服，像威武的狮子王一样，迈着雄健的步伐。}%

@{簡擇諸宿重，安靜審諦士，}@
%{他挑选了一些德高望重的长者，以及沉静、明智的贤士，}%
@{導從百千眾，雲騰昇白山。}@
%{带领着成千上万的人，像云一样涌向白山。}%

@{見菩薩嚴儀，寂靜諸情根，}@
%{他们看到太子端庄的仪容，内心平静，情根都被寂静所摄受，}%
@{端坐山巖室，如月麗青天。}@
%{太子端坐在山洞里，就像明月照耀着夜空一样。}%

@{妙色淨端嚴，猶若法化身，}@
%{他容貌清净，端庄严丽，仿佛是法身化成的形象，}%
@{虔心肅然發，恭步漸親近。}@
%{瓶沙王虔诚地向太子走去，恭敬地一步步靠近。}%

@{猶如天帝釋，詣摩醯首羅，}@
%{这就像天帝释迦牟尼，去拜见摩醯首罗一样，}%
@{歛容執禮儀，敬問彼和安。}@
%{收敛容颜，行礼问候，恭敬地询问太子安好。}%

@{菩薩詳而動，隨順反相酬。}@
%{太子详细地回答，并随顺地与瓶沙王交谈。}%

@{時王勞問畢，端坐清淨石，}@
%{瓶沙王问完话后，端坐在一块干净的石头上，}%
@{瞪矚瞻神儀，顏和情交悅。}@
%{目不转睛地注视着太子的神态，脸上带着和善的笑容，内心感到愉悦。}%

@{伏聞名高族，盛德相承襲，}@
%{我听说你出身名门望族，世代相传着高尚的德行，}%
@{欽情久蘊積，今欲決所疑。}@
%{我对你敬佩已久，现在想解开心中的疑惑。}%

@{日光之元宗，祚隆已萬世，}@
%{你出身于太阳神的光明家族，世代昌隆，已经延续了万世，}%
@{令德紹遺嗣，弘廣萃於今。}@
%{你的德行继承了祖先的遗风，并得到了进一步的弘扬，光耀至今。}%

@{賢明年幼少，何故而出家？}@
%{你如此贤明，而且年纪轻轻，为什么还要出家修行呢？}%

@{超世聖王子，乞食不存榮，}@
%{你是一位超凡脱俗的圣王子，乞讨食物却不计较荣辱，}%
@{妙體應塗香，何故服袈裟？}@
%{你拥有如此美好的身体，应该涂抹香料，为什么却穿着袈裟呢？}%

@{手宜握天下，反以受薄飡，}@
%{你应该掌握天下，却反而接受粗陋的食物，}%
@{若不代父王，受禪享其土，}@
%{如果你不继承父王，接受禅位，享受你的领土，}%
@{吾今分半國，庶望少留情。}@
%{我现在愿意把一半的国家分给你，希望你能够留下来。}%

@{既免逼親嫌，時過隨所從，}@
%{这样就可以避免逼迫亲人的嫌疑，等到时机成熟，你就可以随心所欲，}%
@{當體我誠言，貪德為良隣。}@
%{请你体谅我的真心话，贪图美德，做我的好邻居。}%

@{或恃名勝族，才德容貌兼，}@
%{也许你依靠着显赫的家族，拥有才华、德行和美貌，}%
@{不欲降高節，屈下受人恩。}@
%{不愿降低自己的高尚节操，屈服于别人的恩惠。}%

@{當給勇健士，器仗隨軍資，}@
%{我应该给你配备勇敢的士兵，武器装备和军队的资助，}%
@{自力廣收羅，天下孰不推。}@
%{你自己努力，广招贤才，天下谁不拥戴你呢？}%

@{明人知時取，法財五欲增，}@
%{聪明的人懂得把握时机，既要追求法财，也要享受五欲，}%
@{若不獲三利，終始徒勞勤。}@
%{如果你不能获得这三方面的利益，最终只会白费力气。}%

@{崇法捨財色，財為一分人，}@
%{崇尚佛法，舍弃财色，财富只是用来帮助别人，}%
@{富財捨法欲，此則保財資。}@
%{拥有财富，却舍弃佛法和五欲，这样才能保住财富。}%

@{貧窶而忘法，五欲孰能歡？}@
%{贫困潦倒，却忘记了佛法，五欲又怎么能让你快乐呢？}%
@{是故三事俱，德流而道宣。}@
%{所以，三者兼顾，才能德行流传，道法弘扬。}%

@{法財五欲備，名世大丈夫，}@
%{拥有法财、五欲，才能被称为世间大丈夫，}%
@{無令圓相身，徒勞而無功。}@
%{不要让圆满的相貌，白白浪费，毫无功绩。}%

@{曼陀轉輪王，王領四天下，}@
%{曼陀转轮王，统治着四大天下，}%
@{帝釋分半坐，力不能王天。}@
%{天帝释迦牟尼，就算分给他一半的天界，他也无法统治天界。}%

@{今汝𦟛長臂，足攬人天境，}@
%{你现在拥有如此强大的能力，足以掌控人天两界，}%
@{我不恃王力，而欲強相留。}@
%{我并不是依靠王权，才想要强留你。}%

@{見汝改形好，愛著出家衣，}@
%{我看到你改变了容貌，喜欢上了出家人的衣服，}%
@{既以敬其德，矜苦惜其人，}@
%{既敬佩你的德行，又怜悯你的苦难，珍惜你这个人，}%
@{今見行乞求，我願奉其土。}@
%{现在看到你行乞讨食，我愿意把我的领土奉献给你。}%

@{少壯受五欲，中年習用財，}@
%{年轻的时候享受五欲，中年的时候学习使用财富，}%
@{年耆諸根熟，是乃順法時。}@
%{年老的时候，身心成熟，这是顺应佛法的时候。}%

@{壯年守法財，必為欲所壞，}@
%{壮年的时候，守着法财，必然会被五欲所毁坏，}%
@{老則氣虛微，隨順求寂默，}@
%{年老的时候，气力衰弱，顺应自然，追求寂静，}%
@{耆年愧財欲，行法舉世宗。}@
%{年老的时候，对财富和五欲感到羞愧，修行佛法，就会成为世人的榜样。}%

@{壯年心輕躁，馳騁五欲境，}@
%{壮年的时候，心性轻浮，在五欲的境界里驰骋，}%
@{疇侶契纏綿，情交相感深，}@
%{朋友之间关系密切，感情交织在一起，非常深厚，}%
@{年宿寡綢繆，順法者所宗，}@
%{年老的时候，缺少了这些牵绊，顺应佛法的人，就会成为世人的榜样，}%
@{五欲悉休廢，增長樂法心。}@
%{五欲都消除了，就会增长对佛法的欢喜之心。}%

@{具崇王者法，大會奉天神，}@
%{你应该按照王者的法度，举行盛大的仪式，供奉天神，}%
@{當乘神龍背，受樂上昇天。}@
%{应该骑着神龙，享受快乐，升上天界。}%

@{先勝諸聖王，嚴身寶瓔珞，}@
%{你应该胜过其他的圣王，装饰着珍贵的璎珞，}%
@{祠祀設大會，終歸受天福。}@
%{举行祭祀，举办盛大的宴会，最终都会得到天上的福报。}%

@{如是瓶沙王，種種方便說，}@
%{瓶沙王用各种方法劝说太子，}%
@{太子志堅固，不動如須彌。}@
%{但是太子意志坚定，如同须弥山一样，纹丝不动。}%
@{瓶沙王隨順，安慰勸請已，}@
%{瓶沙王顺从我的意思，安慰我，劝我留下。}%

@{太子敬答謝：深感於來言，}@
%{太子恭敬地回答感谢说：“我深深地感谢你所说的话。}%

@{善得世間宜，所說不乖理，}@
%{你所讲的，符合世间道理，所说的话也合乎逻辑。}%

@{訶梨名族胄，為人善知識，}@
%{你是呵梨族的子孙，是值得尊敬的善知识。}%

@{義懷心虛盡，法應如是說。}@
%{你怀着正义之心，真诚地将佛法宣讲出来，这正是应该如此的。}%

@{世間說凡品，不能處仁義，}@
%{世间所说的凡夫俗子，不能真正做到仁义。}%

@{薄德遇近情，豈達名勝事？}@
%{他们德行薄弱，只顾眼前利益，怎么能成就伟大的事业呢？}%

@{承習先勝宗，崇禮修敬讓，}@
%{只有继承先贤的优秀思想，崇尚礼仪，修养恭敬谦让，}%

@{能於苦難中，周濟不相棄，}@
%{才能在别人遇到困难时，施以援手，不抛弃不放弃，}%

@{是則為世間，真善知識相。}@
%{这样的人才是世间真正的善知识。}%

@{善友財通濟，是名牢固藏，}@
%{朋友之间互相帮助，才是真正的宝藏。}%

@{守惜封己利，是必速亡失。}@
%{只顾自己利益，吝啬不肯帮助他人，最终只会失去朋友，一无所有。}%

@{國財非常寶，惠施為福業，}@
%{国家的财富不是真正的宝藏，只有用来帮助他人，才是真正的福报。}%

@{兼施善知識，雖散後無悔。}@
%{即使把财富分给善知识，即使散尽了也不后悔。}%

@{既知汝厚懷，不為違逆論，}@
%{我知道你心怀善意，不会反对我的决定。}%

@{且今以所見，率心而相告。}@
%{我现在就将我所看到的，坦诚地告诉你。}%

@{畏生老病死，欲求真解脫，}@
%{我害怕生老病死，想要寻求真正的解脱。}%

@{捨親離恩愛，豈還習五欲？}@
%{为了解脱，我放弃亲人，舍弃爱欲，难道还会沉迷于五欲吗？}%

@{不畏盛毒蛇，凍電猛盛火，}@
%{我不害怕剧毒的蛇，也不害怕闪电和猛烈的火焰，}%

@{唯畏五欲境，流轉勞我心。}@
%{我唯独害怕五欲的诱惑，它让我在生死轮回中不停地漂泊，劳苦不堪。}%

@{五欲非常賊，劫人善珍寶，}@
%{五欲就像强盗，会抢夺人们的善念和宝藏。}%

@{詐偽虛非實，猶若幻化人。}@
%{它虚假无实，就像幻影一样，迷惑人心。}%

@{暫思令人惑，況常處其中？}@
%{仅仅想一想它就会让人迷惑，何况长期沉迷其中呢？}%

@{五欲為大礙，永障寂滅法。}@
%{五欲是修行解脱的障碍，永远阻碍着我们进入寂灭的境界。}%

@{天樂尚不可，況處人間欲？}@
%{即使天上的快乐也不能满足我，何况人间五欲的享受呢？}%

@{五欲生渴愛，終無滿足時。}@
%{五欲会滋生贪欲和执着，永远不会满足。}%

@{猶盛風猛火，投薪亦無足，}@
%{就像狂风猛火，即使不断地添加柴薪，也永远填不满。}%

@{世間諸非義，莫過五欲境。}@
%{世间所有不义之事，没有比沉迷于五欲更糟糕的了。}%

@{眾生愚貪故，樂著而不覺，}@
%{众生愚痴贪婪，沉迷于五欲，浑然不觉。}%

@{智者畏五欲，不墮於非義。}@
%{只有智慧的人，才会远离五欲，不堕落到不义的道路上。}%

@{王領四海內，猶外更希求，}@

@{凡人見來賓，應修先後宜，}@
%{普通人遇到客人，应该按照礼仪接待，}%
@{且為設床座，任彼之所安。}@
%{先为他准备床座，让他随意安顿。}%

@{作此要言已，各各正基坐，}@
%{他们说完这些话，就各自端正地坐着，}%
@{如來漸次至，不覺違要言，}@
%{如来慢慢地走过来，没有意识到违背了他们的说法，}%
@{有請讓其坐，有為攝衣鉢，}@
%{有人请他坐下，有人为他整理衣服和钵盂，}%
@{有為洗摩足，有請問所須，}@
%{有人为他洗脚按摩，有人问他需要什么，}%
@{如是等種種，尊敬師奉事，}@
%{他们用各种各样的方式，恭敬地侍奉着如来，}%
@{唯不捨其族，猶稱瞿曇名。}@
%{只是没有放弃他们的族姓，仍然称呼他为瞿昙。}%

@{世尊告彼言：莫稱我本性，}@
%{世尊对他们说：不要称呼我的本性，}%
@{於阿羅呵所，而生𧃳慢言。}@
%{在阿罗汉面前，不要生出傲慢的话语。}%

@{於敬不敬者，我心悉平等，}@
%{我对恭敬和不恭敬的人，内心都平等看待，}%
@{汝等心不恭，當自招其罪。}@
%{你们内心不恭敬，将会招致罪过。}%

@{佛能度世間，是故稱為佛。}@
%{佛能度化世间，所以被称为佛。}%
@{於一切眾生，等心如子想，}@
%{他对一切众生，都怀着如同对待孩子一样的平等之心，}%
@{而稱本名字，如得慢父罪。}@
%{却称呼他的本名，就像得到了父亲的宠爱而犯了傲慢的罪过一样。}%

@{佛以大悲心，哀愍而告彼，}@
%{佛以大悲心，怜悯地对他们说：}%
@{彼率愚騃心，不信正真覺。}@
%{你们这些人太愚蠢，不相信真正的觉悟。}%

@{言先修苦行，猶尚無所得，}@
%{你们说以前修行苦行，尚且没有得到什么，}%
@{今恣身口樂，何因得成佛？}@
%{现在放纵身口之乐，凭什么能成佛呢？}%

@{如是等疑惑，不信得佛道，}@
%{你们带着这样的疑惑，不相信能得到佛道，}%
@{究竟真實義，一切智具足。}@
%{却不知道究竟的真实意义，一切智都具足了。}%

@{如來即為彼，略說其要道：}@
%{如来就为他们略说要道：}%
@{愚夫習苦行，樂行悅諸根，}@
%{愚人修行苦行，喜欢修行，满足自己的感官，}%
@{見彼二差別，斯則為大過，}@
%{看到这两者的差别，这就是大错，}%
@{非是正真道，以違解脫故。}@
%{这不是真正的道，因为违背了解脱。}%

@{疲身修苦行，其心猶馳亂，}@
%{疲惫地修行苦行，内心仍然心猿意马，}%
@{尚不生世智，況能超諸根？}@
%{连世间的智慧都生不起来，怎么能超越感官呢？}%

@{如以水燃燈，終無破闇期，}@
%{就像用水点灯，永远无法驱散黑暗，}%
@{疲身修慧燈，不能壞愚癡。}@
%{疲惫地修行智慧之灯，不能消除愚痴。}%

@{朽木而求火，徒勞而弗獲，}@
%{用朽木求火，只会徒劳无功，}%
@{鑽燧人方便，即得火為用。}@
%{只有用钻燧的方法，才能得到火来使用。}%

@{求道非苦身，而得甘露法，}@
%{求道不是苦行，而是要得到甘露法，}%
@{著欲為非義，愚癡障慧明。}@
%{执着于欲望是不义，愚痴遮蔽了智慧的光明。}%

@{尚不了經論，況得離欲道？}@
%{连经论都理解不了，何况能得到离欲之道呢？}%

@{如人得重病，食不隨病食；}@
%{就像人得了重病，不能吃随意的食物，}%
@{無知之重病，著欲豈能除？}@
%{愚痴的重病，执着于欲望怎么能消除呢？}%

@{放火於曠野，乾草增猛風，}@
%{在旷野放火，干草会助长狂风，}%
@{火盛孰能滅，貪愛火亦然。}@
%{火势旺盛，谁又能灭掉呢？贪爱之火也是如此。}%

@{我已離二邊，心存於中道，}@
%{我已经远离了两个极端，心存中道，}%
@{眾苦畢竟息，安靜離諸過。}@
%{一切苦难最终都会消散，安宁地远离一切过失。}%

@{正見踰日光，平等覺觀佛，}@
%{正见超越阳光，平等觉观佛法，}%
@{正語為舍宅，遊戲正業林，}@
%{正语如同舍宅，在正业之林中游戏，}%
@{正命為豐姿，方便正修塗，}@
%{正命如同丰姿，方便地修行正道，}%
@{正念為城郭，正定為床座，}@
%{正念如同城郭，正定如同床座，}%

@{八道坦平正，}@
%{八正道平坦而正，}%
@{免脫生死苦。從此塗出者，}@
%{可以免除生死之苦。从这条道路走出去的人，}%
@{所作已究竟，不墮於此彼。}@
%{所做的一切都已圆满，不会堕落到任何地方。}%

@{二世苦數中，三界純苦聚，}@
%{在两世苦难中，三界都是苦的集合，}%
@{唯此道能滅，本所未曾聞，}@
%{只有这条道才能灭除苦难，这是以前从未听说过的，}%
@{正法清淨眼，等見解脫道，}@
%{正法清净之眼，平等地看到解脱之道，}%
@{唯我今始超。生老病死苦，}@
%{只有我如今才超越了它。生老病死之苦，}%
@{愛離怨憎會，所求事不果，}@
%{爱别离之苦，怨憎会之苦，求不得之苦，}%
@{及餘種種苦，離欲未離欲，}@
%{以及其他各种各样的苦，离欲未离欲之苦，}%
@{有身及無身，離淨功德者，}@
%{有身之苦，无身之苦，没有清净功德之苦，}%
@{略說斯皆苦。猶如盛火息，}@
%{简单地说，这些都是苦。就像盛火熄灭了，}%
@{雖微不捨熱，寂靜微細我，}@
%{虽然微弱，但余热未消，寂静微细的我，}%
@{大苦性猶存，貪等諸煩惱，}@
%{大苦的本质仍然存在，贪欲等烦恼，}%
@{及種種業過，是則為苦因，}@
%{以及各种各样的业障，这就是苦的根源，}%
@{捨離則苦滅。猶如諸種子，}@
%{舍弃它，苦就会灭除。就像各种种子，}%
@{離於地水等，眾緣不和合，}@
%{离开土壤、水等，各种缘不和合，}%
@{芽葉則不生。有有性相續，}@
%{芽叶就不会生长。有有性相续，}%
@{從天至惡趣，輪廻而不息，}@
%{从天界到恶趣，轮回不断，}%
@{斯由貪欲生，軟中上差降，}@
%{这是由贪欲而生，从软弱到强壮，从低到高，}%
@{種種業為因。若滅於貪等，}@
%{各种各样的业作为因。如果灭除贪欲等，}%
@{則無有相續，種種業盡者，}@
%{就不会有相续，各种各样的业都灭尽了，}%
@{差別苦長息。此有則彼有，}@
%{差别之苦就会永远消失。这个有，那个就有，}%
@{此滅則彼滅，無生老病死，}@
%{这个灭，那个就灭，没有生老病死，}%
@{無地水火風。亦無初中邊，}@
%{没有地水火风。也没有开始、中间、结束，}%
@{亦非欺誑法，賢聖之所住，}@
%{也不是欺骗的法，是贤圣所住的地方，}%
@{無盡之寂滅。所說八正道，}@
%{是无穷无尽的寂灭。我所说八正道，}%
@{是方便非餘，世間所不見，}@
%{是方便法，不是其他的，世间人所看不到，}%
@{彼彼長迷惑。我知苦斷集，}@
%{他们一直都迷惑不解。我知道苦，断除集，}%
@{證滅修正道，觀此四真諦，}@
%{证得灭，修证正道，观照这四谛，}%
@{遂成等正覺。}@
%{最终成就了等正觉。}%

@{謂我已知苦，已斷有漏因，}@
%{我说我已经知道苦，已经断除了有漏的因，}%
@{已滅盡作證，已修八正道。}@
%{已经灭尽苦，证得解脱，已经修证了八正道。}%

@{已知四真諦，清淨法眼成，}@
%{我已经知道四谛，清净的法眼成就了，}%
@{於此四真諦，未生平等眼。}@
%{对这四谛，还没有生出平等之眼。}%

@{不名得解脫，不言作已作，}@
%{不能说已经得到解脱，不能说已经做到了，}%
@{亦不言一切，真實知覺成。}@
%{也不能说已经完全成就了真实的觉悟。}%

@{已知真諦故，自知得解脫，}@
%{因为已经知道真谛，所以自己知道得到了解脱，}%

%{自己知道已经做到了，自己知道已经成就了等正觉。}%

@{說是真實時，憍憐族姓子，}@

%{八万诸天大众，明白了究竟的真实意义，}%
@{遠離諸塵垢，清淨法眼成。}@
%{远离了各种尘垢，清净的法眼成就了。}%

@{天人師知彼，所作事已作，}@
%{天人师知道他，所做的事情已经做到了，}%
@{歡喜師子吼，問憍憐如來。}@
%{欢喜地发出狮子吼，问憍怜如来。}%

@{憍憐即白佛：已知大師法。}@
%{憍怜就对佛说：我已经明白了大师的法。}%

@{以彼知法故，名阿若憍憐，}@
%{因为他明白了佛法，所以被称为阿若憍怜，}%
@{於佛弟子中，最先第一悟。}@
%{在佛弟子中，他是第一个悟道的人。}%

@{彼知正法聲，聞於諸地神，}@
%{他明白了正法的声音，地神们都听到了，}%
@{咸共舉聲唱：善哉見深法。}@
%{一起高声唱道：太好了！看到了深奥的法！}%

@{如來於今日，轉未曾所轉，}@
%{如来今天转动了从未转过的法轮，}%
@{普為諸天人，廣開甘露門。}@
%{普遍地为诸天人，广开甘露之门。}%

@{堅固智為輞，慚愧楔其間。}@

%{正念是法轮的轮毂，成就了真实的法轮，}%
@{正真出三界，不退從邪師。}@

@{如是地神唱，虛空神傳稱，}@
%{地神们这样唱着，虚空神们传递着赞颂，}%
@{諸天轉讚嘆，乃至徹梵天。}@
%{诸天不断地赞叹，直到梵天。}%

@{三界諸天神，始聞大仙說，}@
%{三界诸天神，第一次听到大仙说，}%
@{展轉驚相告，普聞佛興世。}@

@{廣為群生類，轉寂靜法輪，}@

@{風霽雲霧除，空中雨天華，}@
