@{菩提行經卷第二}@
%{菩提行经卷第二}%

@{聖龍樹菩薩集頌}@
%{圣龙树菩萨集颂}%

@{西天中印度惹爛駄囉國密林寺三藏明教大師賜紫沙門臣天息災奉詔譯}@
%{西天中印度惹烂陀罗国密林寺三藏明教大师赐紫沙门臣天息灾奉诏译}%

@{菩提心忍辱波羅蜜多品第四}@
%{菩提心忍辱波罗蜜多品第四}%

@{奉行諸善業，施戒而先導，}@
%{奉行诸善业，施戒而先导，}%
@{供養於如來，百千劫無盡。}@
%{供养于如来，百千劫无尽。}%

@{修行於羼提，嗔罪而不立，}@
%{修行于羼提，嗔罪而不立，}%
@{觀種種體空，是故一心忍。}@
%{观种种体空，是故一心忍。}%

@{不得貪快樂，守意令平等，}@
%{不得贪快乐，守意令平等，}%
@{心有嗔惱病，無睡恒不足。}@
%{心有嗔恼病，无睡恒不足。}%

@{彼此有施主，供給於利養，}@
%{彼此有施主，供给于利养，}%
@{隨彼愛重心，無得生瞋惱。}@
%{随彼爱重心，无得生瞋恼。}%

@{凡諸親近事，不起於憎嫌，}@
%{凡诸亲近事，不起于憎嫌，}%
@{於彼無所瞋，乃得其安樂。}@
%{于彼无所瞋，乃得其安乐。}%

@{忍如是等事，若對於冤家，}@
%{忍如是等事，若对于冤家，}%
@{於瞋若能除，世世獲安樂。}@
%{于瞋若能除，世世获安乐。}%

@{冤若生於心，於愛亦無喜，}@
%{冤若生于心，于爱亦无喜，}%
@{若飱瞋惱食，無忍善不壞。}@
%{若飱瞋恼食，无忍善不坏。}%

@{彼食我大冤，於我無善利，}@
%{彼食我大冤，于我无善利，}%
@{知彼冤不食，是故忍堅牢。}@
%{知彼冤不食，是故忍坚牢。}%

@{凡見冤來去，歡喜而不瞋，}@
%{凡见冤来去，欢喜而不瞋，}%
@{於冤若起瞋，善利終滅盡。}@
%{于冤若起瞋，善利终灭尽。}%

@{忍心常若此，令瞋不得起，}@
%{忍心常若此，令瞋不得起，}%
@{住忍無時節，瞋冤自不生。}@
%{住忍无时节，瞋冤自不生。}%

@{若人自保愛，不作惡口業，}@
%{若人自保爱，不作恶口业，}%
@{口業若不離，後感冤家苦。}@
%{口业若不离，后感冤家苦。}%

@{畏苦不出離，不行眾苦因，}@
%{畏苦不出离，不行众苦因，}%
@{是故堅忍心，獲得諸快樂。}@
%{是故坚忍心，获得诸快乐。}%

@{彼訥陵誐子，邪見求解脫，}@
%{彼讷陵誐子，邪见求解脱，}%
@{刀割火燒身，無利由能忍。}@
%{刀割火烧身，无利由能忍。}%

@{愚癡無正見，虛受大苦惱，}@
%{愚痴无正见，虚受大苦恼，}%
@{我以菩提心，云何苦不忍？}@
%{我以菩提心，云何苦不忍？}%

@{蚊蚤壁虱等，常飢渴苦惱，}@
%{蚊蚤壁虱等，常饥渴苦恼，}%
@{大痒煩苦人，住忍而不見。}@
%{大痒烦苦人，住忍而不见。}%

@{寒熱并雨風，病枷鎖捶打，}@
%{寒热并雨风，病枷锁捶打，}%
@{被諸苦惱事，忍不求快樂。}@
%{被诸苦恼事，忍不求快乐。}%

@{殺他血流迸，堅牢心勇猛，}@
%{杀他血流迸，坚牢心勇猛，}%
@{割身自見血，怕怖而驚倒。}@
%{割身自见血，怕怖而惊倒。}%

@{智者心清淨，常懼瞋惱侵，}@
%{智者心清净，常惧瞋恼侵，}%
@{與煩惱相持，忍心恒勇猛。}@
%{与烦恼相持，忍心恒勇猛。}%

@{蛇腹行在地，喻瞋伏於心，}@
%{蛇腹行在地，喻瞋伏于心，}%
@{殺之謂無勇，殺彼得最勝。}@
%{杀之谓无勇，杀彼得最胜。}%

@{如來大悲者，愍苦說輪迴，}@
%{如来大悲者，愍苦说轮回，}%
@{使識罪根本，住忍而不作。}@
%{使识罪根本，住忍而不作。}%

@{父母何計心，懼子遭淪溺，}@
%{父母何计心，惧子遭沦溺，}%
@{持心離瞋怒，自遠大苦報。}@
%{持心离瞋怒，自远大苦报。}%

@{譬如無智人，令罪而得生，}@
%{譬如无智人，令罪而得生，}%
@{修行而無智，瞋生亦復爾。}@
%{修行而无智，瞋生亦复尔。}%

@{欲住不思議，當須持自心，}@
%{欲住不思議，当须持自心，}%
@{於此生愛重，令瞋不生起。}@
%{于此生爱重，令瞋不生起。}%

@{若貪彼塵境，而生種種罪，}@
%{若贪彼尘境，而生种种罪，}%
@{因彼諸業力，而不得自由。}@
%{因彼诸业力，而不得自由。}%

@{於境若不貪，彼集無因立，}@
%{于境若不贪，彼集无因立，}%
@{和合心無故，是故無有生。}@
%{和合心无故，是故无有生。}%

@{不貪而不生，無得而自說，}@
%{不贪而不生，无得而自说，}%
@{我得如是故，是生不思議。}@
%{我得如是故，是生不思議。}%

@{彼無生不生，是得云何有？}@
%{彼无生不生，是得云何有？}%
@{瞻察於彼此，滅盡得無餘。}@
%{瞻察于彼此，灭尽得无余。}%

@{此心恒清淨，喻隨色摩尼，}@
%{此心恒清净，喻随色摩尼，}%
@{所變悉從因，無因相何有？}@
%{所变悉从因，无因相何有？}%

@{過去行行時，彼行何所作？}@
%{过去行行时，彼行何所作？}%
@{隨彼所行因，等因而感果。}@
%{随彼所行因，等因而感果。}%

@{一切雖由因，因善惡由心，}@
%{一切虽由因，因善恶由心，}%
@{說求性寂靜，如是有何過？}@
%{说求性寂静，如是有何过？}%

@{若取和合因，是樂於苦惱，}@
%{若取和合因，是乐于苦恼，}%
@{此心不可住，智人應自勸。}@
%{此心不可住，智人应自劝。}%

@{是故見冤家，想作善知識，}@
%{是故见冤家，想作善知识，}%
@{因行如是行，當獲得快樂。}@
%{因行如是行，当获得快乐。}%

@{如是諸有情，由業不自在，}@
%{如是诸有情，由业不自在，}%
@{自在若成就，誰肯趣於苦？}@
%{自在若成就，谁肯趣于苦？}%

@{散亂心緣塵，心被刺不覺，}@
%{散乱心缘尘，心被刺不覺，}%
@{食斷食增瞋，於苦而返愛。}@
%{食断食增瞋，于苦而返爱。}%

@{自若無福行，返愛纏縛業，}@
%{自若无福行，返爱缠缚业，}%
@{如飱毒藥食，墮於生死崖。}@
%{如飱毒药食，堕于生死崖。}%

@{自住是煩惱，誠由不自護，}@
%{自住是烦恼，诚由不自护，}%
@{欲解脫他人，此事何由得？}@
%{欲解脱他人，此事何由得？}%

@{煩惱迷昏濁，而致於自殺，}@
%{烦恼迷昏浊，而致于自杀，}%
@{毒盛無有悲，云何瞋不護？}@
%{毒盛无有悲，云何瞋不护？}%

@{自性既愚迷，於他行嬈亂，}@
%{自性既愚迷，于他行娆乱，}%
@{生彼瞋無疑，如火而能燒。}@
%{生彼瞋无疑，如火而能烧。}%

@{有情性愚時，所行諸過失，}@
%{有情性愚时，所行诸过失，}%
@{愚迷故若此，如煙熏虛空。}@
%{愚迷故若此，如烟熏虚空。}%

@{若人瞋不護，愚迷無智故，}@
%{若人瞋不护，愚迷无智故，}%
@{喻持杖勸人，而增彼瞋惱。}@
%{喻持杖劝人，而增彼瞋恼。}%

@{我於過去生，苦惱諸有情，}@
%{我于过去生，苦恼诸有情，}%
@{是故於今身，被苦惱能忍。}@
%{是故于今身，被苦恼能忍。}%

@{我身喻於鐵，受彼燒鎚鍛，}@
%{我身喻于铁，受彼烧锤锻，}%
@{如彼鐵持身，何得有其苦？}@
%{如彼铁持身，何得有其苦？}%

@{我今看此身，如無情形像，}@
%{我今看此身，如无情形像，}%
@{雖被諸苦惱，而瞋無所起。}@
%{虽被诸苦恼，而瞋无所起。}%

@{愚迷起愛業，不知其苦本，}@
%{愚迷起爱业，不知其苦本，}%
@{得苦緣自過，云何生瞋惱？}@
%{得苦缘自过，云何生瞋恼？}%

@{喻受地獄苦，飛禽劍林等，}@
%{喻受地狱苦，飞禽剑林等，}%
@{知自業所生，何處有瞋惱？}@
%{知自业所生，何处有瞋恼？}%

@{我得如是業，此過知所起，}@
%{我得如是业，此过知所起，}%
@{設令入地獄，不由他所作。}@
%{设令入地狱，不由他所作。}%

@{欲盡我之業，無量無有邊，}@
%{欲尽我之业，无量无有边，}%
@{我業既如是，長時受地獄。}@
%{我业既如是，长时受地狱。}%

@{我此過如是，彼實我冤家，}@
%{我此过如是，彼实我冤家，}%
@{云何分別知，愚迷瞋造作？}@
%{云何分别知，愚迷瞋造作？}%

@{若人自護持，對冤忍不恚，}@
%{若人自护持，对冤忍不恚，}%
@{是心功德生，地獄云何入？}@
%{是心功德生，地狱云何入？}%

@{盡我之所行，得因如彼時，}@
%{尽我之所行，得因如彼时，}%
@{不忍瞋不護，破壞於修行。}@
%{不忍瞋不护，破坏于修行。}%

@{意無相無形，散亂即破壞，}@
%{意无相无形，散乱即破坏，}%
@{由身護持故，身苦當忍受。}@
%{由身护持故，身苦当忍受。}%

@{我於口惡業，眾過而不作，}@
%{我于口恶业，众过而不作，}%
@{身不被眾苦，云何心有瞋？}@
%{身不被众苦，云何心有瞋？}%

@{我於今生中，淨心行利行，}@
%{我于今生中，净心行利行，}%
@{於利益既無，何事於食飲？}@
%{于利益既无，何事于食饮？}%

@{凡所作為事，要在於利他，}@
%{凡所作为事，要在于利他，}%
@{彼無利非愛，定獲罪無疑。}@
%{彼无利非爱，定获罪无疑。}%

@{不如今殞沒，無貪邪壽命，}@
%{不如今殞没，无贪邪寿命，}%
@{邪命住雖久，死當墮苦趣。}@
%{邪命住虽久，死当堕苦趣。}%

@{譬如在夢中，百年受快樂，}@
%{譬如在梦中，百年受快乐，}%
@{如真實得樂，覺已知暫非。}@
%{如真实得乐，觉已知暂非。}%

@{喻彼時無常，壽命之延促，}@
%{喻彼时无常，寿命之延促，}%
@{覺此二事已，彼何得快樂？}@
%{觉此二事已，彼何得快乐？}%

@{久處於歡娛，自謂得多益，}@
%{久处于欢娱，自谓得多益，}%
@{如行人被劫，裸形復空手。}@
%{如行人被劫，裸形复空手。}%

@{福利隨過減，罪根還復生，}@
%{福利随过减，罪根还复生，}%
@{福盡罪不生，為獲不瞋利。}@
%{福尽罪不生，为获不瞋利。}%

@{彼何為活命，一向作不善，}@
%{彼何为活命，一向作不善，}%
@{如是不思惟，無善不破壞。}@
%{如是不思惟，无善不破坏。}%

@{無得讚於瞋，破壞有情故，}@
%{无得赞于瞋，破坏有情故，}%
@{如是心利他，彼瞋無由生。}@
%{如是心利他，彼瞋无由生。}%

@{為彼修心人，於忍不住故，}@
%{为彼修心人，于忍不住故，}%
@{見彼煩惱生，是讚忍功德。}@
%{见彼烦恼生，是赞忍功德。}%

@{塔像妙法等，有謗及破壞，}@
%{塔像妙法等，有谤及破坏，}%
@{佛等無苦惱，我於彼不瞋。}@
%{佛等无苦恼，我于彼不瞋。}%

@{於師并眷屬，不作於愛業，}@
%{于师并眷属，不作于爱业，}%
@{今因過去生，見之而自勉。}@
%{今因过去生，见之而自勉。}%

@{覺心觀有情，恒在眾苦惱，}@
%{觉心观有情，恒在众苦恼，}%
@{見彼如是已，於苦惱能忍。}@
%{见彼如是已，于苦恼能忍。}%

@{瞋恚與愚癡，分別過一等，}@
%{瞋恚与愚痴，分别过一等，}%
@{於此毒過咎，何得說無過？}@
%{于此毒过咎，何得说无过？}%

@{云何於過去，而作害他業，}@
%{云何于过去，而作害他业，}%
@{如是諸業因，間斷此何作？}@
%{如是诸业因，间断此何作？}%

@{如佛福亦然，我今一心作，}@
%{如佛福亦然，我今一心作，}%
@{與一切有情，慈心互相覩。}@
%{与一切有情，慈心互相覩。}%

@{喻火燒其舍，舍中而火入，}@
%{喻火烧其舍，舍中而火入，}%
@{舍中若有草，彼火自延蔓。}@
%{舍中若有草，彼火自延蔓。}%

@{如是還喻心，和合於瞋火，}@
%{如是还喻心，和合于瞋火，}%
@{燒彼福功德，剎那無所有。}@
%{烧彼福功德，刹那无所有。}%

@{若人殺在手，放之善可稱，}@
%{若人杀在手，放之善可称，}%
@{地獄苦能免，此善誰不讚？}@
%{地狱苦能免，此善谁不赞？}%

@{若人在世間，少苦不能忍，}@
%{若人在世间，少苦不能忍，}%
@{地獄苦無量，瞋因何不斷？}@
%{地狱苦无量，瞋因何不断？}%

@{我以如是苦，歷百千地獄，}@
%{我以如是苦，历百千地狱，}%
@{一一為利他，所作不自為。}@
%{一一为利他，所作不自为。}%

@{我無如是等，諸大苦惱事，}@
%{我无如是等，诸大苦恼事，}%
@{以離世間故，為利如是行。}@
%{以离世间故，为利如是行。}%

@{離苦獲快樂，彼皆讚功德，}@
%{离苦获快乐，彼皆赞功德，}%
@{得彼如是讚，云何而不喜？}@
%{得彼如是赞，云何而不喜？}%

@{彼既得如此，無礙之快樂，}@
%{彼既得如此，无碍之快乐，}%
@{利他行最上，智者何不勉？}@
%{利他行最上，智者何不勉？}%

@{如是最上行，得快樂不修，}@
%{如是最上行，得快乐不修，}%
@{此見若不捨，破壞於正見。}@
%{此见若不舍，破坏于正见。}%

@{若敬愛於他，以德而稱讚，}@
%{若敬爱于他，以德而称赞，}%
@{他德既稱讚，乃是自敬愛。}@
%{他德既称赞，乃是自敬爱。}%

@{當發菩提心，為一切有情，}@
%{当发菩提心，为一切有情，}%
@{令得諸快樂，云何瞋有情？}@
%{令得诸快乐，云何瞋有情？}%

@{佛為三界供，欲有情成佛，}@
%{佛为三界供，欲有情成佛，}%
@{世利得不實，彼煩惱何作？}@
%{世利得不实，彼烦恼何作？}%

@{若人之骨肉，乃及諸眷屬，}@
%{若人之骨肉，乃及诸眷属，}%
@{養育與命等，不喜瞋何生？}@
%{养育与命等，不喜瞋何生？}%

@{如彼求菩提，當用菩提心，}@
%{如彼求菩提，当用菩提心，}%
@{而不愛有情，福自捨何瞋？}@
%{而不爱有情，福自舍何瞋？}%

@{若人有所求，出財大捨施，}@
%{若人有所求，出财大舍施，}%
@{所求既不獲，不如財在舍。}@
%{所求既不获，不如财在舍。}%

@{清淨功德福，何障而不獲？}@
%{清净功德福，何障而不获？}%
@{得已自不受，如住瞋修行。}@
%{得已自不受，如住瞋修行。}%

@{作罪與作福，不同不隨喜，}@
%{作罪与作福，不同不随喜，}%
@{亦復不依作，當自一無得，}@
%{亦复不依作，当自一无得，}%

@{若愛於冤家，欲求其歡喜，}@
%{若爱于冤家，欲求其欢喜，}%
@{復求諸讚說，此事無因得，}@
%{复求诸赞说，此事无因得，}%

@{雖欲利圓滿，返苦而無樂，}@
%{虽欲利圆满，返苦而无乐，}%
@{菩提心不忍，於利不成就，}@
%{菩提心不忍，于利不成就，}%

@{煩惱之惡鉤，牽人不自在，}@
%{烦恼之恶钩，牵人不自在，}%
@{由如地獄卒，擲人入湯火，}@
%{由如地狱卒，掷人入汤火，}%

@{我本求利他，何要虛稱讚？}@
%{我本求利他，何要虚称赞？}%

%{因为贫穷，不能做到，而愿望不能圆满。}%

@{不施怖者安，不修母快樂，}@
%{不施舍的人安逸，不孝敬母亲的人快乐，}%

@{如入母胎藏，母唯病苦惱。}@
%{就像进入母亲的胎藏，母亲只有病苦烦恼。}%

@{過去為離法，我今得果報，}@
%{过去因为远离法，我现在得到果报，}%

@{所生既如是，當行何法行？}@
%{既然生成了这样，应该修持什么法呢？}%

@{一切善心根，世間之牟尼，}@
%{一切善的心根，世间所有的牟尼，}%

@{彼根恒不退，常得好果報。}@
%{他们的根源永远不退转，总是得到好的果报。}%

@{煩惱苦纏綿，而得種種怖，}@
%{烦恼的痛苦缠绵不绝，就会得到各种各样的恐惧，}%

@{於他愛障難，生罪而自感。}@
%{对他人有爱欲的障碍，就会造罪，而自己感召。}%

@{若人於處處，能起於善願，}@
%{如果一个人在各个地方，都能发起善愿，}%

@{而感彼彼福，獲得供養果。}@
%{就会感召相应的福报，获得供养的果报。}%

@{若人於處處，作罪取快樂，}@
%{如果一个人在各个地方，造罪取乐，}%

@{而感彼彼報，獲得苦器侵。}@
%{就会感召相应的果报，获得苦器的侵害。}%

@{月藏中清涼，廣博妙香潔，}@
%{月藏中清凉，广阔，妙香洁净，}%

@{佛音味第一，非修而不得。}@
%{佛音的味道是第一，不是修行就无法得到。}%

@{而彼善逝子，得解善逝法，}@
%{而那些善逝的弟子，能够理解善逝的法，}%

@{如蓮出最上，亦如仁覺月。}@
%{就像莲花从最上层生长出来，也像仁觉的月亮一样。}%

@{焰魔之獄卒，牽引於罪魂，}@
%{焰魔的地狱卒，牵引着罪恶的灵魂，}%

@{火坑及洋銅，燒煮悉皆入。}@
%{火坑和洋铜，烧煮，全部都进入。}%

@{焰熾殺器杖，斷肉百千斤，}@
%{火焰炽热的杀器杖，断肉，成千上万斤，}%

@{墮落熱鐵地，斯由多不善。}@
%{堕落到滚烫的铁地上，这是由于多造恶业。}%

@{是故心作善，極微細觀察，}@
%{所以心要行善，要极其细微地观察，}%

@{依彼金剛幡，修學而作觀。}@
%{依靠那金刚幡，修行并观察。}%

@{初學觀和合，不觀汝非學，}@
%{初学者观察和合，不观察，你就不是学者，}%

@{而無最上名，汝要迴心作。}@
%{而没有最高的名称，你要回心转意去做。}%

@{生中之所作，增長於罪苦，}@
%{一生中所作的，增长了罪业和痛苦，}%

@{上事業不修，彼下不求勝。}@
%{上面的事业不修行，下面的也不追求超越。}%

@{三種事應知，由業煩惱力，}@
%{三种事情应该知道，是由业力和烦恼的力量，}%

@{將來之惡因，於此何復作？}@
%{将来的恶因，在这里又做什么呢？}%

@{世間之煩惱，拘人不自由，}@
%{世间的烦恼，拘束着人，不自由，}%

@{我如人不能，是故我無作。}@
%{我就像人一样，不能，所以我不做。}%

@{下業之所修，云何令安住？}@
%{低下的业力所修的，怎么才能使它安住呢？}%

@{當觀我無我，而此我所作。}@
%{应该观察我无我，而这是我所做的。}%

@{一滴之甘露，鳥食變金翅，}@
%{一滴甘露，鸟食变成了金翅鸟，}%

@{我意謂微劣，能脫少苦難。}@
%{我以为微不足道，就能解脱少许的苦难。}%

@{嗔作無心難，以不善罪故，}@
%{嗔恨就像无心一样难，因为不善的罪业，}%

@{無心見發起，廣大勝難及。}@
%{无心所见发起，广大而超越，难以企及。}%

@{是故清淨心，頌作此文句，}@
%{所以清净的心，颂作这些文句，}%

@{使知彼三界，我遠離戲論。}@
%{使人知道那三界，我远离戏论。}%

@{我得勝一切，無人能勝我，}@
%{我战胜一切，没有人能战胜我，}%

@{我今而自知，是佛師子子。}@
%{我现在自己知道，我是佛陀的狮子子。}%

@{有情離我人，而彼得最上，}@
%{众生远离我这个人，而得到最上乘，}%

@{不降懈怠冤，懈怠冤自降。}@
%{不降服懈怠的冤家，懈怠的冤家会自己降服。}%

@{以惡趣所牽，身善速破壞，}@
%{因为被恶趣所牵引，身体的善行很快就会破壞，}%

@{由僕從愚惡，寄食而受瘦。}@
%{由于仆人愚蠢邪恶，寄食而变得瘦弱。}%

@{彼受於一切，修行住我慢，}@
%{他们接受一切，修行，住着我慢，}%

@{而此得名聲，下劣云何說？}@
%{而这却得到名声，低劣，怎么可以说呢？}%

@{如是若勇猛，自勝彼冤家，}@
%{如果这样勇猛，自己战胜那些冤家，}%

@{勇猛行此修，慢冤而不勝。}@
%{勇猛地修行，慢的冤家却不能战胜。}%

@{彼慢心若起，此實我冤家，}@
%{那个慢心如果生起，这确实是我的冤家，}%

@{勝果雖欲生，是果悉皆捨。}@
%{虽然想要生起胜果，却把这些果报都舍弃了。}%

@{喻精進師子，煩惱獸中見，}@
%{比喻精进的狮子，在烦恼的野兽中出现，}%

@{煩惱獸千萬，雖眾不能敵。}@
%{烦恼的野兽成千上万，虽然很多，但不能与之抗衡。}%

@{世有大苦惱，人自悉具見，}@
%{世间有很大的痛苦，人自己都亲眼看见，}%

@{煩惱不降伏，乃得如是苦。}@
%{烦恼不降伏，就会得到这样的痛苦。}%

@{我寧使頭落，及刳剔心腸，}@
%{我宁愿头颅落地，以及剖开我的心肠，}%

@{煩惱諸冤家，一切我不降。}@
%{烦恼的冤家，一切我都不会降服。}%

@{因修此精進，得彼慢業盡，}@
%{因为修持这种精进，就能消除那些慢的业力，}%

@{獲得勝果報，自感嬉戲樂。}@
%{获得胜利的果报，自己就会感到欢喜快乐。}%

@{為快樂修因，彼却不獲得，}@
%{为了快乐，修持因，却得不到，}%

@{所修不決定，亦得不殊勝。}@
%{所修的没有确定，也无法得到殊胜。}%

@{輪迴欲不足，喻貪刀刃蜜，}@
%{轮回的欲望永远不会满足，比喻贪吃刀刃上的蜜，}%

@{福甘露若貪，食之後轉美。}@
%{福报的甘露如果贪吃，吃过之后反而更美味。}%

@{是故業寂靜，感妙果隨行，}@
%{所以业力寂静了，就会感召妙果随之而来，}%

@{如日溫月寒，晝夜而相逐。}@
%{就像太阳温暖，月亮寒冷，昼夜相互追逐一样。}%

@{精進之有力，能破於懈怠，}@
%{精进的力量，能够破除懈怠，}%

@{獲得遠離故，深心而愛樂。}@
%{获得远离，内心深处就会热爱。}%

@{煩惱棒堅牢，鬪彼念慧劍，}@
%{烦恼的棍棒坚固，与它斗争的是念头的智慧之剑，}%

@{喻棒劍相持，同彼女人學。}@
%{比喻棍棒和宝剑相互支撑，就像女人学习一样。}%

@{執劍手無力，失之而怖急，}@
%{拿着剑的手没有力量，就会失去，感到恐惧，}%

@{念劍失亦然，地獄而在心。}@
%{念头的剑也一样，就会失去，地狱就在心中。}%

@{世間知善人，不肯飲毒血，}@
%{世间都知道善良的人，不愿意喝毒血，}%

@{心過亦復然，心過而不作。}@
%{心过也是这样，心过就不做。}%

@{出家精進心，喻執持油鉢，}@
%{出家精进的心，比喻拿着油钵，}%

@{鉢墜必當死，墜之故驚怖。}@
%{钵掉下来一定会死，掉下来就会感到恐惧。}%

@{著睡眠懈怠，喻毒蛇在懷，}@
%{沉迷于睡眠懈怠，比喻毒蛇在怀抱里，}%

@{不去當被傷，去之宜須急。}@
%{不去就会被伤害，应该赶快把它拿走。}%

@{一一之深過，要迴心思惟，}@
%{每一个深刻的过失，都要回心转意地思考，}%

@{此過不可守，云何我復作？}@
%{这个过失不能再犯，我为什么还要再犯呢？}%

@{和合之業因，斷以正念劍，}@
%{和合的业因，用正念的宝剑斩断，}%

@{云何名自位，此念而獲得？}@
%{怎么才能被称为自性，用这个念头就能获得。}%

@{正念心不發，纖毫不能滅，}@
%{正念的心不发起，一丝一毫也不能消除，}%

@{來業如往行，一切報皆得。}@
%{过去的业力就像往常一样，所有的果报都会得到。}%

@{如彼覩羅綿，隨風而來往，}@
%{就像那睹罗绵，随着风来来往往，}%

@{精進人亦然，增上如是得。}@
%{精进的人也是一样，不断地增长，这样才能得到。}%



@{菩提行經卷第二}@
