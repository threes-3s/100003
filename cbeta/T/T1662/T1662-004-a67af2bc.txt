@{西天中印度惹爛馱囉國密林寺三藏明教大師賜紫沙門臣天息災奉詔譯}@
%{西天中印度惹烂驮囉国密林寺三藏明教大师赐紫沙门臣天息灾奉诏译}%

@{菩提心般若波羅蜜多品第七}@
%{菩提心般若波罗蜜多品第七}%

@{如來智慧仁，為一切世間，}@
%{如来智慧仁，为一切世间，}%
@{令求遠離苦，是故智慧生。}@
%{令求远离苦，是故智慧生。}%

@{真如及世間，今說此二法，}@
%{真如及世间，今说此二法，}%
@{知佛真如故，說法而智慧。}@
%{知佛真如故，说法而智慧。}%

@{彼世間凡夫，見二種相應，}@
%{彼世间凡夫，见二种相应，}%
@{害及勝害等，乃世相應事。}@
%{害及胜害等，乃世相应事。}%

@{彼二事見已，見之乃為智，}@
%{彼二事见已，见之乃为智，}%
@{智見世間性，是喻於真如。}@
%{智见世间性，是喻于真如。}%

@{此說無去來，智者無不見，}@
%{此说无去来，智者无不见，}%
@{色等甚分明，乃世相應事。}@
%{色等甚分明，乃世相应事。}%

@{不淨而為淨，智者喻有利，}@
%{不净而为净，智者喻有利，}%
@{為知世間故，是說世間性。}@
%{为知世间故，是说世间性。}%

@{為見於真如，見以剎那住，}@
%{为见于真如，见以刹那住，}%
@{世間行相應，此行無過失。}@
%{世间行相应，此行无过失。}%

@{知女人不淨，異世諸害事，}@
%{知女人不净，异世诸害事，}%
@{謂佛福虛幻，使我云何信？}@
%{谓佛福虚幻，使我云何信？}%

@{有情若幻境，云何復生滅？}@
%{有情若幻境，云何复生灭？}%
@{彼因集和合，乃得於幻緣。}@
%{彼因集和合，乃得于幻缘。}%

@{有情種子生，云何有真實？}@
%{有情种子生，云何有真实？}%
@{殺彼虛幻人，無心性等罪。}@
%{杀彼虚幻人，无心性等罪。}%

@{平等心虛幻，罪福得生起，}@
%{平等心虚幻，罪福得生起，}%
@{真言力等持，幻境心無著。}@
%{真言力等持，幻境心无著。}%

@{以彼種種幻，種種因業生，}@
%{以彼种种幻，种种因业生，}%
@{何有於一人，得於一切力。}@
%{何有于一人，得于一切力。}%

@{若住於真如，或住於淨戒，}@
%{若住于真如，或住于净戒，}%
@{如是即佛行，誰云菩提行？}@
%{如是即佛行，谁云菩提行？}%

@{因緣當斷盡，幻化不可得，}@
%{因缘当断尽，幻化不可得，}%
@{因緣若斷盡，無生而自得。}@
%{因缘若断尽，无生而自得。}%

@{若不住疑妄，幻境而不立，}@
%{若不住疑妄，幻境而不立，}%
@{幻境若彼無，一切不可得。}@
%{幻境若彼无，一切不可得。}%

@{如是即真如，得現於心體，}@
%{如是即真如，得现于心体，}%
@{心如是若分，虛幻何由見？}@
%{心如是若分，虚幻何由见？}%

@{心不自見心，世尊之所說，}@
%{心不自见心，世尊之所说，}%
@{如劍刃雖利，雖利不自斷。}@
%{如剑刃虽利，虽利不自断。}%

@{自性由若斯，復喻如燈光，}@
%{自性由若斯，复喻如灯光，}%
@{破闇然得名，而不云自照。}@
%{破闇然得名，而不云自照。}%

@{又若水精珠，體本唯清澈，}@
%{又若水精珠，体本唯清澈，}%
@{因青而有青，影現隨眾色。}@
%{因青而有青，影现随众色。}%

@{非青而現青，如心而自作，}@
%{非青而现青，如心而自作，}%
@{又如彼燈光，智者知此說。}@
%{又如彼灯光，智者知此说。}%

@{智慧此開通，知者何所說？}@
%{智慧此开通，知者何所说？}%
@{雖開而不開，如人無所覩。}@
%{虽开而不开，如人无所覩。}%

@{石女義不生，與此義不二，}@
%{石女义不生，与此义不二，}%
@{亦同無心識，緣念無所得。}@
%{亦同无心识，缘念无所得。}%

@{非念而別生，虛妄念如毒，}@
%{非念而别生，虚妄念如毒，}%
@{謂若因若果，為法而自說。}@
%{谓若因若果，为法而自说。}%

@{有談眼藥方，見瓶而無藥，}@
%{有谈眼药方，见瓶而无药，}%
@{若見聞覺知，此有而非有。}@
%{若见闻觉知，此有而非有。}%

@{念斷於苦因，此實念當念，}@
%{念断于苦因，此实念当念，}%
@{念念而無別，此心當平等。}@
%{念念而无别，此心当平等。}%

@{前塵常惑人，了之無所有，}@
%{前尘常惑人，了之无所有，}%
@{如幻而不實，妄心而自見。}@
%{如幻而不实，妄心而自见。}%

@{住塵處輪迴，喻空無所依，}@
%{住尘处轮回，喻空无所依，}%
@{住塵性亦然，亦無有所得。}@
%{住尘性亦然，亦无有所得。}%

@{若與不善俱，不善汝所得，}@
%{若与不善俱，不善汝所得，}%
@{若心有取捨，施一切如來。}@
%{若心有取舍，施一切如来。}%

@{如是用心意，而有何功德？}@
%{如是用心意，而有何功德？}%
@{幻境一切知，煩惱云何斷？}@
%{幻境一切知，烦恼云何断？}%

@{於彼幻三毒，遠離而不作，}@
%{于彼幻三毒，远离而不作，}%
@{知於煩惱心，彼作而未盡。}@
%{知于烦恼心，彼作而未尽。}%

@{於彼得見時，空有意無力，}@
%{于彼得见时，空有意无力，}%
@{煩惱性非盡，與空而相雜。}@
%{烦恼性非尽，与空而相杂。}%

@{至彼無所學，彼後乃得盡，}@
%{至彼无所学，彼后乃得尽，}%
@{彼性而無得，亦復不能見。}@
%{彼性而无得，亦复不能见。}%

@{彼性若無住，云何住此身？}@
%{彼性若无住，云何住此身？}%
@{若性而無有，身住於無性。}@
%{若性而无有，身住于无性。}%

@{是性如去來，隨現而無著，}@
%{是性如去来，随现而无著，}%
@{劫樹與摩尼，能如意圓滿。}@
%{劫树与摩尼，能如意圆满。}%

@{佛變化亦然，當為斯行願，}@
%{佛变化亦然，当为斯行愿，}%
@{喻法呪林樹，呪成而枯壞。}@
%{喻法呪林树，呪成而枯坏。}%

@{毒等雖久害，彼彼皆消除，}@
%{毒等虽久害，彼彼皆消除，}%
@{菩薩之修行，所作諸事業。}@
%{菩萨之修行，所作诸事业。}%

@{菩提行最勝，佛樹能成就，}@
%{菩提行最胜，佛树能成就，}%
@{以彼平等行，而住於寂靜。}@
%{以彼平等行，而住于寂静。}%

@{及作不思議，供養得何果？}@
%{及作不可思议，供养得何果？}%
@{隨彼所行因，而得於彼果。}@
%{随彼所行因，而得于彼果。}%

@{供養等真實，得果而稱實，}@
%{供养等真实，得果而称实，}%
@{云何得法空，實得解脫法。}@
%{云何得法空，实得解脱法。}%

@{不離牟尼道，當得於菩提，}@
%{不离牟尼道，当得于菩提，}%
@{汝不求大乘，何法求圓滿？}@
%{汝不求大乘，何法求圆满？}%

@{二乘得成就，成就非圓滿，}@
%{二乘得成就，成就非圆满，}%
@{若彼所作因，怖畏於大乘，}@
%{若彼所作因，怖畏于大乘，}%

@{別怖怖非實，此怖實名怖。}@
%{别怖怖非实，此怖实名怖。}%

@{此法要當知，大乘之所論，}@
%{此法要当知，大乘之所论，}%
@{離此為他法，知彼外道論。}@
%{离此为他法，知彼外道论。}%

@{法乃僧根本，僧知法出離，}@
%{法乃僧根本，僧知法出离，}%
@{心若有著處，涅槃不可得。}@
%{心若有著处，涅槃不可得。}%

@{解脫心無著，煩惱得消滅，}@
%{解脱心无著，烦恼得消灭，}%
@{煩惱業消除，斯由解脫力。}@
%{烦恼业消除，斯由解脱力。}%

@{愛取不相緣，以此無執持，}@
%{爱取不相缘，以此无执持，}%
@{愛業而羸劣，是無有癡愛。}@
%{爱业而羸劣，是无有痴爱。}%

@{受愛得相緣，此受而有得，}@
%{受爱得相缘，此受而有得，}%
@{安住有著心，是得名處處。}@
%{安住有著心，是得名处处。}%

@{若心之不空，復得名為著，}@
%{若心之不空，复得名为著，}%
@{心性若云空，如識而無得。}@
%{心性若云空，如识而无得。}%

@{如應正等覺，所說之妙法，}@
%{如应正等觉，所说之妙法，}%
@{是義乃大乘，大乘行平等。}@
%{是义乃大乘，大乘行平等。}%

@{說法之一時，了一切過患，}@
%{说法之一时，了一切过患，}%
@{一味之平等，諸佛無不說。}@
%{一味之平等，诸佛无不說。}%

@{迦葉大尊者，如言之不知，}@
%{迦叶大尊者，如言之不知，}%
@{彼汝云不覺，不受當何作？}@
%{彼汝云不覺，不受当何作？}%

@{解脫力若怖，輪迴得成就，}@
%{解脱力若怖，轮回得成就，}%
@{迷彼苦空事，而得於此果。}@
%{迷彼苦空事，而得于此果。}%

@{迷空彼若此，不得謗於法，}@
%{迷空彼若此，不得谤于法，}%
@{此空審觀察，是故得不疑。}@
%{此空审观察，是故得不疑。}%

@{離闇知煩惱，因法知於空，}@
%{离闇知烦恼，因法知于空，}%
@{欲速一切知，彼言審觀察。}@
%{欲速一切知，彼言审观察。}%

@{若物生於苦，是苦怖得生，}@
%{若物生于苦，是苦怖得生，}%
@{彼苦因空作，彼何得生怖？}@
%{彼苦因空作，彼何得生怖？}%

@{若於彼物怖，斯即名我所，}@
%{若于彼物怖，斯即名我所，}%
@{如是我無所，苦怖云何得？}@
%{如是我无所，苦怖云何得？}%

@{牙齒髮爪甲，骨肉并血髓，}@
%{牙齿头发爪甲，骨肉并血髓，}%
@{鼻洟唾膿涎，脂肪及腸胃，}@
%{鼻涕唾脓涎，脂肪及肠胃，}%
@{便痢汗熱風，九漏并六識，}@
%{便痢汗热风，九漏并六识，}%
@{如是諸法等，一切皆無我。}@
%{如是诸法等，一切皆无我。}%

@{說彼智與聲，聲恒受一切，}@
%{说彼智与声，声恒受一切，}%
@{若說聲智離，彼離云何知？}@
%{若说声智离，彼离云何知？}%

@{若智之不知，彼智難知故，}@
%{若智之不知，彼智难知故，}%
@{彼智既決定，乃近於智智。}@
%{彼智既决定，乃近于智智。}%

@{此智非聲受，彼聲何以聞？}@
%{此智非声受，彼声何以闻？}%
@{彼聲近於心，彼知色如是。}@
%{彼声近于心，彼知色如是。}%

@{若受於色聲，而色復何受？}@
%{若受于色声，而色复何受？}%
@{如彼一父子，思惟無真實。}@
%{如彼一父子，思惟无真实。}%

@{有情塵所翳，無父亦無子，}@
%{有情尘所翳，无父亦无子，}%
@{知聲色如是，亦無於自性。}@
%{知声色如是，亦无于自性。}%

@{彼色如是知，喻樂暫和合，}@
%{彼色如是知，喻乐暂和合，}%
@{彼自性如是，彼一而言有。}@
%{彼自性如是，彼一而言有。}%

@{餘色咸不實，此說色下品，}@
%{余色咸不实，此说色下品，}%
@{彼一切智心，煩惱悉清淨。}@
%{彼一切智心，烦恼悉清净。}%

@{思惟一覺心，彼等彼若無，}@
%{思惟一觉心，彼等彼若无，}%
@{愛若虛不實，云何住於見？}@
%{爱若虚不实，云何住于见？}%

@{無我而無心，此心喻畫像，}@
%{无我而无心，此心喻画像，}%
@{是心智相應，清淨愚癡破。}@
%{是心智相应，清净愚痴破。}%

@{如是之自心，彼作云何作？}@
%{如是之自心，彼作云何作？}%
@{彼愚癡無行，此我而虛作。}@
%{彼愚痴无行，此我而虚作。}%

@{有行自出離，而無惡業果，}@
%{有行自出离，而无恶业果，}%
@{破壞業若為，善果云何得？}@
%{破坏业若为，善果云何得？}%

@{此二之行果，互相破成就，}@
%{此二之行果，互相破成就，}%
@{彼說知不虛，彼自而無事。}@
%{彼说知不虚，彼自而無事。}%

@{因果定相應，惡見要不生，}@
%{因果定相应，恶见要不生，}%
@{此行而實住，作受今當說。}@
%{此行而实住，作受今当说。}%

@{過去未來心，彼我無有生，}@
%{过去未来心，彼我无有生，}%
@{此心生我破，我無復生起。}@
%{此心生我破，我无复生起。}%

@{如芭蕉作柱，無所能勝任，}@
%{如芭蕉作柱，无所能胜任，}%
@{我心生亦然，是得善觀察。}@
%{我心生亦然，是得善观察。}%

@{有情若不有，此行云何為？}@
%{有情若不有，此行云何为？}%
@{彼行今若為，而為有癡事。}@
%{彼行今若为，而为有痴事。}%

@{有情何實無，癡喻其愛事，}@
%{有情何实无，痴喻其爱事，}%
@{若滅於苦惱，當斷於癡事。}@
%{若灭于苦恼，当断于痴事。}%

@{我慢為苦因，癡是得增長，}@
%{我慢为苦因，痴是得增长，}%
@{彼事心不迴，觀空為最上。}@
%{彼事心不迴，观空为最上。}%

@{無足無脛膝，無腰復無腿，}@
%{无足无胫膝，无腰复无腿，}%
@{無臂亦無肩，無臍無胸背，}@
%{无臂亦无肩，无脐无胸背，}%
@{無肋兼無脇，無手亦無鼻，}@
%{无肋兼无脇，无手亦无鼻，}%
@{無項復無頭，骨鎖等皆爾。}@
%{无项复无头，骨锁等皆尔。}%

@{觀此一切身，不行於一處，}@
%{观此一切身，不行于一处，}%
@{彼行於處處，何處自安住？}@
%{彼行于处处，何处自安住？}%

@{以彼身手等，一切處皆住，}@
%{以彼身手等，一切处皆住，}%
@{彼一身如是，乃至於手等。}@
%{彼一身如是，乃至于手等。}%

@{無內無外身，何獨身手等？}@
%{无内无外身，何独身手等？}%
@{手等無分別，云何彼復有？}@
%{手等无分别，云何彼复有？}%

@{彼既無癡身，寧云意手等？}@
%{彼既无痴身，宁云意手等？}%
@{住已近殊勝，觀者知人喻。}@
%{住已近殊胜，观者知人喻。}%

@{若彼因和合，木人此可同，}@
%{若彼因和合，木人此可同，}%
@{若了如是相，彼身同此見。}@
%{若了如是相，彼身同此见。}%

@{如是捨足指，手指亦皆捨，}@
%{如是舍足指，手指亦皆舍，}%
@{彼初觀節合，後見節自離。}@
%{彼初观节合，后见节自离。}%

@{此身破已竟，彼住分別見，}@
%{此身破已竟，彼住分别见，}%
@{分別見此身，得喻如虛空。}@
%{分别见此身，得喻如虚空。}%

@{如是之夢色，智者何所樂？}@
%{如是之梦色，智者何所乐？}%
@{設施若無身，何有男女等？}@
%{设施若无身，何有男女等？}%

@{若喜真得苦，此者何不解？}@
%{若喜真得苦，此者何不解？}%
@{觀察此云何，愛樂深煩惱。}@
%{观察此云何，爱乐深烦恼。}%

@{樂者之不實，如彼無執受，}@
%{乐者之不实，如彼无执受，}%
@{汝苦復云何？如彼自無得。}@
%{汝苦复云何？如彼自无得。}%

@{彼有苦微細，既微而不說，}@
%{彼有苦微细，既微而不说，}%
@{以彼微細故，不說令他喜。}@
%{以彼微细故，不说令他喜。}%

@{因瞋而苦生，既生而有滅，}@
%{因瞋而苦生，既生而有灭，}%
@{若於定有見，於生自不受。}@
%{若于定有见，于生自不受。}%

@{如是而既知，如是觀因果，}@
%{如是而既知，如是观因果，}%
@{禪愛或相應，得生彼疑地。}@
%{禅爱或相应，得生彼疑地。}%

@{善根之所利，皆為於何人？}@
%{善根之所利，皆为于何人？}%
@{彼此何和合？和合而何得？}@

@{明力皆成就，羅剎拏吉儞，}@
%{明力都成就，罗刹拏吉儞，}%

@{斯等皆悲愍。無有苦有情，}@
%{这些都悲悯。没有苦难的有情众生，}%

@{無罪復無病，不輕慢下劣，}@
%{没有罪过又没有病，不轻慢下劣，}%

@{煩惱無所得。讀誦而自在，}@
%{烦恼没有得到。诵读而自在，}%

@{隨意而行住，眾集乃恒常，}@
%{随意而行住，大众集聚乃恒常，}%

@{成就於僧事。苾芻住淨戒，}@
%{成就于僧事。比丘住持净戒，}%

@{復得一切解，觀察於心業，}@
%{又得到一切解脱，观察于心业，}%

@{捨離諸煩惱。苾芻所得利，}@
%{舍离诸烦恼。比丘所得利益，}%

@{當遠離鬪諍，諸出家亦然，}@
%{应该远离斗争，诸出家也一样，}%

@{不得破禁戒。得戒而守護，}@
%{不得破除禁戒。得到戒律而守护，}%

@{恒樂盡諸罪，若彼不破戒，}@
%{恒乐尽除一切罪过，如果他们不破戒，}%

@{得益往天趣。若彼持鉢者，}@
%{得到利益往天趣。如果他们持钵者，}%

@{為得於善利，得清淨種子，}@
%{为了得到善利，得到清净种子，}%

@{名聞滿諸方，永不受罪苦，}@
%{名声满诸方，永远不受罪苦，}%

@{恒行無苦處。無邊諸有情，}@
%{恒行无苦处。无边诸有情，}%

@{供養一切佛，當受一天身。}@
%{供养一切佛，应该得到一天身。}%

@{彼成佛世間，不思議有情，}@
%{他们成佛世间，不可思议的有情众生，}%

@{樂佛而得樂，願為於世間。}@
%{快乐佛而得到快乐，愿为于世间。}%

@{菩薩得成就，彼尊若思惟，}@
%{菩萨得到成就，那些尊者如果思考，}%

@{彼有情令得，辟支佛安樂，}@
%{那些有情众生让他们得到，辟支佛的安乐，}%

@{及得聲聞樂，天人阿修羅，}@
%{以及得到声闻的快乐，天人阿修罗，}%

@{意重而恒護。若彼宿命通，}@
%{意重而恒护。如果他们宿命通，}%

@{出家此恒得，若彼歡喜地，}@
%{出家这里恒能得到，如果他们欢喜地，}%

@{文殊師利住。我若以彼位，}@
%{文殊师利住。我如果以那个地位，}%

@{隨力而能與，若知和合住，}@
%{随力而能给予，如果知道和合住，}%

@{得生於一切。若有欲見者，}@
%{就能生于一切。如果有想要见的人，}%

@{及有欲聞者，如是彼得見，}@
%{以及有想要听的人，这样就能见到，}%

@{文殊師利尊。如日照十方，}@
%{文殊师利尊。如同太阳照耀十方，}%

@{為一切有情，彼文殊修行，}@
%{为了所有有情众生，文殊修行，}%

@{我得如是行。彼或住虛空，}@
%{我得到这样修行。他或者住在虚空，}%

@{或住於世間，今我住亦然，}@
%{或者住在世间，现在我住也一样，}%

@{得壞世間苦。世間若有苦，}@
%{就能破坏世间苦难。世间如果有苦难，}%

@{彼一切我得，世間一切善，}@
%{那些一切我都能得到，世间一切善，}%

@{菩薩之樂得。一藥救世間，}@
%{菩萨的快乐都能得到。一味药救世间，}%

@{一切皆富樂，一切同利養，}@
%{一切皆富乐，一切同利养，}%

@{佛教而久住。以善意清淨，}@
%{佛教而久住。以善意清净，}%

@{歸命於文殊，我說善知識，}@
%{归命于文殊，我说善知识，}%

@{清淨此增長。}@
%{清净此增长。}%

@{菩提行經卷第四}@
%{菩提行经卷第四}%
