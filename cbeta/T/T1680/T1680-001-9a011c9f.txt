@{世尊最殊勝，善斷諸惑種，}@
%{世尊最殊胜，善于断除一切烦恼的根源，}%
@{無量勝功德，總集如來身。}@
%{无量胜妙功德，都集聚在如来的身上。}%

@{唯佛可歸依，可讚可承事，}@
%{只有佛陀可以皈依，可以赞叹，可以供养侍奉，}%
@{如理思惟者，宜應住此教。}@
%{如果能够如理思惟，就应该依止这个教法。}%

@{諸惡煩惱習，護世者已除，}@
%{一切恶业烦恼习气，护世者(佛陀) 已经全部断除，}%
@{福智二俱圓，唯尊不退沒。}@
%{福德智慧二者都圆满，只有尊者(佛陀) 不会退转。}%

@{縱生惡見者，於尊起嫌恨，}@
%{即使生起邪恶的见解，对尊者(佛陀) 产生厌恨，}%
@{伺求身語業，無能得瑕隙。}@
%{也无法从他的身口意三业中找到一丝瑕疵。}%

@{記我得人身，聞法生歡喜，}@
%{我记得自己得到人身，听到佛法就非常欢喜，}%
@{譬如巨海內，盲龜遇楂穴。}@
%{就像在大海里，一只瞎眼的海龟遇到一根木头一样。}%

@{忘念恒隨逐，惑業墮深坑，}@
%{忘记佛法，被烦恼业力所追逐，坠入无边的苦海，}%
@{故我以言詞，歎佛實功德。}@
%{所以我要用语言，赞叹佛陀真实的功德。}%

@{牟尼無量境，聖德無邊際，}@
%{释迦牟尼佛的境界无量无边，圣德无穷无尽，}%
@{為求自利故，我今讚少分。}@
%{为了求取自利，我现在只赞叹佛陀功德的一小部分。}%

@{敬禮無師智，希有眾事性，}@
%{恭敬礼拜无师智(佛陀) ，他具有稀有的、一切事物的本质，}%
@{福慧及威光，誰能知數量？}@
%{他的福德智慧和威光，谁能知道有多少呢？}%

@{如來德無限，無等無能說，}@
%{如来的功德无穷无尽，无等无比，无法用语言形容，}%
@{我今求福利，假讚以名言。}@
%{我现在为了求取福报，就用语言来赞叹他。}%

@{我智力微淺，佛德無崖際，}@
%{我的智慧浅薄，佛陀的功德无边无际，}%
@{唯願大慈悲，拯我無歸處。}@
%{只希望佛陀的大慈悲，能够救度我这个无处可归的人。}%

@{怨親悉平等，無緣起大悲，}@
%{怨恨亲爱都平等对待，没有缘由地生起大悲心，}%
@{普於眾生界，恒作真善友。}@
%{普度一切众生，永远做他们的真朋友。}%

@{內財尚能捨，何況於外財，}@
%{内财(身体) 尚且能舍弃，何况外财(物质) ，}%
@{尊無悋惜心，求者滿其願。}@
%{尊者(佛陀) 没有吝惜之心，求者都能得到满足。}%

@{以身護彼身，以命贖他命，}@
%{用自己的身体保护别人的身体，用自己的生命救赎别人的生命，}%
@{全軀救一鴿，歡喜無慳悋。}@
%{为了救一只鸽子，不惜自己的身体，欢喜地没有任何吝惜。}%

@{尊不畏惡道，亦不貪善趣，}@
%{尊者(佛陀) 不畏惧恶道，也不贪求善趣，}%
@{但為心澄潔，尸羅由此成。}@
%{只为了让自己的心清净，所以成就了清净的戒律。}%

@{常離諸邪曲，恒親質直者，}@
%{永远远离一切邪曲，恒常亲近正直，}%
@{諸業本性空，唯居第一義。}@
%{一切业的本质都是空的，只住于第一义谛。}%

@{眾苦逼其身，尊能善安慮，}@
%{当众苦逼迫他的身体时，尊者(佛陀) 能够善于安顿思虑，}%
@{正智斷諸惑，有過悉興悲。}@
%{用正智断除一切烦恼，如果有过错，就生起悲悯之心。}%

@{殉命濟他難，生無量歡喜，}@
%{为了救助他人，不惜牺牲自己的生命，生起无量的欢喜，}%
@{如死忽重蘇，此喜過於彼。}@
%{就像死后突然复活，这种欢喜超过了前者。}%

@{怨對害其身，一切時恒惱，}@
%{仇敌伤害他的身体，时时刻刻都在恼恨他，}%
@{不觀其過惡，常起大悲心。}@
%{他不看仇敌的过失，常生起大悲心。}%

@{正遍菩提種，心恒所珍玩，}@
%{正遍知菩提的种子，他一直珍藏在心中，}%
@{大雄難勝智，無有能及者。}@
%{大雄无畏的智慧，无人能及。}%

@{無等菩提果，苦行是其因，}@
%{无等无比的菩提果，苦行是它的因缘，}%
@{由此不顧身，勤修諸勝品。}@
%{因此他不顾自己的身体，勤修一切胜妙法门。}%

@{豪貴與貧賤，等引以大悲，}@
%{对富贵和贫贱的人，平等地用大悲心引导他们，}%
@{於諸差別中，而無高下想。}@
%{在一切差别中，没有高低贵贱的念头。}%

@{勝樂等持果，心無有貪著，}@
%{胜乐等持的果位，心中没有贪著，}%
@{普濟諸群生，大悲無間斷。}@
%{普度一切众生，大悲心永不间断。}%

@{尊雖遭極苦，於樂不悕求，}@
%{尊者(佛陀) 虽然遭受极大的苦难，对快乐却毫不贪求，}%
@{妙智諸功德，殊勝無能共。}@
%{他的妙智和一切功德，无比殊胜，无人能比。}%

@{染淨諸雜法，簡偽取其真，}@
%{在染污和清净的各种法中，他舍弃虚假，取其真实，}%
@{如清淨鵝王，飲乳棄其水。}@
%{就像清净的鹅王，饮用乳汁，却舍弃水一样。}%

@{於無量億劫，勇猛趣菩提，}@
%{在无量亿劫中，他勇猛精进地求取菩提，}%
@{於彼生生中，喪身求妙法。}@
%{在每一世中，都为了寻求妙法而舍弃自己的生命。}%

@{三僧祇數量，精勤無懈惓，}@
%{三个僧祇劫的时间，他精勤努力，毫不懈怠，}%
@{持此為勝伴，以證妙菩提。}@
%{以这个作为最好的伴侣，来证得妙菩提。}%

@{尊無嫉姤心，於劣除輕想，}@
%{尊者(佛陀) 没有嫉妒和轻慢之心，对不如自己的人，没有轻视的念头，}%
@{平等無乖諍，勝行悉圓成。}@
%{平等对待，没有争执，一切胜妙的行为都圆满成就。}%

@{尊唯重因行，非求果位圓，}@
%{尊者(佛陀) 只重视因行，而不是追求果位的圆满，}%
@{遍修諸勝業，眾德自成滿。}@
%{遍修一切胜妙的业，各种功德自然圆满成就。}%

@{勤修出離法，超昇眾行頂，}@
%{勤修出离法，超越一切行为的顶峰，}%
@{坐臥經行處，無非勝福田。}@
%{坐、卧、经行，处处都是最好的福田。}%

@{拔除眾過染，增長清淨德，}@
%{拔除一切过失和染污，增长清净的功德，}%
@{斯由積行成，唯尊最無上。}@
%{这是因为积聚了无量功德而成就的，只有尊者(佛陀) 是最无上的。}%

@{眾福皆圓滿，諸過悉蠲除，}@
%{一切福德都圆满，一切过失都消除，}%
@{如來淨法身，塵習皆已斷。}@
%{如来的清净法身，一切尘垢习气都已断除。}%

@{資糧集更集，功歸調御身，}@
%{资粮不断地集聚，功德归于调伏自己的身心，}%
@{欲求於譬類，無能與佛等。}@
%{想要找到可以比拟的人，没有谁能与佛陀相提并论。}%

@{遍觀諸世間，災橫多障惱，}@
%{遍观世间，灾难横祸、障碍烦恼很多，}%
@{縱有少分善，易得為比對；}@
%{即使有少许善法，也很容易找到可以比拟的；}%

@{遠離諸過患，湛然安不動，}@
%{远离一切过失，安然不动，}%
@{最勝諸善根，無能為譬喻。}@
%{最胜妙的善根，没有能比拟的。}%

@{如來智深遠，無底無邊際，}@
%{如来的智慧深远，无底无边际，}%
@{世事喻佛身，牛跡方大海。}@
%{世间万物比喻佛陀的身体，就像牛蹄印比喻大海一样。}%

@{深仁荷一切，世間無有比，}@
%{深厚仁爱承载一切，世间没有能比的，}%
@{大地持重擔，喻此實為輕。}@
%{大地承载着沉重的负担，比喻佛陀的仁爱，就如同轻如鸿毛。}%

@{愚癡闇已除，牟尼光普照，}@
%{愚痴的黑暗已经消除，释迦牟尼佛的光明普照一切，}%
@{世智非能譬，如螢對日光。}@
%{世间的智慧无法比拟，就像萤火虫比喻太阳的光芒一样。}%

@{如來三業淨，秋月皎空池，}@
%{如来的身口意三业清净，就像秋天的月亮映照在空旷的池塘里，}%
@{世潔喻佛身，俱成塵濁性。}@
%{世间清净的事物比喻佛陀的身体，都变成了尘垢浊染的本质。}%

@{如上諸所引，世中殊勝事，}@
%{以上所举的例子，都是世间殊胜的事物，}%
@{佛法逈超過，俗事可哀愍。}@
%{佛法远远超过世间的事物，世俗的事物令人怜悯。}%

@{聖法珍寶聚，佛最居其頂，}@
%{圣法是珍宝的聚合，佛陀居于最顶端，}%
@{無上無比中，唯佛與佛等。}@
%{在无上无比的事物中，只有佛与佛相等。}%

@{如來聖智海，隨樂歎少分，}@
%{如来的圣智如大海般广阔，我随心所欲地赞叹一小部分，}%
@{鄙詞讚勝德，對此實多慚。}@
%{我的语言粗陋，赞叹佛陀的胜德，对此我感到非常惭愧。}%

@{時俗覩降魔，一切咸歸伏，}@
%{当时世人看到佛陀降伏魔军，一切众生都归服，}%
@{觀彼同真性，我謂等輕毛。}@
%{看到那些魔军，我感觉他们就像一根羽毛一样轻微。}%

@{假令大戰陣，智勇能摧伏，}@
%{即使是大规模的战争，智勇双全的人能战胜，}%
@{聖德超世間，降彼非為喻。}@
%{佛陀的圣德超越世间，降伏魔军不能与之相比。}%

@{隣次降魔後，於夜後分中，}@
%{降伏魔军之后，在夜深人静的时候，}%
@{斷諸煩惱習，勝德皆圓滿。}@
%{佛陀断除了一切烦恼习气，他的胜德都圆满成就。}%

@{聖智除眾闇，超過千日光，}@
%{佛陀的圣智除去了众生的黑暗，超过了千个太阳的光芒，}%
@{摧伏諸邪宗，希有無能比。}@
%{摧毁了各种邪恶的宗派，没有谁能与之相比。}%

@{三善根圓滿，永滅貪恚癡，}@
%{三善根圆满，永远消灭了贪嗔痴，}%
@{種習悉已除，清淨無能喻。}@
%{一切种子习气都已断除，清净无染，没有能比拟的。}%

@{妙法尊恒讚，不正法恒非，}@
%{佛陀恒常赞叹妙法，不赞叹不正法，}%
@{於斯邪正處，心無有憎愛。}@
%{在邪正之间，他的心没有憎恨和喜爱。}%

@{於聖弟子眾，及外道師徒，}@
%{对圣弟子和外道师徒，}%
@{於彼違順中，佛心初無二。}@
%{面对他们的违逆顺从，佛陀的心始终如一。}%

@{於德情無著，德者亦非貪，}@
%{对功德没有执着，拥有功德也不贪求，}%
@{善哉極無垢，聖智恒圓潔。}@
%{善哉，佛陀的智慧无比清净，永远圆满无瑕。}%

@{諸根常湛寂，永離迷妄心，}@
%{六根常处于清净的寂静状态，永远远离迷妄之心，}%
@{於諸境界中，現量由親覩。}@
%{在一切境界中，都能直接亲眼观察。}%

@{念慧窮真際，非凡愚所測，}@
%{念力和智慧穷尽真实境界，不是凡夫愚人所能测度的，}%
@{善安立語言，證彼亡言處。}@
%{善于安立语言，证得语言消失的境界。}%

@{寂靜無礙光，皎潔逾輝映，}@
%{寂静无碍的光明，皎洁地超越一切光辉，}%
@{妙色世希有，孰不懷敬心？}@
%{如此奇妙的颜色，世间罕见，谁能不怀着敬畏之心呢？}%

@{若有暫初觀，或復恒瞻覩，}@
%{即使只是短暂地观看，或者长时间地瞻仰，}%
@{妙相曾無二，前後悉同歡。}@
%{奇妙的相貌始终如一，前后都一样令人欢喜。}%

@{最勝威德身，觀者心無厭，}@
%{最胜妙的威德之身，观看者心中没有厌倦，}%
@{縱經無量劫，欣仰似初觀。}@
%{即使经过无量劫，依然像初次观看一样欢喜仰慕。}%

@{所依之德體，能依之德心，}@
%{所依的功德之体，能依的功德之心，}%
@{性相二俱融，能所初無異。}@
%{性相二者融合在一起，能所本来就没有区别。}%

@{如斯善逝德，總集如來躬，}@
%{如此的善逝功德，都集聚在如来的身上，}%
@{離佛相好身，餘非所安處。}@
%{离开佛陀的相好之身，其他地方都不是安身之处。}%

@{我因先世福，幸遇調御師，}@
%{我因为前世积累的福报，幸运地遇到了调伏身心的大师(佛陀) ，}%
@{仰讚功德山，遠酬尊所記。}@
%{仰慕赞叹佛陀的功德如山一样高，远远地报答尊者(佛陀) 的教诲。}%

@{一切有情類，皆因煩惱持，}@
%{一切有情众生，都是因为烦恼而被束缚，}%
@{唯佛能善除，由悲久住世。}@
%{只有佛陀才能善于断除烦恼，由于大悲心，他长久住世。}%

@{誰當先敬禮，唯佛大悲尊，}@
%{谁应该首先恭敬礼拜呢？只有佛陀，大悲的尊者，}%
@{聖德超世間，悲願處生死。}@
%{他的圣德超越世间，他的大悲愿力，能够救度众生脱离生死轮回。}%

@{尊居寂靜樂，處濁為群生，}@
%{尊者(佛陀) 住于寂静的快乐中，却身处污浊之中，为了利益众生，}%
@{永劫久精勤，慈心為一切。}@
%{永远精进努力，慈悲之心对待一切众生。}%

@{從真還利俗，由悲所引生，}@
%{从真实境界回到世间利益众生，是因为大悲心所引导，}%
@{如呪出潛龍，興雲注甘雨。}@
%{就像咒语召唤潜藏的龙，兴起云彩，降下甘露一样。}%

@{恒居勝定位，等觀以怨親，}@
%{佛陀恒常住于最胜的境界，平等地看待怨恨和亲爱，}%
@{兇嶮倡聒人，投身歸聖德。}@
%{那些凶险、吵闹的人，都投身归依佛陀的圣德。}%

@{神通師子吼，宣言三界尊，}@
%{佛陀以神通力，像狮子一样吼叫，宣告他是三界之尊，}%
@{久已厭名聞，由悲自稱讚。}@
%{他早已厌倦了名闻利养，只是出于大悲心，才赞叹自己。}%

@{常修利他行，曾無自利心，}@
%{佛陀恒常修习利他的行为，从未有过自利之心，}%
@{慈念遍眾生，於己偏無愛。}@
%{慈悲之心遍及一切众生，对自己却毫无爱执。}%

@{悲願無邊際，逐器化群生，}@
%{佛陀的大悲愿力无边无际，根据众生的根器来度化他们，}%
@{隨處皆饒益，猶如散祭食。}@
%{无论在哪里，都能够利益众生，就像散布祭品的食物一样。}%

@{深心念一切，恒不捨須臾，}@
%{佛陀以深心念及一切众生，始终不舍弃片刻，}%
@{利彼反遭辱，由咎非佛作。}@
%{利益他人反而受到辱骂，这是因为众生的业力所致，不是佛陀造成的。}%

@{慈音演妙義，誠諦非虛說，}@
%{佛陀以慈悲的声音演说妙法，真诚真实，不是虚假之言，}%
@{廣略任機緣，半滿隨時轉。}@
%{广阔深奥，根据众生的根器，随时改变说法。}%

@{若聞尊演說，孰不歎希奇，}@
%{如果听到佛陀的演说，谁能不感到希奇赞叹呢？}%
@{縱令懷惡心，有智咸歸信。}@
%{即使怀着恶念，有智慧的人都会归信佛法。}%

@{義詞恒善巧，或復出麁言，}@
%{佛陀的语言恒常善巧，有时也会说出粗俗的话，}%
@{利益悉不虛，故並成真妙。}@
