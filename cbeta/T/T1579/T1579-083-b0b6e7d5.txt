@{瑜伽師地論卷第八十三}@
%{瑜伽师地论卷第八十三}%

@{彌勒菩薩說}@
%{弥勒菩萨说}%

@{三藏法師玄奘奉詔譯}@
%{三藏法师玄奘奉诏译}%

@{攝異門分之上}@
%{摄异门分之上}%

@{如是已說攝釋。}@
%{如是已经说了摄释。}%

@{云何攝異門？}@
%{什么是摄异门？}%

@{總嗢拕南曰：}@
%{总嗢拕南说：}%

@{白品與黑品、異門等宣說，}@
%{白品与黑品、异门等等宣说，}%

@{為開悟義覺，略總頌應知。}@
%{为了开悟义觉，略总颂应知。}%

@{別嗢拕南曰：}@
%{别嗢拕南说：}%

@{師、第一、二慧、四種善說等、}@
%{师、第一、二慧、四种善说等等、}%

@{亦有因緣等、施、戒、道廣說。}@
%{也有因缘等等、施、戒、道广说。}%

@{此中大師，所謂如來。}@
%{此中大师，所谓如来。}%

@{紹師即是第一弟子，如彼尊者舍利子等。}@
%{绍师即是第一弟子，如彼尊者舍利子等等。}%

@{言襲師者，謂軌範師、若親教師、若同法者、能開悟者、令憶念者。}@
%{言袭师者，谓轨范师、若亲教师、若同法者、能开悟者、令忆念者。}%

@{大師即是立聖教者。}@
%{大师即是立圣教者。}%

@{紹師即是傳聖教者。}@
%{绍师即是传圣教者。}%

@{襲師即是隨聖教者。}@
%{袭师即是随圣教者。}%

@{開許制止一切應作、不應作故，時時教授教誡轉故。當知即是能說、傳說及隨說者。}@
%{开许制止一切应作、不应作故，时时教授教诫转故。当知即是能说、传说及随说者。}%

@{驅擯造作不應作故，名能獎者。}@
%{驱擯造作不应作故，名能奖者。}%

@{慶慰造作應作事故，名勝獎者。}@
%{庆慰造作应作事故，名胜奖者。}%

@{於前二事能開示故，名至獎者。}@
%{于前二事能开示故，名至奖者。}%

@{隨所生起一切疑惑皆能遣故，名能導者。}@
%{随所生起一切疑惑皆能遣故，名能导者。}%

@{惡作憂悔皆能遣故，名勝導者。}@
%{恶作忧悔皆能遣故，名胜导者。}%

@{一切煩惱及隨煩惱皆能遣故，名至導者。}@
%{一切烦恼及随烦恼皆能遣故，名至导者。}%

@{於諸疑惑能斷除者，謂未顯義能顯發故，已顯發義令明淨故。}@
%{于诸疑惑能断除者，谓未显义能显发故，已显发义令明净故。}%

@{甚深義句以慧通達廣開示故。誓許為作軌範尊重所依止故，名第二伴。}@
%{甚深义句以慧通达广开示故。誓许为作轨范尊重所依止故，名第二伴。}%

@{隨轉伴故，名為善友。}@
%{随转伴故，名为善友。}%

@{宿昔同處居家樂故，名為知識。}@
%{宿昔同处居家乐故，名为知识。}%

@{父母宗親互相繫屬，名憐愍者。}@
%{父母宗亲互相系属，名怜愍者。}%

@{若非眷屬而施恩惠，名有恩者。}@
%{若非眷属而施恩惠，名有恩者。}%

@{言義利者，名所求事能引義利。樂為此故，名樂義利。}@
%{言义利者，名所求事能引义利。乐为此故，名乐义利。}%

@{言利益者，名為善行。樂為此故，名樂利益。}@
%{言利益者，名为善行。乐为此故，名乐利益。}%

@{言安樂者，名安隱住益身心義。樂為此故，名樂安樂。}@
%{言安乐者，名安隐住益身心义。乐为此故，名乐安乐。}%

@{依現法樂，名樂安隱。依後法樂，說名為樂相應安隱。}@
%{依现法乐，名乐安隐。依后法乐，说名为乐相应安隐。}%

@{於一切事現正隨從，故名信順。}@
%{于一切事现正随从，故名信顺。}%

@{若即於彼補特伽羅處所而起，故名為信。}@
%{若即于彼补特伽罗处所而起，故名为信。}%

@{開彼功德及與威力、殊勝慧已，即於彼法處所而起，隨順理門，故名淨信。}@
%{开彼功德及与威力、殊胜慧已，即于彼法处所而起，随顺理门，故名净信。}%

@{即由如是增上力故，身毛為竪，悲泣墮淚，如是等事是淨信相。}@
%{即由如是增上力故，身毛为竪，悲泣堕泪，如是等事是净信相。}%

@{聞彼功德、威力等已，於行住等諸威儀中，恒常信彼實有功德，故名信述。}@
%{闻彼功德、威力等等已，于行住等诸威仪中，恒常信彼实有功德，故名信述。}%

@{所言欲者，若於是處樂作、樂得。}@
%{所言欲者，若于是处乐作、乐得。}%

@{言精進者，發起加行，其心勇悍。}@
%{言精进者，发起加行，其心勇悍。}%

@{言策勵者，既勇悍已，於彼加行正勤修習。}@
%{言策励者，既勇悍已，于彼加行正勤修习。}%

@{言剛決者，發精進已，終無懈廢，不壞不退。}@
%{言刚决者，发精进已，终无懈废，不坏不退。}%

@{言超越者，殷重精進。}@
%{言超越者，殷重精进。}%

@{言威勢者，謂過夜分，或前一更，被服鎧甲，當發精進。}@
%{言威势者，谓过夜分，或前一更，被服铠甲，当发精进。}%

@{言奮發者，如所被服發勤精進，或更昇進，威猛勇悍發勤精進。}@
%{言奋发者，如所被服发勤精进，或更昇进，威猛勇悍发勤精进。}%

@{深見彼果所有勝利，故名勇銳。}@
%{深见彼果所有胜利，故名勇锐。}%

@{於勤修時，堪能忍受寒等淋瀝，故名勇悍。}@
%{于勤修时，堪能忍受寒等淋沥，故名勇悍。}%

@{由善了知前後差別，於其勝上差別證中，深生信順所有精進，名難制伏。}@
%{由善了知前后差别，于其胜上差别证中，深生信顺所有精进，名难制伏。}%

@{於少下劣差別所證進修善中，無怯劣故，名無喜足。}@
%{于少下劣差别所证进修善中，无怯劣故，名无喜足。}%

@{言勵心者，謂於精進所有障處，一切煩惱及隨煩惱諸魔事中，頻頻覺察，令心靜息。}@
%{言励心者，谓于精进所有障处，一切烦恼及随烦恼诸魔事中，频频觉察，令心静息。}%

@{言常恒者，謂即於此正加行中，能常修作，能不捨軛。}@
%{言常恒者，谓即于此正加行中，能常修作，能不舍轭。}%

@{言正信者，謂於大師說正法時，於此正法既聽聞已，獲得淨信。}@
%{言正信者，谓于大师说正法时，于此正法既听闻已，获得净信。}%

@{不放逸者，謂得信已，於樂出離障礙法中，防護其心，恒常發起善法修習。}@
%{不放逸者，谓得信已，于乐出离障碍法中，防护其心，恒常发起善法修习。}%
@{言瑜伽者，受持、讀誦、問論、決擇、正修加行。}@
%{说瑜伽的人，要坚持修行，读诵经文，探讨问题，做出决定，并进行正面的修行。}%

@{言思惟者，隨所受持究竟法義，審諦觀察。}@
%{说思惟的人，要根据自己所持守的究竟佛法，仔细观察。}%

@{言憶念者，於所觀察一切法義，能不忘失；於久所作、久所說中，能正隨念。}@
%{说忆念的人，要记住自己所观察到的所有佛法，不能忘记；对于自己过去所做、过去所说的事情，要能正念地回忆起来。}%

@{言尋思者，即依如是無倒法義，起出離等所有尋思。}@
%{说寻思的人，要根据这些无误的佛法，生起出离等各种思惟。}%

@{所言智者，謂出世間加行妙慧。}@
%{说智者，是指出世间修行的妙智慧。}%

@{所言解者，謂出世間正體妙慧。}@
%{说解者，是指出世间正法本体的妙智慧。}%

@{所言慧者，謂已證得出世間慧，後時所得世間妙慧。}@
%{说慧者，是指已经证悟了出世间智慧，后来所获得的世间妙智慧。}%

@{言觀察者，謂由無倒觀察作意，審諦觀察已斷、未斷、有餘、無餘。}@
%{说观察者，是指通过无误的观察，审慎地观察已经断除的烦恼、未断除的烦恼、有余依的烦恼、无余依的烦恼。}%

@{言梵行者，謂八聖支道，及與遠離非正梵行習婬欲法。}@
%{说梵行者，是指八正道，以及远离不正当的梵行，不染淫欲。}%

@{又言安住餘梵行者，謂三十七菩提分法，彼由三處之所攝受。謂由奢摩他故，由毘鉢舍那故，由修身念故。}@
%{又说安住于其他梵行的人，是指三十七道品，它们由三方面摄受，即由奢摩他(静虑) 摄受，由毗婆舍那(观照) 摄受，由修身念摄受。}%

@{如其所應，彼自性故、彼品類故。}@
%{这三种摄受，是根据它们自身本质和类别而来的。}%

@{此中信念，俱通二品。}@
%{这里所说的信念，适用于两种品类。}%

@{復次，即此大師亦稱第一，自義行故。亦稱為尊，他義行故。亦稱為勝，俱義行故。亦稱為上，映蔽一切諸外道故。亦稱無上，映蔽一切聲聞、獨覺下中乘故。}@
%{其次，这位大师也被称为第一，是因为他自身的修行；也被称为尊，是因为他他人的修行；也被称为胜，是因为他兼具自身和他人的修行；也被称为上，因为他超越了一切外道；也被称为无上，因为他超越了一切声闻、独觉、下中乘。}%

@{復有差別。言第一者，共諸世間善圓滿故。}@
%{这里还有区别。说第一，是因为他圆满了一切世间善法。}%

@{所言尊者，共諸聲聞善圓滿故。}@
%{说尊者，是因为他圆满了一切声闻善法。}%

@{所言勝者，共諸獨覺善圓滿故。}@
%{说胜者，是因为他圆满了一切独觉善法。}%

@{所言上者，於煩惱障得清淨故。}@
%{说上者，是因为他清除了烦恼障。}%

@{言無上者，於所知障得清淨故。}@
%{说无上者，是因为他清除了所知障。}%

@{復有差別。言第一者，於欲行善得圓滿故。}@
%{这里还有区别。说第一，是因为他圆满了一切欲界善法。}%

@{所言尊者，於色行善得圓滿故。所言勝者，無色行善得圓滿故。}@
%{说尊者，是因为他圆满了一切色界善法；说胜者，是因为他圆满了一切无色界善法。}%

@{所言上者，超過一切三界世間善圓滿故。}@
%{说上者，是因为他超越了一切三界世间善法。}%

@{言無上者，出世間善得圓滿故。}@
%{说无上者，是因为他圆满了一切出世间善法。}%

@{無足有情者，如蛇等。}@
%{无足的有情众生，比如蛇等等。}%

@{二足有情者，謂人等。}@
%{二足的有情众生，比如人等等。}%

@{四足有情者，如牛等。}@
%{四足的有情众生，比如牛等等。}%

@{多足有情者，如百足等。}@
%{多足的有情众生，比如百足等等。}%

@{有色有情者，謂從欲界乃至第四靜慮。}@
%{有色身的有情众生，是指从欲界到第四禅天。}%

@{無色有情者，謂從空無邊處乃至非想非非想處。}@
%{无色身的有情众生，是指从空无边处到非想非非想处。}%

@{有想有情者，謂從欲界乃至無所有處，除無想天。}@
%{有想的有情众生，是指从欲界到无所有处，除了无想天。}%

@{無想有情者，謂無想天。}@
%{无想的有情众生，是指无想天。}%

@{非有想非無想有情者，謂非想非非想處所有生天。}@
%{非有想非无想的有情众生，是指非想非非想处所生的天人。}%

@{如是略說品類差別，顯示如來三種第一，謂由蠢動故、由依止故、由心故。}@
%{这样简略地说明了品类的差别，显示了如来的三种第一，即由蠢动而得第一，由依止而得第一，由心而得第一。}%

@{復次，能得慧者，謂總攝一切能引義利所有善慧生長增益。}@
%{其次，说能得慧者，是指能摄受一切能引生义利的所有善慧，使之增长增益。}%

@{廣大慧者，謂軟中上品增進差別。}@
%{说广大慧者，是指从下品到上品，逐渐增进的差别。}%

@{清淨慧者，謂宿世串習，經歷多時其慧成熟。}@
%{说清净慧者，是指经过宿世习气，经历很长时间，其智慧成熟了。}%

@{成辦慧者，謂於諸煩惱遍知永斷。}@
%{说成办慧者，是指对一切烦恼彻底断除，永远不再生起。}%

@{圓滿慧者，謂即此善慧已到究竟。}@
%{说圆满慧者，是指这种善慧已经达到了究竟。}%

@{無退慧者，謂即此善慧成無退法，究竟出離。}@
%{说无退慧者，是指这种善慧已经成为无退法，最终出离轮回。}%

@{言捷慧者，速疾了知故。}@
%{说捷慧者，是指能够迅速地了解。}%

@{言速慧者，慧無滯礙故。}@
%{说速慧者，是指智慧没有障碍。}%

@{言利慧者，盡其所有、如其所有皆善了知故。}@
%{说利慧者，是指能够完全了解所有事物，并且能够很好地理解它们。}%

@{言出慧者，於出離法世間離欲能善了知故。}@
%{说出慧者，是指能够很好地了解出离法，能够在世间离欲。}%

@{決擇慧者，於出世間諸離欲法能了知故。}@
%{说决择慧者，是指能够了解出世间的所有离欲法。}%

@{甚深慧者，於甚深空相應、緣起，隨順諸法能了知故，又於一切甚深義句皆能如實善通達故。此中如來慧能制立，聲聞等慧於所制立能隨覺了。又大慧者，謂即此慧長時串習故。}@
%{说甚深慧者，是指能够了解甚深的空性、缘起，能够随顺一切法，并且能够如实地通达一切甚深的义句。这里如来的智慧能够制定法，声闻等智慧能够随之觉悟。又说大慧者，是指这种智慧经过长时间的习练。}%

@{其廣慧者，謂即此慧無量無邊所行境故。}@
%{说广大慧者，是指这种智慧能够遍行无量无边的境域。}%

@{無等慧者，其餘諸慧無與等故。}@
%{说无等慧者，是指其他所有智慧都无法与之相比。}%

@{言慧寶者，於諸根中，慧最勝故。如末尼珠，顯發輪王毘琉璃寶，令光淨故。與彼相應，故名慧寶皆得成就。}@
%{说慧宝，是指在所有根性中，智慧最胜。就像末尼珠能够显发轮王的毗琉璃宝，使之光芒净洁一样，智慧与其他根性相对应，所以被称为慧宝，能够成就一切。}%
@{又慧眼者，謂俱生慧。}@
%{又说“慧眼”，指的是“俱生慧”。}%

@{言慧明者，謂他所引，則他所引善加行慧。}@
%{说“慧明”，指的是“别人引导”，那么别人引导的善加行慧。}%

@{言慧光者，謂即加行聞思成慧。}@
%{说“慧光”，指的是“直接通过加行闻思而成的慧”。}%

@{言慧曜者，謂即由此修所成慧。}@
%{说“慧曜”，指的是“由此修所成的慧”。}%

@{言慧燈者，謂於如來所說經典，甚深建立等開示故。}@
%{说“慧灯”，指的是“如来所说经典，甚深建立等开示”。}%

@{言慧炬者，謂於法教，隨量隨時能隨轉故。}@
%{说“慧炬”，指的是“对法教，随量随时都能随转”。}%

@{言慧照者，謂於彼彼所有諸法，以其妙慧能善了知；雖善了知，猶隨他轉，而未身證。}@
%{说“慧照”，指的是“对所有诸法，用奇妙的慧能善了知；虽然善了知，但还是随别人转，还没有亲证”。}%

@{慧無闇者，謂身作證。}@
%{“慧无暗”指的是“亲证”。}%

@{其慧根者，謂於他所證，能遍了知增上力故，諸所有慧。}@
%{说“慧根”，指的是“对别人所证，能遍知增上力，所以所有慧”。}%

@{言慧力者，謂於自先後差別所證，能遍了知增上力故，由法道理無退屈慧。}@
%{说“慧力”，指的是“对自身前后差别所证，能遍知增上力，所以由法道理无退屈的慧”。}%

@{言慧財者，謂能招引一切自在最勝富貴，隨獲自心自在轉故。又此慧寶，於一切財最為殊勝，能為一切世間珍財根本因故。}@
%{说“慧财”，指的是“能招引一切自在最胜富贵，随获自心自在转。又此慧宝，在一切财中最殊胜，能为一切世间珍财根本因”。}%

@{如說慧劍及慧刀者，謂能永斷一切結故。言慧杖者，謂能遠防一切煩惱、天惡魔故。}@
%{比如“慧剑”和“慧刀”，指的是“能永远断除一切结”。说“慧杖”，指的是“能远离一切烦恼、天恶魔”。}%

@{言慧轡者，縱意根馬於善行地而馳驟故。}@
%{说“慧辔”，指的是“放任意根马在善行地驰骤”。}%

@{慧無墮者，令諸身分不散壞故。}@
%{“慧无堕”指的是“令诸身分不散坏”。}%

@{慧垣牆者，遍於一切一門轉故。}@
%{“慧垣墙”指的是“遍于一切一门转”。}%

@{慧階陛者，加行道故。}@
%{“慧阶陛”指的是“加行道”。}%

@{慧堂殿者，到究竟故。}@
%{“慧堂殿”指的是“到究竟”。}%

@{為欲顯示垣牆等三，復說三種。所謂界智、種種界智、非一界智。}@
%{为了显示“垣墙”等三，又说三种，即“界智”，“种种界智”，“非一界智”。}%

@{又正見者，能善通達真實法故。}@
%{又“正见”，指的是“能善通达真实法”。}%

@{有學慧者，如理作意，復能引發心善解脫、慧善解脫。又於後時諸有學慧，謂預流果及一來果、不還果攝。}@
%{“有学慧”，指的是“如理作意，又能引发心善解脱、慧善解脱。又在后时诸有学慧，指的是预流果、一来果、不还果”。}%

@{諸無學慧，謂阿羅漢菩提所攝，若諸獨覺菩提所攝，若諸如來最勝無上菩提所攝。}@
%{“诸无学慧”，指的是“阿罗汉菩提所摄，或者独觉菩提所摄，或者如来最胜无上菩提所摄”。}%

@{云何界智？謂能了知種種界故。若能了知十八界者，名非一界智。了知彼界種種品類，名種種界智。通達了知彼界、趣、地、補特伽羅品類差別故。}@
%{什么是“界智”？指的是“能了知种种界”。如果能了知十八界，就叫做“非一界智”。了知彼界种种品类，就叫做“种种界智”。通达了知彼界、趣、地、补特伽罗品类差别。}%

@{又微細者，能入真實甚深義故。}@
%{又“微细”，指的是“能入真实甚深义”。}%

@{言審悉者，具能證入一切義故。}@
%{说“审悉”，指的是“具能证入一切义”。}%

@{言聰明者，謂與引發慧相應故。}@
%{说“聪明”，指的是“与引发慧相应”。}%

@{言叡哲者，謂與俱生慧相應故。或復翻此。}@
%{说“叡哲”，指的是“与俱生慧相应”。或者反过来。}%

@{眼者，能取現見事故。}@
%{“眼”，指的是“能取现见的事”。}%

@{智者，能取不現事故。}@
%{“智”，指的是“能取不现的事”。}%

@{明者，悟入盡所有事。}@
%{“明”，指的是“悟入尽所有事”。}%

@{覺者，悟入如所有事。}@
%{“觉”，指的是“悟入如所有事”。}%

@{言義行者，謂思所成善法攝故。}@
%{说“义行”，指的是“思所成善法所摄”。}%

@{言法行者，謂聞所成善法攝故。}@
%{说“法行”，指的是“闻所成善法所摄”。}%

@{言善行者，施、戒所成善法攝故。}@
%{说“善行”，指的是“施、戒所成善法所摄”。}%

@{調柔行者，謂修所成善法攝故。}@
%{说“调柔行”，指的是“修所成善法所摄”。}%

@{復次，言善說者，謂諸文句善圓滿故。}@
%{其次，说“善说”，指的是“诸文句善圆满”。}%

@{言善覺者，謂能善現等覺義故。}@
%{说“善觉”，指的是“能善现等觉义”。}%

@{言出離者，謂世間道，斷除眾苦得出離故。}@
%{说“出离”，指的是“世间道，断除众苦得出离”。}%

@{趣等覺者，謂出世道，為超眾苦而能真實現等覺故。}@
%{说“趣等觉”，指的是“出世道，为了超越众苦而能真实现等觉”。}%

@{無差別者，師與弟子所說文義相滋潤故，不相違故。}@
%{“无差别”，指的是“师与弟子所说文义相滋润，不相违”。}%

@{有窣堵波者，一切外道、天魔及餘世間不能傾動故。}@
%{“有窣堵波”，指的是“一切外道、天魔及余世间不能倾动”。}%

@{言有依者，具足四依，無失壞故。}@
%{说“有依”，指的是“具足四依，无失坏”。}%

@{大師如來、應、正等覺者，謂所說教善清淨故。}@
%{“大师如来、应、正等觉”，指的是“所说教善清净”。}%

@{此中諸句，略顯四種善說法律最極圓滿。謂初二句顯文義圓滿，次二句顯果圓滿，次三句顯行圓滿；後一句顯師圓滿。}@
%{此中诸句，略显四种善说法律最极圆满。即初二句显文义圆满，次二句显果圆满，次三句显行圆满；后一句显师圆满。}%

@{復次，佛世尊法，有因緣者，謂有緣起，制立一切所學處故。}@
%{其次，佛世尊法，有因缘，指的是“有缘起，制定一切所学处”。}%

@{有出離者，謂有犯已，制立如法還出離故。}@
%{“有出离”，指的是“有犯已，制定如法还出离”。}%

@{言有依者，謂由四依，制立超越一切惡戒諸毀犯故。}@
%{说“有依”，指的是“由四依，制定超越一切恶戒诸毁犯”。}%

@{有超越者，制立遠離受用欲樂、自苦行邊，隨順士用令成就故。}@
%{“有超越”，指的是“制定远离受用欲乐、自苦行边，随顺士用令成就”。}%
@{有神變者，謂由三種所現神變，為令獲得速疾神通，無間制立正教授故。}@
%{有神通的人，据说都是通过三种方式展现神通的，目的是为了快速获得神通，并且能够无间断地建立正法教授。}%

@{復次，解脫捨者，迴向涅槃故，於施果中無繫著故。}@
%{其次，解脱舍施的人，是将功德回向涅槃，所以不会执着于施舍的结果。}%

@{常舒手者，殷重廣施故。}@
%{经常伸出双手的人，是因为他们认真地广施布施。}%

@{樂棄捨者，施前、正施及與施後意悅清淨，無追悔故。}@
%{乐于舍施的人，他们在施舍前、施舍中和施舍后，内心都清净喜悦，没有后悔。}%

@{祠祀施者，一向如法，不以凶暴積集財物，時時數數周遍捨施所施物故。}@
%{祭祀施舍的人，一直都按照正法进行，不会用凶暴的手段积累财富，他们经常周遍地施舍所施舍的物品。}%

@{捨圓滿者，謂於福田而奉獻故。}@
%{舍施圆满的人，是指将供养奉献给福田。}%

@{於惠施中樂分布者，謂於父母、妻子等所，時時平等而分布故。}@
%{乐于在施舍中分配的人，是指对父母、妻子等，经常平等地分配。}%

@{如是一切總有六施。一、無所依施，二、廣大施，三、歡喜施，四、數數施，五、田器施，六、攝受眷屬施。}@
%{像这样，所有施舍加起来共有六种：一、无依施，二、广大施，三、欢喜施，四、数数施，五、田器施，六、摄受眷属施。}%

@{此中依止品類、時、處布施而說。}@
%{这六种施舍是根据品类、时间、地点而说的。}%

@{復次，廣說戒者，中嗢拕南曰：}@
%{其次，广说戒律，中嗢拕南说：}%

@{尸羅、法、殺生、具戒等廣說。}@
%{尸罗、法、杀生、具戒等等，这些都需要详细解释。}%

@{言尸羅者，謂能寂靜毀犯淨戒罪熱惱故，又與清涼義相應故。}@
%{说尸罗，是指能够寂静地消除犯戒罪的热恼，并且与清凉的意义相对应。}%

@{言律儀者，謂是遠離自體相故。}@
%{说律仪，是指远离自身相。}%

@{言具足者，謂正攝受無悔等故。}@
%{说具足，是指正当地摄受无悔等。}%

@{言清淨者，攝受現行三摩地故。}@
%{说清净，是指摄受现行的三摩地。}%

@{又言善者，謂能攝受可愛果故。}@
%{又说善，是指能够摄受可爱的果报。}%

@{言無罪者，謂能攝受自他利故。}@
%{说无罪，是指能够摄受自身和他人的利益。}%

@{言無害者，謂能違拒執持刀杖鬪諍等事。}@
%{说无害，是指能够拒绝执持刀杖争斗等事。}%

@{言隨順者，隨順證得諸沙門果，及餘所有勝功德故。}@
%{说随顺，是指随顺证得诸沙门果，以及其他一切胜功德。}%

@{言隱覆者，謂常隱覆自善法故。}@
%{说隐覆，是指经常隐覆自己的善法。}%

@{言顯發者，謂常發露自惡法故。言端嚴者，謂具攝受諸少欲等所有沙門莊嚴具故。}@
%{说显发，是指经常发露自己的恶法。说端严，是指具备摄受诸少欲等一切沙门庄严具。}%

@{言福田者，攝受正見、軌範、淨命圓滿德故。}@
%{说福田，是指摄受正见、轨范、净命圆满的德行。}%

@{言無熱者，謂正遠離自苦邊故。}@
%{说无热，是指正当地远离自身痛苦的边际。}%

@{言無惱者，遠離受用欲樂邊故。}@
%{说无恼，是指远离享受欲乐的边际。}%

@{言無悔者，謂正遠離染污、不樂、憂慼事故。}@
%{说无悔，是指正当地远离染污、不乐、忧愁痛苦的事。}%

@{復次，善說法者，道理所攝故，任持勝德故。毘奈耶者，隨順一切煩惱滅故。所言聖者，遠離一切雜染污法，令不生故。又言善者，能與無罪可愛果故。言應習者，應習近故。言善哉者，是諸聖賢稱讚事故。}@
%{其次，善于说法的人，是因为他们被道理所摄受，并且能够任持胜德。毘奈耶，是指随顺一切烦恼灭尽。所说的圣者，是指远离一切杂染污法，使之不生。又说善，是指能够给予无罪可爱的果报。说应习，是指应该经常学习。说善哉，是指这是诸圣贤所赞叹的事。}%

@{復次，言殺生者，謂如有一，乃至廣說黑品、白品。當知廣如有尋有伺地中已說。}@
%{其次，说杀生，是指如果有一条生命，乃至广说黑品、白品。应该知道，广说如在有寻有伺地中已经说了。}%

@{復次，言具戒等皆廣說者，謂安住具戒，亦能守護別解脫律儀，乃至廣說。密護根門，若守護念，若常委念，乃至廣說。於食知量，於諸飲食思擇而食，不為充悅，不為憍逸，乃至廣說。進止往來正知而住，乃至廣說。如是一切，廣說應知如聲聞地。}@
%{其次，说具戒等等都应该广说，是指安住具戒，也能够守护别解脱律仪，乃至广说。密护根门，无论是守护念，还是经常委念，乃至广说。在饮食方面知道量，在各种饮食方面思择而食，不为了充悦，不为了憍逸，乃至广说。进止往来正知而住，乃至广说。像这样一切，广说应该知道如声闻地。}%

@{復次，廣說道者，中嗢拕南曰：}@
%{其次，广说道，中嗢拕南说：}%

@{念住、正斷、神足、根力、覺支、道支，}@
%{念住、正断、神足、根力、觉支、道支，}%

@{無量為後。}@
%{无量无边地往下延伸。}%

@{為欲勤修四念住故，發起上品猛利欲者，謂為斷除不正作意諸過失故。}@
%{为了勤修四念住，发起了上品猛利的欲望，是指为了断除不正作意的过失。}%

@{言精進者，謂為斷除縵緩策勤諸過失故。}@
%{说精进，是指为了断除懈怠、策勤的过失。}%

@{言策勤者，謂為斷除惛沈、掉舉二隨煩惱諸過失故。}@
%{说策勤，是指为了断除昏沉、掉举两种随烦恼的过失。}%

@{言勇悍者，不自輕蔑故。}@
%{说勇悍，是指不轻视自己。}%

@{言勇銳者，能抗外敵故。}@
%{说勇锐，是指能够抵抗外敌。}%

@{不可制伏者，於少下劣不生喜足故。}@
%{说不可制伏，是指对微不足道的事物不会感到满足。}%

@{言正念者，不忘教授故。}@
%{说正念，是指不忘教授。}%

@{言正知者，能不毀犯所毀犯故。}@
%{说正知，是指能够不违犯所应该违犯的。}%

@{不放逸者，不捨善軛故。}@
%{说不放逸，是指不舍弃善法。}%

@{住熱光者，能修懈怠對治法故。}@
%{说住热光，是指能够修习懈怠的对治法。}%

@{言正解者，能修毀犯對治法故。}@
%{说正解，是指能够修习违犯的对治法。}%

@{念成辦者，能修忘念對治法故。}@
%{说念成办，是指能够修习忘念的对治法。}%

@{調伏世間者，能修貪、憂一切世法正對治故。}@
%{说调伏世间，是指能够修习贪、忧一切世法的正对治法。}%

@{此中顯示，勤修念住諸苾芻等，應當修習四種對治。}@
%{这里显示，勤修念住的诸苾刍等等，应该修习四种对治法。}%

@{復次，於諸正斷、諸神足中所有異名，廣說應知如聲聞地。}@
%{其次，在诸正断、诸神足中，所有不同的名称，广说应该知道如声闻地。}%
@{復次，於如來所安立正信等，廣說應知如攝決擇分。中安住有勢力、有精進、有勇悍等，廣說應知如菩薩地。}@
%{再说，关于如来所安立的正信等等，广说应该知道就像摄决择分。其中安住有势力、有精进、有勇悍等等，广说应该知道就像菩萨地。}%

@{復次，簡擇諸法、最極簡擇、周遍尋思、周遍觀察，廣說應知如聲聞地。已得無漏真作意故，緣聖諦境，一切無漏作意相應，名為擇法。}@
%{再说，简择诸法、最极简择、周遍寻思、周遍观察，广说应该知道就像声闻地。已经得到无漏真作意，缘圣谛境，一切无漏作意相应，名为择法。}%

@{言簡擇者，總取一切苦法種類為苦聖諦故。}@
%{说简择，就是总取一切苦法种类作为苦圣谛。}%

@{最極簡擇者，各別分別取諸苦故，謂生苦、老苦等。}@
%{最极简择，就是分别取诸苦，比如生苦、老苦等等。}%

@{極簡擇法者，依此處所簡擇契經等法故。所以者何？依止此故，先修所作。}@
%{极简择法，就是依此地方简择契经等等法。为什么呢？依止此，先修所作。}%

@{又簡擇者，謂審定解了。}@
%{又简择，就是审定解了。}%

@{最極簡擇者，謂審定等解了。}@
%{最极简择，就是审定等解了。}%

@{極簡擇法者，謂審定近解了。前是尋求道，今是決定道。}@
%{极简择法，就是审定近解了。以前是寻找道，现在是决定道。}%

@{復有差別。言解了者，於所知事作意發悟。}@
%{还有差别。说解了，就是对所知事作意发悟。}%

@{等解了者，既發悟已，方便尋求。}@
%{等解了，就是已经发悟，方便寻求。}%

@{近解了者，求已決定。}@
%{近解了，就是求已决定。}%

@{復次，黠了者，了知分別體故。通達者，通達所知事故。復有差別。黠了者，了知自相故。通達者，了知共相故。}@
%{再说，黠了，就是了知分别体。通达，就是通达所知事。还有差别。黠了，就是了知自相。通达，就是了知共相。}%

@{審察者，謂能定取盡其所有、如其所有，先後漸次倍增廣故。}@
%{审察，就是能够定取尽其所有、如其所有，先后渐次倍增广。}%

@{聰叡者，先後漸次於彼義中無忘失故。}@
%{聪叡，就是先后渐次于彼义中无忘失。}%

@{覺者，謂堪能簡擇俱生之慧。明者，謂習所得慧。}@
%{觉，就是能够简择俱生之慧。明，就是习所得慧。}%

@{慧行者，謂能受持、讀誦、問論、勝決擇等增上了別，即於彼義轉增明了勤修習慧。}@

@{寂止者，由具忍辱柔和事故。}@

@{不喜樂者，謂於遠離增上意樂極滿足故。}@
%{不喜乐，就是于远离增上意乐极满足。}%

@{不違越者，謂於一切所學眾中無毀犯故，不棄捨故。}@
%{不违越，就是于一切所学众中无毁犯，不弃舍。}%

@{如是廣說，應隨九種心住差別，如聲聞地當知其相。}@

@{智、宣說、善、欲、熾然、獨、遠塵、}@

@{如病等、解釋、我、斷、盡生等、}@
%{如病等、解释、我、断、尽生等、}%

@{并天世眾生、依等、我作等。}@
%{并天世众生、依等、我作等。}%

@{智者，謂聞言說為先慧。}@
%{智，就是闻言说为先慧。}%

@{見者，謂見言說為先慧。}@
%{见，就是见言说为先慧。}%

@{覺者，謂覺言說為先慧。}@
%{觉，就是觉言说为先慧。}%

@{知者，謂知言說為先慧。}@
%{知，就是知言说为先慧。}%

@{智者，謂知不現見境。}@
%{智，就是知不现见境。}%

@{見者，謂見現見現在前境。}@
%{见，就是见现见现在前境。}%
@{明者，謂無明相違解。}@
%{明，就是指能够理解无明，也就是指能够理解无明和它相对立的智慧。}%

@{覺者，謂實有義智。}@
%{觉，就是指能够真正理解事物的本质，也就是指拥有真实的智慧。}%

@{覺者，謂不增益非實有智。}@
%{觉，就是指能够理解事物不会增加也不会减少，不会真的存在，也并不真的不存在，也就是指拥有不增不减的智慧。}%

@{慧者，謂俱生生得慧。}@
%{慧，就是指与生俱来的智慧，也就是指从出生就拥有的智慧。}%

@{明者，謂由加行習所成慧。}@
%{明，就是指通过不断修行而获得的智慧，也就是指通过积累经验而获得的智慧。}%

@{現觀者，謂於內現觀法已，於諸法中非不現見、非緣他智。}@
%{现观，就是指能够在内心观照法，也就是指能够在所有法中，既不是不显现，也不是依靠他人的智慧。}%

@{復次，宣說者，謂因他請問而為記別。}@
%{此外，宣说，就是指因为他人提问而进行解释说明。}%

@{施設者，謂由語及欲，次第編列名句文身。}@
%{设施，就是指通过语言和愿望，将名词、句子和文字按照顺序排列起来。}%

@{安立者，謂次第編列已，略為他說。}@
%{安立，就是指按照顺序排列完之后，简单地向他人讲述。}%

@{分別者，謂略說已，分別開示，解其義趣。}@
%{分别，就是指简单地讲述完之后，再详细地解释说明，并阐明其含义和意义。}%

@{開示者，謂他展轉所生疑惑皆能除遣。}@
%{开示，就是指能够消除他人不断产生的疑惑。}%

@{顯發者，謂自通達甚深義句，為他顯示。}@
%{显发，就是指自己通达深奥的意义和句子，并向他人展示。}%

@{教者，謂不因他發起請問，由哀愍故，說法開示。}@
%{教，就是指并非因为他人提问而开始讲解，而是出于悲悯之心，主动讲解和开示。}%

@{遍開示者，謂無間演說，不作師拳，無所隱覆。}@
%{遍开示，就是指毫无保留地讲解，不打任何师长的架子，没有任何隐瞒。}%

@{復次，初善者，謂聽聞時生歡喜故。}@
%{此外，初善，就是指听到的时候感到高兴。}%

@{中善者，謂修行時無有艱苦，遠離二邊，依中道行故。}@
%{中善，就是指修行的时候没有痛苦，远离两种极端，按照中道修行。}%

@{後善者，謂極究竟離諸垢故，及一切究竟離欲為後邊故。}@
%{后善，就是指最终彻底摆脱所有污垢，以及最终彻底摆脱一切欲望作为最终目标。}%

@{義妙者，謂能引發利益安樂故。}@
%{义妙，就是指能够带来利益和快乐。}%

@{文巧者，謂善緝綴名身等故，及語具圓滿故。}@
%{文巧，就是指能够很好地将名词等组合在一起，以及语言表达完整圆满。}%

@{純一者，謂不與一切外道共故。}@
%{纯一，就是指不与任何外道共同。}%

@{圓滿者，謂無限量故，最尊勝故。清淨者，謂自性解脫故。}@
%{圆满，就是指无量无边，最尊贵，最胜妙。清净，就是指自身本性解脱。}%

@{鮮白者，謂相續解脫故。}@
%{鲜白，就是指连续不断地解脱。}%

@{梵行者，謂八聖支道。當知此道，由純一等四種妙相之所顯說。}@
%{梵行，就是指八正道。应该知道这条道，是由纯一等四种妙相所显现和宣说的。}%

@{諦聽者，謂於如是相法勸令審聽。}@
%{谛听，就是指对于这样的相法，劝告人们认真倾听。}%

@{應善懇到者，謂勸令無倒、無間、殷重如理思惟。}@
%{应该善恳到，就是指劝告人们不要颠倒，不要间断，要认真地按照道理思考。}%

@{復次，猛利欲者，謂我何當於彼處所，乃至廣說。猛利愛者，謂於所修正加行中。}@
%{此外，猛利欲，就是指我什么时候能够到达那个地方，等等，广阔地说。猛利爱，就是指对于所修行的加行。}%

@{猛利樂者，謂於說者及與大師尊重處等。}@
%{猛利乐，就是指对于讲解者以及大师尊贵的地方等等。}%

@{猛利信者，謂於教法教授教誡。}@
%{猛利信，就是指对于教法、教授和教诫。}%

@{復次，能熾然者，謂為證得速疾通慧，終不自暇，推延後期，發勤精進。}@
%{此外，能炽然，就是指为了迅速证得通达智慧，最终不让自己懈怠，推迟时间，而是勤奋精进。}%

@{順瑜伽者，謂隨尊教若等、若勝而修加行，終不減劣。}@
%{顺瑜伽，就是指按照尊者的教诲，无论是平等还是胜过，都修行加行，最终不会减少或退步。}%

@{能永斷者，謂能修習煩惱對治。}@
%{能永断，就是指能够修行烦恼的对治。}%

@{能閑居者，謂依所有邊際臥具遠離而居，修三摩地令現在前，依三摩地修習對治。}@
%{能闲居，就是指依靠所有边际的床具远离而住，修行三摩地使它现在显现，依靠三摩地修行对治。}%

@{復次，獨者，謂處遠離邊際臥具，無有第二而安住故。}@
%{此外，独，就是指住在远离边际的床具处，没有第二个而安住。}%

@{言遠離者，謂諸染污、無記作意不現行故。}@
%{说远离，就是指所有染污和无记作意不再现行。}%

@{無縱逸者，謂於欲等尋思惡法防護心故，又於善中自安處故。}@
%{无纵逸，就是指为了保护自己的心，不思虑欲望等恶法，并在善法中安住。}%

@{言熾然者，謂如前說。}@
%{说炽然，就是指如同前面所说。}%

@{言發遣者，謂除五蓋，內持心故。又由此故，發遣其心，令趣無上安隱處故。}@
%{说发遣，就是指去除五盖，内心持守自己的心。又因此，发遣自己的心，使它走向无上的安稳之处。}%

@{復次，遠塵離垢者，塵謂已生未究竟智，能障現觀有間、無間我慢現轉；垢謂彼品及見斷品所有麁重。令永無故，名遠塵離垢。}@
%{此外，远尘离垢，尘是指已经产生但尚未究竟的智慧，能够阻碍现观有间、无间我慢的现转；垢是指那些品类以及见断品所具有的粗重。为了使之永远消失，所以称为远尘离垢。}%

@{又復塵者，所謂我慢，及見所斷一切煩惱；垢謂二品所有麁重。}@
%{又，尘是指所谓我慢以及见所断的一切烦恼；垢是指这两类品类所具有的粗重。}%

@{於諸法中者，謂於自相、共相所住法中。}@
%{于诸法中，是指在自身相、共同相所住的法中。}%

@{言法眼者，謂如實現證唯有法慧。}@
%{说法眼，是指如同真实地证悟，只有法慧。}%

@{言見法者，謂於苦等如實見故。}@
%{说见法，是指对于苦等如实地见到。}%

@{言得法者，謂隨證得沙門果故。}@
%{说得法，是指随着证得沙门果。}%

@{言知法者，謂證得已，於其所得，能自了知我是預流，我已證得無退墮法故。}@
%{说知法，是指证得之后，对于自己所获得的，能够自己明白我是预流，我已经证得无退堕法。}%

@{至誠法者，謂諦現觀增上力故，獲得證淨。於佛法僧及自所得聖所愛戒，以正信行如實至誠故。}@
%{至诚法，是指由于真实地观照增上力，而获得清净的证悟。对于佛法僧以及自己所获得的圣所爱的戒律，以正信行如实地至诚。}%

@{越渡惑者，謂於自所證。}@
%{越渡惑，是指对于自己所证悟的。}%

@{越渡疑者，謂於他所證。}@
%{越渡疑，是指对于他人所证悟的。}%

@{非緣於他者，謂於此法內自所證，非但隨他聽聞等故。}@
%{非缘于他，是指对于这个法，是自己内证的，并非仅仅是跟随他人听闻等等。}%

@{非餘所引者，謂於大師所有聖教，不為一切外道異論所引奪故。}@
%{非余所引，是指对于大师的所有圣教，不会被任何外道异论所夺取。}%
@{於諸法中得無所畏者，謂於自所證，若他詰問無悚懼故。}@
%{在所有法中，能够做到无所畏惧，指的是对于自己所证悟的真理，即使别人质疑，也不会感到恐惧。}%

@{言逆流者，謂已登聖道故。}@
%{说逆流者，指的是已经登上了圣道。}%

@{言趣向者，謂記神通究竟往趣，無退還故。}@
%{说趣向者，指的是记住了神通的究竟方向，不会退却。}%

@{復有差別。當知建立世俗、勝義二種法故。}@
%{还有区别，要明白世间法和胜义法这两种法。}%

@{復次，如說如病，乃至廣說。}@
%{其次，就像说像生病一样，乃至广说。}%

@{云何顯示彼如病等？非但說彼猶如重病，乃至廣說。}@
%{如何显示像生病一样？不仅仅是说它就像重病一样，乃至广说。}%

@{然修行者，先以如實無常等行，於彼事中如實訶毀，作是思惟：此如病等甚可厭逆。為欲與彼不和合故，是故次說無常行等，如實顯示觀察彼果。}@
%{但是修行者，首先以真实无常等行，在这些事情中真实地呵斥毁谤，这样思惟：像生病一样真是令人厌恶。为了想要与它不和合，所以接下来说无常行等，真实地显示观察它的果报。}%

@{言無常者，顯現生身及與剎那皆展轉故。剎那展轉者，由彼彼觸起盡故，彼彼受起盡。此相續見，由非不現見、非緣他智故。}@
%{说无常者，显示出生身以及刹那都在不断变化。刹那不断变化，是因为每个触碰都生起又灭尽，每个感受都生起又灭尽。这种连续不断的见解，不是不现见，也不是依靠他人的智慧。}%

@{所言苦者，有二種苦，謂生等諸苦，及諸所有受皆說為苦。}@
%{所说的苦，有两种苦，指的是出生等一切苦，以及所有感受都说是苦。}%

@{此二種苦如其所應，由見生身展轉有故，而得悟入。謂死無間有生身生。生已復有老等諸苦，是故說言無常故苦。由見生身展轉有故，悟入苦性。}@
%{这两种苦，正如其应有的那样，由于看到出生身不断变化，而得到悟入。指的是死亡之后无间地有出生身。出生之后又会有衰老等一切苦，所以说无常故苦。由于看到出生身不断变化，悟入苦的本质。}%

@{云何諸所有受皆說為苦？謂諸樂受，變壞故苦；一切苦受，生住故苦；非苦樂受，體是無常滅壞法故，說之為苦。}@
%{所有感受为什么都说成苦呢？指的是所有快乐感受，因为会变化而坏，所以是苦；所有痛苦感受，因为生住而不断，所以是苦；非苦非乐感受，本质是无常灭坏的法，所以说它是苦。}%

@{此中樂受，由無常故必有變壞；一切苦受，由無常故生住相續；皆起於苦。非苦樂受，已滅壞者，由無常故說之為苦。已生起者，滅壞法故亦說為苦。此滅壞法彼二所隨逐故，與二相應故，亦名為苦。}@
%{其中快乐感受，因为无常所以必然会变化而坏；所有痛苦感受，因为无常所以生住相续；都起源于苦。非苦非乐感受，已经灭坏的，因为无常所以说它是苦。已经生起的，因为灭坏法所以也说它是苦。这种灭坏法是这两个所跟随的，与这两个相应，所以也叫做苦。}%

@{云何當觀樂受為苦？謂由此受貪所隨眠，由隨眠故取當來苦，於現法中能生壞苦，如是當觀樂受為苦。云何當觀苦受如箭？謂如毒箭，乃至現前常惱壞故。非苦樂受，體是無常滅壞法者，謂已滅者即是無常，其未滅者是滅壞法。若無常者，從此復生若樂、若苦。滅壞法者，終不解脫苦樂二種。}@
%{如何应该观察快乐感受为苦呢？指的是因为这种感受贪欲所随眠，因为随眠所以取将来之苦，在现法中能生坏苦，这样应该观察快乐感受为苦。如何应该观察痛苦感受像箭一样呢？指的是就像毒箭一样，乃至现前总是恼害。非苦非乐感受，本质是无常灭坏法，指的是已经灭掉的，就是无常，没有灭掉的，是灭坏法。如果是无常，从此又会生起快乐或痛苦。灭坏法，最终不能解脱苦乐两种。}%

%{象迹，指的是不净等想，因为是最重要的，所缘广阔。}%
@{流注者，謂解脫因俱行無常想，能趣涅槃故。}@
%{流注，指的是摆脱因俱行无常想，能够趣向涅槃。}%
@{日出者，謂能對治無明闇故。}@
%{日出，指的是能够对治无明黑暗。}%
@{如輪王者，謂無學無常想。}@
%{如轮王，指的是无学无常想。}%
@{如城王者，謂所餘想。}@
%{如城王，指的是其他的想。}%
@{又或居阿練若、或居樹下、或居空室、或居逈露，由取樹下覆障等故，即攝一切臥具遠離。}@
%{或者住在阿兰若，或者住在树下，或者住在空室，或者住在旷野，因为要取树下的遮蔽等等，所以要远离一切卧具。}%

@{唯有色無常性者，謂唯有色，都無有我，如是正修加行。}@
%{唯有色无常性，指的是只有色，没有我，这样才是正修加行。}%

@{復次，略有四種往趣道障、二種道等。}@
%{其次，简而言之，有四种往趣道障，两种道等等。}%
@{謂由疑故，不能發趣。}@
%{因为怀疑，所以不能发起。}%
@{雖復發趣，由邪尋思，而往餘處。}@
%{虽然发起，但因为邪思，而走向其他地方。}%
@{由邪分尋思見行故，雖無是事，然不堪任教授教誡。}@
%{因为邪分别思见行，虽然没有这件事，但不能承受教授教诲。}%
@{所言忿者，謂他諫諍時。}@
%{所说的忿，指的是别人谏诤的时候。}%
@{言苦惱者，謂出家者不得自在、禁約艱難、麁弊行等。}@
%{所说的苦恼，指的是出家者不得自在，禁戒艰难，粗弊行等等。}%
@{言不樂者，雜瞋事故。}@
%{所说的不乐，是因为夹杂着瞋恨。}%
@{此之二種猶如坑㵎，又此二種能障行路。}@
%{这两种就像坑坎，又能够阻碍行路。}%
@{雖無是事，而由利養及恭敬故，於入山林能為障礙。}@
%{虽然没有这件事，但因为利养和恭敬，能够阻碍进入山林。}%
@{言猛利者，處深稠林故。}@
%{所说的猛利，指的是处在深密的森林。}%
@{所以者何？雖捨所攝受事，而不能捨此故。}@
%{为什么呢？因为虽然舍弃了所摄受的事，但不能舍弃这个。}%

@{復次，言有怖者，謂有盜賊及矯詐故。}@
%{其次，所说的有怖，指的是有盗贼和欺诈。}%
@{言有畏者，謂涉稠林故，有諸惡獸及與非人諸恐畏故。}@
%{所说的有畏，指的是进入密林，有恶兽和非人的恐惧。}%
@{言有刺者，謂一切處多毒刺故。}@
%{所说的有刺，指的是到处都有毒刺。}%
@{言失道者，往餘處故。}@
%{所说的失道，指的是走向其他地方。}%
@{言惡道者，不平正故。}@
%{所说的恶道，指的是不平正。}%
@{如是五種，顯道過失。}@
%{这五种，显现了道的过失。}%
@{弊趣、惡趣者，顯示趣過失。}@
%{弊趣、恶趣，显示了趣的过失。}%
@{失道、惡道而行，及親近不善士者，顯示能行補特伽羅所有過失。}@
%{失道、恶道而行，以及亲近不善之人，显示了能行补特伽罗的过失。}%
@{諸盜賊等名不善士。}@
%{盗贼等等，叫做不善之人。}%

@{復次，無動者，謂一切相皆遠離故。}@
%{其次，无动，指的是一切相都远离。}%
@{無轉者，謂貪愛盡故，於諸境界無轉變故。}@
%{无转，指的是贪爱尽了，在一切境界中没有变化。}%
@{難見者，謂甚深故。}@
%{难见，指的是非常深奥。}%
@{甘露者，謂生老病死皆永盡故。}@
%{甘露，指的是生老病死都永远消失了。}%
@{安隱者，謂超過一切人與非人災橫怖畏故。}@
%{安隐，指的是超越一切人与非人的灾横恐惧。}%
@{清涼者，謂一切苦皆寂滅故，極清涼故。}@
%{清凉，指的是一切苦都寂灭了，非常清凉。}%
@{善事者，謂現法樂住所緣境故。}@
%{善事，指的是现法乐所的所缘境。}%
@{趣吉祥者，謂斷一切煩惱所緣境故。}@
%{趣吉祥，指的是断除一切烦恼的所缘境。}%
@{無愁憂者，謂超過一切愛非愛故。}@
%{无愁忧，指的是超越一切爱和非爱。}%
@{又證得已無失壞故。}@
%{又已经证得，不会失去。}%
@{不死沒者，謂常住故，不退還故。}@
%{不死没，指的是常住，不退还。}%
@{無熾然者，謂清淨故。}@
%{无炽然，指的是清净。}%
@{無熱惱者，謂所欲匱乏永止息故。}@
%{无热恼，指的是所欲的匮乏永远止息。}%
@{無病者，謂一切病諸癰瘡等永寂靜故。}@
%{无病，指的是一切病，诸痈疮等等永远寂静。}%
@{無動亂者，謂一切動亂皆滅盡故。}@
%{无动乱，指的是一切动乱都灭尽了。}%
@{涅槃者，謂一切依皆寂滅故。}@
%{涅槃，指的是一切依都寂灭了。}%
@{復次，我何當不有、我所何當不有者，謂約未來世，於我、我所性所攝，內處、外處所攝自內體性及攝受事，希求不生故。}@
%{再来说说，为什么我会不存在，我的东西也会不存在呢？这是针对未来而言的，因为我、我的东西，本质上都是由内在和外在的因素所构成，而这些因素，无论是内在的本质，还是外在的事物，都是为了避免未来再生而存在的。}%
@{又復顯示希求依止不生故，及希求依彼受不生故。}@
%{也就是说，为了避免未来再生，人们会寻求依靠某种事物，或者寻求依靠某种感受来避免再生。}%
@{我當不有、我所當不有者，謂約現在世說。}@
%{而我为什么不存在，我的东西为什么不存在呢？这是针对现在而言的。}%
@{此觀無常滅，前觀於擇滅。}@
%{现在，人们看到的是无常的灭亡，而之前，人们看到的是选择灭亡。}%
@{又前但有希望故，後於現在因觀無常性故。}@
%{之前，人们只是怀着希望，而现在，人们是因为看到了无常的本质，才选择灭亡。}%

@{復次，不相續者，謂死歿已後，餘識不生故。}@
%{再说，为什么说是不相续呢？因为人死后，剩余的意识不会再生。}%
@{言無取者，謂無所住識、無有趣入名色事故，自體永不生故。}@
%{为什么说是不取呢？因为没有意识可以依附，也没有意识可以进入色身，所以它本身永远不会再生。}%
@{無生長者，謂無有名色更增廣故。}@
%{为什么说是不生长呢？因为色身不会再增长。}%
@{言一切行皆寂止者，謂諸五蘊皆止息故。}@
%{为什么说是一切行为都寂止呢？因为五蕴都停止了。}%

@{復次，所言空者，謂離一切煩惱等故。}@
%{再说，为什么说空呢？因为远离了一切烦恼。}%
@{無所得者，謂離一切所有相故。}@
%{为什么说无所得呢？因为远离了一切拥有。}%
@{言愛盡者，謂不希求未來事故。}@
%{为什么说爱尽呢？因为不再希求未来。}%
@{言離欲者，謂無現在受用憙樂故。}@
%{为什么说离欲呢？因为不再享受现在的快乐。}%
@{所言滅者，謂餘煩惱斷故。}@
%{为什么说灭呢？因为剩余的烦恼断除了。}%
@{言涅槃者，謂無餘依故。}@
%{为什么说涅槃呢？因为没有剩余的依靠。}%

@{復次，言無常者，謂性破壞朽敗法故。}@
%{再说，为什么说无常呢？因为它的本质就是破损、腐朽、衰败。}%
@{言有為者，謂依前際所尋思故。}@
%{为什么说是有为呢？因为它是根据过去的思考而产生的。}%
@{言造作者，謂依後際所希望故。}@
%{为什么说是有造作呢？因为它是根据未来的希望而产生的。}%
@{言緣生者，謂依現世眾因緣力所生起故。}@
%{为什么说是有缘生呢？因为它是根据现在的各种因缘而产生的。}%
@{有盡法者，謂一分盡故。}@
%{为什么说是有尽法呢？因为有一部分已经消失了。}%
@{有沒法者，謂全分滅故。}@
%{为什么说是有没法呢？因为全部都消失了。}%
@{又有盡法者，謂全分滅故。}@
%{为什么说是有尽法呢？因为全部都消失了。}%
@{有沒法者，謂相續變壞故。}@
%{为什么说是有没法呢？因为不断地变化和破损。}%
@{有離欲法者，謂過患相應故。}@
%{为什么说是有离欲法呢？因为它伴随着过患。}%
@{有滅法者，謂一切有為法皆有出離故。}@
%{为什么说是有灭法呢？因为一切有为法都会有出离。}%

@{復次，無餘斷者，謂是總句。}@
%{再说，为什么说无余断呢？这是总括性的说法。}%
@{永棄捨者，諸纏斷故。}@
%{为什么说永弃舍呢？因为一切的烦恼都断除了。}%
@{永變吐者，隨眠斷故。}@
%{为什么说永变吐呢？因为一切的潜在烦恼都断除了。}%
@{言永盡者，過去解脫故。}@
%{为什么说永尽呢？因为过去已经解脱了。}%
@{永離欲者，現在解脫故。}@
%{为什么说永离欲呢？因为现在已经解脱了。}%
@{言永滅者，未來解脫故。}@
%{为什么说永灭呢？因为未来也会解脱。}%
@{永寂靜者，由見道故。}@
%{为什么说永寂静呢？因为已经证得了见道。}%
@{永滅沒者，由修道故。}@
%{为什么说永灭没呢？因为已经证得了修道。}%
@{當知此中，由二種道斷煩惱事，顯無餘斷。}@
%{要知道，这两种道，断除了烦恼，显现出无余断。}%
