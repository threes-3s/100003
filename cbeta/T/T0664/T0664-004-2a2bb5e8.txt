@{合部金光明經卷第四}@
%{合部金光明經卷第四}%

@{隋大興善寺沙門釋寶貴合北涼天竺三藏曇無讖譯}@
%{隋大興善寺沙門釋寶貴合北涼天竺三藏曇無讖譯}%

@{爾時，佛告地神堅牢善女天：過去有王名金龍尊，常以讚歎讚歎去、來、現在諸佛：}@
%{当时，佛陀对地神坚牢善女天说：过去有一个国王，名叫金龙尊，他经常赞叹过去、现在、未来诸佛：}%
@{我今尊重、敬禮、讚歎，去、來、現在，}@
%{我现在恭敬、礼拜、赞叹，过去、现在、未来，}%
@{十方諸佛。諸佛清淨，微妙寂滅，}@
%{十方诸佛。诸佛清净，微妙寂灭，}%
@{色中上色，金光照曜。於諸聲中，}@
%{色中之上色，金光照耀。在各种声音中，}%
@{佛聲最上，猶如大梵，深遠雷音。}@
%{佛声最上，如同大梵天，深远如雷鸣。}%
@{其髮紺黑，光螺焰起，蜂翠孔雀，}@
%{他的头发乌黑发亮，光彩如螺髻火焰，像蜂翠孔雀，}%
@{色不得喻。其齒鮮白，猶如珂雪，}@
%{颜色无法形容。他的牙齿洁白如珂雪，}%
@{顯發金顏，分齊分明。其目脩廣，}@
%{显露出金色的面容，牙齿整齐分明。他的眼睛修长宽广，}%
@{清淨無垢，如青蓮華，映水開敷。}@
%{清净无垢，如同青莲花，映照在水中开放。}%
@{舌相廣長，形色紅暉，光明照耀，}@
%{舌头宽长，形状红润，光明照耀，}%
@{如華初生。眉間毫相，白如珂月，}@
%{如同花朵初生。眉间毫相，洁白如珂月，}%
@{右旋潤澤，如淨琉璃。眉細脩楊，}@
%{右旋润泽，如同净琉璃。眉毛细长修长，}%
@{形如月初，其色黑曜，過於蜂王。}@
%{形状如同月牙初现，颜色黑曜，胜过蜂王。}%
@{鼻高圓直，如鑄金鋌，微妙柔軟，}@
%{鼻子高耸圆直，如同铸造的金铤，微妙柔软，}%
@{當于面門。如來勝相，次第最上，}@
%{当于面门。如来的胜相，次第最上，}%
@{得味真正，無與等者。一一毛孔，}@
%{得到味道真实，无与伦比。每一毛孔，}%
@{一毛旋生，軟細紺青，猶孔雀項。}@
%{一根根毛发旋生，柔软细致，紺青色，如同孔雀的脖子。}%
@{即於生時，身放大光，普照十方，}@
%{就在出生的时候，身体放射出光芒，普照十方，}%
@{無量國土，滅盡三界，一切諸苦，}@
%{无量国土，灭尽三界，一切苦难，}%
@{令諸眾生，悉受快樂。地獄、畜生、}@
%{让一切众生，都得到快乐。地狱、畜生，}%
@{及以餓鬼、諸人、天等，安隱無患，}@
%{以及饿鬼、人、天等，安稳无患，}%
@{悉滅一切，無量惡趣。身色微妙，}@
%{全部灭尽一切，无量恶趣。身体颜色微妙，}%
@{如鎔金聚；面貌清淨，如月盛滿；}@
%{如同熔化的金子聚集；面容清净，如同月亮圆满；}%
@{佛身明曜，如日初出；進止威儀，}@
%{佛身光明照耀，如同太阳初升；进退威仪，}%
@{猶如師子；脩臂下垂，立過于膝，}@
%{如同狮子；修长的胳膊垂下，站立超过膝盖，}%
@{猶如風動，娑羅樹枝；圓光一尋，}@
%{如同风吹动，娑罗树枝；圆光一寻，}%
@{能照無量，猶如聚集，百千日月。}@
%{能够照亮无量，如同聚集，百千万日月。}%
@{佛身淨妙，無諸垢穢，其明普照，}@
%{佛身清净美好，没有污垢，光芒普照，}%
@{一切佛剎。佛光巍巍，明焰火盛，}@
%{一切佛刹。佛光巍巍，光明火焰旺盛，}%
@{悉能隱蔽，無量日月。佛日燈炬，}@
%{全部能够遮蔽，无量日月。佛日如同灯炬，}%
@{照無量界，皆令眾生，尋光見佛。}@
%{照亮无量世界，让一切众生，寻着光找到佛。}%
@{本所修集，百千行業，聚集功德，}@
%{本来所修集的，百千万种行为，聚集功德，}%
@{莊嚴佛身。臂𦟛纖圓，如象王鼻，}@
%{庄严佛身。胳膊纤细圆润，如同象王的鼻子，}%
@{手足淨軟，敬愛無厭。去、來諸佛，}@
%{手足清净柔软，敬爱无厌。过去、未来诸佛，}%
@{數如微塵；現在諸佛，亦復如是。}@
%{数量如同微尘；现在诸佛，也如同如此。}%
@{如是如來，我今悉禮，身、口清淨、}@
%{这样的如来，我现在全部礼拜，身、口清净，}%
@{意亦如是。以好華香，供養奉獻，}@
%{意念也如同如此。用美好的花香，供养奉献，}%
@{百千功德，讚詠歌歎。設以百舌，}@
%{百千万种功德，赞咏歌颂。假设用一百条舌头，}%
@{於千劫中，歎佛功德，不能得盡。}@
%{在千劫中，赞叹佛陀的功德，也不能完全说完。}%
@{如來所有，現世功德，種種深固，}@
%{如来所有，现世的功德，种种深固，}%
@{微妙第一。設復千舌，欲讚一佛，}@
%{微妙第一。假设再用一千条舌头，想要赞叹一位佛陀，}%
@{尚不能盡，功德少分；況欲歎美，}@
%{尚且不能说完，功德的少部分；何况想要赞美，}%
@{諸佛功德？大地及天，以為大海，}@
%{诸佛的功德？大地和天空，当作大海，}%
@{乃至有頂，滿其中水，尚可以毛，}@
%{乃至有顶，充满其中的水，尚且可以用毛发，}%
@{知其渧數；無有能知，佛一功德。}@
%{知道它的滴数；没有谁能知道，佛陀的一项功德。}%
@{我今已禮，讚歎諸佛，身、口、意業，}@
%{我现在已经礼拜，赞叹诸佛，身、口、意业，}%
@{悉皆清淨。一切所修，無量善業，}@
%{全部都清净。一切所修的，无量善业，}%
@{與諸眾生，證無上道。如是人王，}@
%{与一切众生，证得无上道。这样的国王，}%
@{讚歎佛已，復作如是，無量誓願：}@
%{赞叹佛陀之后，又做出这样的，无量誓愿：}%
@{若我來世，無量無邊，阿僧祇劫，}@
%{如果我未来世，无量无边，阿僧祇劫，}%
@{在在生處，常於夢中，見妙金鼓，}@
%{在在生处，经常在梦中，见到奇妙的金鼓，}%
@{得聞懺悔，深奧之聲。今所讚歎，}@
%{听到忏悔，深奥的声音。现在所赞叹的，}%
@{面貌清淨，願我來世，亦得如是。}@
%{面容清净，愿我未来世，也得到这样。}%
@{諸佛功德，不可思議，於百千劫，}@
%{诸佛的功德，不可思議，在百千万劫中，}%
@{甚難得值，願於當來，無量之世，}@
%{非常难得遇到，愿在未来，无量世界，}%
@{夜則夢見、晝如實說。我當具足，}@
%{晚上梦到，白天如实地说。我应该具足，}%
@{修行六度，濟拔眾生，越於苦海，}@
%{修行六度，济拔众生，越过苦海，}%
@{然後我身，成無上道，令我世界，}@
%{然后我的身体，成就无上道，使我的世界，}%
@{無與等者。奉貢金鼓，讚佛因緣，}@
%{无与伦比。奉献金鼓，赞叹佛陀的因缘，}%
@{以此果報，當來之世，值釋迦佛，}@
%{以此果报，未来世，遇到释迦佛，}%
@{得授記莂；并令二子，金龍、金光，}@
%{得到授记；并且让两个儿子，金龙、金光，}%
@{常生我家，同共授記。若有眾生，}@
%{经常出生在我家，共同得到授记。如果有众生，}%
@{無救護者，眾苦逼切，無所依止。}@
%{没有救护者，众苦逼迫，无所依止。}%
@{我於當來，為是等輩，作大救護、}@
%{我在未来，为这些众生，做大救护、}%
@{及依止處，能除眾苦，悉令滅盡，}@
%{以及依靠之处，能够消除众苦，全部使之灭尽，}%
@{施與眾生，諸善安樂。我未來世，}@
%{施舍给众生，一切善法安乐。我未来世，}%
@{行菩提道，不計劫數，如盡本際。}@
%{行菩提道，不计劫数，如同尽到本际。}%
@{以此金光，懺悔因緣，使我惡海、}@
%{以此金光，忏悔因缘，使我的恶海、}%
@{及以業海、煩惱大海，悉竭無餘；}@
%{以及业海、烦恼大海，全部枯竭无余；}%
@{我功德海，願悉成就；智慧大海，}@
%{我的功德海，愿全部成就；智慧大海，}%
@{清淨具足。無量功德，助菩提道，}@
%{清净具足。无量功德，帮助菩提道，}%
@{猶如大海，珍寶具足。以此金光，}@
%{如同大海，珍宝具足。以此金光，}%
@{懺悔力故，菩提功德，光明無礙。}@
%{忏悔力故，菩提功德，光明无碍。}%
@{慧光無垢，照徹清淨，我當來世，}@
%{智慧的光芒无垢，照彻清净，我未来世，}%
@{身光普照。功德威神，光明焰盛，}@
%{身体光芒普照。功德威神，光明火焰旺盛，}%
@{於三界中，最勝殊特。諸功德力，}@
%{在三界中，最胜殊特。一切功德力，}%
@{無所減少，當度眾生，越於苦海，}@
%{无所减少，当度化众生，越过苦海，}%
@{并復安置，功德大海。來世多劫，}@
%{并且安置，功德大海。未来世多劫，}%
@{行菩提道，如昔諸佛，行菩提道。}@
%{行菩提道，如同过去诸佛，行菩提道。}%
@{三世諸佛，淨妙國土；諸佛至尊，}@
%{三世诸佛，清净美好的国土；诸佛至尊，}%
@{無量功德。令我來世，得此殊異，}@
%{无量功德。使我未来世，得到这样的殊异，}%
@{功德淨土，如佛世尊。信相當知：}@
%{功德净土，如同佛世尊。信者应该知道：}%
@{爾時國王，金龍尊者，則汝身是；}@
%{当时的国王，金龙尊者，就是你的身体；}%
@{爾時二子，金龍、金光，今汝二子，}@
%{当时的两个儿子，金龙、金光，现在你的两个儿子，}%
@{銀相等是。}@
%{银相等等就是。}%
@{無量餘經，已廣說空，是故此中，}@
%{很多经文已经详细地阐述了空性，所以在这里，}%
@{略而解說。眾生根鈍，尠於智慧，}@
%{就简单地解释一下。众生的根性愚钝，智慧很少，}%
@{不能廣知，無量空義。故此尊經，}@
%{不能广泛地了解无量空性的道理。所以这部尊贵的经文，}%
@{略而說之，異妙方便，種種因緣。}@
%{就简略地阐述它，以种种因缘巧妙方便的方式。}%

@{為鈍根故，起大悲心，今我演說，}@
%{因为众生根性愚钝，我生起了大悲心，现在我来宣说，}%
@{此妙經典。如我所解，知眾生意，}@
%{这部奇妙的经典。正如我所理解的，我知道众生的意识，}%
@{是身虛偽，猶如空聚。六入村落，}@
%{是虚假的身体，如同空聚。六根就像村落，}%
@{結賊所止，一切自住，各不相知。}@
%{是结贼所住的地方，一切事物都各自存在，彼此不相识。}%

@{眼根受色、耳分別聲、鼻嗅諸香、}@
%{眼睛感受颜色，耳朵分辨声音，鼻子嗅闻香气，}%
@{舌嗜於味、所有身根，貪受諸觸、}@
%{舌头贪嗜味道，所有身体的根，贪婪地感受各种触觉，}%
@{意根分別，一切諸法。六情諸根，}@
%{意根分辨一切法。六根的意识，}%
@{各各自緣，諸塵境界，不行他緣。}@
%{各自缘于各自的尘境，不缘于其他的境。}%

@{心如幻化，馳騁六情，而常妄想，}@
%{心如同幻化，在六根意识中驰骋，而常常妄想，}%
@{分別諸法。猶如世人，馳走空聚，}@
%{分别各种法。如同世人，在空聚中奔走，}%
@{六賊所害，愚不知避。心常依止，}@
%{被六贼所害，愚痴不知躲避。心常常依止，}%
@{六根境界，各各自知，所伺之處。}@
%{六根的境界，各自了解，所观察的地方。}%

@{隨行色、聲、香、味、觸、法，心處六情，}@
%{跟随颜色、声音、香气、味道、触觉、法，心住在六根意识中，}%
@{如鳥投網。其心在在，常處諸根，}@
%{如同鸟儿飞入网中。它的心到处都在，常常住在六根中，}%
@{隨逐諸塵，無有暫捨。身空虛偽，}@
%{跟随各种尘境，没有片刻舍弃。身体是空虚的，}%
@{不可長養，無有諍訟、亦無正主。}@
%{不能培养，没有争执，也没有主人。}%

@{從諸因緣，和合而有，無有堅實，}@
%{从各种因缘和合而成，没有坚实，}%
@{妄想故起。業力機關，假偽空聚，}@
%{是妄想所生。业力的机关，是虚假的空聚，}%
@{地、水、火、風，合集成立，隨時增減，}@
%{地、水、火、风，组合在一起形成，随时增减，}%
@{共相殘害，猶如四蛇，同處一篋。}@
%{互相残害，如同四条蛇，一同在一个箱子里。}%

@{四大蚖蛇，其性各異，二上、二下，}@
%{四大就像四条蛇，它们的性情各不相同，两个向上，两个向下，}%
@{諸方亦二，如是蛇大，悉滅無餘。}@
%{各个方向也有两个，这样大的蛇，全部都灭尽无余。}%
@{地、水二蛇，其性沈下；風、火二蛇，}@
%{地、水这两条蛇，它们的性情沉下去；风、火这两条蛇，}%
@{性輕上昇。心、識二性，躁動不停，}@
%{性情轻浮上升。心、识这两个性，躁动不停，}%
@{隨業受報，天人諸趣，隨所作業，}@
%{随着业力受报，天人各种趣，随着所做的业，}%
@{而墮三有。水、火、風種，散滅壞時，}@
%{而堕入三界。水、火、风这三种，散灭坏掉的时候，}%

@{大小不淨，盈流於外，體生諸蟲，}@
%{大小不净，充满流到外面，身体生出各种虫子，}%
@{無可愛樂，捐棄塚間，如朽敗木。}@
%{没有可爱之处，被抛弃在坟墓里，如同腐烂的木头。}%

@{善女當觀：諸法如是，何處有人、}@
%{善女人应该观察：诸法就是这样，哪里有人，}%
@{及以眾生？本性空寂，無明故有。}@
%{以及众生？本性空寂，因为无明而有。}%

@{如是諸大，一一不實，本自不生，}@
%{这样的大地，每一个都不真实，本来就不生，}%
@{性無和合。以是因緣，我說諸大，}@
%{性没有和合。因为这个缘故，我说大地，}%
@{從本不實，和合而有。無明體性，}@
%{从本不真实，和合而成。无明的体性，}%
@{本自不有，妄想因緣，和合而生。}@
%{本来就没有，是妄想的因缘，和合而生。}%

@{無所有故，假名無明，是故我說，}@
%{因为没有，所以假名为无明，所以我说，}%
@{名曰無明。行、識、名色、六入、觸、受、}@
%{叫做无明。行、识、名色、六入、触、受，}%
@{愛、取、有、生、老、死、憂、惱，眾苦行業，}@
%{爱、取、有、生、老、死、忧、恼，众苦的行业，}%
@{不可思議。生死無際，輪轉不息，}@
%{不可思议。生死无边，轮回不止，}%

@{本無有生，亦無和合。不善思惟，}@
%{本来就没有生，也没有和合。不善思惟，}%
@{心行所造，我斷一切，諸見纏等。}@
%{心行所造作，我断一切见惑等等。}%
@{以智慧刀，裂煩惱網，五陰舍宅，}@
%{用智慧的刀，斩断烦恼的网，五蕴的住宅，}%
@{觀悉空寂。證無上道，微妙功德，}@
%{观察都空寂。证得无上道，微妙的功德，}%

@{開甘露門、示甘露器、入甘露城、}@
%{打开甘露之门，示现甘露之器，进入甘露之城，}%
@{處甘露室，令諸眾生，食甘露味。}@
%{住在甘露之室，让所有众生，吃甘露的味道。}%
@{吹大法螺、擊大法鼓、然大法燈、}@
%{吹响大法螺，敲击大法鼓，燃点大法灯，}%
@{雨勝法雨。我今摧伏，一切怨結，}@

@{是時，如意寶光耀善女天於大眾中從座而起，偏袒右肩，右膝著地，合掌恭敬以偈白佛：}@
%{这时候，如意宝光耀善女天从座位上站起来，在众人面前，袒露右肩，右膝跪地，双手合十，恭敬地用偈颂向佛陀致敬：}%

@{我問照世界，兩足最勝尊，}@
%{我向照耀世界的、两足最尊贵的佛陀询问：}%

@{菩薩正行法，惟願垂聽許。}@
%{菩萨的正修行法，希望您能允许我聆听。}%

@{佛言：善女天！汝若有疑者，}@
%{佛陀说：“善女天！如果你有任何疑问，}%

@{隨汝意所問，吾當分別說。}@
%{就随你的心意提问，我会一一解释。”}%

@{云何諸菩薩，行菩提正行、}@
%{“诸位菩萨是如何行持菩提正行，}%

@{離生死涅槃，利益自他故？}@
%{脱离生死涅槃，为了利益自己和他人呢？”}%

@{佛言：善女天！依於法界行菩提法、修平等行。}@
%{佛陀说：“善女天！要依于法界行持菩提法，修习平等行。}%

@{善女天！云何依於法界行菩提法、修平等行？}@
%{善女天！什么是依于法界行持菩提法、修习平等行呢？}%

@{善女天！五陰能現法界、法界是五陰，五陰亦不可說、非五陰亦不可說。}@
%{善女天！五蕴能显现法界，法界就是五蕴，五蕴不可说，非五蕴也不可说。}%

@{何以故？若五陰是法界則是斷見、若離五陰即是常見，離於二邊、不著二邊，不可見過，所見無名、無相，是則名為說於法界。}@
%{为什么呢？如果五蕴就是法界，那就是断见；如果离开五蕴，那就是常见。要远离两边，不执着于两边，不可见有，所见无名、无相，这才是说于法界。}%

@{善女天！云何五陰能現法界？}@
%{善女天！五蕴是如何显现法界的呢？}%

@{善女天！如是五陰不從因緣生。}@
%{善女天！这样的五蕴，不是从因缘而生。}%

@{何以故？若從因緣生，已生故得生、未生故得生？}@
%{为什么呢？如果从因缘而生，已经生了就应该生，没生就应该生。}%

@{若已得生，何因緣生？若已生，不從因緣生。}@
%{如果已经生了，还有什么因缘生呢？如果已经生了，就不从因缘而生。}%

@{若未生時，不可得生。}@
%{如果还没有生的时候，就不可能生。}%

@{何以故？未生諸法則是不有，無名、無相，非算數譬喻之所能知、非因緣所生。}@
%{为什么呢？因为还没有生的法则是无有的，无名、无相，不是算数譬喻所能知，不是因缘所生。}%

@{善女天！譬如鼓聲，依木、依皮、依捊、依人工等故得出聲。}@
%{善女天！比如鼓声，依靠木头、依靠皮革、依靠鼓槌、依靠人工等等才发出声音。}%

@{是鼓聲空，過去亦空、未來亦空、現在亦空。}@
%{这个鼓声是空的，过去也空，未来也空，现在也空。}%

@{何以故？是鼓音聲不從木生、不從皮生、不從捊生、不從人工生，是聲不於三世生，是則不生。}@
%{为什么呢？因为鼓声不是从木头生，不是从皮革生，不是从鼓槌生，不是从人工生，这个声音不在三世中生，所以它是不生。}%

@{若不可得生則不可滅；若不可滅，無所從來；若無所從來，亦無處去；若無處去，不常、不斷；若不常、不斷，則不一、不異。}@
%{如果不能生，就不能灭；如果不能灭，就无处而来；如果无处而来，也无处可去；如果无处可去，就不常、不断；如果就不常、不断，那么就不一、不异。}%

@{何以故？若一，不異法界，若爾者，凡夫人則見真諦，得於無上安樂涅槃；是義不然，是故不一。}@
%{为什么呢？如果一，就不异于法界，如果这样的话，凡夫就能见到真谛，获得无上安乐涅槃；但事实并非如此，所以不一。}%

@{若言其異者，一切諸佛菩薩行相即是執著，未得解脫煩惱繫縛則不能得阿耨多羅三藐三菩提。}@
%{如果说它异，那么一切佛菩萨的行相就是执着，没有解脱烦恼束缚，就不能获得阿耨多罗三藐三菩提。}%

@{何以故？一切聖人於行、非行法中同智慧行，是故不異。}@
%{为什么呢？因为一切圣人在行、非行法中都以智慧行，所以不异。}%

@{是故，五陰非有，不從因緣生、非不有五陰。}@
%{所以，五蕴并非有，不是从因缘而生，也不是没有五蕴。}%

@{不過聖境界故，非言語之所能及，無名、無相，無因、無緣，無有境界、無有譬喻，始終寂靜、本來自空。}@
%{只是圣境的境界，不是语言所能及，无名、无相，无因、无缘，无有境界、无有譬喻，始终寂静，本来就空。}%

@{是故，五陰能現法界。}@
%{所以，五蕴能显现法界。}%

@{善女天！若善男子、善女人欲求阿耨多羅三藐三菩提，真異、俗異，如是難可思量，於聖凡境界不異思惟，不捨於俗、不捨於真，依於法界行菩提行。}@
%{善女天！如果善男子、善女人想要寻求阿耨多罗三藐三菩提，真谛和俗谛，这样难以思量，要在圣凡境界不异的思维中，不舍俗谛、不舍真谛，依于法界行持菩提行。}%

@{爾時，世尊作是語已，時善女天踊躍歡喜，即從坐起，偏袒右肩，右膝著地，合掌恭敬一心頂禮，而白佛言：世尊！如上所說，菩提正行我今當學。}@

@{是時，娑婆世界主大梵天王於大眾中問如意寶光耀善女天：此菩提行難可修行，汝心云何於此菩提行而得自在？}@
%{这时候，娑婆世界的主人大梵天王在众人面前问如意宝光耀善女天：“这菩提行很难修行，你的心是如何在这菩提行中获得自在的呢？”}%

@{時善女天答梵王言：大梵王！若佛所說是真甚深，一切凡夫不得其味，是聖境界微妙難知。}@
%{善女天回答大梵天王说：“大梵王！佛陀所说的是真实的，非常深奥，一切凡夫都无法体会到其中的滋味，这是圣境的境界，微妙难知。}%

@{若使我心依於此法得安樂住，是真實語者，願令一切五濁惡世無量無邊眾生皆得金色、三十二相，非男、非女，坐寶蓮華受無量快樂，雨天妙華，天諸音樂不鼓自鳴，一切供養皆悉具足。}@

@{是時善女天說是語已，一切五濁惡世所有眾生皆悉金色，具足三十二相，非男、非女，坐寶蓮華受無量快樂。}@
%{这时候，善女天说完这些话，一切五浊恶世的所有众生都变成了金色，具足三十二相，不男不女，坐在宝莲华上，享受无量快乐。}%

@{猶如他化自在天宮，無諸惡道，寶樹行列，七寶蓮華遍滿世界，雨眾七寶、上妙天華，作天伎樂。}@
%{就像他化自在天宫一样，没有恶道，宝树排列成行，七宝莲华遍满世界，降下各种七宝、上妙天华，演奏天上的音乐。}%

@{如意寶光耀善女天即轉女形作梵天身。}@
%{如意宝光耀善女天立刻转变女身，成为梵天的身形。}%

@{時大梵天王問如意寶光耀菩薩言：汝昔以何行菩提行？}@
%{这时候，大梵天王问如意宝光耀菩萨说：“你以前用什么方法行持菩提行呢？”}%

@{菩薩答言：梵王！若水中月能行菩提行者，我亦已行菩提之行；若夢見行菩提行，我亦行菩提行；若焰、露行菩提行，我亦行菩提行；若聲響行菩提行，我亦行菩提行。}@
%{菩萨回答说：“梵王！如果水中月能行持菩提行，我也已经行持菩提之行；如果梦中行持菩提行，我也行持菩提行；如果火焰、露水行持菩提行，我也行持菩提行；如果声音行持菩提行，我也行持菩提行。”}%
@{時大梵王聞此說已語菩薩言：汝依何等而說此語？}@
%{大梵王听到这些话之后，就问菩萨说：“你根据什么来说这些话呢？”}%

@{答言：梵王！無有一法而有實相，因果相成故。}@
%{菩萨回答说：“梵王！没有一种法是真实存在的，都是因为因果相成才产生的。”}%

@{梵王又白菩薩言：若如此者，諸凡夫人皆悉應得阿耨多羅三藐三菩提。}@
%{大梵王又对菩萨说：“如果真是这样的话，那么所有凡夫众生都应该能得到无上正等正觉。”}%

@{菩薩答言：以何思惟而作是說？梵王！愚癡人異、智慧人異，菩提異、非菩提異，解脫異、非解脫異。梵王！如是，諸法平等無異。於此法界如如不異，無有中間而可執持，無增、無減。}@
%{菩萨回答说：“你用什么想法来这样说呢？梵王！愚痴的人和智慧的人不同，菩提和非菩提不同，解脱和非解脱也不同。梵王！就是这样，所有法都是平等无二的。在这个法界中，一切如如不动，没有中间可以执着，没有增加，也没有减少。”}%

@{梵王！譬如幻師善巧幻術，及幻弟子於四衢道取諸土沙、草木、葉等聚在一處，作種種幻術，使人覩見象眾、馬眾、車眾、軍眾、七寶之聚、種種倉庫。若有眾生愚癡無智、不能思惟、不知幻本，若見若聞作是思惟：如我所見，象、馬眾等謂是真實，如見、如聞。隨能隨力各執所見，自言是實、於他非真，後不重思惟。有智之人則能思惟了於幻本，若見若聞作是思惟：如我所見，象、馬等眾非是真實，惟有幻事惑人眼目。是處說名象、馬等眾及諸庫倉，惟有名字、無有實體。如我所見、如我所聞。隨能隨力不執所見、自言是實、於他非真、後不重思惟。是諸智人隨說世語，皆欲令他知實義故，如見如聞思惟則不如是。}@
%{梵王！就像魔术师精通魔术，他的徒弟在十字路口收集泥沙、草木、树叶等东西堆在一起，做各种各样的魔术，让人看到象群、马群、车队、军队、七宝的聚合、各种各样的仓库。如果有一些众生愚痴无智，不能思考，不知道魔术的本质，看到或听到这些，就认为：就像我看到的那样，象群、马群等等都是真实的，就像我看到、听到的一样。他们尽力地执着于自己所看到的，自称是真实的，认为别人看到的都是假的，以后不再重新思考。有智慧的人就能思考魔术的本质，看到或听到这些，就认为：就像我看到的那样，象群、马群等等都不是真实的，只是魔术迷惑人的眼睛。这里说叫象群、马群等等，还有仓库，只是名字而已，没有实体。就像我看到、听到的一样。他们尽力地不执着于自己所看到的，自称是真实的，认为别人看到的都是假的，以后不再重新思考。这些有智慧的人所说的话，都是为了让别人知道真相，就像看到、听到那样思考，就和他们不一样。}%

@{如是，梵王！若有眾生——愚癡凡夫未得出世聖智慧故——未知一切諸法如如不可言說，是諸凡愚若見若聞行、非行法，作是思惟：實有如是諸法，如我所見、如我所聞。是諸凡夫人如見、如聞，隨能隨力執著所見，自言是實、於他非真，後不重思惟。若有眾生——非凡夫人，已見第一義諦，得出世聖慧——知一切法如如不可言說，是諸聖人若見若聞行法、非行法，隨能隨力不執著所見、自言是實、於他非真、後不重思惟。無實行法、無實非行法，如我所見、如我所聞，惟妄思惟行、非行相、或人智慧。是處說名行、非行法，惟有名字，無有實體。如我所見、如我所聞。隨能隨力不執所見、自言是實、於他非真、後不重思惟。是諸聖人如世語言隨順其說，為欲令他知真實義。如是，梵王！是諸聖人聖智見故，不可言法如如攝行、非行法，是法如如，為他證知故說種種名。}@
%{梵王！就像这样，如果有一些众生——愚痴的凡夫没有获得出世的圣智慧——不知道所有法都是如如不可言说，这些愚痴的凡夫看到或听到行法、非行法，就认为：确实有这样的法，就像我看到、听到的一样。这些凡夫众生看到或听到，尽力地执着于自己所看到的，自称是真实的，认为别人看到的都是假的，以后不再重新思考。如果有一些众生——不是凡夫众生，已经看到了第一义谛，获得了出世的圣慧——知道所有法都是如如不可言说，这些圣人看到或听到行法、非行法，尽力地不执着于自己所看到的，自称是真实的，认为别人看到的都是假的，以后不再重新思考。没有真实的行法，没有真实的非行法，就像我看到、听到的一样，只是虚妄地思考行法、非行法的相状，或者人的智慧。这里说叫行法、非行法，只是名字而已，没有实体。就像我看到、听到的一样。他们尽力地不执着于自己所看到的，自称是真实的，认为别人看到的都是假的，以后不再重新思考。这些圣人就像世俗语言一样，顺应他们的说法，为了让别人知道真相。梵王！就像这样，这些圣人因为圣智的见地，所以不可言说法的如如摄受行法、非行法，这个法是如如的，为了让他们证知，所以说各种各样的名字。}%

@{時大梵王問如意寶光耀菩薩言：有幾眾生能解能通如是微妙甚深正法？}@
%{这时，大梵王问如意宝光耀菩萨说：“有多少众生能够理解和通达这样微妙深奥的正法呢？”}%

@{菩薩答言：梵王！凡是若干眾幻化人心數，若干眾生能解能通是甚深法。}@
%{菩萨回答说：“梵王！凡是所有众生幻化人心数，有多少众生就能理解和通达这个深奥的法。”}%

@{梵王又言：此幻化人即是不有，如是心數從何而得？}@
%{大梵王又说：“这个幻化的人本来就不存在，那么这些心数是从哪里来的呢？”}%

@{菩薩答言：梵王！如是法界不有、不無，如是眾生能解能通是甚深義。}@
%{菩萨回答说：“梵王！这个法界既不有也不无，能够理解和通达这个深奥义理的众生，也就是这样。”}%

@{是時梵王白佛言：世尊！是如意寶光耀菩薩不可思議，通達如是甚深之義。}@
%{这时，大梵王对佛说：“世尊！这个如意宝光耀菩萨不可思议，通达这样深奥的义理。”}%

@{佛言：如是，如是。梵王！如汝所言。何以故？是如意寶光耀菩薩已教梵王學觀無生忍法。}@
%{佛说：“是的，是的。梵王！正如你所说。为什么呢？因为如意宝光耀菩萨已经教导梵王学习观无生忍法。”}%

@{於是大梵天王與諸梵眾從坐而起，偏袒右肩，右膝著地，合掌恭敬頂禮如意寶光耀菩薩足，說如是言：希有，希有。我等今日得見大師、得聞正法。}@
%{于是，大梵天王和所有的梵众从座位上站起来，袒露右肩，右膝着地，合掌恭敬地顶礼如意宝光耀菩萨的脚，说：“希有，希有。我们今天有幸见到大师，听到正法。”}%

@{爾時，世尊於一切法通達無礙，告梵王言：是如意寶光耀菩薩於未來世當得作佛，號曰德寶焰吉上藏如來、應供、正遍知，說是金光明微妙經典，三千億菩薩得不退阿耨多羅三藐三菩提、八千億天子得無垢淨於法成就清淨法眼、無量無數國王臣民得法眼淨。}@
%{那时，世尊对于一切法都通达无碍，对梵王说：“这个如意宝光耀菩萨在未来世将会成佛，号称德宝焰吉上藏如来、应供、正遍知，他所说的是金光明微妙经典，三千亿菩萨将获得不退无上正等正觉，八千亿天子将获得无垢净于法，成就清净法眼，无量无数的国王臣民将获得法眼清净。”}%

@{五十億比丘行菩提行欲退菩提心，聞如意寶光耀菩薩說法，得堅固不可思議滿足之願，更復還發菩提之心，各自脫衣供養菩薩，重發無上勝進心。發無上勝進心已：願令我等功德善根悉皆不退，迴向阿耨多羅三藐三菩提。}@
%{五亿比丘在行菩提道的时候，想要退却菩提心，听到如意宝光耀菩萨说法，获得了坚固不可思议的圆满愿望，再次发起了菩提心，他们各自脱下衣服供养菩萨，重新发起了无上胜进心。发起了无上胜进心之后，他们发愿：愿我们所有的功德善根都不要退转，回向无上正等正觉。}%
@{如是比丘依此功德修行，過九十大劫當得成就，是諸比丘出於生死，佛為授記：過三十阿僧祇劫當得作佛，號難勝光王，其國名曰無垢光，同時皆得阿耨多羅三藐三菩提，皆同一號，名曰願莊嚴間廁王佛。}@
%{如果这些比丘按照这个功德修行，经过九十个大劫就能成就，这些比丘就能脱离生死，佛陀给他们授记：经过三十个阿僧祇劫就能成佛，佛号叫难胜光王，他的国土名叫无垢光，同时他们都将获得无上正等正觉，佛号都一样，叫做愿庄严间厕王佛。}%

@{爾時，佛告梵王：是金光明經，正聞正聽有大神力。梵王！百千大劫行六波羅蜜無有方便，若有善男子、善女人已得聽聞是金光明經，書寫半月、半月一過轉讀，是善功德聚於前功德，百千分不及一分，乃至算數譬喻所不能及。梵王！是故，我今當令修學、受持、為他廣說。何以故？如是甚深微妙經典，我行菩薩道時如於戰陣不惜身命，得通此經、受持、讀誦、為他解說。}@
%{这时，佛陀对梵天说：这部金光明经，正听正闻有大神力。梵天！经过上千个大劫修行六度，也没有什么方便法门，如果有人已经听到这部金光明经，抄写半个月，半个月后就翻阅一遍，这种善功德积累在以前的功德之上，百千万分之一都不到，甚至用算术和比喻都无法形容。梵天！所以，我现在要让人们学习、受持、广泛传播这部经。为什么呢？这部经如此深奥微妙，我在修行菩萨道的时候，就像在战场上不惜生命一样，得到了这部经，受持、诵读、为别人讲解。}%

@{梵王！譬如轉輪聖王，若王在世七寶不滅；王若過世，一切七寶自然而盡。梵王！是金光明微妙經典若現在世，大正法寶皆悉不滅。是故，當依金光明經，聽聞、讀誦、受持、為他解說、令他書寫，於功德中行精進波羅蜜，不惜壽命、不憚疲勞，我諸弟子應當如是精勤修學。}@
%{梵天！就像转轮圣王，如果国王在世，七宝就不会灭；如果国王去世了，所有的七宝就会自然消失。梵天！这部金光明经如果在世，正法宝就不会灭。所以，要依金光明经，听闻、诵读、受持、为别人讲解、让人们抄写，在功德中精进修行，不惜生命、不畏劳苦，我的弟子们应该精勤地学习。}%

@{是時，天主大梵天王與無量梵眾、帝釋四王及夜叉眾俱從坐起，偏袒右肩，右膝著地，合掌恭敬而白佛言：我等一切為守護流通是金光明微妙經典，說法法師若有諸難，我當除却，令具諸善、色味滿足、辯才無礙、身心開泰，時會之眾皆令快樂；是處國土若飢饉、怨賊、非人之畏，我等禳却，使其人民豐盛歡逸，皆是我等四王恩力；若有供養是經卷者，我亦當為作大擁護，如佛不異。}@
%{这时，天主大梵天王和无数梵众、四大天王以及夜叉众都从座位上站起来，袒露右肩，右膝着地，合掌恭敬地对佛陀说：我们一切众生为了守护、流通这部金光明经，如果讲经的法师遇到困难，我们都会帮助他，让他拥有各种善法，色香味俱全、辩才无碍、身心舒畅，让听法的人们都感到快乐；如果这个地方发生饥荒、怨贼、非人的恐怖，我们都会消除灾难，使人民富足安乐，这都是我们四天王的恩力；如果有人供养这部经卷，我们也会为他大力护持，就像佛陀一样。}%
