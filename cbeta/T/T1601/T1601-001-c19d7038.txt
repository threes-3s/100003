@{辯相品第一}@
%{辩相品第一}%

@{唯相障真實，及修諸對治，}@
%{只有相障碍着真实，以及修习各种对治方法，}%

@{即此修分位，得果無上乘。}@
%{就在这个修行的次第中，就能获得无上果位。}%

@{虛妄分別有，於此二都無，}@
%{虚妄分别是有，在这两种中都没有，}%

@{此中唯有空，於彼亦有此。}@
%{其中只有空，在那也同样有这个。}%

@{故說一切法，非空非不空，}@
%{所以说一切法，不是空也不是不空，}%

@{有無及有故，是則契中道。}@
%{有无以及有故，这就是契合中道。}%

@{識生變似義、有情、我及了，}@
%{识生变化类似于义、有情、我以及了，}%

@{此境實非有，境無故識無。}@
%{这个境界实际上并非有，境界没有所以识也没有。}%

@{虛妄分別性，由此義得成，}@
%{虚妄分别性，由此义而成就，}%

@{非實有全無，許滅解脫故。}@
%{并非真实存在完全没有，允许灭度解脱，所以。}%

@{唯所執、依他，及圓成實性，}@
%{只有所执、依他，以及圆成实性，}%

@{境故分別故，及二空故說。}@
%{因为境界，所以分别，以及因为二空，所以说。}%

@{依識有所得，境無所得生；}@
%{依识有所得，境界没有所得而生；}%

@{依境無所得，識無所得生。}@
%{依境界没有所得，识没有所得而生。}%

@{由識有得性，亦成無所得，}@
%{由于识有得性，也成就无所得，}%

@{故知二有得，無得性平等。}@
%{所以知道二有得，无得性平等。}%

@{三界心心所，是虛妄分別，}@
%{三界心心所，是虚妄分别，}%

@{唯了境名心，亦別名心所。}@
%{唯了解境界叫做心，也别称为心所。}%

@{一則名緣識，第二名受者，}@
%{一种叫做缘识，第二种叫做受者，}%

@{此中能受用，分別推心所。}@
%{其中能受用，分别推演心所。}%

@{覆障及安立、將導、攝、圓滿、}@
%{覆障以及安立、引导、摄、圆满，}%

@{三分別、受用、引起并連縛、}@
%{三种分别、受用、引起并连缚，}%

@{現前、苦果故，唯此惱世間，}@
%{现前、苦果故，只有这个烦恼世间，}%

@{三、二、七雜染，由虛妄分別。}@
%{三、二、七杂染，由虚妄分别。}%

@{諸相及異門，義差別成立，}@
%{诸相以及异门，义差别成立，}%

@{應知二空性，略說唯由此。}@
%{应该知道二空性，略说唯由此。}%

@{無二有無故，非有亦非無，}@
%{无二无有故，非有亦非无，}%

@{非異亦非一，是說為空相。}@
%{非异亦非一，是说为空相。}%

@{略說空異門，謂真如、實際、}@
%{略说空异门，谓真如、实际、}%

@{無相、勝義性、法界等應知。}@
%{无相、胜义性、法界等应该知道。}%

@{由無變無倒，相滅聖智境，}@
%{由无变无倒，相灭圣智境界，}%

@{及諸聖法因，異門義如次。}@
%{以及诸圣法因，异门义如次。}%

@{此雜染、清淨，由有垢、無垢，}@
%{这个杂染、清净，由有垢、无垢，}%

@{如水界金空，淨故許為淨。}@
%{如同水界中的金空，净故许为净。}%

@{能食及所食，此依身所住，}@
%{能食以及所食，这个依身所住，}%

@{能見如此理，所求二淨空。}@
%{能见如此道理，所求二净空。}%

@{為常益有情、為不捨生死、}@
%{为了常益有情、为了不舍生死、}%

@{為善無窮盡，故觀此為空。}@
%{为了善无穷尽，所以观此为空。}%

@{為種性清淨、為得諸相好、}@
%{为了种性清净、为了得诸相好、}%

@{為淨諸佛法，故菩薩觀空。}@
%{为了净诸佛法，所以菩萨观空。}%

@{補特伽羅法，實性俱非有，}@
%{补特伽罗法，实性俱非有，}%

@{此無性、有性，故別立二空。}@
%{这个无性、有性，所以分别立二空。}%

@{此若無雜染，一切應自脫；}@
%{这个如果无杂染，一切应该自脱；}%

@{此若無清淨，功用應無果。}@
%{这个如果无清净，功用应该无果。}%

@{非染非不染，非淨非不淨，}@
%{非染非不染，非净非不净，}%

@{心性本淨故，由客塵所染。}@
%{心性本净故，由客尘所染。}%

@{辯障品第二}@
%{辩障品第二}%

@{具分及一分，增盛與平等，}@
%{具分以及一分，增盛与平等，}%

@{於生死取捨，說障二種性。}@
%{于生死取舍，说障两种性。}%

@{九種煩惱相，謂愛等九結，}@
%{九种烦恼相，谓爱等九结，}%

@{初二障厭捨，餘七障真見，}@
%{初二障厌舍，余七障真见，}%

@{謂能障身見，彼事滅道寶，}@
%{谓能障身见，彼事灭道宝，}%

@{利養恭敬等，遠離遍知故。}@
%{利养恭敬等，远离遍知故。}%

@{無加行非處，不如理不生，}@
%{无加行非处，不如理不生，}%

@{不起正思惟，資糧未圓滿；}@
%{不起正思惟，资粮未圆满；}%

@{闕種性善友，心極疲厭性，}@
%{阙种性善友，心极疲厌性，}%

@{及闕於正行，鄙惡者同居；}@
%{以及阙于正行，鄙恶者同居；}%

@{倒麁重三餘，般若未成就，}@
%{倒粗重三余，般若未成就，}%

@{及本性麁重，懈怠放逸性；}@
%{以及本性粗重，懈怠放逸性；}%

@{著有著資財，及心性下劣，}@
%{着有着资财，以及心性下劣，}%

@{不信無勝解，如言而取義；}@
%{不信无胜解，如言而取义；}%

@{輕法重名利，於有情無悲，}@
%{轻法重名利，于有情无悲，}%

@{匱聞及少聞，不修治妙定。}@
%{匮闻以及少闻，不修治妙定。}%

@{善菩提攝受，有慧無亂障，}@
%{善菩提摄受，有慧无乱障，}%

@{迴向不怖慳，自在名善等。}@
%{迴向不怖悭，自在名善等。}%

@{如是善等十，各有前三障。}@
%{如是善等十，各有前三障。}%

@{於覺分度地，有別障應知。}@
%{于觉分度地，有别障应知。}%

@{於事不善巧，懈怠、定減二，}@
%{于事不善巧，懈怠、定减二，}%

@{不植羸劣性，見麁重過失。}@
%{不植羸劣性，见粗重过失。}%

@{障富貴、善趣、不捨諸有情、}@
%{障富贵、善趣、不舍诸有情、}%

@{於失德減增、令趣入、解脫，}@
%{于失德减增、令趣入、解脱，}%

@{障施等諸善、無盡亦無間、}@
%{障施等诸善、无尽亦无间、}%

@{所作善決定、受用法成熟。}@
%{所作善决定、受用法成熟。}%

@{遍行與最勝，勝流及無攝，}@
%{遍行与最胜，胜流以及无摄，}%

@{相續無差別，無雜染清淨，}@
%{相续无差别，无杂染清净，}%

@{種種法無別，及不增不減，}@
%{种种法无别，以及不增不减，}%

@{并無分別等，四自在依義。}@
%{并无分别等，四自在依义。}%

@{於斯十法界，有不染無明，}@
%{于斯十法界，有不染无明，}%

@{障十地功德，故說為十障。}@
%{障十地功德，故说为十障。}%

@{已說諸煩惱，及諸所知障，}@
%{已说诸烦恼，以及诸所知障，}%

@{許此二盡故，一切障解脫。}@
%{许此二尽故，一切障解脱。}%

@{辯真實品第三}@
%{辩真实品第三}%

@{真實唯有十，謂根本與相、}@
%{真实唯有十，谓根本与相、}%

@{無顛倒、因果，及麁細真實、}@
%{无颠倒、因果，以及粗细真实、}%

@{極成、淨所行、攝受并差別，}@
%{极成、净所行、摄受并差别，}%

@{十善巧真實，皆為除我見。}@
%{十善巧真实，皆为除我见。}%

@{許於三自性，唯一常非有，}@
%{许于三自性，唯一常非有，}%

@{一有而不真，一有無真實。}@
%{一有而不真，一无有真实。}%

@{於法數取趣，及所取能取，}@
%{于法数取趣，以及所取能取，}%

@{有非有性中，增益損減見；}@
%{有非有性中，增益损减见；}%

@{知此故不轉，是名真實相。}@
%{知此故不转，是名真实相。}%

@{無性與生滅、垢淨三無常；}@
%{无性与生灭、垢净三无常；}%

@{所取及事相、和合苦三種；}@
%{所取以及事相、和合苦三种；}%

@{空亦有三種，謂無、異、自性；}@
%{空亦有三種，谓无、异、自性；}%

@{無相及異相、自相三無我。}@
%{无相以及异相、自相三无我。}%

@{如次四三種，依根本真實。}@
%{如次四三种，依根本真实。}%

@{苦三相已說，集亦有三種，}@
%{苦三相已说，集亦有三種，}%

@{謂習氣、等起，及相未離繫。}@
%{谓习气、等起，以及相未离繫。}%

@{自性、二不生、垢寂二三滅，}@
%{自性、二不生、垢寂二三灭，}%

@{遍知及永斷、證得三道諦。}@
%{遍知以及永断、证得三道谛。}%

@{應知世俗諦，差別有三種，}@
%{应知世俗谛，差别有三種，}%

@{謂假、行、顯了，如次依本三。}@
%{谓假、行、显了，如次依本三。}%

@{勝義諦亦三，謂義、得、正行，}@
%{胜义谛亦三，谓义、得、正行，}%

@{依本一無變，無倒二圓實。}@
%{依本一无变，无倒二圆实。}%

@{世極成依一，理極成依三。}@
%{世极成依一，理极成依三。}%

@{淨所行有二，依一圓成實。}@
%{净所行有二，依一圆成实。}%

@{名遍計所執，相分別依他，}@
%{名遍计所执，相分别依他，}%

@{真如及正智，圓成實所攝。}@
%{真如以及正智，圆成实所摄。}%

@{流轉與安立，邪行依初二，}@
%{流转与安立，邪行依初二，}%

@{實相唯識淨，正行依後一。}@
%{实相唯识净，正行依后一。}%

@{於蘊等我見，執、一、因、受者、}@
%{于蕴等我见，执、一、因、受者、}%

@{作者、自在轉、增上義及常、}@
%{作者、自在转、增上义以及常、}%

@{雜染清淨依、觀、縛解者性。}@
%{杂染清净依、观、缚解者性。}%

@{此所執分別，法性義在彼。}@
%{此所执分别，法性义在彼。}%

@{非一及總略、分段義名蘊。}@
%{非一以及总略、分段义名蕴。}%

@{能所取、彼取，種子義名界。}@
%{能所取、彼取，种子义名界。}%

@{能受、所了境，用門義名處。}@
%{能受、所了境，用门义名处。}%

@{緣起義於因、果、用無增減。}@
%{缘起义于因、果、用无增减。}%

@{於非愛、愛、淨、俱生及勝主、}@
%{于非爱、爱、净、俱生以及胜主、}%

@{得、行不自在，是處非處義。}@
%{得、行不自在，是处非处义。}%

@{根於取、住、續、用、二淨增上。}@
%{根于取、住、续、用、二净增上。}%

@{因果已未用，是世義應知。}@
%{因果已未用，是世义应知。}%

@{受及受資糧，彼所因諸行，}@
%{受以及受资粮，彼所因诸行，}%

@{二寂滅對治，是諦義應知。}@
%{二寂灭对治，是谛义应知。}%

@{由功德過失，及無分別智，}@
%{由功德过失，以及无分别智，}%

@{依他自出離，是乘義應知。}@
%{依他自出离，是乘义应知。}%

@{有為、無為義，謂若假、若因、}@
%{有为、无为义，谓若假、若因、}%

@{若相，若寂靜、若彼所觀義。}@
%{若相，若寂静、若彼所观义。}%
@{辯修對治品第四}@
%{辩修对治品第四}%

@{以麁重愛因，我事無迷故，}@
%{因为粗重的爱执，我的事没有迷惑，}%

@{為入四聖諦，修念住應知。}@
%{为了进入四圣谛，应该修念住。}%

@{已遍知障治，一切種差別，}@
%{已经遍知所有障碍的治法，所有种类的差别，}%

@{為遠離修集，勤修四正斷。}@
%{为了远离修集，勤修四正断。}%

@{依住堪能性，為一切事成，}@
%{依靠住着堪能性，为了所有事成就，}%

@{滅除五過失，勤修八斷行。}@
%{灭除五种过失，勤修八断行。}%

@{懈怠、忘聖言，及惛沈、掉舉、}@
%{懈怠，忘记圣言，以及昏沉、掉举，}%

@{不作行作行，是五失應知。}@

@{為斷除懈怠，修欲、勤、信、安，}@
%{为了断除懈怠，修习欲、勤、信、安，}%

@{即所依、能依，及所因、能果。}@
%{就是所依、能依，以及所因、能果。}%

@{為除餘四失，修念、智、思、捨，}@
%{为了除掉另外四种过失，修习念、智、思、舍，}%

@{記言覺沈掉，伏行滅等流。}@
%{记住圣言，觉知昏沉掉举，伏住行为，灭除等流。}%

@{已種順解脫，復修五增上，}@
%{已经种下顺解脱，再修习五种增上，}%

@{謂欲、行、不忘、不散亂、思擇。}@
%{就是欲、行、不忘、不散乱、思择。}%

@{即損障名力，因果立次第。}@
%{就是损害障碍的名力，因果建立次第。}%

@{順決擇二二，在五根、五力。}@
%{顺决择二十二，在五根、五力。}%

@{覺支略有五，謂所依、自性、}@
%{觉支略有五，就是所依、自性，}%

@{出離并利益，及三無染支。}@
%{出离并利益，以及三无染支。}%

@{由因緣所依，自性義差別，}@
%{由于因缘所依，自性的意义差别，}%

@{故輕安、定、捨，說為無染支。}@
%{所以轻安、定、舍，说为无染支。}%

@{分別及誨示，令他信有三，}@
%{分别以及教诲，使他相信有三种，}%

@{對治障亦三，故道支成八。}@
%{对治障碍也三种，所以道支成就八种。}%

@{表見戒遠離，令他深信受，}@
%{表明见戒远离，使他深信受，}%

@{對治本隨惑，及自在障故。}@
%{对治本随惑，以及自在障碍，}%

@{有倒順無倒，無倒有倒隨，}@
%{有倒顺无倒，无倒有倒随，}%

@{無倒無倒隨，是修治差別。}@
%{无倒无倒随，是修治的差别。}%

@{菩薩所修習，由所緣作意，}@
%{菩萨所修习，由所缘作意，}%

@{證得殊勝故，與二乘差別。}@
%{证得殊胜，所以与二乘有差别。}%

@{辯無上乘品第七}@
%{辩无上乘品第七}%

@{總由三無上，說為無上乘，}@
%{总由三种无上，说为无上乘，}%

@{謂正行、所緣，及修證無上。}@
%{就是正行、所缘，以及修证无上。}%

@{正行有六種，謂最勝、作意、}@
%{正行有六种，就是最胜、作意，}%

@{隨法、離二邊、差別、無差別。}@

@{最勝有十二，謂廣大、長時、}@
