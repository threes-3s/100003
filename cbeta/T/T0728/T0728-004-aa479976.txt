@{諸法集要經卷第四}@
%{诸法集要经卷第四}%

@{觀無畏尊者集}@
%{观无畏尊者集}%

@{西天譯經三藏朝散大夫試鴻臚卿宣梵大師賜紫沙門臣日稱等奉詔譯}@
%{西天译经三藏朝散大夫试鸿胪卿宣梵大师赐紫沙门臣日称等奉诏译}%

@{訶厭五欲品第七之餘}@
%{呵厌五欲品第七之余}%

@{欲境如夢事，亦猶尋香城，}@
%{欲望就像梦境，也像寻找香城一样，}%

@{猛熾若焰然，諸天由是墮。}@
%{燃烧得厉害，如同火焰一样，诸天因此堕落。}%

@{若於欲生愛，後則為所損，}@
%{如果对欲望产生爱，后来就会被它所损害，}%

@{曲戾無正思，諸天由是墮。}@
%{歪曲不正，没有正思，诸天因此堕落。}%

@{極下惡可厭，流注若河源，}@
%{极度低劣，令人厌恶，如同河流的源头一样，}%

@{譬之深險坑，諸天由是墮。}@
%{比喻成深险的坑，诸天因此堕落。}%

@{欲性本動搖，猶風浪水月，}@
%{欲望的本性本来就摇摆不定，就像风浪中的水月一样，}%

@{如蛇舌不停，諸天由是墮。}@
%{如同蛇的舌头不停地吐着，诸天因此堕落。}%

@{欲如飛電轉，亦如於陽焰，}@
%{欲望如同飞快的闪电，也如同阳光下的幻影一样，}%

@{如聚沫不堅，諸天由是墮。}@
%{如同泡沫一样不坚固，诸天因此堕落。}%

@{欲如迅河流，如象耳常動，}@
%{欲望如同迅猛的河流，如同大象的耳朵一样不停地晃动，}%

@{如芭蕉不實，諸天由是墮。}@
%{如同芭蕉一样不结果实，诸天因此堕落。}%

@{欲如彼幻事，如金播歌果，}@
%{欲望如同幻境，如同金色的播歌果一样，}%

@{如魚吞其鉤，諸天由是墮。}@
%{如同鱼儿吞食鱼钩一样，诸天因此堕落。}%

@{當以真實智，斷除於欲境，}@
%{应该用真实的智慧，断除对欲望的执着，}%

@{解脫不善果，及諸不饒益。}@
%{解脱不善的果报，以及一切不饶益的东西。}%

@{起妄想思惟，於欲生欣樂，}@
%{产生妄想和思惟，对欲望产生欢喜，}%

@{則為欲羂拘，壽命豈能久？}@
%{就会被欲望的网绳所束缚，寿命怎么可能长久呢？}%

@{眾生心輕動，咸為欲所牽，}@
%{众生心轻浮，容易被欲望所牵引，}%

@{愚癡無覺知，彼為自欺誑。}@
%{愚痴没有觉知，他们是在欺骗自己。}%

@{若為欲境動，則是諸苦本，}@
%{如果被欲望的境界所动摇，那就是一切痛苦的根源，}%

@{如乾闥婆城，當知不久住。}@
%{就像乾闼婆城一样，应该知道不会长久存在。}%

@{若於欲生貪，彼瞋則隨轉，}@
%{如果对欲望产生贪婪，那么瞋恨就会随之而来，}%

@{如是諸眾生，速趣於惡道。}@
%{这样，众生就会很快堕入恶道。}%

@{是故彼正士，捨欲除瞋恚，}@
%{所以，那些正士，舍弃欲望，消除瞋恨，}%

@{離癡等過失，顯發於明慧。}@
%{远离愚痴等过失，显发智慧的光明。}%

@{若人厭欲境，悟彼如深冤，}@
%{如果一个人厌恶欲望的境界，认识到它如同深重的冤仇一样，}%

@{以智為良朋，速證真常果。}@
%{以智慧为良友，就能快速证得永恒的果实。}%

@{於欲不生著，得離諸垢染，}@
%{对欲望不生执着，就能远离一切污垢和染污，}%

@{斯為具智人，諸天咸敬奉。}@
%{这就是具有智慧的人，诸天都会敬奉他。}%

@{善超欲淤泥，能與眾生樂，}@
%{能够超越欲望的泥潭，能给众生带来快乐，}%

@{心離縛寂靜，降伏諸魔軍。}@
%{内心远离束缚，寂静安宁，降伏一切魔军。}%

@{離愛品第八}@
%{离爱品第八}%

@{愛火起於心，迷者謂清涼，}@
%{爱的火焰从心中燃烧起来，迷惑的人以为它是清凉的，}%

@{猶勝地獄火，此乃遍三界。}@
%{却比地狱的火焰还要厉害，它遍布三界。}%

@{又彼地獄中，能生於劫火，}@
%{而且在地狱中，还会产生劫火，}%

@{極炎猛熾然，皆從愛所起。}@
%{极其猛烈地燃烧，都是从爱所产生的。}%

@{地獄苦眾生，業盡必當出，}@
%{地狱中的苦难众生，业障消除后，一定会离开，}%

@{三界諸有情，愛火無休息。}@
%{三界中的所有有情众生，爱的火焰永无止息。}%

@{由愛所縛故，輪迴無有窮，}@
%{由于被爱所束缚，轮回无有穷尽，}%

@{何況地獄中，更生於愛火？}@
%{何况在地狱中，还会再生爱的火焰呢？}%

@{又地獄業火，但能然彼身，}@
%{而且地狱的业火，只能燃烧身体，}%

@{愛火損眾生，燒心及其體；}@
%{爱的火焰却损害众生，燃烧他们的心和身体；}%

@{是二種差別，今當分別之，}@
%{这两种火焰有区别，现在就来分别一下，}%

@{獄火雖熾炎，愛火復過彼。}@
%{地狱的火焰虽然炽烈，但爱的火焰比它更厉害。}%

@{由三業所起，遍三有燒然，}@
%{由身口意三业所产生，遍布三界燃烧，}%

@{損害於善因，唯愛火為毒。}@
%{损害善因，只有爱的火焰是毒药。}%

@{貪火燒諸天，瞋火亦復爾，}@
%{贪婪的火焰燃烧诸天，瞋恨的火焰也是如此，}%

@{癡火逐愚夫，愛火隨所至；}@
%{愚痴的火焰追逐愚蠢的人，爱的火焰则无处不在；}%

@{嫉慢復如火，從我執薪生，}@
%{嫉妒和我慢如同火焰，从我执的柴薪中产生，}%

@{愛火燒世間，無薪常熾盛。}@
%{爱的火焰燃烧着世间，没有柴薪却永远燃烧。}%

@{於境生戀著，為愛蛇所傷，}@
%{对境界产生贪恋，就会被爱的毒蛇所伤害，}%

@{瞻視及承迎，展轉而增長。}@
%{瞻仰和迎合，不断地增长。}%

@{如火增其薪，相續而不絕，}@
%{如同火势遇到柴薪就会增强，不断地延续下去，}%

@{世火猶可防，愛火無能制。}@
%{世间的火焰还能防备，但爱的火焰无法控制。}%

@{若為愛欺罔，隨世間流轉，}@
%{如果被爱欺骗，随着世间轮回，}%

@{彼則如冤敵，無有能勝者。}@
%{它就像冤家仇敌一样，没有人能够战胜它。}%

@{由愛之所覆，趣海求諸珍，}@
%{由于被爱所蒙蔽，就到大海中寻找珍宝，}%

@{恐怖軍陣中，深入而鬪戰。}@
%{在恐怖的战场中，深入其中进行战斗。}%

@{王者愛其國，乃互相侵競，}@
%{国王爱护自己的国家，就互相争斗，}%

@{以至母及子，因財而起諍。}@
%{甚至母亲和儿子，因为财富而发生争执。}%

@{若解脫彼愛，捐捨諸珍玩，}@
%{如果能够解脱爱，舍弃珍宝和玩物，}%

@{視之如瓦礫，則近菩提道。}@
%{把它们看作瓦砾，就能接近菩提之道。}%

@{由愛於財故，墮地獄趣中，}@
%{由于贪爱财富，就堕入地狱道中，}%

@{彼熱惱難堪，是故當遠離。}@
%{那里的热恼难以忍受，所以应该远离。}%

@{當以其智水，沃之令永滅，}@
%{应该用智慧的水，浇灭它，让它永远消失，}%

@{若不除愛火，去菩提則遠。}@
%{如果不消除爱的火焰，就会远离菩提。}%

@{若遠離其愛，於諸珍不著，}@
%{如果远离爱，对珍宝不执着，}%

@{是人於世間，則無微少苦。}@
%{这个人世间就不会有丝毫的痛苦。}%

@{如網捕於魚，悉遺諸螺蜆，}@
%{如同网捕鱼，却把所有的螺和蚬都遗弃了，}%

@{愛縛彼眾生，無有能免者。}@
%{爱束缚着众生，没有一个人能够逃脱。}%

@{如鹿中毒箭，則四向馳走，}@
%{如同鹿中了毒箭，就四处奔走，}%

@{毒處處相隨，寧逃於苦惱。}@
%{毒药无处不在，怎么能够逃脱痛苦呢？}%

@{愛火亦如是，彼毒常隨逐，}@
%{爱的火焰也是如此，它的毒药永远跟随，}%

@{燒愚癡凡夫，何由能出離？}@
%{燃烧着愚痴的凡夫，怎么能够解脱呢？}%

@{暫生於適意，果報常燒然，}@
%{暂时产生快乐，但果报却永远燃烧，}%

@{求出世樂者，應當去其愛。}@
%{想要获得出世间的快乐，就应该舍弃爱。}%

@{如魚食其餌，彼必當趣死，}@
%{如同鱼儿吃食鱼饵，它一定会走向死亡，}%

@{人為愛所牽，中夭無疑惑。}@
%{人被爱所牵引，夭折是毫无疑问的。}%

@{墮鬼境界中，熱惱遍馳走，}@
%{堕入鬼道中，热恼遍布全身，}%

@{及地獄有情，多由心起愛。}@
%{以及地狱中的有情众生，大多都是因为心生爱而堕落的。}%

@{乃至向他門，求乞而活命，}@
%{甚至向别人乞讨，才能活命，}%

@{皆由愛使然，是佛之所說。}@
%{都是因为爱所致，这是佛陀所说。}%

@{愛火燒諸天，不須薪少許，}@
%{爱的火焰燃烧诸天，不需要一点柴薪，}%

@{由著於境界，從六根發起。}@
%{因为执着于境界，从六根发起的。}%

@{受具足快樂，常迷醉其心，}@
%{享受着完全的快乐，却永远迷醉自己的心，}%

@{墮落不覺知，由愛之所誤。}@
%{堕落而不自知，都是因为爱的错误。}%

@{一切輪迴因，皆從愛所得，}@
%{一切轮回的根源，都是从爱而来的，}%

@{愛鎖拘有情，令墮於惡趣。}@
%{爱的锁链束缚着有情众生，让他们堕入恶趣。}%

@{又天中愛火，欲境常圍繞，}@
%{而且天界中的爱的火焰，欲望的境界总是包围着，}%

@{由愚癡自在，燒彼著欲者。}@
%{由于愚痴放纵，燃烧着那些执着于欲望的人。}%

@{如火得乾薪，騰焰則彌盛，}@
%{如同火遇到干柴，火焰就会更加旺盛，}%

@{欲樂適其心，其愛轉增長。}@
%{欲望的快乐满足了他们的心，爱就会更加增长。}%

@{世火極炎猛，人皆能遠之，}@
%{世间的火焰极其猛烈，人们都能远离它，}%

@{愛火燒世間，無能免斯害。}@
%{但爱的火焰燃烧着世间，没有人能够免受它的伤害。}%

@{若人不生愛，得最上寂靜，}@
%{如果人不生爱，就能获得最高的寂静，}%

@{出過患稠林，能超於苦海。}@
%{走出过患的丛林，能够超越苦海。}%

@{離和合過失，則斷愛欲索，}@
%{远离和合的过失，就能断除爱的绳索，}%

@{解脫諸罪垢，乃是無憂者。}@
%{解脱一切罪垢，就是无忧无虑的人。}%

@{百千俱胝劫，常為愛所欺，}@
%{千千万万劫，一直被爱所欺骗，}%

@{愚夫不棄捨，為幻網維縶。}@
%{愚蠢的人不舍弃它，被幻网所束缚。}%

@{謂由愛所覆，樂親近承事，}@
%{以为因为爱的蒙蔽，喜欢亲近和侍奉，}%

@{如彼傭力人，渴飲其醎水；}@
%{就像那些雇佣劳力的人，口渴了就喝咸水；}%

@{飲已渴暫息，須臾喉復乾，}@
%{喝了之后，口渴暂时缓解，一会儿喉咙又干了，}%

@{渴愛在於心，非道何由止？}@
%{渴爱还在心中，不走正道，怎么能够停止呢？}%

@{是故當遠離，諸惡由之生，}@
%{所以应该远离它，一切恶事都是由它而生的，}%

@{為愛所伏者，沈淪無出期。}@
%{被爱所征服的人，就会沉沦其中，没有出头之日。}%

@{受天中最勝，上妙五欲樂，}@
%{享受着天界中最美好的五欲之乐，}%

@{終為愛索牽，復墮於惡趣。}@
%{最终还是被爱的绳索所牵引，再次堕入恶趣。}%

@{若人近於愛，苦惱常充滿，}@
%{如果一个人接近爱，痛苦就会永远充满，}%

@{依正教所明，此定非饒益。}@
%{根据正法教义，这肯定不是饶益。}%

@{若著愛境界，則無有厭足，}@
%{如果执着于爱的境界，就没有厌足的时候，}%

@{能棄彼愛者，是人無憂患。}@
%{能够舍弃爱的人，这个人就不会有忧患。}%

@{諸天由愛故，則生於放逸，}@
%{诸天因为爱，就产生放逸，}%

@{耽著復追求，後墮於地獄。}@
%{沉溺其中，不断追求，后来就堕入地狱。}%

@{諸天若退失，為第一慚恥，}@
%{诸天如果退失，是最大的耻辱，}%

@{由著上妙樂，則受極重苦。}@
%{因为执着于美好的快乐，就会受到极大的痛苦。}%

@{愛覆於自心，其心則狂亂，}@
%{爱蒙蔽了他们的心，他们的心就变得狂乱，}%

@{不怖於輪迴，長時而縱逸。}@
%{不害怕轮回，长期放纵。}%

@{眾生由欲樂，復增長其愛，}@
%{众生因为欲望的快乐，又增长了他们的爱，}%

@{愛火地獄火，為彼彼燒害。}@
%{爱的火焰和地狱的火焰，都燃烧着他们，伤害着他们。}%

@{彼愛若增長，展轉則無窮，}@
%{如果爱增长，就会不断地增长，没有穷尽，}%

@{已有生防護，未得常追求。}@
%{已经拥有了生命，却还不断地追求，}%

@{由起追求故，其心常不足，}@
%{因为不断地追求，他们的心永远都不满足，}%

@{是人無樂分，如來所印可。}@
%{这个人没有快乐的份，如来所印证。}%

@{由心無厭故，常思其欲味，}@
%{因为内心没有厌倦，就总是想着欲望的滋味，}%

@{則為彼愛火，相續而燒然。}@
%{就会被爱的火焰不断地燃烧。}%

@{若於欲生怖，離愛火所逼，}@
%{如果对欲望产生恐惧，远离爱的火焰的逼迫，}%

@{解脫愛垢染，不復墮惡道。}@
%{解脱爱的污垢和染污，就不会再堕入恶道。}%

@{由自邪思惟，起三毒塵坌，}@
%{由于自己邪恶的思惟，产生了贪嗔痴三毒的尘垢，}%

@{溺放逸深淵，常貪於女色，}@
%{沉溺于放逸的深渊，总是贪恋女色，}%

@{歌樂妙音聲，引生於散亂，}@
%{歌舞和美妙的声音，引起散乱，}%

@{其心無暫停，猶如於駃水。}@
%{他们的心没有片刻的停歇，如同奔流的河水一样。}%

@{愛如深險河，欲如漏舡舫，}@
%{爱如同深险的河流，欲望如同漏水的船只，}%

@{愚者之所乘，即為彼沈喪。}@
%{愚蠢的人乘坐它，就会沉沦其中。}%

@{愛如猛熾焰，三毒如乾薪，}@
%{爱如同猛烈的火焰，三毒如同干柴，}%

@{放逸如迅風，燒諸天不覺。}@
%{放逸如同迅猛的风，燃烧着诸天，他们却不自知。}%

@{諸天著欲樂，則為愛降伏，}@
%{诸天执着于欲望的快乐，就被爱所征服，}%

@{於頃刻須臾，暫無其少暇。}@
%{在片刻之间，都没有片刻的空闲。}%

@{愛為欲所依，生百千障礙，}@
%{爱依附于欲望，产生千千万万的障碍，}%

@{諸天著樂故，於善不能作。}@
%{诸天执着于快乐，就不能做善事。}%

@{愛蛇有五首，其性極暴惡，}@
%{爱的毒蛇有五个头，它的本性极其凶恶，}%

@{螫彼貪欲人，是苦難堪忍。}@

@{於彼彼快樂，後則皆散壞。}@
%{他们以为快乐，但最终都会散坏。}%

@{眾生酒所迷，其心常醉亂，}@
%{众生被酒迷住，他们的心常醉乱。}%

@{為彼癡所牽，耽著其美味。}@
%{被愚痴牵引，沉迷于美味。}%

@{當知酒如繩，癡愛常難解，}@
%{要知道，酒就像一根绳索，愚痴的爱难以解脱。}%

@{寧墮地獄中，於酒不應觸。}@
%{宁愿堕入地狱，也不应该触碰酒。}%

@{因觸聞其香，癡人即樂飲，}@
%{因为接触到酒香，愚痴的人就喜欢喝酒。}%

@{是故於彼酒，見已當捨去。}@
%{所以，见到酒就应该舍弃。}%

@{若見即生貪，若觸香即發，}@
%{看到酒就生贪心，接触到酒香就发作。}%

@{由聞彼香故，其心不能止。}@
%{因为闻到酒香，心就无法止住。}%

@{是故酒為毒，生過失非一，}@
%{所以，酒是毒药，产生的过失不止一个。}%

@{壞色力名聞，皆因飲彼酒。}@
%{损害色力名闻，都是因为喝酒。}%

@{口出於狂言，瞪目無定往，}@
%{口中说出狂言，瞪着眼睛，没有定所。}%

@{時臥不覺知，所作皆廢忘。}@
%{有时躺下就不知道，所做的事情都忘记了。}%

@{由偃仆於地，為女人所笑，}@
%{因为倒在地上，被女人嘲笑。}%

@{其身不動轉，如枯木相似。}@
%{身体不能转动，就像枯木一样。}%

@{彼醉酒而臥，瞥見謂其死，}@
%{他喝醉酒躺下，瞥见就以为他死了。}%

@{知者咸告言，由飲酒如是。}@
%{有智慧的人就告诉他，这是因为喝酒。}%

@{常樂飲酒者，住三十六失，}@
%{常喜欢喝酒的人，会犯三十六种过失。}%

@{當了知彼過，此則常安隱。}@
%{应该了解这些过失，这样才能常得安稳。}%

@{具勝族名稱，由酒之所污，}@
%{拥有高贵家族名称的人，因为酒而被玷污。}%

@{是人如蘆花，不久自輕棄。}@
%{这个人就像芦苇花，不久就会被抛弃。}%

@{若人樂飲酒，展轉為境牽，}@
%{如果人喜欢喝酒，就会被境界牵引。}%

@{墮放逸水中，漂流難出離。}@
%{堕入放逸的水中，漂流难以出离。}%

@{為境所牽故，不知善不善，}@
%{因为被境界牵引，不知道善恶。}%

@{於清勝園林，何用復飲酒？}@
%{在清净美好的园林里，为什么还要喝酒呢？}%

@{若樂於酒味，則生諸險難，}@
%{如果喜欢酒的味道，就会产生各种危险。}%

@{墮於地獄中，具受諸苦惱。}@
%{堕入地狱，承受各种痛苦。}%

@{飲已發生癡，由癡造眾罪，}@
%{喝了酒就产生愚痴，因为愚痴而造下各种罪业。}%

@{愚人心愛樂，何能生遠離？}@
%{愚痴的人喜欢它，怎么会生起远离之心呢？}%

@{起增上耽著，受極重苦報，}@
%{产生增上贪着，承受极重的苦报。}%

@{若能離彼過，則無諸憂惱。}@
%{如果能远离这些过失，就没有各种忧恼。}%

@{初則損其慧，後則壞其樂，}@
%{开始会损害智慧，后来会破坏快乐。}%

@{是故彼智人，於酒常厭捨。}@
%{所以，智者常厌舍酒。}%

@{若人近於酒，彼則如飛鳶，}@
%{如果人靠近酒，就像飞鸢一样。}%

@{常為癡所盲，故說酒為毒。}@
%{常被愚痴所蒙蔽，所以说酒是毒药。}%

@{於酒作毒想，最上第一樂，}@
%{对酒产生毒药的念头，是最高的第一快乐。}%

@{由持淨戒故，寧飲於銅汁。}@
%{因为持守清净戒律，宁愿喝铜汁。}%

@{若樂飲酒者，於罪則不免，}@
%{如果喜欢喝酒，就无法免除罪业。}%

@{彼增上愚癡，常處於惡道。}@
%{他增上愚痴，常处在恶道。}%

@{飲酒雖一罪，能生一切惡，}@
%{喝酒虽然是一个罪过，却能生起一切恶业。}%

@{是故當制之，心戒則為本。}@
%{所以应该制止它，心戒是根本。}%

@{比丘樂飲酒，則捨阿蘭若，}@
%{比丘喜欢喝酒，就会舍弃阿兰若。}%

@{離心一境性，不思惟正法。}@
%{远离心一境性，不思惟正法。}%

@{由樂飲酒故，心常生熱惱，}@
%{因为喜欢喝酒，心常生烦恼。}%

@{習近於非法，壞二世善利。}@
%{习近非法，破坏两世的善利。}%

@{無威儀道行，廢說法是分，}@
%{没有威仪道行，废弃说法是份。}%

@{與言行相違，空說有何益？}@
%{与言行相违背，空说有什么益处呢？}%

@{自不能達解，何由悟於他？}@
%{自己都不能理解，怎么才能使别人明白呢？}%

@{發麁獷言詞，此非善說法，}@
%{说出粗俗的话语，这不是善说法。}%

@{違背於正理，識者咸譏誚，}@
%{违背正理，有智慧的人都会嘲笑。}%

@{貧弊人所輕，皆由飲於酒。}@
%{贫穷的人都会轻视他，都是因为喝酒。}%

@{過去無憶念，現在復忘失，}@
%{过去没有记忆，现在又忘记了。}%

@{未來何所知？由酒迷三世。}@
%{未来又怎么知道呢？因为酒迷了三世。}%

@{失名稱威德，令心常馳散，}@
%{失去名声威德，让心常散乱。}%

@{引生諸過咎，斯為酒所困。}@
%{招致各种过失，这是被酒困住了。}%

@{若遠離酒者，具戒定清淨，}@
%{如果远离酒，具足戒定清净。}%

@{住最上安隱，得至不滅處。}@
%{住于最高安稳，到达不灭之处。}%

@{諸法集要經卷第四}@
%{诸法集要经卷第四}%
