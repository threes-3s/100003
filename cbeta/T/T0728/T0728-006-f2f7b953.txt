@{諸法集要經卷第六}@
%{诸法集要经卷第六}%

@{觀無畏尊者集}@
%{观无畏尊者集}%

@{西天譯經三藏朝散大夫試鴻臚卿宣梵大師賜紫沙門臣日稱等奉詔譯}@
%{西天译经三藏朝散大夫试鸿胪卿宣梵大师赐紫沙门臣日称等奉诏译}%

@{福非福業品第十三}@
%{福非福业品第十三}%

@{所造作諸業，謂福及非福，}@
%{所造作诸业，谓福及非福，}%
%{你所做的所有业，分为福业和非福业，}%

@{能縛諸有情，定各招其報。}@
%{能缚诸有情，定各招其报。}%
%{这些业能束缚所有有情众生，必定会招来相应的报应。}%

@{愚夫心如魚，依愛波而住，}@
%{愚夫心如鱼，依爱波而住，}%
%{愚蠢的人，就像鱼儿一样，沉迷于爱欲的波浪中，}%

@{含笑造諸惡，悲啼而自受。}@
%{含笑造诸恶，悲啼而自受。}%
%{他们一边笑着做恶事，一边又悲痛地承受着恶报。}%

@{昔同造諸罪，謂僕使營從，}@
%{昔同造诸罪，谓仆使营从，}%
%{过去，你曾经和仆人、使者、营帐的跟随者一起犯过罪，}%

@{後受其苦報，彼則無相代。}@
%{后受其苦报，彼则无相代。}%
%{现在你承受着苦报，他们却无法替你承受。}%

@{由親眷朋屬，和合造眾罪，}@
%{由亲眷朋属，和合造众罪，}%
%{因为亲人、朋友和属下一起犯了罪，}%

@{於他世相隨，唯所作惡業。}@
%{於他世相随，唯所作恶业。}%
%{在来世，他们会跟随你，只有你所做的恶业相随。}%

@{如花所至處，其香不捨離，}@
%{如花所至处，其香不舍离，}%
%{就像花朵所到之处，香味不会离开一样，}%

@{善惡業亦然，在處常隨逐。}@
%{善恶业亦然，在处常随逐。}%
%{善恶业也是如此，无论你在哪里，它们都会跟随你。}%

@{眾生由自業，因果常相應，}@
%{众生由自业，因果常相應，}%
%{众生因为自己的业，因果总是相对应的，}%

@{作善生諸天，而受殊勝樂。}@
%{作善生诸天，而受殊胜乐。}%
%{做善事，就会生到天界，享受殊胜的快乐。}%

@{若惡業果報，則受極重苦，}@
%{若恶业果报，则受极重苦，}%
%{如果作恶，就会承受极重的苦报，}%

@{墮三惡趣中，彼苦無相似。}@
%{堕三恶趣中，彼苦无相似。}%
%{堕入三恶道，那里的苦楚是无法比拟的。}%

@{謂由彼三業，造作遍三界，}@
%{谓由彼三业，造作遍三界，}%
%{因为身、口、意三业，造作遍布三界，}%

@{常起於三毒，則墮三惡道。}@
%{常起于三毒，则堕三恶道。}%
%{经常起贪、嗔、痴三毒，就会堕入三恶道。}%

@{諸愚夫異生，由因緣和合，}@
%{诸愚夫异生，由因缘和合，}%
%{愚蠢的人，因为各种因缘的和合，}%

@{流轉三界中，皆隨於自業；}@
%{流转三界中，皆随于自业；}%
%{在三界中轮回，都跟随自己的业力；}%

@{非自作他受，非他作我受，}@
%{非自作他受，非他作我受，}%
%{不是自己做，别人受，也不是别人做，我受，}%

@{當知所造業，招報唯決定。}@
%{当知所造业，招报唯决定。}%
%{应该知道，你所造的业，招来的报应是绝对的。}%

@{業雖有眾多，受處有其九，}@
%{业虽有众多，受处有其九，}%
%{业虽然有很多种，但受报的地方只有九种，}%

@{由彼互相資，成四十種惡。}@
%{由彼互相资，成四十种恶。}%
%{这些业互相资助，形成了四十种恶业。}%

@{自造作一業，定受其一報，}@
%{自造作一业，定受其一报，}%
%{自己造作一种业，就一定会承受一种报应，}%

@{墮於險道中，則無其伴侶。}@
%{堕于险道中，则无其伴侣。}%
%{堕入险道，就没有人陪伴你。}%

@{或為他勸請，而造作惡業，}@
%{或为他劝请，而造作恶业，}%
%{有时，别人劝你做恶事，你就做了，}%

@{後受苦報時，彼則不能救。}@
%{后受苦报时，彼则不能救。}%
%{等到你承受苦报的时候，他们就救不了你了。}%

@{業熟非初後，及此生他世，}@
%{业熟非初后，及此生他世，}%
%{业成熟的时间，不一定是现在，也不一定是将来，不一定是今生，也不一定是来世，}%

@{謂於此造作，或於餘處受。}@
%{谓于此造作，或于余处受。}%
%{你可能是在这里造业，却在别的地方受报。}%

@{由善惡業故，隨輪迴流轉，}@
%{由善恶业故，随轮回流转，}%
%{因为善恶业，你就会随着轮回不断流转，}%

@{為業風所吹，而招苦樂報。}@
%{为业风所吹，而招苦乐报。}%
%{就像被风吹着，招来苦乐的报应。}%

@{愚夫心散亂，於欲常樂著，}@
%{愚夫心散乱，于欲常乐著，}%
%{愚蠢的人，心散乱，总是沉迷于欲望的快乐中，}%

@{無正慧揀擇，諸惡則增長。}@
%{无正慧拣择，诸恶则增长。}%
%{没有正确的智慧去辨别，恶业就会增长。}%

@{彼著樂眾生，為癡之所覆，}@
%{彼著乐众生，为痴之所覆，}%
%{那些执着于快乐的人，被愚痴所蒙蔽，}%

@{惡報現其前，則墮黑暗處。}@
%{恶报现其前，则堕黑暗处。}%
%{恶报出现在他们面前，他们就会堕入黑暗之中。}%

@{由於佛正法，心不生欣樂，}@
%{由于佛正法，心不生欣乐，}%
%{因为佛陀的正法，内心没有生起欢喜，}%

@{在彼地獄中，長時受其苦。}@
%{在彼地狱中，长时受其苦。}%
%{他们就会堕入地狱，长时间受苦。}%

@{從無始輪迴，為業網纏縛，}@
%{从无始轮回，为业网缠缚，}%
%{从无始以来，就被业网缠缚着，}%

@{此滅彼復生，皆由心造作；}@
%{此灭彼复生，皆由心造作；}%
%{这个灭了，那个又生，都是由心造作的；}%

@{或從天墮落，或地獄生天，}@
%{或从天堕落，或地狱生天，}%
%{有时从天界堕落，有时从地狱升到天界，}%

@{或生於人中，或受餓鬼報；}@
%{或生于人中，或受饿鬼报；}%
%{有时生为人，有时受饿鬼报；}%

@{謂彼苦樂因，皆由己所造，}@
%{谓彼苦乐因，皆由己所造，}%
%{应该知道，这些苦乐的因，都是自己所造的，}%

@{各互相生起，非自在天作。}@
%{各互相生起，非自在天作。}%
%{不是由天帝所造成的。}%

@{輪迴生死中，造無數惡業，}@
%{轮回生死中，造无数恶业，}%
%{在轮回生死中，造了无数恶业，}%

@{唯佛當證知，餘智不能了。}@
%{唯佛当证知，余智不能了。}%
%{只有佛陀才能证知，其他的智慧都无法了解。}%

@{若非法招善，此因為顛倒，}@
%{若非法招善，此因为颠倒，}%
%{如果用不正当的方法来求取善报，这是颠倒的，}%

@{當知所受果，皆與因相似。}@
%{当知所受果，皆与因相似。}%
%{应该知道，你所受的果报，都会和因相似的。}%

@{若因果相應，則順於正理，}@
%{若因果相應，则顺于正理，}%
%{如果因果相对应，那就符合道理，}%

@{是有為諸法，無不從緣起。}@
%{是有为诸法，无不从缘起。}%
%{所有有为法，都无不从缘起而生。}%

@{未見無罪者，而趣於地獄，}@
%{未见无罪者，而趣于地狱，}%
%{没有见过无罪的人，却堕入地狱，}%

@{定由惡業故，則受其苦報。}@
%{定由恶业故，则受其苦报。}%
%{一定是由于恶业，才会承受苦报。}%

@{決定造諸惡，堅著而無悔，}@
%{决定造诸恶，坚著而无悔，}%
%{坚决地做恶事，执着不悔，}%

@{彼為業所縛，則墮於惡道。}@
%{彼为业所缚，则堕于恶道。}%
%{他们就被业力所束缚，就会堕入恶道。}%

@{未見不善業，引生於樂果，}@
%{未见不善业，引生于乐果，}%
%{没有见过不善业，能带来快乐的果报，}%

@{唯佛真實言，示彼對治道。}@
%{唯佛真实言，示彼对治道。}%
%{只有佛陀的真实言，才能指明对治的方法。}%

@{如因燈有光，如由業招報，}@
%{如因灯有光，如由业招报，}%
%{就像灯有光是因为有灯芯一样，就像业招来报应一样，}%

@{諸有所作者，皆因緣生故。}@
%{诸有所作者，皆因缘生故。}%
%{所有发生的事情，都是由因缘而生的。}%

@{謂由彼彼因，各各果隨轉，}@
%{谓由彼彼因，各各果随转，}%
%{因为有因，就会有果，果随着因而转，}%

@{善達如是相，則名真實見。}@
%{善达如是相，则名真实见。}%
%{能够明白这种道理，就叫做真实见。}%

@{非同自在天，無因而建立，}@
%{非同自在天，无因而建立，}%
%{不像天帝，没有因就能建立，}%

@{諸法皆緣生，是如來所說。}@
%{诸法皆缘生，是如来所说。}%
%{所有法都是缘起而生，这是如来所说。}%

@{由無始輪迴，業報常相似，}@
%{由无始轮回，业报常相似，}%
%{从无始以来，轮回的业报总是相似的，}%

@{非顛倒分別，從因緣而有。}@
%{非颠倒分别，从因缘而有。}%
%{不是颠倒的分别，而是从因缘而有的。}%

@{眾生癡所迷，於愛欲無厭，}@
%{众生痴所迷，于爱欲无厌，}%
%{众生被愚痴所迷惑，对爱欲没有厌倦，}%

@{若不了業報，何由獲寂靜？}@
%{若不了业报，何由获寂静？}%
%{如果不明白业报，怎么能获得寂静呢？}%

@{若人於佛教，不達道非道，}@
%{若人于佛教，不达道非道，}%
%{如果有人学习佛教，却不明白什么是正道，什么是邪道，}%

@{由癡無正慧，常生於熱惱。}@
%{由痴无正慧，常生于热恼。}%
%{因为愚痴，没有正确的智慧，就会经常生起烦恼。}%

@{見他如意樂，彼樂從因生，}@
%{见他如意乐，彼乐从因生，}%
%{看到别人如意快乐，那是因为他们种下了善因，}%

@{諸法皆唯心，各各隨自行。}@
%{诸法皆唯心，各各随自行。}%
%{所有法都是唯心所现，每个人都会随着自己的行为而得到结果。}%

@{有為皆無常，如水泡非久，}@
%{有为皆无常，如水泡非久，}%
%{所有有为法都是无常的，就像水泡一样，不能长久，}%

@{應當行善行，為二世饒益。}@
%{应当行善行，为二世饶益。}%
%{应该行善，为了今生和来世都得到利益。}%

@{覩世間業報，及諸天退墮，}@
%{覩世间业报，及诸天退堕，}%
%{看到世间众生受业报，看到天人堕落，}%

@{若樂放逸者，彼定無少樂。}@
%{若乐放逸者，彼定无少乐。}%
%{如果喜欢放逸，他们一定不会得到快乐。}%

@{業索極修長，堅固而難脫，}@
%{业索极修长，坚固而难脱，}%
%{业力就像一根非常长的绳索，非常坚固，很难摆脱，}%

@{纏縛彼愚夫，去菩提則遠。}@
%{缠缚彼愚夫，去菩提则远。}%
%{缠缚着愚蠢的人，让他们离菩提越来越远。}%

@{智慧如利劍，於彼能除斷，}@
%{智慧如利剑，于彼能除断，}%
%{智慧就像一把利剑，能够斩断这些业力，}%

@{離愚癡熱惱，令至於彼岸。}@
%{离愚痴热恼，令至于彼岸。}%
%{远离愚痴的烦恼，使他们到达彼岸。}%

@{由業受彼果，隨善惡相應，}@
%{由业受彼果，随善恶相應，}%
%{因为业力，就会得到相应的果报，无论善恶，}%

@{智者不暫忘，因果常決定。}@
%{智者不暂忘，因果常决定。}%
%{智者不会忘记，因果是绝对的。}%

@{由因緣和合，生肢分骨鎖，}@
%{由因缘和合，生肢分骨锁，}%
%{因为各种因缘的和合，就形成了身体的骨骼，就像锁链一样，}%

@{纏縛諸有情，輪迴無解脫。}@
%{缠缚诸有情，轮回无解脱。}%
%{缠缚着所有有情众生，让他们在轮回中无法解脱。}%

@{由彼纏縛故，逼迫難堪任，}@
%{由彼缠缚故，逼迫难堪任，}%
%{因为这种缠缚，让人感到逼迫，难以承受，}%

@{當修解脫因，得盡諸苦際。}@
%{当修解脱因，得尽诸苦际。}%
%{应该修习解脱的因，才能彻底断除所有的苦难。}%

@{彼業善鉤名，復能牽眾生，}@
%{彼业善钩名，复能牵众生，}%
%{业力就像钩子一样，能牵引众生，}%

@{於在所生處，隨業而受報。}@
%{于在所生处，随业而受报。}%
%{在他们所生的地方，就随着业力而受报。}%

@{彼業果如輪，於三有旋轉，}@
%{彼业果如轮，于三有旋转，}%
%{业力的果报就像车轮一样，在三界中不停地转动，}%

@{當離諸過患，常修殊勝行。}@
%{当离诸过患，常修殊胜行。}%
%{应该远离所有的过患，经常修习殊胜的法行。}%

@{布施如淨器，貯戒勤慧水，}@
%{布施如净器，贮戒勤慧水，}%
%{布施就像一个干净的容器，里面装满了戒律、精进和智慧的水，}%

@{智者善持用，滅三有業火。}@
%{智者善持用，灭三有业火。}%
%{智者善于运用，就能灭掉三界的业火。}%

@{若縱彼三業，三毒則隨轉，}@
%{若纵彼三业，三毒则随转，}%
%{如果你放纵身、口、意三业，三毒就会跟着转动，}%

@{馳騁三界中，由癡三種行。}@
%{驰骋三界中，由痴三种行。}%
%{在三界中奔跑，都是由愚痴的三种行为造成的。}%

@{一切諸眾生，為苦所逼迫，}@
%{一切诸众生，为苦所逼迫，}%
%{所有众生，都被苦难所逼迫，}%

@{皆隨自作業，常依止而住。}@
%{皆随自作业，常依止而住。}%
%{都随着自己的业力而生活。}%

@{若無彼善因，何能有少樂？}@
%{若无彼善因，何能有少乐？}%
%{如果没有善因，怎么能得到一点快乐呢？}%

@{隨業受彼報，如種生其果。}@
%{随业受彼报，如种生其果。}%
%{随着业力，就会得到相应的报应，就像种子会结出果实一样。}%

@{又如陽春時，能滋榮卉木，}@
%{又如阳春时，能滋荣卉木，}%
%{就像春天的时候，能滋养草木一样，}%

@{彼果從因生，無因則不起。}@
%{彼果从因生，无因则不起。}%
%{果实是从因缘而生的，没有因缘，就不会生起。}%

@{為業索所拘，百千生往返，}@
%{为业索所拘，百千生往返，}%
%{被业力所束缚，在百千生中轮回往返，}%

@{如世間車輪，由機關而轉。}@
%{如世间车轮，由机关而转。}%
%{就像世间的车轮，由机关控制着转动。}%

@{彼三毒堅牢，眾生難出離，}@
%{彼三毒坚牢，众生难出离，}%
%{贪、嗔、痴三毒非常坚固，众生很难摆脱，}%

@{離貪等過患，則善超三有。}@
%{离贪等过患，则善超三有。}%
%{远离贪等过患，才能超越三界。}%

@{若人慶快心，修彼殊勝行，}@
%{若人庆快心，修彼殊胜行，}%
%{如果有人欢喜快乐地修习殊胜的法行，}%

@{以是因緣故，受莊嚴勝報。}@
%{以是因缘故，受庄严胜报。}%
%{因为这个因缘，就会得到庄严殊胜的报应。}%

@{業如彼畫師，善圖諸形像，}@
%{业如彼画师，善图诸形像，}%
%{业力就像画家一样，善于描绘各种形象，}%

@{或天上人間，所畫無不盡。}@
%{或天上人间，所画无不尽。}%
%{无论是天界还是人间，他都能画得淋漓尽致。}%

@{彼畫無數量，皆由業變化，}@
%{彼画无数量，皆由业变化，}%
%{这些画的数量无穷无尽，都是由业力变化而成的，}%

@{不施眾彩飾，亦無能見者。}@
%{不施众彩饰，亦无能见者。}%
%{没有用各种颜色装饰，也无法看见。}%

%{在世间不停地旋转，而这一切，都是由我们的心念所驱使的。}%

@{由心造善業，  引生於天中，}@
%{如果我们用心造作善业，就会被引导到天界，}%

@{為境界所迷，  不思惟後苦。}@
%{但如果沉迷于天界的快乐境界，不去思考未来的痛苦，}%

@{於樂及非樂，  當審慮而行，}@
%{那么最终还是会堕落。因此，我们对于快乐和痛苦，都应当仔细思量，谨慎取舍，}%

@{苦樂業雖殊，  皆從因緣起。}@
%{要知道苦乐虽然不同，但都是由因缘和合而生，并非永恒不变的。}%

@{世間無有樂，  皆為業所牽，}@
%{世间没有真正的快乐，一切都是被业力所牵引，}%

@{樂壞苦現前，  由心而造作。}@
%{快乐过后，痛苦就会随之而来，而这一切都是由我们自己的心念造作而成的。}%

@{眾生由業風，  吹至所生處，}@
%{众生就像是被业风吹动，飘荡到不同的生命形态中，}%

@{於彼生愛樂，  則為業所拘。}@
%{如果对所生的环境产生贪恋，就会被业力牢牢地束缚住。}%

@{唯善不善業，  後世常相逐，}@
%{只有善业和恶业，才会在未来世一直伴随着我们，}%

@{猶如採其花，  彼香則隨至。}@
%{就像我们采摘花朵，花香也会随之而来，如影随形。}%

@{眾生自業使，  隨生滅流轉，}@
%{众生被自己的业力驱使着，随着生灭之法不断流转，}%

@{譬若彼鞦韆，  昇墜無休息。}@
%{就像秋千一样，上下起伏，永无止境。}%

@{於天人脩羅，  六趣而往返，}@
%{我们在天、人、阿修罗等六道中不断地轮回，}%

@{為癡之所覆，  不生真實見。}@
%{就是因为被愚痴所蒙蔽，无法生起真实的智慧。}%

@{又如世間輪，  依手而旋轉，}@
%{轮回也像世间车轮一样，需要依靠外力才能转动，}%

@{彼為業所催，  速疾無與等。}@
%{而这股推动力量就是业力，它快速迅猛，无与伦比。}%

@{由業之所纏，  十二支和合，}@
%{正是由于被业力所缠缚，十二因缘支才会环环相扣，}%

@{是名緣生輪，  世間無知者。}@
%{形成这轮回不息的缘起之轮，而世间的愚痴之人却对此一无所知。}%

@{諸天癡所覆，  常著於欲境，}@
%{即使是天上的神仙，也被愚痴所蒙蔽，总是执着于五欲的享受，}%

@{唯業果長存，  彼樂無積聚。}@
%{却不知道快乐是无常的，只有业果才会长久存在。}%

@{彼不知善業，  如良藥明燈，}@
%{他们不知道，善业就像良药和明灯，}%

@{除暗獲輕安，  能為作歸救。}@
%{能够驱除黑暗，获得光明和安乐，最终引领我们走向解脱。}%

@{受難堪極苦，  及種種怖畏，}@
%{而恶业却会让我们遭受难以忍受的痛苦，以及各种各样的恐惧，}%

@{是業有大力，  而不生疲勞。}@
%{业力的力量无比强大，它不会感到疲倦，会一直推动着我们承受果报。}%

@{天滅生人中，  人歿墮地獄，}@
%{天人福报享尽，就会堕落到人间；人死后如果造作恶业，就会堕入地狱，}%

@{獄出作傍生，  復墮於鬼趣。}@
%{从地狱出来后，可能还会投胎为畜生，或者堕入饿鬼道，}%

@{皆由彼業風，  飄轉而無定，}@
%{这一切都是被业风吹动，飘忽不定，没有丝毫自主权。}%

@{彼愚癡眾生，  未甞生覺悟。}@
%{而那些愚痴的众生，却始终没有觉醒，依然在轮回中迷茫度日。}%

@{眾生乘業車，  能行於三界，}@
%{众生乘坐着用业力打造的车子，在欲界、色界、无色界三界中奔驰，}%

@{餘乘則不然，  速疾無相似。}@
%{没有任何其他的车辆，能够比它更快，更便捷。}%

@{若所作清淨，  則受其福報，}@
%{如果我们所作所为清净无染，就会获得相应的福报，}%

@{唯於彼現生，  則知其自業。}@
%{而且在今生今世就能感受到自己善业带来的快乐。}%

@{彼業如彩繪，  皆從心所起，}@
%{这业力就像我们亲手绘制的图画，一切都是由我们自己的心念所创造，}%

@{所畫無不周，  長時而不滅。}@
%{它描绘得细致入微，而且能够长时间地保存，不会轻易消失。}%

@{謂廣大福報，  皆從業所生，}@
%{我们所说的广大福报，都是来自于善业的积累，}%

@{福業若盡時，  彼樂則散壞；}@
%{但如果福报的业力耗尽，快乐也会随之消失；}%

@{於善若不廢，  彼樂則增長，}@
%{反之，如果我们坚持行善，不曾懈怠，快乐就会不断增长，}%

@{是故於善因，  展轉常修作。}@
%{所以我们应当努力种下善因，并且不断地去培育它，让它开花结果。}%

@{於彼百千生，  受形軀骨鎖，}@
%{在漫长的轮回中，我们经历了无数次的生死，承受着各种各样的痛苦，}%

@{為業之所纏，  曾無安樂想。}@
%{被业力紧紧地束缚着，几乎忘记了安乐是什么滋味。}%

@{若造種種因，  則受種種報，}@
%{如果我们造作各种各样的因，就会承受各种各样的果报，}%

@{當於此生中，  勤修諸善行。}@
%{所以应当趁着今生，抓紧时间勤修各种善行，为未来积累福德资粮。}%

@{業畫極工巧，  皆依心造作，}@
%{业力就像一位技艺高超的画师，它根据我们内心的一思一念，描绘出不同的生命图景，}%

@{業盡果則亡，  剎那不久住。}@
%{当业力耗尽时，果报也会随之消失，就像电光石火，转瞬即逝。}%

@{唯自業為親，  於他何所得？}@
%{只有我们自己造作的业，才是最亲近、最可靠的，依靠别人是无法获得真正的解脱。}%

@{善調伏其心，  如理而安住。}@
%{所以，我们应当努力调伏自己的心念，让它安住在正法之中，}%

@{以業自莊嚴，  則非餘所作，}@
%{用善业来庄严自己，而不是依靠外在的装饰。}%

@{於百生千生，  而未曾暫捨。}@
%{这份善业的功德，将会伴随着我们生生世世，永远不会消失。}%

@{若了知生滅，  及真實因果，}@
%{如果我们能够真正明白世间万物的生灭变化，以及因果法则的真实不虚，}%

@{則離諸罪垢，  得至不滅處。}@
%{就能够彻底断除一切烦恼和罪垢，最终到达永恒不灭的解脱彼岸。}%

@{所造作諸業，  迂曲常相隨，}@
%{我们过去所造作的各种业，无论善恶，都会像影子一样紧紧跟随，不会消失，}%

@{如輻依彼輪，  於世間旋轉。}@
%{就像车轮的辐条，依附着车轮，在世间不断地旋转。}%

@{當以慧揀擇，  如理而修作，}@
%{我们应当用智慧仔细地分辨善恶，选择正确的道路，如理如法地修行，}%

@{是為調御師，  永脫諸煩惱。}@
%{这才是真正的调伏大师，能够帮助我们永远摆脱一切烦恼，获得究竟的解脱。}%

@{教示眾生品第十四}@
%{教示众生品第十四}%

@{謂貪恚癡垢，及老病死苦，}@
%{说贪婪、嗔恨、愚痴的污垢，以及衰老、疾病、死亡的痛苦，}%

@{此六如深冤，能損諸含識。}@
%{这六种就像深深的冤仇，能伤害一切有情众生。}%

@{又五境如賊，能劫功德財，}@
%{还有五种境界就像强盗，能劫夺功德和财富，}%

@{初如彼親朋，後則為冤害。}@
%{一开始就像亲朋好友，后来就变成冤仇。}%

@{由心生放逸，於欲境囂馳，}@
%{由于内心放逸，在欲望的境界里狂奔，}%

@{能令諸眾生，趣地獄餓鬼。}@
%{就能让众生走向地狱、饿鬼道。}%

@{貪為其熾火，瞋則如彼冤，}@
%{贪婪就像熊熊烈火，嗔恨就像冤仇，}%

@{黑暗說為癡，是三皆可畏。}@
%{黑暗就叫做愚痴，这三种都可怕。}%

@{謂三十六業，及彼四十行，}@
%{说三十六种业，以及四十种行为，}%

@{九十八煩惱，周遍於三界。}@
%{九十八种烦恼，遍布三界。}%

@{離十二因緣，一百八煩惱，}@
%{离开十二因缘，一百零八种烦恼，}%

@{善解法非法，常獲無量樂；}@
%{善于理解法与非法，就能常得无量快乐；}%

@{於十六現觀，及彼十六空，}@
%{对十六种现观，以及十六种空性，}%

@{了我法二相，是名為智者。}@
%{明白了法和我的两种相，这叫做智慧者。}%

@{善達道非道，及彼四究竟，}@
%{善于辨别道与非道，以及四种究竟，}%

@{解脫四瀑流，能滅諸罪垢。}@
%{解脱四种瀑流，就能灭除一切罪垢。}%

@{修習八聖道，出二種生死，}@
%{修习八正道，就能脱离两种生死，}%

@{顯現彼十力，得證菩提果。}@
%{显现十种力量，就能证得菩提果。}%

@{明真俗二諦，及彼四念處，}@
%{明白真谛和俗谛，以及四种念处，}%

@{除三際無知，不為魔所伏。}@
%{就能消除三世无明，不被魔所降伏。}%

@{是五欲境界，初甘後則苦，}@
%{这五种欲乐的境界，开始甜美，后来就苦，}%

@{令墮諸險難，是故常遠離。}@
%{会让人堕入种种险难，所以要常远离。}%

@{以正智思惟，伏斷諸煩惱，}@
%{用正知正见思惟，就能伏断一切烦恼，}%

@{斯為具智人，世世常安隱。}@
%{这才是智慧具足的人，世世常得安稳。}%

@{智能斷諸惑，猶火焚乾薪，}@
%{智慧能断除一切迷惑，就像火烧干柴，}%

@{正智若增明，令三寶顯現。}@
%{正知正见如果增明，就能让三宝显现。}%

@{若樂智境界，常住寂靜法，}@
%{如果乐于智慧的境界，常住寂静法，}%

@{煩惱如毒蛇，則能害諸善。}@
%{烦恼就像毒蛇，就能伤害一切善法。}%

@{若具真實見，能利於自他，}@
%{如果具足真实的见解，就能利益自己和他人，}%

@{離老死過患，住最上寂靜。}@
%{脱离老死患难，住于最上的寂静。}%

@{若樂於輪迴，常為彼纏縛，}@
%{如果乐于轮回，就常被它缠缚，}%

@{是煩惱冤賊，遍三有逼迫。}@
%{这烦恼就像冤仇强盗，遍布三界逼迫着你。}%

@{若人知佛教，為眾生演說，}@
%{如果有人知道佛法，为众生演说，}%

@{常修純淨行，得生於梵天。}@
%{常修清净的行为，就能生到梵天。}%

@{若厭離三毒，常供養諸佛，}@
%{如果厌离三毒，常供养诸佛，}%

@{破壞彼輪迴，如燃於槁木。}@
%{就能破坏轮回，就像燃烧枯木。}%

@{若人知苦因，而不造諸罪，}@
%{如果有人知道苦的根源，而不造作罪业，}%

@{無量煩惱聚，於彼無能縛。}@
%{无量的烦恼聚，对他无能为力。}%

@{智為勝光明，癡為極黑暗，}@
%{智慧是光明，愚痴是黑暗，}%

@{若能善分別，此說為智者。}@
%{如果能善于分别，这叫做智慧者。}%

@{若離癡過失，則無諸險難，}@
%{如果离开愚痴的过失，就没有种种险难，}%

@{為癡之所覆，何能得解脫？}@
%{被愚痴所蒙蔽，怎么能得到解脱呢？}%

@{寧觸於猛火，及毒蛇共處，}@
%{宁愿触碰猛火，和毒蛇共处，}%

@{善求寂滅樂，不應與癡俱。}@
%{也要善于寻求寂灭的快乐，不应该和愚痴在一起。}%

@{愚人無正智，如盲處黑暗，}@
%{愚人没有正知正见，就像盲人处在黑暗中，}%

@{不怖畏輪迴，常造非法行。}@
%{不害怕轮回，常造作非法行为。}%

@{眾生為癡誑，常起於愛染，}@
%{众生被愚痴所蒙蔽，常起贪爱，}%

@{受世間貧窮，為衰老逼迫。}@
%{遭受世间贫穷，被衰老逼迫。}%

@{由三世業果，從地獄生天，}@
%{由于三世业果，从地狱生到天，}%

@{或天墮畜生，或受餓鬼報；}@
%{或者从天堕落为畜生，或者受饿鬼报；}%

@{眾生由彼貪，隨業往諸趣，}@
%{众生由于贪婪，随业力往生各种趣，}%

@{復為癡羂拘，輪回三有海。}@
%{又被愚痴的网罗拘束，轮回三界大海。}%

@{無始造諸罪，受種種生死，}@
%{无始以来造作罪业，受种种生死，}%

@{由彼慣習故，曾不生疲倦。}@
%{由于习惯了，从没有疲倦。}%

@{諸天為樂損，人世匱乏苦，}@
%{天人享受着快乐，却会损耗，人世间充满了贫乏和苦难，}%

@{地獄常燒然，傍生互相噉，}@
%{地狱常燃着火焰，旁生互相吞食，}%

@{餓鬼飢渴逼，皆由彼癡故，}@
%{饿鬼饥渴逼迫，都是由于愚痴，}%

@{長處於輪迴，何曾有少樂？}@
%{长期处在轮回中，哪里有过一点快乐呢？}%

@{愚癡著欲樂，由樂而受苦，}@
%{愚痴执着于欲乐，由于快乐而受苦，}%

@{不近善知識，無正法救護。}@
%{不亲近善知识，没有正法救护。}%

@{若人具真實，常樂聞正法，}@
%{如果有人具足真实，常乐于听闻正法，}%

@{修習諸禪定，彼則無憂苦。}@
%{修习各种禅定，他就没有忧愁和痛苦。}%

@{諸佛宣正法，如燈常照明，}@
%{诸佛宣说正法，就像灯火常照亮，}%

@{慈念諸眾生，過如彼父母。}@
%{慈悲地念着众生，超过了父母。}%

@{眾生由三因，造三種過失，}@
%{众生由于三种因，造作三种过失，}%

@{循環三界中，三受常相逐。}@
%{在三界中循环，三受常相继。}%

@{由三業所起，趣三惡險難，}@
%{由于三业所起，走向三恶道，}%

@{眾生著樂故，馳騁於三有。}@
%{众生执着于快乐，在三界中奔驰。}%

@{若尊重三寶，當得三菩提，}@
%{如果尊重三宝，就能得到三菩提，}%

@{遠離三種見，則不生諸苦。}@
%{远离三种错误见解，就不会生起各种痛苦。}%

@{於彼晝夜中，三時常觀察，}@
%{在白天和黑夜，三时常观察，}%

@{謂彼老病死，三種過失藏。}@
%{说衰老、疾病、死亡，这三种过失隐藏着。}%

@{三業離邪思，善住三平等，}@
%{三业远离邪念，善于住于三平等，}%

@{不著於輪迴，永離諸憂惱。}@
%{不执着于轮回，永远离开忧愁烦恼。}%

@{於彼道非道，及空有等相，}@
%{对道与非道，以及空与有等相，}%

@{慈心善觀察，當證無上道。}@
%{慈悲心善于观察，就能证得无上道。}%

@{是人意清淨，不觸諸染欲，}@
%{这个人内心清净，不触碰染污的欲望，}%

@{永離諸垢濁，得解脫安樂。}@
%{永远离开一切污垢，得到解脱安乐。}%

@{離三有貪求，常生於正念，}@
%{离开三界的贪求，常生起正念，}%

@{是人於正道，決定無退轉。}@
%{这个人走在正道上，决定不会退转。}%

@{眾生癡所蔽，於智不通達，}@
%{众生被愚痴所蒙蔽，对智慧不通达，}%

@{起無量貪愛，常為苦纏縛。}@
%{起无量的贪爱，常被痛苦缠缚。}%

@{懈怠無慚愧，習近惡知識，}@
%{懈怠没有羞愧，亲近恶知识，}%

@{為地獄種子，智者善防護。}@
%{这是堕入地狱的种子，智慧者要善于防范。}%

@{由彼無慚愧，常造作眾罪，}@
%{由于没有羞愧，常造作罪业，}%

@{後墮於險道，徒生於悔惱。}@
%{后来堕入险道，徒然生起悔恨烦恼。}%

@{起憍慢瞋恚，嫉妬并覆惱，}@
%{起傲慢、嗔恨、嫉妒、覆恼，}%
