@{不放逸品第六之餘}@
%{不放逸品第六之余}%

@{於善當奉行，見惡則如毒，}@
%{对好的事应该去奉行，看到不好的事就好像看到毒药一样，}%

@{故說此放逸，為第一險道。}@
%{所以说这种放逸，是第一条危险的道路。}%

@{彼放逸眾生，為美言所誑，}@
%{那些放逸的众生，被美言所欺骗，}%

@{常迷醉其心，於苦唯己分。}@
%{常常迷醉自己的心，对苦难只认为是自己的份。}%

@{不怖於苦惱，不求天快樂，}@
%{不害怕苦恼，不追求天上的快乐，}%

@{由無智觀察，與傍生無異。}@
%{因为愚痴的观察，和旁生没有区别。}%

@{於飲食婬欲，其心無暫捨，}@
%{对饮食和淫欲，内心没有片刻的舍弃，}%

@{如是諸有情，則同其畜類。}@
%{像这样的众生，就和畜生一样了。}%

@{樂著嬉戲者，住琰摩口中，}@
%{喜欢玩乐的人，住在阎摩的口中，}%

@{彼死即現前，是苦難堪忍。}@
%{他死就马上到来，这种苦难难以忍受。}%

@{眾生心放逸，樂壞即夭喪，}@
%{众生的心放逸，快乐坏了就夭折，}%

@{由彼放逸故，終為死磨滅。}@
%{因为放逸，最终被死亡消灭。}%

@{唯有一善法，令壽命安隱，}@
%{只有一件善法，能让寿命安稳，}%

@{復能生諸天，說名不放逸。}@
%{还能生到诸天，叫做不放逸。}%

@{放逸不放逸，生過失功德，}@
%{放逸和不放逸，产生过失和功德，}%

@{善惡皆由心，解縛亦如是。}@
%{善恶都由心，解脱也一样。}%

@{眾生不了知，快樂如冤害，}@
%{众生不明白，快乐就像冤仇一样，}%

@{彼意極愚癡，遠離於佛智。}@
%{他们的想法非常愚痴，远离佛智。}%

@{放逸如毒樹，聳幹有其三，}@
%{放逸就像毒树，高耸的树干有三部分，}%

@{為彼老病死，常依止而住。}@
%{就是老、病、死，常常依附着它而存在。}%

@{具足修正行，不為彼所侵，}@
%{具足修正的行为，不被它侵犯，}%

@{智者處輪迴，常樂不放逸。}@
%{智者处在轮回中，常常快乐不放逸。}%

@{若樂不放逸，則離諸過咎，}@
%{如果快乐不放逸，就能远离过失，}%

@{解脫諸恐怖，得最上快樂。}@
%{解脱一切恐惧，得到最高的快乐。}%

@{若具放逸者，我則生怖畏，}@
%{如果具足放逸，我就生起恐惧，}%

@{彼若有解脫，於樂不須斷。}@
%{如果他能够解脱，对快乐就不需要断绝。}%

@{由貪起放逸，當墮蓮華獄，}@
%{因为贪婪而产生放逸，就会堕落到莲花地狱，}%

@{彼於欲自在，常止住其中。}@
%{他们在欲望中自在，常常停留其中。}%

@{若住於放逸，生四種顛倒，}@
%{如果住在放逸中，就会生起四种颠倒，}%

@{能壞於善行，如世間冤害。}@
%{能破坏善行，就像世间冤仇一样。}%

@{無量諸疑惑，恐怖常逼切，}@
%{无数的疑惑，恐惧常常逼迫着，}%

@{流轉生死中，皆由放逸行。}@
%{在生死中轮回，都是因为放逸的行为。}%

@{此一放逸行，常樂諸欲樂，}@
%{这种放逸的行为，常常喜欢各种欲望的快乐，}%

@{則遠離一切，無漏清淨法。}@
%{就远离一切无漏清净法。}%

@{諸天放逸故，展轉無窮極，}@
%{诸天因为放逸，转来转去没有穷尽，}%

@{愚癡不厭離，何由得寂靜？}@
%{愚痴不厌离，怎么能够得到寂静？}%

@{離染污思惟，其心則寂靜，}@
%{远离染污的思惟，内心就会寂静，}%

@{能利於自他，無復諸熱惱。}@
%{能利益自己和他人，没有再多的烦恼。}%

@{諸天著欲樂，所得何曾見！}@
%{诸天执着欲乐，得到什么曾经见过？}%

@{是樂有為生，無常不久住。}@
%{这种快乐是有为而生的，无常不久住。}%

@{此有為色相，決定而破壞，}@
%{这种有为的色相，一定会破灭，}%

@{彼樂若壞時，則生於苦惱。}@
%{当这种快乐破灭的时候，就会生起苦恼。}%

@{若人於欲境，放逸心狂亂，}@
%{如果人在欲望的境界中，放逸的心狂乱，}%

@{決定有乖離，後受於熱惱。}@
%{一定会产生分离，后来就会受到烦恼。}%

@{於五欲境界，貪愛得自在，}@
%{在五欲的境界中，贪爱得到自在，}%

@{彼欲誑諸天，放逸則隨轉。}@
%{这些欲望欺骗诸天，放逸就随着它转动。}%

@{若人作放逸，是諸難根本，}@
%{如果人做放逸，这是诸难的根本，}%

@{於財起貪求，廣造諸不善。}@
%{对财富产生贪求，广泛地造作不善。}%

@{若人於此生，具足諸快樂，}@
%{如果人在今生，具足各种快乐，}%

@{當隨智惠行，如魚逐流水。}@
%{应该随着智慧的行为，就像鱼儿追逐流水一样。}%

@{諸天若放逸，為女色所伏，}@
%{诸天如果放逸，被女色所征服，}%

@{則為彼燒然，常受於苦惱。}@
%{就会被它燃烧，常常受到苦恼。}%

@{是故諸天人，一心當遠離，}@
%{所以诸天人，一心应该远离，}%

@{於意常止足，不為欲所縛。}@
%{在内心常常止足，不被欲望所束缚。}%

@{愚癡迷罪福，不知法非法，}@
%{愚痴迷于罪福，不知道什么是法什么是非法，}%

@{此人於涅盤，少分不可得。}@
%{这个人对涅槃，一点都得不到。}%

@{於輕重律儀，及甚深法要，}@
%{对轻重律仪，以及甚深的法要，}%

@{而常樂修習，則獲彼安樂。}@
%{而常常快乐地修行，就会获得安乐。}%

@{常樂宣正法，滌除諸垢穢，}@
%{常常快乐地宣说正法，涤除一切污垢，}%

@{不作放逸行，則常獲妙樂。}@
%{不做放逸的行为，就会常常获得美妙的快乐。}%

@{昔樂行放逸，常迷醉愚癡，}@
%{以前喜欢放逸，常常迷醉愚痴，}%

@{若天若世人，皆不應親近。}@
%{无论是天人还是世间人，都不应该亲近。}%

@{造放逸過失，未曾有間斷，}@
%{造作放逸的过失，从来没有间断，}%

@{彼人命欲終，則趣於惡道。}@
%{这个人命将终结，就会趣向恶道。}%

@{此說彼放逸，於理非和合，}@
%{这里说放逸，在道理上是不和谐的，}%

@{諸天常癡迷，當墮於地獄。}@
%{诸天常常愚痴，会堕落到地狱。}%

@{諸天欲所迷，非愛而生愛，}@
%{诸天被欲望所迷惑，不是爱却产生了爱，}%

@{癡暗不覺知，何有於少樂？}@
%{愚痴黑暗不知道，有什么少的快乐？}%

@{諸天於五欲，常樂著迷醉，}@
%{诸天对五欲，常常快乐地迷醉，}%

@{如彼生盲人，不見於正道。}@
%{就像天生盲人，看不到正道。}%

@{由彼心散亂，不生真實見，}@
%{因为他们的心散乱，不生起真实的见解，}%

@{放逸鎮燒然，是故當遠離。}@
%{放逸常常燃烧着，所以应该远离。}%

@{意地諸善法，由放逸破壞，}@
%{意地的善法，因为放逸而破坏，}%

@{八聖道昭然，畢竟無能見。}@
%{八圣道明明在眼前，最终却看不到。}%

@{是放逸為毒，能壞十善法，}@
%{放逸是毒，能破坏十善法，}%

@{如是放逸者，失四種禪定。}@
%{像这样的放逸者，就会失去四种禅定。}%

@{諸佛離五欲，常讚不放逸，}@
%{诸佛远离五欲，常常赞叹不放逸，}%

@{是身老死侵，當厭離放逸。}@
%{因为身体会衰老死亡，应该厌离放逸。}%

@{放逸唯有苦，離此即解脫，}@
%{放逸只有苦，远离它就解脱，}%

@{為放逸所牽，無智不能斷。}@
%{被放逸所牵引，愚痴的人不能断除。}%

@{不放逸最勝，諸天樂親近，}@
%{不放逸最胜，诸天都喜欢亲近，}%

@{若作放逸者，定知當退沒。}@
%{如果做放逸，一定会知道会退没。}%

@{一切諸眾生，輪迴不解脫，}@
%{一切众生，轮回不能解脱，}%

@{由放逸羂索，縈纏難出離，}@
%{因为放逸的绳索，缠绕着很难离开，}%

@{造惡則無福，作善非招罪。}@
%{造恶就没有福报，做善不是招罪。}%

@{放逸諸有情，常生顛倒見，}@
%{放逸的众生，常常生起颠倒的见解，}%

@{云何為罪福？非世俗所說。}@
%{什么叫罪什么叫福？不是世俗所说。}%

@{智者善明了，不讚於放逸，}@
%{智者善于明白，不赞叹放逸，}%

@{諸天若放逸，則減失善業，}@
%{诸天如果放逸，就会减少善业，}%

@{當知此惡因，定招於苦果。}@
%{应该知道这种恶因，一定会招致苦果。}%

@{若意著諸欲，暫時生少樂，}@
%{如果内心执着于欲望，暂时生起少许快乐，}%

@{則為彼破壞，退失生大苦。}@
%{就会被它破坏，退失而生起大苦。}%

@{境界常現前，愚夫不厭足，}@
%{境界常常出现在眼前，愚蠢的人不知足，}%

@{癡迷著女人，則見其墮落。}@
%{愚痴地迷恋女人，就会看到自己堕落。}%

@{由貪生放逸，習近諸女人，}@
%{因为贪婪而产生放逸，习惯接近女人，}%

@{於己命終時，彼則咸觀看。}@
%{当自己命终的时候，他们就都会观看。}%

@{諸天若退沒，獨逝而無侶，}@
%{诸天如果退没，独自离去没有同伴，}%

@{唯有善惡業，於後而隨逐。}@
%{只有善恶业，在后面跟随。}%

@{遠離彼惡作，專修諸善行，}@
%{远离那些恶作，专心修持善行，}%

@{棄捨斯過咎，常樂不放逸。}@
%{舍弃这些过失，常常快乐不放逸。}%

@{放逸輪迴本，離此為寂靜，}@
%{放逸是轮回的根本，远离它就是寂静，}%

@{彼二種差別，此說其自相。}@
%{这两种差别，这里说的是它们各自的相貌。}%

@{智者常思惟，樂修於諸善，}@
%{智者常常思惟，快乐地修持善法，}%

@{正人依法行，則無於少苦。}@
%{正直的人依法而行，就不会有丝毫的苦难。}%

@{眾生處輪迴，皆隨於自業，}@
%{众生处在轮回中，都随着自己的业力，}%

@{云何彼世間，為放逸破壞？}@
%{怎么能够说世间，被放逸破坏？}%

@{放逸第一縛，復能壞諸善，}@
%{放逸是第一条束缚，还能破坏善法，}%

@{為彼墮落因，作地獄苦本。}@
%{是他们堕落的原因，是造地狱苦的根本。}%

@{顯示一放逸，為諸苦惱因，}@
%{显示放逸，是各种苦恼的原因，}%

@{若樂饒益者，常離彼過失。}@
%{如果想要利益众生，常常远离那些过失。}%

@{若離於放逸，得至不死處，}@
%{如果远离放逸，就能到达不死的地方，}%

@{由不放逸故，則近菩提道。}@
%{因为不放逸，就能接近菩提道。}%

@{若樂不放逸，住最上涅盤，}@
%{如果快乐不放逸，就能住到最高的涅槃，}%

@{故說彼放逸，為其墮落因。}@
%{所以说放逸，是他们堕落的原因。}%

@{若人作放逸，此惡無過上，}@
%{如果人做放逸，这种恶没有比它更厉害的，}%

@{為世所欺輕，死墮餓鬼趣。}@
%{被世间所欺骗轻视，死后堕落到饿鬼道。}%

@{若樂於放逸，則生彼顛倒，}@
%{如果喜欢放逸，就会生起颠倒，}%

@{由如是業果，隨生死流轉。}@
%{因为这样的业果，随着生死而轮回。}%

@{彼地獄熾火，常燒放逸人，}@
%{地狱的炽热火焰，常常燃烧着放逸的人，}%

@{若樂解脫者，則捨於放逸。}@
%{如果想要解脱，就应该舍弃放逸。}%

@{若人離放逸，則生於明智，}@
%{如果人远离放逸，就会生起智慧，}%

@{永斷於諸惑，常受彼妙樂。}@
%{永远断除一切烦恼，常常享受美妙的快乐。}%

@{眾生心散亂，為業而纏縛，}@
%{众生的心散乱，被业力所缠缚，}%

@{流轉三有中，當樂不放逸。}@
%{在三界中轮回，应该快乐不放逸。}%

@{於苦謂為樂，樂壞生於苦，}@
%{把苦当作快乐，快乐坏了就生起苦，}%

@{夫死轉為妻，當樂不放逸。}@
%{丈夫死了就转世为妻子，应该快乐不放逸。}%

@{妻死或為母，母死或為妻，}@
%{妻子死了就转世为母亲，母亲死了就转世为妻子，}%

@{於此流轉中，當樂不放逸。}@
%{在这种轮回中，应该快乐不放逸。}%

@{放逸能生貪，由貪復生恚，}@
%{放逸能生起贪婪，因为贪婪又生起嗔恨，}%

@{為過患之源，沈輪於惡道。}@
%{是过患的根源，沉沦到恶道。}%

@{放逸不放逸，智者皆不著，}@
%{放逸和不放逸，智者都不执着，}%

@{此最上安隱，昇智慧樓閣。}@
%{这是最高的安稳，升到智慧的楼阁。}%

@{若能離放逸，善住安樂處，}@
%{如果能够远离放逸，善于住到安乐的地方，}%

@{為大智丈夫，速證真常果。}@
%{就是大智丈夫，快速证得真常的果位。}%

@{示放逸過失，與道極相違，}@
%{显示放逸的过失，和道极度相违背，}%

@{則斷法橋梁，壞善心種子。}@
%{就会断绝法桥，破坏善心的种子。}%

@{能壞解脫法，引生諸妄念，}@
%{能破坏解脱法，引生各种妄念，}%

@{墮彼險惡趣，皆從放逸起。}@
%{堕落到那些险恶的趣向，都是从放逸而起。}%

@{自不求利益，為他人所棄，}@
%{自己不求利益，被他人所抛弃，}%

@{無言無所作，彼則如死者。}@
%{不说话不做任何事，就像死人一样。}%

@{雖具天形質，愚癡同畜類，}@
%{虽然具有天人的形质，愚痴却和畜生一样，}%

@{常居放逸中，作歌舞戲笑。}@
%{常常住在放逸中，做歌舞玩乐。}%

@{若人著放逸，即趣三有海，}@
%{如果人执着放逸，就会趣向三界的海，}%

@{滅已復還生，悉見其破壞。}@
%{灭了又再生，都看到它的破坏。}%

@{由此造諸罪，為業所纏縛，}@
%{由此造作各种罪业，被业力所缠缚，}%

@{放逸轉為冤，違背諸善法。}@
%{放逸就会变成冤仇，违背一切善法。}%

@{是業不住內，亦不在於外，}@
%{这种业力不在内心，也不在外，}%

@{當知放逸者，皆由心破壞。}@
%{应该知道放逸，都是因为心而破坏。}%

@{於最上境界，其心無厭足，}@
%{在最高的境界中，内心不知足，}%

@{樂歌舞嬉戲，不知墮滅處。}@
%{喜欢歌舞玩乐，不知道堕落的地方。}%

@{愚癡放逸者，以怖為歡悅，}@
%{愚痴放逸的人，把恐惧当作快乐，}%

@{彼天同生盲，不知道非道。}@
%{他们就像天生盲人，不知道什么是道什么是非道。}%

@{由彼放逸行，從諸禪墮落，}@
%{因为他们的放逸行为，从禅定中堕落，}%

@{輪轉欲界中，馳流於五趣。}@
%{在欲界中轮回，奔流于五趣。}%

@{若住無色界，獲四種空定，}@
%{如果住在无色界，获得四种空定，}%

@{由行放逸故，流轉於世間。}@
%{因为行为放逸，就在世间轮回。}%

@{周流三界中，愚癡無知覺，}@
%{周游三界中，愚痴没有知觉，}%

@{放逸為桎梏，愛索而纏縛。}@
%{放逸是枷锁，爱欲的绳索缠绕着。}%
