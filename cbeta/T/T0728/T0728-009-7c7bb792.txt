@{西天譯經三藏朝散大夫試鴻臚卿宣梵大師賜紫沙門臣日稱等奉詔譯}@

@{寂靜品第二十八}@
%{寂静品第二十八}%

@{若盡諸煩惱，則得最上樂，}@
%{如果能消除所有的烦恼，就能获得最高的快乐，}%

@{此為寂靜通，智者如實說。}@
%{这就是寂静通，智慧的人如实地说。}%

@{又彼諸如來，常讚寂靜法，}@
%{而且那些如来，经常赞叹寂静法，}%

@{得至不滅處，諸苦則不生。}@
%{获得永不灭的境界，所有的痛苦都不会产生。}%

@{若人離放逸，無貪欲過患，}@
%{如果一个人远离放逸，没有贪欲的过患，}%

@{樂修寂靜行，去菩提不遠。}@
%{乐于修习寂静行，离菩提就不远了。}%

@{若能離貪愛，於境心不亂，}@
%{如果能远离贪爱，面对境界内心不乱，}%

@{及捨惡知識，去菩提不遠。}@
%{并且舍弃恶知识，离菩提就不远了。}%

@{若能善觀察，不著善不善，}@
%{如果能善于观察，不执着于善和不善，}%

@{離輪迴怖畏，去菩提不遠。}@
%{远离轮回的恐惧，离菩提就不远了。}%

@{若能斷諸惑，除懈怠垢穢，}@
%{如果能断除所有的惑，消除懈怠的污垢，}%

@{了自他無我，去菩提不遠。}@
%{明白自己和他人都没有我，离菩提就不远了。}%

@{離三毒過患，令諸根寂靜，}@
%{远离三毒的过患，让所有的根都寂静下来，}%

@{明了於四諦，去菩提不遠。}@
%{明白四谛，离菩提就不远了。}%

@{於精麁飲食，而不生貪厭，}@
%{对于精粗的饮食，不生贪厌，}%

@{智境兩相如，去菩提不遠。}@
%{智慧和境界两者相一致，离菩提就不远了。}%

@{於苦樂二種，亦不生執著，}@
%{对于苦乐两种，也不生执着，}%

@{得離諸怖畏，去菩提不遠。}@
%{获得远离所有的恐惧，离菩提就不远了。}%

@{若人怖諸罪，當離諸放逸，}@
%{如果一个人害怕所有的罪过，就应该远离放逸，}%

@{善求趣菩提，得最上寂靜。}@
%{善于求取菩提，获得最高的寂静。}%

@{聖道品第二十九}@
%{圣道品第二十九}%

@{若人於四諦，以智善觀察，}@
%{如果一个人以智慧善于观察四谛，}%

@{解脫諸輪迴，趣無為彼岸。}@
%{就能解脱所有的轮回，趣向无为的彼岸。}%

@{若無正思惟，愚癡著諸欲，}@
%{如果没有正思惟，愚痴地执着于欲望，}%

@{不厭離生死，為輪迴所縛。}@
%{不厌离生死，就被轮回所束缚。}%

@{不畏廣大苦，無出離方便，}@
%{不畏惧广大的苦，没有出离的方便，}%

@{為欲箭所中，當墮於惡道。}@
%{被欲箭所中，就会堕落到恶道。}%

@{愚人著諸欲，如蛾愛燈光，}@
%{愚人执着于欲望，就像飞蛾爱着灯光，}%

@{不知大怖畏，畢竟無少樂。}@
%{不知道大的恐惧，最终没有一点快乐。}%

@{佛說真實道，謂苦空無常，}@
%{佛说真实之道，就是苦、空、无常，}%

@{及無我作者，能免脫輪迴。}@
%{以及没有我造作者，能够免脱轮回。}%

@{智境本平等，一切皆心造，}@
%{智慧和境界本来是平等的，一切都是心所造，}%

@{於此證解已，故不說三種。}@
%{对此证悟了解了，所以不讲三种。}%

@{離飲食過患，不依邪活命，}@
%{远离饮食的过患，不依靠邪恶的方式谋生，}%

@{起無分別智，證出世間法。}@
%{发起无分别智，证悟出世间法。}%

@{善住真實智，作十六行相，}@
%{善于住于真实智，作十六种行相，}%

@{然後能了知，達諸法次第。}@
%{然后才能了解，到达诸法的次第。}%

@{永離三毒垢，出三有逼迫，}@
%{永远远离三毒的垢，超出三有的逼迫，}%

@{趣越三惡趣，是名須陀洹。}@
%{趣向超越三恶趣，这就是名为须陀洹。}%

@{不善法為因，定墮於惡道，}@
%{不善法作为因，一定会堕落到恶道，}%

@{由依解脫法，得須陀洹果。}@
%{由于依靠解脱法，就能获得须陀洹果。}%

@{善修習諸定，引生於輕安，}@
%{善于修习各种定，引导产生轻安，}%

@{純淨業相應，得超三有海。}@
%{纯净的业相应该，就能超越三有海。}%

@{永斷諸疑惑，離貪及不害，}@
%{永远断除所有的疑惑，远离贪欲和不害，}%

@{乘白業船舫，能到於彼岸。}@
%{乘着白业的船舫，就能到达彼岸。}%

@{謂有為諸法，從因緣生起，}@
%{就是说有为法，从因缘生起，}%

@{知彼四聖諦，為染淨因果。}@
%{知道那四圣谛，就是染净的因果。}%

@{若了達四諦，決定得解脫，}@
%{如果了解四谛，就能决定获得解脱，}%

@{愚夫著欲境，隨三有旋轉。}@
%{愚夫执着于欲境，随着三有而旋转。}%

@{若能離貪欲，則住寂靜道，}@
%{如果能远离贪欲，就能住于寂静道，}%

@{此道無過上，智者所遊履。}@
%{这条道没有比它更高的，是智慧的人所游履的。}%

@{若人於諸佛，常淨心恭敬，}@
%{如果一个人对诸佛，经常保持清净心恭敬，}%

@{生生獲快樂，離佛無解脫。}@
%{生生世世获得快乐，离佛就没有解脱。}%

@{若人於正法，常淨心愛樂，}@
%{如果一个人对正法，经常保持清净心爱乐，}%

@{生生獲快樂，離法無解脫。}@
%{生生世世获得快乐，离法就没有解脱。}%

@{若人於眾僧，常淨心供養，}@
%{如果一个人对众僧，经常保持清净心供养，}%

@{生生獲快樂，離僧無解脫。}@
%{生生世世获得快乐，离僧就没有解脱。}%

@{若人於四諦，常淨心信解，}@
%{如果一个人对四谛，经常保持清净心信解，}%

@{生生獲快樂，離此無解脫。}@
%{生生世世获得快乐，离此就没有解脱。}%

@{若人於聖道，常淨心修習，}@
%{如果一个人对圣道，经常保持清净心修习，}%

@{生生獲快樂，離此無解脫。}@
%{生生世世获得快乐，离此就没有解脱。}%

@{若求證真如，當安住淨慧，}@
%{如果要证悟真如，就应该安住于清净的智慧，}%

@{不樂著遊戲，此為天中天。}@
%{不乐于玩乐，这就是天中天。}%

@{若樂著嬉戲，心不生暫捨，}@
%{如果乐于玩乐，内心不生暂时的舍弃，}%

@{增長諸病惱，斯則實非樂。}@
%{就会增长各种病恼，这才是真正的不是快乐。}%

@{善解四諦法，及修施等行，}@
%{善于理解四谛法，并且修习布施等行，}%

@{當知如是人，住最上安隱。}@
%{应该知道这样的人，住于最上的安隐。}%

@{教誡比丘品第三十}@
%{教诫比丘品第三十}%

@{常樂行慈忍，不害諸有情，}@
%{经常乐于行慈忍，不伤害一切有情，}%

@{得一切眾生，敬之如其父。}@
%{就能获得一切众生，敬爱他如同他们的父亲。}%

@{身業常清淨，諸根善相應，}@
%{身业经常清净，诸根善于相应该，}%

@{心不生慳貪，遠離不與取。}@
%{内心不生悭贪，远离不与取。}%

@{於彩繪女人，亦不應觀視，}@
%{对于彩绘的女人，也不应该观视，}%

@{斷堅固欲想，為世解脫者。}@
%{断除坚固的欲想，是为世间解脱者。}%

@{樂習諸禪定，得離諸憂畏，}@
%{乐于修习各种禅定，就能远离各种忧畏，}%

@{不觸煩惱蛇，視金如瓦礫。}@
%{不触烦恼蛇，看金子如同瓦砾。}%

@{於苦樂安危，及盛衰等事，}@
%{对于苦乐安危，以及盛衰等事，}%

@{其心不傾動，此名為比丘。}@
%{内心不倾动，这就是名为比丘。}%

@{善降伏諸根，不為境所嬈，}@
%{善于降伏诸根，不被境界所扰，}%

@{以智如實知，於冤親平等。}@
%{以智慧如实知道，对冤亲平等。}%

@{若具足明智，了欲境如毒，}@
%{如果具足明智，明白欲境如同毒药，}%

@{我記如是人，當得菩提道。}@
%{我记得这样的人，应该能获得菩提道。}%

@{具真實正見，悟剎那生滅，}@
%{具足真实的正见，悟解刹那生灭，}%

@{安住如須彌，超出輪迴海。}@
%{安住如同须弥山，超出轮回海。}%

@{於草及旃檀，等視無差別，}@
%{对于草和檀香，平等看待没有差别，}%

@{珍饍及名衣，皆不生愛樂。}@
%{珍馐和名贵衣服，都不生爱乐。}%

@{於利養名聞，觀之如熾火，}@
%{对于利养名闻，看它如同炽热的火，}%

@{心常生止足，依草而坐臥。}@
%{内心经常生起止足，依草而坐卧。}%

@{於過現所作，種種諸事業，}@
%{对于过去现在所作，种种诸事业，}%

@{離顛倒思惟，則不生染著。}@
%{远离颠倒思惟，就不会生起染着。}%

@{樂聞於正法，不隨流轉因，}@
%{乐于听闻正法，不随着流转的因，}%

@{以慧善揀擇，常修殊勝行。}@
%{以智慧善于拣择，经常修习殊胜的行。}%

@{欲境雖熾然，由彼心寂靜，}@
%{欲境虽然炽热，但是由于内心寂静，}%

@{驅策如僮僕，則無諸苦惱。}@
%{驱策它如同僮仆，就没有各种苦恼。}%

@{若諸根調順，則不生流蕩，}@
%{如果诸根调顺，就不会生起流荡，}%

@{離欲纏縛者，與牟尼無異。}@
%{远离欲缠缚者，与牟尼没有区别。}%

@{善人如金寶，見者咸貴重，}@
%{善人如同金宝，见到的人都会珍惜，}%

@{樂修寂靜行，令他生喜心。}@
%{乐于修习寂静行，让别人生起欢喜心。}%

@{棲止阿蘭若，不樂居樓觀，}@
%{栖息在阿兰若，不乐于住在楼观，}%

@{知足持毳衣，常行於乞食。}@
%{知足持着粗毛衣，经常行乞食。}%

@{善修身語心，不生苦樂想，}@
%{善于修习身语心，不生苦乐想，}%

@{離分別執著，得最上安隱。}@
%{远离分别执着，获得最上的安隐。}%

@{由習諸禪定，能破彼魔怨，}@
%{由于修习各种禅定，就能破除那些魔怨，}%

@{常以真實言，引導於群品。}@
%{经常以真实的话语，引导大众。}%

@{善乘智慧車，摧六根貪使，}@
%{善于乘着智慧的车，摧毁六根的贪使，}%

@{當知如是人，得近菩提道。}@
%{应该知道这样的人，就能接近菩提道。}%

@{樂住阿蘭若，則永離諸過，}@
%{乐于住在阿兰若，就能永远远离各种过失，}%

@{如風於空中，吹雲無障礙。}@
%{就像风在空中，吹动云彩没有障碍。}%

@{以清淨三業，勤修諸勝行，}@
%{以清净的三业，勤修各种胜行，}%

@{具真實正見，破壞諸魔教。}@
%{具足真实的正见，破除各种魔教。}%

@{於彼貪等行，本性而不染，}@
%{对于那些贪欲等行，本性上就不染着，}%

@{常起慈悲心，是比丘所作。}@
%{经常生起慈悲心，这就是比丘应该做的。}%

@{了知色等境，為彼纏縛因，}@
%{明白色等境界，是缠缚的因，}%

@{是人無憂惱，得至寂滅處。}@
%{这个人没有忧恼，就能到达寂灭处。}%

@{知諸因緣法，善惡皆決定，}@
%{知道各种因缘法，善恶都决定，}%

@{樂聞解脫法，於貪則不著。}@
%{乐于听闻解脱法，对于贪欲就不执着。}%

@{於未修善業，而常生愛樂，}@
%{对于没有修行的善业，却经常生起爱乐，}%

@{是人如月光，本性淨無垢。}@
%{这个人如同月光，本性清净无垢。}%

@{焚燒罪惡法，如火投乾薪，}@
%{焚烧罪恶法，如同火投到干柴，}%

@{棄背諸苦因，於三有稱勝。}@
%{舍弃各种苦的因，在三有中称胜。}%

@{志求於解脫，不著世間法，}@
%{志向求取解脱，不执着世间法，}%

@{超出諸輪迴，如鳥隨空往。}@
%{超出各种轮回，如同鸟儿随着空而去。}%

@{如實知其因，決定而受果，}@
%{如实知道它的因，就能决定而受果，}%

@{彼於三有中，是名真解脫。}@
%{他在三有中，这就是名为真解脱。}%

@{於苦樂精麁，皆無有所著，}@
%{对于苦乐精粗，都没有任何执着，}%

@{此最上比丘，觀世間如焰。}@
%{这个最上等的比丘，看世间如同火焰。}%

@{心不生散亂，樂求於正法，}@
%{内心不生散乱，乐于求取正法，}%

@{勤修本白業，彼如阿羅漢。}@
%{勤修本来的白业，他就如同阿罗汉。}%

@{樂近善知識，遠離於親屬，}@
%{乐于接近善知识，远离亲属，}%

@{捐在家垢染，彼如阿羅漢。}@
%{舍弃在家时的垢染，他就如同阿罗汉。}%

@{由慧寂諸根，不著於境界，}@
%{由于智慧寂静诸根，不执着于境界，}%

@{行視二足指，彼如阿羅漢。}@
%{行走看如同指着脚趾，他就如同阿罗汉。}%

@{不詣王履道，城邑四衢巷，}@
%{不前往王履道，城邑的四衢巷，}%

@{販賣為譏呵，彼如阿羅漢。}@
%{贩卖会被讥讽，他就如同阿罗汉。}%

@{不觀於歌舞，不樂相鬪諍，}@
%{不观看歌舞，不乐于互相争斗，}%

@{住曠野空閑，彼如阿羅漢。}@
%{住在旷野空闲的地方，他就如同阿罗汉。}%

@{一日唯一食，未得不希求，}@
%{一天只吃一顿饭，不希求其他，}%

@{知分量止足，彼如阿羅漢。}@
%{知道分量止足，他就如同阿罗汉。}%

@{常持糞掃衣，不樂上妙服，}@
%{经常穿着粪扫衣，不乐于穿华丽的衣服，}%

@{與食行相應，彼如阿羅漢。}@
%{与饮食行相应该，他就如同阿罗汉。}%

@{由不造諸業，離一切虛妄，}@
%{由于不造作各种业，远离一切虚妄，}%

@{不起亦不樂，彼如阿羅漢。}@
%{不起也不乐，他就如同阿罗汉。}%

@{不起恚怒相，離貪欲愚癡，}@
%{不起恚怒相，远离贪欲愚痴，}%

@{解脫諸惡法，彼如阿羅漢。}@
%{解脱各种恶法，他就如同阿罗汉。}%

@{離一切瞋惱，超越諸結使，}@
%{远离一切瞋恼，超越各种结使，}%

@{住正念思惟，彼如阿羅漢。}@
%{住于正念思惟，他就如同阿罗汉。}%

@{修習八聖道，善住於寂靜，}@
%{修习八圣道，善于住于寂静，}%

@{破諸煩惱怨，彼如阿羅漢。}@
%{破除各种烦恼怨，他就如同阿罗汉。}%

@{寂靜根堅固，超貪欲淤泥，}@
%{寂静根坚固，超越贪欲的淤泥，}%

@{住心一境性，彼如阿羅漢。}@
%{住于心一境性，他就如同阿罗汉。}%

@{於自經行所，及他遊止處，}@
%{对于自己经行的地方，以及他人游止的地方，}%

@{知已如實說，彼如阿羅漢。}@
%{已经如实地知道，他就如同阿罗汉。}%

@{知漏無漏法，皆從因緣生，}@
%{知道漏法和无漏法，都是从因缘生起，}%

@{決定無有疑，彼如阿羅漢。}@
%{决定没有怀疑，他就如同阿罗汉。}%

@{離惛沈睡眠，無懈依時起，}@
%{远离昏沉睡眠，没有懈怠依时起床，}%

@{勤修諸梵行，彼如阿羅漢。}@
%{勤修各种梵行，他就如同阿罗汉。}%

@{喜住阿蘭若，修毘鉢舍那，}@
