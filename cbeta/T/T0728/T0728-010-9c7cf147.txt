@{生天品第三十二}@
%{生天品第三十二}%

@{若人修眾善，清淨心質直，}@
%{如果一个人修持各种善法，内心清净质朴，}%

@{當得生天中，牟尼之所說。}@
%{就应该能够生到天界，正如佛陀所说。}%

@{樂布施愛語，慈心常相應，}@
%{喜欢布施和说爱语，慈悲心常相伴随，}%

@{護念諸眾生，此因為真實。}@
%{护念一切众生，这就是真实的。}%

@{由彼心清淨，為白法依止，}@
%{因为内心清净，所以就能依靠白法，}%

@{受天中快樂，身出於光明。}@
%{在天界享受快乐，身体散发光明。}%

@{是光明最勝，相續而不絕，}@
%{这种光明是最胜的，不断相续，}%

@{如世有燈明，不墮諸險難。}@
%{就像世间灯明，不会陷入各种险难。}%

@{若人心清淨，如摩尼無垢，}@
%{如果一个人内心清净，像摩尼宝石一样没有污垢，}%

@{平等常謙和，得生於諸天。}@
%{平等、谦和，就能生到诸天。}%

@{若心善調伏，持戒修諸定，}@
%{如果内心善于调伏，持戒修持禅定，}%

@{清淨如真金，得生於諸天。}@
%{清净如真金，就能生到诸天。}%

@{於一切有情，心常生憐愍，}@
%{对一切有情众生，内心常生怜悯，}%

@{不造殺生罪，得生於诸天。}@
%{不造杀生罪，就能生到诸天。}%

@{處世身光潔，不染諸罪法，}@
%{在世间，身体光洁，不染各种罪法，}%

@{離一切損害，得生於諸天。}@
%{远离一切损害，就能生到诸天。}%

@{見欲境如毒，視金寶如草。}@
%{看到欲望就像毒药，看待金宝就像草芥。}%

@{離貪欲過患，得生於諸天。}@
%{远离贪欲的过患，就能生到诸天。}%

@{若人離貪欲，心不為境牽，}@
%{如果一个人远离贪欲，内心不被境相牵引，}%

@{脫怖畏險難，得生於諸天。}@
%{就能摆脱恐惧和险难，生到诸天。}%

@{遠親眷朋屬，彼互相纏縛，}@
%{远离亲人、朋友和眷属，他们互相缠缚，}%

@{單己而修行，得生於諸天。}@
%{独自修行，就能生到诸天。}%

@{具足殊勝行，安住於淨慧，}@
%{具足殊胜的修行，安住于清净智慧，}%

@{善降伏貪欲，得生於諸天。}@
%{善于降伏贪欲，就能生到诸天。}%

@{三業離毀犯，樂修諸禪定，}@
%{三业远离毁犯，喜欢修持禅定，}%

@{為眾所稱讚，得生於諸天。}@
%{受到大众的赞叹，就能生到诸天。}%

@{棄背惡知識，遠離愛毒箭，}@
%{抛弃恶知识，远离爱的毒箭，}%

@{不為女索縛，得生於諸天。}@
%{不被女色所束缚，就能生到诸天。}%

@{善離彼愛纏，如劍斷朽索，}@
%{善于远离爱的缠缚，就像宝剑斩断朽绳，}%

@{安隱離諸怖，得生於諸天。}@
%{安稳地远离一切恐惧，就能生到诸天。}%

@{慧力極堅勇，常專求正法，}@
%{智慧的力量非常坚强勇敢，常专心寻求正法，}%

@{修施戒禪定，得生於諸天。}@
%{修持布施、持戒、禅定，就能生到诸天。}%

@{又彼諸眾生，廣修其善行，}@
%{那些众生，广泛修持善行，}%

@{感果生天中，諸天常恭敬。}@
%{感果生到天界，诸天常恭敬他们。}%

@{人作善生天，天修福為人，}@
%{人做善事生到天界，天人修福为人类，}%

@{常依於正法，互相有力能。}@
%{常依正法，互相帮助，互相有力量。}%

@{或從天退沒，或餘趣生天，}@
%{有时从天界退没，有时从其他趣生到天界，}%

@{若離彼善業，則趣三惡道。}@
%{如果远离那些善业，就会堕入三恶道。}%

@{諸寂靜善法，為樂之因本，}@
%{一切寂静的善法，都是快乐的根本原因，}%

@{乃至於夢中，不應捨善法。}@
%{甚至在梦中，也不应该舍弃善法。}%

@{善法滅非法，真實摧虛妄，}@
%{善法消灭非法，真实摧毁虚妄，}%

@{諸天降非天，智慧破愚鈍。}@
%{诸天降伏非天，智慧破除愚痴。}%

@{善法為階梯，智者能昇蹈，}@
%{善法是阶梯，智者能够攀登，}%

@{往趣諸天中，受上妙快樂。}@
%{前往诸天，享受上妙的快乐。}%

@{謂身語七支，不行殺盜等，}@
%{所谓身语七支，不杀盗等，}%

@{此七能為梯，得生於天界。}@
%{这七种能作为阶梯，使人能够生到天界。}%

@{由淨業嚴身，光潔極可愛，}@
%{因为清净的业力庄严身体，光洁无比，非常可爱，}%

@{猶如彼明燈，從自身而發。}@
%{就像明亮的灯，从自身发出。}%

@{是故諸天人，皆以戒為本，}@
%{所以诸天人，都以持戒为根本，}%

@{常生適悅心，受樂無窮極。}@
%{内心常生欢喜，享受无穷无尽的快乐。}%

@{諸天女侍衛，如眾星拱月，}@
%{诸天女侍卫，像众星拱月，}%

@{遊戲於天中，皆善因所得。}@
%{在天界游玩，都是善因所得。}%

@{隨所欲從心，得已無減失，}@
%{随心所欲，得到之后没有损失，}%

@{彼樂常增長，皆善因所得。}@
%{快乐常增长，都是善因所得。}%

@{善修布施行，憐愍諸有情，}@
%{善于修持布施，怜悯一切有情，}%

@{與慈心相應，得生於天上。}@
%{与慈悲心相伴随，就能生到天上。}%

@{遠離殺生罪，不害諸眾生，}@
%{远离杀生罪，不伤害一切众生，}%

@{與善行相應，得生於天上。}@
%{与善行相伴随，就能生到天上。}%

@{遠離不與取，而樂行布施，}@
%{远离不与取，喜欢行布施，}%

@{不悋惜少物，得生於天上。}@
%{不吝惜少物，就能生到天上。}%

@{棄捨欲邪行，常依彼正道，}@
%{抛弃欲望的邪恶行为，常依正道，}%

@{離垢心寂靜，得生於天上。}@
%{内心远离污垢，寂静安宁，就能生到天上。}%

@{離飲酒過失，意不生迷亂，}@
%{远离饮酒的过失，内心不生迷乱，}%

@{不為人輕笑，得生於天上。}@
%{不被别人轻笑，就能生到天上。}%

@{諸天受快樂，不應生放逸，}@
%{天人享受快乐，不应该生起放逸，}%

@{當求寂靜樂，彼則常安隱。}@
%{应该追求寂静的快乐，那样才能常得安稳。}%

@{受天中妙欲，及最勝宮殿，}@
%{享受天界妙欲，以及最胜的宫殿，}%

@{念念即無常，不久後當盡。}@
%{念念都是无常，不久之后就会消失。}%

@{彌盧極高勝，善業能過彼，}@
%{弥卢天极高胜，善业能够超越它，}%

@{乃至究竟天，匪善何能詣？}@
%{乃至究竟天，没有善业怎么能到达呢？}%

@{於欲境無厭，天中更殊勝，}@
%{对欲境没有厌倦，天界更加殊胜，}%

@{由愛轉增長，何由得寂靜？}@
%{因为爱而不断增长，怎么能够得到寂静呢？}%

@{諸天由愛故，著樂無休息，}@
%{诸天因为爱，执着于快乐，没有休息，}%

@{常為愛火燒，於樂何能得？}@
%{常被爱的火焰燃烧，怎么能够得到快乐呢？}%

@{作善有三品，三類為三因，}@
%{做善事有三品，三类为三种因，}%

@{三有三現行，三業感三果。}@
%{三种有三种现行，三业感得三种果报。}%

@{由汝樂善法，修施忍不害，}@
%{因为你喜爱善法，修持布施、忍辱、不伤害，}%

@{真實行相應，得生於天上。}@
%{真实地行持，就能生到天上。}%

@{具足勝莊嚴，花鬘寶纓絡，}@
%{具足胜妙的庄严，花鬘、宝珠、璎珞，}%

@{受樂於天中，皆由善業得。}@
%{在天界享受快乐，都是由善业所得。}%

@{又天中妙樂，上中下差別，}@
%{天界妙乐，有上中下差别，}%

@{如是三品因，隨福果所現。}@
%{像这样三种因，随福报的果报而显现。}%

@{隨修彼善行，不失二種報，}@
%{随修持那些善行，不会失去两种报应，}%

@{或人或天中，則受於快樂。}@
%{或者在人间，或者在天界，都会享受到快乐。}%

@{汝昔修福業，善護七支戒，}@
%{你过去修持福业，善于护持七支戒，}%

@{今來此天中，自受其樂果。}@
%{现在来到天界，享受着自己所修的快乐果报。}%

@{有妙蓮花池，生清涼香風，}@
%{有妙美的莲花池，散发清凉香风，}%

@{珍妙樓閣中，諸天共遊戲。}@
%{在珍贵的楼阁里，诸天共同游戏。}%

@{極殊妙金山，琉璃為峰頂，}@
%{极其殊妙的金山，琉璃为峰顶，}%

@{寶樹多花果，諸天共遊戲。}@
%{宝树上有很多花果，诸天共同游戏。}%

@{上妙劫波林，枝葉悉滋茂，}@
%{上妙的劫波林，枝叶都非常茂盛，}%

@{清泉繞其中，諸天共遊戲。}@
%{清泉环绕其中，诸天共同游戏。}%

@{復有七寶山，河流四圍繞，}@
%{还有七宝山，河流环绕四周，}%

@{金沙布其底，諸天共遊戲。}@
%{金沙铺在山底，诸天共同游戏。}%

@{青蓮花妙香，曼陀羅花林，}@
%{青莲花散发妙香，曼陀罗花林，}%

@{眾鳥出妙音，諸天共遊戲。}@
%{各种鸟儿发出美妙的歌声，诸天共同游戏。}%

@{復有餘勝處，皆眾寶林木，}@
%{还有其他更加胜妙的地方，都是各种宝树，}%

@{宮殿寶莊嚴，諸天共遊戲。}@
%{宫殿宝珠庄严，诸天共同游戏。}%

@{微妙五音樂，及最勝歌舞，}@
%{微妙的五种音乐，以及最胜的歌舞，}%

@{聞已咸適悅，諸天共遊戲。}@
%{听到之后，都感到欢喜，诸天共同游戏。}%

@{戒為其種子，出生諸樂果，}@
%{持戒是种子，生出各种快乐的果实，}%

@{於上妙五欲，隨心而受用。}@
%{在天界上妙的五欲中，随心所欲地享受。}%

@{諸天光明鬘，殊勝淨無垢，}@
%{诸天光明的光环，殊胜清净无垢，}%

@{乃至諸快樂，皆善因所感。}@
%{乃至一切快乐，都是善因所感。}%

@{與諸天遊戲，受無量快樂，}@
%{与诸天游戏，享受无量快乐，}%

@{如是快樂因，此因非我作。}@
%{像这样的快乐因，这个因不是我做的。}%

@{最勝妙樓閣，眾寶而莊嚴，}@
%{最胜妙的楼阁，用各种宝珠装饰，}%

@{由夙植善因，得其中安住。}@
%{因为过去种下的善因，才能在其中安住。}%

@{受廣大五欲，不生染著心，}@
%{享受广大的五欲，不生染着之心，}%

@{由離三種縛，為帝釋天王。}@
%{因为远离三种束缚，成为帝释天王。}%

@{若人心質直，修定除散亂，}@
%{如果一个人内心质朴，修持禅定，消除散乱，}%

@{來生此天中，自業為現證。}@
%{来世生到天界，自己所造的业力就会显现。}%

@{廣修眾善業，受樂常相續，}@
%{广泛修持各种善业，享受快乐常相续，}%

@{悟昔所行因，今來受斯果。}@
%{明白过去所做的一切因，现在来享受这个果报。}%

@{此現生修善，猶如植林木，}@
%{现在生中修持善法，就像种植树木，}%

@{令相續滋榮，是人為智者。}@
%{让它不断繁荣昌盛，这个人就是智者。}%

@{於在所生處，作善或不善，}@
%{在所生之处，做善事或不善事，}%

@{各各如其因，而受種種報。}@
%{都会根据其因，而接受各种果报。}%

@{若人修善因，得生於天中，}@
%{如果一个人修持善因，就能生到天界，}%

@{造彼不善因，當墮於地獄。}@
%{造作不善因，就会堕落到地狱。}%

@{愚癡著欲人，捨善作不善，}@
%{愚痴执着欲望的人，舍弃善法，做不善事，}%

@{不怖命終時，彼為自損壞。}@
%{不害怕生命终结的时候，他们就是自己伤害自己。}%

@{善不善業報，如種各滋長，}@
%{善业和不善业的果报，就像种子各自生长，}%

@{愚夫心著樂，而未甞暫捨。}@
%{愚蠢的人内心执着快乐，却从来没有暂时舍弃过。}%

@{修三品善業，觀身語七支，}@
%{修持三品善业，观察身语七支，}%

@{解脫彼三毒，得生於天中。}@
%{就能摆脱三毒，生到天界。}%

@{於欲境不動，亦不生讚美，}@
%{面对欲境，内心不动，也不生起赞美，}%

@{住離染清淨，得生於天中。}@
%{住于远离染污的清净，就能生到天界。}%

@{善修種種施，治彼慳悋心，}@
%{善于修持各种布施，治疗悭吝之心，}%

@{越諸苦難處，得生於天中。}@
%{就能超越一切苦难，生到天界。}%

@{不害眾生命，而常生愛護，}@
%{不伤害众生，而常生爱护之心，}%

@{住寂靜慈心，得生於天中。}@
%{住于寂静的慈悲心，就能生到天界。}%

@{遠離不與取，其心如熾火，}@
%{远离不与取，内心像炽热的火焰，}%

@{安住正思惟，得生於天中。}@
%{安住于正思维，就能生到天界。}%

@{不染欲淤泥，見他色如母，}@
%{不染欲望的泥潭，看到他人的色相就像母亲，}%

@{以慧善觀察，得生於天中。}@
%{用智慧善于观察，就能生到天界。}%

@{自心懷獄火，從舌薪發起，}@
%{内心怀着地狱之火，从舌头上的柴火燃起，}%

@{斯由說妄言，離此生天中；}@
%{这是因为说妄语，远离它就能生到天界；}%

@{由舌無真實，指虛而談有。}@
%{因为舌头没有真实，指着虚无说有。}%

@{離詭飾言詞，得生於天中。}@
%{远离虚假欺骗的言词，就能生到天界。}%

@{常遠彼兩舌，慈心相愛敬，}@
%{常远离两舌，慈悲心互相爱敬，}%

@{不說離間語，得生於天中。}@
%{不说离间语，就能生到天界。}%

@{惡語如刀杖，智者當遠離，}@
%{恶语就像刀杖，智者应该远离，}%

@{常出美妙言，得生於天中。}@
%{常说出美妙的语言，就能生到天界。}%

@{善護斯七戒，則能生諸天，}@
%{善于护持这七支戒，就能生到诸天，}%

@{智者當了知，此諸佛所說。}@
%{智者应该明白，这是诸佛所说。}%

@{以眾善莊嚴，得生於天上，}@
%{用各种善法庄严自己，就能生到天上，}%

@{若廢昔修行，彼則生後悔。}@
%{如果废弃过去的修行，就会生起后悔。}%

@{天上妙林藤，修蔓四垂布，}@
%{天上的妙林藤，枝蔓四面垂下，}%

@{好花香莊嚴，諸天憩其下。}@
%{美丽的鲜花散发香气，庄严美丽，诸天休息在下面。}%

@{若修眾善行，當獲諸快樂，}@
%{如果修持各种善行，就会获得各种快乐，}%

@{往趣於天中，得見如是果。}@
%{前往天界，就能看到这样的果报。}%

@{又彼諸天眾，受三品快樂，}@
%{那些天界众生，享受三品快乐，}%

@{各各如先業，而得彼彼果。}@
%{每个人都根据过去的业力，而获得相应的果报。}%

@{因與果相似，而不肯修因，}@
%{因和果相似，却不愿意修因，}%

@{因果善了知，當獲於樂分。}@
%{因果关系要善于明白，才能获得快乐的份额。}%

@{勝妙五欲境，諸天生樂著，}@
%{胜妙的五欲境，诸天生起快乐和执着，}%

@{唯仁恕和平，當得為天主。}@
%{只有仁慈宽恕，才能成为天主。}%

@{不造諸罪惡，不妄生喜慍，}@
%{不造罪业，不无缘无故地喜怒哀乐，}%

@{心離彼垢染，當得為天主。}@
%{内心远离污垢，就能成为天主。}%

@{當決擇思惟，然後隨所作，}@
%{应该慎重思考，然后才去做事，}%

@{依正法而行，當得為天主。}@
%{按照正法来行事，就能成为天主。}%

@{不嗜諸飲食，常樂於正法，}@
%{不贪图美食，经常喜欢正法，}%

@{獲清淨輕安，為智中智者。}@
%{获得清净安乐，成为智慧中的智慧者。}%

@{由解彼正法，愛育於黎民，}@
%{因为懂得正法，所以爱护百姓，}%

@{彼王具福慧，天龍常守護；}@
%{这样的君王拥有福德和智慧，天龙常来守护；}%

@{輪迴極長遠，如絲緒不絕，}@
%{轮回非常漫长，就像一根线一样不断，}%

@{若入解正法，於彼善超越。}@
%{如果能够理解正法，就能超越轮回。}%

@{於如來所說，十善真實法，}@
%{佛陀所宣说的十善法，是真实的，}%

@{彼王能奉行，依法而治世，}@
%{如果君王能够奉行十善法，就能依法治理天下，}%

@{善達因緣法，及福非福業，}@
%{懂得因缘法，以及福报和非福报的业力，}%

@{見色離其貪，常生大覺悟；}@
%{看到色相，就能远离贪欲，经常生起大觉悟；}%

@{以上妙諸物，奉施於如來，}@
%{用最好的东西供养佛陀，}%

@{由是得人天，展轉常恭敬。}@
%{就能获得人天福报，不断地恭敬佛陀。}%

@{清淨心無染，護后妃眷屬，}@
%{内心清净无染，保护妻子和眷属，}%

@{遠離彼邪非，知足無憂惱。}@
%{远离邪恶，知足常乐，没有忧愁烦恼。}%

@{凡夫為境牽，智者心無垢，}@
%{凡夫会被外境所牵引，有智慧的人内心没有污垢，}%

@{當樂修正行，於戒能守護。}@
%{应该喜欢修习正行，能够守护戒律。}%

@{離彼毀戒者，憐念諸善人，}@
%{远离那些破坏戒律的人，怜悯那些善良的人，}%

@{住正見思惟，常樂於法樂。}@
%{住于正见，经常思考，享受法喜。}%

@{以正法治國，護大臣人民，}@
%{用正法来治理国家，保护大臣和人民，}%

@{彼王於世間，等諸天無異。}@
%{这样的君王在世间，与天人没有区别。}%

@{由王修淨德，臣佐依正行，}@
%{因为君王修习清净的品德，大臣也按照正法行事，}%

@{民庶悉清淨，如月麗秋空。}@
%{百姓都清净无染，就像秋天的月亮一样美丽。}%

@{了知因果相，則無相攻罰，}@
%{明白因果关系，就不会互相攻伐惩罚，}%

@{一切處吉祥，自他獲安隱。}@
%{一切地方都吉祥，自身和他人都能获得安宁。}%
@{稱讚功德品第三十六}@
%{赞叹功德品第三十六}%

@{歸依正遍知，為世間之父，}@
%{皈依正遍知，他是世间之父，}%

@{能斷三有縛，令登於覺路。}@
%{能够断除三有束缚，让人登上了觉悟之路。}%

@{歸依淨智眼，能破諸疑暗，}@
%{皈依清净智慧之眼，能够破除一切疑暗，}%

@{善摧眾異論，令住於正見。}@
%{善于摧毁各种邪说，让人住于正见。}%

@{歸依良福田，滋榮諸善果，}@
%{皈依良好的福田，滋养培育一切善果，}%

@{離三毒過患，令離垢清淨。}@
%{远离三毒的过患，让人远离垢染，清净无染。}%

@{歸依最上慧，住勝三摩地，}@
%{皈依最上的智慧，住于胜妙的三摩地，}%

@{以最勝法寶，為眾生開示。}@
%{以最胜的法宝，为众生开示。}%

@{稱讚佛世尊，相好諸功德，}@
%{赞叹佛世尊，赞叹他相好的一切功德，}%

@{能令彼見者，適悅心清淨。}@
%{能够让见到他的人，心生欢喜，心清净。}%

@{若人意清淨，常禮敬諸佛，}@
%{如果一个人心清净，经常礼敬诸佛，}%

@{獲最上吉祥，得離諸恐怖。}@
%{就会获得最上的吉祥，远离一切恐怖。}%

@{若人意清淨，善說微妙法，}@
%{如果一个人心清净，善于说微妙法，}%

@{能至於菩提，獲畢竟安隱。}@
%{就能到达菩提，获得究竟的安稳。}%

@{若人意清淨，常起正思惟，}@
%{如果一个人心清净，经常起正思维，}%

@{乘彼智慧舟，能渡於彼岸。}@
%{乘着智慧之舟，就能渡到彼岸。}%

@{智眼最清淨，能矚於幽顯，}@
%{智慧之眼最清净，能够洞察幽冥显现，}%

@{自他情非情，普遍無不盡。}@
%{自己、他人、情、非情，无一不尽。}%

@{若離貪等過，心不生惛濁，}@
%{如果远离贪等过患，心不生昏浊，}%

%{就像清净的琉璃，一切都能明了。}%
