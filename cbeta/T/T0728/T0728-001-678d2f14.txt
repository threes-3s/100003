@{諸法集要經卷第一}@
%{诸法集要经卷第一}%

@{觀無畏尊者集(總二千六百八十四頌)}@
%{观无畏尊者集(总二千六百八十四颂) }%

@{西天譯經三藏朝散大夫試鴻臚卿宣梵大師賜紫沙門臣日稱等奉詔譯}@
%{西天译经三藏朝散大夫试鸿胪卿宣梵大师赐紫沙门臣日称等奉诏译}%

@{伏除煩惱品第一}@
%{伏除烦恼品第一}%

@{稽首三有最勝尊，吉祥無垢盡諸漏，}@
%{稽首三有最胜尊，吉祥无垢尽诸漏，}%
@{愚夫為惑所沈溺，能以等慈而拔濟。}@
%{愚夫为惑所沉溺，能以等慈而拔济。}%

@{依正法念處，廣大契經海，}@
%{依正法念处，广大契经海，}%
@{集成此伽陀，為作世間眼。}@
%{集成此伽陀，为作世间眼。}%

@{當發生淨信，精進不放逸，}@
%{当发生净信，精进不放逸，}%
@{棄背於六塵，修習微妙智，}@
%{弃背于六尘，修习微妙智，}%

@{樂布施持戒，忍辱諸禪定，}@
%{乐布施持戒，忍辱诸禅定，}%
@{以四無量心，利樂諸含識。}@
%{以四无量心，利乐诸含识。}%

@{若散亂放逸，起暴惡邪見，}@
%{若散乱放逸，起暴恶邪见，}%
@{常發虛妄言，當墮於地獄。}@
%{常发虚妄言，当堕于地狱。}%

@{若慳悋憎嫉，遠離於善法，}@
%{若悭吝憎嫉，远离于善法，}%
@{樂損惱眾生，定墮餓鬼趣。}@
%{乐损恼众生，定堕饿鬼趣。}%

@{若破壞正法，愚癡著染欲，}@
%{若破坏正法，愚痴着染欲，}%
@{耽飲食睡眠，當獲畜生報。}@
%{耽饮食睡眠，当获畜生报。}%

@{身語意三種，造作諸不善，}@
%{身语意三种，造作诸不善，}%
@{為業之所纏，數數而增長。}@
%{为业之所缠，数数而增长。}%

@{如是彼愚夫，展轉無休息，}@
%{如是彼愚夫，辗转无休息，}%
@{智者善修習，當得寂靜樂。}@
%{智者善修习，当得寂静乐。}%

@{了知是業果，棄捨於放逸，}@
%{了知是业果，弃舍于放逸，}%
@{以智慧揀擇，此善無過上。}@
%{以智慧拣择，此善无过上。}%

@{常樂修惠施，堅固持淨戒。}@
%{常乐修惠施，坚固持净戒。}%

@{憐愍諸有情，令所願成就，}@
%{怜愍诸有情，令所愿成就，}%
@{與慈悲相應，不生於希望，}@
%{与慈悲相应，不生于希望，}%

@{為攝受於他，決定得成就。}@
%{为摄受于他，决定得成就。}%

@{解脫彼貪瞋，知足離過患，}@
%{解脱彼贪瞋，知足离过患，}%
@{愛念於眾生，決定得成就。}@
%{爱念于众生，决定得成就。}%

@{以平等質直，無有冤親想，}@
%{以平等质直，无有冤亲想，}%
@{永脫諸惡趣，決定得成就。}@
%{永脱诸恶趣，决定得成就。}%

@{具足諸威儀，善平等說法，}@
%{具足诸威仪，善平等说法，}%
@{離慳嫉過失，決定得成就。}@
%{离悭嫉过失，决定得成就。}%

@{樂承事師尊，持戒修諸定，}@
%{乐承事师尊，持戒修诸定，}%
@{明了罪福相，決定得成就。}@
%{明了罪福相，决定得成就。}%

@{起勇猛精進，坐禪若讀誦，}@
%{起勇猛精进，坐禅若读诵，}%
@{愛語無諂曲，決定得成就。}@
%{爱语无谄曲，决定得成就。}%

@{於時及彼方，或作或止息，}@
%{于时及彼方，或作或止息，}%
@{皆方便了知，決定得成就。}@
%{皆方便了知，决定得成就。}%

@{無瞋伏瞋恚，忍辱除暴惡，}@
%{无瞋伏瞋恚，忍辱除暴恶，}%
@{正法捨非法，光明滅黑暗，}@
%{正法舍非法，光明灭黑暗，}%

@{真實遣虛妄，寂靜摧兩舌，}@
%{真实遣虚妄，寂静摧两舌，}%
@{憐愍息毀呰，無縛解纏縛，}@
%{怜愍息毁呰，无缚解缠缚，}%

@{慈心止殺害，樂施銷慳垢。}@
%{慈心止杀害，乐施销悭垢。}%

@{以淨善對治，不如理作意，}@
%{以净善对治，不如理作意，}%
@{以智照無明，無常破常執。}@
%{以智照无明，无常破常执。}%

@{猶日昃月虧，遷流本如是，}@
%{犹日昃月亏，迁流本如是，}%
@{當自善思惟，觀察離邪欲。}@
%{当自善思惟，观察离邪欲。}%

@{八聖道為因，能越諸惡趣，}@
%{八圣道为因，能越诸恶趣，}%
@{住四無所畏，能降諸恐怖，}@
%{住四无所畏，能降诸恐怖，}%

@{正念祛妄念，勝智摧邪智，}@
%{正念祛妄念，胜智摧邪智，}%
@{樂住阿蘭若，淡泊絕貪欲。}@
%{乐住阿兰若，淡泊绝贪欲。}%

@{譬如眾山中，須彌為最勝，}@
%{譬如众山中，须弥为最胜，}%
@{如大海深廣，能生諸珍寶；}@
%{如大海深广，能生诸珍宝；}%
@{如皎日光明，映蔽諸宿曜。}@
%{如皎日光明，映蔽诸宿曜。}%

@{無生止有生，如火然槁木，}@
%{无生止有生，如火然槁木，}%
@{亦如妙飲食，能除於飢渴。}@
%{亦如妙饮食，能除于饥渴。}%

@{如師子進止，能伏諸群獸，}@
%{如狮子进止，能伏诸群兽，}%
@{知足絕貪求，無畏亦如是。}@
%{知足绝贪求，无畏亦如是。}%

@{仁慈世共稱，捨此咸輕鄙，}@
%{仁慈世共称，舍此咸轻鄙，}%
@{懈怠與顓愚，精進能除斷。}@
%{懈怠与颛愚，精进能除断。}%

@{如有暴惡人，非理相加謗，}@
%{如有暴恶人，非理相加谤，}%
@{智者以誠言，安忍能除遣。}@
%{智者以诚言，安忍能除遣。}%

@{諸天降脩羅，由信樂正法，}@
%{诸天降修罗，由信乐正法，}%
@{佛出於世間，能制諸外道。}@
%{佛出于世间，能制诸外道。}%

@{說法品第二}@
%{说法品第二}%

@{若人善說法，能使彼開解，}@
%{若人善说法，能使彼开解，}%
@{為眾之導師，令至安隱處，}@
%{为众之导师，令至安隐处，}%

@{示菩提正路，畢竟令趣入，}@
%{示菩提正路，毕竟令趣入，}%
@{生死險道中，永斷癡纏縛。}@
%{生死险道中，永断痴缠缚。}%

@{是法無過上，非世俗所說，}@
%{是法无过上，非世俗所说，}%
@{若有聞是者，能渡諸有海。}@
%{若有闻是者，能渡诸有海。}%

@{若有智慧人，於此勤修習，}@
%{若有智慧人，于此勤修习，}%
@{有四種福田，能生諸善果。}@
%{有四种福田，能生诸善果。}%

@{若奉持佛教，得諸根具足，}@
%{若奉持佛教，得诸根具足，}%
@{不生愛樂心，於後徒悔惱。}@
%{不生爱乐心，于后徒悔恼。}%

@{若人於五欲，常追求耽染，}@
%{若人于五欲，常追求耽染，}%
@{戀著於妻孥，當墮於惡趣。}@
%{恋著于妻孥，当堕于恶趣。}%

@{起不正思惟，積集諸過失，}@
%{起不正思惟，积集诸过失，}%
@{皆由於自心，為妄想縈縶。}@
%{皆由于自心，为妄想萦縶。}%

@{是心難降伏，多攀緣欲境，}@
%{是心难降伏，多攀缘欲境，}%
@{若能善制之，獲清涼安隱。}@
%{若能善制之，获清凉安隐。}%

@{是心如惡馬，以正法調伏，}@
%{是心如恶马，以正法调伏，}%
@{聞已當憶持，數數而觀察。}@
%{闻已当忆持，数数而观察。}%

@{若人聞正法，聞已悉明了，}@
%{若人闻正法，闻已悉明了，}%
@{發生於善根，遠離諸過咎。}@
%{发生于善根，远离诸过咎。}%

@{由聞正法故，心淨無垢染，}@
%{由闻正法故，心净无垢染，}%
@{生踊躍歡喜，增長於明慧。}@
%{生踊跃欢喜，增长于明慧。}%

@{由聞正法故，不造作眾罪，}@
%{由闻正法故，不造作众罪，}%
@{知業果不虛，當得菩提道。}@
%{知业果不虚，当得菩提道。}%

@{由聞正法故，知佛諸功德，}@
%{由闻正法故，知佛诸功德，}%
@{法為解脫因，是為真智者。}@
%{法为解脱因，是为真智者。}%

@{由聞正法故，知法相常住，}@
%{由闻正法故，知法相常住，}%
@{是故當一心，於事勤修作。}@
%{是故当一心，于事勤修作。}%

@{由聞正法故，解脫輪迴海，}@
%{由闻正法故，解脱轮回海，}%
@{斷種種貪愛，當證於實際。}@
%{断种种贪爱，当证于实际。}%

@{由聞正法故，悟彼生滅相，}@
%{由闻正法故，悟彼生灭相，}%
@{具四種因緣，當明了信解。}@
%{具四种因缘，当明了信解。}%

@{由聞正法故，了知蘊處界，}@
%{由闻正法故，了知蕴处界，}%
@{與生滅相應，令正智明顯。}@
%{与生灭相应，令正智明显。}%

@{是三種過患，為輪迴之本，}@
%{是三种过患，为轮回之本，}%
@{樂多聞正法，當斷令永盡。}@
%{乐多闻正法，当断令永尽。}%

@{若樂於多聞，處世無過上，}@
%{若乐于多闻，处世无过上，}%
@{於動不動法，悉究諸源底。}@
%{于动不动法，悉究诸源底。}%

@{是人命終時，無復諸憂怖，}@
%{是人命终时，无复诸忧怖，}%
@{善達彼正法，不生於少苦。}@
%{善达彼正法，不生于少苦。}%

@{能以正智火，焚燒煩惱薪，}@
%{能以正智火，焚烧烦恼薪，}%
@{由樂於多聞，後苦不復受。}@
%{由乐于多闻，后苦不复受。}%

@{若親近多聞，則生安隱樂，}@
%{若亲近多闻，则生安隐乐，}%
@{離放逸燒然，此為善根本。}@
%{离放逸烧然，此为善根本。}%

@{當承事耆德，欣樂彼宣說，}@
%{当承事耆德，欣乐彼宣说，}%
@{出離老死因，得證真常處。}@
%{出离老死因，得证真常处。}%

@{了達一切法，解脫諸障染，}@
%{了达一切法，解脱诸障染，}%
@{引發菩提心，多聞為最上。}@
%{引发菩提心，多闻为最上。}%

@{若習近多聞，樂欲修正慧，}@
%{若习近多闻，乐欲修正慧，}%
@{當以身語心，尊重常恭敬。}@
%{当以身语心，尊重常恭敬。}%

@{若樂多聞者，善住於法性，}@
%{若乐多闻者，善住于法性，}%
@{堅固勤修作，能越三有海。}@
%{坚固勤修作，能越三有海。}%

@{若人有多聞，則具諸財寶，}@
%{若人有多闻，则具诸财宝，}%
@{無聞雖富饒，愚懵同貧窶。}@
%{无闻虽富饶，愚懵同贫窶。}%

@{若人無法財，遠離於師範，}@
%{若人无法财，远离于师范，}%
@{虛受彼形軀，常懷於憂赧。}@
%{虚受彼形躯，常怀于忧赧。}%

@{若近惡知識，生放逸懈怠，}@
%{若近恶知识，生放逸懈怠，}%
@{猶如磽田中，虛擲於種子。}@
%{犹如磽田中，虚掷于种子。}%

@{多聞具法眼，雖瞽亦明覺，}@
%{多闻具法眼，虽瞽亦明觉，}%
@{無目無多聞，是為暗鈍者。}@
%{无目无多闻，是为暗钝者。}%

@{若遠離正法，依止於非法，}@
%{若远离正法，依止于非法，}%
@{猶如捨良醫，而求愈篤疾。}@
%{犹如舍良医，而求愈笃疾。}%

@{諸法無限量，積學方悟入，}@
%{诸法无限量，积学方悟入，}%
@{滴雨成駛流，皆由於漸次。}@
%{滴雨成驶流，皆由于渐次。}%

@{無始輪迴海，發起菩提心，}@
%{无始轮回海，发起菩提心，}%
@{至金剛道場，成佛果亦爾。}@
%{至金刚道场，成佛果亦尔。}%

@{淨心持正法，不著於諸禪，}@
%{净心持正法，不著于诸禅，}%
@{非欲境所牽，決定常安隱。}@
%{非欲境所牵，决定常安隐。}%

@{有智親智人，當捨離無智，}@
%{有智亲智人，当舍离无智，}%
@{以智德修身，斯人甚希有！}@
%{以智德修身，斯人甚希有！}%

@{以信心求法，常生於勝處，}@
%{以信心求法，常生于胜处，}%
@{設墮險難中，諸天常捄護。}@
%{设堕险难中，诸天常捄护。}%

@{於暗作明燈，於病為良藥，}@
%{于暗作明灯，于病为良药，}%
@{貧乏與珍財，盲者使能視。}@
%{贫乏与珍财，盲者使能视。}%

@{於世間瀑流，為作彼舡筏，}@
%{于世间瀑流，为作彼舡筏，}%
@{若醉傲放逸，決定為自損。}@
%{若醉傲放逸，决定为自损。}%

@{是先佛所說，當具足信受，}@
%{是先佛所说，当具足信受，}%
@{令正智現前，修習忘疲倦。}@
%{令正智现前，修习忘疲倦。}%

@{厭離自身品第三}@
%{厌离自身品第三}%

@{謂利養名聞、飲食臥具等，}@
%{就是说，对于钱财名声、吃喝住行这些东西，}%

@{無少分希求，於我何所作？}@
%{一点儿都不想拥有，对我来说有什么用呢？}%

@{是身可厭患，損害如冤賊，}@
%{这身体真是令人厌烦，就像仇敌一样害我，}%

@{造作諸過愆，常樂非梵行。}@
%{经常造作各种罪过，还以为是快乐，这不是正道。}%

@{又復此身者，為眾病依止，}@
%{再说，这身体是各种疾病的住处，}%

@{不淨常盈流，實罪惡之器；}@
%{污秽不断流淌，实际上是罪恶的工具；}%

@{與心為近住，無思惟覺知，}@
%{和心紧紧相连，没有思考觉知，}%

@{剎那命終時，惡色深可畏。}@
%{一瞬间生命终结的时候，恐怖的颜色非常可怕。}%

@{壽煖識三緣，俱時而棄捨，}@
%{寿命、温暖、意识这三种缘分，同时被抛弃，}%

@{如枯木無知，形消流穢汁。}@
%{就像枯木没有知觉，形体消散，流出污秽的汁液。}%

@{愚夫當盛年，迷亂多憍恣，}@
%{愚蠢的人正值壮年，迷迷糊糊，骄纵放肆，}%

@{須臾不暫停，變異成衰老。}@
%{片刻不停，就变成了衰老。}%

@{恃財富奢逸，廣造諸惡業，}@
%{依靠财富奢侈放纵，广造各种恶业，}%

@{是人命欲終，受極苦熱惱。}@
%{这样的人，生命将要终结的时候，会承受极大的痛苦煎熬。}%

@{不樂於正法，何異彼非人？}@
%{不喜欢正法，跟那些非人有什么区别？}%

@{違背涅盤城，棲止於邪道。}@
%{违背涅槃城，栖息在邪道上。}%

@{廣大修福報，由是生人間，}@
%{广修福报，因此出生在人间，}%

@{當乘智慧舟，永越於有海。}@
%{应该乘着智慧的船，永远渡过这有海。}%

@{是身如掣電，類乾闥婆城，}@
%{这身体就像闪电一样，如同乾闼婆城，}%

@{云何於他人，數生於喜怒？}@
%{为什么对别人，反复产生喜怒呢？}%

@{此身非堅牢，暫時而動轉，}@
%{这身体并不坚固，暂时存在，不断变化，}%

@{常諂曲為心，寧免老死怖；}@
%{经常奉承曲意讨好内心，难道就能免除生老病死的恐惧吗？}%

@{為疾病城邑，是憂惱舍宅，}@
%{是疾病的城邑，是忧愁烦恼的住所，}%

@{亦如於田疇，生善不善種。}@
