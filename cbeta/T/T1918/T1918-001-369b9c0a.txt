@{四念處卷第一}@
%{四念处卷第一}%

@{隋天台山修禪寺智者大師說}@
%{隋天台山修禅寺智者大师说}%

@{門人章安灌頂記}@
%{门人章安灌顶记}%

@{一切諸法皆不可思議。不可思想圖度。不可言語商略。何以故。言語道斷故不可議。心行處滅故不可思。}@
%{一切诸法都不可思議，不可思想图度，不可言语商略。为什么呢？因为言语道断，所以不可议；心行处灭，所以不可思。}%

@{大經云。生生不可說。生不生不可說。不生生不可說。不生不生不可說。既不可說亦不可思。}@
%{大经云：生生不可说，生不生不可说，不生生不可说，不生不生不可说。既然不可说，也就不可以思。}%

@{大品云。色不可說。乃至識不可說。眼不可說。乃至意不可說。色不可說。乃至法不可說。眼界不可說。乃至法界不可說。}@
%{大品云：色不可说，乃至识不可说，眼不可说，乃至意不可说，色不可说，乃至法不可说，眼界不可说，乃至法界不可说。}%

@{當知五陰十二入。十八界。皆不可說。此指俗諦不可說也。}@
%{当知五阴十二入、十八界，皆不可说。这里指的是俗谛不可说。}%

@{四念處不可說。乃至根力覺道。皆不可說。須陀洹不可說。乃至阿羅漢亦不可說。此指真諦不可說。}@
%{四念处不可说，乃至根力觉道，皆不可说，须陀洹不可说，乃至阿罗汉亦不可说。这里指的是真谛不可说。}%

@{佛十力不可說。四無畏。十八不共。三十二相。八十種好等。皆不可說。此指中道第一義諦不可說也。}@
%{佛十力不可说，四无畏，十八不共，三十二相，八十种好等，皆不可说。这里指的是中道第一义谛不可说。}%

@{大論云。實法不顛倒。念想觀已除。言語法皆滅。無量眾罪除。清淨心常一。如是尊妙人。則能見般若。}@
%{大论云：实法不颠倒，念想观已除，言语法皆灭，无量众罪除，清净心常一，如是尊妙人，则能见般若。}%

@{此總指三諦不可思說也。}@
%{这里总指的是三谛不可思说。}%

@{法華云。諸法寂滅相不可以言宣。}@
%{法华云：诸法寂灭相不可以言宣。}%

@{又云。是法非思量分別之所能解。}@
%{又云：是法非思量分别之所能解。}%

@{當知不可以心思口議。亦不可以無心口思議。}@
%{当知不可以心思口议，亦不可以无心口思議。}%

@{不可以有無。不可以非有。非無。非方。非數。非法。非諭。寂然無為群經之極說。眾聖之誠言。深深若此。明明若彼。}@
%{不可以有無，不可以非有，非无，非方，非数，非法，非谕。寂然无为群经之极说，众圣之诚言，深深若此，明明若彼。}%

@{境智雙冥能所俱寂。}@
%{境智双冥，能所俱寂。}%

@{淨名諸菩薩以言言於無言。}@
%{净名诸菩萨以言言于无言。}%

@{文殊以無言言於無言。}@
%{文殊以无言言于无言。}%

@{淨名不以言不以無言。默然無言。}@
%{净名不以言不以无言，默然无言。}%

@{文殊歎曰真入不二法門當知。玄妙玄妙不可思議。復不可思議但可冥悟。}@
%{文殊叹曰：真入不二法门当知，玄妙玄妙不可思議，复不可思議，但可冥悟。}%

@{不可以彰辯。}@
%{不可以彰辩。}%

@{問若不可說不可說。何故言滿龍宮。}@
%{问：若不可说不可说，何故言满龙宫？}%

@{若不可思不可思。何故雪山深思此義。}@
%{若不可思不可思，何故雪山深思此义？}%

@{答佛常樂寂而哀於蒙。欲訥於言而敏於行。}@
%{答：佛常乐寂，而哀于蒙，欲讷于言，而敏于行。}%

@{慈悲權巧指畫虛空。撝方示月作種種說。}@
%{慈悲权巧，指画虚空，撝方示月，作种种说。}%

@{或作生生說。或作生不生說。或作不生生說。或作不生不生說。}@
%{或作生生说，或作生不生说，或作不生生说，或作不生不生说。}%

@{若作生生說時。又非一種。或作有說或作無說。或作亦有亦無說。或作非有非無說。}@
%{若作生生说时，又非一种，或作有说或作无说，或作亦有亦无说，或作非有非无说。}%

@{引諸根性從四方門入清涼池聞悅善生惡亡理顯。得四利益餘三句亦如是。}@
%{引诸根性，从四方门入清凉池，闻悦善生恶亡理显，得四利益，余三句亦如是。}%

@{欲令此義明了。更引諸經。}@
%{欲令此义明了，更引诸经。}%

@{大經。名諸佛法界佛即果人。法界即果法。於無果中而作果說。}@
%{大经，名诸佛法界佛即果人，法界即果法，于无果中而作果说。}%

@{大品云。諸法實相慧。名摩訶般若。此慧有能度所度。今取能度名般若。於無因中而作因說。}@
%{大品云：诸法实相慧，名摩訶般若，此慧有能度所度，今取能度名般若，于无因中而作因说。}%

@{大集云。菩薩觀一切法平等。眾生性同涅槃性。若觀平等即因也。同涅槃即果也。}@
%{大集云：菩萨观一切法平等，众生性同涅槃性，若观平等即因也，同涅槃即果也。}%

@{此約亦因亦果說也。}@
%{此约亦因亦果说也。}%

@{華嚴云。遊心法界如虛空。則知諸佛之境界。遊心即因。佛境界即果如前說。}@
%{华严云：游心法界如虚空，则知诸佛之境界，游心即因，佛境界即果如前说。}%

@{法華云。是法不可示言辭相寂滅。此即非因非果說也。}@
%{法华云：是法不可示言辞相寂灭，此即非因非果说也。}%

@{法門甚眾廣說令智退。略則義不周。我今處中說方則四句。略則因果。}@
%{法门甚众，广说令智退，略则义不周，我今处中说，方则四句，略则因果。}%

@{聞而修行名之為因。與法相應名之為果。}@
%{闻而修行，名之为因，与法相应，名之为果。}%

@{約果更修故言因因。從因又得果故言果果。}@
%{约果更修，故言因因，从因又得果，故言果果。}%

@{若初若後究竟寂滅。故言非因非果。}@
%{若初若后，究竟寂灭，故言非因非果。}%

@{初生生四門亦作因果四說。乃至不生不生亦作四門。}@
%{初生生四门，亦作因果四说，乃至不生不生，亦作四门。}%

@{門門亦作因果之說。一一說中悅種種眾生。立種種善根。治種種罪垢。}@
%{门门亦作因果之说，一一说中悦种种众生，立种种善根，治种种罪垢。}%

@{從於種種入第一義。故初中後重說無咎。}@
%{从于种种入第一义，故初中后重说无咎。}%

@{說即是教。稟教修觀。以修觀故名修四念處。}@
%{说即是教，禀教修观，以修观故名修四念处。}%

@{此義廣可思之。}@
%{此义广可思之。}%

@{元佛出世為一大事因緣。開發眾生諸覺寶藏。}@
%{元佛出世为一大事因缘，开发众生诸觉宝藏。}%

@{譬如日出先照高山。先喜先利先治先益。}@
%{譬如日出先照高山，先喜先利先治先益。}%

@{大經云。若欲盛貯先用完淨。若欲耕墾先種肥良。若欲乘御先駕調壯。若欲教詔先教孝明。}@
%{大经云：若欲盛贮先用完净，若欲耕垦先种肥良，若欲乘御先驾调壮，若欲教诏先教孝明。}%

@{斯皆積習深厚煩惱障薄。先聞雷震先沐甘雨。先出籠樊先獲正觀。皆由往昔數數勤修。今世道成最初四益。}@
%{斯皆积习深厚，烦恼障薄，先闻雷震，先沐甘雨，先出笼樊，先获正观，皆由往昔数数勤修，今世道成最初四益。}%

@{其未度者更設方便而塗熨之。}@
%{其未度者，更设方便而涂熨之。}%

@{隱其無量神德以貧所樂法趣波羅奈。便有涅槃音。}@
%{隐其无量神德，以贫所乐法趣波罗奈，便有涅槃音。}%

@{法僧差別名則是從頓次漸而調熟之。}@
%{法僧差别名，则是从顿次渐而调熟之。}%

@{今從此義粗為四說說。即是教依教修觀。即是四種四念處所為。}@
%{今从此义粗为四说，说即是教，依教修观，即是四种四念处所为。}%

@{生生四念處。乃至不生不生四念處。亦名三藏通別圓四念處。}@
%{生生四念处，乃至不生不生四念处，亦名三藏通别圆四念处。}%

@{若三藏四念處。更為三。大意五停四念。}@
%{若三藏四念处，更为三，大意五停四念。}%

@{所言三者。其義有八。謂理教智斷行位因果。}@
%{所言三者，其义有八，谓理教智断行位因果。}%

@{理三者。聲聞謂理在正使外。緣覺謂理在習氣外。菩薩謂理在正習外。}@
%{理三者，声闻谓理在正使外，缘觉谓理在习气外，菩萨谓理在正习外。}%

@{三人出三種外。方乃見理故言理三也。}@
%{三人出三种外，方乃见理，故言理三也。}%

@{教三者。聲聞稟四諦。緣覺稟十二緣。菩薩稟六度。}@
%{教三者，声闻禀四谛，缘觉禀十二缘，菩萨禀六度。}%

@{聲聞修總相智。緣覺修別相智。菩薩修總別智。}@
%{声闻修总相智，缘觉修别相智，菩萨修总别智。}%

@{聲聞斷正。緣覺斷習。菩薩斷正習。}@
%{声闻断正，缘觉断习，菩萨断正习。}%

@{聲聞為自修戒定慧。緣覺為自修獨善寂。菩薩為眾生修六度五通。}@
%{声闻为自修戒定慧，缘觉为自修独善寂，菩萨为众生修六度五通。}%

@{聲聞住學無學。緣覺住無學。菩薩三僧祇登道場。}@
%{声闻住学无学，缘觉住无学，菩萨三僧祇登道场。}%

@{聲聞帶果行因。緣覺望果行因。菩薩伏惑行因。}@
%{声闻带果行因，缘觉望果行因，菩萨伏惑行因。}%

@{聲聞斷正如燒木為炭。緣覺斷習如燒木為灰。菩薩正習盡如燒木無灰炭。}@
%{声闻断正，如烧木为炭，缘觉断习，如烧木为灰，菩萨正习尽，如烧木无灰炭。}%

@{具此八三故言三也。}@
%{具此八三，故言三也。}%

@{藏者。謂修多羅藏。毘尼藏。阿毘曇藏。}@
%{藏者，谓修多罗藏，毘尼藏，阿毘曇藏。}%

@{修多羅藏。謂四阿含。增一阿含。說人天因果。長阿含破邪見。雜阿含明諸禪法。中阿含明諸深義。}@
%{修多罗藏，谓四阿含，增一阿含，说人天因果，长阿含破邪见，杂阿含明诸禅法，中阿含明诸深义。}%

@{具如彼(云云)。}@
%{具如彼(云云) 。}%

@{毘尼藏明持犯輕重。如律藏中說。}@
%{毘尼藏明持犯轻重，如律藏中说。}%

@{阿毘曇藏名無比分別無比。分別(云云)。}@
%{阿毘曇藏名无比分别无比，分别(云云) 。}%

@{大經云。此是契經甚深之義。此是戒律輕重之義。此是阿毘曇分別法句。}@
%{大经云：此是契经甚深之义，此是戒律轻重之义，此是阿毘曇分别法句。}%

@{皆佛自釋三藏名也。今不具論。}@
%{皆佛自释三藏名也，今不具论。}%

@{然佛臨涅槃。阿難心沒憂海。}@
%{然佛临涅槃，阿难心没忧海。}%

@{阿㝹樓馱語云。汝持佛法人應問將來事云何啼哭。}@
%{阿㝹楼馱语云：汝持佛法人应问将来事云何啼哭？}%

@{阿難即醒悟乃問四事。佛皆具答。今出其二。}@
%{阿难即醒悟，乃问四事，佛皆具答，今出其二。}%

@{比丘當依四念處行道。}@
%{比丘当依四念处行道。}%

@{依波羅提木叉住木叉有二。一舊二客舊又二。一邪。二正。}@
%{依波罗提木叉住木叉有二，一旧二客，旧又二，一邪，二正。}%

@{正謂十善。邪謂鷄狗牛馬等(云云)。}@
%{正谓十善，邪谓鸡狗牛马等(云云) 。}%

@{定有二。一舊二客。舊又二。一正二邪。}@
%{定有二，一旧二客，旧又二，一正二邪。}%

@{正謂三四十二。邪謂邪禪鬼定等(云云)。}@
%{正谓三四十二，邪谓邪禅鬼定等(云云) 。}%

@{慧有二。一舊二客。舊又二。一正。二邪。}@
%{慧有二，一旧二客，旧又二，一正，二邪。}%

@{正謂識因識果。邪謂撥無因果。}@
%{正谓识因识果，邪谓拨无因果。}%

@{三種舊邪墮三惡道。佛棄而不用。}@
%{三种旧邪，堕三恶道，佛弃而不用。}%

@{三種舊正昇三善道。佛會而取之。}@
%{三种旧正，昇三善道，佛会而取之。}%

@{更說三種客法。客戒者。謂三歸五戒。二百五十等。}@
%{更说三种客法，客戒者，谓三归五戒，二百五十等。}%

@{客定者。謂九想八背等。}@
%{客定者，谓九想八背等。}%

@{客慧者。謂四諦智也。}@
%{客慧者，谓四谛智也。}%

@{佛之遺囑以戒為師。師訓七支。弟子奉行莫令污染。}@
%{佛之遗嘱，以戒为师，师训七支，弟子奉行莫令污染。}%

@{仁讓貞信和雅真正。戰戰兢兢動靜和諧。故言以戒為師也。}@
%{仁让贞信，和雅真正，战战兢兢，动静和谐，故言以戒为师也。}%

@{依木叉住者。木叉名保得解脫。若依木叉住者。保脫世間熱惱。}@
%{依木叉住者，木叉名保得解脱，若依木叉住者，保脱世间热恼。}%

@{所謂居家逼迫。牢獄熱惱妻子榔檔。繫縛熱惱。財業產貨怨賊熱惱。}@
%{所谓居家逼迫，牢狱热恼，妻子榔档，系缚热恼，财业产货，怨贼热恼。}%

@{王難逼迫水火熱惱。若依木叉保脫如此熱惱也。}@
%{王难逼迫，水火热恼，若依木叉，保脱如此热恼也。}%

@{又復住者。未來住安樂之處也。}@
%{又复住者，未来住安乐之处也。}%

@{若能住戒者不耐惡聲。惟欣善法。未來保住四天王住處也。}@
%{若能住戒者，不耐恶声，惟欣善法，未来保住四天王住处也。}%

@{若住戒者。則是供養三寶供養父母。未來保住三十三天也。}@
%{若住戒者，则是供养三宝，供养父母，未来保住三十三天也。}%

@{若住戒者。能發麁住保得住於炎摩天住處也。}@
%{若住戒者，能发麁住，保得住于炎摩天住处也。}%

@{若住戒者。能發細住保住兜率陀天。}@
%{若住戒者，能发细住，保住兜率陀天。}%

@{若持戒者能得欲界定。保住自在天。}@
%{若持戒者，能得欲界定，保住自在天。}%

@{若住戒者。能發未來禪。保住他化自在天。}@
%{若住戒者，能发未来禅，保住他化自在天。}%

@{若持戒者。能發四禪四空。保得住於色無色界諸天住處也。}@
%{若持戒者，能发四禅四空，保得住于色无色界诸天住处也。}%

@{是名依波羅提木叉住也。}@
%{是名依波罗提木叉住也。}%

@{依念處行道者。若無念處慧。一切行法皆非佛法非行道人。}@
%{依念处行道者，若无念处慧，一切行法皆非佛法，非行道人。}%

@{皆空剃頭如放牧者。空著染衣如木頭旛。}@
%{皆空剃头，如放牧者，空著染衣，如木头旛。}%

@{雖執鉢錫如病人乞具。雖讀誦經書如盲人誦賦。}@
%{虽执鉢锡，如病人乞具，虽读诵经书，如盲人诵赋。}%

@{雖復禮拜如碓上下。雖復坐禪如樹木葉。}@
%{虽复礼拜，如碓上下，虽复坐禅，如树木叶。}%

@{雖復興造。媒衒客作種樹貿易沈淪生死。蠶繭自縛無解脫期。}@
%{虽复兴造，媒衒客作种树贸易，沈沦生死，蚕茧自缚，无解脱期。}%

@{捨身命財但得名施非波羅蜜。}@
%{舍身命财，但得名施，非波罗蜜。}%

@{雖復持戒不免鷄狗。雖復精進精進無繡媚。}@
%{虽复持戒，不免鸡狗，虽复精进精进，无绣媚。}%

@{雖復坐禪如彼株杌。雖復知解狂顛智慧。常在此岸。不到彼岸。}@
%{虽复坐禅，如彼株杌，虽复知解狂颠智慧，常在此岸，不到彼岸。}%

@{不降愛見不破取相。不得入道品。非賢聖位不成四枯樹。非波羅蜜。}@
%{不降爱见，不破取相，不得入道品，非贤圣位，不成四枯树，非波罗蜜。}%

@{何以故。無念慧故。}@
%{何以故？无念慧故。}%

@{以念慧能破邪顯正。}@
%{以念慧能破邪显正。}%

@{大經云舊醫乳藥其實是毒。如蟲食木偶成字耳。}@
%{大经云：旧医乳药，其实是有毒，如虫食木偶成字耳。}%

@{是蟲不知是字非字。更有新醫從遠方來曉八種術。}@
%{是虫不知是字非字，更有新医从远方来晓八种术。}%

@{謂四枯四榮以新四枯破其舊乳。}@
%{谓四枯四荣，以新四枯破其旧乳。}%

@{法華亦云。大火從四面起。即斯例也。}@
%{法华亦云：大火从四面起，即斯例也。}%

@{執倒略有三種。一一切智。六師故六臣白王。我師是一切智。}@
%{执倒略有三種，一切智，六师故六臣白王，我师是一切智。}%

@{王若見者罪垢消滅此執世性也。}@
%{王若见者，罪垢消滅，此执世性也。}%

@{大論云。得宿命智。見八萬劫事。過是已不復能知。}@
%{大论云：得宿命智，见八万劫事，过是已不复能知。}%

@{但見初受胎身中陰之識。而自思惟此識。不應無因緣。}@
%{但见初受胎身中阴之识，而自思惟此识，不应无因缘。}%

@{憶想分別有法名。世性非五情所知。極微細故於世性冥初生覺。}@
%{忆想分别有法名，世性非五情所知，极微细故于世性冥初生觉。}%

@{覺即中陰識。從覺生我。從我生五塵。謂色聲香味觸。}@
%{觉即中阴识，从觉生我，从我生五尘，谓色声香味触。}%

@{從聲塵生空大。從聲觸生風大。從色觸生火大。}@
%{从声尘生空大，从声触生风大，从色触生火大。}%

@{從色聲觸味生水大。從色聲觸味香生地大。}@
%{从色声触味生水大，从色声触味香生地大。}%

@{從空生耳根。從風生身根。從火生眼根。從水生舌根。從地生鼻根。}@
%{从空生耳根，从风生身根，从火生眼根，从水生舌根，从地生鼻根。}%

@{如是漸漸從細至麁。還從麁至細。}@
%{如是渐渐从细至麁，还从麁至细。}%

@{譬如泥丸中具有瓶盆等性。瓶盆等破還為泥。都無所失世性。}@
%{譬如泥丸中具有瓶盆等性，瓶盆等破还为泥，都无所失世性。}%

@{是常無所從來。此僧佉所執也。}@
%{是常无所从来，此僧佉所执也。}%

@{復有執微塵。微塵常不可破。至微細故但待罪福因緣。}@
%{复有执微尘，微尘常不可破，至微细故但待罪福因缘。}%

@{因緣和合故有身。若天人地獄等以父母故罪福盡則散壞。}@
%{因缘和合故有身，若天人地狱等，以父母故罪福尽则散坏。}%

@{復有執自然為世界始。貧富貴賤非願行所得。}@
%{复有执自然为世界始，贫富贵贱非愿行所得。}%

@{復有人言。天主是世界主。始造吉凶滅時還天攝取。}@
%{复有人言，天主是世界主，始造吉凶灭时还天摄取。}%

@{復有人言。世世受苦樂盡自到邊。譬如山上投縷丸盡自止。}@
%{复有人言，世世受苦乐尽自到边，譬如山上投缕丸尽自止。}%

@{受罪福命歸於盡。是諸邪外皆於禪定知見。不從他聞。亦不從韋陀中聞。}@
%{受罪福命归于尽，是诸邪外，皆于禅定知见，不从他闻，亦不从韦陀中闻。}%

@{執此明利故云是事實餘妄語即此意也。}@
%{执此明利故，云是事实，余妄语即此意也。}%

@{復計自在天。復計父母等。皆是一切智之執也。}@
%{复计自在天，复计父母等，皆是一切智之执也。}%

@{二者神通。六師修得五通。停河在耳十二年。變釋為羊千根在體。}@
%{二者神通，六师修得五通，停河在耳十二年，变释为羊，千根在体。}%

@{毘富羅城變為鹵土等是也。}@
%{毘富罗城变为卤土等是也。}%

@{三者韋陀。六師解星文地理十八大經知吉凶等是也。}@
%{三者韦陀，六师解星文地理十八大经知吉凶等是也。}%

@{雖知世性無神通。是小知世性通是。次知三種備足。是大六師阿毘曇中明三種念處。}@
%{虽知世性无神通，是小知世性通是，次知三种备足，是大六师阿毘曇中明三种念处。}%

@{謂性共緣對破。}@
%{谓性共缘对破。}%

@{此三外道有人釋性念處。謂觀無生淺名為生深細觀無生見細法皆生死苦諦名性念處。}@
%{此三外道，有人释性念处，谓观无生浅，名为生深细观无生，见细法皆生死苦谛，名性念处。}%

@{有人專用慧。數緣無生空理。發真斷結得慧解脫羅漢。}@
%{有人专用慧，数缘无生空理，发真断结，得慧解脫罗汉。}%

@{對破邪因緣。無因緣顛倒執性。一切智外道也。}@
%{对破邪因缘，无因缘颠倒执性，一切智外道也。}%

@{共念處者。以禪定助道。正助合修亦名事理共觀。發得無漏三明。六通八解。成俱解脫羅漢。}@
%{共念处者，以禅定助道，正助合修，亦名事理共观，发得无漏三明，六通八解，成俱解脫罗汉。}%

@{對破根本愛慢。得五通外道也。}@
%{对破根本爱慢，得五通外道也。}%

@{緣念處者。緣佛三藏十二部文言。及一切世間名字。所緣處廣非如支佛出無佛世。}@
%{缘念处者，缘佛三藏十二部文言，及一切世间名字，所缘处广，非如支佛出无佛世。}%

@{不稟聲教。但作神通以悅眾生。不能說法。}@
%{不禀声教，但作神通以悦众生，不能说法。}%

@{緣念處人了達根性。善知四辯堪集法藏。成無疑大羅漢。}@
%{缘念处人了达根性，善知四辩，堪集法藏，成无疑大罗汉。}%

@{對破世間韋陀。星文地理文字鄙狹。當知邪正真偽。猶金比鐵螢日跡海。}@
%{对破世间韦陀，星文地理文字鄙狭，当知邪正真伪，犹金比铁，萤日迹海。}%

@{故大經云。於諸想中無常為最。於諸耕中秋耕為最。於諸跡中象跡為最。}@
%{故大经云：于诸想中无常为最，于诸耕中秋耕为最，于诸迹中象迹为最。}%

@{無常譬性耕。譬神通跡。譬文字也。}@
%{无常譬性，耕譬神通迹，譬文字也。}%

@{經云。諸優婆塞善解諸法對治之門。所謂常無常等。}@
%{经云：诸优婆塞善解诸法对治之门，所谓常无常等。}%

@{故知心行理外未入正真。破邪之義可解。}@
%{故知心行理外，未入正真，破邪之义可解。}%

@{二明五停心者。謂數息。不淨。慈心界方便因緣觀也。}@
%{二明五停心者，谓数息，不净，慈心界方便因缘观也。}%

@{停名停住。行人聞生滅四諦。發心觀苦諦欲出生死。}@
%{停名停住，行人闻生灭四谛，发心观苦谛，欲出生死。}%

@{為煩惱風搖動慧燈。照境不了為除此障。修五停心。}@
%{为烦恼风摇动慧灯，照境不了，为除此障，修五停心。}%

@{息門有數隨觀等。不淨門有九想八背等。慈心界方便十二因緣。}@
%{息门有数随观等，不净门有九想八背等，慈心界方便十二因缘。}%

@{皆如禪門中廣說。今不委論。}@
%{皆如禅门中广说，今不委论。}%

@{此五停門復有五意。謂對轉不轉。兼亦對亦轉亦不轉。}@
%{此五停门复有五意，谓对转不转，兼亦对亦转亦不转。}%

@{亦兼對者。數息對覺觀。不淨對貪欲。慈心對瞋恚。方便對我。因緣對癡。}@
%{亦兼对者，数息对觉观，不净对贪欲，慈心对瞋恚，方便对我，因缘对癡。}%

@{對治若成煩惱不起又發禪定。禪定持心安隱出入。故云五停心也。}@
%{对治若成，烦恼不起，又发禅定，禅定持心安隐出入，故云五停心也。}%

@{心既調定乃可習觀。若對治未益。更須用後四種治之。}@
%{心既调定，乃可习观，若对治未益，更须用后四种治之。}%

@{行者善用四隨巧修五治煩惱。不能障觀心停住。即入初賢位。}@
%{行者善用四随巧修五治烦恼，不能障观心停住，即入初贤位。}%

@{具如禪門廣說。}@
%{具如禅门广说。}%
