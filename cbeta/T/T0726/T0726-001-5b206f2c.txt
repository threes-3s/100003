@{馬鳴菩薩集}@
%{馬鳴菩薩集}%

@{歸命三世尊，正等覺所說，}@
%{向三世佛陀顶礼，这是正等觉者所说，}%
@{常行於利他，積集諸功德。}@
%{他们总是行利他之事，积聚各种功德。}%

@{若自身口意，所作善惡業，}@
%{如果自身言语意念所造的善恶业，}%
@{感果定非差，無別造作者。}@
%{感召的果报一定不会差，没有其他人替你造业。}%

@{最勝之導師，現證垂慈愍，}@
%{最殊胜的导师，现身证悟，慈悲怜悯，}%
@{普為諸有情，說隨業受報。}@
%{普度一切有情众生，讲述他们随业受报的道理。}%

@{此相應正理，聞已當領受，}@
%{这些道理与正法相符，听到之后应该要领受，}%
@{作業皆自心，為因馳六趣。}@
%{造业都是出自自身的心，这是原因，导致你流转六道。}%

@{由三毒怖畏，販賣諸物命，}@
%{因为贪嗔痴三毒的恐惧，贩卖各种生物的生命，}%
@{養已殺於他，當墮等活獄；}@
%{为了养活自己而杀害他人，就会堕入等活地狱；}%
@{彼壽百千歲，加刀杖捶打，}@
%{在那里寿命长达上千岁，却遭受刀杖捶打，}%
@{死已更復生，受如是苦報。}@
%{死后又会再生，承受这样的苦报。}%

@{於父母朋屬，而生於損害，}@
%{对父母朋友亲属，而生起伤害，}%
@{起妄語欺誑，當墮黑線獄；}@
%{说妄语欺骗，就会堕入黑绳地狱；}%
@{黑線絣其身，如世間解木，}@
%{黑色的绳索捆绑着身体，就像世间用来解剖木材的工具，}%
@{鋸發焰熾然，受如是苦報。}@
%{用锯子锯头发，火焰炽盛，承受这样的苦报。}%

@{以火焚山川，林木及原野，}@
%{用火焚烧山川，林木和原野，}%
@{燒害諸有情，當墮炎熱獄；}@
%{烧害众生，就会堕入炎热地狱；}%
@{火焰遍燒然，苦叫聲不絕，}@
%{火焰遍地燃烧，痛苦的叫声不断，}%
@{兩目無有明，受如是苦報。}@
%{双眼失明，承受这样的苦报。}%

@{謂法說非法，無根而誹謗，}@
%{说正法为非法，无缘无故地诽谤，}%
@{令他生熱惱，墮極炎熱獄；}@
%{让别人产生烦恼，就会堕入极炎热地狱；}%
@{是諸罪眾生，為大火逼切，}@
%{这些罪业众生，被大火逼迫，}%
@{燒然不暫停，受如是苦報。}@
%{燃烧不停，承受这样的苦报。}%

@{於猪羊狐兔，及餘生類等，}@
%{对猪羊狐兔，以及其他生灵，}%
@{殺害彼無邊，當墮眾合獄；}@
%{杀害他们无边无际，就会堕入众合地狱；}%
@{生彼地獄已，備受諸楚毒，}@
%{出生在那里之后，会承受各种酷刑毒害，}%
@{拷掠滅還生，受如是苦報。}@
%{拷打折磨，死了又再生，承受这样的苦报。}%

@{起惡身語意，讒構相離間，}@
%{起恶的言语意念，挑拨离间，}%
@{如是罪眾生，當墮號叫獄；}@
%{这样的罪业众生，就会堕入号叫地狱；}%
@{彼熾猛火焰，燒然無休息，}@
%{那里火焰猛烈，燃烧不停息，}%
@{常發大惡聲，受如是苦報。}@
%{总是发出巨大的恶声，承受这样的苦报。}%

@{若於佛法僧，及諸貧乏者，}@
%{如果对佛法僧，以及贫穷的人，}%
@{剽竊彼財物，墮大號叫獄；}@
%{偷窃他们的财物，就会堕入大号叫地狱；}%
@{為火所燒炙，受最極熱惱，}@
%{被火烧烤，承受最极端的热恼，}%
@{出大猛惡聲，受如是苦報。}@
%{发出巨大的恶声，承受这样的苦报。}%

@{於父母師長，及有德賢者，}@
%{对父母师长，以及有德行的人，}%
@{起增上殺害，定墮無間獄；}@
%{起增上杀害，一定会堕入无间地狱；}%
@{炎熾大火聚，洞然徹骨髓，}@
%{那里火焰炽盛，直透骨髓，}%
@{長時受極苦，決定無暫樂。}@
%{长时间承受极度痛苦，绝对没有片刻安乐。}%

@{起冤敵鬪諍，戈戟相殘害，}@
%{起冤仇敌对，刀枪互相残害，}%
@{死墮銅爪獄，其狀極鋒利，}@
%{死后就会堕入铜爪地狱，那里铜爪极其锋利，}%
@{互相而擘裂，或變為刀杖，}@
%{互相撕裂，或者变成刀杖，}%
@{競斫刺其身，受如是苦報。}@
%{互相砍刺身体，承受这样的苦报。}%

@{強暴侵他色，後墮鐵刺獄，}@
%{强暴侵犯他人色身，之后就会堕入铁刺地狱，}%
@{刺長十六指，削肉而穿骨；}@
%{那里铁刺长达十六指，削肉穿骨；}%
@{復有大鐵女，焰牙甚可怖，}@
%{还有巨大的铁女，火焰般的牙齿非常可怕，}%
@{噉頂至其踵，受如是苦報。}@
%{从头顶吃到脚跟，承受这样的苦报。}%

@{若陰謀害他，墮彼劍葉林，}@
%{如果阴谋害人，就会堕入剑叶林，}%
@{猘狗及鵰鷲，奔逐競分食。}@
%{那里有恶狗和秃鹫，追逐争抢着分食。}%

@{若劫奪他財，當受其極苦，}@
%{如果抢劫他人财物，就会承受极度痛苦，}%
@{常吞熱鐵丸，復灌以銅汁。}@
%{总是吞食滚烫的铁丸，再灌入铜汁。}%

@{無辜害眾生，當墮鐵鷹獄，}@
%{无辜杀害众生，就会堕入铁鹰地狱，}%
@{利爪而搏擊，長時受苦惱。}@
%{那里铁鹰用锋利的爪子搏击，长时间承受痛苦。}%

@{若人好傷殺，水族諸物命，}@
%{如果喜欢杀害水族生物，}%
@{後墮鎔銅河，為業火燒煮。}@
%{之后就会堕入熔铜河，被业火烧煮。}%

@{若自貪盛事，而掩斥他善，}@
%{如果自己贪图盛事，而排斥别人的善行，}%
@{墮彼鐵磨獄，悲號無救度。}@
%{就会堕入铁磨地狱，悲号无救度。}%

@{若人不饒益，當受多種形，}@
%{如果人不饶益他人，就会承受多种形态，}%
@{兩山夾其身，逼切復加㨝。}@
%{两座山夹着身体，逼迫又加压。}%

@{若人說非法，破壞於橋梁，}@
%{如果说非法，破坏桥梁，}%
@{鋒刃以為路，驅彼而來往。}@
%{用锋利的刀刃作为道路，驱赶他们来回行走。}%

@{以甲傷蚤虱，墮二羺頭山，}@
%{用盔甲伤害跳蚤虱子，就会堕入两羺头山，}%
@{相擊死還生，展轉而受苦。}@
%{互相撞击，死了又再生，辗转承受痛苦。}%

@{依止出離道，而不護禁戒，}@
%{依止出离之道，却不好好守护戒律，}%
@{墮煻煨獄中，肢體皆消爛。}@
%{就会堕入煻煨地狱，肢体都腐烂消散。}%

@{矯現諸威儀，苟求邪活命，}@
%{假装各种威仪，苟且求活，}%
@{墮屍糞獄中，為蛆虫𠯗食。}@
%{就会堕入尸粪地狱，被蛆虫啃食。}%

@{見五穀中虫，不擇而礱伐，}@
%{看到五谷中的虫子，不加选择地碾磨，}%
@{墮鐵碓獄中，常為彼舂擣。}@
%{就会堕入铁碓地狱，经常被舂捣。}%

@{見他苦生喜，諂曲多疑惑，}@
%{看到别人痛苦就感到高兴，谄媚曲意，多疑，}%
@{常懷忿惡心，作焰摩羅卒。}@
%{总是怀着怨恨恶毒的心，成为焰摩罗的卒兵。}%

@{諸苦果種子，少略而分別，}@
%{这些苦果的种子，简单地做一下分别，}%
@{清淨身語心，畢竟常遠離。}@
%{清净身语意，最终要永远远离。}%

@{地獄趣竟。}@
%{地狱趣说完。}%

@{若人不樂施，復盜眾飲食，}@
%{如果人不乐意布施，又偷盗众人的饮食，}%
@{墮大癭鬼中，常噉諸糞穢。}@
%{就会堕入大瘿鬼中，总是吃各种粪便。}%

@{若障人布施，於己物生悋，}@
%{如果阻碍别人布施，对自己的东西生起吝啬，}%
@{墮針口鬼中，腹大常飢渴。}@
%{就会堕入针口鬼中，肚子很大总是饥渴。}%

@{無嗣慳財物，不捨不受用，}@
%{没有子嗣，贪吝财物，不舍不使用，}%
@{墮匱乏鬼中，得他享殘棄。}@
%{就会堕入匮乏鬼中，只能得到别人享用剩下的东西。}%

@{希望他施惠，自少施生悔，}@
%{希望别人施舍，自己却很少布施，还生起悔恨，}%
@{墮下劣鬼中，常食於涎吐。}@
%{就会堕入下劣鬼中，总是吃别人的口水。}%

@{樂聞他過失，加惡語宣傳，}@
%{喜欢听别人的过失，还用恶语宣扬，}%
@{墮焰口鬼中，長受諸苦惱。}@
%{就会堕入焰口鬼中，长时间承受各种痛苦。}%

@{好起諸鬪諍，無少慈愍心，}@
%{喜欢争斗，没有一点慈悲心，}%
@{墮疲極鬼中，食蜎蠕虫類。}@
%{就会堕入疲极鬼中，吃蠕虫之类的东西。}%

@{恐愶取他財，得已或少施，}@
%{害怕别人抢夺自己的财物，得到之后又很少布施，}%
@{墮極醜鬼中，他怖獲微祀。}@
%{就会堕入极丑鬼中，让人害怕，只能得到微薄的供养。}%

@{若人多讌樂，廣殺諸物命，}@
%{如果人喜欢宴乐，大量杀害生物，}%
@{自食復與他，後墮羅剎鬼。}@
%{自己吃，也给别人吃，之后就会堕入罗刹鬼道。}%

@{於供養香花，或嗅或私取，}@
%{对供养的香花，要么闻，要么私自取走，}%
@{由起少貪心，後作尋香鬼。}@
%{因为起了少许贪心，之后就会成为寻香鬼。}%

@{若人相崇奉，於己有所求，}@
%{如果别人尊敬你，向你有所求，}%
@{怒色希彼財，後作猛惡鬼。}@
%{你却怒气冲冲，希望得到他们的财物，之后就会成为凶猛的恶鬼。}%

@{若人於他娶，常樂作媒伐，}@
%{如果人喜欢给别人做媒，总是乐于撮合婚事，}%
@{後懷惡相離，死作步多鬼。}@
%{之后怀着恶念，互相离间，死后就会成为步多鬼。}%

@{若人樂飲酒，過量復顛酗，}@
%{如果人喜欢喝酒，过量又醉酒，}%
@{或持勸他人，後作藥叉鬼。}@
%{或者劝别人喝酒，之后就会成为药叉鬼。}%

@{於父母師長，所欲相違背，}@
%{对父母师长，想要违背他们的意愿，}%
@{後生藥叉宮，勇健多卒暴。}@
%{之后就会出生在药叉宫，勇猛健壮，经常暴虐。}%

@{彼慳貪過失，常生餓鬼中，}@
%{这些贪婪吝啬的过失，经常导致人出生在饿鬼道，}%
@{苦樂隨自因，是故勿復造。}@
%{苦乐随自己的因缘，所以不要再造这些恶业。}%
@{餓鬼趣竟。}@
%{饿鬼道结束了。}%

@{人天三惡趣，唯自能救拔，}@
%{人、天、三恶道，只有自己才能救拔自己，}%

@{奔馳六趣中，如夢境和合。}@
%{在六道中奔波，就像梦境一样，虚幻无常。}%

@{覩自他眷屬，傷嘆豈長久，}@
%{看到自己和亲人，悲伤叹息，又能持续多久呢？}%

@{如彼俳優人，數數易形色。}@
%{就像那些演员，不断地变换着角色和装扮。}%

@{受地獄苦畢，或生於天中，}@
%{受完地狱的苦难，也许会生到天界，}%

@{福盡復沈淪，墮彼畜生趣。}@
%{福报尽了又会沉沦，堕落到畜生道。}%

@{受種種形色，後得生人間，}@
%{受着各种各样的身形，后来才能投生到人间，}%

@{貧苦極艱辛，譬車輪旋轉。}@
%{贫苦困顿，艰辛无比，就像车轮一样不停地转动。}%

@{畜生趣竟。}@
%{畜生道结束了。}%

@{彼人趣壽命，分量本長遠，}@
%{人道寿命，本来就比较长，}%

@{多造殺生因，由此而減少。}@
%{但如果造了太多杀生的业因，寿命就会因此而减少。}%

@{縈纏諸病苦，癩瘦時疫等，}@
%{会遭受各种疾病的折磨，比如麻风病、瘦弱、瘟疫等等，}%

@{為鬼魅所著，及王法捶打。}@
%{还会被鬼魅附身，或者被法律惩罚。}%

@{若人於財利，勞役廣希求，}@
%{如果一个人为了钱财，拼命地劳碌奔波，}%

@{無少惠施心，後作守財鬼。}@
%{一点也不愿意布施，死后就会变成守财鬼。}%

@{若人盜他財，用已或能施，}@
%{如果一个人偷了别人的财物，自己用过之后，也许会布施出去，}%

@{後墮鬼趣中，隨得隨散失。}@
%{但死后堕落到饿鬼道，得到的财富就会立刻消失。}%

@{若人於己財，隨分而行施，}@
%{如果一个人用自己的财物，适当地进行布施，}%

@{當獲於富饒，不為他侵損。}@
%{就会获得富裕，不会被别人侵犯。}%

@{若人以淨財，離慳廣行施，}@
%{如果一个人用干净的钱财，远离吝啬，广泛地布施，}%

@{得上妙飲食，所欲皆如意。}@
%{就会得到美味的食物，想要什么就能得到什么。}%

@{若人以珍饌，淨心而奉施，}@
%{如果一个人用珍贵的食物，清净心地布施，}%

@{其所得福報，色力命安隱。}@
%{就会得到色力寿命安稳的福报。}%

@{若人以衣施，彼得生愛樂，}@
%{如果一个人布施衣服，可以让接受的人感到快乐，}%

@{獲色相端嚴，具慚愧衣服。}@
%{自己也会得到端庄的相貌，并且拥有惭愧的衣服。}%

@{若人造僧坊，歡喜而用施，}@
%{如果一个人建造寺院，欢喜地布施，}%

@{感宮殿莊嚴，五欲皆具足。}@
%{就会感得宫殿庄严，五欲都能够满足。}%

@{若人建橋梁，及車乘等施，}@
%{如果一个人建造桥梁，以及车乘等布施，}%

@{得最上安隱，珍寶之輦輿。}@
%{就会得到最安稳的宝车。}%

@{若人於曠野，施池井泉流，}@
%{如果一个人在荒野中，布施水井泉水，}%

@{於在所生處，無渴乏熱惱。}@
%{投生的地方就不会有干渴和炎热之苦。}%

@{若人以嬖妾，嚴飾而為施，}@
%{如果一个人用自己的爱妾，装饰后布施，}%

@{彼所得果報，具欲樂富貴。}@
%{就会得到拥有欲望、快乐和富贵的果报。}%

@{若人以經教，及世俗文典，}@
%{如果一个人用经书和世俗典籍，}%

@{能持施於他，感博學大智。}@
%{能够布施给别人，就会感得博学多才。}%

@{若人以醫藥，及以無畏施，}@
%{如果一个人用医药，以及无畏布施，}%

@{彼所得果報，安樂離恐怖。}@
%{就会得到安乐无忧的果报。}%

@{若施以燈明，其眼常清淨；}@
%{如果布施灯明，眼睛就会永远清净；}%

@{若施以音樂，得其聲美妙；}@
%{如果布施音乐，就会得到美妙的声音；}%

@{若施以臥具，獲安隱快樂；}@
%{如果布施卧具，就会得到安稳快乐；}%

@{若施以僮僕，營從常圍繞；}@
%{如果布施僮仆，就会有侍从永远围绕；}%

@{施乳牛等物，得色力壽命；}@
%{如果布施乳牛等物，就会得到色力寿命；}%

@{若施以良田，得倉庫盈溢。}@
%{如果布施良田，就会得到仓库盈满。}%

@{如彼彼求者，各各隨所欲，}@
%{就像那些乞求者，每个人都根据自己的愿望，}%

@{花果及清泉，愛語善安慰。}@
%{得到花果、清泉、爱语和安慰。}%

@{復有雖行施，役他而少與，}@
