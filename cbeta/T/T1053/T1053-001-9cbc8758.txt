@{聖觀自在菩薩功德讚}@
%{聖觀自在菩薩功德讚}%

@{西方賢聖集}@
%{西方賢聖集}%

@{歸命最聖觀自在，滿月妙相蓮華生，}@
%{我向最尊贵的观世音菩萨顶礼膜拜，他就像一轮圆满的月亮，从莲花中诞生。}%
@{能以無畏施有情，我今稱讚彼功德：}@
%{他以无畏的精神，救度众生，我今天要赞叹他的功德。}%

@{一切善法悉具足，頂戴諸佛大智冠，}@
%{他拥有所有善法，头戴着佛陀的大智慧之冠。}%
@{福慧莊嚴最上尊，是故歸命蓮華手。}@
%{他是福德智慧圆满的最高尊者，所以我们向他顶礼膜拜。}%

@{兩頰圓滿紅白色，鼻相脩直妙端嚴，}@
%{他的两颊圆润，红白相间，鼻子端正秀丽，非常庄严。}%
@{齒如珂雪密復齊，四牙平正而具足；}@
%{牙齿洁白如珂雪，排列整齐，四颗大牙平整而完整。}%

@{梵音清響復深妙，一切聞者生愛心，}@
%{他的声音清脆悦耳，深奥微妙，所有听到的人都生起爱慕之心。}%
@{凡所宣說眾語言，柔和善順而甘美；}@
%{他宣说的各种语言，柔和顺畅，甜美动听。}%

@{常以大悲方便力，救度一切苦眾生，}@
%{他始终以大悲心和巧妙的方法，救度所有受苦的众生。}%
@{自利利他眾導師，是故歸命蓮華手。}@
%{他是利益自己和他人，引导众生的导师，所以我们向他顶礼膜拜。}%

@{雙眉同彼初生月，身相猶如真紫金，}@
%{他的双眉如同初生的月亮，身体如同真正的紫金。}%
@{眾寶為鬘以莊嚴，一切見者無厭足；}@
%{他用各种珍宝装饰身体，所有看到他的人，都看不够。}%

@{敷鹿皮衣而為座，最上自在眾中尊，}@
%{他坐在鹿皮做的垫子上，是所有自在者中最尊贵的。}%
@{身諸相分悉周圓，是故歸命蓮華手。}@
%{他的身体相貌，完美无缺，所以我们向他顶礼膜拜。}%

@{上妙細㲲為絡腋，摩尼珠寶作耳環，}@
%{他用最精美的丝绸装饰腋下，用摩尼宝珠做耳环。}%
@{世出世寶所莊嚴，最上自在最尊勝；}@
%{他用世间和出世间所有的珍宝来装饰，是最自在、最尊贵的。}%

@{諸善功德皆滿足，一切世間大法王，}@
%{他拥有所有善法功德，是世间一切大法之王。}%
@{能開種種方便門，是故歸命蓮華手。}@
%{他能开启各种方便之门，所以我们向他顶礼膜拜。}%

@{罪業眾生微塵數，無量劫來處輪迴，}@
%{罪业深重的众生，如同微尘般多，无量劫以来，在轮回中漂泊。}%
@{菩薩常生方便心，化度悉令歸智聚；}@
%{菩萨始终以巧妙的方法，度化他们，让他们回归智慧的集合。}%

@{放慈悲光照一切，以最上法施眾生，}@
%{他放射慈悲的光芒，照耀一切，用最高的法度化众生。}%
@{普令有情證菩提，是故歸命蓮華手。}@
%{让所有有情众生都证得菩提，所以我们向他顶礼膜拜。}%

@{所有阿鼻大地獄，菩薩慈悲往現身，}@
%{在所有无间地狱中，菩萨以慈悲心，前往现身。}%
@{破壞彼趣悉無餘，閻摩獄卒生驚怖；}@
%{他彻底摧毁地狱，阎罗的狱卒都感到恐惧。}%

@{彼諸業報眾生類，咸皆離苦罪銷除，}@
%{那些受业报的众生，都摆脱了痛苦，罪业消除了。}%
@{惡趣能以智火焚，是故歸命蓮華手。}@
%{恶趣被智慧之火焚烧，所以我们向他顶礼膜拜。}%

@{具足悲智行願力，欲色界中皆現身，}@
%{他具足悲心、智慧、行持和愿力，在欲界和色界中，都显现身。}%
@{於法自在利群生，閻摩眷屬悉歡喜；}@
%{他在法中自在，利益众生，阎罗的眷属都感到欢喜。}%

@{破壞一切罪業境，令諸苦惱不復生，}@
%{他摧毁一切罪业的境界，让众生的痛苦不再产生。}%
@{有情離苦得清涼，是故歸命蓮華手。}@
%{有情众生摆脱了痛苦，获得清凉，所以我们向他顶礼膜拜。}%

@{我今稱讚大聖者，最上色相無等倫，}@
%{我今天赞叹这位伟大的圣者，他的相貌是世间最美好的，无与伦比。}%
@{無礙自在三界尊，普施眾生無所畏；}@
%{他在三界中自在无碍，无畏地救度所有众生。}%

@{度脫有情出苦海，斷彼三毒煩惱根，}@
%{他度化众生，让他们脱离苦海，斩断贪、嗔、痴三毒的根源。}%
@{咸令得住安樂中，是故歸命蓮華手。}@
%{让他们都安住在快乐之中，所以我们向他顶礼膜拜。}%

@{所有一切餓鬼眾，針咽大腹及臭毛，}@
%{所有饿鬼，他们咽喉如同针眼，肚子很大，身上长着臭毛。}%
@{如是等類數甚多，晝夜受彼飢渴苦，}@
%{这类饿鬼数量很多，他们昼夜承受着饥渴的痛苦。}%

@{由彼飢渴火所逼，互相吞噉苦惱生，}@
%{被饥渴的火焰逼迫，他们互相吞食，痛苦丛生。}%
@{菩薩悲願力所持，往彼現身而救濟，}@
%{菩萨以悲心和愿力，前往那里现身，救济他们。}%

@{先化種種美飲食，各各令飽益身胑，}@
%{他先变化出各种美味的食物，让每个人都吃饱，身体得到滋养。}%
@{復為宣說正法門，一切得離諸苦惱；}@
%{然后，他宣说正法，让他们都摆脱痛苦。}%

@{菩薩悲智方便力，能入一切趣類中，}@
%{菩萨以悲心、智慧和巧妙的方法，能进入所有众生的类别中。}%
@{利行同事攝有情，是故歸命蓮華手。}@
%{他利益众生，与他们同行，摄受他们，所以我们向他顶礼膜拜。}%

@{身有大摩尼珠寶，常出淨妙眾光明，}@
%{他的身体如同巨大的摩尼宝珠，不断放射出清净的光芒。}%
@{普照世間諸暗暝，映蔽日月而不現，}@
%{这光芒照耀世间所有黑暗，使日月都黯然失色。}%

@{所有夜叉羅剎等，蒙光照者悉歸依，}@
%{所有夜叉、罗刹等，被这光芒照耀，都皈依他。}%
@{觀眾生心令斷疑，是故歸命蓮華手。}@
%{他观照众生的心，让他们断除疑虑，所以我们向他顶礼膜拜。}%

@{大力阿修羅王眾，鬪諍勇猛而難調，}@
%{强大的阿修罗王众，好斗勇猛，难以调伏。}%
@{菩薩亦現彼趣中，方便說法而化度，}@
%{菩萨也出现在他们那里，巧妙地宣说法，度化他们。}%

@{彼聞法已斷疑惑，各各生於慈善心，}@
%{他们听到佛法，断除了疑惑，都生起慈悲心。}%
@{能施最上妙法門，是故歸命蓮華手。}@
%{他施予他们最高妙的法门，所以我们向他顶礼膜拜。}%

@{菩薩曾遊諸天界，入一妙環天子宮，}@
%{菩萨曾经游历诸天界，进入妙环天子的宫殿。}%
@{隨順方便開化門，彼天悉無諸所施，}@
%{他顺应方便，开启度化之门，但那些天人却一无所有，无法施舍。}%

@{是時天子生苦惱，菩薩即為現珍財，}@
%{这时，天子感到痛苦，菩萨就显现珍宝给他。}%
@{令彼天子滿施心，是故歸命蓮華手。}@
%{让天子充满施舍之心，所以我们向他顶礼膜拜。}%

@{所有一切貪欲者，常為煩惱火所燒，}@
%{所有贪欲的人，总是被烦恼之火燃烧。}%
@{得見菩薩善威容，即能迴心思正法，}@
%{他们看到菩萨的善相威仪，就能回心转意，思慕正法。}%

@{世間貪瞋癡盛者，恭敬頂禮大聖尊，}@
%{世间贪、嗔、痴盛行的人，恭敬地顶礼大圣尊。}%
@{三毒煩惱獲銷除，是故歸命蓮華手。}@
%{他们三毒烦恼消除了，所以我们向他顶礼膜拜。}%

@{眾生界分無邊際，蜫蟲螻蟻數無窮，}@
%{众生界无边无际，蜎虫、蝼蚁数量无穷。}%
@{菩薩亦以慈悲心，處處方便而化度，}@
%{菩萨也以慈悲心，处处以巧妙的方法，度化他们。}%

@{世間或時值飢饉，即變所食濟群生，}@
%{世间有时发生饥荒，他就会变化出食物，救济众生。}%
@{咸令息彼飢渴心，是故歸命蓮華手。}@
%{让所有众生都平息了饥渴之心，所以我们向他顶礼膜拜。}%

@{若見眾生處大海，舡筏破壞欲漂沈，}@
%{如果看到众生在海中，船只破损，快要沉没。}%
@{菩薩慈悲現馬身，令其攀附而得渡，}@
%{菩萨以慈悲心，显现马身，让他们攀附上来，渡过难关。}%

@{眾生乘舡在大海，菩薩隨意使其風，}@
%{众生乘船在大海中，菩萨随心所欲地控制着风。}%
@{若順若逆濟眾生，咸令遠離諸怖畏。}@
%{无论是顺风还是逆风，他都救度众生，让他们远离恐惧。}%

@{菩薩已離三界苦，圓滿三摩地法門，}@
%{菩萨已经脱离了三界痛苦，圆满成就了三摩地法门。}%
@{復能攝化諸有情，數等虛空無邊際；}@
%{他还能摄受度化所有有情众生，数量如同虚空般无边无际。}%

@{菩薩已離種種怖，所謂羅剎諸部多，}@
%{菩萨已经远离各种恐惧，比如罗刹等各种鬼怪。}%
@{必舍左及拏吉泥，邪迷一切惡鬼眾，}@
%{他远离了左道和邪恶的鬼怪，也远离了所有迷惑众生的恶鬼。}%

@{大水大火并盜賊，虎狼蟲獸及刀兵，}@
%{他远离了洪水、火灾、盗贼、虎狼、虫兽、刀兵。}%
@{疾病毒藥與邪明，乃至王刑禁縛等；}@
%{他远离了疾病、毒药、邪术，甚至远离了王法刑罚和囚禁。}%

@{菩薩自得遠離已，而復為彼諸眾生，}@
%{菩萨自己已经远离了这些痛苦，却仍然为了这些众生。}%
@{自利利他大導師，最上大悲無畏者；}@
%{他是利益自己和他人，引导众生的伟大导师，是最高的大悲无畏者。}%

@{菩薩已具諸梵行，如日月分照世間，}@
%{菩萨已经具备了所有梵行，如同日月照耀世间。}%
@{人天稱讚大聖尊，是為最上真梵行；}@
%{人天都赞叹这位伟大的圣者，这是最真实的梵行。}%

@{菩薩常住三摩地，無邊功德為所依，}@
%{菩萨常住三摩地，无边的功德作为他的依托。}%
@{隨順方便現所居，補陀落山為住止，}@
%{他随顺方便，显现住处，在补陀落山居住。}%

@{其山高廣復殊妙，種種珍寶以莊嚴，}@
%{这座山高大广阔，非常奇妙，用各种珍宝装饰。}%
@{彼有寶樹數甚多，低羅迦及瞻波等，}@
%{那里有很多宝树，比如低罗迦树、瞻波树等。}%

@{有諸異鳥止其上，常出清淨妙好音，}@
%{各种奇异的鸟儿栖息在树上，不断发出清净美妙的声音。}%
@{如是莊嚴聖所居，禮敬瞻仰而獲福，}@
%{如此庄严的圣者住处，礼敬瞻仰，就能获得福报。}%

@{俱尾囉并嚩嚕乃，及彼閻摩等三天，}@
%{俱尾囉天、嚩嚕乃天，以及阎罗等三天，}%
@{乾闥婆王修羅王，夜叉王及諸龍主，}@
%{乾闥婆王、阿修罗王、夜叉王，以及所有龙王，}%

@{乃至部多等諸類，若天若神一切王，}@
%{还有部多等各种类别，无论是天人、神灵，还是一切国王，}%
@{晝夜各生恭敬心，歸依稱讚真聖者，}@
%{他们昼夜都生起恭敬心，皈依赞叹这位真实的圣者。}%

@{彼等如是稱讚已，一切所欲皆隨心，}@
%{他们如此赞叹之后，所有愿望都能心想事成。}%
@{大富快樂壽命長，勇猛精進具威德，}@
%{他们富贵快乐，寿命长久，勇猛精进，具有威德。}%

@{息除無量諸苦惱，夢中怖畏亦不生，}@
%{他们消除了无量的痛苦，梦中恐惧也不再产生。}%
@{乃至壽命欲終時，菩薩現身而安慰。}@
%{即使寿命将尽，菩萨也会现身安慰他们。}%

@{聖觀自在功德海，無量無邊無有窮，}@
%{观世音菩萨的功德如同大海，无量无边，无穷无尽。}%
@{假以百舌千劫中，稱揚讚歎不能盡。}@
%{即使用一百条舌头，千劫的时间，也无法完全赞叹他的功德。}%

@{聖觀自在菩薩功德讚}@
%{观世音菩萨功德赞}%
