@{佛說護國尊者所問經卷第二}@
%{佛说护国尊者所问经卷第二}%

@{西天譯經三藏朝散大夫試鴻臚卿傳法大師臣施護奉詔譯}@
%{西天译经三藏朝散大夫试鸿胪卿传法大师臣施护奉诏译}%

@{爾時，世尊告尊者護國言：有四種法，於諸菩薩而為縛法。}@
%{当时，世尊对护国尊者说：有四种法，对所有菩萨来说都是束缚之法。}%

@{何等四法？一者、輕慢他人，二者、於世間事方便趣求，三者、散亂用心如行嶮難，四者、於其眷屬一心貪著。如是四法為菩薩縛。}@
%{这四种法是什么呢？第一，轻慢他人；第二，在世间事上方便地追求；第三，心散乱，像走在险峻的山路上一样；第四，一心贪着于自己的眷属。这四种法就是菩萨的束缚。}%

@{復說頌曰：}@
%{再说一首颂歌：}%

@{若行輕慢於他人、方便唯求世間事、}@
%{如果轻慢他人，只方便地追求世间事，}%

@{散亂如行嶮道中，如象陷身深泥裏；}@
%{心散乱像走在险峻的山道中，像大象陷入泥潭里；}%

@{於自眷屬生愛著，常懷貪戀如迷醉，}@
%{对自己的眷属生出爱恋，常常怀着贪恋像醉酒一样，}%

@{如是種種被纏縛，增長愚癡覆大智。}@
%{这样种种被束缚，增长愚痴蒙蔽大智慧。}%

@{若人怖苦厭生死，求出沈淪趣解脫，}@
%{如果有人害怕痛苦厌恶生死，想要出离轮回获得解脱，}%

@{捨於輕慢世間等，是名菩薩所行道。}@
%{舍弃轻慢世间等等，这才是菩萨的修行之道。}%

@{滅盡無邊諸苦已、及彼煩惱諸眷屬，}@
%{灭尽无边的苦，以及那些烦恼的眷属，}%

@{究竟安樂無所求，圓滿菩提寂靜道。}@
%{最终安乐无所求，圆满菩提寂静道。}%

@{所行六種波羅蜜、三身、五智、十力等，}@
%{所修行的六种波罗蜜、三身、五智、十力等等，}%

@{一切功德悉具足，如是永離無邊苦。}@
%{一切功德都具备，这样就能永远远离无边的苦。}%

@{過去修行無量劫，為眾生故求菩提，}@
%{过去修行无量劫，为了众生而求菩提，}%

@{一切眾善悉皆修，遠離諸惡眷屬等。}@
%{一切善法都修习，远离一切恶的眷属等等。}%

@{恒樂深山寂靜處，遠離聲色想真空，}@
%{恒乐深山寂静之处，远离声色想，进入真空，}%

@{如是精進不間修，獲大丈夫圓滿慧。}@
%{这样精进不断地修行，就能获得大丈夫圆满的智慧。}%

@{見彼世間眾生行，五趣輪迴無有窮，}@
%{看见世间众生，在五趣轮回中无穷无尽，}%

@{我於過去發慈心，捨自身命及妻子、}@
%{我在过去发慈悲心，舍弃自己的生命，以及妻子、}%

@{國城、大地及珍寶，如是求佛無數劫。}@
%{国城、大地以及珍宝，这样求佛无量劫。}%

@{我昔居山行忍辱，花菓池沼悉清淨，}@
%{我曾经住山修行忍辱，花果池沼都清净，}%

@{歌利王來截手足，心生慈忍無瞋恚。}@
%{歌利王来砍断我的手足，我心中生起慈忍，没有瞋恚。}%

@{昔住深山名闍摩，我為仙人婆囉多，}@
%{曾经住在深山名叫闍摩，我是仙人婆囉多，}%

@{時有天子射我身，亦無瞋恨生其惡，}@
%{当时有天子射我身体，我依然没有瞋恨，没有生起恶念，}%

@{不惜身命如頑石，志求菩提心不退。}@
%{不惜生命像顽石一样，志求菩提心不退转。}%

@{我昔曾為薩埵時，見彼餓虎欲食子，}@
%{我曾经是萨埵的时候，看见那只饿虎想要吃孩子，}%

@{投崖捨命濟彼飢，天人稱讚大精進，}@
%{就投崖舍命救济它，天人称赞我精进，}%

@{常樂布施救眾生，不悋身命及財寶。}@
%{常常乐于布施救济众生，不吝惜生命和财宝。}%

@{我昔曾為摩曩縛，廣行布施盡寶海，}@
%{我曾经是摩曩縛，广行布施，耗尽宝海，}%

@{捨大摩尼令富他，如是求證菩提果。}@
%{舍弃大摩尼让别人富裕，这样求证菩提果。}%

@{往昔作大蘇摩王，名稱普聞我修行，}@
%{往昔做大苏摩王，名声远播，我修行，}%

@{爾時入縛為他人，於彼百王得解脫。}@
%{当时入狱为了他人，让那百王获得解脱。}%

@{我昔曾為能捨王，一切所求皆充足，}@
%{我曾经是能舍王，一切所求都充足，}%

@{乃至身命及珍財，令他大富離貧苦。}@
%{乃至生命和珍宝，都让别人富裕，离开贫苦。}%

@{昔有飛鴿來投我，即割身肉濟彼命，}@
%{曾经有一只飞鸽投靠我，我就割下自己的肉救它性命，}%

@{如是持刀割肉時，無驚無怖心安隱。}@
%{这样拿着刀割肉的时候，没有惊恐，没有害怕，心安稳。}%

@{亦於過去捨王位，盡世行彼波羅蜜，}@
%{也曾经在过去舍弃王位，终生修行波罗蜜，}%

@{復自化身為妙藥，捨己身命濟群品。}@
%{又把自己化身成妙药，舍弃自己的生命救济众生。}%

@{又昔曾為師子王，常為世間行利樂，}@
%{又曾经是狮子王，常常为世间行利益，}%

@{棄捨王位及眷屬，一心志求無上道。}@
%{舍弃王位和眷属，一心志求无上道。}%

@{又昔曾為妙牙王，當時獲壽於千歲，}@
%{又曾经是妙牙王，当时获得千岁寿命，}%

@{八十四年修苦行，發大精進施珍財，}@
%{八十四年修苦行，发大精进，施舍珍宝，}%

@{於佛塔前燃己身，志心恭敬作供養。}@
%{在佛塔前燃烧自己的身体，志心恭敬地作供养。}%

@{又昔曾為無垢王，時有惡眼婆羅門，}@
%{又曾经是无垢王，当时有个恶眼婆罗门，}%

@{來詣深宮乞我頭，即便捨頭而施與。}@
%{来到深宫向我乞求头颅，我就立刻舍弃头颅施舍给他。}%

@{又昔曾為月光王，普救眾生作善利，}@
%{又曾经是月光王，普救众生，行善利，}%

@{一切城隍聚落中，四衢道路施良藥。}@
%{在所有城隍聚落中，在所有四通八达的道路上，施舍良药。}%

@{千女端嚴妙色相，金寶、真珠廣莊嚴，}@
%{千个女子端庄美丽，金宝、珍珠广为装饰，}%

@{捨彼千女自修行，如是所作福無等。}@
%{舍弃那千个女子自己修行，这样所做的福德无与伦比。}%

@{又昔曾為輸婆王，所戴寶冠世希有，}@
%{又曾经是输婆王，所戴的宝冠世间罕见，}%

@{香花、眾寶共莊嚴，捨施他人無所悋。}@
%{香花、珠宝共同装饰，舍施给别人，毫不吝惜。}%

@{又昔曾為寶髻王，手足柔軟如兜羅，}@
%{又曾经是宝髻王，手足柔软像兜罗，}%

@{細滑微妙色如蓮，捨自手足利眾生。}@
%{细滑微妙，颜色像莲花，舍弃自己的手足利益众生。}%

@{又昔曾為安意王，時有商主名星賀，}@
%{又曾经是安意王，当时有个商主名叫星贺，}%

@{領諸商客泛海中，忽然漂墮羅剎國。}@
%{带着众商客航海，突然漂泊到罗刹国。}%

@{彼有百千夜叉女，無慚大惡唯食人，}@
%{那里有上百上千的夜叉女，无耻，极恶，只吃人，}%

@{商客不識夜叉女，見此端正生愛心，}@
%{商客不认识夜叉女，看见她们端庄美丽，就生起爱心，}%

@{五百商旅將被食，我親救度俱脫難。}@
%{五百商旅将要被吃，我亲自救度，他们都脱离了困境。}%

@{又昔曾為妙眼王，四兆女人常圍繞，}@
%{又曾经是妙眼王，四兆个女人常常围绕着我，}%

@{端正殊妙如天女，捨彼出家求佛道。}@
%{端庄殊妙，像天女一样，舍弃她们出家求佛道。}%

@{又昔曾為福光王，無垢清淨黃金色，}@
%{又曾经是福光王，无垢清净，黄金色，}%

@{手指纖長世所希，捨此手指利群品。}@
%{手指纤长，世间罕见，舍弃自己的手指利益众生。}%

@{又昔曾為法財王，紺目清淨如青蓮，}@
%{又曾经是法财王，绀目清净，像青莲，}%

@{於身所愛最難捨，人來求者亦與之。}@
%{对自己身体最爱的东西最难舍，有人来求，我也会给他。}%

@{又昔曾為蓮目王，愍見眾生在苦惱，}@
%{又曾经是莲目王，怜悯众生在苦恼中，}%

@{時有女人懷憂病，我行悲愍令解脫。}@
%{当时有个女人怀着忧愁，生病了，我行悲悯，让她解脱。}%

@{又昔曾為大醫王，常救病苦諸眾生，}@
%{又曾经是大医王，常常救治生病痛苦的众生，}%

@{或出身血、及髓、腦，救療疾病令除愈，}@
%{或者取出他们的血、髓、脑，救治疾病，让他们痊愈，}%

@{如是勇猛精進心，未曾暫捨於情物。}@
%{这样勇猛精进的心，从未暂时舍弃过任何情物。}%

@{又昔曾為成利王，以自所愛如蓮目，}@
%{又曾经是成利王，用自己所爱的东西，像莲目一样，}%

@{施諸眾生療彼疾，一心為求無上道。}@
%{施舍给众生，治疗他们的疾病，一心求证无上道。}%

@{又昔為王名普現，慈愍有情行救度，}@
%{又曾经是王，名叫普现，慈悲怜悯有情，行救度，}%

@{爾時捨彼四大洲、國土、人民及眾寶、}@
%{当时舍弃那四大洲、国土、人民以及珍宝，}%

@{乃至割身血肉等，施與眾生心歡喜。}@
%{乃至割下自己的血肉等等，施舍给众生，心欢喜。}%

@{又為王女稱大智，身嚴金色體柔軟，}@
%{又曾经是王女，名叫大智，身体装饰着金色，柔软，}%

@{時有一女名色相，此是商人所生女，}@
%{当时有个女子名叫色相，她是商人家的女儿，}%

@{飢羸困苦無飲食，我捨雙乳濟彼命。}@
%{饥饿瘦弱，困苦没有食物，我舍弃自己的双乳救济她性命。}%

@{又昔為王號多聞，所有珍寶、妙衣服、}@
%{又曾经是王，名叫多闻，拥有所有的珍宝、妙衣服，}%

@{象、馬、車乘、財帛等，如是布施無有數。}@
%{象、马、车乘、财帛等等，这样布施，无数。}%

@{復見商人漂海浪，我於海內救得彼，}@
%{又看见商人漂泊在海浪中，我在海里面救了他，}%

@{背恩復乞我眼睛，我亦施之無瞋恚。}@
%{他背恩，又向我乞求眼睛，我也施舍给他，没有瞋恚。}%

@{棄捨大地諸眷屬，觀彼不著如蟻子，}@
%{舍弃大地，舍弃所有的眷属，看待他们就像蚂蚁一样，没有执着，}%

@{如是往昔濟群生，心無退動生疲苦。}@
%{这样，过去救济众生，内心没有退转，没有生起疲苦。}%

@{復覩孤老貧窮人，給足所須而承事，}@
%{又看见孤苦贫穷的人，给予他们所需的，并且侍奉他们，}%

@{恒行敬愛無慢心，亦無慚𧹞、無人我。}@
%{恒常行敬爱，没有我慢心，也没有惭愧，没有“我”和“人”。}%

@{又昔作猿猴身，與彼同類共遊行，}@
%{又曾经做猿猴，和同类一起游玩，}%

@{時遇獵師縛彼身，我即替他令脫命。}@
%{当时遇到猎人捆绑它，我就替它解脱，让它免于死亡。}%

@{如是以我奉國王，王令後宮縻繫我，}@
%{这样，我奉献给国王，国王命令后宫囚禁我，}%

@{思念父母年孤老，所有飲饌無心食，}@
%{思念父母年老孤苦，所有饮食都没有心思吃，}%

@{如是忍苦懷慈孝，是故得脫王宮難。}@
%{这样忍耐痛苦，怀着慈孝之心，所以能够脱离王宫的困境。}%

@{又昔曾作大熊身，常處深山行慈忍，}@
%{又曾经做大熊，常常住在深山，行慈忍，}%

@{忽逢樵士遭大雨，引彼山巖令迴避。}@
%{突然遇到樵夫遇到大雨，就引导他到山岩下躲避。}%

@{過是七日至天晴，告彼樵人：莫說我。}@
%{过了七天，天气晴朗，就告诉那个樵夫：不要说我。}%

@{爾時樵士安隱歸，招引獵師來殺害，}@
%{当时樵夫安然回家，就招引猎人来杀害我，}%

@{如是背恩殺我身，亦無瞋恨生慈忍。}@
%{这样背恩杀害我，我依然没有瞋恨，生起慈忍。}%

@{又昔曾為白象王，求佛菩提行十善，}@
%{又曾经是白象王，求佛菩提，修行十善，}%

@{時有獵師射我身，我即捨牙心歡喜。}@
%{当时有猎人射我身体，我就舍弃象牙，心欢喜。}%

@{昔有惡人帝哩子，以火焚燒大山野，}@
%{曾经有个恶人帝哩子，用火焚烧大山野，}%

@{我見此火運慈心，天雨香花火自滅。}@
%{我看见这场火，就运起慈悲心，天降香花，火就自己熄灭了。}%

@{又昔曾為大鹿王，金寶莊嚴體殊妙，}@
%{又曾经是大鹿王，金宝装饰，身体殊妙，}%

@{入彼河中救溺人，令得安隱全身命。}@
%{进入河中救溺水的人，让他安然无恙，全身性命得救。}%

@{告言：勿說我居山，恐彼惡人來獵我。}@
%{告诉他说：不要说我住在山里，恐怕那些恶人会来猎杀我。}%

@{時彼溺人背其恩，指告國王令採捕，}@
%{当时那个溺水的人背恩，指着我告诉国王，让他来捉捕我，}%

@{指已兩手俱墮地，我時無有少瞋恚。}@
%{手指已经掉到地上，我当时没有一点瞋恚。}%

@{昔有五百商人眾，為求珍寶泛海中，}@
%{曾经有五百个商人在海中，为了寻找珍宝，}%

@{商主所有資糧竭，商眾飢羸無飲食。}@
%{商主所有的粮食都用完了，商人们饥饿瘦弱，没有食物。}%

@{是時我作大龜王，捨身濟彼商人命，}@
%{当时我化身成大龟王，舍身救济那些商人的性命，}%

@{以我慈心利他故，俱得安然到海岸。}@
%{因为我慈悲心利益他人，他们都平安到达海岸。}%

@{我昔變身為藥蟲，此蟲名曰俱蘇摩，}@
%{我曾经化身成药虫，这种虫叫做俱苏摩，}%

@{一切疾病食我身，俱獲安隱無諸患。}@
%{所有疾病，吃我的身体，都能获得安稳，没有病患。}%

@{我昔復為師子王，大力無畏行慈愍，}@
%{我曾经又是狮子王，大力无畏，行慈悲，}%

@{有大獵師射我身，亦無瞋恨、無忿怒。}@
%{有个大猎人射我身体，我也没有瞋恨，没有愤怒。}%

@{我昔亦作白馬王，常行菩薩慈悲行，}@
%{我曾经也做白马王，常常行菩萨慈悲行，}%

@{救彼商人羅剎難，擔負眾人出海中。}@
%{救那些商人脱离罗刹的困境，背负着他们出海。}%

@{昔作飛鳥軍拏羅，遠離色欲無散亂，}@
%{曾经做飞鸟军拏羅，远离色欲，没有散乱，}%

@{令彼同類眾飛禽，亦復而行清淨行。}@
%{让那些同类的飞禽，也一起修行清净行。}%

@{又昔因中作兔王，與諸群兔宣法行，}@
%{又曾经在因缘中做兔子王，和所有兔子宣讲佛法，}%

@{見一仙人飢無食，即捨自身濟彼命。}@
%{看见一个仙人饥饿没有食物，就舍弃自己的身体救济他性命。}%

@{又昔曾作鸚鵡身，常居花菓樹林中，}@
%{又曾经做鹦鹉，常常住在花果树林中，}%

%{護國尊者又問：不應該講法還有什麼道理呢？}%

@{佛言：護國！若有補特伽羅無決定者，我不說菩提；於虛妄者，我不說清淨行；於懈怠者，我不說菩薩行；於慳悋者，我不說供佛行；於我慢者，我不說波羅蜜清淨；於無慧者，我不說斷疑法；於嫉妬者，我不說心清淨；於無信根者，我不說總持法；於無德者，我不說善逝法；於貪親愛者，我不說身清淨；於不善律儀者，我不說謗佛有過失法；於妄言者，我不說語清淨；於我慢者，我不說恭敬法；於無識者，我不說修學法；於為身命者，我不說求於道法。如是補特伽羅不應說法。}@
%{佛祖說：護國尊者！如果有人沒有決心，我不講菩提道；對虛妄的人，我不講清淨行；對懶惰的人，我不講菩薩行；對吝嗇的人，我不講供養佛陀的行；對驕傲的人，我不講六度波羅蜜的清淨；對沒有智慧的人，我不講斷除疑慮的方法；對嫉妒的人，我不講心清淨；對沒有信心的人，我不講總持法；對沒有德行的人，我不講善逝法；對貪圖親情的人，我不講身體清淨；對不守戒律的人，我不講誹謗佛陀的過失；對說謊的人，我不講言語清淨；對驕傲的人，我不講恭敬法；對沒有智慧的人，我不講修學的方法；對為了自身生命的人，我不講求道法。這樣的人不應該講法。}%

@{時護國白言：於意云何？}@
%{護國尊者說：您怎麼看呢？}%

@{佛言：護國！為此有情愚癡迷惑、心識顛倒、虛妄分別、不依法教，乃至天上人間不應為說。}@
%{佛祖說：護國尊者！因為這些眾生愚癡迷惑，心識顛倒，胡亂分別，不按規矩修行，所以，即使是天上人間，也不應該為他們講法。}%

@{爾時，世尊而說頌曰：}@
%{當時，佛祖就說了這首偈子：}%

@{不定諸有情，補特伽羅等，}@
%{那些不定性的人，那些有缺陷的人，}%

@{我慢自貢高，貪著於利養，}@
%{驕傲自大，貪圖財利，}%

@{恒行不律儀，深著於五欲，}@
%{經常不守戒律，沉迷於五欲，}%

@{增長諸煩惱，遠離佛菩提。}@
%{加重自己的煩惱，遠離佛陀的菩提。}%

@{退失於善法，懈怠不修習，}@
%{退失了善法，懶惰不修行，}%

@{猶豫多散亂，於其戒法言，}@
%{猶豫不決，心神散亂，對戒律法語，}%

@{而不生信受。因為貧窮逼，}@
%{沒有生起信心。因為貧困逼迫，}%

@{方便求出家，設得作苾芻，}@
%{才勉強出家，即使成為了比丘，}%

@{輕捨於道法，如棄金寶擔，}@
%{卻輕視道法，就像丟掉金寶，}%

@{荷負於麻擔。雖欲入深山，}@
%{卻背負著麻袋。雖然想進入深山，}%

@{到彼寂靜處，無意樂修禪，}@
%{到達寂靜的地方，卻沒有興趣修禪定，}%

@{邪思而散亂，障礙於辯才，}@
%{邪念叢生，心神散亂，阻礙了智慧和辯才，}%

@{沈沒大智慧，墜墮惡趣中。}@
%{沉淪於大智慧，墮落到惡道之中。}%

@{設復得人身，醜陋不具足，}@
%{即使得到了人身，卻又醜陋，不具備好的條件，}%

@{懈怠性愚癡，不行眾善法。}@
%{懶惰愚蠢，不修行各種善法。}%

@{諸根常暗鈍，墮大嶮難中，}@
%{六根經常昏鈍，墮落到巨大的險難之中，}%

@{經彼俱胝劫，迷沒不解脫。}@
%{經歷無數劫，迷失在輪迴中，無法解脫。}%

@{若行邪利濟，得證佛菩提，}@
%{如果為了邪惡的利益，而想要證得佛陀的菩提，}%

@{調達不正知，應成善逝果。}@
%{就像調達王不了解正法，卻想成為佛陀一樣。}%

@{若人貪利養，墜墮於眾生，}@
%{如果有人貪圖財利，就會墮落到眾生之中，}%

@{如空大風力，能墮諸飛鳥，}@
%{就像空中的大風，能吹落飛鳥一樣，}%

@{邪福勢盡時，其義亦如是。}@
%{邪惡的福報力量消失時，道理也是一樣的。}%

@{無信破戒者，見善如盲人，}@
%{沒有信心，破戒的人，就像盲人一樣看不見善法，}%

@{譬如焚屍柴，不吉人嫌棄。}@
%{就像焚燒屍體的柴火，不吉利的人會厭惡它。}%

@{雖復發善心，無彼廣大智，}@
%{雖然發了善心，卻沒有廣大的智慧，}%

@{謗法不信故，解脫非究竟。}@
%{因為誹謗佛法，沒有信心，所以無法徹底解脫。}%

@{譬如畫無膠，莊嚴色不久，}@
%{就像沒有膠水的畫，裝飾的顏色不會持久，}%

@{我慢自貢高，其義亦如是。}@
%{驕傲自大，道理也是一樣的。}%

@{若求佛菩提，不惜於身命，}@
%{如果想要追求佛陀的菩提，不惜生命，}%

@{於法甚深言，勇猛勤習學。}@
%{對佛法深奧的教義，勇猛精進地學習。}%

@{捨善行非法，所行增過失，}@
%{拋棄善法，行持惡法，所作所為會增加過失，}%

@{墮於大火坑。若聞如是法，}@
%{墮落到巨大的火坑之中。如果聽到這些法語，}%

@{依法而受行、斷除貪愛心、}@
%{按照佛法修行，斷除貪愛之心，}%

@{修植眾德本，乃至於一句，}@
%{修持各種德行，即使是一句法語，}%

@{通達悉明了，如是積功德，}@
%{都能通達明瞭，這樣積累功德，}%

@{成就最上道，永離於愚盲。}@
%{就能成就最殊勝的道路，永遠擺脫愚昧。}%
@{爾時，世尊說此頌已，告尊者護國言：我於過去無量無邊不可說不可說阿僧祇劫，時有佛出世，號曰成義意如來、應供、正遍知、明行足、善逝、世間解、無上士、調御丈夫、天人師、佛、世尊。}@
%{话说，佛祖讲完这首偈子后，就对护国尊者说：在无量无边、不可说不可说阿僧祇劫的遥远过去，曾经有一位佛出世，他的名字叫成义意如来，应供、正遍知、明行足、善逝、世间解、无上士、调御丈夫、天人师、佛、世尊。}%

@{爾時，有大國王名曰發光，主閻浮提，其地廣闊一萬六千由旬，其中州城數滿二萬。其發光王所居城邑名曰寶光，其城東西長十二由旬，南北闊七由旬，城有七重七寶所作。彼王善行八正之道，種族豪盛有千俱胝，其國人民壽十俱胝歲。}@
%{当时，有一位名叫发光的国王，统治着阎浮提，他的国家疆域广阔，一万六千由旬，其中有二万座城池。发光王居住的城名叫宝光城，城东西长十二由旬，南北宽七由旬，城墙由七层七宝堆砌而成。这位国王善行八正之道，他的家族兴盛，有千俱胝人口，国家人民寿命长达十俱胝岁。}%

@{王有太子名曰福光，諸根具足，色相端嚴，殊妙第一。太子生時，有千寶藏從地涌出——內有一藏現王殿前，滿中七寶，高七人量——復令一切眾生所作如意，乃至禁縛之者俱得解脫。又，彼大子生得七日，一切伎藝、工巧、算術皆悉明了，乃至世、出世間一切事業無不通解。}@
%{国王有一个太子，名叫福光，他根性具足，相貌端庄，非常殊妙，是第一等的。太子出生的时候，有千座宝藏从地里涌出，其中有一座宝藏出现在王宫殿前，里面装满了七宝，高七个人那么高，而且能够满足一切众生的愿望，即使被禁锢的人也能获得解脱。此外，太子出生七天后，就精通一切技艺、巧艺、算术，甚至世间、出世间的一切事情都无所不通。}%

@{於夜分中有淨光天子來為說法，告太子言：福光諦聽！汝須息心，不應散亂，於諸塵境常當遠離。晝夜思惟有為之法，當觀無常、壽命盡時誰是救者，於諸非法而生怖畏。}@
%{在夜半时分，净光天子来到太子面前说法，对太子说：福光，你要专心听讲！你应该静心，不要心猿意马，要远离一切尘世间的境相。白天黑夜都要思考有为法，要观察无常，当生命终结的时候，谁是你的救主？你要对一切非法产生恐惧。}%
