@{持法王名者，其福亦如是。}@
%{持诵佛陀的名号，得到的福报也是如此。}%

@{若人持佛名，七寶華中生，}@
%{如果有人持诵佛名号，就会出生在七宝莲花中，}%
@{其華千億葉，威光相具足。}@
%{这朵莲花有千亿片花瓣，威光具足，光芒四射。}%

@{若人持佛名，父母諸眷屬，}@
%{如果有人持诵佛名号，他的父母和亲人，}%
@{和顏無諍訟，永不生別離。}@
%{都会和颜悦色，没有争吵，永远不会分离。}%

@{若有諸女人，受持此經者，}@
%{如果有人是女性，持诵这部经书，}%
@{捨離女人形，轉生智男身；}@
%{就能舍弃女身，转生为智慧的男性；}%
@{得男子身已，即成於菩提，}@
%{得到男身之后，就能成就菩提，}%
@{轉無上法輪，隨意入涅槃。}@
%{转动无上法轮，随心所欲地进入涅槃。}%

@{若人持佛名，刀杖不能害，}@
%{如果有人持诵佛名号，刀杖不能伤害他，}%
@{水火不焚漂，縣官不能殺。}@
%{水火不能烧毁他，官府也不能杀害他。}%

@{若人持佛名，眾魔及波旬，}@
%{如果有人持诵佛名号，各种魔和波旬，}%
@{行住坐臥處，不能得其便。}@
%{在他行走、站立、坐卧的地方，都不能得逞。}%

@{若人持佛名，世世所生處，}@
%{如果有人持诵佛名号，生生世世，}%
@{身通遊虛空，能至無邊剎，}@
%{身体能够自由地穿梭于虚空，能够到达无边世界，}%
@{面覩於諸佛，能問甚深義，}@
%{亲眼见到佛陀，能够询问深奥的佛法，}%
@{彼等無量佛，即知其心意，}@
%{那些无量佛陀，马上就能知道他的心意，}%
@{為說微妙法，授彼菩提記。}@
%{为他讲述微妙的佛法，给他授予菩提记。}%

@{得聞授記已，心生大歡喜，}@
%{听到授记之后，心中充满欢喜，}%
@{即於諸佛教，決定無有疑。}@
%{对佛陀的教诲，坚定不移，没有丝毫的怀疑。}%

@{彼諸佛世尊，所說甚深法，}@
%{那些佛陀世尊，所讲的深奥佛法，}%
@{其人聞法已，受持永不失。}@
%{这个人听到之后，持诵而不忘记。}%

@{於後得成就，六度及諸地，}@
%{后来就能成就六度和菩萨地，}%
@{無畏諸力等，眾相及諸好，}@
%{获得无畏的力量，以及种种相好，}%
@{佛剎及眾生，任意隨所取。}@
%{能够随意选择佛国和众生，}%
@{生於彼剎中，速能成正覺，}@
%{出生在那个佛国，就能很快成就正觉，}%
@{轉無上法輪，隨意入涅槃。}@
%{转动无上法轮，随心所欲地进入涅槃。}%

@{若有善男子，及善女人等，}@
%{如果有人，不管是男性还是女性，}%
@{受持此經者，如上說功德。}@
%{持诵这部经书，就能获得以上所说的功德。}%

@{若持此經者，謹慎莫放逸，}@
%{如果持诵这部经书，一定要谨慎小心，不要懈怠，}%
@{聞如前福業，若滿於一劫，}@
%{听到这些福报，如果能够持诵满一劫，}%
@{若減於一劫，常須勤讀誦，}@
%{如果少于一劫，也要经常勤奋地读诵，}%
@{不生懈怠心。我勅有如是，}@
%{不要产生懈怠之心。我这样宣告，}%
@{此等諸佛者，常住功德藏，}@
%{这些佛陀，永远住持功德宝藏，}%
@{虛空雲中王，常住於彼處。}@
%{就像虚空中的云中之王，永远住持在那里。}%

@{若人聞佛名，一劫減一劫，}@
%{如果有人听到佛陀的名号，即使只持诵了一劫减一劫，}%
@{聞已生敬心，彼是最健人，}@
%{听到之后，就生起敬畏之心，这个人是最健壮的人，}%
@{亦是大智慧，亦是人中最，}@
%{也是大智慧的人，也是人中之最，}%
@{亦是健丈夫，於世大名聞。}@
%{也是强壮的男子汉，在世间名声远扬。}%

@{是故我今言，彌勒當聽我，}@
%{所以我现在对你说，弥勒，你应该听我讲，}%
@{有大智慧人，巧解方便者，}@
%{有大智慧的人，善于巧妙运用方便法门的人，}%
@{常勤不放逸，恒近善知識，}@
%{经常勤奋努力，不懈怠，经常亲近善知识，}%
@{得聞此經典，後趣菩提道。}@
%{听到这部经书，将来就能趣入菩提道。}%

@{佛說此經已，彌勒菩薩摩訶薩，及諸菩薩眾、大比丘眾，天、龍、夜叉、乾闥婆、阿修羅、迦樓羅、緊那羅、摩睺羅伽、人非人等，聞佛所說，歡喜奉行。}@
