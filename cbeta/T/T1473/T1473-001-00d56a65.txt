@{佛說沙彌十戒儀則經(計七十二頌)}@
%{佛說沙彌十戒儀則經(計七十二頌)}%

@{西天譯經三藏朝散大夫試鴻臚少卿傳法大師臣施護奉 詔譯}@
%{西天譯經三藏朝散大夫試鴻臚少卿傳法大師臣施護奉 詔譯}%

@{頂禮一切智，妙法及聖眾，}@
%{向一切智慧、妙法和圣众顶礼膜拜，}%
@{略說沙彌行，令發出家心。}@
%{简单地说一下沙弥的修行，让他们能够生出出家的心。}%
@{於彼釋迦教，堅持於禁戒，}@
%{在释迦牟尼的教法中，要坚持遵守戒律，}%
@{持戒如護身，防守勿令犯。}@
%{持戒就像保护自己的身体一样，要严守不犯。}%

@{夜臥從早起，淨口及牙齒，}@
%{晚上睡觉，早上起来，洗漱干净，}%
@{念誦至天明，頂禮正等覺。}@
%{念诵经文直到天亮，然后向正等觉者顶礼膜拜。}%

@{參近於師房，以手輕擊門，}@
%{来到师父的房间，用手轻轻地敲门，}%
@{入已問訊師，身體安樂不？}@
%{进门后问候师父，身体是否安好？}%

@{如是所作已，復作曼拏羅，}@
%{做完这些事之后，再制作曼荼罗，}%
@{供養而恭敬，早辰觀水內，}@
%{恭敬地供养它，清晨观察水里，}%
@{中後時亦然，甕器盂鉢等。}@
%{中午和下午也一样，观察坛、器皿、钵等。}%

@{水蟲有大小，志心恒觀照，}@
%{水里的虫子有大有小，要用心观察，}%
@{以羅淨濾水，審觀而可用。}@
%{用过滤网过滤水，仔细观察后才能使用。}%

@{食飲懷慈愍，勿令殺害蟲，}@
%{吃喝时要怀着慈悲心，不要杀害虫子，}%
@{乃至草木上，塗壇牛糞中，}@
%{甚至草木上，涂坛牛粪中，}%
@{如是受用時，救護於含識。}@
%{在使用这些东西的时候，要保护有知觉的生命。}%

@{或彼床座內，田地糞土中，}@
%{或者在床铺、田地、粪土中，}%
@{一一子細觀，是名出家行。}@
%{都要仔细观察，这就是出家人的行为。}%

@{一切承事師，洗鉢令清淨，}@
%{所有侍奉师父的事，都要洗干净钵，}%
@{依時三作禮，合掌向師前。}@
%{按时向师父行三拜礼，合掌站在师父面前。}%

@{問師何所食？作食治飢病，}@
%{问师父吃什么？做饭来治疗饥饿和疾病，}%
@{如是若為食，用前清淨水，}@
%{如果要吃饭，就用之前净化的水，}%
@{以水灌手淨，如法默然食。}@
%{用水洗净双手，按照方法默默地吃饭。}%

@{食已說二偈，迴向於信施，}@
%{吃完饭后，念诵两句偈语，回向给信施的人，}%
@{如說而依行，增長正念行。}@
%{就像经文中说的一样，按照方法去做，就能增长正念。}%

@{信解淨戒法，依法而修持，}@
%{相信并理解清净戒律，按照戒律修行，}%
@{正觀自相應，出家心安樂。}@
%{正观与自己相符，出家的心就会安乐。}%

@{若人自犯戒，他見生輕毀，}@
%{如果有人自己犯戒，别人看到就会轻视和毁谤，}%
@{二人皆得罪，持者須一心。}@
%{两个人都会犯错，持戒的人要一心一意。}%

@{持戒或破戒，乃至病患人，}@
%{持戒或者破戒，甚至生病的人，}%
@{說法不當機，他聞心不重，}@
%{说法不符合时机，别人听了心不重视，}%
@{命終得大苦。若人行毒藥，}@
%{死后就会得到很大的苦报。如果有人下毒药，}%
@{及行呪法等，令他得命終，}@
%{或者使用咒法等，让他死亡，}%
@{又復以種種，方便而行殺，}@
%{又用各种各样的方法去杀生，}%
@{如是破戒因，永沈三惡道，}@
%{这样破戒的因缘，就会永远沉沦到三恶道，}%
@{地獄鬼畜生，如次受罪報。}@
%{地狱、鬼道、畜生道，依次承受罪报。}%

@{及彼諸天輩，若或亦殺生，}@
%{即使是天上的众生，如果也杀生，}%
@{還墮惡趣中，受彼諸惡報。}@
%{也会堕落到恶趣之中，承受各种恶报。}%

@{若有出家者，恣用身口意，}@
%{如果有人出家，任意使用身口意，}%
@{誤殺螻蟻類，於其三業中，}@
%{误杀了蚂蚁之类的虫子，在他的身口意三业中，}%
@{而得三種罪。若人以拳棒，}@
%{就会得到三种罪过。如果有人用拳头、棍棒，}%
@{土石及塼瓦，打擲於有情，}@
%{石头、砖瓦，去打有情众生，}%
@{駝騾禽獸等，亦得破戒罪。}@
%{比如骆驼、骡子、禽兽等，也会犯破戒的罪。}%

@{駝騾象馬類，而欲輒乘騎，}@
%{如果想要骑乘骆驼、骡子、象、马等，}%
@{往來逼迫他，亦得破戒罪。}@
%{来来往往地逼迫它们，也会犯破戒的罪。}%

@{若有出家者，不得行偷盜，}@
%{如果有人出家，不能偷盗，}%
@{若自若教他，偷得於財物，}@
%{无论是自己偷盗，还是教唆别人偷盗财物，}%
@{迦哩沙波拏，四分中一分，}@
%{都会犯戒，而且是四分之一的重罪。}%

@{如是破戒因，而成最重罪。}@
%{这样破戒的因缘，就会成为最重的罪过。}%

@{飲食穀米等，花菓草木類，}@
%{饮食、谷米等，花果草木类，}%
@{地生或水生，而行於偷竊，}@
%{无论是地上的，还是水里的，都不要偷窃，}%
@{或偷官稅利，私過於關津，}@
%{或者偷盗官税、利润，私自越过关口，}%
@{及盜於有情，二足與多足。}@
%{或者偷盗有情众生，比如两足和多足的动物。}%

@{或彼出家人，受用雖具足，}@
%{即使出家人拥有足够的资用，}%
@{貪心復盜財，俱獲於重罪。}@
%{如果贪心又去偷盗财物，都会得到严重的罪过。}%

@{若自衣鉢等，被賊所劫盜，}@
%{如果自己的衣服、钵等，被贼人抢劫，}%
@{勿得強取之，說法方便化，}@
%{不要强行夺回来，要用说法来巧妙地化解，}%
@{或復而迴買，不允隨他意。}@
%{或者用钱赎回来，不要随他的意愿。}%

@{是名出家心，不動三業瘡，}@
%{这就是出家的心，三业不会有病痛，}%
@{戒性自成就。若有出家者，}@
%{戒律自然圆满。如果有人出家，}%
@{不得行婬欲。女男及黃門，}@
%{不能行淫欲。无论是男女，还是宦官，}%
@{自來相慕欲，愚迷而愛著，}@
%{彼此互相爱慕，愚痴迷惑而执着，}%
@{得彼根本罪。若彼故行婬，}@
%{就会犯根本的罪过。如果因此而行淫，}%
@{如蛇如毒藥，損壞於自身，}@
%{就像蛇和毒药一样，会损害自己的身体，}%
@{得大地獄苦，障礙於涅槃，}@
%{会堕入大地狱，障碍涅槃，}%
@{不出於生死。正覺非如是，}@
%{不能脱离生死轮回。正觉不是这样的，}%
@{未曾身犯戒，持戒獲大利。}@
%{没有身体犯戒，持戒就能得到很大的利益。}%

@{若有出家者，不得出妄語。}@
%{如果有人出家，不能说妄语。}%
@{若言見天人，與我同言語，}@
%{如果说看到天人，跟我说话，}%
@{大人所住處，我自亦曾到，}@
%{或者说去过天人居住的地方，}%
@{乃至乾闥婆，恭伴龍夜叉，}@
%{甚至乾闥婆、龙、夜叉，}%
@{摩睺羅伽輩，毘舍緊那羅，}@
%{摩睺罗伽、紧那罗，}%
@{鉢哩哆鬼等，與我恒言話，}@
%{鉢哩哆鬼等，都跟我说话，}%
@{此皆為妄語。或言得五通，}@
%{这些都是妄语。或者说得到五通，}%
@{正道及四果，曉了甚深法，}@
%{或者说证得正道和四果，明了深奥的佛法，}%
@{未得言為得，如斯之妄語，}@
%{如果还没有得到，却说已经得到，这种妄语，}%
@{永沈於惡趣。不得起兩舌，}@
%{会永远沉沦到恶趣。不能挑拨离间，}%
@{離別他善友，及彼麁惡言，}@
%{离间别人的朋友，或者说粗俗恶劣的话，}%
@{綺說非與是。設被他毀罵，}@
%{花言巧语，说是非曲直。即使被别人辱骂，}%
@{勿得酬對他，種種鬪諍言，}@
%{也不要反驳他，说各种争斗的话，}%
@{一一須忍受。若彼不依行，}@
%{都要忍耐。如果他不按方法去做，}%
@{而得犯戒罪；智者一心持，}@
%{就会犯戒的罪过；智者要一心持戒，}%
@{獲離口業過。若以穀米等，}@
%{就能远离口业的过失。如果用谷米等，}%
@{為酒醉於人，不得而故飲。}@
%{酿酒让别人喝醉，不能故意饮酒。}%

@{或以甘蔗花，蒲萄菓實等，}@
%{或者用甘蔗花、葡萄果实等，}%
@{修醞可醉人，不得而故飲。}@
%{酿制可以让人喝醉的酒，不能故意饮酒。}%

@{自飲教他飲，迷亂而失念，}@
%{自己喝，教别人喝，迷乱而失去正念，}%
@{增長放逸心，飲者得重罪。}@
%{会增长放逸的心，喝的人会得到重罪。}%

@{是故世尊言，若人以草葉，}@
%{所以世尊说，如果有人用草叶，}%
@{滴酒於口中，增長於過失。}@
%{滴酒到嘴里，就会增长过失。}%

@{歌舞兼倡妓，故往而觀看，}@
%{歌舞和妓女，故意去看，}%
@{而得犯戒罪。香油塗飾身，}@
%{就会犯破戒的罪。用香油涂抹身体，}%
@{旃檀欝金等，及以好花鬘，}@
%{檀香、郁金等，以及美丽的花环，}%
@{金銀珠寶類，種種莊嚴身，}@
%{金银珠宝等，各种装饰身体，}%
@{而得犯戒罪。若於眼目上，}@
%{就会犯破戒的罪。如果在眼睛上，}%
@{點畫令端正，而得犯戒罪。}@
%{点画来装饰，使它看起来端正，就会犯破戒的罪。}%

@{座床臥床等，量高可一肘，}@
%{坐床、卧床等，高度可以到一肘，}%
@{亦不作莊嚴，復不令廣大，}@
%{也不要装饰，也不要让它宽大，}%
@{放逸不依行，而得犯戒罪。}@
%{放逸不按照方法去做，就会犯破戒的罪。}%

@{佛說出家人，過失宜遠離。}@
%{佛说出家人，应该远离过失。}%

@{若受齋食時，不得過中午，}@
%{如果接受斋食，不能超过中午，}%
@{日出至午前，可許受齋飯；}@
%{从日出到午前，可以接受斋饭；}%
@{非時而喫食，佛說得重罪。}@
%{不在规定的时间吃饭，佛说会得到重罪。}%

@{如有比丘病，治病救於身，}@
%{如果有比丘生病，为了治病救治身体，}%
@{可許中後食；無病不依時，}@
%{可以允许在中午和下午吃饭；如果没病，不按照时间吃饭，}%
@{而得犯戒罪。金銀珍寶等，}@
%{就会犯破戒的罪。金银珠宝等，}%
@{出家不得觸，受者生於貪，}@
%{出家人不能接触，接受的人会产生贪心，}%
@{而得犯戒罪。若有檀信施，}@
%{就会犯破戒的罪。如果有檀越施主，}%
@{供養佛法僧，為彼興福利，}@
%{供养佛法僧，为他们兴旺福利，}%
@{可許而受用。一切快樂具，}@
%{可以接受他们的供养。所有快乐的东西，}%
@{勿得生愛著，若有貪著者，}@
%{不要产生贪爱，如果有贪爱的人，}%
@{彼人須擯出。所有戲笑等，}@
%{这个人必须被驱逐出去。所有嬉戏玩耍等，}%
@{沙彌不得作，作者得犯罪。}@
%{沙弥不能做，做的人会犯罪。}%

@{若有呵欠時，以手遮蓋口，}@
%{如果打哈欠，用手遮住嘴，}%
@{不依而得罪。上座作啑噴，}@
%{不按照方法去做就会犯错。上座打喷嚏，}%
@{不審而頂禮，新戒等啑噴，}@
%{不仔细就顶礼，新受戒的人打喷嚏，}%
@{上座須呪願，長壽無其病。}@
%{上座要念咒，祝愿他长寿无病。}%

@{沙彌向師前，不得於洟唾，}@
%{沙弥在师父面前，不能擤鼻涕，}%
@{不得刷牙齒。不得向師前，}@
%{不能刷牙。不能在师父面前，}%
@{經行及對座。耆年宿德者，}@
%{经行和对面坐着。年老德高的人，}%
@{如法而尊重。若入觸溷時，}@
%{要按照方法尊敬。如果进入厕所，}%
@{如有後來者，低聲而警覺。}@
%{如果有后来的人，要低声提醒。}%

@{所濾無蟲水，用添於淨瓶，}@
%{过滤过没有虫子的水，用来添到净瓶里，}%
@{淨手安詳行，不失出家行。}@
%{洗净双手，安详地行走，就不会失去出家人的行为。}%

@{師自座臥床，恒用淨襯褥，}@
%{师父自己坐卧的床，要经常用干净的垫子，}%
@{所有床椅等，不得沙彌座。}@
%{所有的床椅等，沙弥不能坐。}%

@{師自若不在，依止清淨僧，}@
%{如果师父不在，要依止清净的僧人，}%
@{明了律儀者，承事而侍奉；}@
%{明白戒律的人，侍奉他们；}%
@{勿更依別僧，違者而得罪。}@
%{不要依止其他的僧人，违反了就会犯错。}%

@{比丘若出行，而為求緣事，}@
%{比丘如果外出，为了求缘的事情，}%
@{五日內復歸，得利同行眾；}@
%{五天内回来，得到的利益要与同行的人分享；}%
@{眾得利亦然，復與外來者，}@
%{大家得到的利益也一样，也要与外来的人分享，}%
@{如過五日後，二俱各無分。}@
%{如果超过五天，两个人就都没有份了。}%

@{沙彌大小便，須問佛僧地，}@
%{沙弥大小便，要问佛僧指定的地方，}%
@{不得亂往來，行依自界分。}@
%{不能乱跑，要按照自己的界域活动。}%

@{一切凡所作，合掌先問師，}@
%{所有做的事情，都要合掌先问师父，}%
@{如是所作為，不失於正行。}@
%{这样去做，就不会失去正道。}%

@{日沒禮佛塔，禮已復問師，}@
%{日落的时候，礼拜佛塔，礼拜完后，再问师父，}%
@{與師濯雙足，事畢一心聽，}@
%{和师父洗脚，事情做完后，一心一意地听，}%
@{顒望於所須。初夜及後夜，}@
%{恭敬地等待师父的指示。初夜和后夜，}%
@{持誦勿須停。中夜睡眠時，}@
%{持诵经文不要停止。半夜睡觉的时候，}%
@{如彼師子臥。所說一切法，}@
%{就像狮子一样睡觉。所有讲过的法，}%
@{志心而依行，薰識賢種子，}@
%{要用心去实践，熏习贤善的种子，}%
@{煩惱自斷除，速成無上覺。}@
%{烦恼自然断除，就能迅速成就无上觉悟。}%

@{佛說沙彌十戒儀則經}@
%{佛说沙弥十戒仪则经}%
