@{舉。世尊一日。見文殊在門外立。乃云。文殊文殊何不入門來。文殊云。我不見一法在門外。何以教我入門。}@
%{举。世尊有一天，看见文殊在门外站着，就说：“文殊文殊，为什么不进门来？”文殊说：“我没看见一法在门外，怎么教我入门呢？”}%

@{代云。啟予者多。}@
%{代云。启予者多。}%

@{舉。世尊因。外道問。昨日說何法。云。說定法。又問。今日說何法。云說不定法。外道云。昨日說定。今日何說不定。世尊云。昨日定今日不定。}@
%{举。世尊因为外道问：“昨天讲了什么法？”世尊说：“讲定法。”又问：“今天讲了什么法？”世尊说：“讲不定法。”外道说：“昨天讲定，今天怎么讲不定？”世尊说：“昨天定，今天不定。”}%

@{代外道。相顧而去。}@
%{代外道。相顾而去。}%

@{舉。世尊臨入涅槃。文殊請再轉法輪。世尊咄云。吾四十九年住世。未曾說一字。汝請再轉法輪。是吾曾轉法輪耶。}@
%{举。世尊临入涅槃，文殊请求再转法轮。世尊呵斥道：“我四十九年住世，从未说过一句话。你请求再转法轮，是我曾经转过法轮吗？”}%

@{代文殊云。世尊末後殷勤。}@
%{代文殊云。世尊末后殷勤。}%

@{舉。梁武帝請傅大士講經。士纔陞座。以尺拊案一下。便下座。帝愕然。誌公乃問。陛下還會麼。帝云。不會。誌公云。大士講經竟。}@
%{举。梁武帝请傅大士讲经，大士刚上座，就用尺子拍了一下案子，便下座。帝愕然，志公就问：“陛下会吗？”帝说：“不会。”志公说：“大士讲经完了。”}%

@{代武帝云。實為罕聞。}@
%{代武帝云。实为罕闻。}%

@{舉。忠國師因。肅宗皇帝問。師在曹溪得何法。師云。陛下還見空中一片雲麼。帝云。見。師云。丁釘著懸挂著。}@
%{举。忠国师因为肃宗皇帝问：“你在曹溪得什么法？”师说：“陛下还看见空中一片云吗？”帝说：“看见。”师说：“钉钉着悬挂着。”}%

@{代云。曹溪何在。}@
%{代云。曹溪何在。}%

@{舉。忠國師問南泉。甚處來。泉云。江西。師云。還將得馬大師真來否。泉云。只者是。師云。背後底聻。泉休去。}@
%{举。忠国师问南泉：“从哪里来？”泉说：“江西。”师说：“还将得马大师真来否？”泉说：“只者是。”师说：“背后底聻。”泉休去。}%

@{代南泉。以脚一抹便行。}@
%{代南泉。以脚一抹便行。}%

@{舉。忠國師因。麻谷到。乃振錫一下。卓然而立。師云。汝既如是。何用見吾。谷又授錫一下。}@
%{举。忠国师因为麻谷到，就振锡一下，卓然而立。师说：“你既然如此，何用见我？”谷又授锡一下。}%

@{代云。子合見吾。}@
%{代云。子合见我。}%

@{舉。忠國師因。虞軍容問。師住白崖山時。如何修行。師喚童子。以手摩頂云。惺惺直言惺惺。歷歷直言歷歷。向後莫受人謾。軍容無語。}@
%{举。忠国师因为虞军容问：“你在白崖山时，如何修行？”师唤童子，用手摩顶说：“惺惺直言惺惺，历历直言历历。向后莫受人瞒。”军容无言。}%

@{代云。弟子此去。更不禮佛。}@
%{代云。弟子此去，更不礼佛。}%

@{舉。豐干因。寒山子問。古鏡不磨。如何照燭。干云。氷壺無影像。猿猴探水月。山云。此是不照燭也。更請道。干云。萬德不將來。教我道甚麼。寒拾俱作禮而退。}@
%{举。丰干因为寒山子问：“古镜不磨，如何照烛？”干云：“冰壶无影像，猿猴探水月。”山云：“此是不照烛也。更请道。”干云：“万德不将来，教我道什么？”寒拾俱作礼而退。}%

@{代云。因我致得。}@
%{代云。因我致得。}%

@{舉。布袋和尚。常將布袋并破蓆。於通衢往來。布袋內盛鉢盂木履魚飯菜肉瓦石土木諸般總有。或於稠人處。打開布袋內物。撒下云。看看。又一一將起問人云。者箇喚作甚麼。眾無對。}@
%{举。布袋和尚，常将布袋并破席，于通衢往来。布袋内盛钵盂木履鱼饭菜肉瓦石土木诸般总有。或于稠人处，打开布袋内物，撒下云：“看看。”又一一将起问人云：“者箇唤作什么？”众无对。}%

@{代云。醜婦顰眉。}@
%{代云。丑妇颦眉。}%

@{舉。無著往臺山。文殊問。大德從何方而來。云。南方。云。南方佛法如何住持。云。末法比丘少奉戒律。云。多少眾。云。或三百或五百。無著却問。和尚此間如何住持。云。凡聖同居。龍蛇混雜。云。多少眾。云。前三三後三三。}@
%{举。无著往台山，文殊问：“大德从何方而来？”云：“南方。”云：“南方佛法如何住持？”云：“末法比丘少奉戒律。”云：“多少众？”云：“或三百或五百。”无著却问：“和尚此间如何住持？”云：“凡圣同居，龙蛇混杂。”云：“多少众？”云：“前三三后三三。”}%

@{代云。幾不到此。}@
%{代云。几不到此。}%

@{舉。無著臺山遇文殊。喫茶次。殊拈起玻璃盞問。南方還有者箇麼。云。無。殊云。尋常將甚麼喫茶。著無對。}@
%{举。无著台山遇文殊，喫茶次，殊拈起玻璃盏问：“南方还有者箇麼？”云：“无。”殊云：“寻常将什么喫茶？”著无对。}%

@{代一揖便起。}@
%{代一揖便起。}%

@{舉。南泉示眾。王老師賣身去也。還有人買麼。時有僧出云。某甲買泉。云。不作貴。不作賤。作麼生買。僧無對。}@
%{举。南泉示众：“王老师卖身去也，还有人买吗？”时有僧出云：“某甲买泉。”云：“不作贵，不作贱，作么生买？”僧无对。}%

@{代云。若有人看事。某也不辭。}@
%{代云。若有人看事，某也不辞。}%

@{舉。南泉示眾。法身具四大。有人道。得。與他一腰裩。道吾出云。性地非空。空非性地。此是地大。三大亦然。泉不違前言。乃與一腰裩。}@
%{举。南泉示众：“法身具四大。”有人道：“得。”与他一腰裩。道吾出云：“性地非空，空非性地。此是地大，三大亦然。”泉不违前言，乃与一腰裩。}%

@{代道吾。叩齒揑訣。}@
%{代道吾。叩齿揑诀。}%

@{舉。黃檗示眾。汝等諸人。盡是噇酒糟漢。與麼行脚。何處有今日。還知大唐國裏。無禪師麼。時有僧出云。只如諸方𭅕徒領眾。又作麼生。云。不道無禪。只是無師。}@
%{举。黄檗示众：“汝等诸人，尽是噇酒糟汉，与么行脚，何处有今日？还知大唐国里，无禅师吗？”时有僧出云：“只如诸方𭅕徒领众，又作么生？”云：“不道无禅，只是无师。”}%

@{代僧云。深荷成褫。}@
%{代僧云。深荷成褫。}%
@{舉。趙州在南泉。時井樓上。見泉行過。乃垂一足云。相救相救。泉於蹈梯上打云。一二三四五。州具威儀。上方丈云。適來謝和尚相救。}@
%{赵州和尚在南泉寺，当时在井楼上，看到泉水经过，就伸出一只脚说：“救救我，救救我。”泉水就从梯子上跳下来打了他一下，说：“一二三四五。”赵州和尚就站好，走到方丈那里说：“刚刚来感谢和尚救我。”}%

@{代云。老僧不著便。}@
%{泉水说：“老僧不穿鞋子。”}%

@{舉。西堂因。路逢天使。留齋次。偶驢子鳴。使云。頭陀。堂舉頭。使指驢。堂却指使。使無對。}@
%{西堂和尚在路上遇到天使，留下来一起吃饭。这时，驴子叫了起来，天使说：“头陀。”西堂和尚抬起头，天使指着驴子，西堂和尚却指着天使。天使无言以对。}%

@{代云。下官罪過。}@
%{天使说：“下官有罪了。”}%

@{舉。歸宗因。李渤問。須彌納芥子。渤即不疑。芥子納須彌。莫是妄否。宗云。人傳史君讀萬卷書。是否。曰然。宗云。身如椰子大。萬卷書著向甚處。李公俛首而已。}@
%{归宗和尚，李渤问他：“须弥山能容纳芥子吗？”李渤毫不怀疑。归宗和尚说：“芥子能容纳须弥山吗？这难道不是妄言吗？”李渤说：“人们都说史君读过万卷书，是不是？”归宗和尚说：“你身体像椰子那么大，万卷书放在哪里？”李渤低下头，无言以对。}%

@{代李云。將謂無人。}@
%{李渤说：“以为没有人知道。”}%

@{舉。大梅因。龐居士問。久響大梅。未審梅子熟也未。梅云。爾向甚處下口。士云。百雜碎。梅云。還我核子來。}@
%{大梅和尚，庞居士问他：“久仰大梅和尚，不知梅子熟了没有？”大梅和尚说：“你从哪里下口？”居士说：“从百杂碎。”大梅和尚说：“还我核子来。”}%

@{代云。平出。}@
%{大梅和尚说：“平淡无奇。”}%

@{舉。茱萸因。趙州上法堂。東覷西覷。萸云。作甚麼。云。探水。云。我者裏一滴也無。探箇甚麼。州將主丈。靠壁而出。}@
%{茱萸和尚，赵州和尚登上法堂，东张西望。茱萸和尚说：“你在做什么？”赵州和尚说：“探水。”茱萸和尚说：“我这里一滴水都没有，你探什么？”赵州和尚就用手撑着墙，退了出去。}%

@{代云。莫道得便宜。}@
%{茱萸和尚说：“别以为占了便宜。”}%

@{舉。洞山因。僧問。三身中那箇說法。山云。吾常於此切。僧後問曹山。洞山道。吾常於此切。意旨如何。曹云。要頭斫將去。僧又問雪峯。峯以主丈。劈口打。云。我也曾到洞山來。僧無語。}@
%{洞山和尚，一个僧人问他：“三身中哪一个说法？”洞山和尚说：“我常在此处。”后来，这个僧人又问曹山和尚，洞山和尚说：“我常在此处。”曹山和尚说：“把你的头砍掉。”这个僧人又问雪峰和尚，雪峰和尚用手撑着墙，劈口打了他一下，说：“我也曾到洞山去过。”这个僧人无言以对。}%

@{代云。氣急殺人。}@
%{曹山和尚说：“气急了会杀人。”}%

@{舉。石室因。杏山入碓坊。相見乃云。不易行者。室云。有甚不易。開心椀子盛將來。無蓋盤子合將去。杏無語。}@
%{石室和尚，杏山和尚走进碓坊，见面就说：“不容易啊。”石室和尚说：“有什么不容易？开心的碗盛着来，没有盖子的盘子收着去。”杏山和尚无言以对。}%

@{代云。龍門之下。}@
%{石室和尚说：“龙门之下。”}%

@{舉。章敬因。小師遊方回。乃問。汝離此多少年。云。自離和尚。將及八載。敬云。辨得箇甚麼。小師就地上。畫一圓相。敬云。只者箇。更別有。小師畫破圓相。作禮而退。}@
%{章敬和尚，一个小和尚游方回来，章敬和尚问他：“你离开这里多少年了？”小和尚说：“离开师父将近八年了。”章敬和尚说：“你领悟到什么了？”小和尚就在地上画了一个圆圈。章敬和尚说：“就这一个，还有其他的吗？”小和尚就划破了圆圈，作礼退去。}%

@{代云。家無小使。不成君子。}@
%{章敬和尚说：“家里没有小使，不成君子。”}%

@{舉。雪峯因。僧辭問。甚處去。云禮拜徑山和尚去。云徑山忽問爾此間佛法如何。爾作麼生。道云。待問即道。峯便打。却回問鏡清云。者僧過在甚處。便喫棒。清云。問得徑山徹困也。峯云。徑山在浙中。因甚問得徹困。清云。不見道。遠問近對。峯休去。}@
%{雪峰和尚，一个僧人辞别他，问他：“你要去哪里？”僧人说：“去拜访径山和尚。”雪峰和尚说：“径山和尚如果问你这里佛法如何，你怎么办？”僧人说：“等他问了我就说。”雪峰和尚就打了他一下。然后雪峰和尚问镜清和尚：“这个僧人错在哪里？”镜清和尚也被打了一下。镜清和尚说：“问得径山和尚精疲力尽了。”雪峰和尚说：“径山和尚在浙江，为什么问得精疲力尽？”镜清和尚说：“不知道，远问近答。”雪峰和尚说：“你别去了。”}%

@{代云。魯般繩墨。}@
%{雪峰和尚说：“鲁班的绳墨。”}%

@{舉。雪峯辭洞山。山云。甚處去。云。歸嶺中去。云。當時從甚路出。云。從飛猿嶺出。云。今向甚路去。云。飛猿嶺去。云。有一人不從飛猿嶺去。子還識麼。云。不識。云。為甚不識。云。他無面目。云。子既不識。爭知無面目。峯無對。}@
%{雪峰和尚辞别洞山和尚，洞山和尚问他：“你要去哪里？”雪峰和尚说：“回山里去。”洞山和尚说：“当时你从什么路出去？”雪峰和尚说：“从飞猿岭出去。”洞山和尚说：“现在你从什么路去？”雪峰和尚说：“从飞猿岭去。”洞山和尚说：“有一个人不从飞猿岭去，你认识吗？”雪峰和尚说：“不认识。”洞山和尚说：“为什么不认识？”雪峰和尚说：“他没有面目。”洞山和尚说：“你既然不认识，怎么知道他没有面目？”雪峰和尚无言以对。}%

@{代云。不以眼見。}@
%{洞山和尚说：“不可以用眼睛去看。”}%

@{舉。雪峯因。鼓山來參。纔入門。峯搊住云。是甚麼。山契悟。舉手作舞云。子作道理耶。云。何道理之有。峯印之。}@
%{雪峰和尚，鼓山和尚来拜访他，刚进门，雪峰和尚就抓住他，说：“是什么？”鼓山和尚顿时开悟，举手跳舞，说：“你在讲道理吗？”雪峰和尚说：“有什么道理？”雪峰和尚就印证了他。}%

@{代鼓山云。和尚終不謾小子。}@
%{鼓山和尚说：“和尚最终没有欺骗我。”}%

@{舉。靈雲問僧。甚處去。云。雪峯去。我有信寄雪峯。云。便請雲脫履。拋向面前。僧便去。峯問。甚處來。云。靈雲。云。和尚安否。云。有信相寄。道了脫履。拋向面前。峯休去。}@
%{灵云和尚问一个僧人：“你要去哪里？”僧人说：“去雪峰那里，我有信要寄给雪峰和尚。”灵云和尚说：“那就请你脱掉鞋子，扔到我面前。”僧人就走了。雪峰和尚问他：“你从哪里来？”僧人说：“灵云那里。”雪峰和尚说：“和尚好吗？”僧人说：“有信要寄给你。”说完就脱掉鞋子，扔到雪峰和尚面前。雪峰和尚说：“你别去了。”}%

@{代云。念汝遠來。}@
%{灵云和尚说：“念你远道而来。”}%

@{舉。長慶一日陞堂。眾集定。慶拽出一僧云。大眾禮拜者僧著。又云。者僧有甚長處。却教大眾禮拜。眾無對。}@
%{长庆和尚有一天上法堂，大众都聚集起来，长庆和尚拉出一个僧人，说：“大众都要礼拜这个僧人。”又说：“这个僧人有什么长处，却要大众礼拜他？”大众无言以对。}%

@{代云。得得得。}@
%{长庆和尚说：“得得得。”}%

@{舉。靈樹因。有尼送甆鉢盂與師。遂托起問云。者箇出在甚處。尼云。出在定州。樹乃撲破。尼無對。}@
%{灵树和尚，有一个尼姑送给他一个瓦钵和一个盂，灵树和尚就拿起来问：“这个是从哪里来的？”尼姑说：“是从定州来的。”灵树和尚就打碎了瓦钵，尼姑无言以对。}%

@{代云。某甲不會和尚意。}@
%{尼姑说：“我不会理解和尚的意思。”}%

@{舉。雲居膺和尚。聞山下有一僧住庵。令侍者送袴一腰與之。僧云。我自有孃生袴。竟不受。再令侍者去問。孃未生時。著箇甚麼。僧無語。}@
%{云居膺和尚，听说山下有一个僧人住在庵里，就让侍者送了一条裤子给他。僧人说：“我自有母亲生的裤子。”就拒绝了。云居膺和尚又让侍者去问：“你母亲还没生你的时候，穿什么？”僧人无言以对。}%
@{代云。唯敢相昧。}@
%{你说的是，我只能这样说了。}%

@{舉。雲門甞舉。馬大師道。一切語言是提婆宗。以者箇為主。乃云。好語只是無人問我。時有僧便問。如何是提婆宗。門云。西天九十六種。爾是最下種。}@
%{云门禅师曾经举过一个例子，马大师说：一切语言都是提婆宗。以这个为主，才算得上是好话。当时就有人问：什么是提婆宗？云门禅师说：西天有九十六种，你说的这个是最下等的。}%

@{代。禮拜了退。}@
%{说完就礼拜，然后退下了。}%

@{舉。雲門問首座。山河大地。與汝自己。是同是別。云。同。門云。一切物命。蛾蛘蟻子。與汝自己。是同是別。云。同。門云。為甚干戈相待。}@
%{云门禅师问首座：山河大地，和你自己，是相同还是不同？首座说：相同。云门禅师说：一切有生命的，包括蛾子、蛘虫、蚂蚁，和你自己，是相同还是不同？首座说：相同。云门禅师说：那为什么还要互相残杀呢？}%

@{代首座。吐舌側立。}@
%{首座听到这里，吐舌侧立，无言以对。}%

@{舉。雲門問僧。甚處來。云。禮塔來。門云。謔我。僧云。某甲實禮塔來。門云。五戒也不持。僧無語。}@
%{云门禅师问一个僧人：你从哪里来？僧人说：我从礼拜佛塔那里来。云门禅师说：你是在戏弄我吗？僧人说：我确实是去礼拜佛塔了。云门禅师说：你连五戒都不持，还敢说去礼拜佛塔？僧人无言以对。}%

@{代云。某甲乍入道。}@
%{这僧人说：我刚入道不久。}%

@{舉。葉縣省和尚問僧。日暮投林。朝離何處。云。某甲不曾參禪。縣云。爾生身入地獄。僧無語。}@
%{叶县省和尚问一个僧人：你傍晚投宿在树林里，明天早上从哪里离开？僧人说：我还没有参禅。叶县省和尚说：你这样下去，要下地狱的。僧人无言以对。}%

@{代云。誰不承恩。}@
%{谁会不承蒙佛祖的恩典呢？}%

@{舉。葉縣問僧。近離甚處。云。襄州縣。拽童子。摑一摑。便喝出。僧無語。}@
%{叶县省和尚问一个僧人：你最近从哪里来？僧人说：我从襄州来。叶县省和尚一把抓住一个小沙弥，打了他的脸，然后大声喝斥。僧人无言以对。}%

@{代云。入水見長人。}@
%{就像掉进水里看到一个巨人一样。}%

@{舉。韓文公問僧。承聞講得肇論是否。云。是。公云。肇有四不遷是否。云。是。公將茶盞。撲破云。者箇是遷不遷。僧無語。}@
%{韩文公问一个僧人：听说你精通肇论，是吗？僧人说：是的。韩文公说：肇论里说有四种不迁移，是吗？僧人说：是的。韩文公拿起茶杯，把它摔碎了，说：这个是迁移还是不迁移？僧人无言以对。}%

@{代云。不經一事。不長一智。}@
%{不经历一件事，就学不到一个道理。}%

@{舉。洛浦一日因。侍者云。肇法師製得四論。也甚奇怪。浦云。肇公甚奇怪。要且不見祖師。者無語。}@
%{洛浦禅师有一天，他的侍者说：肇法师写了四论，真是太奇怪了。洛浦禅师说：肇公真是奇怪，他连祖师都没有见过，还写什么论？侍者无言以对。}%

@{代云。打草蛇驚。}@
%{就像打草惊蛇一样。}%

@{舉。寶應顒和尚因。僧問。從上諸聖。向甚麼處去。應云。不上天堂。即入地獄。僧云。和尚又作麼生。應云。還知寶應老漢落處麼。僧擬議。應以拂子驀口打。復喚近前云。令合是汝行。又打一拂子。}@
%{宝应顒和尚有一天，一个僧人问他：从前的诸位圣人，都去哪里了？宝应顒和尚说：他们不上天堂，也不入地狱。僧人说：和尚，你又怎么说？宝应顒和尚说：你知道我宝应老汉要去哪里吗？僧人正要说话，宝应顒和尚就用拂子打他的嘴，然后把他叫到跟前，说：你应该这样做！又打了他一拂子。}%

@{代僧擬議處。以目視以手指。}@
%{僧人正要说话的时候，用眼睛看着宝应顒和尚，还用手指指着他。}%

@{舉。真點胸。在歸宗南和尚會中。為首座。時歸宗問云。承首座常將女子出定話。為人是否。真云。無。宗云。奢而不儉。儉而不奢。為甚道無。真云。若是本分衲僧。也少他鹽醬不得。宗喚侍者。報典座。來日只煮白粥。}@
%{真点胸在归宗南和尚的寺院里做首座。有一天，归宗南和尚问他：听说你经常用女子出定的话来度人，是吗？真点胸说：没有。归宗南和尚说：奢侈而不节俭，节俭而不奢侈，为什么说没有呢？真点胸说：如果是一个本分的僧人，也是需要一些盐酱的。归宗南和尚叫来侍者，告诉典座，明天只煮白粥。}%

@{代云。負心人不放債。}@
%{负心的人不借钱。}%

@{舉。石霜諸禪師因。僧辭。霜問。舡去陸去。僧云。遇舡即舡。遇陸即陸。霜云。我道半途稍難。僧無語。}@
%{石霜诸禅师有一天，一个僧人向他告别。石霜禅师问：你是坐船走还是走路走？僧人说：遇到船就坐船，遇到陆地就走路。石霜禅师说：我以为你半路上会遇到一些困难呢。僧人无言以对。}%

@{代云。造次顛沛。不忘付囑。}@
%{即使在慌乱颠沛的时候，也不要忘记佛祖的教诲。}%

@{舉。明招到泉州坦長老處。坦云。夫參學。一人所在亦須到。半人所在亦須到。招便問。一人所在即不問。作麼生。是半人所在。坦無語。後却令小師問招。招云。欲識半人麼。也只是弄泥團漢。}@
%{明招禅师来到泉州坦长老那里，坦长老说：参禅学道，一个人在那里，你也要去；半个人在那里，你也要去。明招禅师就问：一个人在那里，我就不问了，那么半个人在那里，是什么？坦长老无言以对。后来，坦长老叫一个小沙弥去问明招禅师。明招禅师说：你想知道半个人吗？那也只是在玩泥巴的小孩子。}%

@{代坦長老云却得禮謝。}@
%{坦长老听到这里，只能向明招禅师礼拜谢过。}%

@{舉。鶴林素禪師因。僧敲門。林問。是甚麼人。云。是僧。林云。非但是僧。佛來亦不著。僧云。佛來為甚麼不著。林云。無汝止泊處。}@
%{鹤林素禅师有一天，一个僧人敲门。鹤林素禅师问：你是谁？僧人说：我是僧人。鹤林素禅师说：你不仅是僧人，即使佛来，也留不住他。僧人说：佛来为什么留不住他？鹤林素禅师说：因为你没有地方可以让他停留。}%

@{代云。不枉賣却布單。}@
%{真是没有白白卖掉布单。}%

@{舉。婺州新建和尚。一生不畜沙彌。有座主云。和尚年尊。何不討箇小師侍奉。云。若有眼暗耳聾口啞底。為我討一箇來。主無對。}@
%{婺州新建和尚，一生没有收过小沙弥。有一天，一位首座说：和尚您年纪大了，为什么不收一个小沙弥来侍奉您？新建和尚说：如果有眼睛瞎了、耳朵聋了、嘴巴哑了的小沙弥，你帮我找一个来。首座无言以对。}%

@{代云。某甲可以奉掃灑。}@
%{我可以负责打扫卫生。}%

@{舉。昔有老宿。問一座主。疏鈔解義廣略如何。主云。鈔解疏。疏解經。宿云。經解甚麼。主無對。}@
%{以前有一位老和尚，问一位首座：疏、钞、解、义，哪一个更广？哪一个更略？首座说：钞解疏，疏解经。老和尚说：经解什么？首座无言以对。}%

@{代云。看取科頭。}@
%{看看你的帽子吧。}%

@{舉。天台幽棲和尚。一日鳴鐘上堂。眾纔集。乃問。誰打鐘。僧云。維那。云。近前來。僧近前。棲與一掌。便歸方丈。}@
%{天台幽栖和尚有一天，敲响钟声，上堂说法。僧众刚聚集起来，他就问：谁敲的钟？僧人说：是维那。幽栖和尚说：你过来。僧人走近前，幽栖和尚给了他一掌，然后就回到方丈室。}%

@{代云。只恐和尚別有長處。}@
%{我怕和尚还有别的长处。}%

@{舉。昔有行者。隨法師入佛殿。行者向佛唾。云。行者少去就。何以唾佛。者云。將無佛處來與某甲唾。師無對。}@
%{以前有一个修行人，跟着法师进入佛殿。修行人对着佛像吐口水。法师说：修行人，你退后一点，为什么要吐佛像？修行人说：我从没有佛的地方来，要吐口水，谁让你吐的？法师无言以对。}%
@{代云。蛇形鼈鼻。}@
%{就是说，像蛇一样弯弯曲曲，像鳖一样鼻子又尖又长。}%

@{舉。昔有僧到翠嵓相看。適值不在。遂看主事。事云。參見和尚也未。云。未。事乃指狗子云。上人要見和尚。但禮拜者狗子。僧無語。}@
%{从前有个和尚到翠巖去看望住持，正好住持不在，于是就去找管事，管事问：“您见过和尚了吗？”和尚说：“没有。”管事就指着一条狗说：“这位上人想要见和尚，您就拜这条狗吧。”和尚没说话。}%

@{代云。鈍置翠巖。則可。}@
%{就是说，这个和尚太笨了，把他放在翠巖里，应该就可以了。}%

@{舉。昔高麗國來錢唐。刻觀音聖像。及舁。上舡。竟不能動。因請入明州開元寺供養。後有設問。無剎不現身。聖像為甚不去高麗國。}@
%{从前高丽国的人来到钱塘，刻了一尊观音圣像。当他们抬着圣像要上船的时候，怎么也抬不动。于是就请到明州开元寺供养。后来有人问：“观音菩萨无处不在，为什么圣像不去高丽国？”}%

@{代云。熟處難忘。}@
%{就是说，因为圣像是在高丽国刻的，所以它对那里有感情，舍不得离开。}%

@{舉。昔有僧到曹溪。因守衣鉢。僧提起衣云。此是大庾嶺頭提不起底。僧云。為甚在上座手裏。僧無語。}@
%{从前有个和尚来到曹溪，因为要守住衣钵，他就拿起衣钵说：“这件衣钵是大庾岭上抬不起来的。”另一个和尚说：“为什么会在上座手里？”第一个和尚没说话。}%

@{代云。仁者見之謂之仁。}@
%{就是说，仁者看到这件衣钵，就认为它很仁慈，所以就留在了上座手里。}%

@{舉。昔有僧。持鉢到長者家。偶為犬傷。長者云。龍披一縷。金翅不吞。大德全披法服。為甚却被狗咬。僧無語。}@
%{从前有个和尚拿着钵盂到一个长者家里，正好被狗咬了。长者说：“龙披着一根头发，金翅鸟都不敢吞它。您身穿法衣，为什么会被狗咬？”和尚没说话。}%

@{代云。甜瓜徹蔕甜。}@
%{就是说，甜瓜从根到蒂都是甜的，和尚身穿法衣，也应该不受狗咬。}%

@{舉。昔有施主入院。行眾僧隨年襯。主事云。聖僧前著一分。施主云。聖僧年多少。主事無對。}@
%{从前有个施主来到寺院，看到很多和尚，就按照年龄给和尚布施。管事说：“给这位圣僧多布施一些。”施主问：“这位圣僧多少岁？”管事答不上来。}%

@{代。倒指示之。}@
%{就是说，管事应该反过来问施主，圣僧是谁？}%

@{舉。襄州哲和尚因。檀越請首座開堂。主事來白。哲云。是即是。欠悟在。首座聞乃束裝他去。哲以主丈打出。}@
%{襄州的哲和尚，檀越请首座开堂说法，管事来报告。哲和尚说：“就是那样，就是那样，只是你们没有领悟而已。”首座听到后就收拾东西走了。哲和尚就用戒尺把管事打出去。}%

@{代云。不敢孤負和尚。}@
%{就是说，管事没有领悟哲和尚的意思，所以被哲和尚打出去了。}%

@{舉。昔有道士。在殿前背佛而坐。僧云。道流莫背佛。士云。大德佛身充滿於法界。教我向甚處坐。}@
%{从前有个道士在殿前背着佛像坐着。和尚说：“道友，不要背着佛像。”道士说：“大师，佛身遍布法界，您让我往哪里坐？”}%

@{代云。忘子為異教之人。}@
%{就是说，和尚忘记了道士是异教徒，所以才说他背着佛像。}%

@{舉。翠微無學禪師問丹霞。如何是諸佛師。霞咄云。幸自可憐生。須要執巾帚作麼。微退身三步。霞云。錯。微乃進前。霞亦云。錯。微翹一足旋身一轉而去。霞云。得即得。孤他諸佛師。}@
%{翠微无学禅师问丹霞禅师：“什么是诸佛之师？”丹霞禅师呵斥说：“可怜的家伙，你还需要拿着毛巾扫帚吗？”无学禅师后退三步，丹霞禅师说：“错了。”无学禅师又上前，丹霞禅师又说：“错了。”无学禅师翘起一只脚，转过身就走了。丹霞禅师说：“得到就得到了，孤单地做诸佛之师吧。”}%

@{別云。錯錯。}@
%{就是说，无学禅师两次都错了，最后才得到丹霞禅师的认可。}%

@{舉。忠國師因。西堂藏與馬祖馳書至。師問。汝師說甚麼法。藏從西過東立。師云。只者箇別更有麼。藏從東過西立。師云。者箇是馬祖底。仁者底作麼生。藏云。早箇呈似和尚了也。}@
%{忠国师，西堂藏和马祖互相写信。忠国师问：“你师父说的是什么法？”西堂藏从西往东站着，忠国师说：“就这个，还有别的吗？”西堂藏从东往西站着，忠国师说：“这个是马祖的，仁者的呢？是怎么样的？”西堂藏说：“早就呈给您了。”}%

@{別西堂後語云。來時教請取回書。}@
%{就是说，西堂藏来的时候，马祖让他把信带回去。}%

@{舉。馬祖因。有僧於前作四畫。上一畫長。下三畫短。乃云。不得道一畫長三畫短。離四句絕百非。請師答。祖乃畫一畫云。不得道一畫長三畫短。答汝了也。}@
%{马祖，有个和尚在他面前画了四条线，上面一条长，下面三条短，然后说：“不得道一画长三画短，离四句绝百非，请师父回答。”马祖就画了一条线说：“不得道一画长三画短，回答你了吧。”}%

@{別請答處云。老僧年庚在坎。}@
%{就是说，马祖的回答是，他的年龄在坎卦，也就是六十岁左右。}%

@{舉。南泉示眾。江西馬祖說即心即佛。王老師不恁麼。不是心。不是佛。不是物。恁麼道。還有過麼。時趙州出禮拜了退。續有僧問趙州。上座禮拜意作麼生。州云。汝却問取和尚。僧又問南泉。適來諗上座意作麼生。泉云。他却領得老僧意。}@
%{南泉禅师在讲法，他说：“江西马祖说即心即佛，王老师不这样，不是心，不是佛，不是物，这样说，还有过错吗？”这时赵州禅师出来拜了一下就退下。后来有个和尚问赵州禅师：“您刚才拜的意思是什么？”赵州禅师说：“你去问和尚吧。”和尚又去问南泉禅师：“刚才赵州禅师拜的意思是什么？”南泉禅师说：“他领会了我的意思。”}%

@{別。面壁而坐。}@
%{就是说，南泉禅师的意思是，赵州禅师领会了他的意思，所以就面壁而坐。}%

@{舉。歸宗剗草次。有座主來參。偶見一蛇過。宗遂鉏斷。主云。久嚮歸宗。元來是箇麁行沙門。宗云。爾麁我麁。}@
%{归宗禅师在除草的时候，有个住持来参拜。正好看到一条蛇经过，归宗禅师就用锄头把它砍断了。住持说：“我早就想拜访归宗禅师了，原来您是个粗鲁的沙门啊。”归宗禅师说：“你粗鲁，我也粗鲁。”}%

@{別云。好心不得好報。}@

@{舉。南泉問僧。夜來好風。僧云。夜來好風。泉云。吹折門前一枝松。僧云。吹折門前一枝松。次問一僧。夜來好風。僧云。是甚麼風。泉云。吹折門前一枝松。僧云。是甚麼松。泉云。一得一失。}@
%{举。南泉问僧：夜来好风。僧云：夜来好风。泉云：吹折门前一枝松。僧云：吹折门前一枝松。次问一僧：夜来好风。僧云：是甚么风？泉云：吹折门前一枝松。僧云：是甚么松？泉云：一得一失。}%

@{別南泉後語咨嗟云。明皇幸蜀。}@
%{别南泉后语咨嗟云：明皇幸蜀。}%

@{舉。文殊令善財採藥云。是藥者採將來。善財遍採。無不是藥。却來白云。無不是藥者。殊云。是藥採將來。善財乃拈一枝草。度與文殊。殊拈起示眾云。此藥亦能殺人亦能活人。}@
%{举。文殊令善财采药云：是药者采将来。善财遍采，无不是药。却来白云，无不是药者。殊云：是药采将来。善财乃拈一枝草，度与文殊。殊拈起示众云：此药亦能杀人亦能活人。}%
@{別文殊接草處。與一嗅復度與善財。}@
%{文殊菩萨离开接草的地方，闻了闻草香，就又渡过河去了，继续带着善财童子前进。}%

@{舉。孚上座初參雪峯。跨門纔見雪峯。便參主事。次日却來禮拜云。昨日觸忤和尚。峯云。知是般事便休。}@
%{有个叫孚的人，第一次去拜见雪峰禅师。他刚跨过门槛，还没见到雪峰禅师，就跑去问禅师的侍者。第二天，他又来拜见雪峰禅师，说：“昨天我冒犯了您”。雪峰禅师说：“既然知道这是冒犯，那就别再犯了”。}%

@{別云。我此間來千去萬。}@
%{意思是说：我在这世间来来往往，经历了无数次，已经看透了世间的一切。}%

@{舉。石梯一日。見侍者托鉢上堂。乃喚侍者。者應喏。梯云。甚處去。云。上堂齋去。梯云。我豈不知汝上堂齋去。云。除此外別。道箇甚麼。梯云。我只問汝本分事。云。若問本分事。某甲實是上堂齋去。梯云。不謬為吾侍者。}@
%{有一天，石梯禅师看到侍者拿着钵盂去上堂吃饭，就叫住了侍者。侍者答应了一声。石梯禅师问：“你去哪里？”侍者回答说：“去上堂吃饭”。石梯禅师说：“我当然知道你上堂吃饭，除了吃饭，你还想说什么？”侍者说：“您问我什么？”石梯禅师说：“我只问你本分的事”。侍者说：“如果问本分的事，我确实是去上堂吃饭”。石梯禅师说：“你没有说错，你是我的侍者”。}%

@{別云。若不再勘。難以結欵。}@
%{意思是说：如果不能再继续深入探讨，就很难真正理解佛法。}%

@{舉。日容遠和尚因。奯上堂參。容拊掌。三下云。猛虎當軒。誰是敵者。奯云。俊鷂冲天。阿誰捉得。容云。彼此難當。奯云。且休。未斷者公案。容將主丈。舞歸方丈。奯無語。容云。死却者漢也。}@
%{有一天，日容远禅师在法堂上讲法，奯禅师去参禅。日容远禅师拍了几下掌说：“猛虎当轩，谁是敌者？”奯禅师回答说：“俊鹞冲天，阿谁捉得？”日容远禅师说：“彼此都难以对付”。奯禅师说：“且休，未断者公案”。日容远禅师便带着侍者，跳舞着回到了方丈室。奯禅师无言以对。日容远禅师说：“死却者汉也”。}%

@{別云。贏我一著。}@
%{意思是说：奯禅师在这次辩论中，胜过了日容远禅师。}%

@{舉。奯上座到百丈。丈云。闍梨有事。相借問得麼。奯云。幸自非言。何須譗𧫡。丈云。收得安南。又憂塞北。奯擘開胸云。與麼不與麼。丈云。要且難搆。要且難搆。奯云知即得。知即得。}@
%{奯禅师来到百丈禅师那里，百丈禅师说：“您有事吗？可以问一问”。奯禅师说：“幸自非言，何须譗𧫡”。百丈禅师说：“收得安南，又忧塞北”。奯禅师打开胸膛说：“与么不与么？”百丈禅师说：“要且难搆，要且难搆”。奯禅师说：“知即得，知即得”。}%

@{別云。山長水遠。人面獸心。}@
%{意思是说：世事难料，人心难测。}%

@{舉。西院思明和尚未住時。在許州。聞汝州南院出世。元與同參。遂特去人事了云。某甲啟和尚。無可人事。自許州來。收得江西剃刀一柄。上和尚。院云。汝從許州來。為甚却有江西剃刀。明遂於院手上。掐一掐。院云。侍者收。明以衣袖。一拂便行。院云。阿喇喇。阿喇喇。}@
%{西院思明禅师还没有住持寺院的时候，住在许州。听说汝州南院禅师出世了，他们以前是同参道友，于是思明禅师特地去拜访南院禅师。思明禅师说：“我向您问候，没什么别的事。我从许州来，带了一把江西的剃刀，送给您”。南院禅师说：“你从许州来，为什么会有江西的剃刀？”思明禅师于是掐了掐南院禅师的手。南院禅师说：“侍者收起来”。思明禅师用袖子拂了一下，就走了。南院禅师说：“阿喇喇，阿喇喇”。}%

@{別云。容欵曲敘舊。}@
%{意思是说：思明禅师和南院禅师之间，是老朋友，所以思明禅师的拜访，只是为了叙旧。}%

@{舉。廓侍者因。華嚴上堂。今日賜卿無畏。若是臨際．德山．高亭．大愚．鳥窠．舡子兒孫。不用如何若何。便請單刀直入。華嚴與汝證據。廓便出作禮。起。便喝。嚴亦喝。廓又喝。嚴又喝。廓禮拜起云。大眾看。者漢一場敗闕。又喝一喝。拍手歸眾。華嚴歸方丈。時風穴充維那。上問訊。嚴云。浙客尀耐。者守廓。今日把老僧搊絕。如今集眾。打一頓趁出。穴云。趁他遲也。況和尚言太過。道單刀直入。他是臨際兒孫。本分與麼。嚴方息怒。穴下來。與廓說是事。廓云。爾著甚來由。勸者老漢。我未問前。早要棒喫得我話行。如今不打我搨却我。者話不行。穴云。此話已行也。}@
%{廓侍者在华严禅师上堂说法的时候，华严禅师说：“今天赐给你无畏。如果是临济、德山、高亭、大愚、鸟窠、舡子这些禅师的弟子，不用如何若何，就请单刀直入。我与你验证”。廓侍者便出来作礼，站起来就喝了一声。华严禅师也喝了一声。廓侍者又喝了一声，华严禅师又喝了一声。廓侍者作礼拜起来，说：“大众看，这个人真是失败了”。又喝了一声，拍拍手回到大众中。华严禅师回到方丈室。当时风穴禅师担任维那，去向华严禅师问讯。华严禅师说：“浙客尀耐，这个人是守廓。今天他把老僧打得毫无招架之力。现在他召集大众，要打一顿赶他出去”。风穴禅师说：“赶他出去已经迟了，况且和尚的话太过了，说单刀直入，他是临济的弟子，本分就是如此”。华严禅师正要发怒，风穴禅师走下来，对廓侍者说这件事。廓侍者说：“你有什么资格劝我？老汉，我还没问之前，你就想要打我，让我说话。现在你不打我，反而推开我，这算什么？”风穴禅师说：“这话已经说出来了”。}%

@{別風穴云。廓兄爾也秤椎裏。覓甚麼汁。}@
%{风穴禅师对廓侍者说：“廓兄，你难道是在秤椎里找汁吗？”}%

@{舉。藥山因。僧問。平田淺草麈鹿成群。如何射得麈中麈。山云。看箭。僧便放身倒。山云。侍者拕出。僧便走。山云。弄泥團漢。有甚麼限。}@
%{有一天，药山禅师遇到一个僧人问：“平田浅草麈鹿成群，如何射得麈中麈？”药山禅师说：“看箭”。僧人便倒在地上。药山禅师说：“侍者把他抬出去”。僧人便跑掉了。药山禅师说：“弄泥团的人，有什么限制？”}%

%{意思是说：一个人在老房子里住，会感到孤独寂寞。}%

@{舉。長髭曠和尚問僧。甚處來。僧云。九華山控石庵。髭云。庵主是甚麼人。云。馬祖下尊宿。云。名甚麼。云。不委他法號。髭云。他不委。爾不委。云。尊宿眼在甚處。髭云。若是庵主親來。今日也須喫棒。僧云。賴遇和尚放過某甲。髭云。百年後討箇師僧。也難得。}@
%{长髭旷禅师问一个僧人：“你从哪里来？”僧人说：“九华山控石庵”。长髭旷禅师说：“庵主是什么人？”僧人说：“马祖的弟子”。长髭旷禅师说：“叫什么名字？”僧人说：“我不知道他的法号”。长髭旷禅师说：“他不知道，你也不知道？”僧人说：“尊宿的眼睛在哪里？”长髭旷禅师说：“如果庵主亲自来，今天也必须挨打”。僧人说：“幸亏遇到您放过了我”。长髭旷禅师说：“百年后找一个师父，也很难得到”。}%

%{别说了，和尚们一起打他一顿就行了。}%
