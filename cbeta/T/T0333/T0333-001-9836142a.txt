%{无论是六欲天还是初禅天，都被淫欲之火所吞噬，谁又能拯救呢？}%

@{如是耽女色，婬火大熾燃，}@
%{沉迷于女色，淫欲之火就会熊熊燃烧，}%
@{焚燒於有情，違損不可救。}@
%{烧毁了众生，最终只能带来伤害，无法拯救。}%

@{人身速不淨，穢惡諸物成，}@
%{人的身体很快就会变得不洁，充满了污秽之物，}%
@{指爪與髮毛，涎唾并結聹，}@
%{手指甲、头发、唾液、鼻涕，}%
@{垢汗大小便，肪膏及腦膜，}@
%{污垢、汗水、大小便、脂肪、脑膜，}%
@{皮肉兼骨髓，膿血筋脈連，}@
%{皮肉、骨髓、脓血、筋脉，}%
@{脾腎心共肺，腸胃膽與肝，}@
%{脾脏、肾脏、心脏、肺脏、肠胃、胆囊、肝脏，}%
@{生藏對熟藏，赤痰共白痰。}@
%{生肉、熟肉、黄痰、白痰。}%

@{又復八萬戶，微細蟲𠯗食，}@
%{还有八万户微小的虫子，在身体里不断地啃食，}%
@{常住於身中，愚人那知覺。}@
%{它们一直住在你的身体里，愚蠢的人却浑然不觉。}%

@{於身起貪愛，如蠅慕膿血，}@
%{你对自己的身体产生贪爱，就像苍蝇喜欢臭血一样，}%
@{臭氣覺馨香，苦中而為樂。}@
%{闻到臭味却觉得香甜，在痛苦中寻找快乐。}%

@{如是耽慾人，執杖相敺擊，}@
%{沉迷于欲望的人，就像拿着棍子互相驱赶，}%
@{慾火競來燒，迷醉誰能悟？}@
%{欲望之火不断地燃烧，他们却迷失其中，无法醒悟。}%

@{愚癡著樂味，如狗在空房，}@
%{愚蠢的人执着于快乐的滋味，就像狗在空房子里乱叫，}%
@{亦似底囉聲，究竟成妄想。}@
%{如同空洞的叫声，最终都是虚幻的妄想。}%

@{又如於猿猴，攀緣常在樹，}@
%{就像猴子一样，总是攀缘在树上，}%
@{乃至到無常，不離於樹上。}@
%{直到死亡，也无法离开树木。}%

@{如是貪慾人，追求於色境，}@
%{沉迷于欲望的人，总是追求着色欲，}%
@{墜墮惡趣中，不離生死苦。}@
%{最终堕落到恶趣之中，无法摆脱生死的痛苦。}%

@{愚癡婬慾人，彼處命終後，}@
%{愚蠢的人沉迷于淫欲，死后就会被投入到铁锅中，}%
@{擲在鐵鑊中，如是住一劫。}@
%{在其中煎熬一劫。}%

@{浮沈如煮豆，其鑊大小量，}@
%{就像煮豆子一样，铁锅的大小，}%
@{六十四俱胝，眾生所依彼。}@
%{可以容纳六十四俱胝的众生。}%

@{一一墮落者，較量不能知，}@
%{每一个堕落的人，数量多得无法计数，}%
@{煎煮於鑊中，受苦滿百劫。}@
%{他们在铁锅里被煎煮，要承受一百劫的痛苦。}%

@{或二三四劫，隨彼業輕重，}@
%{有的要承受两劫、三劫、四劫，根据他们罪业的轻重而定，}%
@{皮肉俱爛壞，骨現似白螺。}@
%{他们的皮肉都被烧烂了，骨头露出来，就像白色的螺壳一样。}%

@{又隨自業力，手捉尖利鉤，}@
%{根据他们自己的业力，他们会被用锋利的钩子钩住，}%
@{擲在炎鐵槽，死已而還活。}@
%{投入到炽热的铁槽里，死了又活，活了又死。}%

@{又被於獄卒，手執鐵杵搗，}@
%{狱卒拿着铁杵，不停地捣他们的骨髓，}%
@{骨髓皆成粖，風吹而却活。}@
%{骨髓都被捣碎了，却依然被风吹着，再次复活。}%

@{或以鐵棒打，劈裂如斧斫，}@
%{有的被铁棒打，就像用斧头砍一样，被劈成两半，}%
@{鐵獸三四五，隨後而咬嚙。}@
%{还有铁兽，三头、四头、五头，紧随其后，撕咬着他们的身体。}%

@{又復為鐵烏，鐵狗及犲狗，}@
%{有的变成了铁乌鸦、铁狗、恶狗，}%
@{牙嘴利如劍，食罪人腦髓。}@
%{它们的牙齿和嘴巴锋利如剑，吃着罪人的脑髓。}%

@{若人造罪業，墮落於糞河，}@
%{如果有人造了罪业，就会堕落到粪便河流里，}%
@{或落刀劍上，一切皆臭穢。}@
%{或者掉在刀剑上，一切都充满了臭秽。}%

@{若人造罪業，墮在極炎熱，}@
%{如果有人造了罪业，就会堕落到极度炎热的地方，}%
@{號叫、大號叫，黑繩及燒燃。}@
%{他们大声惨叫，被黑色的绳子捆绑，被火焰焚烧。}%

@{若人造罪業，墮在於灰河，}@
%{如果有人造了罪业，就会堕落到灰烬河流里，}%
@{重重入由增，痛苦不可忍。}@
%{他们一次又一次地沉入其中，痛苦难以忍受。}%

@{若人造罪業，死墮地獄中，}@
%{如果有人造了罪业，死后就会堕落到地狱里，}%
@{飢吞熱鐵丸，渴復飲銅汁。}@
%{饿了就吞下滚烫的铁丸，渴了就喝铜汁。}%

@{若人造罪業，墮在鐵山間，}@
%{如果有人造了罪业，就会堕落到铁山之间，}%
@{眾山一時合，拶碎身如粉。}@
%{所有的山峰同时合拢，把他们压成粉末。}%

@{若人造罪業，恒受於苦惱，}@
%{如果有人造了罪业，就会永远遭受痛苦，}%
@{獲得如是果，無有能救者。}@
%{最终得到这样的果报，没有任何人能够拯救他们。}%

@{此處非安樂，先世業所招，}@
%{这里不是安乐的地方，是前世所造的罪业招来的，}%
@{父母與妻兒，何能相救濟？}@
%{父母、妻子、儿女，怎么能够救济他们呢？}%

@{下劣婬慾行，直往於無間，}@
%{低劣的淫欲行为，会直接把你送往无间地狱，}%
@{受苦不可當，三世佛皆說。}@
%{承受的痛苦难以忍受，三世佛都这样说过。}%

@{是故下劣人，與女人同處，}@
%{所以，低劣的人和女人在一起，}%
@{如擔於糞袋，愚癡到處行。}@
%{就像背着粪袋到处走一样，真是愚蠢至极。}%

@{丈夫為女人，纏縛如枷鎖，}@
%{男人因为女人，被束缚得像枷锁一样，}%
@{地獄火燃身，無能得安樂。}@
%{地狱的火焰焚烧着他们的身体，他们无法得到安乐。}%

@{剎那智慧生，如是聞佛法，}@
%{如果能够在一瞬间生起智慧，听到佛法，}%
@{離一切婬慾，出家得解脫。}@
%{就能远离一切淫欲，出家修行，获得解脱。}%
@{佛告大王：若丈夫行婬慾行，當墮地獄受斯大苦。}@
%{佛陀告诉大王：如果男人放纵淫欲，就会堕入地狱，承受极大的痛苦。}%

@{是故，大王！恒常念佛念法，觀察身心勿令起過。}@
%{所以，大王！你要时刻念佛念法，观察身心，不要让它产生过失。}%

@{日子王言：如是，世尊！於如來處，發深信心，白言：世尊！甚為希有，如來、應、正等覺，善說女人、丈夫之過。}@
%{日子王说：是的，世尊！我在佛陀面前，生起了深信，并向佛陀说：世尊！真是太罕见了，如来、应、正等觉，能如此善妙地阐述女人、男人的过失。}%

@{我當受持，歸佛、歸法、歸苾芻眾，今後棄捨婬慾刀杖等過，慜念饒益一切眾生。}@
%{我将受持佛法，归依佛陀、归依佛法、归依僧众，从此以后，我会舍弃淫欲、刀杖等过失，用心去利益一切众生。}%

@{說此語時，會中日子王，及諸苾芻、菩薩摩訶薩、天、龍、藥叉、阿素囉、㜸路茶、健闥婆、莫呼落迦、人非人等，聞佛所說，皆大歡喜，作禮而去。}@
%{当日子王说出这些话的时候，会场中的日子王，以及所有的比丘、菩萨摩诃萨、天人、龙、药叉、阿修罗、紧那罗、乾闼婆、摩呼罗迦、人非人等，听到佛陀所说，都非常欢喜，向佛陀礼拜后就离开了。}%
