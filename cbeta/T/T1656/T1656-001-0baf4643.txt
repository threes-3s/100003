@{寶行王正論一卷}@
%{宝行王正论一卷}%

@{陳天竺三藏真諦譯}@
%{陈天竺三藏真谛译}%

@{安樂解脫品第一}@
%{安乐解脱品第一}%

@{解脫一切障，圓德所莊嚴，}@
%{解脱一切障碍，圆德所庄严，}%
@{禮一切智尊，眾生真善友。}@
%{礼一切智尊，众生真善友。}%

@{正法決定善，為愛法大王，}@
%{正法决定善，为爱法大王，}%
@{我當說由法，流注法器人。}@
%{我当说由法，流注法器人。}%

@{先說樂因法，後辯解脫法，}@
%{先说乐因法，后辩解脱法，}%
@{眾生前安樂，次後得解脫。}@
%{众生前安乐，次后得解脱。}%

@{善道具名樂，解脫謂惑盡，}@
%{善道具名乐，解脱谓惑尽，}%
@{略說此二因，唯信智二根。}@
%{略说此二因，唯信智二根。}%

@{因信能持法，由智如實了，}@
%{因信能持法，由智如实了，}%
@{二中智最勝，先藉信發行。}@
%{二中智最胜，先藉信发行。}%

@{由癡貪瞋怖，而能不壞法，}@
%{由痴贪瞋怖，而能不坏法，}%
@{當知是有信，吉祥樂名器。}@
%{当知是有信，吉祥乐名器。}%

@{已能熟簡擇，身口意三業，}@
%{已能熟简择，身口意三业，}%
@{恒利益自他，說為有智人。}@
%{恒利益自他，说为有智人。}%

@{殺生盜邪婬、妄言及兩舌、}@
%{杀生盗邪婬、妄言及两舌、}%
@{惡罵不應語、貪瞋與邪見、}@
%{恶骂不应语、贪瞋与邪见、}%

@{此法名十惡，翻此即十善。}@
%{此法名十恶，翻此即十善。}%

@{離酒清淨命，無逼惱心施，}@
%{离酒清净命，无逼恼心施，}%
@{供養所應敬，略說法當爾。}@
%{供养所应敬，略说法当尔。}%

@{若但行苦行，決不生善法，}@
%{若但行苦行，决不生善法，}%
@{以離智悲故；若唯有苦行，}@
%{以离智悲故；若唯有苦行，}%

@{不能除損他，與救濟利益，}@
%{不能除损他，与救济利益，}%
@{施戒修所明，正法大夷路。}@
%{施戒修所明，正法大夷路。}%

@{若棄行邪道，自苦受牛罰，}@
%{若弃行邪道，自苦受牛罚，}%
@{是生死曠澤，無飲食樹陰，}@
%{是生死旷泽，无饮食树阴，}%

@{或狼所食噉，長遠於中行。}@
%{或狼所食噉，长远於中行。}%

@{因殺生短壽，逼惱招多病，}@
%{因杀生短寿，逼恼招多病，}%
@{由盜致乏財，侵他境多怨，}@
%{由盗致乏财，侵他境多怨，}%

@{妄語遭誹謗，兩舌親愛離，}@
%{妄语遭诽谤，两舌亲爱离，}%
@{惡口聞不愛，綺語他憎嫉，}@
%{恶口闻不爱，绮语他憎嫉，}%

@{由貪害所求，瞋恚受驚怖，}@
%{由贪害所求，瞋恚受惊怖，}%
@{邪見生僻執，飲酒心訥亂，}@
%{邪见生僻执，饮酒心讷乱，}%

@{不施故貧窮，邪命逢欺誑，}@
%{不施故贫穷，邪命逢欺誑，}%
@{不恭生卑賤，嫉妬無威德，}@
%{不恭生卑贱，嫉妬无威德，}%

@{恒恨形色醜，不問聰故癡。}@
%{恒恨形色丑，不问聪故痴。}%

@{此報在人道，先已受惡趣，}@
%{此报在人道，先已受恶趣，}%
@{殺生等罪法，如所說果報；}@
%{杀生等罪法，如所说果报；}%

@{無貪等及業，說名善習因。}@
%{无贪等及业，说名善习因。}%

@{惡修及諸苦，皆從邪法生；}@
%{恶修及诸苦，皆从邪法生；}%
@{諸善道安樂，皆因善法起，}@
%{诸善道安乐，皆因善法起，}%

@{常離一切惡，恒行一切善。}@
%{常离一切恶，恒行一切善。}%

@{由身口意業，應知此二法，}@
%{由身口意业，应知此二法，}%
@{由一法能脫，地獄等四趣，}@
%{由一法能脱，地狱等四趣，}%

@{第二法能感，人天王富樂，}@
%{第二法能感，人天王富乐，}%
@{由定梵住空，得受梵等樂，}@
%{由定梵住空，得受梵等乐，}%

@{如是略說名，樂因及樂果。}@
%{如是略说名，乐因及乐果。}%

@{復次解脫法，微細深難見，}@
%{复次解脱法，微细深难见，}%
@{無耳心凡夫，聞則生驚怖。}@
%{无耳心凡夫，闻则生惊怖。}%

@{我無當不生，現來我所無，}@
%{我无当不生，现来我所无，}%
@{凡人思此畏，智者怖永盡。}@
%{凡人思此畏，智者怖永尽。}%

@{世間我見生，他事執所繫，}@
%{世间我见生，他事执所系，}%
@{佛由至道證，依悲為他說。}@
%{佛由至道证，依悲为他说。}%

@{我有及我所，此二實皆虛，}@
%{我有及我所，此二实皆虚，}%
@{由見如實理，二執不更生。}@
%{由见如实理，二执不更生。}%

@{諸陰我執生，我執由義虛，}@
%{诸阴我执生，我执由义虚，}%
@{若種子不實，芽等云何真？}@
%{若种子不实，芽等云何真？}%

@{若見陰不實，我見則不生，}@
%{若见阴不实，我见则不生，}%
@{由我見滅盡，諸陰不更起。}@
%{由我见灭尽，诸阴不更起。}%

@{如人依淨鏡，得見自面影，}@
%{如人依净镜，得见自面影，}%
@{此影但可見，一向不真實。}@
%{此影但可见，一向不真实。}%

@{我見亦如是，依陰得顯現，}@
%{我见亦如是，依阴得显现，}%
@{如實撿非有，猶如鏡面影。}@
%{如实捡非有，犹如镜面影。}%

@{如人不執鏡，不見自面影，}@
%{如人不执镜，不见自面影，}%
@{如此若析陰，我見即不有。}@
%{如此若析阴，我见即不有。}%

@{因聞如是義，大淨命阿難，}@
%{因闻如是义，大净命阿难，}%
@{即得淨法眼，恒為他說此。}@
%{即得净法眼，恒为他说此。}%

@{陰執乃至在，我見亦恒存，}@
%{阴执乃至在，我见亦恒存，}%
@{由有我見故，業及有恒有。}@
%{由有我见故，业及有恒有。}%

@{生死輪三節，無初中後轉，}@
%{生死轮三节，无初中后转，}%
@{譬如旋火輪，生起互相由。}@
%{譬如旋火轮，生起互相由。}%

@{從自他及二，三世不有故，}@
%{从自他及二，三世不有故，}%
@{證此我見滅，次業報亦然。}@
%{证此我见灭，次业报亦然。}%

@{如此見因果，生起及滅盡，}@
%{如此见因果，生起及灭尽，}%
@{故不執實有，世間有及無。}@
%{故不执实有，世间有及无。}%

@{愚人聞此法，能盡一切苦，}@
%{愚人闻此法，能尽一切苦，}%
@{由無智生怖，於無怖畏處。}@
%{由无智生怖，於无怖畏处。}%

@{涅槃處無此，汝云何生怖？}@
%{涅槃处无此，汝云何生怖？}%
@{如所說實空，云何令汝怖？}@
%{如所说实空，云何令汝怖？}%

@{解脫無我陰，汝若受此法，}@
%{解脱无我阴，汝若受此法，}%
@{捨我及諸陰，汝何不樂？}@
%{舍我及诸阴，汝何不乐？}%

@{無尚非涅槃，何況當是有？}@
%{无尚非涅槃，何況当是有？}%
@{有無執淨盡，佛說名涅槃。}@
%{有無执净尽，佛说名涅槃。}%

@{若略說邪見，謂撥無因果，}@
%{若略说邪见，谓拨无因果，}%
@{此今非福滿，惡道因最重。}@
%{此今非福满，恶道因最重。}%

@{若略說正見，謂信有因果，}@
%{若略说正见，谓信有因果，}%
@{能令福德滿，善道因最上。}@
%{能令福德满，善道因最上。}%

@{由智有無寂，超度福非福，}@
%{由智有無寂，超度福非福，}%
@{故離善惡道，佛說名解脫。}@
%{故离善恶道，佛说名解脱。}%

@{若見生有因，智人捨無執，}@
%{若见生有因，智人舍无执，}%
@{由見滅共因，是故捨有執。}@
%{由见灭共因，是故舍有执。}%

@{先俱生二因，實義則非因，}@
%{先俱生二因，实义则非因，}%
@{假名無依故，及生非實故。}@
%{假名无依故，及生非实故。}%

@{若此有彼有，譬如長及短，}@
%{若此有彼有，譬如长及短，}%
@{由此生彼生，譬如燈與光。}@
%{由此生彼生，譬如灯与光。}%

@{先長後為短，不然非性故，}@
%{先长后为短，不然非性故，}%
@{光明不生故，燈亦非實有。}@
%{光明不生故，灯亦非实有。}%

@{如此因果生，若見不執無，}@
%{如此因果生，若见不执无，}%
@{已信世真實，由亂心所生。}@
%{已信世真实，由乱心所生。}%

@{見滅非虛故，即證得真如，}@
%{见灭非虚故，即证得真如，}%
@{是故不執有，不依二解脫。}@
%{是故不执有，不依二解脱。}%

@{色是遠所見，若近最分明，}@
%{色是远所见，若近最分明，}%
@{鹿渴若實已，云何近不見？}@
%{鹿渴若实已，云何近不见？}%

@{若遠於實智，即見世間有，}@
%{若远於实智，即见世间有，}%
@{證實則不見，無相如鹿渴。}@
%{证实则不见，无相如鹿渴。}%

@{如鹿渴似水，非水非實物，}@
%{如鹿渴似水，非水非实物，}%
@{如此陰似人，非人非實法。}@
%{如此阴似人，非人非实法。}%

@{計鹿渴為水，往彼若飲此，}@
%{计鹿渴为水，往彼若饮此，}%
@{若無執為水，如此人愚癡。}@
%{若无执为水，如此人愚痴。}%

@{世間如鹿渴，若執實有無，}@
%{世间如鹿渴，若执实有無，}%
@{此即是無明，癡故無解脫。}@
%{此即是无明，痴故无解脱。}%

@{執無墮惡趣，執有生善道，}@
%{执无堕恶趣，执有生善道，}%
@{若能知如實，不二依解脫。}@
%{若能知如实，不二依解脱。}%

@{不樂有無執，由擇真實義，}@
%{不乐有無执，由择真实义，}%
@{若墮於無執，何不說墮有？}@
%{若堕於无执，何不說堕有？}%

@{若言由破有，義至故墮無，}@
%{若言由破有，义至故堕无，}%
@{如此破無故，云何不墮有？}@
%{如此破无故，云何不堕有？}%

@{無言行及心，由依菩提故，}@
%{无言行及心，由依菩提故，}%
@{若說彼墮無，何因不墮有？}@
%{若说彼堕无，何因不堕有？}%

@{僧佉鞞世師，尼揵說人陰，}@
%{僧佉鞞世师，尼揵說人阴，}%
@{約世汝問彼，若說過有無，}@
%{约世汝问彼，若说过有無，}%

@{是不可言法，以過有無故。}@
%{是不可言法，以过有無故。}%

@{汝應知甚深，佛正教甘露，}@
%{汝应知甚深，佛正教甘露，}%
@{如曉無去來，亦無一念住，}@
%{如晓无去来，亦无一念住，}%

@{若體過三世，何世為實有？}@
%{若体过三世，何世为实有？}%

@{二世無去來，現在實不住，}@
%{二世无去来，现在实不住，}%
@{世生及住滅，此言云何實？}@
%{世生及住灭，此言云何实？}%

@{若恒有變異，何法不念滅？}@
%{若恒有变异，何法不念灭？}%
@{若無念念滅，云何有變異？}@
%{若无念念灭，云何有变异？}%

@{若言念念滅，分具分滅故，}@
%{若言念念灭，分具分灭故，}%
@{不等證見故，此二無道理。}@
%{不等证见故，此二无道理。}%

@{若念滅皆盡，云何有故物？}@
%{若念灭皆尽，云何有故物？}%
@{若堅無念滅，故物云何成？}@
%{若坚无念灭，故物云何成？}%

@{如剎那後際，前中際亦有，}@
%{如刹那后际，前中际亦有，}%
@{由剎那三分，故世念無住。}@
%{由刹那三分，故世念无住。}%

@{是一念三際，應擇際如念，}@
%{是一念三际，应择际如念，}%
@{前中後三際，不由自他成。}@
%{前中后三际，不由自他成。}%

@{非一念分故，若無分何有？}@
%{非一念分故，若无分何有？}%
@{離一多云何？離有何法無？}@
%{离一多云何？离有何法无？}%

@{由滅及對治，若言有成無，}@
%{由灭及对治，若言有成无，}%
@{此無及對治，何法有無故？}@
%{此无及对治，何法有無故？}%

@{是故世涅槃，由義不成有，}@
%{是故世涅槃，由义不成有，}%
@{世間有後際，他問佛默然。}@
%{世间有后际，他问佛默然。}%

@{是尊一切智，故智人識佛，}@
%{是尊一切智，故智人识佛，}%
@{由此甚深法，不說非器處。}@
%{由此甚深法，不说非器处。}%

@{如此解脫法，甚深無繫攝，}@
%{如此解脱法，甚深无繫摄，}%
@{諸佛一切智，故說無依底，}@
%{诸佛一切智，故说无依底，}%

@{於無依著法，過有無二邊。}@
%{於无依著法，过有無二边。}%

@{世人受依著，由癡驚怖失，}@
%{世人受依著，由痴惊怖失，}%
@{彼自失壞他，怖畏無依處。}@
%{彼自失坏他，怖畏无依处。}%

@{王願汝不動，莫由彼自壞，}@
%{王愿汝不动，莫由彼自坏，}%
@{為汝成不壞，我當說真理。}@
%{为汝成不坏，我当说真理。}%

@{由依無倒合，離有無二執，}@
%{由依无倒合，离有無二执，}%
@{此過福非福，甚深義明了。}@
%{此过福非福，甚深义明了。}%

@{非身見怖空，二人境當說，}@
%{非身见怖空，二人境当说，}%
@{四大及空識，一聚俱非人。}@
%{四大及空识，一聚俱非人。}%

@{若合離非人，云何執人有？}@
%{若合离非人，云何执人有？}%
@{如六界非人，聚故虛非實，}@
%{如六界非人，聚故虚非实，}%

@{一一界同然，由聚故非實。}@
%{一一界同然，由聚故非实。}%

@{陰非我我所，離陰我不顯，}@
%{阴非我我所，离阴我不显，}%
@{不如薪火雜，何依陰成我？}@
%{不如薪火杂，何依阴成我？}%

@{地界非三大，地中亦無三，}@
%{地界非三大，地中亦无三，}%
@{三中亦無地，相離互不成。}@
%{三中亦无地，相离互不成。}%

@{地水火風大，各自性不成，}@
%{地水火风大，各自性不成，}%
@{一離三不成，三離一亦爾。}@
%{一离三不成，三离一亦尔。}%

@{一三及三一，相離若不成，}@
%{一三及三一，相离若不成，}%
@{各各自不成，彼相離云何？}@
%{各各自不成，彼相离云何？}%

@{若各離自成，離薪何無火，}@
%{若各离自成，离薪何无火，}%
@{動礙及相聚，水風地亦然。}@
%{动碍及相聚，水风地亦然。}%

@{若火不自成，三云何各立？}@
%{若火不自成，三云何各立？}%
@{三大緣生義，相違云何成？}@
%{三大缘生义，相违云何成？}%

@{若彼各自成，云何更互有？}@
%{若彼各自成，云何更互有？}%
@{若各自不成，云何互成有？}@
%{若各自不成，云何互成有？}%

@{若言不相離，諸大各自成，}@
%{若言不相离，诸大各自成，}%
@{不雜則不共，若雜非獨成。}@
%{不杂则不共，若杂非独成。}%

@{諸大非各成，云何各性相？}@
%{诸大非各成，云何各性相？}%
@{各成無偏多，故相假名說。}@
%{各成无偏多，故相假名说。}%

@{色聲香味觸，簡擇義如大，}@
%{色声香味触，简择义如大，}%
@{眼色識無明，業生擇亦爾。}@
%{眼色识无明，业生择亦尔。}%

@{作者業及事，數合因果世，}@
%{作者业及事，数合因果世，}%
@{短長及名想，非想擇亦然。}@
%{短长及名想，非想择亦然。}%

@{地水風火等，長短及小大，}@
%{地水风火等，长短及小大，}%
@{善惡言識智，智中滅無餘。}@
%{善恶言识智，智中灭无余。}%

@{如識處無形，無邊遍一切，}@
%{如识处无形，无边遍一切，}%
@{此中地等大，一切皆滅盡。}@
%{此中地等大，一切皆灭尽。}%

@{我離一切惡，攝持一切善，}@

@{眾生諸善行，隨喜及順行。}@
%{所有众生的善行，随喜和顺行。}%

@{願為轉法輪，窮生死後際。}@

@{從此行我德，已作及未作，}@

@{因此願眾生，皆發菩提心。}@
%{因此希望所有众生，都发菩提心。}%

%{穷尽后际无尽，希望众生如此。}%

@{恒於一切時，明足得圓滿。}@
%{恒常在一切时，明足圆满。}%

@{勝形貌威德，好色他愛見，}@
%{胜妙的形貌和威德，好色的人爱慕而见，}%

%{没有病痛，力量充足，寿命长久，希望他们如此。}%

%{圆满福慧的行持，相好光明照耀，}%

%{希望他们不可思议，行十地无障碍。}%

@{與此德相應，餘德所莊嚴，}@

@{圓滿一切善，及眾生所樂，}@

%{能够消除其他众生的苦难，希望我恒常如此。}%

%{如果他们有恐惧，在一切时间和一切地方，}%

@{由唯憶我名，得脫一切苦。}@
%{只要忆念我的名字，就能解脱一切苦难。}%

@{敬信我及瞋，若見及憶持，}@
%{敬信我，以及瞋恨我，如果看到我，或者忆持我，}%

@{乃至聞我名，願彼定菩提。}@
%{乃至听到我的名字，希望他们都能成就菩提。}%

@{願我得五通，恒隨一切生，}@
%{希望我得到五通，恒常跟随一切众生，}%

%{希望遍断他们的恶行，如理使他们修持善法。}%

%{就像地水火风，野草和树木，}%

%{就像他们想要享用，希望我自己忍受。}%
