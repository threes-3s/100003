@{既繫念已，再入觀門，先想頂內，}@

@{其華八葉。於蓮華內，想金剛地，}@
%{这莲花有八瓣。在莲花里面，想金刚地，}%

@{金剛地上，想日月輪。於日輪上，}@
%{金刚地上，想日月轮。在日轮上，}%

@{想十二字，月輪之上，想十六字，}@
%{想十二字，在月轮之上，想十六字，}%

@{梵字之母。出生諸法，不可得故，}@
%{梵字的母亲。出生一切法，不可得故，}%

@{於月輪上，想安唵字。又觀唵字，}@
%{在月轮上，想安置唵字。又观想唵字，}%

@{想為菩薩，旋復而去，十方世界，}@
%{想它成为菩萨，旋转而去，十方世界，}%

@{隨意供養，諸佛聖賢。既供養已，}@
%{随意供养，诸佛圣贤。既然供养已，}%

@{良久而來，作輪子形。復想一返，}@
%{良久而来，变成轮子形状。再想一返，}%

@{為毘盧佛，餘四方佛，次第想安，}@
%{成为毘卢佛，其余四方佛，依次想安置，}%

@{悉皆同等。先想種智，及想梵號，}@
%{都平等无二。先想种智，以及想梵号，}%

@{兼想五印，四波羅密，准四佛作，}@
%{兼想五印，四波罗蜜，按照四佛的做法，}%

@{而居四隅，坐三重座，無其月輪。}@
%{而住在四个角落，坐在三重座上，没有月轮。}%

%{往下到明王，都跟上面一样。以上九位，}%

%{住在第一院。里面的八位菩萨，想种智字，}%

@{金剛阿闍黎，與弟子灌頂，}@
%{金刚阿阇黎，给弟子灌顶，}%
@{如上先想壇，亦應誦密言。}@
%{就像上面说的，先想到坛，也应该念密语。}%
@{所為四明王，鉢囉抳也怛迦，}@
%{这四个明王，就是鉢囉抳也怛迦，}%
@{真言用稻華，打弟子五障。}@
%{用真言和稻草花，打掉弟子的五障。}%
@{鉢納麼明王，而燒白芥子，}@
%{鉢納麼明王，然后烧白芥子，}%
@{熏授法之者，一切諸煩惱。}@
%{用香熏授法的人，消除一切烦恼。}%
@{吒枳囉惹王，加持於淨水，}@
%{吒枳囉惹王，加持净水，}%
@{以洒弟子身，而除諸罪垢。}@
%{洒在弟子身上，去除一切罪垢。}%
@{播多羅真言，用加持生飯，}@
%{播多羅真言，加持生米饭，}%
@{供養於諸天，龍神八部等。}@
%{供养诸天、龙神、八部众等等。}%

@{說是祕密壇加持法已，告彌勒菩薩曰：願授灌頂，先結大曼拏羅壇而授灌頂。}@
%{说完这个秘密坛加持法之后，对弥勒菩萨说：我想要授予灌顶，先结一个大曼荼罗坛，然后授予灌顶。}%
@{爾時世尊即說五如來灌頂真言曰(真言如經)。}@
%{当时世尊就说五如来灌顶真言(真言如经上所说) 。}%
