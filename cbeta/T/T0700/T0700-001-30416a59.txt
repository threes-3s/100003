@{右繞佛塔功德經}@
%{右绕佛塔功德经}%

@{大周于闐國三藏沙門實叉難陀等奉制譯}@
%{大周于阗国三藏沙门实叉难陀等奉制译}%

@{如是我聞：}@
%{如是我闻：}%

@{一時，佛在舍衛國祇樹給孤獨園，與大比丘僧及餘無量眾俱，前後圍遶。}@
%{一时，佛在舍卫国祇树给孤独园，与大比丘僧及余无量众俱，前后围绕。}%

@{爾時，長老舍利弗即從坐起，偏袒右肩，右膝著地，合掌向佛，以偈請曰：}@
%{尔时，长老舍利弗即从坐起，偏袒右肩，右膝着地，合掌向佛，以偈请曰：}%

@{大威德世尊，願為我等說，}@
%{大威德世尊，愿为我等说，}%

@{右遶於佛塔，所得之果報。}@
%{右绕于佛塔，所得之果报。}%

@{爾時世尊以偈答曰：}@
%{尔时世尊以偈答曰：}%

@{右繞於佛塔，所得諸功德，}@
%{右绕于佛塔，所得诸功德，}%

@{我今說少分，汝等咸善聽。}@
%{我今说少分，汝等咸善听。}%

@{一切諸天龍、夜叉、鬼神等，}@
%{一切诸天龙、夜叉、鬼神等，}%

@{皆親近供養，斯由右繞塔。}@
%{皆亲近供养，斯由右绕塔。}%

@{在在所生處，遠離於八難，}@
%{在在所生处，远离于八难，}%

@{常生無難處，斯由右繞塔。}@
%{常生无难处，斯由右绕塔。}%

@{於一切生處，念慧常無失，}@
%{于一切生处，念慧常无失，}%

@{具足妙色相，斯由右繞塔。}@
%{具足妙色相，斯由右绕塔。}%

@{往來天人中，福命悉長遠，}@
%{往来天人中，福命悉长远，}%

@{常獲大名稱，斯由右繞塔。}@
%{常获大名称，斯由右绕塔。}%

@{在於閻浮提，常生最尊勝，}@
%{在于阎浮提，常生最尊胜，}%

@{清淨種姓中，斯由右繞塔。}@
%{清净种姓中，斯由右绕塔。}%

@{儀貌常端正，富貴多財寶，}@
%{仪貌常端正，富贵多财宝，}%

@{恒食大封邑，斯由右繞塔。}@
%{恒食大封邑，斯由右绕塔。}%

@{財寶常盈積，而無慳悋心，}@
%{财宝常盈积，而无悭悋心，}%

@{勇猛廣惠施，斯由右繞塔。}@
%{勇猛广惠施，斯由右绕塔。}%

@{色相淨微妙，見者皆欣仰，}@
%{色相净微妙，见者皆欣仰，}%

@{所住常安樂，斯由右繞塔。}@
%{所住常安乐，斯由右绕塔。}%

@{或為忉利王，妻子悉具足，}@
%{或为忉利王，妻子悉具足，}%

@{威勢力自在，斯由右繞塔。}@
%{威势力自在，斯由右绕塔。}%

@{或作婆羅門，持戒善通達，}@
%{或作婆罗门，持戒善通达，}%

@{呪術、圍陀典，斯由右繞塔。}@
%{呪術、围陀典，斯由右绕塔。}%

@{或作大長者，豪貴多財產，}@
%{或作大长者，豪贵多财产，}%

@{倉廩常豐足，斯由右繞塔。}@
%{仓廪常丰足，斯由右绕塔。}%

@{或作正法王，自在王閻浮，}@
%{或作正法王，自在王阎浮，}%

@{率土咸歸化，斯由右繞塔。}@
%{率土咸归化，斯由右绕塔。}%

@{或為具七寶，大勢轉輪王，}@
%{或为具七宝，大势转轮王，}%

@{十善御群生，斯由右繞塔。}@
%{十善御群生，斯由右绕塔。}%

@{從此生天上，常有大威德，}@
%{从此生天上，常有大威德，}%

@{淨信於佛法，斯由右繞塔。}@
%{净信于佛法，斯由右绕塔。}%

@{淨信速成已，於法無迷惑，}@
%{净信速成已，于法无迷惑，}%

@{見諸行皆空，斯由右繞塔。}@
%{见诸行皆空，斯由右绕塔。}%

@{從天上捨命，下生於人中，}@
%{从天上舍命，下生于人中，}%

@{入胎不迷亂，斯由右繞塔。}@
%{入胎不迷乱，斯由右绕塔。}%

@{在於母胎中，垢穢所不染，}@
%{在于母胎中，垢秽所不染，}%

@{如淨摩尼珠，斯由右繞塔。}@
%{如净摩尼珠，斯由右绕塔。}%

@{在胎及生時，令母常安樂，}@
%{在胎及生时，令母常安乐，}%

@{飲乳亦復然，斯由右繞塔。}@
%{饮乳亦复然，斯由右绕塔。}%

@{父母及親戚，一切共鞠養，}@
%{父母及亲戚，一切共鞠养，}%

@{乳母常不離，斯由右繞塔。}@
%{乳母常不离，斯由右绕塔。}%

@{眷屬皆愛念，超過於父母，}@
%{眷属皆爱念，超过于父母，}%

@{資財自增長，斯由右繞塔。}@
%{资财自增长，斯由右绕塔。}%

@{夜叉諸惡鬼，不能暫驚怖，}@
%{夜叉诸恶鬼，不能暂惊怖，}%

@{所須自然得，斯由右繞塔。}@
%{所须自然得，斯由右绕塔。}%

@{經於百千劫，其身轉清淨，}@
%{经于百千劫，其身转清净，}%

@{妙色相成滿，斯由右繞塔。}@
%{妙色相成满，斯由右绕塔。}%

@{淨眼修且廣，猶如青蓮花，}@
%{净眼修且广，犹如青莲花，}%

@{兼得淨天眼，斯由右繞塔。}@
%{兼得净天眼，斯由右绕塔。}%

@{妙色常圓滿，諸相自莊嚴，}@
%{妙色常圆满，诸相自庄严，}%

@{成就大勢力，斯由右繞塔。}@
%{成就大势力，斯由右绕塔。}%

@{或生帝釋宮，大威勢自在，}@
%{或生帝释宫，大威势自在，}%

@{忉利天中尊，斯由右繞塔。}@
%{忉利天中尊，斯由右绕塔。}%

@{或生須夜摩、兜率陀天宮、}@
%{或生须夜摩、兜率陀天宫、}%

@{化樂及他化，斯由右繞塔。}@
%{化乐及他化，斯由右绕塔。}%

@{或復生梵天，梵世最自在，}@
%{或复生梵天，梵世最自在，}%

@{諸天常供養，斯由右繞塔。}@
%{诸天常供养，斯由右绕塔。}%

@{億那由他劫，常為諸智人，}@
%{亿那由他劫，常为诸智人，}%

@{恭敬而供養，斯由右繞塔。}@
%{恭敬而供养，斯由右绕塔。}%

@{其身及衣服，億劫常無垢，}@
%{其身及衣服，亿劫常无垢，}%

@{具足白淨法，斯由右繞塔。}@
%{具足白净法，斯由右绕塔。}%

@{具大精進力，勤修種種行，}@
%{具大精进力，勤修种种行，}%

@{未嘗有疲懈，斯由右繞塔。}@
%{未尝有疲懈，斯由右绕塔。}%

@{勇猛常精進，堅固不可壞，}@
%{勇猛常精进，坚固不可坏，}%

@{所作速成就，斯由右繞塔。}@
%{所作速成就，斯由右绕塔。}%

@{深遠微妙音，聞者皆歡喜，}@
%{深远微妙音，闻者皆欢喜，}%

@{安樂常無病，斯由右繞塔。}@
%{安乐常无病，斯由右绕塔。}%

@{如我所演說，厭捨三有苦，}@
%{如我所演说，厌舍三有苦，}%

@{成就出世智，斯由右繞塔。}@
%{成就出世智，斯由右绕塔。}%

@{常在四念處，及以四正勤，}@
%{常在四念处，及以四正勤，}%

@{四如意神足，斯由右繞塔。}@
%{四如意神足，斯由右绕塔。}%

@{了達四真諦、根、力、七覺分，}@
%{了达四真谛、根、力、七觉分，}%

@{正道及聖果，斯由右繞塔。}@
%{正道及圣果，斯由右绕塔。}%

@{滅一切煩惱，具足大威德，}@
%{灭一切烦恼，具足大威德，}%

@{無漏六神通，斯由右繞塔。}@
%{无漏六神通，斯由右绕塔。}%

@{永離貪、恚、癡，及一切障礙，}@
%{永离贪、恚、癡，及一切障碍，}%

@{證獨覺菩提，斯由右繞塔。}@
%{证独觉菩提，斯由右绕塔。}%

@{得妙紫金色，相好莊嚴身，}@
%{得妙紫金色，相好庄严身，}%

@{現作天人師，斯由右繞塔。}@
%{现作天人师，斯由右绕塔。}%

@{皆由以身業，及語業讚歎，}@
%{皆由以身业，及语业赞叹，}%

@{右繞於佛塔，獲此大利益。}@
%{右绕于佛塔，获此大利益。}%

@{右繞諸佛塔，所得諸功德，}@
%{右绕诸佛塔，所得诸功德，}%

@{我今隨所聞，略說詎能盡？}@
%{我今随所闻，略说詎能尽？}%

@{爾時世尊說此偈已，舍利弗等一切眾會，皆大歡喜，信受奉行。}@
%{尔时世尊说此偈已，舍利弗等一切众会，皆大欢喜，信受奉行。}%

@{右繞佛塔功德經}@
%{右绕佛塔功德经}%
