@{此佛菩提大智門，如應觀想得成就。}@

@{如上是為諸部大菩提智成就教理。}@
%{以上是所有部大菩提智成就的教义。}%

@{復次，金剛手菩薩摩訶薩宣說諸部隨愛樂成就教理：}@
%{其次，金刚手菩萨摩诃萨宣说了所有部随爱乐成就的教义：}%

@{謂先通說一切如來隨愛樂成就教理：}@
%{首先，我们来谈谈所有如来随爱乐成就的教义：}%

@{一切義利遍所作，即自求佛菩提者，}@
%{所有一切的利益，都遍布于所作的一切，这就是你自己想要求佛菩提的人。}%

@{念佛三昧法若成，即得一切佛愛樂。}@
%{如果你成就了念佛三昧法，你就能得到所有佛的喜爱。}%
@{此佛所說金剛愛樂法門。}@
%{这个佛说的金刚爱乐法门。}%

@{如來部隨愛樂成就教理：}@
%{如来部随爱乐成就教理：}%

@{隨諸境界所成已，金剛薩埵作成就，}@
%{随着各种境界成就之后，金刚萨埵成就了，}%

@{真實警悟法無虛，速得所愛皆成就。}@
%{真实警悟法无虚，快速得到所爱都成就了。}%

@{此佛所說金剛薩埵法門。}@
%{这个佛说的金刚萨埵法门。}%

@{金剛部隨愛樂成就教理：}@
%{金刚部随爱乐成就教理：}%

@{如佛教勅作大利，普為調伏諸惡者，}@
%{如佛所教令作大利，普遍为调伏诸恶者，}%

@{清淨忿恚諸有情，此即金剛隨愛樂。}@
%{清净忿恚诸有情，这就是金刚随爱乐。}%

@{此佛所說降三世法門。}@
%{这个佛说的降三世法门。}%

@{蓮華部隨愛樂成就教理：}@
%{莲花部随爱乐成就教理：}%

@{貪法乃為觀照慈，悲心即是正法語，}@
%{贪法乃为观照慈悲，悲心即是正法语，}%

@{一切無畏施有情，即得諸佛隨愛樂。}@
%{一切无畏施予有情，即得诸佛随爱乐。}%

@{此佛所說金剛眼法門。}@
%{这个佛说的金刚眼法门。}%

@{寶部隨愛樂成就教理：}@
%{宝部随爱乐成就教理：}%

@{如應廣施灌頂法，及施諸妙珍財藏，}@
%{如应广施灌顶法，及施予诸妙珍财藏，}%

@{彼等得佛利相應，此即諸佛隨愛樂。}@
%{彼等得佛利相应，这就是诸佛随爱乐。}%

@{此佛所說虛空藏法門。}@
%{这个佛说的虚空藏法门。}%

@{如上是為諸部隨愛樂成就教理。}@
%{如上是为诸部随爱乐成就教理。}%

@{復次，金剛手菩薩摩訶薩宣說諸部敬愛成就教理：}@
%{复次，金刚手菩萨摩诃萨宣说诸部敬爱成就教理：}%

@{謂先通說一切如來敬愛成就教理：}@
%{谓先通说一切如来敬爱成就教理：}%

@{應知貪性勿厭離，即清淨樂施亦然，}@
%{应知贪性勿厌离，即清净乐施亦然，}%

@{利諸有情法相應，此即諸佛敬愛理。}@
%{利诸有情法相应，这就是诸佛敬爱理。}%

@{此佛所說覺智法門。}@
%{这个佛说的觉智法门。}%

@{如來部敬愛成就教理：}@
%{如来部敬爱成就教理：}%

@{己身妙樂欲順行，稱蘇囉多薩怛鑁，}@
%{己身妙乐欲顺行，称苏囉多萨怛鑁，}%

@{金剛薩埵成就門，真實警悟作敬愛。}@
%{金刚萨埵成就门，真实警悟作敬爱。}%

@{此佛所說普賢法門。}@
%{这个佛说的普贤法门。}%

@{金剛部敬愛成就教理：}@
%{金刚部敬爱成就教理：}%

@{佛為清淨諸有情，普令有情得無畏，}@
%{佛为清净诸有情，普遍令有情得无畏，}%

@{由於佛教作護持，殺諸魔惡令敬愛。}@
%{由于佛教作护持，杀诸魔恶令敬爱。}%

@{此佛所說降三世法門。}@
%{这个佛说的降三世法门。}%

@{蓮華部敬愛成就教理：}@
%{莲花部敬爱成就教理：}%

@{觀察貪性本清淨，譬彼蓮華正開敷，}@
%{观察贪性本清净，譬彼莲花正开敷，}%

@{此中若染若愛時，如應調伏作敬愛。}@
%{此中若染若爱时，如应调伏作敬爱。}%

@{此佛所說觀自在法門。}@
%{这个佛说的观自在法门。}%

@{寶部敬愛成就教理：}@
%{宝部敬爱成就教理：}%

@{為求諸佛灌頂故，日日以大金剛寶，}@
%{为求诸佛灌顶故，日日以大金刚宝，}%

@{安置頂間作法成，即得諸佛敬愛理。}@
%{安置顶间作法成，即得诸佛敬爱理。}%

@{此佛所說虛空藏法門。}@
%{这个佛说的虚空藏法门。}%

@{如上是為諸部敬愛成就教理。}@
%{如上是为诸部敬爱成就教理。}%

@{復次，金剛手菩薩摩訶薩宣說諸部破惡成就教理：}@
%{复次，金刚手菩萨摩诃萨宣说诸部破恶成就教理：}%

@{謂先通說一切如來破惡成就教理：}@
%{谓先通说一切如来破恶成就教理：}%

@{不見有情有所利，亦非佛性可清淨，}@
%{不见有情有所利，亦非佛性可清净，}%

@{身口意業所作中，當得破惡法成就。}@
%{身口意业所作中，当得破恶法成就。}%

@{此佛所說如實法門。}@
%{这个佛说的如实法门。}%

@{如來部破惡成就教理：}@
%{如来部破恶成就教理：}%

@{以無所得一切成，由無瞋恚忿怒生，}@
%{以无所得一切成，由无瞋恚忿怒生，}%

@{此大金剛摧擊時，速得破惡法成就。}@
%{此大金刚摧击时，速得破恶法成就。}%

@{此佛所說金剛手法門。}@
%{这个佛说的金刚手法门。}%

@{金剛部破惡成就教理：}@
%{金刚部破恶成就教理：}%

@{無惡可制無忿怒，亦無有情可調伏，}@
%{无恶可制无忿怒，亦无有情可调伏，}%

@{以己苦法皆施他，即降三世破諸惡。}@
%{以己苦法皆施予他，即降三世破诸恶。}%

@{此佛所說降三世法門。}@
%{这个佛说的降三世法门。}%

@{蓮華部破惡成就教理：}@
%{莲花部破恶成就教理：}%

@{法中無悲亦無慈，無惡有情可清淨，}@
%{法中无悲亦无慈，无恶有情可清净，}%

@{於己違背語言中，自得破惡法成就。}@
%{于己违背语言中，自得破恶法成就。}%

@{此佛所說觀自在法門。}@
%{这个佛说的观自在法门。}%

@{寶部破惡成就教理：}@
%{宝部破恶成就教理：}%

@{自本無財可能捨，他財不取亦不求，}@
%{自本无财可能舍，他财不取亦不求，}%

@{一切有情貧亦然，自得破惡法成就。}@
%{一切有情贫亦然，自得破恶法成就。}%

@{此佛所說圓滿一切願法門。}@
%{这个佛说的圆满一切愿法门。}%

@{無所有法得相應，我於諸佛不和合，}@
%{无所有法得相应，我于诸佛不和合，}%

@{彼為救拔利益故，速疾施我勝成就。}@
%{彼为救拔利益故，速疾施予我胜成就。}%

@{此佛所說聖普賢法門。}@
%{这个佛说的圣普贤法门。}%

@{如上是為諸部破惡成就教理。}@
%{如上是为诸部破恶成就教理。}%

@{復次，金剛手菩薩摩訶薩宣說諸部加護印成就教理：}@
%{复次，金刚手菩萨摩诃萨宣说诸部加护印成就教理：}%

@{謂先通說一切如來加護印成就教理：}@
%{谓先通说一切如来加护印成就教理：}%

@{常不捨離諸有情，我常供養於諸佛，}@
%{常不舍离诸有情，我常供养于诸佛，}%

@{歸命諸佛常所尊，最上希有此加護。}@
%{归命诸佛常所尊，最上希有此加护。}%

@{如來部加護印成就教理：}@
%{如来部加护印成就教理：}%
@{金剛薩埵勝名號，一稱中間能攝受，}@
%{金刚萨埵这个名字，只要念诵一遍，就能把你摄受住，}%
@{此加護為廣大門，剎那常得施成就。}@
%{这就是通往广阔解脱之门，瞬间就能成就。}%

@{此佛所說執金剛法門。}@

%{金刚部加护印成就的教理是：}%

@{信奉降三世聖尊，此與自他為加護。}@

@{此佛所說降三世法門。}@
%{这就是佛陀所说的降三世法门。}%

@{蓮華部加護印成就教理：}@
%{莲花部加护印成就的教理是：}%

@{所有貪清淨大慈，常施一切無所畏，}@
%{所有贪欲都清净了，就拥有了大慈悲，能常施一切众生，无所畏惧，}%

%{这就是佛陀所说的法金刚法门。}%

@{寶部加護印成就教理：}@
%{宝部加护印成就的教理是：}%

@{晝日所作無虛過，隨其力修方便行，}@
%{白天所做的事，没有虚假，根据自己的能力，修习方便法门，}%
@{常施佛及諸有情，最上希有此加護。}@
%{常施佛陀和一切众生，就能得到最殊胜的加持。}%

@{此佛所說金剛護法門。}@
%{这就是佛陀所说的金刚护法门。}%

@{爾時，金剛手菩薩摩訶薩前白佛言：世尊！我向所說一切如來諸部最上成就教理，願佛攝受，令諸有情於一切教中證法自在，速疾成就阿耨多羅三藐三菩提果。}@
%{这时，金刚手菩萨摩诃萨站在佛陀面前，对佛陀说：“世尊！我之前所说的一切如来各个部的最殊胜成就教理，希望佛陀能摄受，让一切众生，在一切教法中证得法自在，快速成就无上正等正觉果位。”}%

@{爾時，世尊讚金剛手菩薩摩訶薩言：善哉，善哉！金剛手！汝善宣說如上法門，我今為汝加持攝受。}@
%{这时，世尊赞叹金刚手菩萨摩诃萨说：“善哉，善哉！金刚手！你很好地宣说了以上法门，我现在为你加持摄受。”}%
