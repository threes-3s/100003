@{大智度論釋發趣品第二十之餘(卷五十)}@
%{大智度論釋發趣品第二十之餘(卷五十)}%

@{龍樹菩薩造}@
%{龍樹菩薩造}%

@{云何菩薩不著我？}@
%{菩萨怎么才能不执着于“我”？}%
@{畢竟無我故。}@
%{因为本来就没有“我”啊。}%

@{云何菩薩不著眾生、不著壽命、不著眾數，乃至知者、見者？}@
%{菩萨怎么才能不执着于众生、寿命、数量，甚至包括知者、见者呢？}%
@{是諸法畢竟不可得故。}@
%{因为这些都是无常的，最终无法获得。}%

@{云何菩薩不著斷見？}@
%{菩萨怎么才能不执着于断见？}%
@{無有法斷，諸法畢竟不生故。}@
%{因为没有法能断灭，一切法本来就不生。}%

@{云何菩薩不著常見？}@
%{菩萨怎么才能不执着于常见？}%
@{若法不生，是不作常。}@
%{如果法本来就不生，那它就不可能恒常存在。}%

@{云何菩薩不應取相？}@
%{菩萨怎么才能不执着于相？}%
@{無諸煩惱故。}@
%{因为没有烦恼，就不会执着于相。}%

@{云何菩薩不應作因見？}@
%{菩萨怎么才能不执着于因见？}%
@{諸見不可得故。}@
%{因为一切见解都是不可得的。}%

@{云何菩薩不著名色？}@
%{菩萨怎么才能不执着于名色？}%
@{名色處相無故。}@
%{因为名色本来就没有实质的相。}%

@{云何菩薩不著五眾、不著十八界、不著十二入？}@
%{菩萨怎么才能不执着于五众、十八界、十二入呢？}%
@{是諸法性無故。}@
%{因为这些都是无常的，没有本质。}%

@{云何菩薩不著三界？}@
%{菩萨怎么才能不执着于三界？}%
@{三界性無故。}@
%{因为三界也是无常的，没有本质。}%

@{云何菩薩不應作著心？云何菩薩不應作願？云何菩薩不應作依止？}@
%{菩萨怎么才能不执着于心？怎么才能不执着于愿望？怎么才能不执着于依靠呢？}%
@{是諸法性無故。}@
%{因为这些都是无常的，没有本质。}%

@{云何菩薩不著依佛見？}@
%{菩萨怎么才能不执着于依靠佛的见解呢？}%
@{作依見，不見佛故。}@
%{因为执着于依靠，就无法真正见到佛。}%

@{云何菩薩不著依法見？}@
%{菩萨怎么才能不执着于依靠法的见解呢？}%
@{法不可見故。}@
%{因为法本身是不可见的。}%

@{云何菩薩不著依僧見？}@
%{菩萨怎么才能不执着于依靠僧的见解呢？}%
@{僧相無為，不可依故。}@
%{因为僧相是无为的，不可依靠。}%

@{云何菩薩不著依戒見？}@
%{菩萨怎么才能不执着于依靠戒律的见解呢？}%
@{罪、無罪不著故。}@
%{因为罪、无罪都是不执着的。}%

@{是為菩薩住七地中，二十法所不應著。}@
%{这就是菩萨在七地中，不应执着的二十种法。}%

@{云何菩薩應具足空？}@
%{菩萨怎么才能具足空呢？}%
@{具足諸法自相空故。}@
%{因为一切法都是空性的，所以菩萨具足空性。}%

@{云何菩薩無相證？}@
%{菩萨怎么才能无相证呢？}%
@{不念諸相故。}@
%{因为不执着于相，所以菩萨就能无相证。}%

@{云何菩薩知無作？}@
%{菩萨怎么才能知道无作呢？}%
@{於三界中不作故。}@
%{因为在三界中不造作，所以菩萨就能知道无作。}%

@{云何菩薩三分清淨？}@
%{菩萨怎么才能三分清净呢？}%
@{十善道具足故。}@
%{因为十善法具足，所以菩萨就能三分清净。}%

@{云何菩薩一切眾生中慈悲智具足？}@
%{菩萨怎么才能在一切众生中具足慈悲智慧呢？}%
@{得大悲故。}@
%{因为获得了大悲心，所以菩萨就能在一切众生中具足慈悲智慧。}%

@{云何菩薩不念一切眾生？}@
%{菩萨怎么才能不执着于一切众生呢？}%
@{淨佛世界具足故。}@
%{因为清净佛国世界具足，所以菩萨就能不执着于一切众生。}%

@{云何菩薩一切法等觀？}@
%{菩萨怎么才能平等看待一切法呢？}%
@{於諸法不損益故。}@
%{因为对一切法不增不减，所以菩萨就能平等看待一切法。}%

@{云何菩薩知諸法實相？}@
%{菩萨怎么才能知道诸法的实相呢？}%
@{諸法實相無知故。}@
%{因为诸法的实相是不可知的，所以菩萨就能知道诸法的实相。}%

@{云何菩薩無生忍？}@
%{菩萨怎么才能获得无生忍呢？}%
@{為諸法不生不滅、不作故。}@
%{因为一切法本来就不生不灭，不造作，所以菩萨就能获得无生忍。}%

@{云何菩薩無生智？}@
%{菩萨怎么才能获得无生智呢？}%
@{知名色不生故。}@
%{因为知道名色本来就不生，所以菩萨就能获得无生智。}%

@{云何菩薩說諸法一相？}@
%{菩萨怎么才能说诸法一相呢？}%
@{心不行二相故。}@
%{因为心不执着于二相，所以菩萨就能说诸法一相。}%

@{云何菩薩破分別相？}@
%{菩萨怎么才能破除分别相呢？}%
@{一切法不分別故。}@
%{因为一切法本来就不分别，所以菩萨就能破除分别相。}%

@{云何菩薩轉憶想？}@
%{菩萨怎么才能转忆想呢？}%
@{小、大、無量想轉故。}@
%{因为小、大、无量想都能转，所以菩萨就能转忆想。}%

@{云何菩薩轉見？}@
%{菩萨怎么才能转见呢？}%
@{於聲聞、辟支佛地見轉故。}@
%{因为能转声闻、辟支佛地的见解，所以菩萨就能转见。}%

@{云何菩薩轉煩惱？}@
%{菩萨怎么才能转烦恼呢？}%
@{斷諸煩惱故。}@
%{因为断除了烦恼，所以菩萨就能转烦恼。}%

@{云何菩薩等定慧地？}@
%{菩萨怎么才能等定慧地呢？}%
@{所謂得一切種智故。}@
%{因为获得了一切种智，所以菩萨就能等定慧地。}%

@{云何菩薩調意？}@
%{菩萨怎么才能调意呢？}%
@{於三界不動故。}@
%{因为在三界中不动心，所以菩萨就能调意。}%

@{云何菩薩心寂滅？}@
%{菩萨怎么才能心寂灭呢？}%
@{制六根故。}@
%{因为制伏了六根，所以菩萨就能心寂灭。}%

@{云何菩薩無礙智？}@
%{菩萨怎么才能获得无碍智呢？}%
@{得佛眼故。}@
%{因为获得了佛眼，所以菩萨就能获得无碍智。}%

@{云何菩薩不染愛？}@
%{菩萨怎么才能不染爱呢？}%
@{捨六塵故。}@
%{因为舍弃了六尘，所以菩萨就能不染爱。}%

@{是為菩薩住七地中，具足二十法。}@
%{这就是菩萨在七地中，具足的二十种法。}%

@{云何菩薩順入眾生心？}@
%{菩萨怎么才能顺入众生心呢？}%
@{菩薩以一心知一切眾生心及心數法。}@
%{因为菩萨能以一心了解一切众生的心和心数法，所以就能顺入众生心。}%

@{云何菩薩遊戲諸神通？}@
%{菩萨怎么才能游戏诸神通呢？}%
@{以是神通從一佛國至一佛國，亦不作佛國想。}@
%{因为菩萨能以神通从一个佛国到另一个佛国，而且不执着于佛国，所以就能游戏诸神通。}%

@{云何菩薩觀諸佛國？}@
%{菩萨怎么才能观诸佛国呢？}%
@{自住其國，見無量諸佛國，亦無佛國想。}@
%{因为菩萨能住在一个佛国，同时看到无数个佛国，而且不执着于佛国，所以就能观诸佛国。}%

@{云何菩薩如所見佛國，自莊嚴其國？}@
%{菩萨怎么才能像所见佛国一样，自己庄严其国呢？}%
@{住轉輪聖王地，遍至三千大千世界以自莊嚴。}@
%{因为菩萨能住转轮圣王地，遍至三千大千世界，以自身庄严，所以就能像所见佛国一样，自己庄严其国。}%

@{云何菩薩如實觀佛身？}@
%{菩萨怎么才能如实观佛身呢？}%
@{如實觀法身故。}@
%{因为菩萨能如实观法身，所以就能如实观佛身。}%

@{是為菩薩住八地中具足四法。}@
%{这就是菩萨在八地中，具足的四种法。}%

@{云何菩薩知上下諸根？}@
%{菩萨怎么才能知道上下诸根呢？}%
@{菩薩住佛十力，知一切眾生上下諸根。}@
%{因为菩萨住佛十力，所以就能知道一切众生上下诸根。}%
@{云何菩薩淨佛世界？}@
%{菩萨如何净化佛的世界呢？}%

@{淨眾生故。}@
%{因为净化众生，佛的世界自然就净化了。}%

@{云何菩薩如幻三昧？}@
%{菩萨如何进入如幻三昧呢？}%

@{住是三昧，能成辦一切事，亦不生心相。}@
%{处于这种三昧中，可以成就一切事情，但不会产生任何心相。}%

@{云何菩薩常入三昧？}@
%{菩萨如何常常进入三昧呢？}%

@{菩薩得報生三昧故。}@
%{因为菩萨获得了报生三昧。}%

@{云何菩薩隨眾生所應善根受身？}@
%{菩萨如何根据众生的善根而受身呢？}%

@{菩薩知眾生所應生善根而為受身，成就眾生故。}@
%{菩萨知道众生应该生起的善根，而为了成就众生，就相应地受身。}%

@{是為菩薩住八地中，具足五法。}@
%{这就是菩萨住在八地中，具足五种法。}%

@{云何菩薩受無邊世界所度之分？}@
%{菩萨如何承担度化无边世界的众生呢？}%

@{十方無量世界中眾生，如諸佛法所應度者而度脫之。}@
%{十方无量世界中的众生，只要是佛法应该度化的，菩萨就会度化他们。}%

@{云何菩薩得如所願？}@
%{菩萨如何能如愿以偿呢？}%

@{六波羅蜜具足故。}@
%{因为六波罗蜜具足了。}%

@{云何菩薩知諸天、龍、夜叉、犍闥婆語？}@
%{菩萨如何知道诸天、龙、夜叉、犍闼婆的语言呢？}%

@{辭辯力故。}@
%{因为有辩才无碍的力量。}%

@{云何菩薩胎生成就？}@
%{菩萨如何成就胎生呢？}%

@{菩薩世世常化生故。}@
%{因为菩萨世世代代都是化生。}%

@{云何菩薩家成就？}@
%{菩萨如何成就家呢？}%

@{常在大家生故。}@
%{因为菩萨常常在大家庭中出生。}%

@{云何菩薩所生成就？}@
%{菩萨如何成就所生呢？}%

@{若剎利家生，若婆羅門家生故。}@
%{因为菩萨可以出生在刹利家庭，也可以出生在婆罗门家庭。}%

@{云何菩薩性成就？}@
%{菩萨如何成就姓呢？}%

@{如過去菩薩所生姓，從此中生故。}@
%{因为菩萨从过去的菩萨所生的姓氏中出生。}%

@{云何菩薩眷屬成就？}@
%{菩萨如何成就眷属呢？}%

@{純諸菩薩摩訶薩為眷屬故。}@
%{因为菩萨的眷属都是清净的菩萨摩诃萨。}%

@{云何菩薩出生成就？}@
%{菩萨如何成就出生呢？}%

@{生時光明遍照無量無邊世界，亦不取相故。}@
%{因为菩萨出生时光明遍照无量无边世界，而且不执着于任何相。}%

@{云何菩薩出家成就？}@
%{菩萨如何成就出家呢？}%

@{出家時，無量百千億諸天侍從出家，是一切眾生必至三乘。}@
%{因为菩萨出家时，无量百千亿诸天侍从一起出家，这是所有众生必定要到达三乘的。}%

@{云何菩薩莊嚴佛樹成就？}@
%{菩萨如何成就庄严佛树呢？}%

@{是菩提樹以黃金為根，七寶為莖、節、枝、葉，莖、節、枝、葉光明，遍照十方阿僧祇三千大千世界。}@
%{因为菩提树以黄金为根，七宝为茎、节、枝、叶，茎、节、枝、叶光明遍照十方无量无边三千大千世界。}%

@{云何菩薩一切諸善功德成滿具足？}@
%{菩萨如何成就一切善功德圆满具足呢？}%

@{菩薩得眾生清淨，佛國亦淨。}@
%{因为菩萨令众生清净，佛国也自然清净了。}%

@{是為菩薩住九地中具足十二法。}@
%{这就是菩萨住在九地中，具足十二种法。}%

@{云何菩薩住十地中當知如佛？}@
%{菩萨住在十地中，如何知道像佛一样呢？}%

@{若菩薩摩訶薩具足六波羅蜜、四念處乃至十八不共法，一切種智具足滿，斷一切煩惱及習；是為菩薩摩訶薩住十地中當知如佛。須菩提！菩薩摩訶薩住是十地中，以方便力故，行六波羅蜜，行四念處，乃至十八不共法，過乾慧地、性地、八忍地、見地、薄地、離欲地、已作地、辟支佛地、菩薩地，過是九地，住於佛地。是為菩薩十地。如是，須菩提！是名菩薩摩訶薩大乘發趣。}@
%{如果菩萨摩诃萨具足六波罗蜜、四念处乃至十八不共法，一切种智圆满具足，断除一切烦恼和习气；这就是菩萨摩诃萨住在十地中，知道像佛一样。须菩提！菩萨摩诃萨住在十地中，因为方便力的缘故，修行六波罗蜜，修行四念处，乃至十八不共法，超越乾慧地、性地、八忍地、见地、薄地、离欲地、已作地、辟支佛地、菩萨地，超越这九地，住在佛地。这就是菩萨十地。这样，须菩提！这就是菩萨摩诃萨大乘发趣。}%

@{論}@
%{论}%

@{者言：}@
%{这里说：}%

@{我等二十法不可得故不著。不可得因緣，如先種種說。我見乃至知者、見者，佛見、僧見，是入眾生空故，是見不應著。餘斷、常乃至戒見，是法空故，不應著。}@
%{我们二十种法不可得，所以不执着。不可得的因缘，就像前面所说的种种。我见乃至知者、见者，佛见、僧见，这是进入众生空，所以这种见不应该执着。其他的断、常乃至戒见，这是法空，所以不应该执着。}%

@{問曰：}@
%{问曰：}%

@{餘者可知，因見云何？}@
%{其他的可以知道，因见是什么呢？}%

@{答曰：}@
%{答曰：}%

@{一切有為法展轉為因果，是法中著心取相生見，是名因見，所謂非因說因，或因果一異等。}@
%{一切有为法互相转为因果，执着于这种法中，取相而生起见，这就是因见，所谓的非因说因，或者因果一异等等。}%

@{具足空者，若菩薩能盡行十八空，是名具足空。}@
%{具足空者，如果菩萨能够圆满修行十八空，这就是具足空。}%

@{復次，能行二種空：眾生空、法空，是名具足空。}@
%{其次，能够修行两种空：众生空、法空，这就是具足空。}%

@{復次，若菩薩能行畢竟空，於中不著，是名具足空。}@
%{其次，如果菩萨能够修行毕竟空，在其中不执着，这就是具足空。}%

@{問曰：}@
%{问曰：}%

@{若爾者，佛此中何以但說自相空？}@
%{如果是这样的话，佛在这里为什么只说自相空呢？}%

@{答曰：}@
%{答曰：}%

@{此三種空皆是自相空。以住六地菩薩福德故利根，利根故分別諸法取相；以是故，七地中以相空為具足空。}@
%{这三种空都是自相空。因为住在六地菩萨福德的缘故，所以利根，利根的缘故，所以分别诸法取相；因此，在七地中，以相空为具足空。}%

@{佛或時說有為空、無為空名具足空，或時說不可得空名具足空。}@
%{佛有时候说有为空、无为空叫做具足空，有时候说不可得空叫做具足空。}%

@{無相證者，無相即是涅槃，可證，不可修；不可修故，不得言知；無量無邊不可分別故，不得言具足。}@
%{无相证者，无相就是涅槃，可以证得，不可修行；因为不可修行，所以不能说知道；因为无量无边不可分别，所以不能说具足。}%

@{知無作者，三事雖通是知，二事更義立其名，無作但有知名。}@
%{知道无作者，虽然三事都通称为知，但是两事意义不同而立其名，无作只有知名。}%
@{三分清淨者，所謂十善道：身三、口四、意三，是名三分。上以說三解脫門故，此中不復說。三分清淨者，或有人身業清淨，口業不清淨；口業清淨，身業不清淨；或身口業清淨，意業不清淨；或有世間三業清淨，而未能離著；是菩薩三業清淨及離著故，是名三分清淨。}@
%{三分清净，指的是十善道：身三、口四、意三，这就是三分。之前已经讲过三解脱门，所以这里不再重复。三分清净，有的人可能身业清净，但口业不清净；口业清净，但身业不清净；或者身口业清净，但意业不清净；或者世间三业都清净，却不能脱离执着；而菩萨的三业清净，而且能脱离执着，所以叫做三分清净。}%

@{一切眾生中具足慈悲智者，悲有三種：眾生緣、法緣、無緣。此中說無緣大悲名具足。所謂法性空，乃至實相亦空，是名無緣大悲。菩薩深入實相，然後悲念眾生；譬如人有一子，得好寶物，則深心愛念欲以與之。}@
%{在所有众生中，具足慈悲智慧的人，悲心有三种：众生缘、法缘、无缘。这里讲的无缘大悲，就是具足的悲心。所谓法性空，乃至实相也空，这就是无缘大悲。菩萨深入实相，然后悲悯众生；就像一个人有一个儿子，得到宝贵的东西，就会深切地爱念，想要把它送给儿子。}%

@{不念一切眾生者，所謂淨世界具足故。}@
%{不念一切众生，是因为净世界具足的缘故。}%

@{問曰：}@
%{问曰：}%

@{若不念眾生者，云何能淨佛世界？}@
%{如果不念众生，怎么能够净佛世界呢？}%

@{答曰：}@
%{答曰：}%

@{菩薩令眾生住十善道，為莊嚴佛國；雖莊嚴，未得無礙莊嚴。今菩薩教化眾生，不取眾生相，諸善根福德清淨；諸善根福德清淨故，是無礙莊嚴。}@
%{菩萨让众生住持十善道，来庄严佛国；虽然庄严，但还没有得到无碍的庄严。现在菩萨教化众生，不执着众生相，一切善根福德都清净；因为诸善根福德清净，所以是无碍的庄严。}%

@{一切法等觀者，如法等忍中說。此中佛自說：於諸法不增損。}@
%{一切法等观，就像法等忍中所说。这里佛自己说：对于诸法不增不减。}%

@{知諸法實相者，如先種種因緣廣說。}@
%{知道诸法实相，就像以前种种因缘中广说的一样。}%

@{無生法忍者，於無生滅諸法實相中，信受、通達、無礙、不退，是名無生忍。}@
%{无生法忍，在无生灭诸法实相中，信受、通达、无碍、不退，这就是无生忍。}%

@{無生智者，初名忍，後名智；麁者忍，細者智。佛自說：知名色不生故。}@
%{无生智，最初叫做忍，后来叫做智；粗糙的叫做忍，细微的叫做智。佛自己说：因为知道名色不生，所以叫做无生智。}%

@{說諸法一相者，菩薩知內外十二入皆是魔網，虛誑不實，於此中生六種識亦是魔網虛誑。何者是實？唯不二法——無眼、無色，乃至無意、無法等，是名實。令眾生離十二入故，常以種種因緣說是不二法。}@
%{说诸法一相，菩萨知道内外十二入都是魔网，虚妄不实，在此生起的六种识也是魔网虚妄。什么才是真实？只有不二法——无眼、无色，乃至无意、无法等等，这就是真实。为了让众生脱离十二入，所以常常以种种因缘讲述不二法。}%

@{破分別相者，菩薩住是不二法中，破所緣男女、長短、大小等分別諸法。}@
%{破除分别相，菩萨住在不二法中，破除所缘男女、长短、大小等分别诸法。}%

@{轉憶想者，破內心憶想分別諸法等。}@
%{转忆想，破除内心忆想分别诸法等等。}%

@{轉見者，是菩薩先轉我見、邊見等邪見，然後入道；今轉法見、涅槃見，以諸法無定相。轉涅槃者，轉聲聞、辟支佛見，直趣佛道。}@
%{转见，是菩萨先转我见、边见等邪见，然后入道；现在转法见、涅槃见，以诸法无定相。转涅槃，转声闻、辟支佛见，直接趣向佛道。}%

@{轉煩惱者，菩薩以福德持戒力故，折伏麁煩惱，安隱行道，唯有愛、見、慢等微細者在，今亦離細煩惱。復次，菩薩用實智慧，觀是煩惱即是實相；譬如神通人，能轉不淨為淨。}@
%{转烦恼，菩萨以福德持戒力，折伏粗烦恼，安稳行道，只有爱、见、慢等微细的烦恼存在，现在也离细烦恼。再次，菩萨用真实的智慧，观照这些烦恼就是实相；就像神通人，能转不净为净。}%

@{等定慧地者，菩薩於初三地慧多定少，未能攝心故；後三地定多慧少，以是故，不得入菩薩位。今眾生空、法空，定慧等故，能安隱行菩薩道；從阿鞞跋致地，漸漸得一切種智慧地。}@
%{等定慧地，菩萨在最初三地，智慧多定少，不能摄心；后三地定多慧少，因此，不能进入菩萨位。现在众生空、法空，定慧等，所以能安稳行菩萨道；从阿鞞跋致地，逐渐得到一切种智慧地。}%

@{調意者，是菩薩先憶念老、病、死、三惡道，慈愍眾生故，調伏心意；今知諸法實相故不著三界，不著三界故調伏。}@
%{调意，是菩萨先忆念老、病、死、三恶道，慈悯众生，所以调伏心意；现在知道诸法实相，所以不执着三界，不执着三界，所以调伏心意。}%

@{心寂滅者，菩薩為涅槃故，先於五欲中折伏五情，意情難折伏故；今住七地，意情寂滅。}@
%{心寂灭，菩萨为了涅槃，先在五欲中折伏五情，意情很难折伏；现在住七地，意情寂灭。}%

@{無礙智者，菩薩得般若波羅蜜，於一切實、不實法中無礙；得是道慧，將一切眾生令入實法，得無礙解脫、得佛眼，於一切法中無礙。}@
%{无碍智，菩萨得到般若波罗蜜，在一切实、不实法中无碍；得到这种道慧，将一切众生令入实法，得到无碍解脱、得到佛眼，在一切法中无碍。}%

@{問曰：}@
%{问曰：}%

@{是七地中，何以說得佛眼？}@
%{这七地中，为什么说得到佛眼？}%

@{答曰：}@
%{答曰：}%

@{是中應學佛眼，於諸法無礙，似如佛眼。}@
%{这里应该学习佛眼，在诸法中无碍，就像佛眼一样。}%

@{不染愛者，是菩薩雖於七地得智慧力，猶有先世因緣，有此肉身，入禪定不著，出禪定時有著氣；隨此肉眼所見，見好人親愛，或愛是七地智慧實法。是故佛說：於六塵中行捨心，不取好惡相。(七地竟)}@
%{不染爱，是菩萨虽然在七地得到智慧力，但还有前世因缘，有这个肉身，入禅定不执着，出禅定时还执着气；随着这个肉眼所见，见到好人亲爱，或者爱这七地智慧实法。所以佛说：在六尘中行舍心，不取好恶相。(七地竟) }%

@{順入眾生心者，菩薩住是八地中，順觀一切眾生心之所趣，動發思惟，深念順觀，以智慧分別知：是眾生永無得度因緣；是眾生過無量阿僧祇劫然後可度，是眾生或一劫、二劫乃至十劫可度，是眾生或一世、二世乃至今世可度，是眾生或即時可度者，是熟、是未熟；是人可以聲聞乘度，是人可以辟支佛乘度。譬如良醫診病，知差久近、可治不可治者。}@
%{顺入众生心，菩萨住八地中，顺观一切众生心之所趣，发动思惟，深念顺观，以智慧分别知道：这个众生永远没有度化的因缘；这个众生要过无量阿僧祇劫才能度化，这个众生可能一劫、两劫乃至十劫才能度化，这个众生可能一世、两世乃至今世才能度化，这个众生可能立即就能度化，这个成熟了，这个还没有成熟；这个人可以用声闻乘度化，这个人可以用辟支佛乘度化。就像良医诊断病人，知道病情的轻重缓急、可治不可治。}%
@{遊戲諸神通者，先得諸神通，今得自在遊戲，能至無量無邊世界。}@
%{玩弄神通的人，首先要获得神通，现在已经能够自由自在地玩弄神通，可以抵达无数无边无际的世界。}%

@{菩薩住七地中時，欲取涅槃；爾時，有種種因緣及十方諸佛擁護，還生心欲度眾生。}@
%{菩萨住在七地的时候，想要获得涅槃；这时，因为种种因缘以及十方诸佛的护持，他再次生起想要度化众生的心。}%

@{好莊嚴神通，隨意自在，乃至無量無邊世界中無所罣礙，見諸佛國，亦不取佛國相。}@
%{他乐于装饰神通，随心所欲，即使在无数无边无际的世界中也毫无挂碍，看到诸佛国土，也不执着于佛国土的相貌。}%

@{觀諸佛國者，有菩薩以神通力飛到十方，觀諸清淨世界，取相欲自莊嚴其國。}@
%{观察诸佛国土的人，有的菩萨凭借神通力飞到十方，观察诸佛清净世界，取相想要装饰自己的国土。}%

@{有菩薩佛將至十方，示清淨世界，取淨國相，自作願行；如世自在王佛，將法積比丘至十方，示清淨世界。}@
%{有的菩萨佛将要前往十方，示现清净世界，取净国相，自己发愿修行；就像世自在王佛，带着法积比丘前往十方，示现清净世界。}%

@{或有菩薩自住本國，用天眼見十方清淨世界，初取淨相，後得不著心故還捨。}@
%{有的菩萨自己住在本国，用天眼观察十方清净世界，最初取净相，后来因为不执着心而放弃了。}%

@{如所見佛國，自莊嚴其國者，如先說。是八地名轉輪地。}@
%{就像所看到的佛国，自己装饰自己的国土，就像前面所说，这就是八地中的转轮地。}%

@{如轉輪王寶輪至處，無礙無障，無諸怨敵；菩薩住是地中，能雨法寶，滿眾生願，無能障礙，亦能取所見淨國相而自莊嚴其國。}@
%{就像转轮王的宝轮走到哪里，就没有任何阻碍和障碍，没有敌人；菩萨住在转轮地中，能够降下法宝，满足众生的愿望，没有任何障碍，也能够取所看到的净国相而装饰自己的国土。}%

@{如實觀佛身者，觀諸佛身如幻如化，非五眾、十二入、十八界所攝，若長若短，若干種色，隨眾生先世業因緣所見。}@
%{真实观察佛身的人，观察诸佛身就像幻化一样，不受五众、十二入、十八界所摄，无论是长还是短，有多少种颜色，都是根据众生前世业因缘所看到的。}%

@{此中佛自說：見法身者，是為見佛。法身者，不可得法空；不可得法空者，諸因緣邊生法，無有自性。}@
%{佛在这个时候自己说：看到法身，就是看到佛。法身，不能用空来获得；不能用空来获得法身，是因为诸法缘起，没有自性。}%

@{知上下諸根者，如十力中說。菩薩先知一切眾生心所行，誰鈍、誰利，誰布施多、誰智慧多，因其多者而度脫之。}@
%{知道上下诸根的人，就像十力中所说。菩萨先知道一切众生心的所行，谁愚钝、谁敏捷，谁布施多、谁智慧多，就根据他们多的方面来度化他们。}%

@{淨佛世界者，有二種淨：一者、菩薩自淨其身；二者、淨眾生心，令行清淨道。}@
%{净化佛国土，有两种净化：一是菩萨自己净化自己的身；二是净化众生的心，让他们走上清净的道。}%

@{以彼我因緣清淨故，隨所願得清淨世界。}@
%{因为彼此的因缘清净，所以就能随心所欲地获得清净世界。}%

@{入如幻三昧者，如幻人一處住，所作幻事，遍滿世界，所謂四種兵眾，宮殿城郭，飲食歌舞，殺活、憂苦等；菩薩亦如是，住是三昧中，能於十方世界變化，遍滿其中：先行布施等充滿眾生，次說法教化破壞三惡道，然後安立眾生於三乘；一切所可利益之事，無不成就。}@
%{进入如幻三昧的人，就像幻化的人在一个地方住着，所做的幻事遍布世界，比如四种军队、宫殿城郭、饮食歌舞、杀生、忧苦等等；菩萨也是这样，住在如幻三昧中，能够在十方世界变化，遍布其中：首先以布施等等充满众生，然后说法教化，破除三恶道，最后安置众生在三乘中；一切可以利益的事，没有不成就的。}%

@{是菩薩心不動，亦不取心相。}@
%{这个菩萨的心不动，也不执着于心相。}%

@{常入三昧者，菩薩得如幻等三昧，所役心能有所作；今轉身得報生三昧，如人見色，不用心力。}@
%{经常进入三昧的人，菩萨获得了如幻等等三昧，所役使的心能够有所作为；现在转而获得报生三昧，就像人看到颜色，不需要用心力。}%

@{住是三昧中，度眾生安隱，勝於如幻三昧，自然成事，無所役用；如人求財，有役力得者，有自然得者。}@
%{住在报生三昧中，度化众生安稳，胜过如幻三昧，自然成就，不需要役使；就像人求财，有的用力量获得，有的自然获得。}%

@{隨眾生所應善根受身者，菩薩得二種三昧，二種神通，行得、報得；知以何身、以何語、以何因緣、以何事、以何道、以何方便而為受身，乃至受畜生身而化度之。(八地竟) }@
%{随顺众生应得的善根受身的人，菩萨获得两种三昧，两种神通，行得、报得；知道用什么身、用什么语言、用什么因缘、用什么事、用什么道、用什么方便而受身，乃至受畜生身而化度他们。(八地结束) }%

@{受無邊世界所度之分者，無量阿僧祇十方世界六道中眾生，是菩薩教化所應度者而度之。}@
%{承受无边世界所度化的一部分的人，是无数阿僧祇十方世界六道中的众生，是菩萨教化应该度化的人而度化他们。}%

@{是世界有三種：有淨、不淨、有雜。}@
%{这些世界有三种：有清净的、不净的、有杂的。}%

@{是三種世界中眾生所可應度有利益者，皆攝取之。}@
%{这三种世界中，众生可以度化，有利益的，都摄取他们。}%

@{譬如然燈，為有目之人，不為盲者；菩薩亦如是，或先有因緣者，或始作因緣者。}@
%{比如点燃灯，是为有眼睛的人，而不是为盲人；菩萨也是这样，有的先有因缘，有的开始建立因缘。}%

@{復次，三千大千世界名一世界，一時起、一時滅；如是等十方如恒河沙等世界，是一佛世界如是一佛世界數，如恒河沙等世界，是一佛世界海如是佛世界海，數如十方恒河沙世界，是佛世界種。}@
%{再说，三千大千世界称为一个世界，一时兴起，一时灭亡；像这样十方像恒河沙一样多的世界，是一个佛世界，像这样数量的佛世界，像恒河沙一样多的世界，是一个佛世界海，像这样数量的佛世界海，像十方恒河沙世界一样多，是佛世界的种子。}%

@{如是世界種，十方無量，是名一佛世界。}@
%{像这样的世界种子，十方无数，这就是一个佛世界。}%

@{於一切世界中取如是分，是名一佛所度之分。}@
%{在一切世界中取这样的一部分，这就是一个佛所度化的一部分。}%

@{得如所願者，是菩薩福德、智慧具足故，無願不得。}@
%{能够获得如所愿，是因为菩萨福德、智慧具足，没有愿望不能获得。}%

@{聽者聞無量無邊世界所度之分，疑不可得；以是故，次說所願如意。}@
%{听众听到无数无边世界所度化的一部分，怀疑不可获得；因此，接下来说所愿如意。}%

@{此中佛自說六波羅蜜具足：五度則福德具足，般若則智慧具足。}@
%{佛在这个时候自己说六波罗蜜具足：五度就是福德具足，般若就是智慧具足。}%

@{知諸天、龍、夜叉、犍闥婆語者，我上說福德、智慧具足，所願如意；知他人種種語，即是所願事。}@
%{知道诸天、龙、夜叉、乾闼婆语言的人，我上面说福德、智慧具足，所愿如意；知道他人种种语言，就是所愿的事。}%

@{復次，菩薩得宿命智清淨故，知處處生一切語。}@
%{再说，菩萨获得宿命智清净，所以知道各个地方生的一切语言。}%

@{復次，得願智故，知立名者心，強作種種名字語言。}@
%{再说，获得愿智，所以知道立名的人的心，强作种种名字语言。}%

@{復次，菩薩得解眾生語言三昧故，通一切語無礙。}@
%{再说，菩萨获得解众生语言三昧，所以通晓一切语言，没有障碍。}%

@{復次，自得四無礙智，又復學佛四無礙智；以是故，知眾生語言音聲。}@
%{再说，自己获得四无碍智，又学习佛的四无碍智；因此，知道众生的语言音声。}%
@{處胎成就者，有人言：菩薩乘白象，與無量兜率諸天圍遶、恭敬、供養、侍從，入母胎。}@
%{有人说，菩萨骑着白象，带着无数兜率天的诸天，围着它恭敬、供养、侍奉，进入母亲的胎里。}%

@{有人言：菩薩母得如幻三昧力故，令腹廣大無量，一切三千大千世界菩薩及天龍鬼神皆得入出；胎中有宮殿、臺觀，先莊嚴床座，懸繒幡蓋，散花，燒香，皆是菩薩福德業因緣所感，然後菩薩來下處之。亦以三昧力故，下入母胎，於兜率天上如故。}@
%{也有人说，菩萨的母亲因为获得了如幻三昧的力量，肚子变得无比宽广，三千大千世界所有的菩萨，以及天龙鬼神都能进进出出；胎里还有宫殿、楼台，先装饰好床铺，悬挂着锦缎、幡盖，撒花、烧香，都是菩萨的福德业因缘感召来的，然后菩萨才来到这里。也因为三昧的力量，进入母亲的胎里，跟在兜率天上的时候一样。}%

@{生成就者，菩薩欲生時，諸天龍鬼神莊嚴三千大千世界；是時，有七寶蓮花座，自然而有；從母胎中，有無量菩薩先出，坐蓮華上，叉手讚歎，俟待菩薩；及諸天龍、鬼神、仙聖、諸玉女等皆合手一心，欲見菩薩生。然後菩薩從母右脇出，如滿月從雲中出，放大光明，照無量世界。是時，有大名聲，遍滿十方世界，唱言：某國菩薩末後身生。或有菩薩化生蓮華；於四生中，菩薩胎生、化生。於四種人中，菩薩生剎利、婆羅門二姓中——生此二種姓，人所貴故。}@
%{菩萨要出生的时候，诸天龙鬼神装饰三千大千世界；这个时候，有一个七宝莲花座，自然而然地出现了；从母亲的胎里，有无数的菩萨先出来，坐在莲花上，双手合十赞叹，等待着菩萨；以及诸天龙、鬼神、仙人、玉女等等，都双手合十，一心想要看到菩萨出生。然后菩萨从母亲的右肋出来，就像满月从云中出来一样，放出光明，照耀无数世界。这个时候，有巨大的名声，遍布十方世界，高喊着：某国的菩萨，最后的身躯出生了。或者有菩萨化生在莲花上；在四生中，菩萨胎生、化生。在四种人中，菩萨出生在刹利、婆罗门这两个姓氏中——出生在这两个姓氏中，是因为人们认为它们高贵。}%

@{家成就者，婆羅門家有智慧，剎利家有力勢；婆羅門利益後世，剎利利益今世。是二種於世有益，是故菩薩在此中生。復次，諸功德法家，所謂不退轉生，是名家生。}@
%{家成就指的是，婆罗门家有智慧，刹利家有力量；婆罗门利益后世，刹利利益今世。这两种姓氏对世间都有益处，所以菩萨出生在这里。再次，诸功德法家，所谓不退转生，就是指家生。}%

@{姓成就者，菩薩兜率天上觀世間何姓為貴、能攝眾生，即於是姓中生。如七佛中，初三佛，憍陳如姓中生；次三佛，迦葉姓中生；釋迦文尼佛，憍曇姓中生。復次，菩薩初深心牢固，是名諸佛姓。有人言：得無生法忍，是諸佛姓；是時，得佛一切種智氣分故，如聲聞法中姓地人。}@
%{姓成就指的是，菩萨在兜率天上观察世间哪个姓氏最尊贵、能够摄受众生，就在哪个姓氏中出生。就像七佛中，前三佛，憍陈如姓中出生；后三佛，迦叶姓中出生；释迦牟尼佛，憍曇姓中出生。再次，菩萨最初的深心牢固，这就是诸佛的姓氏。有人说，获得无生法忍，就是诸佛的姓氏；这个时候，获得了佛的一切种智的氣分，就像声闻法中的姓地人。}%

@{眷屬成就者，皆是智人、善人，世世集功德。此中佛自說：純以菩薩為眷屬。如《不可思議經》中說：瞿毘耶是大菩薩，一切眷屬皆是住阿毘跋致地菩薩；以方便三昧變化力，為男、為女，共為眷屬。如轉輪聖王居士寶是夜叉、鬼神現作人身，與人共事。}@
%{眷属成就指的是，都是智者、善人，世世代代积累功德。佛陀在这里自己说：纯粹以菩萨为眷属。就像《不可思議經》中说：瞿毘耶是大菩萨，所有的眷属都是住阿毘跋致地的菩萨；以方便三昧的变化力量，化身为男、化身为女，共同成为眷属。就像转轮圣王的居士宝是夜叉、鬼神，现化为人身，与人一起做事。}%

@{出家成就者，如釋迦文菩薩，夜於宮殿，見諸婇女皆如死狀；十方諸天鬼神齎持幡華供養之具，奉迎將出。是時車匿雖先受淨飯王勅，而隨菩薩意，自牽馬至；四天王使者接捧馬足，踰城而出。為破諸煩惱及魔人，示一切眾人在家之穢。如此大功德貴重之人，猶尚出家，況諸凡細！如是等因緣，名出家成就。}@
%{出家成就指的是，就像释迦牟尼菩萨，晚上在宫殿里，看到所有的美女都像死了一样；十方诸天鬼神拿着幡花供养的器具，迎接他出去。这个时候，车匿虽然先接受了净饭王的命令，但还是按照菩萨的意思，自己牵着马过来了；四天王的使者接捧着马的脚，越过城墙出去。为了破除一切烦恼和魔人，向所有众生展示在家中的污秽。如此大功德、贵重的人，尚且出家，何况其他的凡夫俗子！像这样的因缘，就叫做出家成就。}%

@{莊嚴佛樹成就者，莊嚴菩提樹，如先說。佛此中自說：是菩提樹，以黃金為根，七寶為莖、節、枝、葉，莖、節、枝、葉光明遍照十方無數阿僧祇諸佛世界。或有佛以菩薩七寶莊嚴佛樹，或有不如是者。所以者何？諸佛神力不可思議，為眾生故現種種莊嚴。}@
%{庄严佛树成就指的是，庄严菩提树，就像之前说的一样。佛陀在这里自己说：这棵菩提树，以黄金为根，七宝为茎、节、枝、叶，茎、节、枝、叶的光明遍照十方无数阿僧祇诸佛世界。或者有佛以菩萨七宝庄严佛树，或者有不如是者。为什么呢？诸佛的神力不可思議，为了众生，显现种种庄严。}%

@{一切諸善功德成滿具足者，菩薩住七地中，破諸煩惱，自利具足；住八地、九地，利益他人，所謂教化眾生，淨佛世界。自利、利他深大故，一切功德具足。如阿羅漢、辟支佛，自利雖重，利他輕故，不名具足；諸天及小菩薩，雖能利益他，而自未除煩惱故，亦不具足。是名功德具足。(九地竟)}@
%{一切诸善功德圆满具足指的是，菩萨住在七地中，破除一切烦恼，自利具足；住在八地、九地，利益他人，所谓教化众生，净化佛的世界。自利、利他深大，所以一切功德具足。就像阿罗汉、辟支佛，自利虽然重要，利他轻微，所以不叫做具足；诸天和小菩萨，虽然能够利益他人，但是自己还没有除掉烦恼，所以也不叫做具足。这就是功德具足。(九地结束) }%

@{當知如佛者，菩薩坐如是樹下，入第十地，名為法雲地。譬如大雲澍雨，連下無間；心自然生無量無邊清淨諸佛法，念念無量。爾時，菩薩作是念：欲界魔王心未降伏。放眉間光，令百億魔宮闇蔽不現。魔即瞋惱，集其兵眾來逼菩薩。菩薩降魔已，十方諸佛慶其功勳，皆放眉間光，從菩薩頂入。是時，十地所得功德，變為佛法，斷一切煩惱習，得無礙解脫，具十力、四無所畏、四無礙智、十八不共法、大慈大悲等無量無邊諸佛法。是時，地為六種震動，天雨華香，諸菩薩、天人皆合手讚歎。是時，放大光明，遍照十方無量世界；十方諸佛、菩薩、天人，大聲唱言：某方某國某甲菩薩，坐於道場，成具佛事，是其光明。是名十地當知如佛。}@
%{要知道，像佛陀一样，菩萨坐在这样的树下，进入第十地，叫做法云地。譬如大云降雨，连续不断地下；心自然而然地生起无数无边清净的佛法，念念无量。这个时候，菩萨心想：欲界魔王的心还没有降伏。放出眉间的光，让百亿魔宫黑暗遮蔽，不显现。魔王顿时愤怒，集合他的军队来逼迫菩萨。菩萨降伏了魔王之后，十方诸佛庆贺他的功勋，都放出眉间的光，从菩萨头顶进入。这个时候，十地所得的功德，转化为佛法，断除一切烦恼习气，获得无碍解脱，具有十力、四无所畏、四无碍智、十八不共法、大慈大悲等等无数无边的佛法。这个时候，大地发生六种震动，天上下着花香，诸菩萨、天人，都双手合十赞叹。这个时候，放出光明，遍照十方无数世界；十方诸佛、菩萨、天人，大声喊着：某方某国某甲菩萨，坐在道场，成就了佛事，这就是他的光明。这就是十地当知如佛。}%
@{復次，佛此中更說第十地相，所謂菩薩行六波羅蜜，以方便力故，過乾慧地，乃至菩薩地，住於佛地。佛地，即是第十地。}@
%{接着，佛陀在这里进一步阐述了第十地的特点，也就是菩萨修行的六度波罗蜜，凭借方便的力量，超越了乾慧地，最终到达菩萨地，住在了佛地。佛地，就是第十地。}%

@{菩薩能如是行十地，是名發趣大乘。}@
%{菩萨能够这样修行十地，这被称为发趣大乘。}%

@{大智度論釋出到品第二十一}@

%{【经文】}%

@{佛告須菩提：汝所問是乘何處出，至何處住者。佛言：是乘從三界中出，至薩婆若中住，以不二法故。何以故？摩訶衍、薩婆若，是二法共，不合、不散，無色、無形、無對，一相所謂無相。}@
%{佛陀对须菩提说：“你所问的这个乘，是从哪里出发的，最终停留在哪里？"佛陀说：“这个乘，是从三界中诞生的，最终停留在无上正等正觉中，这是因为不二法的缘故。为什么呢？因为大乘和无上正等正觉，这两个法是合一的，既不分离也不散乱，没有颜色、没有形状、没有对立，只有一个相，叫做无相。”}%

@{若人欲使實際出，是人為欲使無相法出；若人欲使如、法性、不可思議性出，是人為欲使無相法出。}@
%{如果有人想要让实相显现出来，这个人就是想要让无相法显现出来；如果有人想要让如、法性、不可思议性显现出来，这个人就是想要让无相法显现出来。}%

@{若人欲使色空出，是人為欲使無相法出；若人欲使受、想、行、識空出，是人為欲使無相法出。何以故？須菩提！色空相不出三界，亦不住薩婆若；受、想、行、識空相不出三界，亦不住薩婆若。所以者何？色，色相空；受、想、行、識，識相空故。}@
%{如果有人想要让色空显现出来，这个人就是想要让无相法显现出来；如果有人想要让受、想、行、识空显现出来，这个人就是想要让无相法显现出来。为什么呢？须菩提！色空相不会超出三界，也不会停留在无上正等正觉中；受、想、行、识空相不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为色，色相是空的；受、想、行、识，识相是空的。}%

@{若人欲使眼空出，是人為欲使無相法出；若人欲使耳、鼻、舌、身、意空出，是人為欲使無相法出；若人欲使乃至意觸因緣生受空出，是人為欲使無相法出。何以故？須菩提！眼空不出三界，亦不住薩婆若；乃至意觸因緣生受空不出三界，亦不住薩婆若。所以者何？眼，眼相空；乃至意觸因緣生受，意觸因緣生受相空故。}@
%{如果有人想要让眼空显现出来，这个人就是想要让无相法显现出来；如果有人想要让耳、鼻、舌、身、意空显现出来，这个人就是想要让无相法显现出来；如果有人想要让乃至意触因缘生受空显现出来，这个人就是想要让无相法显现出来。为什么呢？须菩提！眼空不会超出三界，也不会停留在无上正等正觉中；乃至意触因缘生受空不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为眼，眼相是空的；乃至意触因缘生受，意触因缘生受相是空的。}%

@{若人欲使夢出，是人為欲使無相法出；若人欲使幻、焰、響、影、化出，是人為欲使無相法出。何以故？須菩提！夢相不出三界，亦不住薩婆若；幻、焰、響、影、化相亦不出三界，亦不住薩婆若。}@
%{如果有人想要让梦显现出来，这个人就是想要让无相法显现出来；如果有人想要让幻、焰、响、影、化显现出来，这个人就是想要让无相法显现出来。为什么呢？须菩提！梦相不会超出三界，也不会停留在无上正等正觉中；幻、焰、响、影、化相也不会超出三界，也不会停留在无上正等正觉中。}%

@{須菩提！若人欲使檀波羅蜜出，是人為欲使無相法出；若人欲使尸羅波羅蜜、羼提波羅蜜、毘梨耶波羅蜜、禪波羅蜜、般若波羅蜜出，是人為欲使無相法出。何以故？檀波羅蜜相不出三界，亦不住薩婆若；尸羅波羅蜜乃至般若波羅蜜不出三界，亦不住薩婆若。所以者何？檀波羅蜜，檀波羅蜜相空；尸羅波羅蜜、羼提波羅蜜、毘梨耶波羅蜜、禪波羅蜜、般若波羅蜜，般若波羅蜜相空故。}@
%{须菩提！如果有人想要让布施波罗蜜显现出来，这个人就是想要让无相法显现出来；如果有人想要让持戒波罗蜜、忍辱波罗蜜、精进波罗蜜、禅定波罗蜜、般若波罗蜜显现出来，这个人就是想要让无相法显现出来。为什么呢？布施波罗蜜相不会超出三界，也不会停留在无上正等正觉中；持戒波罗蜜乃至般若波罗蜜不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为布施波罗蜜，布施波罗蜜相是空的；持戒波罗蜜、忍辱波罗蜜、精进波罗蜜、禅定波罗蜜、般若波罗蜜，般若波罗蜜相是空的。}%

@{若人欲使內空出乃至無法有法空出，是人為欲使無相法出。何以故？須菩提！內空相乃至無法有法空相，不出三界，亦不住薩婆若。所以者何？內空、內空性空，乃至無法有法空、無法有法空性空故。}@
%{如果有人想要让内空显现出来，乃至想要让无法有法空显现出来，这个人就是想要让无相法显现出来。为什么呢？须菩提！内空相乃至无法有法空相，不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为内空，内空性是空的，乃至无法有法空，无法有法空性是空的。}%

@{若人欲使四念處出，是人為欲使無相法出。何以故？四念處性不出三界，亦不住薩婆若。所以者何？四念處，四念處性空故。若人欲使四正勤、四如意足、五根、五力、七覺分、八聖道分出，是人為欲使無相法出。何以故？八聖道分性不出三界，亦不住薩婆若。所以者何？八聖道分，八聖道分性空故。乃至十八不共法亦如是。}@
%{如果有人想要让四念处显现出来，这个人就是想要让无相法显现出来。为什么呢？四念处性不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为四念处，四念处性是空的。如果有人想要让四正勤、四如意足、五根、五力、七觉分、八圣道分显现出来，这个人就是想要让无相法显现出来。为什么呢？八圣道分性不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为八圣道分，八圣道分性是空的。乃至十八不共法也是如此。}%

@{須菩提！若人欲使阿羅漢出生處，是人為欲使無相法出；若人欲使辟支佛出生處，是人為欲使無相法出；若人欲使多陀阿伽度、阿羅訶、三藐三佛陀出生處，是人為欲使無相法出。何以故？須菩提！阿羅漢性、辟支佛性、佛性不出三界，亦不住薩婆若。所以者何？阿羅漢，阿羅漢性空；辟支佛，辟支佛性空；佛，佛性空故。若人欲使須陀洹果、斯陀含果、阿那含果、阿羅漢果、辟支佛道、佛道、一切種智出，是人為欲使無相法出，如上說。}@
%{须菩提！如果有人想要让阿罗汉的出生处显现出来，这个人就是想要让无相法显现出来；如果有人想要让辟支佛的出生处显现出来，这个人就是想要让无相法显现出来；如果有人想要让多陀阿伽度、阿罗汉、正等正觉的出生处显现出来，这个人就是想要让无相法显现出来。为什么呢？须菩提！阿罗汉性、辟支佛性、佛性不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为阿罗汉，阿罗汉性是空的；辟支佛，辟支佛性是空的；佛，佛性是空的。如果有人想要让须陀洹果、斯陀含果、阿那含果、阿罗汉果、辟支佛道、佛道、一切种智显现出来，这个人就是想要让无相法显现出来，就像上面说的那样。}%

@{若人欲使名字、假名、施設、相、但有語言出，是人為欲使無相法出。何以故？名字空，不出三界，亦不住薩婆若。所以者何？名字相，名字相空故。乃至施設亦如是。}@
%{如果有人想要让名字、假名、方便、相、仅仅是语言显现出来，这个人就是想要让无相法显现出来。为什么呢？名字是空的，不会超出三界，也不会停留在无上正等正觉中。这是为什么呢？因为名字相，名字相是空的。乃至方便也是如此。}%
@{若人欲使不生不滅法、不垢不淨、無作法出，是人為欲使無相法出。何以故？不生乃至無作法性不出三界，亦不住薩婆若。所以者何？不生性乃至無作性，性空故。}@
%{如果有人想要讓不生不滅法、不垢不淨、無作法從三界中出來，那他就是想要讓無相法從三界中出來。為什麼呢？因為不生不滅法、不垢不淨、無作法這些法性，不會從三界中出來，也不會住於薩婆若。這是為什麼呢？因為不生不滅法、不垢不淨、無作法這些法性，其本性是空的。}%

@{須菩提！以是因緣故，摩訶衍從三界中出，至薩婆若中住，不動法故。}@
%{須菩提啊！因為這個原因，大乘法從三界中出來，住於薩婆若中，因為大乘法是不動的法。}%

@{論}@
%{論}%

@{問曰：}@
%{問道：}%

@{佛已知須菩提所問，今何以更稱而答？}@
%{佛已經知道須菩提所問的問題了，為什麼還要再稱呼他並回答呢？}%

@{答曰：}@
%{回答說：}%

@{是《摩訶般若波羅蜜》有十萬偈，三百二十萬言，與《四阿含》等，此非一坐說盡。又上須菩提所問，已答二事，異時異日故稱第三問而答。}@
%{這《摩訶般若波羅蜜經》有十萬偈，三百二十萬字，和《四阿含經》一樣多，這不是一坐就能說完的。而且上面須菩提所問的問題，已經回答了兩件事，因為是不同的時間和不同的日子，所以才稱作第三個問題來回答。}%

@{復次，有人言：聲聞法中無有不可思議事，不得一日一坐中說盡。佛有無礙解脫，菩薩有不可思議三昧，能令多時作少時、少時作多時，亦能以大色入小、小色作大。又如六十小劫說《法華經》，人謂從旦至食。}@
%{另外，有人說：聲聞法中沒有不可思議的事情，不可能在一天之內一坐就能說完。佛有無礙解脫，菩薩有不可思議的三昧，能使長時間變成短時間，短時間變成長時間，也能使大的顏色變成小的，小的顏色變成大的。就像說《法華經》需要六十小劫的時間，人們卻覺得只是從早晨到中午的時間而已。}%

@{問曰：}@
%{問道：}%

@{色有形可見，時無形但有名，云何得以近為遠、以遠為近？}@
%{顏色有形狀可以看見，時間沒有形狀只有名字，怎麼能把近的變成遠的，把遠的變成近的？}%

@{答曰：}@
%{回答說：}%

@{以是故說以不可思議神通力。如人夢中，夢有所見，自以為覺；夢中復夢，如是展轉，故是一夜。}@
%{所以說要用不可思議的神通力。就像人做夢的時候，夢到了一些東西，自己以為是醒著；夢中又做夢，這樣反覆，所以就好像是一夜的時間。}%

@{以是故，更稱其問而答。}@
%{所以，再次稱呼他的問題來回答。}%

@{是乘何處出，至何處住者，佛答：是乘從三界中出，至薩婆若中住。}@
%{這大乘法從哪裡出來，到哪裡住呢？佛回答說：這大乘法從三界中出來，住於薩婆若中。}%

@{問曰：}@
%{問道：}%

@{是乘為是佛法？為是菩薩法？若是佛法，云何從三界出？若是菩薩法，云何薩婆若中住？}@
%{這大乘法是佛法呢？還是菩薩法呢？如果是佛法，怎麼會從三界中出來呢？如果是菩薩法，怎麼會住於薩婆若中呢？}%

@{答曰：}@
%{回答說：}%

@{是乘是菩薩法，乃至金剛三昧；是諸功德清淨，變為佛法。是乘有大力，能有所去，直以至佛，更無勝處可去，故言住。譬如劫盡火燒三千世界，勢力甚大，更無所燒，故便自滅；摩訶衍亦如是，斷一切煩惱，集諸功德，盡其邊際，更無所斷，更無所知，更無所集故，便自歸滅。}@
%{這大乘法是菩薩法，直到金剛三昧；這些功德清淨了，就變成佛法。這大乘法有很大的力量，能有所去，直接到達佛的境界，再也沒有比這更好的地方可以去，所以說住。就像劫盡的時候，火燒了三千世界，力量非常大，再也沒有東西可以燒了，所以就自己滅了；大乘法也是一樣，斷了一切煩惱，集聚了所有的功德，達到了極限，再也沒有東西可以斷，再也沒有東西可以知道，再也沒有東西可以集聚了，所以就自己歸於寂滅。}%

@{不二法者，斷諸菩薩著故說。此中佛自說：大乘、薩婆若，是二法不一故不合，不異故不散；六情所知盡虛妄故，無色、無形、無對、一相。}@
%{不二法，是為了斷除菩薩的執著而說的。這裡佛自己說：大乘法和薩婆若，這兩種法不是一個，所以不能合在一起，不是不同的，所以不會散開；六根所認識的都是虛妄的，所以沒有顏色、沒有形狀、沒有對待、只有一相。}%

@{問曰：}@
%{問道：}%

@{先言不一故不合，今何以言一相？}@
%{之前說不是一個所以不能合在一起，現在為什麼又說只有一相呢？}%

@{答曰：}@
%{回答說：}%

@{此中言一相，所謂無相，無相則無有出至佛道；為引導凡夫人故，說言一相。}@
%{這裡說只有一相，指的是無相，無相就沒有從三界中出來到達佛道；為了引導凡夫眾生，所以說只有一相。}%

@{實際者，是諸法末後實相，無出無入；若有狂人欲使實際出至佛道者，此人則欲使無相法出。}@
%{實際，是所有法的最後的實相，沒有出來，也沒有進去；如果有人像瘋子一樣想要讓實際從三界中出來到達佛道，那他就是想要讓無相法從三界中出來。}%

@{如、法性、法相，如先說。}@
%{如、法性、法相，就像之前說的一樣。}%

@{不可思議性者，有人言：即是如、法性、實際，無量無邊，心心數法滅，故言不可思議。復有人言：過實際、涅槃，更求諸法實，若有若無，是名不可思議。復次，一切諸佛法，無有能思惟籌量者，故名不可思議。復有人言：一切諸法，分別思惟，皆同涅槃相，是不可思議。}@
%{不可思議性，有人說：就是如、法性、實際，無量無邊，心心數法滅，所以說不可思議。又有人說：超越實際、涅槃，再尋找所有法的實相，有也像沒有，沒有也像有，這就是不可思議。另外，所有佛法，沒有人能思惟和籌量，所以叫做不可思議。又有人說：所有法，分別思惟，都和涅槃的相狀一樣，這就是不可思議。}%

@{若人欲使空中出，此人則欲使無相法中出。此中佛自說：五眾空相，不能出三界，不能至薩婆若，五眾中五眾相空故，十二入乃至意觸因緣生受空亦如是。}@
%{如果有人想要讓空性從三界中出來，那他就是想要讓無相法從三界中出來。這裡佛自己說：五蘊的空相，不能從三界中出來，不能到達薩婆若，因為五蘊中五蘊的相狀是空的，十二入乃至意觸因緣生受的空性也是一樣。}%

@{夢等空譬喻，亦如是；自相空故，無出無至。}@
%{夢境等空性的比喻，也是一樣；因為自身是空的，所以沒有出來，也沒有進去。}%

@{若人欲使六波羅蜜出，此人則為欲使無相法出。何以故？六波羅蜜因緣和合故無自性，自性無故空。菩薩著六波羅蜜，墮邪道，故為說空。十八空乃至一切種智，亦如是。}@
%{如果有人想要讓六波羅蜜從三界中出來，那他就是想要讓無相法從三界中出來。為什麼呢？因為六波羅蜜是由因緣和合而成的，所以沒有自性，沒有自性所以是空的。菩薩執著於六波羅蜜，就會墮入邪道，所以佛才說空。十八空乃至一切種智，也是一樣。}%

@{問曰：}@
%{問道：}%

@{六波羅蜜有道俗，俗可著故可說空；出世間六波羅蜜、三十七品乃至十八不共法，無所著故，何以說空？}@
%{六波羅蜜有世間的和出世間的，世間的可以執著，所以可以說空；出世間的六波羅蜜、三十七品乃至十八不共法，沒有可以執著的，為什麼還要說空呢？}%

@{答曰：}@
%{回答說：}%
@{諸菩薩漏未盡，以福德、智慧力故行是法，或取相愛著故。凡夫法虛妄顛倒，此法從凡夫法邊生，云何是實？以是故，佛說是亦空，以喻無相法。}@
%{那些菩萨还有漏没有断尽，依靠福德和智慧的力量来修行这个法，或者因为执着于相而生起爱着。凡夫的法都是虚妄颠倒的，这个法是从凡夫法那里生出来的，怎么说是真实的呢？因此，佛说这个法也是空的，用这个来比喻无相的法。}%

@{是大乘即是無相，無相云何有出、有至？}@
%{大乘法就是无相的，无相怎么会有出来和到达呢？}%

@{諸法皆空，但有名字、相、假名、語言；今名字等亦空，以喻無相，第一義中不可得；世俗法中有相。名字等假名相義，如先說。}@
%{所有法都是空的，只是有名字、相、假名、语言；现在名字等等也是空的，用这个来比喻无相，在第一义谛中是不可得的；世俗法中是有相的。名字等等这些假名相的意义，就像前面说的那样。}%

@{用如是法，從三界出，至薩婆若中住；非是實法，亦無所動。}@
%{用这样的法，从三界中出来，到达无漏智中住着；这不是真实的法，也没有任何的动摇。}%

@{【經】}@
%{《经》}%

@{須菩提！汝所問是乘何處住者，須菩提！是大乘無住處。何以故？一切法無住相故。是乘若住，不住法住。須菩提！譬如法性，不生不滅、不垢不淨、無起無作，非住非不住。須菩提！是乘亦如是，非住非不住。何以故？法性相乃至無作相，非住非不住。所以者何？法性相性空故，乃至無作性、無作性性空故。諸餘法亦如是。須菩提！以是因緣故，是乘無住處，以不住法、不動法故。}@
%{须菩提！你所问的大乘法住在哪里，须菩提！大乘法是无住处的。为什么呢？因为一切法都没有住相。大乘法如果住，不是住在法上。须菩提！譬如法性，不生不灭、不垢不净、无起无作，既不是住也不是不住。须菩提！大乘法也是这样，既不是住也不是不住。为什么呢？因为法性相乃至无作相，既不是住也不是不住。为什么呢？因为法性相空，乃至无作性、无作性性空。其他法也是这样。须菩提！因为这个缘故，大乘法是无住处的，因为不住法、不动法。}%

@{【論】}@
%{《论》}%

@{問曰：}@
%{问曰：}%

@{上言是乘到薩婆若，更無勝法可去，今何以復說是乘無住處？}@
%{上面说大乘法到达无漏智，再没有比这更好的法可以去，现在为什么又说大乘法是无住处的呢？}%

@{答曰：}@
%{答曰：}%

@{先說以空、不二法故言住，如幻、如夢，雖有坐臥行住，非實是住；菩薩亦如是，雖言到薩婆若住，亦無定住。}@
%{前面说因为空、不二的法，所以说住，就像幻境、梦境一样，虽然有坐卧行走，但不是真实的住；菩萨也是这样，虽然说到达无漏智住着，但也没有固定的住处。}%

@{佛此中自說：一切法從本已來無住相，云何獨大乘有住？若有所住，以畢竟空法住。}@
%{佛在这里自己说：一切法从本来就无住相，怎么唯独大乘法有住处呢？如果有住处，就是以毕竟空的法来住。}%

@{譬如：如、法性、法相、實際，非住、非不住，不生不滅，不垢不淨，不起不作。}@
%{譬如：如、法性、法相、实相，既不是住也不是不住，不生不灭，不垢不净，不起无作。}%

@{不住者，自相中不住；非不住者，異相中不住。}@
%{不住，是在自相中不住；非不住，是在异相中不住。}%

@{不住者，說空破有；非不住者，說世諦方便有住。}@
%{不住，是说空破有；非不住，是说世俗谛方便的有住。}%

@{不住者，說無常破常相；非不住者，破滅相。}@
%{不住，是说无常破常相；非不住，是破灭相。}%

@{此中佛自說：法性、法性相空。何以故？自相空故。乃至無起無作諸餘法亦如是。}@
%{佛在这里自己说：法性、法性相空。为什么呢？因为自相空。乃至无起无作，其他法也是这样。}%

@{【經】}@
%{《经》}%

@{須菩提！汝所問誰當乘是乘出者，無有人乘是乘出者。何以故？是乘及出者、所用法及出時，是一切法皆無所有；若一切法無所有，用何等法當出？何以故？我不可得，乃至知者、見者不可得，畢竟淨故。不可思議性不可得，畢竟淨故。眾、入、界不可得，畢竟淨故。檀波羅蜜不可得，畢竟淨故；乃至般若波羅蜜不可得，畢竟淨故。內空不可得，畢竟淨故；乃至無法有法空不可得，畢竟淨故。四念處不可得，乃至十八不共法不可得，畢竟淨故。須陀洹不可得，乃至阿羅漢、辟支佛、菩薩、佛不可得，畢竟淨故。須陀洹果，乃至阿羅漢果、辟支佛道、佛道、一切種智不可得，畢竟淨故。不生不滅，不垢不淨，無起無作不可得，畢竟淨故。過去世、未來世、現在世，生、住、滅不可得，畢竟淨故。增、減不可得，畢竟淨故。何法不可得故不可得？法性不可得故不可得，如、實際、不可思議性、法性、法相、法位；檀波羅蜜不可得故不可得，乃至般若波羅蜜不可得故不可得。內空不可得故不可得，乃至無法有法空不可得故不可得。四念處不可得故不可得，乃至十八不共法不可得故不可得。須陀洹不可得故不可得，乃至佛不可得故不可得。須陀洹果不可得故不可得，乃至佛道不可得故不可得。不生不滅，乃至不起不作不可得故不可得。}@
%{须菩提！你所问谁应该乘着大乘法出来，没有人乘着大乘法出来。为什么呢？因为大乘法和出来者、所用法和出来的时候，都是一切法皆无所有；如果一切法皆无所有，用什么法来出来呢？为什么呢？因为我不可得，乃至知者、见者不可得，因为毕竟清净。不可思議性不可得，因为毕竟清净。众、入、界不可得，因为毕竟清净。檀波罗蜜不可得，因为毕竟清净；乃至般若波罗蜜不可得，因为毕竟清净。内空不可得，因为毕竟清净；乃至无法有法空不可得，因为毕竟清净。四念处不可得，乃至十八不共法不可得，因为毕竟清净。须陀洹不可得，乃至阿罗汉、辟支佛、菩萨、佛不可得，因为毕竟清净。须陀洹果，乃至阿罗汉果、辟支佛道、佛道、一切种智不可得，因为毕竟清净。不生不灭，不垢不净，无起无作不可得，因为毕竟清净。过去世、未来世、现在世，生、住、灭不可得，因为毕竟清净。增、减不可得，因为毕竟清净。什么法不可得，所以不可得呢？因为法性不可得，所以不可得，如、实相、不可思議性、法性、法相、法位；因为檀波罗蜜不可得，所以不可得，乃至因为般若波罗蜜不可得，所以不可得。因为内空不可得，所以不可得，乃至因为无法有法空不可得，所以不可得。因为四念处不可得，所以不可得，乃至因为十八不共法不可得，所以不可得。因为须陀洹不可得，所以不可得，乃至因为佛不可得，所以不可得。因为须陀洹果不可得，所以不可得，乃至因为佛道不可得，所以不可得。因为不生不灭，乃至因为不起无作不可得，所以不可得。}%

@{復次，須菩提！初地不可得故不可得，乃至第十地不可得故不可得，畢竟淨故。云何為初地乃至十地？所謂乾慧地、性地、八人地、見地、薄地、離欲地、已作地、辟支佛地、菩薩地、佛地。內空中初地不可得，乃至無法有法空中初地不可得；內空乃至無法有法空中，第二、第三、第四、第五、第六、第七、第八、第九、第十地不可得。何以故？須菩提！初地非得非不得，乃至十地非得非不得，畢竟淨故。內空乃至無法有法空中，成就眾生不可得，畢竟淨故。內空乃至無法有法空中，淨佛世界不可得，畢竟淨故。內空乃至無法有法空中，五眼不可得，畢竟淨故。如是，須菩提！菩薩摩訶薩，以一切諸法不可得故，乘是摩訶衍出薩婆若。}@
%{其次，须菩提！初地不可得，所以不可得，乃至第十地不可得，所以不可得，因为毕竟清净。什么叫初地乃至十地呢？就是指乾慧地、性地、八人地、见地、薄地、离欲地、已作地、辟支佛地、菩萨地、佛地。内空中初地不可得，乃至无法有法空中初地不可得；内空乃至无法有法空中，第二、第三、第四、第五、第六、第七、第八、第九、第十地不可得。为什么呢？须菩提！初地既不是得也不是不得，乃至十地既不是得也不是不得，因为毕竟清净。内空乃至无法有法空中，成就众生不可得，因为毕竟清净。内空乃至无法有法空中，净佛世界不可得，因为毕竟清净。内空乃至无法有法空中，五眼不可得，因为毕竟清净。这样，须菩提！菩萨摩诃萨，因为一切诸法不可得，所以乘着大乘法出来，到达无漏智。}%
@{者言：}@
%{有人说：}%

@{出者，行是乘到佛道邊出，又復以成就故名出，以是乘成就薩婆若，是名為出。}@
%{“出”的意思，是指通过修行乘佛道，到达佛道边际而出来，也因为成就了佛道而被称为“出”，因为乘佛道成就了无上正等正觉，所以被称为“出”。}%

@{此中佛自說空因緣：乘者，是六波羅蜜。所用法者，是慈悲、方便等諸法，六波羅蜜所不攝。出者，是菩薩。是三法皆空。}@
%{佛陀在这里解释“出”的空性缘起：乘，指的是六度波罗蜜；所用法，指的是慈悲、方便等诸法，这些法不包括在六度波罗蜜中；出，指的是菩萨。这三种法都是空的。}%

@{此中佛復說因緣：我不可得，乃至知者、見者不可得，畢竟空故。}@
%{佛陀又解释了“我”不可得的缘起：我不可得，乃至知觉者、观察者不可得，因为一切都毕竟空。}%

@{五眾、十二入、十八界，檀波羅蜜乃至十八不共法，須陀洹乃至薩婆若，不生不滅、不垢不淨，乃至三世、三相、增減等，是名法空。}@
%{色、受、想、行、识五蕴，十二处，十八界，布施波罗蜜乃至十八不共法，须陀洹乃至无上正等正觉，不生不灭、不垢不净，乃至三世、三相、增减等等，这些都是法的空性。}%

@{我乃至知者、見者，須陀洹乃至佛，是名眾生空。}@
%{我、知觉者、观察者，须陀洹乃至佛，这些都是众生的空性。}%

@{問曰：}@
%{有人问：}%

@{有二種不可得：一者、有法，智慧少故不能得；二者、有大智慧，推求不能得。此云何不可得？}@
%{有两种不可得：一是，有法，因为智慧不够，所以不能得到；二是，有大智慧，却推求不到。这两种情况，为什么说不可得呢？}%

@{答曰：}@
%{回答说：}%

@{是法無故不可得。}@
%{因为法本来就是空的，所以不可得。}%

@{問曰：}@
%{有人问：}%

@{一切法本末不可得，於人有何利益？}@
%{一切法从根源到结果都是不可得的，对人有什么益处呢？}%

@{答曰：}@
%{回答说：}%

@{此中佛自說：畢竟清淨故。}@
%{佛陀在这里说：因为一切法毕竟清净。}%

@{畢竟者，若行者依無而破有，於有得清淨，於無未清淨，以依止故。}@
%{“毕竟”的意思是，修行者依靠空性来破除有，在有中获得清净，在空性中还没有获得清净，因为依靠空性而修行。}%

@{此中佛自說不可得因緣：一切眾生不可得，一切法不可得；譬如如、法性、實際等，乃至不作、不起不可得。}@
%{佛陀在这里解释了“不可得”的缘起：一切众生不可得，一切法不可得；比如如、法性、实相等等，乃至不造作、不起心动念，这些都是不可得的。}%

@{復次，十八空故，法性不可得，乃至不起、不作。}@
%{此外，因为十八空，所以法性不可得，乃至不起心动念、不造作。}%

@{十八空中，無初地乃至十地，無成就眾生，無淨佛世界，無五眼；以十八空故空，畢竟清淨故不可得。}@
%{十八空里面，没有初地乃至十地，没有成就的众生，没有清净的佛国世界，没有五眼；因为十八空而空，因为毕竟清净，所以不可得。}%

@{菩薩用不可得法，乘是乘出薩婆若。}@
%{菩萨用不可得的法，乘着这个乘，出离无上正等正觉。}%
