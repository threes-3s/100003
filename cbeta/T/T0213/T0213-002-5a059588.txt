@{法集要頌經正道品第十二}@
%{法集要颂经正道品第十二}%

@{正道四聖諦，智慧所觀察，}@
%{正道四圣谛，智慧所观察，}%
@{破壞愛輪迴，如風吹塵散。}@
%{破坏爱轮回，如风吹尘散。}%

@{能見聖諦者，寂靜應觀察，}@
%{能见圣谛者，寂静应观察，}%
@{滅除煩惱見，如雨灑微塵。}@
%{灭除烦恼见，如雨洒微尘。}%

@{八正最上道，四諦為法迹，}@
%{八正最上道，四谛为法迹，}%
@{是道名無為，智燈照愚暗。}@
%{是道名无为，智灯照愚暗。}%

@{道為八真妙，聖諦四句上，}@
%{道为八真妙，圣谛四句上，}%
@{無欲法之最，明眼善觀察。}@
%{无欲法之最，明眼善观察。}%

@{智為出世長，快樂證無為，}@
%{智为出世长，快乐证无为，}%
@{知受正教者，永盡生老死。}@
%{知受正教者，永尽生老死。}%

@{一切行無常，如慧所觀察，}@
%{一切行无常，如慧所观察，}%
@{若能覺此苦，行道淨其迹。}@
%{若能觉此苦，行道净其迹。}%

@{一切諸行苦，如慧之所見，}@
%{一切诸行苦，如慧之所见，}%
@{若能覺此苦，行道淨其迹。}@
%{若能觉此苦，行道净其迹。}%

@{一切諸行空，如慧之所見，}@
%{一切诸行空，如慧之所见，}%
@{若能覺此苦，行道淨其迹。}@
%{若能觉此苦，行道净其迹。}%

@{一切法無我，如慧之所見，}@
%{一切法无我，如慧之所见，}%
@{若能覺此苦，行道淨其迹。}@
%{若能觉此苦，行道净其迹。}%

@{吾已說道迹，愛箭而為射，}@
%{吾已说道迹，爱箭而为射，}%
@{宜以自勗勵，諦受如來言。}@
%{宜以自勗励，谛受如来言。}%

@{吾已說道迹，拔愛堅固刺，}@
%{吾已说道迹，拔爱坚固刺，}%
@{宜以自勗勵，諦受如來言。}@
%{宜以自勗励，谛受如来言。}%

@{此道無別法，見諦之所淨，}@
%{此道无别法，见谛之所净，}%
@{趣向滅眾苦，能壞魔羅軍。}@
%{趣向灭众苦，能坏魔罗军。}%

@{此道無有餘，見諦能證果，}@
%{此道无有余，见谛能证果，}%
@{趣向滅眾苦，能破魔羅軍。}@
%{趣向灭众苦，能破魔罗军。}%

@{是道更無過，一趣如淵流，}@
%{是道更无过，一趣如渊流，}%
@{如能仁入定，在眾頻演道，}@
%{如能仁入定，在众频演道，}%
@{一入見生死，得道為祐助。}@
%{一入见生死，得道为祐助。}%

@{此道度當度，截流至彼岸，}@
%{此道度当度，截流至彼岸，}%
@{究竟道清淨，已盡生死源。}@
%{究竟道清净，已尽生死源。}%

@{辯才無邊界，明見宣說道，}@
%{辩才无边界，明见宣说道，}%
@{可趣服甘露，前未聞法輪。}@
%{可趣服甘露，前未闻法轮。}%

@{轉為哀眾生。禮拜奉事者，}@
%{转为哀众生。礼拜奉事者，}%
@{化之度三有，三念可念善，}@
%{化之度三有，三念可念善，}%
@{三念當離惡。從念而有行，}@
%{三念当离恶。从念而有行，}%
@{滅之為正斷，三觀為轉念，}@
%{灭之为正断，三观为转念，}%
@{逮獲無上道。得三除三窟，}@
%{逮获无上道。得三除三窟，}%
@{無量修念待，能除三有垢，}@
%{无量修念待，能除三有垢，}%
@{攝定用縛意，智慧禪定力，}@
%{摄定用缚意，智慧禅定力，}%
@{已定攝外亂。世間生滅法，}@
%{已定摄外乱。世间生灭法，}%
@{一一彼無邊，覺道獲解脫，}@
%{一一彼无边，觉道获解脱，}%
@{快樂無窮盡。積善得善行，}@
%{快乐无穷尽。积善得善行，}%
@{讚歎得名譽，逮賢聖八品，}@
%{赞叹得名誉，逮贤圣八品，}%
@{修道甘露果。}@
%{修道甘露果。}%

@{法集要頌經利養品第十三}@
%{法集要颂经利养品第十三}%

@{芭蕉以實死，竹蘆實亦然，}@
%{芭蕉以实死，竹芦实亦然，}%
@{駏驉坐妊終，人為貪利喪。}@
%{駏驉坐妊终，人为贪利丧。}%

@{如是貪無利，當知從癡生，}@
%{如是贪无利，当知从痴生，}%
@{愚為此害賢，首落分于地。}@
%{愚为此害贤，首落分于地。}%

@{貪利不善性，苾芻勿羨之，}@
%{贪利不善性，苾刍勿羡之，}%
@{住處多愛戀，希望他供養。}@
%{住处多爱恋，希望他供养。}%

@{在家及出家，族姓諸愚迷，}@
%{在家及出家，族姓诸愚迷，}%
@{貪利興嫉心，我為降伏彼。}@
%{贪利兴嫉心，我为降伏彼。}%

@{愚為愚計想，欲慢日夜增，}@
%{愚为愚计想，欲慢日夜增，}%
@{異哉得利養，圓寂趣不同。}@
%{异哉得利养，圆寂趣不同。}%

@{能論知足者，苾芻真佛子，}@
%{能论知足者，苾刍真佛子，}%
@{不貪著名譽，喜悅是智人。}@
%{不贪著名誉，喜悦是智人。}%

@{不愛著一切，不諂於他人，}@
%{不爱著一切，不谄于他人，}%
@{不依他活命，當自守法行。}@
%{不依他活命，当自守法行。}%

@{自利尚無貪，豈貴他名譽？}@
%{自利尚无贪，岂贵他名譽？}%
@{百味如膏車，支形得行道。}@
%{百味如膏车，支形得行道。}%

@{苾芻貪利養，不得三摩地，}@
%{苾刍贪利养，不得三摩地，}%
@{知足常寂靜，止觀可成就。}@
%{知足常寂静，止观可成就。}%

@{苾芻遠利譽，常足不貪求，}@
%{苾刍远利誉，常足不贪求，}%
@{但三衣飲食，真活命快樂。}@
%{但三衣饮食，真活命快乐。}%

@{苾芻不捨利，如毒蛇同室，}@
%{苾刍不舍利，如毒蛇同室，}%
@{坐臥睡寐畏，皆由貪活命。}@
%{坐卧睡寐畏，皆由贪活命。}%

@{苾芻不捨利，下劣中劣喜，}@
%{苾刍不舍利，下劣中劣喜，}%
@{一法應觀察，少智難得脫，}@
%{一法应观察，少智难得脱，}%
@{謹慎常依戒，無貪智者讚。}@
%{谨慎常依戒，无贪智者赞。}%

@{淨行正根力，應當自思惟，}@
%{净行正根力，应当自思惟，}%
@{具足得三明，解脫獲無漏。}@
%{具足得三明，解脱获无漏。}%

@{寡智尠識人，無所憶念知，}@
%{寡智尠识人，无所忆念知，}%
@{其於諸飲食，依於他人得，}@
%{其于诸饮食，依于他人得，}%
@{而有惡法生。由利養憎嫉，}@
%{而有恶法生。由利养憎嫉，}%
@{自利多結怨，徒服三法衣，}@
%{自利多结怨，徒服三法衣，}%
@{但望美飲食，不奉諸佛教。}@
%{但望美饮食，不奉诸佛教。}%

@{當知是過失，利養為大怖，}@
%{当知是过失，利养为大怖，}%
@{少智不審慮，苾芻應釋心。}@
%{少智不审虑，苾刍应释心。}%

@{苾芻說出家，三業應調伏，}@
%{苾刍说出家，三业应调伏，}%
@{不邪命自活，心善常思惟。}@
%{不邪命自活，心善常思惟。}%

@{微細病難忍，利養最難離，}@
%{微细病难忍，利养最难离，}%
@{供養心不動，天龍致禮拜。}@
%{供养心不动，天龙致礼拜。}%
@{法集要頌經怨家品第十四}@
%{法集要颂经怨家品第十四}%

@{不怨而興怨，不謗而造業，}@
%{不怨恨别人却招致怨恨，不诽谤别人却造下罪业，}%

@{愚迷受輪迴，今世及後世。}@
%{愚昧无知，受轮回之苦，今生今世，乃至来世都无法解脱。}%

@{先自作漏業，然後害他人，}@
%{首先自己造下恶业，然后再去伤害别人，}%

@{彼此相興害，如鳥墮羅網。}@
%{彼此互相伤害，就像鸟儿落入罗网一样，无法逃脱。}%

@{破他還自破，冤家遇冤家，}@
%{想要伤害别人，最终却伤害了自己，冤家遇到冤家，}%

@{毀他還自毀，瞋他還自瞋。}@
%{毁谤别人，最终毁谤了自己，憎恨别人，最终憎恨了自己。}%

@{斯何沙門行，不知正法本，}@
%{这哪是出家人应该做的行为？他们根本不懂正法的根本，}%

@{壽既獲短促，捨冤復結冤。}@
%{寿命短暂，却还放不下怨恨，反而结下新的怨恨。}%

@{眾相共毀謗，各發恚怒聲，}@
%{大家互相诽谤，各自怒火中烧，}%

@{歡心平等忍，此忍最無比。}@
%{只有以欢喜心，平等忍辱，才是最无与伦比的忍辱。}%

@{斷骨而命終，牛馬死財失，}@
%{断骨而死，牛马死亡，财物丢失，}%

@{國界則喪亂，聚集還復得。}@
%{国家陷入混乱，最终还是会恢复平静。}%

@{汝等不興惡，此法得離怨，}@
%{你们不要造恶，这样就能远离怨恨，}%

@{他怨能忍受，說之名為智。}@
%{能够忍受别人的怨恨，这才是真正的智慧。}%

@{若知此說勝，愚迷求快樂，}@
%{如果能明白这个道理，就能放下愚昧，寻求真正的快乐，}%

@{現在無怨意，未來亦無恨。}@
%{现在没有怨恨，未来也无怨恨。}%

@{不可怨以怨，終已得快樂，}@
%{不要以怨恨来报答怨恨，最终才能获得真正的快乐，}%

@{行忍怨自息，此名如來法。}@
%{以忍辱化解怨恨，这才是如来的教法。}%

@{若人致毀罵，彼勝我不勝，}@
%{如果有人辱骂我，他赢了，我并没有输，}%

@{快樂從意者，怨終得休息。}@
%{因为快乐来自于内心，怨恨最终会消散。}%

@{若人親善友，共遊於世間，}@
%{如果有人是亲密的朋友，一起在世间游历，}%

@{不積有冤餘，專念同其意。}@
%{不要积累怨恨，一心一意地与朋友同心同德。}%

@{設不得善友，獨遊無伴侶，}@
%{如果没有善友，独自一人旅行，}%

@{應觀諸國土，獨善不造惡。}@
%{就应该观察各个地方，独自做好事，不要造恶。}%

@{學無同伴侶，又不得親友，}@
%{学习没有同伴，也没有亲友，}%

@{寧獨守善行，不與愚人偕。}@
%{宁愿独自坚持善行，也不要和愚蠢的人在一起。}%

@{樂戒學法行，奚用伴侶為？}@
%{喜欢戒律，学习佛法，修行佛法，为什么要依靠同伴呢？}%

@{如龍好深淵，如象樂曠野。}@
%{就像龙喜欢深潭，大象喜欢旷野一样。}%

%{经常练习善语，沙门思坐起来，}%
@{一坐而所樂，欲求於息心。}@
%{一坐下来就感到快乐，想要寻求内心平静。}%

@{一坐而一臥，獨步而無伴，}@
%{一坐一卧，独自行走，没有同伴，}%
@{當自降伏心，自樂居山林。}@
%{应该降伏自己的心，独自快乐地住在山林。}%

@{千千而為敵，一夫能勝之，}@
%{面对千千万万的敌人，一个人也能战胜他们，}%
@{莫若自伏心，便為戰中勝。}@
%{不如降伏自己的心，就能在战斗中获得胜利。}%

@{自勝而為上，如彼眾生心，}@
%{战胜自己才是最高境界，就像那些众生的心，}%
@{自降為大士，眾行則具足。}@
%{自我降伏成为大士，各种行为才能圆满具足。}%

@{非天彥達嚩，非魔及梵天，}@
%{不是天人，不是魔，也不是梵天，}%
@{棄勝最為上，如智慧苾芻。}@
%{战胜自己才是最高境界，就像智慧的比丘。}%

@{先自而正己，然後正他人，}@
%{先要修正自己，然后才能修正他人，}%
@{若自而正者，乃謂之上士。}@
%{如果能修正自己，才是真正的上士。}%

@{先自而正己，然後正他人，}@
%{先要修正自己，然后才能修正他人，}%
@{若自而正者，不侵名真智。}@
%{如果能修正自己，才不会侵犯真智的名号。}%

@{當自而修剋，隨其教訓之，}@
%{应该自我修行克制，按照教诲去做，}%
@{己不被教訓，焉能教訓他？}@
%{自己没有接受教诲，怎么能教诲他人呢？}%

@{念自而修剋，使彼而信解，}@
%{念念自修克制，使他人信解，}%
@{我己意專心，智者所習學。}@
%{我一心一意专心致志，这是智者所学习的。}%

@{為己或為彼，多有不成就，}@
%{为了自己或为了他人，往往难以成就，}%
@{其有學此者，自正兼訓彼。}@
%{那些学习这个道理的人，应该修正自己，同时教诲他人。}%

@{身全得存道，爾時豈容彼？}@
%{身心完整才能保存道法，那个时候还容得下他人吗？}%
@{己以被降伏，智者演其義。}@
%{自己已经被降伏，智者阐述道理。}%

@{自己心為師，不隨他為師，}@
%{以自己的心为师，不跟随他人为师，}%
@{自己為師者，獲真智人法。}@
%{以自己为师的人，就能获得真智的人法。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，得譽獲利樂。}@
%{以自己为师的人，就能获得赞誉，获得利益，获得快乐。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，獲智為天人。}@
%{以自己为师的人，就能获得智慧，成为天人。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，久受生天樂。}@
%{以自己为师的人，就能长久享受天上的快乐。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，親族中最勝。}@
%{以自己为师的人，就能成为亲族中最优秀的人。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，煩惱中無憂。}@
%{以自己为师的人，就能在烦恼中没有忧愁。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，斷除一切縛。}@
%{以自己为师的人，就能断除一切束缚。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，能破諸惡趣。}@
%{以自己为师的人，就能破除一切恶趣。}%

%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，長作真智師。}@
%{以自己为师的人，就能长久成为真智的老师。}%

@{自己心為師，不依他為師，}@
%{以自己的心为师，不依靠他人为师，}%
@{自己為師者，解脫一切苦。}@
%{以自己为师的人，就能解脱一切苦。}%
