%{西天中印度惹烂陀罗国密林寺三藏明教大师赐紫沙门臣天息灾奉诏译}%

@{能覺悟煩惱，宜發歡喜心，}@
%{能觉悟烦恼，宜发欢喜心，}%
@{今聽我所集，佛所宣法頌。}@
%{今听我所集，佛所宣法颂。}%
@{如是佛世尊，一切智中師，}@
%{如是佛世尊，一切智中师，}%
@{慈悲為有情，廣說真實語。}@
%{慈悲为有情，广说真实语。}%

@{一切行非常，皆是興衰法，}@
%{一切行非常，皆是兴衰法，}%
@{夫生輒還終，寂滅最安樂。}@
%{夫生輒还终，寂灭最安乐。}%
@{如燭熾焰時，擲物在暗處，}@
%{如烛炽焰时，掷物在暗处，}%
@{不使智燈尋，恒為煩惱覆。}@
%{不使智灯寻，恒为烦恼覆。}%

@{人身有形器，棄散在諸方，}@
%{人身有形器，弃散在诸方，}%
@{骸骨如鴿色，觀斯有何樂？}@
%{骸骨如鸽色，观斯有何乐？}%
@{譬如人初夜，識託住母胎，}@
%{譬如人初夜，识托住母胎，}%
@{日涉多遷變，逝而定不還。}@
%{日涉多迁变，逝而定不还。}%

@{晨朝覩好事，夜至則不現，}@
%{晨朝睹好事，夜至则不现，}%
@{昨所瞻視者，今夕則或無。}@
%{昨所瞻视者，今夕则或无。}%
@{榮富焰熾盛，無常無時節，}@
%{荣富焰炽盛，无常无时节，}%
@{不揀擇貴賤，常被死王降。}@
%{不拣择贵贱，常被死王降。}%

@{或有在胎殞，或初誕亦亡，}@
%{或有在胎殒，或初诞亦亡，}%
@{盛壯不免死，老耄甘心受。}@
%{盛壮不免死，老耄甘心受。}%
@{若老或少年，及與中年者，}@
%{若老或少年，及与中年者，}%
@{恒被死來侵，云何不懷怖？}@
%{恒被死来侵，云何不怀怖？}%

@{命如菓自熟，常恐會零落，}@
%{命如菓自熟，常恐会零落，}%
@{生已必有終，誰能免斯者？}@
%{生已必有终，谁能免斯者？}%
@{譬如陶家師，埏埴作坏器，}@
%{譬如陶家师，埏埴作坏器，}%
@{諸有悉破壞，人命亦如是。}@
%{诸有悉破壞，人命亦如是。}%

@{如人彈琴瑟，具足眾妙音，}@
%{如人弹琴瑟，具足众妙音，}%
@{絃斷無少聲，人命亦如是。}@
%{弦断无少声，人命亦如是。}%
@{如囚被繫縛，拘牽詣都市，}@
%{如囚被系缚，拘牵诣都市，}%
@{動則向死路，壽命亦如是。}@
%{动则向死路，寿命亦如是。}%

@{如河急駛流，往而悉不還，}@
%{如河急驶流，往而悉不还，}%
@{人生亦如是，逝者皆不迴。}@
%{人生亦如是，逝者皆不回。}%
@{諸患集成身，生多眾苦惱，}@
%{诸患集成身，生多众苦恼，}%
@{人命亦如是，為老死所伺。}@
%{人命亦如是，为老死所伺。}%

@{所造成功勞，永世獲安樂，}@
%{所造成功劳，永世获安乐，}%
@{如杖擊急水，暫開還却合。}@
%{如杖击急水，暂开还却合。}%
@{如人操杖行，牧牛飲飼者，}@
%{如人操杖行，牧牛饮饲者，}%
@{人命亦如是，亦即養命去。}@
%{人命亦如是，亦即养命去。}%

@{夫人欲立德，晝夜勿空過，}@
%{夫人欲立德，昼夜勿空过，}%
@{既獲得人身，一心思命盡。}@
%{既获得人身，一心思命尽。}%
@{不寐覺夜長，疲倦道路長，}@
%{不寐觉夜长，疲倦道路长，}%
@{愚迷生死長，希聞於妙法。}@
%{愚迷生死长，希闻于妙法。}%

@{有子兼有財，慳惜遇散壞，}@
%{有子兼有财，慳惜遇散坏，}%
@{愚夫不自觀，何恃有財子？}@
%{愚夫不自观，何恃有财子？}%
@{百千非算數，族姓富男女，}@
%{百千非算数，族姓富男女，}%
@{積聚多財產，無不皆衰滅。}@
%{积聚多财产，无不皆衰灭。}%

@{富貴非聖財，恒為無常伺，}@
%{富贵非圣财，恒为无常伺，}%
@{猶如盲眼人，不能自觀察。}@
%{犹如盲眼人，不能自观察。}%
@{聚集還散壞，崇高必墜落，}@
%{聚集还散坏，崇高必坠落，}%
@{生者皆盡終，有情亦如是。}@
%{生者皆尽终，有情亦如是。}%

@{行惡入地獄，修善則生天，}@
%{行恶入地狱，修善则生天，}%
@{若能修善者，漏盡得涅槃。}@
%{若能修善者，漏尽得涅槃。}%

@{諸佛與菩薩，緣覺及聲聞，}@
%{诸佛与菩萨，缘觉及声闻，}%
@{尚捨有為身，何況諸有情？}@
%{尚舍有为身，何況诸有情？}%
@{非空非海中，非入山窟間，}@
%{非空非海中，非入山窟间，}%
@{無有地方所，脫止不受死。}@
%{无有地方所，脱止不受死。}%

@{若住現在世，過去及未來，}@
%{若住现在世，过去及未来，}%
@{一切有為事，終歸於盡壞。}@
%{一切有为事，终归于尽坏。}%
@{智者能離繫，恒正念觀察，}@
%{智者能离系，恒正念观察，}%
@{常思無漏道，是名真智者。}@
%{常思无漏道，是名真智者。}%

@{如囚被繫縛，所欲無能益，}@
%{如囚被系缚，所欲无能益，}%
@{亦如朽故車，不久見破壞。}@
%{亦如朽故车，不久见破壞。}%
@{色變為老耄，戀家如在獄，}@
%{色变为老耄，恋家如在狱，}%
@{不覺死來侵，愚夫不能知。}@
%{不觉死来侵，愚夫不能知。}%

@{雖壽滿百歲，亦被死相隨，}@
%{虽寿满百岁，亦被死相随，}%
@{為老病所逼，患終至後際。}@
%{为老病所逼，患终至后际。}%
@{老至苦纏身，晝夜多痛惱，}@
%{老至苦缠身，昼夜多痛恼，}%
@{辛楚有千般，如魚入灰火。}@
%{辛楚有千般，如鱼入灰火。}%

@{江河無停止，駛流去不迴，}@
%{江河无停止，驶流去不回，}%
@{保惜膿漏軀，雖戀不能住。}@
%{保惜脓漏躯，虽恋不能住。}%
@{四大聚集身，無常詎久留，}@
%{四大聚集身，无常詎久留，}%
@{地種散壞時，神識空何用？}@
%{地种散坏时，神识空何用？}%

@{此身多障惱，膿漏恒疾患，}@
%{此身多障恼，脓漏恒疾患，}%
@{愚迷貪愛著，不厭求寂滅。}@
%{愚迷贪爱著，不厌求寂灭。}%
@{今歲雖云在，冬夏不久停，}@
%{今岁虽云在，冬夏不久停，}%
@{凡夫貪世樂，中間不驚怖。}@
%{凡夫贪世乐，中间不惊怖。}%

@{父母與兄弟，妻子并眷屬，}@
%{父母与兄弟，妻子并眷属，}%
@{無常來牽引，無能救濟者。}@
%{无常来牵引，无能救济者。}%
@{如是諸有情，舉動貪榮樂，}@
%{如是诸有情，举动贪荣乐，}%
@{無常老病侵，不覺生苦惱。}@
%{无常老病侵，不觉生苦恼。}%

@{剃髮為苾芻，宜應修止觀，}@
%{剃发为苾刍，宜应修止观，}%
@{魔羅不能伺，度生到彼岸。}@
%{魔罗不能伺，度生到彼岸。}%

@{法集要頌經愛欲品第二}@
%{法集要颂经 爱欲品第二}%

@{慾我知汝根，意以思想生，}@
%{欲我知汝根，意以思想生，}%
@{我不思惟汝，則汝慾不有。}@
%{我不思惟汝，则汝欲不有。}%

@{因慾生煩惱，因慾生怖畏，}@
%{因欲生烦恼，因欲生怖畏，}%
@{離慾得解脫，無怖無煩惱。}@
%{离欲得解脱，无怖无烦恼。}%
@{從愛生煩惱，從愛生怖畏，}@
%{从爱生烦恼，从爱生怖畏，}%
@{離愛得解脫，無怖無煩惱。}@
%{离爱得解脱，无怖无烦恼。}%

@{菓先甜後苦，愛慾亦如斯，}@
%{菓先甜后苦，爱欲亦如斯，}%
@{後受地獄苦，燒煮無數劫。}@
%{后受地狱苦，烧煮无数劫。}%
@{愚迷貪愛慾，戀著於妻子，}@
%{愚迷贪爱欲，恋著于妻子，}%
@{為愛染纏縛，堅固難出離。}@
%{为爱染缠缚，坚固难出离。}%

@{賢聖示愛慾，莊嚴諸眷屬，}@
%{贤圣示爱欲，庄严诸眷属，}%
@{遠離於妻子，堅固能利益。}@
%{远离于妻子，坚固能利益。}%
@{貪欲難解脫，離欲真出家，}@
%{贪欲难解脱，离欲真出家，}%
@{不貪受快樂，智者無所欲。}@
%{不贪受快乐，智者无所欲。}%

@{世間貪欲人，種種非思惟，}@
%{世间贪欲人，种种非思惟，}%
@{若能調伏者，是名真離欲。}@
%{若能调伏者，是名真离欲。}%
@{若人恒貪欲，處縛難解脫，}@
%{若人恒贪欲，处缚难解脱，}%
@{唯慧能分別，煩惱斷不生。}@
%{唯慧能分别，烦恼断不生。}%

@{正念常興起，寂靜欲易除，}@
%{正念常兴起，寂静欲易除，}%
@{自制以法戒，不犯善增長。}@
%{自制以法戒，不犯善增长。}%
@{常行貪欲人，愚者共狎習，}@
%{常行贪欲人，愚者共狎习，}%
@{念定不放逸，次第獲無漏。}@
%{念定不放逸，次第获无漏。}%

@{剎那修止觀，能離諸罪垢，}@
%{刹那修止观，能离诸罪垢，}%
@{我慢自消除，解脫獲安樂。}@
%{我慢自消除，解脱获安乐。}%
@{若人不斷欲，如皮入火燒，}@
%{若人不断欲，如皮入火烧，}%
@{剎那見燋壞，受罪無央數。}@
%{刹那见燋坏，受罪无央数。}%

@{苾芻慎欲樂，放逸多憂愁，}@
%{苾刍慎欲乐，放逸多忧愁，}%
@{若離於愛欲，正念受快樂。}@
%{若离于爱欲，正念受快乐。}%
@{無厭有何足？不足何有樂？}@
%{无厌有何足？不足何有乐？}%
@{無樂有何憂？有愛有何樂？}@
%{无乐有何忧？有爱有何乐？}%

@{寂靜智慧足，能長無漏道，}@
%{寂静智慧足，能长无漏道，}%
@{貪愛若不足，非法受中夭。}@
%{贪爱若不足，非法受中夭。}%
@{見色心迷惑，不自觀無常，}@
%{见色心迷惑，不自观无常，}%
@{愚以為美善，不知其非真。}@
%{愚以为美善，不知其非真。}%

@{愚以貪自縛，不求度彼岸，}@
%{愚以贪自缚，不求度彼岸，}%
@{貪財為愛欲，害人亦自縛。}@
%{贪财为爱欲，害人亦自缚。}%
@{世容眾妙欲，此欲最味少，}@
%{世容众妙欲，此欲最味少，}%
@{若比天上樂，迦哩灑跛拏。}@
%{若比天上乐，迦哩灑跛拏。}%

@{眾山盡為金，猶如鐵圍山，}@
%{众山尽为金，犹如铁围山，}%
@{此猶無厭足，正覺盡能知。}@
%{此犹无厌足，正觉尽能知。}%
@{世間苦果報，皆因於貪欲，}@
%{世间苦果报，皆因于贪欲，}%
@{智者善調伏，應依此中學。}@
%{智者善调伏，应依此中学。}%


@{法集要頌經貪品第三}@
%{法集要颂经贪品第三}%

@{極貪善顯現，有情懷疑慮，}@
%{贪心过盛，显露出来，众生就会产生怀疑，心生犹豫。}%

@{若復增貪意，自作堅固縛。}@
%{如果再加深贪心，就会把自己捆绑得更牢固。}%

@{離貪善觀察，疑慮得消除，}@
%{远离贪心，好好观察，疑虑就会消除。}%

@{棄捨彼貪愛，堅固縛自壞。}@
%{放弃贪爱，捆绑你的东西自然就会消失。}%

@{以欲網自弊，以愛蓋自覆，}@
%{用欲望的网把自己困住，用爱把自己蒙蔽。}%

@{愚情自恣縛，如魚入釣手，}@
%{愚蠢的情感自我放纵，把自己捆绑起来，就像鱼儿掉进渔网一样。}%

@{死命恒來逼，如犢逐愛母。}@
%{死亡的威胁始终逼近，就像小牛犊追着母牛一样。}%

@{貪著放逸者，如猿逢果樹，}@
%{贪图享乐，放纵自己的人，就像猴子遇到果树一样。}%

@{貪意甚堅牢，趣而還復趣。}@
%{贪心非常坚固，一味追求，又反复追求。}%

@{夫貪愛潤澤，思想為滋蔓，}@
%{贪爱就像雨露，滋润着思想，让它蔓延生长。}%

@{貪欲深無底，老死是用增。}@
%{贪欲就像无底深渊，只会让人加速衰老和死亡。}%

@{貪欲多虛誑，貪欲懷悋惜，}@
%{贪欲充满了虚假和欺骗，贪欲总是让人心怀吝啬和贪婪。}%

@{若以慧分別，正觀獲安樂。}@
%{如果你用智慧来分辨，用正念观察，就能获得真正的快乐。}%

@{由貪受生死，奔波樂向前，}@
%{因为贪心，所以轮回生死，不停地追求着眼前的快乐。}%

@{群生無慧眼，不能自觀察。}@
%{芸芸众生没有智慧的眼睛，不能自己观察。}%

@{愚迷貪所執，沈淪豈覺知？}@
%{愚蠢迷茫，执着于贪心，沉沦其中，怎么还会知道呢？}%

@{若修瑜伽行，魔王不能伺。}@
%{如果你修习瑜伽行，魔王就无法靠近你。}%

@{貪垢難消釋，如犢戀愛母，}@
%{贪婪的污垢很难消除，就像小牛犊依恋母牛一样。}%

@{離貪免沈淪，離貪得解脫。}@
%{远离贪心，就能避免沉沦，获得解脱。}%

@{因貪增喧諍，因愛饒毀謗，}@
%{因为贪心，所以争吵不断，因为爱，所以充满诽谤。}%

@{苾芻修止觀，證得寂靜果。}@
%{比丘修习止观，就能证得寂静的果位。}%

@{貪意如良田，遇風雨增長，}@
%{贪心就像一块良田，遇到风吹雨打，就会更加茂盛。}%

@{若遠離貪愛，煩惱不能侵。}@
%{如果你远离贪爱，烦恼就无法侵袭你。}%

@{貪欲若薄劣，如水滴蓮上，}@
%{如果贪欲很微弱，就像水滴落在荷叶上一样。}%

@{彼煩惱易除，可說為智者。}@
%{那些烦恼很容易消除，这就可以称作是智慧。}%

@{伐樹不伐根，雖伐猶增長，}@
%{砍伐树木，不砍伐树根，虽然砍伐了，却还会继续生长。}%

@{拔貪不盡根，雖伐還復生。}@
%{拔除贪心，不拔除它的根，虽然拔除了，却还会再生。}%

@{貪欲如種田，耕之去雜穢，}@
%{贪欲就像种田，要耕地除掉杂草。}%

@{愛苗若不耘，善果不堅貞。}@
%{如果不去除爱苗，善果就不会坚固。}%

@{貪心與愛心，分別本不二，}@
%{贪心和爱心，本质上是一样的。}%

@{造惡俱苦受，云何不生悔？}@
%{造恶都会带来痛苦，为什么不悔改呢？}%

@{貪性初為種，愛性受胞胎，}@
%{贪心最初是种子，爱则是受孕的胎儿。}%

@{有情戀不息，往來難出離。}@
%{众生对爱恋永不停止，来来往往，难以解脱。}%

@{諸天及人民，依愛而止住，}@
%{天人和人，都依靠爱而停留。}%

@{愛往眾結隨，剎那亦不停，}@
%{爱随着各种各样的结缘，一刻不停。}%

@{時過復生憂，入獄方自覺。}@
%{时间过去，又会产生忧愁，进了监狱才醒悟。}%

@{緣流愛不住，欲網覆瘡根，}@

@{善聞從慧得，得脫一切縛。}@
%{从智慧中听到佛法，就能摆脱一切束缚。}%

@{信之與戒法，慧意則能行，}@
%{有了信仰和戒律，智慧才能发挥作用，}%

@{健夫度恚怒，從是得脫淵。}@
%{强壮的人能战胜愤怒，从此就能摆脱深渊。}%

@{信使戒成就，亦獲壽及慧，}@
%{信仰能使戒律圆满，也能获得长寿和智慧，}%

@{在在則能行，處處見供養。}@
%{无论在哪里都能修行，处处都能受到供养。}%

@{施共與鬪集，此業智不處，}@
%{布施和争斗，这两种行为智慧不会停留，}%

@{施時非鬪時，速施何疑慮？}@
%{布施的时候不是争斗的时候，赶快布施，还有什么疑虑呢？}%

@{此方出世利，慧信為智母，}@
%{这是出离轮回的利益，智慧和信仰是智慧的母亲，}%

@{是財出世寶，家產則非常。}@
%{这是出离轮回的宝藏，是永恒的财富。}%

@{欲見諸真者，樂聽聞法教，}@
%{想要见到诸佛菩萨，就应该乐于听闻佛法，}%

@{能捨慳垢心，此乃為上信。}@
%{能够舍弃吝啬贪婪的心，这就是最高的信仰。}%

@{信能渡有河，其福難侵奪，}@
%{信仰能渡过生死之河，它的福报难以被夺取，}%

@{能禁止竊盜，閑靜沙門樂。}@
%{能禁止偷盗，清净的沙门会感到快乐。}%

@{沙門恒來至，智者所見樂，}@
%{沙门经常来到这里，智者看到他们会感到快乐，}%

@{及餘篤信者，聞則生歡喜。}@
%{以及其他虔诚的信徒，听到佛法就会生起欢喜心。}%

@{若人懷懊惱，貪他人衣食，}@
%{如果有人心中充满烦恼，贪图别人的衣食，}%

@{彼人晝夜寐，不獲三摩地。}@
%{这个人昼夜沉睡，得不到三摩地。}%

@{若人能斷貪，如截多羅樹，}@
%{如果有人能够断除贪心，就像砍断多罗树，}%

@{彼人則晝夜，及獲三摩地。}@
%{这个人就能昼夜安住于三摩地。}%

@{無信不修行，好剝正言說，}@
%{没有信仰的人不修行，喜欢歪曲佛法，}%

@{如掘取清泉，掘泉揚其泥。}@
%{就像挖取清泉，却把泥土扬起来。}%

@{智者習信行，樂仰清淨流，}@
%{智者学习信仰，乐于仰望清净的佛法之流，}%

@{如善取泉水，思冷不擾濁。}@
%{就像善于取用泉水，想到清凉，不扰动浑浊。}%

@{信智不染他，惟智與賢仁，}@
%{信仰和智慧不受外物污染，只有智慧和贤德的人，}%

@{非好則遠之，可好則近學。}@
%{不好的就远离，好的就靠近学习。}%

@{樂信與不樂，寂默自應思，}@
%{喜欢信仰和不喜欢信仰，应该静默地思考，}%

@{遠離無信者，信仁應行之。}@
%{远离没有信仰的人，信仰仁义应该去实行。}%

@{法集要頌經沙門品第十一}@
%{法集要颂经沙门品第十一}%

@{斷漏降伏他，離欲名梵行，}@
%{断除烦恼，降伏他人，远离欲望，叫做梵行，}%

@{不犯牟尼戒，無一願不滿。}@
%{不违犯佛陀的戒律，没有一个愿望不能满足。}%

@{行力若緩慢，作善與不善，}@
%{修行懈怠缓慢，做善事和恶事，}%

@{梵行不清淨，不獲於大果。}@
%{梵行不纯净，得不到大果。}%

@{所有緩慢業，劣意盡除之，}@
%{所有懈怠的行为，都要彻底消除掉，}%

@{修習清淨行，獲果盡無餘。}@
%{修行清净的行为，就能获得圆满的果报。}%

@{譬如執利劍，執緩則傷手，}@
%{比如拿着锋利的宝剑，拿着松懈就会伤手，}%

@{沙門不禁制，地獄縛牽引。}@
%{沙门不约束自己，就会被地狱的业力牵引。}%

@{又如執利劍，執緊不傷手，}@
%{又比如拿着锋利的宝剑，握紧了就不会伤手，}%

@{沙門禁制戒，漸近涅槃路。}@
%{沙门约束自己遵守戒律，就能逐渐接近涅槃之路。}%

@{難曉則難了，沙門少智慧，}@
%{难以理解就难以明白，沙门智慧不够，}%

@{諸想多擾亂，愚者致苦惱。}@
%{各种念头扰乱心神，愚蠢的人就会带来痛苦。}%

@{沙門為何行？如意不自禁，}@
%{沙门为什么要修行？随心所欲，不能控制自己，}%

@{步步數黏著，但隨思想走。}@
%{一步一步地被念头黏住，只跟着思想走。}%

@{學難捨罪難，居在家亦難，}@
%{学习很难，舍弃罪业也很难，住在家里也很难，}%

@{會止同利難，艱難不過是。}@
%{聚会止息，利益相同也很难，世界上最难的事不过如此。}%

@{袈裟在肩披，為惡不捐棄，}@
%{穿着袈裟，却不做恶事，}%

@{常念行惡者，斯則墮惡道。}@
%{经常想着做恶事的人，就会堕落到恶道。}%

@{畏罪懷驚懼，假名為沙門，}@
%{害怕罪业，心中充满恐惧，假装自己是沙门，}%

@{身披僧伽胝，如刳娑羅皮。}@
%{身上穿着僧伽梨，就像剥了娑罗树皮一样。}%

@{所謂長老者，不必以耆年，}@
%{所谓长者，不一定要年老，}%

@{形熟鬢髮白，愚憃不知罪。}@
%{身体衰老，头发变白，却愚蠢糊涂，不知道罪过。}%

@{能知罪福者，身淨修梵行，}@
%{能知道罪业和福报的人，身心清净，修行梵行，}%

@{明遠純清潔，是名為長老。}@
%{明理通达，纯洁清净，这才是真正的长者。}%

@{所謂沙門者，不必剃鬚髮，}@
%{所谓沙门，不一定要剃除胡须头发，}%

@{妄語多貪愛，有欲如凡夫。}@
%{说谎，贪图欲望，就像普通人一样。}%

@{世稱名沙門，汝亦言沙門，}@
%{世俗称他为沙门，你也说自己是沙门，}%

@{形服似沙門，譬如鶴伺魚。}@
%{外表和穿着像沙门，就像鹤等待着鱼一样。}%

@{如離實不離，袈裟除不除，}@
%{就像离开实质，不离开，袈裟去掉也不去掉，}%

@{持鉢實不持，非俗非沙門。}@
%{拿着钵盂，实际上不拿着，既不是俗人，也不是沙门。}%

@{所言沙門者，消除窣兔羅，}@
%{所谓沙门，要消除一切烦恼，}%

@{守護微細愆，是名真梵行。}@
%{守护细微的过失，这才是真正的梵行。}%

@{所言沙門者，息心滅意想，}@
%{所谓沙门，要熄灭心念，消除妄想，}%

@{穢垢盡消除，故說為出家。}@
%{一切污垢都消除掉，所以被称为出家。}%
