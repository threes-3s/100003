@{無信懷憎嫉，鬪亂彼此人，}@

@{智者所棄嫌，愚習以為樂。}@
%{智者所厌弃，愚人却以此为乐。}%

@{有信無憎嫉，精進信多聞，}@

@{不親惡知識，不與非法會，}@
%{不亲近恶知识，不参加非法集会，}%

@{親近善知識，恒與正法會。}@
%{亲近善知识，经常参加正法集会。}%

@{行路念防慮，持戒多聞人，}@
%{行路时要多加防范，持戒多闻的人，}%

@{思慮無量境，聞彼善言教。}@
%{要思虑无量法界，听闻他们的善言教诲。}%

@{各各知差別，近惡自陷溺，}@
%{各自知道差别，亲近恶人会自己陷入困境，}%

@{習善致名稱。妙者恒自妙，}@
%{学习善法才能获得好名声。妙法永远是妙法，}%

@{此由身真正；善者終以善，}@
%{这是因为自身真实不虚；善法最终以善法成就，}%

@{斯由親近善。智慧為最上，}@
%{这是因为亲近善法。智慧是最上乘，}%

@{持戒永寂滅。}@
%{持戒才能永远寂灭。}%
@{如魚湊臭爛，人貪競取之，}@
%{就像鱼儿喜欢聚集在臭烂的东西旁边，人贪婪地争着去取，}%
@{意者不覺臭，習惡亦如是。}@
%{他们自己却意识不到臭味，习染恶习也是如此。}%

@{多誐波羅葉，眾生往採取，}@
%{人们争着去采摘波罗蜜树叶，}%
@{葉薰香遠布，習善亦如是。}@
%{叶子散发出香气，传布很远，习染善法也是如此。}%

@{親近惡知識，罪垢日夜增，}@
%{亲近恶知识，罪过日夜增加，}%

%{就像猪的身子不干净，自己污秽，还污染别人。}%

@{己自不習惡，親近習惡者，}@
%{自己不习染恶习，却亲近习染恶习的人，}%
@{為人所輕笑，惡名日夜熾。}@
%{会被别人嘲笑，恶名日夜增长。}%

@{觀習而習之，知近而親近，}@
%{看到别人习染恶习，自己也去习染，知道恶人，就亲近恶人，}%
@{毒箭在其束，淨者被其污。}@
%{就像毒箭藏在箭袋里，干净的人靠近了，也会被污染。}%

@{勇夫能除污，去惡不為伴，}@
%{勇敢的人能除掉污垢，远离恶人，不做朋友，}%
@{是故知果報，智人悉分別。}@
%{所以，知道因果报应的人，都会明辨是非。}%

@{非親慎莫習，習當近於賢，}@
%{不要亲近不贤的人，学习要亲近贤人，}%
@{苾芻修行道，忍苦盡諸漏。}@
%{比丘修行佛法，要忍耐痛苦，断除一切烦恼。}%

%{愚人一辈子侍奉明智的人，}%

@{法集要頌經圓寂品第二十六}@
%{法集要颂经圆寂品第二十六}%

@{如龜藏其六，苾芻攝意想，}@
%{就像乌龟把六根藏起来一样，比丘要摄住自己的意念，}%
@{無倚無害彼，圓寂無言說。}@
%{不依靠任何东西，不伤害任何人，这就是圆寂，不用言语来表达。}%

@{忍辱第一道，佛說圓寂最，}@
%{忍辱是第一条修行之道，佛陀说圆寂是最好的，}%
@{不以懷煩熱，害彼為沙門。}@
%{不要怀着烦恼，伤害别人，成为沙门。}%

@{言當莫麁獷，所說應辯才，}@
%{说话不要粗鲁，所说的话应该有智慧，}%
@{少聞共論難，反受彼屈伏。}@
%{不要和少闻的人辩论，反而会被他们压制。}%

@{數自興煩惱，猶彼器敗壞，}@
%{经常自己生烦恼，就像器皿破损一样，}%
@{生死數流轉，長沒無出期。}@
%{生死轮回不断，永远沉沦，没有出头之日。}%

@{若不自煩惱，猶器完牢具，}@
%{如果自己不生烦恼，就像器皿完好坚固一样，}%
@{如是至圓寂，永無諸塵翳。}@
%{这样才能达到圆寂，永远没有尘埃遮蔽。}%

@{無病第一利，知足第一富，}@
%{没有病痛是最大的利益，知足是最大的财富，}%
@{知親第一友，圓寂第一樂。}@
%{了解亲人是最真挚的朋友，圆寂是最大的快乐。}%

@{飢為第一患，行為第一苦，}@
%{饥饿是最痛苦的，行为是最大的苦恼，}%
@{如實知此者，圓寂第一樂。}@
%{如实知道这些的人，圆寂是最大的快乐。}%

@{最妙聖言教，流布無窮際，}@
%{最妙的圣言教诲，流布到无穷无尽的境界，}%
@{世共傳習者，實無有厭時。}@
%{世间共同学习的人，永远不会厌倦。}%

@{如是無等倫，所說善言教，}@
%{像这样无与伦比的，佛陀所说善妙的教诲，}%
@{身苦所逼迫，何過飢患苦？}@
%{身体的痛苦，怎么能比得上饥饿和患病的痛苦呢？}%

@{趣善之徒少，趣惡之徒多，}@
%{追求善法的人很少，追求恶法的人很多，}%
@{如實知此者，速求於圓寂。}@
%{如实知道这些的人，要尽快寻求圆寂。}%

@{有緣生善處，有緣生惡趣，}@
%{有缘的人，会生到善道，有缘的人，会生到恶道，}%
@{有緣般涅槃，如斯皆有緣。}@
%{有缘的人，会证得涅槃，这一切都是有缘的。}%

@{鹿歸於田野，鳥歸於虛空，}@
%{鹿归于田野，鸟归于虚空，}%
@{義歸於分別，真人歸寂滅。}@
%{道理归于分别，真人归于寂灭。}%

@{不以懈怠意，怯弱有所至，}@
%{不要因为懈怠，胆怯而有所退却，}%
@{欲求於圓寂，焚燒諸縛著。}@
%{想要寻求圆寂，就要烧毁一切束缚。}%

@{苾芻速杼船，以杼便當輕，}@
%{比丘要快速地撑船，撑船的工具要轻便，}%
@{求斷貪欲情，然後至圓寂。}@
%{要断除贪欲的念头，才能达到圆寂。}%

@{我有本以無，本有我今無，}@
%{我本来是空的，本来有的我，现在也不存在了，}%
@{非無亦非有，如今不可獲。}@
%{既不是空，也不是有，现在已经无法获得。}%

@{難見諦不動，善觀而不動，}@
%{很难见到真理而不动摇，要善于观察，也不要动摇，}%
@{當察愛盡源，是謂名業際。}@
%{要观察爱的根源，这就是业际。}%

@{斷愛除其欲，竭河無流兆，}@
%{断除爱欲，就像干涸的河流，没有水流的迹象，}%
@{能明此愛本，是謂名苦際。}@
%{能够明了爱的本源，这就是苦际。}%

@{無身滅其想，諸痛得清涼，}@
%{没有身体，就断除对身体的执着，一切痛苦都会得到清凉，}%

%{观察清净而使自己清净。如果五根不具足，}%
@{於食無厭足，斯等凡品行，}@
%{在饮食上不知满足，这些凡夫俗子的行为，}%

@{轉增於欲意。如屋壞穿漏，}@
%{只会增加欲望。就像房屋破损漏雨，}%
@{當觀不淨行，諸根無缺漏，}@
%{应该观察不净的行为，五根没有缺漏，}%
@{於食知止足。有信執精進，}@
%{在饮食上懂得知足。有信心，坚持精进，}%
@{不恣於欲意，如風吹泰山。}@
%{不要放纵欲望，就像风吹泰山。}%

@{空閑甚可樂，然人不樂彼，}@
%{空闲非常快乐，然而人们不喜欢它，}%
@{無欲常居之，非欲之所處。}@
%{没有欲望才能长久地住在这里，不是欲望所安住的地方。}%

@{難移難可動，如彼重雪山，}@
%{很难移动，很难撼动，就像那座高耸的雪山，}%
@{非賢則不現，猶夜射冥室。}@
%{不是贤者就不会出现，就像在夜晚射向黑暗的房间。}%

@{賢者有千數，智者在叢林，}@
%{贤者有千人之多，智者在丛林之中，}%
@{義理極深邃，智者所分別。}@
%{义理极其深奥，是智者所分别的。}%

@{多有眾生類，非射而不值，}@
%{很多众生，不射箭就射不中，}%
@{今觀此義理，無戒人所恥。}@
%{现在观察这些义理，没有戒律的人所耻笑。}%

@{觀有知恐怖，變易知有無，}@
%{观察有，知道恐怖，变化无常，知道有和无，}%
@{是故不樂有，當念遠離有。}@
%{所以不喜欢有，应该念念不忘远离有。}%

@{無信無反復，穿牆而盜竊，}@
%{没有信心，没有反复，穿墙去偷窃，}%
@{斷彼希望思，是名為勇士。}@
%{断除那些希望和思虑，这叫做勇士。}%

@{除其父母緣，王家及二種，}@
%{除掉父母的缘分，王室和两种，}%
@{徧滅其境界，無垢為梵行。}@
%{遍灭一切境界，无垢就是梵行。}%

@{若人無所依，知彼所貴食，}@
%{如果人没有依靠，知道他们所珍贵的食物，}%
@{空及無相願，思惟以為行。}@
%{空和无相的愿望，思惟并以此作为修行。}%

@{鳥飛於虛空，而無足跡現，}@
%{鸟儿飞翔在虚空，却没有留下足迹，}%
@{如彼行行人，言說無所趣。}@
%{就像那些行走的人，言语没有所趣。}%

@{諸能斷有本，不依於未然，}@
%{那些能断除有根的人，不依靠于未来，}%
@{空及無相願，思惟以為行。}@
%{空和无相的愿望，思惟并以此作为修行。}%

@{希有諸眾生，多不順其性，}@
%{稀有的众生，大多不顺从自己的本性，}%
@{有度不度者，為滅甚為難。}@
%{有的度化，有的不度化，要灭度他们非常困难。}%

@{諸有平等說，法法共相觀，}@
%{一切法平等的说法，法法相互观察，}%
@{盡斷諸結使，無復有熱惱。}@
%{彻底断除一切烦恼，不再有热恼。}%

@{行路無復憂，終日得解脫，}@
%{行走不再忧虑，终日得解脱，}%
@{一切結使盡，無復有眾惱。}@
%{一切烦恼都断除了，不再有众恼。}%

@{如鳥飛虛空，而無有所礙，}@
%{就像鸟儿飞翔在虚空，没有受到任何阻碍，}%
@{彼人獲無漏，空無相願定。}@
%{那个人获得无漏，空无相的愿望禅定。}%

@{如鳥飛虛空，而無有所礙，}@
%{就像鸟儿飞翔在虚空，没有受到任何阻碍，}%
@{行人到彼岸，空無相願定。}@
%{行者到达彼岸，空无相的愿望禅定。}%

@{無造無有造，造者受煩惱，}@
%{不造，也没有造，造的人会受到烦恼，}%
@{非造非無造，前憂後亦憂。}@
%{不是造，也不是不造，前面忧愁，后面也忧愁。}%

@{造者為善妙，以作不懷憂，}@
%{造的人是善妙的，做事不怀忧愁，}%
@{造而樂而造，生天受歡樂。}@
%{造作而快乐地造作，生到天上享受快乐。}%

@{虛空無轍迹，沙門無外意，}@
%{虚空没有车辙，沙门没有外意，}%
@{眾人盡樂惡，唯佛淨無穢。}@
%{众人都喜欢恶，只有佛是清净无染的。}%

@{虛空無轍迹，沙門無外意，}@
%{虚空没有车辙，沙门没有外意，}%
@{世間皆無常，佛無我所有。}@
%{世间一切都是无常的，佛没有我所拥有。}%

@{諸天及世人，一切行相應，}@
%{诸天和世间人，一切行为相应该，}%
@{得脫一切苦，離愛免輪迴。}@
%{就能摆脱一切苦，远离爱，免除轮回。}%

@{諸天及世人，一切行相應，}@
%{诸天和世间人，一切行为相应该，}%
@{能遠諸惡業，不墮於惡趣。}@
%{就能远离一切恶业，不堕落到恶道。}%

@{亦復不知論，賢愚無差別，}@
%{也同样不知道论，贤愚没有差别，}%
@{若復知論義，所說無垢跡。}@
%{如果知道论的义理，所说的话语就无垢无染。}%

@{說應法議論，當豎仙人幢，}@
%{说应该法的议论，应该竖立仙人幢，}%
@{法幢為仙人，仙人為法幢。}@
%{法幢就是仙人，仙人就是法幢。}%

@{或有寂然罵，或有在眾罵，}@
%{有的默默地骂，有的在众人面前骂，}%
@{或有未聲罵，世無不罵者。}@
%{有的没有声音地骂，世间没有不骂的人。}%

@{一毀及一譽，但利養其名，}@
%{一毁坏和一赞誉，只是为了利益和养活自己的名声，}%
@{非有亦非有，則亦不可知。}@
%{不是有，也不是没有，那么也无法知道。}%

@{智人所稱譽，若好兼及醜，}@
%{智者所称赞的，如果好的和丑的都有，}%
@{智人無缺漏，慧定得解脫。}@
%{智者没有缺漏，智慧禅定才能解脱。}%

@{如紫磨真金，內外徹清淨，}@
%{就像紫磨真金，内外彻底清净，}%
@{猶如安明山，不為風所動，}@
%{就像安稳的明山，不会被风所动，}%
@{智人亦如是，不為毀譽動。}@
%{智者也是这样，不会被毁誉所动。}%

@{如樹無有根，無枝況有葉？}@
%{就像树没有根，没有枝，何况有叶子？}%
@{健者以解縛，誰能毀其德？}@
%{健壮的人已经解除了束缚，谁能毁坏他的德行？}%

@{無垢無有住，身壍種苦子，}@
%{无垢没有安住，身体是苦的种子，}%
@{最勝無有愛，天世人不知。}@
%{最胜的没有爱，天人和世间人不知道。}%

@{猶如網叢林，無愛況有餘？}@
%{就像网状的丛林，没有爱，何况还有剩余？}%

@{佛有無量行，無跡誰跡將？}@
%{佛有无量行，没有足迹，谁的足迹能比得上？}%

@{若有不欲生，以生不受有，}@
%{如果有人不想出生，就以出生来不受有，}%
@{佛有無量行，無跡誰跡將？}@
%{佛有无量行，没有足迹，谁的足迹能比得上？}%

@{若欲滅其想，內外無諸因，}@
%{如果想要灭除它的想法，内外没有一切因缘，}%
@{亦無過色想，四應不受生。}@
%{也无超过色相的想法，四种应缘就不会出生。}%

@{捨前及捨後，捨間越於有，}@
%{舍弃前面，舍弃后面，舍弃中间，超越于有，}%
@{一切盡皆捨，不復受生老。}@
%{一切全部舍弃，不再受生老病死。}%

@{法集要頌經卷第三}@
%{法集要颂经卷第三}%
