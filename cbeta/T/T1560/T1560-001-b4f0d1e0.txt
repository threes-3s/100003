@{阿毘達磨俱舍論本頌(說一切有部)}@
%{阿毘達磨俱舍論本頌(說一切有部)}%

@{世親菩薩造}@
%{世親菩薩造}%

@{諸一切種諸冥滅，拔眾生出生死泥，}@
%{所有一切众生，都处在黑暗的轮回之中，被生死苦海所困，}%
@{敬禮如是如理師，對法藏論我當說。}@
%{我敬礼那些能够引导众生脱离苦海的明师，现在我要讲解《对法藏论》。}%

@{淨慧隨行名對法，及能得此諸慧論，}@
%{能够消除烦恼、获得智慧的修行方法，被称为“对法”，}%
@{攝彼勝義依彼故，此立對法俱舍名。}@
%{而能够阐述这些智慧的理论，则被称为“俱舍”。}%

@{若離擇法定無餘，能滅諸惑勝方便，}@
%{如果脱离了选择正确的修行方法，就无法彻底消除烦恼，}%
@{由惑世間漂有海，因此傳佛說對法。}@
%{因为众生被烦恼所迷惑，如同漂泊在苦海之中，所以佛陀才传授了“对法”。}%

@{有漏無漏法，除道餘有為，}@
%{除了涅槃之道以外，所有有为法都可以分为有漏法和无漏法。}%
@{於彼漏隨增，故說名有漏。}@
%{有漏法是指伴随着烦恼而增长的法，所以被称为有漏法。}%

@{無漏謂道諦，及三種無為，}@
%{无漏法是指涅槃之道，以及三种无为法。}%
@{謂虛空二滅，此中空無礙。}@
%{这三种无为法分别是虚空、二灭，其中虚空是无障碍的。}%

@{擇滅謂離繫，隨繫事各別，}@
%{二灭是指脱离了烦恼的束缚，而随烦恼的束缚，事物就有所差别。}%
@{畢竟礙當生，別得非擇滅。}@
%{因为最终还是会产生障碍，所以这种差别得不到解脱，并非真正的二灭。}%

@{又諸有為法，謂色等五蘊，}@
%{所有有为法，可以分为色、受、想、行、识五蕴。}%
@{亦世路言依，有離有事等。}@
%{在世间的说法中，有“有”和“无”的区别。}%

@{有漏名取蘊，亦說為有諍，}@
%{有漏法被称为“取蕴”，也称为“有争”。}%
@{及苦集世間，見處三有等。}@
%{它包括了苦、集、世间、见处、三有等等。}%

@{色者唯五根，五境及無表，}@
%{色蕴指的是五根、五境以及无表色。}%
@{彼識依淨色，名眼等五根。}@
%{眼识依附于净色，被称为眼根等等五根。}%

@{色二或二十，聲唯有八種，}@
%{色蕴有两种或者二十种，声蕴只有八种。}%
@{味六香四種，觸十一為性。}@
%{味蕴有六种，香蕴有四种，触蕴的本质是十一。}%

@{亂心無心等，隨流淨不淨，}@
%{无表色是指乱心、无心等等，随着心念的流动，有清净和不清净两种。}%
@{大種所造性，由此說無表。}@
%{无表色是由四大所造，所以被称为无表色。}%

@{大種謂四界，即地水火風，}@
%{四大是指地、水、火、风四界。}%
@{能成持等業，堅濕煖動性。}@
%{四大分别具有成就、持、等业，以及坚、湿、暖、动的特性。}%

@{地謂顯形色，隨世想立名，}@
%{地界是指显现形色的，根据世俗的观念而命名。}%
@{水火亦復然，風即界亦爾。}@
%{水界和火界也一样，风界就是指风。}%

@{此中根與境，即說十處界，}@
%{五根和五境，被称为十处界。}%
@{受領納隨觸，想取像為體。}@
%{受蕴是领纳，随触而生，想蕴是取像，以像为体。}%

@{四餘名行蘊，如是受等三，}@
%{剩下的四种被称为行蕴，包括受、想、以及无表色和无为法。}%
@{及無表無為，名法處法界。}@
%{无表色和无为法，被称为法处和法界。}%

@{識謂各了別，此即名意處，}@
%{识蕴是指分别了知，被称为意处。}%
@{及七界應知，六識轉為意。}@
%{应该知道，六识转变为意，包括七界。}%

@{由即六識身，無間滅為意，}@
%{由于六识的无间灭，就成为意。}%
@{成第六依故，十八界應知。}@
%{因为六识成为第六识的依处，所以应该知道十八界。}%

@{總攝一切法，由一蘊處界，}@
%{所有法都可以归纳到一个蕴、处、界中。}%
@{攝自性非餘，以離他性故。}@
%{因为它们具有自性，而没有其他的性。}%

@{類境識同故，雖二界體一，}@
%{因为它们的类别、境界、识都相同，虽然两个界体的本质是相同的，}%
@{然為令端嚴，眼等各生二。}@
%{但为了使之更加清晰，眼等等分别产生了两个。}%

@{聚生門種族，是蘊處界義，}@
%{聚生门、种族，就是蕴、处、界的含义。}%
@{愚根樂三故，說蘊處界三。}@
%{因为愚痴的人喜欢三，所以说蕴、处、界有三种。}%

@{諍根生死因，及次第因故，}@
%{因为争执的根源是生死，以及次第的因缘，}%
@{於諸心所法，受想別為蘊。}@
%{所以将所有心所法中的受、想分别作为蕴。}%

@{蘊不攝無為，義不相應故，}@
%{因为蕴不包含无为法，两者意义不相应，}%
@{隨麁染器等，界別次第立。}@
%{所以根据粗、染、器等等，界就有了不同的次第。}%

@{前五境唯現，四境唯所造，}@
%{前五境只存在于现在，而四境只存在于所造。}%
@{餘用遠速明，或隨處次第。}@
%{其他的则根据用途、速度、明暗，或者根据处所的次第。}%

@{為差別最勝，攝多增上法，}@
%{为了区分得更加清晰，将许多增上法归纳到一起，}%
@{故一處名色，一名為法處。}@
%{所以一个处所被称为色处，一个处所被称为法处。}%

@{牟尼說法蘊，數有八十千，}@
%{佛陀说法蕴的数量有八十千，}%
@{彼體語或名，此色行蘊攝。}@
%{它们包括了体、语、名等等，都被色蕴和行蕴所摄。}%

@{有言諸法蘊，量如彼論說，}@
%{有人说所有法蕴的数量如佛陀所说，}%
@{或隨蘊等言，如實行對治。}@
%{或者根据蕴等等，如实地进行对治。}%

@{如是餘蘊等，各隨其所應，}@
%{像这样，其他的蕴等等，都应该根据各自的类别，}%
@{攝在前說中，應審觀自相。}@
%{归纳到前面所说的内容中，应该仔细观察它们的本质。}%

@{空界謂竅隙，傳說是明闇，}@
%{空界是指空隙，传说它有光明和黑暗两种。}%
@{識界有漏識，有情生所依。}@
%{识界是指有漏识，是众生存在的根本。}%

@{一有見謂色，十有色有對，}@
%{一有的见是指色，十有的色是指对。}%
@{此除色聲八，無記餘三種。}@
%{除了色、声八种之外，还有三种无记。}%

@{欲界繫十八，色界繫十四，}@
%{欲界有十八界，色界有十四界。}%
@{除香味二識，無色繫後三。}@
%{除了香味二识之外，无色界还有后面的三界。}%

@{意法意識通，所餘唯有漏，}@
%{意、法、意识是共通的，其他的都是有漏的。}%
@{五識唯尋伺，後三三餘無。}@
%{五识只有寻伺，后面的三种则没有。}%

@{說五無分別，由計度隨念，}@
%{五无分别是由于计度、随念而产生的，}%
@{以意地散慧，意諸念為體。}@
%{以意地、散慧，以及意念作为它的本质。}%

@{七心法界半，有所緣餘無，}@
%{七心法界中，有一半是有所缘的，另一半则没有。}%
@{前八界及聲，無執受餘二。}@
%{前八界和声，没有执受，其他的还有两种。}%

@{觸界中有二，餘九色所造，}@
%{触界中有两种，其他的九种都是由色所造。}%
@{法一分亦然，十色可積集。}@
%{法界也一样，十种色可以积聚。}%

@{謂唯外四界，能斫及所斫，}@
%{只有外四界，可以砍伐和被砍伐，}%
%{亦所燒能稱，能燒所稱諍。}%
%{也可以燃烧和被燃烧，以及燃烧和被燃烧的争执。}%

@{內五有熟養，聲無異熟生，}@
%{内五蕴有熟养，声蕴没有异熟生，}%
@{八無礙等流，亦異熟生性。}@
%{八无碍等流，也是异熟生的性质。}%

@{餘三實唯法，剎那唯後三，}@
%{剩下的三种，实际上只有法，只有后面的三种是刹那的。}%
@{眼與眼識界，獨俱得非等。}@
%{眼界和眼识界，是独俱得，而不是等无间。}%

@{內十二眼等，色等六為外，}@
%{内十二界是眼等等，色等等六界是外。}%
@{法同分餘二，作不作自業。}@
%{法界和同分界还有两种，分别是作业和不作业。}%

@{十五唯修斷，後三界通三，}@
%{前十五界只有修断，后面的三界则三种都通。}%
@{不染非六生，色定非見斷。}@
%{不染是指六道生死，色定是指见断。}%

@{眼法界一分，八種說名見，}@
%{眼法界中的一部分，被称为八种见。}%
@{五識俱生慧，非見不度故。}@
%{五识俱生的智慧，不是见，因为不能度脱。}%

@{眼見色同分，非彼能依識，}@
%{眼见色是同分，而不是能够依附于识。}%
@{傳說不能觀，被障諸色故。}@
%{传说不能观察，是因为被障碍了。}%

@{或二眼俱時，見色分明故，}@
%{或者两只眼睛同时看到，所以能够清晰地看到色。}%
@{眼耳意根境，不至三相違。}@
%{眼、耳、意根的境界，不会违背三相。}%

@{應知鼻等三，唯取等量境，}@
%{应该知道，鼻等等三根，只取等量的境界。}%
@{後依唯過去，五識依或俱。}@
%{后面的依处只有过去，五识的依处可能是过去的，也可能是现在的。}%

@{隨根變識異，故眼等名依，}@
%{随着根的变化，识也随之变化，所以眼等等被称为依处。}%
@{彼及不共因，故隨根說識。}@
%{因为它们和不共因，所以根据根来说识。}%

@{眼不下於身，色識非上眼，}@
%{眼不低于身，色识不高于眼。}%
@{色於識一切，二於身亦然。}@
%{色对识来说是所有，而对身来说也是所有。}%

@{如眼耳亦然，次三皆自地，}@
%{像眼、耳一样，后面的三根都是来自自身。}%
@{身識自下地，意不定應知。}@
%{身识来自下地，意是不定的，应该知道。}%

@{五外二所識，常法界無為，}@
%{五外、二所识、常、法界、无为法，}%
@{法一分是根，并內界十二。}@
%{法界中的一部分是根，以及内界十二。}%
@{傳說五於四，四根於二種，}@
%{传说的五根归于四根，四根归于两种。}%
@{五八染淨中，各別為增上。}@
%{这五根和八根，在染污和清净中，分别为增上。}%
@{了自境增上，總立於六根，}@
%{了知自己和境地的增上，总立于六根。}%
@{從身立二根，女男性增上。}@
%{从身立二根，男女性的增上。}%
@{於同住雜染，清淨增上故，}@
%{在同住杂染中，清净的增上，所以}%
@{應知命五受，信等立為根。}@
%{应该知道命根、五受根、信根等，立为根。}%
@{未當知已知，具知根亦爾，}@
%{未曾知道而已经知道，具知根也是如此。}%
@{於得後後道，涅槃等增上。}@
%{在得到之后，后道、涅槃等增上。}%
@{心所依此別，此住此雜染，}@
%{心所依此分别，此住此杂染。}%
@{此資糧此淨，由此量立根。}@
%{此资粮此清净，由此量立根。}%
@{或流轉所依，及生住受用，}@
%{或者流转所依，以及生住受用。}%
@{建立前十四，還滅後亦然。}@
%{建立前十四根，还灭之后也是如此。}%
@{身不悅名苦，即此悅名樂，}@
%{身不悦，名为苦，即此悦，名为乐。}%
@{及三定心悅，餘處此名喜。}@
%{以及三定心悦，其余地方，此名为喜。}%
@{心不悅名憂，中捨二無別，}@
%{心不悦，名为忧，中舍二种，没有区别。}%
@{見修無學道，依九立三根。}@
%{见修无学道，依九种立三根。}%
@{唯無漏後三，有色命憂苦，}@
%{唯有无漏的后三种，有色根、命根、忧根、苦根。}%
@{當知唯有漏，通二餘九根。}@
%{应该知道唯有漏根，通于二种，其余九种根。}%
@{命唯是異熟，憂及後八非，}@
%{命根唯是异熟根，忧根及后八根不是。}%
@{色意餘四受，一一皆通二。}@
%{色根、意根、其余四种受根，一一都通于二种。}%
@{憂定有異熟，前八後三無，}@
%{忧根、定根有异熟根，前八根、后三根没有。}%
@{意餘受信等，一一皆通二。}@
%{意根、其余根、信根等，一一都通于二种。}%
@{唯善後八根，憂通善不善，}@
%{唯有善根的后八根，忧根通于善根和不善根。}%
@{意餘受三種，前八唯無記。}@
%{意根、其余受根三种，前八根唯是无记根。}%
@{欲色無色繫，如次除後三，}@
%{欲界、色界、无色界，依次除去后三种。}%
@{兼女男憂苦，并餘色喜樂。}@
%{兼有男女根、忧根、苦根，并有其余色根、喜根、乐根。}%
@{意三受通三，憂見修所斷，}@
%{意根、三受根通于三种，忧根、见修根、所断根。}%
@{九唯修所斷，五修非三非。}@
%{九种根唯是修所断根，五种根既不是修所断根，也不是三种根。}%
@{欲胎卵濕生，初得二異熟，}@
%{欲界、胎生、卵生、湿生，最初得到两种异熟根。}%
@{化生六七八，色六上唯命。}@
%{化生得到六根、七根、八根，色界六根以上，唯是命根。}%
@{正死滅諸根，無色三色八，}@
%{正死灭诸根，无色界三根、色界八根。}%
@{欲頓十九八，漸四善增五。}@
%{欲界顿得到十九根、八根，渐渐得到四善根，增加五根。}%
@{九得邊二果，七八九中二，}@
%{九根得到边二果，七根、八根、九根中得到二果。}%
@{十一阿羅漢，依一容有說。}@
%{十一根阿罗汉，依一根，可以有说。}%
@{成就命意捨，各定成就三，}@
%{成就命根、意根、舍根，各定成就三根。}%
@{若成就樂身，各定成就四，}@
%{若成就乐根、身根，各定成就四根。}%
@{成眼等及喜，各定成五根，}@
%{成就眼根等及喜根，各定成就五根。}%
@{若成就苦根，彼定成就七，}@
%{若成就苦根，彼定成就七根。}%
@{若成女男憂，信等各成八。}@
%{若成就男女根、忧根，信根等各成就八根。}%
@{二無漏十一，初無漏十三，}@
%{二无漏根十一根，初无漏根十三根。}%
@{極少八無善，成受身命意。}@
%{最少八根无善根，成就受根、身根、命根、意根。}%
@{愚生無色界，成善命意捨，}@
%{愚痴生于无色界，成就善根、命根、意根、舍根。}%
@{極多成十九，二形除三淨，}@
%{最多成就十九根，二形根除去三净根。}%
@{聖者未離欲，除二淨一形。}@
%{圣者未曾离开欲界，除去二净根，一形根。}%
@{欲微聚無聲，無根有八事，}@
%{欲界微细聚合，无声，没有根，有八事。}%
@{有身根九事，十事有餘根。}@
%{有身根九事，十事有其余根。}%
@{心心所必俱，諸行相或得，}@
%{心心所必俱，诸行相或得。}%
@{心所且有五，大地法等異。}@
%{心所且有五种，大地法等不同。}%
@{受想思觸欲，慧念與作意，}@
%{受想思触欲，慧念与作意。}%
@{勝解三摩地，遍於一切心。}@
%{胜解三摩地，遍于一切心。}%
@{信及不放逸，輕安捨慚愧，}@
%{信根及不放逸根，轻安根、舍根、惭愧根。}%
@{二根及不害，勤唯遍善心。}@
%{二根及不害根，勤根唯遍于善心。}%
@{癡逸怠不信，惛掉恒唯染，}@
%{痴根、逸根、怠根、不信根，惛掉根恒唯染污。}%
@{唯遍不善心。無慚及無愧，}@
%{唯遍于不善心。无惭根及无愧根。}%
@{忿覆慳嫉惱，害恨諂誑憍，}@
%{忿根、覆根、悭根、嫉根、恼根，害根、恨根、谄根、诳根、憍根。}%
@{如是類名為，小煩惱地法，}@
%{如此类，名为小烦恼地法。}%
@{欲有尋伺故。於善心品中，}@

%{空无边等三种禅定，是根据修行的次第而命名的。}%

@{非想非非想，昧劣故立名。}@
%{非想非非想，因为昏昧劣等，所以才这样命名。}%

@{此本等至八，前七各有三，}@
%{从初禅到第八禅，前面七种禅定每种都有三种。}%

@{謂味淨無漏，後味淨二種。}@
%{指的是味、净、无漏，后面还有味净两种。}%

@{味謂愛相應，淨謂世間善，}@
%{味是指与爱相对应的，净是指世间善法。}%

@{此即所味著，無漏謂出世。}@
%{这就是所爱所著，无漏是指出世法。}%

@{靜慮初五支，尋伺喜樂定，}@

@{第二有四支，內淨喜樂定，}@
%{第二禅有四支，分别是内净、喜、乐、定。}%

@{第三具五支，捨念慧樂定，}@
%{第三禅有五支，分别是舍、念、慧、乐、定。}%

@{第四有四支，捨念中受定。}@
%{第四禅有四支，分别是舍、念、中受、定。}%

@{此實事十一，初二樂輕安，}@
%{这实际上是十一支，初禅的乐、轻安。}%

@{內淨即信根，喜即是喜受。}@
%{内净就是信根，喜就是喜受。}%

@{染如次從初，無喜樂內淨，}@
%{染污的禅定依次从初禅开始，没有喜、乐、内净。}%

@{正念慧捨念，餘說無安捨。}@
%{正念、慧、舍念，其他的禅定没有轻安和舍。}%

@{第四名不動，離八災患故，}@
%{第四禅称为不动禅，因为远离了八种灾患。}%

@{八者謂尋伺，四受入出息。}@
%{八种灾患指的是寻、伺、四种受、入、出、息。}%

@{生靜慮從初，有喜樂捨受，}@
%{从初禅开始，有喜、乐、舍、受。}%

@{及喜捨樂捨，唯捨受如次。}@
%{还有喜舍、乐舍，只有舍、受依次出现。}%

@{生上三靜慮，起三識表心，}@
%{生起上面的三种禅定，会产生三种识，显现心。}%

@{皆初靜慮攝，唯無覆無記。}@
%{都归属于初禅，只有无覆无记。}%

@{全不成而得，淨由離染生，}@
%{完全不成就而得到，净定是由远离染污而生。}%

@{無漏由離染，染由生及退。}@
%{无漏禅定也是由远离染污而生，染污禅定是由生起和退失而生。}%

@{無漏次生善，上下至第三，}@
%{无漏禅定依次生起善法，从下到上到第三禅。}%

@{淨次生亦然，兼生自地染。}@
%{净定也是依次生起，同时还会生起本地的染污。}%

@{染生自淨染，并下一地淨，}@
%{染污禅定生起，同时会生起本地的净定，以及下一地的净定。}%

@{死淨生一切，染生自下染。}@
%{死后净定生起一切，染污禅定生起，同时会生起下地的染污。}%

@{淨定有四種，謂即順退分，}@
%{净定有四种，指的是即、顺、退、分。}%

@{順住順勝進，順決擇分攝。}@
%{顺住、顺胜、进，属于顺决择分。}%

@{如次順煩惱，自上地無漏，}@
%{依次顺烦恼，从上地生起无漏禅定。}%

@{互相望如次，生二三三一。}@
%{互相望向，依次生起二、三、三、一。}%

@{二類定順逆，均間次及超，}@
%{两种禅定顺逆，平均、间、次、超。}%

@{至間超為成，三洲利無學。}@
%{到达间、超，就成就了，利益三界无学。}%

@{諸定依自下，非上無用故，}@
%{各种禅定依次从下地生起，不是说上面的禅定没有用。}%

@{唯生有頂聖，起下盡餘惑。}@
%{只有生起有顶圣，才能从下地开始，尽除余惑。}%

@{味定緣自繫，淨無漏遍緣，}@
%{味定缘于自身，净定和无漏禅定遍缘。}%

@{根本善無色，不緣下有漏。}@
%{根本善法是无色，不缘于下地的有漏法。}%

@{無漏能斷惑，及諸淨近分，}@
%{无漏禅定能断除烦恼，以及各种净定近分。}%

@{近分八捨淨，初亦聖或三。}@
%{近分有八种舍净，初禅也可能是圣者或第三禅。}%

@{中靜慮無尋，具三唯捨受，}@
%{中禅没有寻，只有三支，分别是舍、受、定。}%

@{初下有尋伺，中唯伺上無。}@
%{初禅有寻、伺，中禅只有伺，上禅没有。}%

@{空謂空非我，無相謂滅四，}@
%{空是指空非我，无相是指灭四谛。}%

@{無願謂餘十，諦行相相應。}@
%{无愿是指其余十谛，以及行相相对应的。}%

@{此通淨無漏，無漏三脫門，}@
%{这些禅定都通于净定和无漏禅定，无漏禅定是三解脱门。}%

@{重二緣無學，取空非常相，}@
%{重复两次缘于无学，取空非常相。}%

@{後緣無相定，非擇滅為靜。}@
%{后面缘于无相定，非择灭为静。}%

%{有漏众生有时会离开上面的七种近分。}%

%{为了得到现法快乐，修行各种善法禅定。}%

@{為得勝知見，修淨天眼通，}@
%{为了得到胜妙的智慧，修行净天眼通。}%

%{慈悲无瞋，喜舍无贪。}%

%{这些行相依次出现，带来快乐和拔除痛苦。}%

@{欣慰有情等，緣欲界有情。}@
