@{唐天后代佛授記寺翻經沙門慧智奉 制譯}@
%{唐天后代佛授記寺翻經沙門慧智奉 制譯}%

@{菩薩號為觀世音，神通無礙難可量，}@
%{菩萨号称观世音，神通无碍难以衡量，}%
@{搖山竭海震大地，悲愍眾生同一體，}@
%{摇山竭海震动大地，悲悯众生如同一体，}%
@{記憶名號福不虛，是故常應稱念彼，}@
%{记住名号福报不虚，所以应该常念观世音菩萨，}%
@{我今至誠念彼德，以是敬心而讚歎。}@
%{我如今真诚地念诵观世音菩萨的德行，以此敬畏之心而赞叹。}%

@{菩薩智慧深如海，方便善權無能測，}@
%{菩萨智慧深如大海，方便善巧难以测度，}%
@{所有諸天阿脩羅，摩睺羅伽及人眾，}@
%{所有诸天阿修罗，摩睺罗伽以及人众，}%
@{咸以微妙清淨偈，歷劫讚之無懈倦，}@
%{都以微妙清净的偈颂，历劫赞叹观世音菩萨，毫无倦怠，}%
@{我今宣說春陽頌，述其功德之少分。}@
%{我如今宣说春阳颂，述说观世音菩萨功德的一小部分。}%

@{世間形色尊嚴者，諸天莫踰於梵王，}@
%{世间形色尊严者，诸天莫过于梵王，}%
@{況聖自在本色身，百千萬分不及一。}@
%{何况观世音菩萨圣洁自在的本色身，百千万分之一也比不上。}%

@{尊者首飾甚嚴好，冠以曼陀及金花，}@
%{观世音菩萨头饰甚为精美，戴着曼陀罗和金花，}%
@{虹蜺美麗以莊嚴，復如半月映山王，}@
%{虹光美丽地装饰着，又像半月映照着山王，}%
@{又似百寶成須彌。尊者身相甚微妙，}@
%{又像百宝堆成的须弥山。观世音菩萨身相极其微妙，}%

@{猶如輕雨籠寶岳，伊尼鹿彩覆其肩，}@
%{犹如轻雨笼罩着宝山，伊尼鹿彩覆盖着肩膀，}%
@{光明晃朗普周遍，巍巍挺特若金山。}@
%{光明晃朗遍布周身，巍峨挺拔如同金山。}%

@{亦如滿月處虛空，又似薝波迦花色，}@
%{也像满月在虚空之中，又像薝波迦花色，}%
@{勝彼摩醯首羅身，徒以白龍為瓔珞。}@
%{胜过摩醯首罗的身躯，仅仅以白龙作为璎珞。}%

@{右手執持金蓮花，毘琉璃寶以為莖，}@
%{右手执持金莲花，用玻璃宝作为茎，}%
@{以慈開敷極香潔，令諸智者生愛樂。}@
%{以慈悲之心开敷，极其香洁，令诸智者生起爱乐。}%

@{尊手所持勝淨花，莊嚴殊妙甚暉麗，}@
%{观世音菩萨手中所持的胜净花，庄严殊妙，极其光辉美丽，}%
@{如月光曜須彌頂。聖身所處蓮花臺，}@
%{如同月光照耀着须弥山顶。观世音菩萨所处之处是莲花台，}%

@{由尊福德之所感，我以至誠慇重心，}@
%{是由观世音菩萨的福德所感召，我以至诚恳切之心，}%
@{歸命敬禮賜願者，諸天供養所讚歎，}@
%{归命敬礼赐予愿望者，诸天供养赞叹观世音菩萨，}%
@{我今淨心歸命禮。尊者螺髻玄蜂色，}@
%{我如今净心归命礼拜。观世音菩萨螺髻是玄蜂色，}%

@{牟尼妙像處其中，光明映曜嚴尊首，}@
%{牟尼妙像在其间，光明映照着观世音菩萨的尊首，}%
@{如燒珊瑚流電色，又如雌黃耀黑山，}@
%{如同燃烧的珊瑚流出电光之色，又像雌黄照耀着黑山，}%
@{上下莊嚴大聖身，冒敢為文稱歎彼。}@
%{上下庄严着伟大的圣者之身，我冒昧地用文字赞叹观世音菩萨。}%

@{四面狂象利牙齒，若人居中將被害，}@
%{四面狂象利齿，如果有人身处其中将要被伤害，}%
@{其人專心念菩薩，捨離苦惱而無畏。}@
%{这个人专心念诵观世音菩萨，便能摆脱苦恼，无所畏惧。}%

@{暴龍哮吼放烟毒，嘊喍瞋怒厲惡色，}@
%{暴龙哮吼放出烟毒，嘊喍瞋怒，厉声恶色，}%
@{將噬其人甚可怖，憶念觀音便解脫。}@
%{将要吞噬这个人，非常可怕，忆念观世音菩萨便能解脱。}%

@{海潮風浪出音聲，摩竭巨魚皆震激，}@
%{海潮风浪发出声音，摩竭巨鱼都震动激荡，}%
@{若人船破誤落中，念彼觀音獲安隱。}@
%{如果有人船破误落水中，念诵观世音菩萨便能获得安稳。}%

@{高山嶮谷懸巖裏，奔流濺石眾難處，}@
%{高山险谷悬崖之中，奔流溅石，众人难以到达之处，}%
@{若人墮中無救護，心念觀音登彼岸。}@
%{如果有人坠落其中，没有救护，心中念诵观世音菩萨便能登上彼岸。}%

@{烈火猛焰惡風起，延燒屋宅焚及人，}@
%{烈火猛焰恶风起，蔓延烧毁房屋宅院，焚烧到人，}%
@{至心念彼功德藏，應時災火不為害。}@
%{至心念诵观世音菩萨的功德宝藏，应时灾火便不会伤害。}%

@{若人繫縛在牢獄，身嬰枷鎖及杻械，}@
%{如果有人被囚禁在牢狱，身上戴着枷锁和脚镣，}%
@{至心稱念觀音故，其人須臾便解脫。}@
%{至心称念观世音菩萨，这个人须臾之间便能解脱。}%

@{若人墮落大火坑，專志稱名觀世音，}@
%{如果有人堕落到大火坑，专心致志地称念观世音菩萨，}%
@{其人除熱得清淨，猶若池中華開敷。}@
%{这个人便能消除热恼，获得清净，如同池塘中莲花盛开。}%

@{若人經於大林中，忽遇雷電驟風雨，}@
%{如果有人经过大森林，突然遇到雷电暴风雨，}%
@{霹靂將燒極危殆，稱念觀音免災難。}@
%{霹雳将要烧毁，极其危险，称念观世音菩萨便能免除灾难。}%

@{若人逢值猛師子，牙爪銛利如刀劍，}@
%{如果有人遇到猛狮子，牙齿爪子锋利如同刀剑，}%
@{瞋目吼喚來食人，稱念觀音便免害。}@
%{瞋目怒吼着要来吃人，称念观世音菩萨便能免除伤害。}%

@{惡賊突厥彌戾車，兇險無慈若羅剎，}@
%{恶贼突厥弥戾车，凶险无慈如同罗刹，}%
@{復以鐵鎖繫縛人，稱名發念皆解脫。}@
%{又用铁锁捆绑人，称念观世音菩萨，便能解脱。}%

@{深山大澤險難處，劫賊伺人將抄奪，}@
%{深山大泽险难之处，劫贼伺机要抢夺人，}%
@{稱名念力願救護，其人釋然無患苦。}@
%{称念观世音菩萨的名号，念力祈求救护，这个人便能释然无忧，没有痛苦。}%

@{呪呾讒訟擬中傷，恒以惡心求方便，}@
%{咒诅诽谤，想要中伤人，总是怀着恶意寻找机会，}%
@{由彼常念觀音故，其人竟不能為害。}@
%{由于这个人常念观世音菩萨，那个人最终不能加害于他。}%

@{鬪戰旗鼓兩相當，鋒鏑交橫競殘刃，}@
%{战斗旗鼓两相抵，锋鏑交错，争着残杀，}%
@{此時心念及稱名，菩薩於中施無畏。}@
%{此时心中念诵观世音菩萨的名号，菩萨便在其中施予无畏。}%

@{大力惡鬼執捉人，鳩槃荼等飲人髓，}@
%{大力恶鬼捉拿人，鸠槃荼等饮人髓，}%
@{若能繫念觀音者，擁護其人銷疫難。}@
%{如果能系念观世音菩萨，便能护佑这个人，消除疫病灾难。}%

@{鬼病熱病腹脹病，疳病黃病心痛病，}@
%{鬼病、热病、腹胀病、疳病、黄病、心痛病，}%
@{癩病癲病霍亂病，血氣羸瘦種種病，}@
%{癞病、癫病、霍乱病，血气羸瘦各种疾病，}%
@{諸惡黑業纏其身，稱名繫念皆除愈。}@
%{种种恶业缠绕其身，称念观世音菩萨便能消除治愈。}%

@{有諸餓鬼腹如山，脣口乾焦飢渴惱，}@
%{有些饿鬼肚子像山一样大，嘴唇嘴巴干枯，饥渴难受，}%
@{菩薩見之生慈悲，次第令其得充足。}@
%{观世音菩萨看到他们，生起慈悲，次第令他们得到充足。}%

@{餓鬼或遭寒凍逼，身體皮肉皆破壞，}@
%{饿鬼有时遭受寒冷冻逼，身体皮肉都被破坏，}%
@{舉彼兩手生極苦，神通方便令熅適。}@
%{举起两只手非常痛苦，观世音菩萨的神通方便使他们感到温暖舒适。}%

@{若金翅鳥以嘴爪，搏撮水陸諸龍等，}@
%{如果金翅鸟用嘴爪，搏捉水陆诸龙等，}%
@{其龍專心念菩薩，即得離苦獲安樂。}@
%{这些龙专心念诵观世音菩萨，便能脱离苦难，获得安乐。}%

@{若有欲求勝果報，象馬車乘及奴婢，}@
%{如果有人想要求得胜果报，象马车乘以及奴婢，}%
@{衣服飲食諸珍寶，由彼常念觀音故，}@
%{衣服饮食诸珍宝，由于这个人常念观世音菩萨，}%
@{應其求願自然至。若有欲得長生術，}@
%{应其所求愿望自然会实现。如果有人想要获得长生术，}%

@{坐臥雲間恣空行，由彼常念觀音故，}@
%{坐卧云间，任意在空中行走，由于这个人常念观世音菩萨，}%
@{當獲神呪及仙藥。我見壁畫觀音像，}@
%{便能获得神咒和仙药。我看到墙壁上的观世音菩萨画像，}%
@{遍視色相諸功德，及見神通大自在，}@
%{遍观观世音菩萨的色相和种种功德，以及看到观世音菩萨的神通和自在，}%
@{故起至誠而讚歎。所有一切諸功德，}@
%{所以生起至诚之心而赞叹。所有一切功德，}%

@{於菩薩中最第一，無邊善巧大方便，}@
%{在菩萨中是最第一，无边善巧，大方便，}%
@{示現清淨妙色身，遠離有無諸分別，}@
%{示现清净妙色身，远离有无一切分别，}%
@{利益無量如虛空。以此讚歎功德藏，}@
%{利益无量如同虚空。以此赞叹观世音菩萨的功德宝藏，}%
@{願證如來一切智。}@
%{愿证得如来一切智。}%

@{以此讚歎觀世音菩薩功德，上資國主聖神皇帝陛下善轉金輪，色力壽命永無窮盡，智慧果報不可思議，恒御閻浮長居震旦，擁護三寶利益群生，一切含靈皆蒙聖福。}@
%{以此赞叹观世音菩萨的功德，上资国主圣神皇帝陛下善转金轮，色力寿命永无穷尽，智慧果报不可思议，恒御阎浮，长居震旦，拥护三宝，利益群生，一切含灵都蒙受圣福。}%
