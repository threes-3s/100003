@{佛說內身觀章句經}@
%{佛說內身觀章句經}%

@{失譯人名在後漢錄}@
%{失譯人名在後漢錄}%

@{一切一其心，皆聽美言訓，}@
%{所有的心都应该听从美好的教诲，}%
@{佛所從得道，且聽我彼經。}@
%{佛陀从哪里得到道呢？请听我讲述那部经书。}%
@{彼空亦不斷，有行皆非常，}@
%{那虚空也不断，所有的行为都不常存在，}%
@{夫行不敗壞，佛以講授經。}@
%{因为行为不会败坏，所以佛陀才讲授经书。}%

@{深微難見事，非章原之句，}@
%{那些深奥难见的事，不是用一般的语句就能表达清楚的，}%
@{通彼能敷演，是以故為師。}@
%{只有通达的人才能详细阐述，所以他们才成为老师。}%
@{從本以存本，有造法之積，}@
%{从根本上保存根本，有造作法的积累，}%
@{從慧以除棄，上士所講說。}@
%{从智慧上舍弃它，这是上士所讲的。}%

@{所從因緣有，有行皆無有，}@
%{所有行为都有因缘，行为本身并不存在，}%
@{亦有前世除，無知彼諸行；}@
%{也因为前世的缘故，不知那些行为；}%
@{亦如欲演身，日親之所講，}@
%{就像想要阐述身体，每天亲近的人都会讲，}%
@{大災患無數，皆以歸流身。}@
%{无穷无尽的大灾难，都归结到身体上。}%

@{刪定如本文，所演眾要言，}@
%{删定如同本文，所阐述的重要言论，}%
@{靡麗五字句，明誓以勸勉。}@
%{没有华丽的五字句，用明确的誓言来劝勉。}%
@{以頌文具足，偶字音商備，}@
%{用颂文来使它完整，偶字音韵也具备，}%
@{若花實雜糅，皆聽彼我誨。}@
%{就像花和果实混合在一起，都听从那我的教诲。}%

@{身非人非命，不丈夫非士，}@
%{身体不是人，不是命，不是丈夫，不是士，}%
@{若體若不艶，斯事都無彼。}@
%{如果身体不美好，这些事情就都没有了。}%
@{夫身造而有，以有即敗亡，}@
%{身体是由造作而有，有了就会败亡，}%
@{無強皆歸命，如沫蹈踐碎；}@
%{没有强求都归命，就像泡沫被踩踏一样破碎；}%

@{無強則無常，無常即無樂；}@
%{没有强求就无常，无常就没有快乐；}%
@{無常亦為苦，非身身非我；}@
%{无常也是苦，身体不是身体，不是我；}%
@{身非常亦苦，非彼應為身。}@
%{身体不常也是苦，不应该把它看成是身体。}%

@{若體艶我有，身都無可有；}@
%{如果身体美好，我存在，身体就没有任何东西可以拥有；}%
@{若以都無體，有存亦有亡。}@
%{如果身体没有任何东西，存在就会有消失。}%
@{慧者以本末，何彼已有體？}@
%{智慧的人从根本和末端来观察，那身体怎么会本来就存在呢？}%

@{若以無主者，不得以自由，}@
%{如果身体没有主人，就无法自由，}%
@{有校計若此，何彼體我有？}@
%{有这种计算，那身体怎么会是我呢？}%
@{斯體身為空，體我有為虛，}@
%{这个身体是空的，身体是我也是虚假的，}%
@{應因緣為有，非身逝心造，}@
%{应该因为因缘而有，不是身体消亡后心造出来的，}%

@{非眾人身造，亦非自身造，}@
%{不是众人造的，也不是自身造的，}%
@{都無身有造，亦身無不造。}@
%{完全没有身体存在的造作，也没有身体不存在的造作。}%
@{以身無造者，從彼為得諦。}@
%{因为身体没有造作，从那里获得真理。}%

@{亦非都骨節，}@
%{也不是完全由骨节组成的，}%
@{從前世方來，亦非天造身，}@
%{是从前世而来，也不是天造的身体，}%
@{非神所化城，非無行無本，}@
%{不是神所化的城，不是没有行为没有根本，}%
@{無因為自有，是身有所由，}@
%{没有原因就自己存在，这个身体是有来源的，}%

@{有事亦有物。有本有所起，}@
%{有事情就有事物。有根本就有起源，}%
@{稍稍為生有，本以癡亦愛，}@
%{一点点地就生存在了，根本是因为愚痴和贪爱，}%
@{心與愛有漏，亦有縛亦結，}@
%{心和爱有漏，也有束缚和结缚，}%
@{行二品為漏。斯為本亦餘，}@
%{行为的两种品类是漏。这就是根本，还有其他，}%

@{世與受因緣，以由彼斯身，}@
%{世界和受的因缘，由此而来，这个身体，}%
@{以漸能致有。初始有精沫，}@
%{逐渐就能存在。最初有精沫，}%
@{精沫為轉凝，足為生兩兩，}@
%{精沫转而凝固，足以生出两个，}%
@{以兩為轉厚，以序有四體，}@
%{以两个转而变厚，依次就有了四肢，}%

@{為生頭第五，若干骨積聚，}@
%{生出头是第五，许多骨头聚集在一起，}%
@{從行為用成，頭九為髑髏，}@
%{从行为中用它而成，头部九块是头骨，}%
@{頞䪼為二骨，齒根三十二，}@
%{颌骨有两块骨头，牙齿根部有三十两块，}%
@{齒三十二骨，頸為與耳本，}@
%{牙齿有三十两块骨头，脖子是耳朵的根部，}%

@{鼻為與上腭，心與頸嚨喉，}@
%{鼻子是上颚的一部分，心脏与脖子喉咙相连，}%
@{凡為骨八十。頰車與頸四，}@
%{总共有八十块骨头。腮骨和脖子有四块，}%
@{咽亦為骨四，左臂與右臂，}@
%{咽喉也有四块骨头，左胳膊和右胳膊，}%
@{凡有骨五十。若其斯左脇，}@
%{总共有五十块骨头。如果左边肋骨，}%

@{應有十三肋；亦斯右脇然，}@
%{应该有十三根肋骨；右边肋骨也是一样，}%
@{應有肋十三，斯為四十八。}@
%{应该有十三根肋骨，总共是四十八根。}%
@{三三三相連，二為二相連，}@
%{三根三根相连，两根两根相连，}%
@{其餘不相連，身者以為彊。}@
%{其他的不相连，身体就认为是强壮。}%

@{如束葦無彊，脊膂三十二，}@
%{就像捆绑的芦苇没有强壮，脊椎骨有三十两块，}%
@{尻與腰為三，若其斯左髀，}@
%{臀部和腰部有三块，如果左边大腿，}%
@{骨為二十五；右髀為亦然，}@
%{骨头有二十五块；右边大腿也是一样，}%
@{為骨二十五；肩髀有骨四，}@
%{有二十五块骨头；肩胛骨有四块骨头，}%

@{凡三百二十。敷演名之諦，}@
%{总共三百二十块。详细阐述名相的真理，}%
@{佛以日斷嗣，彼叚悉以聚，}@
%{佛陀用每天断除后裔，那些残余的都聚在一起，}%
@{諸根為以縫，非瘡而裹之，}@
%{各种根系用它来缝合，不是伤口就用它来包裹，}%
@{肉血以塗𢴉，如木機關縷，}@
%{用肉和血来涂抹，就像木头机关的丝线，}%

@{為如幻師幻。骨機關亦然，}@
%{就像幻术师的幻术。骨头的机关也是一样，}%
@{以筋纏縛成，合聚骨若此，}@
%{用筋缠绕捆绑而成，把骨头聚集在一起，就像这样，}%
@{以為是形體。愚者莫不著，}@
%{就认为是身体。愚蠢的人没有不执着的，}%
@{智者而不著。生革以隱蔽，}@
%{智慧的人不执着。生皮用来遮蔽，}%

@{九孔為大瘡，周匝為滲漏，}@
%{九个孔洞是大的伤口，周围都渗漏，}%
@{不淨腐臭處，口喙如為孔，}@
%{不洁净腐烂臭的地方，嘴巴就像是一个孔，}%
@{滿之以諸穀，是身為若此。}@
%{用各种谷物填满它，身体就是这样。}%

@{以進若干腑，毛髮與爪齒，}@
%{还有很多内脏，毛发和指甲牙齿，}%
@{塵埃亦皮革，骨節亦骨髓，}@
%{尘埃也是皮革，骨节也是骨髓，}%
@{為凡筋與脈，胃與心亦脾，}@
%{还有筋和脉，胃和心还有脾，}%
@{大腸亦小腸，肝與肺亦腎，}@
%{大肠和小肠，肝和肺还有肾，}%

@{脂𦠻亦大便，淚與唾亦汗，}@
%{脂肪也是大便，眼泪和唾液还有汗水，}%
@{鼻涕膏亦血，寒熱肪小便，}@
%{鼻涕膏也是血，寒热脂肪还有小便，}%
@{𮌗之與𮌗膜，皆以沈沒彼，}@
%{脓液和脓膜，都沉没在那里，}%
@{如泥塗老牛，如其成不知。}@
%{就像泥巴涂在老牛身上，就像它形成的时候不知道。}%

@{身之內與外，夫城骨與牆，}@
%{身体的内部和外部，就像城墙的骨头和墙，}%
@{肉血為塗嚴，為怨所破壞。}@
%{用肉和血来涂抹装饰，被怨恨所破坏。}%
@{恒為以內外，彼央若干百，}@
%{总是内外，那里有数百个，}%
@{以為貪其肉，其外亦災害，}@
%{以为贪图它的肉，外面也有灾难，}%

@{皆以多尤彼，斯身腐敗壞。}@
%{都用很多罪过责备它，这个身体腐烂败坏。}%
@{譬如久故城，晝夜供侍之，}@
%{比如一座古老的城，昼夜侍奉它，}%
@{壞如已復壞，如坎與空聚；}@
%{坏就像已经坏掉，就像坑和空虚聚集在一起；}%
@{恒盜賊俱止，取之欺殆人，}@
%{总是盗贼一起居住，偷窃欺骗人，}%

@{身為毒虺穴，夫毒虺劇毒；}@
%{身体是毒蛇的巢穴，毒蛇剧毒；}%
@{恒于身居止，喜怒毒奸弊，}@
%{总是住在身体里，喜怒毒奸邪弊病，}%
@{孚不和大毒。正使滿百歲，}@
%{不诚实不和善的大毒。即使活到一百岁，}%
@{恒以和安隱，忿則無反復，}@
%{总是和睦安宁，愤怒就无法复原，}%

@{須臾復不安。斯身為災禍，}@
%{须臾就再次不安。这个身体是灾祸，}%
@{嫌而有恐畏，如虎遊荒澤，}@
%{厌恶它就感到恐惧，就像老虎在荒野中游荡，}%
@{有畏多恐怖。諸念為以仰，}@
%{有恐惧就感到很多恐怖。各种念头都仰仗它，}%
@{為一切苦器，亦為諸劇事。}@
%{成为一切痛苦的工具，也是各种剧烈的事情。}%

@{斯身主為受，沈於苦之法，}@
%{这个身体是感受的主人，沉溺于痛苦的法则，}%
@{一切諸病宅，為老死之法。}@
%{一切疾病的住所，是衰老死亡的法则。}%
@{身為增恩愛，苦以寒與熱，}@
%{身体使恩爱增长，痛苦来自寒冷和炎热，}%
@{或風而不和，是則病如生。}@
%{或者风不调和，疾病就会像这样产生。}%

@{為敗壞諸根，失疾賊害人，}@
%{使根系败坏，失去健康，伤害人，}%
@{老死笮厄人，如雪聚得火，}@
%{衰老死亡折磨人，就像雪堆遇到火，}%
@{疾而為解釋。斯若此無數，}@
%{疾病就立刻消失。像这样无穷无尽，}%
@{身之多災禍，吾所演一切，}@
%{身体的灾难很多，我所阐述的一切，}%

@{未能已備具。要以為諸苦，}@
%{还不能完全具备。总之，身体是各种痛苦，}%
@{腐身為不實，多因緣以成，}@
%{腐烂的身体是不真实的，由很多因缘而成，}%
@{大耶以為軀。}@
%{总之，身体是躯壳。}%

@{十一因緣章}@
%{十一因缘章}%

@{佛言：行者有十一因緣，滅道制令人不墮惡道，當不識者，謂萬物。一為大會，謂人眾；二者多食，謂諸美，亦謂過飽；三者為多行，謂多業；四為多喧，謂多語；五為多睡眠；六為會聚，謂禪中；七為習行，謂多事；八為愛身；九為輕，謂非法語；十為貪，謂多欲；十一為不好善處居，謂惡人中。行道者當斷是十一因緣，得道疾。}@
%{佛陀说：修行人有十一因缘，灭除道，制止人堕落恶道，对于那些不认识的人来说，就是指万物。一是大会，指人群；二是多食，指各种美味，也指过饱；三是多行，指多做业；四是多喧，指多说话；五是多睡眠；六是会聚，指禅定中；七是习行，指多做事；八是爱惜身体；九是轻慢，指不正确的语言；十是贪婪，指多欲；十一是不喜欢善处居住，指恶人之中。修行人应该断除这十一因缘，就能快速得道。}%
