@{入楞伽經卷第四元魏天竺三藏菩提留支譯}@
%{入楞伽經卷第四元魏天竺三藏菩提留支譯}%

@{集一切佛法品第三之三}@
%{集一切佛法品第三之三}%

@{爾時聖者大慧菩薩摩訶薩復白佛言：世尊！如世尊說，十二因緣從因生果，不說自心妄想分別見力而生。}@
%{当时大慧菩萨又对佛说：世尊！您说十二因缘是从因生果，但没有说从自心妄想分别见力而生。}%
@{世尊！若爾，外道亦說從因生果。}@
%{世尊！如果这样，外道也说从因生果。}%
@{世尊！外道說言：從於自性、自在天、時、微塵等因生一切法。}@
%{世尊！外道说：从自性、自在天、时间、微尘等因生一切法。}%
@{如來亦說依於因緣而生諸法，而不說有自建立法。}@
%{如来也说依于因缘而生诸法，但没有说有自建立的法。}%
@{世尊！外道亦說：從於有無而生諸法。}@
%{世尊！外道也说：从于有无而生诸法。}%
@{世尊說言，諸法本無，依因緣生生已還滅，世尊說從無明緣行乃至於有，依眼識等生一切法。}@
%{世尊说：诸法本来没有，依因缘生，生已还灭，世尊说从无明缘行乃至于有，依眼识等生一切法。}%
@{如世尊說，亦有諸法無因而生。}@
%{正如世尊所说，也有一些法无因而生。}%
@{何以故？不從因生一時無前後生，以因此法生此法，世尊自說，因虛妄因法生此法，非次第生故。}@
%{为什么呢？因为不从因生，一时无前后生，因为此法生此法，世尊自己说，因虚妄因法生此法，并非次第生。}%
@{世尊！若爾外道說法勝而如來不如。}@
%{世尊！如果这样，外道说法胜过如来。}%
@{何以故？世尊！外道說因無因緣能生果，如來說法因亦依果果亦依因，若爾因緣無因無果。}@
%{为什么呢？世尊！外道说因无因缘能生果，如来说法因也依果，果也依因，如果这样，因缘无因无果。}%
@{世尊！若爾彼此因果展轉無窮。}@
%{世尊！如果这样，彼此因果展转无穷。}%
@{世尊說言，從此法生彼法，若爾無因生法。}@
%{世尊说：从此法生彼法，如果这样，就无因生法了。}%

@{佛告聖者大慧菩薩摩訶薩言：大慧！我今當說，因此法生彼法，不同外道所立因果，無因之法亦從因生。}@
%{佛对大慧菩萨说：大慧！我现在要告诉你，因此法生彼法，不同于外道所立因果，无因之法也从因生。}%
@{我不如是。}@
%{我不这样认为。}%
@{我說諸法從因緣生，非無因緣亦不雜亂，亦無展轉無窮之過。}@
%{我说诸法从因缘生，并非无因缘，也不杂乱，也没有展转无穷的过失。}%
@{何以故？以無能取可取法故。}@
%{为什么呢？因为没有能取可取的法。}%
@{大慧！外道不知自心見故，執著能取可取之法，不知不覺惟自心見內外法故。}@
%{大慧！外道不知自心见故，执着能取可取之法，不知不觉惟自心见内外法。}%
@{大慧！彼諸外道不知自心內境界故見有無物，是故外道有如是過，非我過也。}@
%{大慧！那些外道不知自心内境界，所以见有无物，因此外道有这样的过失，并非我的过失。}%
@{我常說言，因緣和合而生諸法，非無因生。}@
%{我常说：因缘和合而生诸法，并非无因生。}%

@{大慧復言：世尊！有言語說應有諸法。}@
%{大慧又说：世尊！有言语说应有诸法。}%
@{世尊！若無諸法者應不說言語。}@
%{世尊！如果没有诸法，就应该不说言语。}%
@{世尊！是故依言說應有諸法。}@
%{世尊！所以依言语说应有诸法。}%

@{佛告大慧：亦有無法而說言語，謂兔角龜毛石女兒等，於世間中而有言說。}@
%{佛对大慧说：也有无法而说言语，比如兔角龟毛石女儿等，在世间中就有言说。}%
@{大慧！彼兔角非有非無而說言語。}@
%{大慧！那个兔角非有非无，却有言说。}%
@{大慧！汝言以有言說應有諸法者，此義已破。}@
%{大慧！你说以有言说应有诸法，这个道理已经破了。}%
@{大慧！非一切佛國土言語說法。}@
%{大慧！并非一切佛国土都用言语说法。}%
@{何以故？以諸言語惟是人心分別說故。}@
%{为什么呢？因为诸言语只是人心分别说。}%
@{是故，大慧！有佛國土直視不瞬口無言語名為說法；有佛國土直爾示相名為說法；有佛國土但動眉相名為說法；有佛國土惟動眼相名為說法；有佛國土笑名說法；有佛國土欠呿名說法；有佛國土咳名說法；有佛國土念名說法；有佛國土身名說法。}@
%{所以，大慧！有些佛国土直视不瞬，口无言语，名为说法；有些佛国土直尔示相，名为说法；有些佛国土但动眉相，名为说法；有些佛国土惟动眼相，名为说法；有些佛国土笑，名为说法；有些佛国土欠呿，名为说法；有些佛国土咳，名为说法；有些佛国土念，名为说法；有些佛国土身，名为说法。}%
@{大慧！如無瞬世界及眾香世界，於普賢如來、應、正遍知，彼菩薩摩訶薩觀察如來目不暫瞬，得無生法忍，亦得無量勝三昧法。}@
%{大慧！如同无瞬世界和众香世界，在普贤如来、应、正遍知那里，那些菩萨观察如来目不暂瞬，得无生法忍，也得到无量胜三昧法。}%
@{是故，大慧！汝不得言有言語說應有諸法。}@
%{所以，大慧！你不能说有言语说应有诸法。}%
@{大慧！如來亦見諸世界中，一切微蟲蚊虻蠅等眾生之類，不說言語共作自事而得成辦。}@
%{大慧！如来也看到诸世界中，一切微虫蚊虻蝇等众生之类，不说言语，共同做自己的事，而能成功。}%

@{爾時世尊重說偈言：}@
%{当时世尊说偈言：}%

@{如虛空兔角，及與石女兒；}@
%{如同虚空兔角，以及石女儿；}%
@{無而有言說，如是妄分別。}@
%{无而有言说，如同是妄分别。}%
@{因緣和合法，愚癡分別生；}@
%{因缘和合法，愚痴分别生；}%
@{不知如實法，輪迴三有中。}@
%{不知如实法，轮回三有中。}%

@{爾時聖者大慧菩薩摩訶薩復白佛言：世尊！世尊說常語法，依何等法作如是說？}@
%{当时大慧菩萨又对佛说：世尊！您说常语法，依何等法作如是说？}%

@{佛告聖者大慧菩薩言：大慧！依迷惑法我說為常。}@
%{佛对大慧菩萨说：大慧！依迷惑法我说为常。}%
@{何以故？大慧！聖人亦見世間迷惑法非顛倒心。}@
%{为什么呢？大慧！圣人亦见世间迷惑法非颠倒心。}%
@{大慧！譬如陽焰、火輪、毛輪、乾闥婆城、幻、夢、水中月、鏡中像，世間非智慧者見有諸像，顛倒見故；有智慧者不生分別，非不見彼迷惑之事。}@
%{大慧！譬如阳焰、火轮、毛轮、乾闥婆城、幻、梦、水中月、镜中像，世间非智慧者见有诸像，颠倒见故；有智慧者不生分别，并非不见彼迷惑之事。}%
@{大慧！有智慧者見彼種種迷惑之事不生實心。}@
%{大慧！有智慧者见彼种种迷惑之事不生实心。}%
@{何以故？離有無法故。}@
%{为什么呢？因为离有无法。}%

@{佛復告聖者大慧菩薩言：大慧！云何迷惑法離於有無？}@
%{佛又对大慧菩萨说：大慧！云何迷惑法离于有无？}%
@{謂諸愚癡凡夫見有種種境界，如諸餓鬼大海恒河見水不見。}@
%{谓诸愚痴凡夫见有种种境界，如诸饿鬼大海恒河见水不见。}%
@{大慧！是迷惑法不得言有不得言無。}@
%{大慧！是迷惑法不得言有不得言无。}%
@{大慧！餘眾生見彼是水故不得言無。}@
%{大慧！其余众生见彼是水故不得言无。}%
@{大慧！迷惑之事亦復如是，以諸聖人離顛倒見故。}@
%{大慧！迷惑之事亦复如是，因为诸圣人离颠倒见故。}%
@{大慧！言迷惑法常者，以想差別故。}@
%{大慧！说迷惑法常者，是因为想差别故。}%
@{大慧！因迷惑法見種種相，而迷惑法不分別異差別。}@
%{大慧！因为迷惑法见种种相，而迷惑法不分别异差别。}%
@{是故，大慧！迷惑法常。}@
%{所以，大慧！迷惑法常。}%
@{大慧！云何迷惑法名之為實？}@
%{大慧！云何迷惑法名之为实？}%
@{以諸聖人迷惑法中不生顛倒心，亦不生實心。}@
%{因为诸圣人迷惑法中不生颠倒心，也不生实心。}%
@{大慧！而諸聖人見彼迷惑法起少心想，不生聖智事相。}@
%{大慧！而诸圣人见彼迷惑法起少心想，不生圣智事相。}%
@{大慧！起少想者是謂凡夫非謂聖人。}@
%{大慧！起少想者是谓凡夫，非谓圣人。}%
@{大慧！分別彼迷惑法顛倒非顛倒者，能生二種性。}@
%{大慧！分别彼迷惑法颠倒非颠倒者，能生两种性。}%
@{何等二種？}@
%{何等两种？}%
@{一者、能生凡夫性；二者、能生聖人性。}@
%{一者、能生凡夫性；二者、能生圣人性。}%
@{大慧！彼聖人性者，能生三種差別之性，所謂聲聞辟支佛佛國土差別性故。}@
%{大慧！彼圣人性者，能生三种差别之性，所谓声闻辟支佛佛国土差别性故。}%
@{大慧！云何毛道凡夫分別迷惑法，而能生彼聲聞乘性。}@
%{大慧！云何毛道凡夫分别迷惑法，而能生彼声闻乘性。}%
@{大慧！所謂執著彼迷惑法自相同相，能成聲聞乘性。}@
%{大慧！所谓执着彼迷惑法自相同相，能成声闻乘性。}%
@{大慧！是名迷惑法能生能成聲聞乘性。}@
%{大慧！是名迷惑法能生能成声闻乘性。}%
@{大慧！云何愚癡凡夫分別迷惑法，而能生彼辟支佛乘性？}@
%{大慧！云何愚痴凡夫分别迷惑法，而能生彼辟支佛乘性？}%
@{大慧！所謂執著彼迷惑法，觀察諸法自相同相不樂憒閙，能生辟支佛乘性。}@
%{大慧！所谓执着彼迷惑法，观察诸法自相同相不乐憒闹，能生辟支佛乘性。}%
@{大慧！是名迷惑法能生能成辟支佛乘性。}@
%{大慧！是名迷惑法能生能成辟支佛乘性。}%
@{大慧！云何智者即分別彼迷惑之法，能生佛乘性？}@
%{大慧！云何智者即分别彼迷惑之法，能生佛乘性？}%
@{大慧！所謂見彼能見可見惟是自心，而不分別有無法故。}@
%{大慧！所谓见彼能见可见惟是自心，而不分别有无法故。}%
@{大慧！如是觀察迷惑之法，能生能成如來乘性。}@
%{大慧！如是观察迷惑之法，能生能成如来乘性。}%
@{大慧！如是名為性義。}@
%{大慧！如是名为性义。}%
@{大慧！何者一切毛道凡夫即分別彼迷惑之法見種種事，能生世間所有乘性？}@
%{大慧！何者一切毛道凡夫即分别彼迷惑之法见种种事，能生世间所有乘性？}%
@{以觀察諸法如是如是決定不異。}@
%{因为观察诸法如是如是决定不异。}%
@{是故，大慧！彼迷惑法，愚癡凡夫虛妄分別種種法體。}@
%{所以，大慧！彼迷惑法，愚痴凡夫虚妄分别种种法体。}%
@{大慧！彼迷惑法非是實事非不實事。}@
%{大慧！彼迷惑法非是实事，非不实事。}%
@{何以故？大慧！聖人觀察彼迷惑法不虛妄分別，是故聖人能轉心、意、意識身相，離煩惱習故。}@
%{为什么呢？大慧！圣人观察彼迷惑法不虚妄分别，所以圣人能转心、意、意识身相，离烦恼习故。}%
@{是故聖人轉彼迷惑法，名為真如。}@
%{所以圣人转彼迷惑法，名为真如。}%
@{大慧！此名何等法？}@
%{大慧！此名何等法？}%
@{大慧！此名真如法，離分別法故。}@
%{大慧！此名真如法，离分别法故。}%
@{大慧！為此義故，我重宣說真如法體離分別法，彼真如中無彼虛妄分別法故。}@
%{大慧！为此义故，我重宣说真如法体离分别法，彼真如中无彼虚妄分别法故。}%
@{大慧菩薩復白佛言：世尊！彼迷惑法為有為無？}@
%{大慧菩萨又对佛说：世尊！那个迷惑法是有还是没有呢？}%

@{佛告大慧：彼迷惑法執著種種相故名有。大慧！彼迷惑法於妄想中若是有者，一切聖人皆應不離，執著有無虛妄法故。大慧！如外道說十二因緣，有從因生不從因生，此義亦如是。}@
%{佛告诉大慧：那个迷惑法因为执着种种相，所以叫它有。大慧！那个迷惑法如果在妄想中是有的话，那么一切圣人都应该不离开它，因为执着有无虚妄法。大慧！就像外道说十二因缘，有的是从因生，有的不是从因生，这个道理也是一样的。}%

@{大慧言：世尊！若迷惑法如幻見者，此迷惑法異於迷惑，以迷惑法能生法故。}@
%{大慧说：世尊！如果迷惑法像幻觉一样的话，那么这个迷惑法就不同于迷惑，因为迷惑法能生法。}%

@{佛告大慧：大慧！非迷惑法生煩惱過。大慧！若不分別迷惑法者不生諸過。復次，大慧！一切幻法依於人功呪術而生，非自心分別煩惱而生。是故，大慧！彼迷惑法不生諸過，惟是愚癡人見迷惑法故。大慧！愚癡凡夫執著虛妄微細之事而生諸過，非謂聖人。}@
%{佛告诉大慧：大慧！不是迷惑法生烦恼过失。大慧！如果不去分别迷惑法，就不会生出各种过失。再说，大慧！一切幻法都是依靠人的功力、咒术而生，不是由自己的心分别烦恼而生。所以，大慧！那个迷惑法不会生出各种过失，只是愚痴人看到迷惑法才会这样。大慧！愚痴凡夫执着虚妄细小的事情而生出各种过失，不是说圣人。}%

@{爾時世尊重說偈言：}@
%{当时世尊就吟诵了一首偈语：}%

@{聖不見迷惑，世間亦無實；}@
%{圣人不见迷惑，世间也没有真实；}%

@{迷惑即是實，實法次迷惑。}@
%{迷惑就是真实，真实法次于迷惑。}%

@{捨離諸迷惑，若有相生者；}@
%{舍离一切迷惑，如果有相生的话；}%

@{即彼是迷惑，不淨猶如翳。}@
%{那就是迷惑，不净就像翳障一样。}%

@{復次，大慧！汝不得言幻是無故，一切諸法亦無如幻。}@
%{再说，大慧！你不能说幻是无，一切诸法也没有像幻一样。}%

@{大慧言：世尊！為執著諸法如幻相故，言諸法如幻；為執著諸法顛倒相故，言諸法如幻也。世尊！若執著諸法如幻相者，世尊不得言一切法皆如幻相；若執著諸法顛倒相故言如幻者，不得言一切法如幻。何以故？世尊！色有種種因相見故。世尊！無有異因色有諸相可見如幻。是故，世尊！不得說言執著諸法一切如幻。}@
%{大慧说：世尊！因为执着诸法如幻相，所以说诸法如幻；因为执着诸法颠倒相，所以说诸法如幻。世尊！如果执着诸法如幻相，世尊就不能说一切法都如幻相；如果执着诸法颠倒相而说如幻，就不能说一切法如幻。为什么呢？世尊！因为色有各种各样的因相可見。世尊！没有其他的因，色有诸相可以看见，就像幻觉一样。所以，世尊！不能说执着诸法一切如幻。}%

@{佛告大慧：非謂執著種種法相，說言諸法一切如幻。大慧！諸法顛倒速滅如電故言如幻。大慧！一切諸法譬如電光即見即滅，凡夫不見。大慧！一切諸法亦復如是，以一切法自心分別同相異相，以不能觀察故不如實見，以妄執著色等法故。}@
%{佛告诉大慧：不是说执着种种法相，就说诸法一切如幻。大慧！诸法颠倒迅速灭亡，就像闪电一样，所以说如幻。大慧！一切诸法就像闪电一样，一见就灭，凡夫看不见。大慧！一切诸法也是这样，因为一切法都是由自己的心分别同相异相，因为不能观察，所以不能如实地看见，因为妄执着色等法。}%

@{爾時世尊重說偈言：}@
%{当时世尊就吟诵了一首偈语：}%

@{非見色等法，說言無幻法；}@
%{不是看见色等法，就说没有幻法；}%

@{故不違上下。我說一切法，}@
%{所以不违背上下。我说一切法，}%

@{不見有本性，如幻無生體。}@
%{不见有本性，就像幻觉一样没有生体。}%

@{大慧菩薩復白佛言：世尊！如世尊說諸法不生，復言如幻，將無世尊前後所說自相違耶？以如來說一切諸法不如幻故。}@
%{大慧菩萨又对佛说：世尊！就像世尊说诸法不生，又说如幻，难道世尊前后所说自相矛盾吗？因为如来说一切诸法不像幻觉一样。}%

@{佛告大慧：我說一切法不生如幻者，不成前後有相違過。何以故？以諸一切愚癡凡夫不見生法及不生法，不能覺知自心有無、外法有無。何以故？以不能見不生法故。大慧！我如是說諸法前後無有相違。大慧！我遮外道建立因果義不相當，是故我說諸法不生。大慧！一切外道愚癡群聚作如是說，從於有無生一切法；不說自心分別執著因緣而生。大慧！我說諸法有亦不生無亦不生。是故，大慧！我說諸法不生不滅。大慧！我說一切諸法有者，護諸弟子令知二法。何等為二？一者、攝取諸世間故；二者、為護諸斷見故。何以故？以依業故有種種身攝六道生，是故我說言有諸法攝取世間。大慧！我說一切法如幻者，為令一切愚癡凡夫畢竟能離自相同相故，以諸凡夫癡心執著墮於邪見，以不能知但是自心虛妄見故，令離執著因緣生法，是故我說一切諸法如幻如夢無有實體。何以故？若不如是說者，愚癡凡夫執邪見心，欺誑自身及於他身，離如實見一切法故。大慧！云何住如實見？謂入自心見諸法故。}@
%{佛告诉大慧：我说一切法不生如幻，不构成前后有矛盾。为什么呢？因为一切愚痴凡夫都看不到生法和不生法，不能觉知自己的心有没有，外法有没有。为什么呢？因为不能看到不生法。大慧！我这样说诸法前后没有矛盾。大慧！我遮止外道建立因果义不相应，所以我说诸法不生。大慧！一切外道愚痴群聚，就说从有无生一切法；不说自己的心分别执着因缘而生。大慧！我说诸法有也不生，无也不生。所以，大慧！我说诸法不生不灭。大慧！我说一切诸法有，是为了保护弟子们让他们知道两种法。哪两种呢？第一，摄取诸世间；第二，为了保护断见。为什么呢？因为依业力，有各种各样的身摄六道生，所以我说有诸法摄取世间。大慧！我说一切法如幻，是为了让一切愚痴凡夫最终能离开自相同相，因为诸凡夫痴心执着，堕落于邪见，因为不能知道这只是自己的心虚妄见，所以让他们离开执着因缘生法，所以我说一切诸法如幻如梦，没有实体。为什么呢？如果不这样说，愚痴凡夫执着邪见心，欺骗自己和别人，离开如实地看见一切法。大慧！怎么样才能住于如实见呢？就是进入自己的心，看见诸法。}%

@{爾時世尊重說偈言：}@
%{当时世尊就吟诵了一首偈语：}%

@{如汝言諸法，一切不生者；}@
%{就像你说诸法，一切不生；}%

@{是則謗因果，不生如實見。}@
%{那就是诽谤因果，不能生出如实见。}%

@{我說有生法，攝受諸世間；}@
%{我说有生法，摄受诸世间；}%

@{見諸法同幻，不取諸見相。}@
%{看见诸法都像幻觉一样，不取诸见相。}%

@{復次佛告聖者大慧菩薩言：大慧！我今為諸菩薩摩訶薩，說名句字身相，以諸菩薩善知名句字身相故，依名句字身相，速得阿耨多羅三藐三菩提，得菩提已為眾生說名句字相。}@
%{再说，佛告诉圣者大慧菩萨说：大慧！我现在为诸菩萨摩诃萨，说名句字身相，因为诸菩萨善知名句字身相，所以依名句字身相，迅速得到阿耨多罗三藐三菩提，得到菩提后，为众生说名句字相。}%
@{大慧菩薩白佛言：善哉！世尊！惟願速說。}@
%{大慧菩萨对佛说：太好了！世尊！请您赶紧讲讲吧。}%

@{大慧！何者名身？謂依何等何等法作名，名身事物名異義一。大慧！是名我說名身。}@
%{大慧！什么是名身呢？就是说，依靠什么法来命名，名身、事物、名、异、义，这五个字的意思都是一样的。大慧！这就是我说的名身。}%

@{大慧！何者是句身？謂義事決定究竟見義故。大慧！是名我說句身。}@
%{大慧！什么是句身呢？就是说，意思、事物确定、究竟、见到意思，所以叫句身。大慧！这就是我说的句身。}%

@{大慧！何者是字身？謂文句畢竟故。大慧！復次名身者，依何等法了別名句，能了知自形相故。}@
%{大慧！什么是字身呢？就是说，文句最终的意义，所以叫字身。大慧！再说，名身，就是依靠什么法来辨别名和句，能够认识到它自身的形相。}%

@{大慧！復次句身者，謂句事畢竟故。大慧！復次名身者，所謂諸字從名差別，從阿字乃至呵字，名為名身。}@
%{大慧！再说，句身，就是说，句子的意思最终是什么，所以叫句身。大慧！再说，名身，就是说，所有字从名称上区分，从“阿”字到“呵”字，都叫做名身。}%

@{大慧！復次字身者，謂聲長短音韻高下，名為字身。大慧！復次句身者，謂巷路行迹，如人象馬諸獸行迹等，得名為句。}@
%{大慧！再说，字身，就是说，声音的长度、短促、音韵的高低，叫做字身。大慧！再说，句身，就是说，巷道、道路上的行迹，比如人、象、马、各种动物的行迹等等，都叫做句身。}%

@{大慧！復次名字者，謂無色四陰依名而說。大慧！復次名字相者，謂能了別名字相故。}@
%{大慧！再说，名字，就是说，无色的四蕴，依靠名称来表达。大慧！再说，名字相，就是说，能够辨别名字的相貌。}%

@{大慧！是名名句字身相。大慧！如是名句字相，汝應當學為人演說。}@
%{大慧！这就是名、句、字的相貌。大慧！像这样名、句、字的相貌，你应该学习，然后为别人讲解。}%

@{爾時世尊重說偈言：}@
%{当时世尊就用偈颂来说：}%

@{名身與句身，及字身差別；}@
%{名身、句身、字身，它们之间是有区别的；}%

@{凡夫癡計著，如象溺深泥。}@
%{凡夫愚痴地执着于它们，就像大象陷入泥潭一样。}%

@{復次，大慧！未來世中無智慧者，以邪見心不知如實法故，因世間論自言智者，有智者問如實之法，離邪見相一異俱不俱。}@
%{再说，大慧！未来世中没有智慧的人，因为邪见的心，不知道如实法，所以就根据世间的理论，自称是智者，有智者问如实法，离开邪见，相是一样的，还是不同的，或者两者兼而有之呢？}%

@{而彼愚人作如是言：是問非是非正念問，謂色等法常無常為一為異？如是涅槃有為，諸行為一為異？}@
%{而那些愚人就说：这个问题问得不对，不是正念问，是说色等法，是常还是无常，是一样的，还是不同的？比如涅槃是有为法，还是无为法，两者是一样的，还是不同的？}%

@{相中所有能見所見為一為異？作者所作為一為異？四大中色香味觸為一為異？能見所見為一為異？}@
%{相中所有能见和所见，是一样的，还是不同的？作者和所作，是一样的，还是不同的？四大中，色、香、味、触，是一样的，还是不同的？能见和所见，是一样的，还是不同的？}%

@{泥團微塵為一為異？智者所知為一為異？如是等上上次第相，上上無記置答，佛如是說是為謗我。}@
%{泥团、微尘，是一样的，还是不同的？智者所知，是一样的，还是不同的？像这样层层递进的相貌，层层无记，置之不答，佛这样说，就是诽谤我。}%

@{大慧！而我不說如是法者，為遮外道邪見說故。何以故？大慧！外道等說謂身即命、身異命異，如是等法外道所說是無記法。}@
%{大慧！而我不说这些法，是为了遮蔽外道邪见的说法。为什么呢？大慧！外道等说，身体就是生命，身体和生命是不同的，像这样说法，是外道说的无记法。}%

@{大慧！外道迷於因果義故，是故無記，非我法中名無記也。大慧！我佛法中離能見可見虛妄之想無分別心，是故我法中無有置答；}@
%{大慧！外道迷于因果的道理，所以是无记，不是我法中所说的无记。大慧！我佛法中，离开能见、可见、虚妄的妄想，没有分别心，所以在我法中，没有置之不答；}%

@{諸外道等執著可取能取，不知但是自心見法，為彼人故我說言有四種問法，無記置答非我法中。}@
%{那些外道等执着于可取和能取，不知道这仅仅是自心所见的法，为了他们，我说了有四种问法，无记置之不答，不是我法中。}%

@{大慧！諸佛、如來、應、正遍知，為諸眾生有四種說言置答者，大慧！為待時故說如是法，為根未熟非為根熟，是故我說置答之義。}@
%{大慧！诸佛、如来、应、正遍知，为了众生，有四种说法，置之不答，大慧！为了等待时机，所以说这些法，为了根器未熟，而不是为了根器成熟，所以我说置之不答的道理。}%

@{復次，大慧！一切諸法若離作者及因不生，以無作者故，是故我說諸法不生。}@
%{再说，大慧！一切诸法，如果离开作者和因缘，就不会生起，因为没有作者，所以我说诸法不生。}%

@{佛告大慧：一切諸法無有體相。}@
%{佛对大慧说：一切诸法，没有实体和相貌。}%

@{大慧白佛言：世尊！何故一切諸法無實體相？}@
%{大慧对佛说：世尊！为什么一切诸法没有实体和相貌呢？}%

@{佛告大慧：自智觀察一切諸法自相同相不見諸法，是故我說一切諸法無實體相。}@
%{佛对大慧说：你自己用智慧观察，一切诸法自身相同，没有相貌，所以我说一切诸法没有实体和相貌。}%

@{佛告大慧：一切諸法亦無取相。}@
%{佛对大慧说：一切诸法，也没有取相。}%

@{大慧言：世尊！以何義故一切諸法亦無取相？}@
%{大慧说：世尊！为什么一切诸法也没有取相呢？}%

@{佛告大慧：自相同相無法可取，是故我說無法可取。}@
%{佛对大慧说：自身相同，没有相貌，就没有办法可以取，所以我说没有办法可以取。}%

@{佛告大慧：一切諸法亦無捨相。}@
%{佛对大慧说：一切诸法，也没有舍相。}%

@{大慧言：世尊！何故諸法亦無捨相？}@
%{大慧说：世尊！为什么诸法也没有舍相呢？}%

@{佛告大慧：觀察自相同相法無法可捨，是故我說一切諸法亦無捨相。}@
%{佛对大慧说：观察自身相同，没有相貌的法，就没有办法可以舍弃，所以我说一切诸法也没有舍相。}%

@{佛告大慧：諸法不滅。}@
%{佛对大慧说：诸法不灭。}%

@{大慧言：世尊！何故一切諸法不滅？}@
%{大慧说：世尊！为什么一切诸法不灭呢？}%

@{佛告大慧：觀一切法自相同相無體相故，是故我說諸法不滅。}@
%{佛对大慧说：观察一切法自身相同，没有实体和相貌，所以我说诸法不灭。}%

@{佛告大慧：諸法無常。}@
%{佛对大慧说：诸法无常。}%

@{大慧言：世尊！何故一切諸法無常？}@
%{大慧说：世尊！为什么一切诸法无常呢？}%

@{佛告大慧：一切諸法常無常想常不生相，是故我說諸法無常。復次，大慧！我說一切諸法無常。}@
%{佛对大慧说：一切诸法，常、无常的念头，经常不生起相貌，所以我说诸法无常。再说，大慧！我说一切诸法无常。}%

@{大慧言：世尊！何故一切諸法無常？}@
%{大慧说：世尊！为什么一切诸法无常呢？}%

@{佛告大慧：以相不生、以不生體相，是故常無常，是故我說諸法無常。}@
%{佛对大慧说：因为相貌不生起，因为不生起实体和相貌，所以是常、无常，所以我说诸法无常。}%
@{爾時世尊重說偈言：}@
%{当时世尊就说了一首偈子：}%

@{記論有四種，直答反質答；}@
%{论述有四种，直接回答，反问回答；}%

@{分別答置答，以制諸外道。}@
%{分别回答，置换回答，用来制伏外道。}%

@{有及非有生，僧佉毘世師，}@
%{有和无的产生，僧佉毘世师，}%

@{而說悉無記，彼作如是說。}@
%{却说一切都是无记，他们这样说。}%

@{正智慧觀察，自性不可得；}@
%{用正智慧观察，自性不可得；}%

@{是故不可說，及說無體相。}@
%{所以不可说，也说无体相。}%

@{爾時聖者大慧菩薩摩訶薩白佛言：世尊！惟願世尊為我等說須陀洹等行差別相，我及一切菩薩摩訶薩等，善知須陀洹等修行相已，如實知須陀洹、斯陀含、阿那含、阿羅漢等，如是如是為眾生說；眾生聞已入二無我相、淨二種障，次第進取地地勝相，得如來不可思議境界修行；得修行處已，如如意寶隨眾生念，受用境界身口意行故。}@
%{当时，大慧菩萨摩诃萨对佛说：世尊！希望世尊能为我们讲解须陀洹等行的差别相，我和一切菩萨摩诃萨等，都已善知须陀洹等的修行相，能如实了解须陀洹、斯陀含、阿那含、阿罗汉等，就如实地为众生说；众生听了以后，就能进入二无我相，净化两种障碍，次第前进，获得每一地的殊胜相，得到如来不可思议境界的修行；得到修行处以后，就像如意宝一样，随着众生的念头，受用境界，身口意行都圆满。}%

@{佛告大慧言：善哉！善哉！善哉大慧！諦聽！諦聽！今為汝說。}@
%{佛对大慧说：善哉！善哉！善哉大慧！仔细听！仔细听！现在就为你讲解。}%

@{大慧白佛言：善哉世尊！唯然聽受。}@
%{大慧对佛说：善哉世尊！我一定仔细听。}%

@{佛告大慧言：大慧！須陀洹有三種果差別。}@
%{佛对大慧说：大慧！须陀洹有三种果位的差别。}%

@{大慧言：何等三種？}@
%{大慧说：哪三种？}%

@{佛告大慧：謂下中上。大慧！何者須陀洹下？謂三有中七返受生。大慧！何者為中？謂三生五生入於涅槃。大慧！何者為上？謂即一生入於涅槃。大慧！是三種須陀洹有三種結，謂下中上。大慧！何者三結？謂身見、疑、戒取。大慧！彼三種結上上勝進得阿羅漢果。大慧！身見有二種。何等為二？一者、俱生；二者、虛妄分別而生，如因緣分別法故。大慧！譬如依諸因緣法相虛妄分別而生實相，彼因緣法中非有非無，以分別有無非實相故；愚癡凡夫執著種種法相，如諸禽獸見於陽炎取以為水。大慧！是名須陀洹分別身見。何以故？以無智故；無始世來虛妄取相故。大慧！此身見垢見人無我乃能遠離。大慧！何者須陀洹俱生身見？所謂自身他身俱見。彼二四陰無色，色陰生時，依於四大及四塵等，彼此因緣和合生色，而須陀洹知已能離有無邪見斷於身見，斷身見已不生貪心。大慧！是名須陀洹身見之相。大慧！何者須陀洹疑相？謂得證法善見相已，先斷身見及於二見分別之心，是故於諸法中不生疑心，復不生心於餘尊者以為尊相為淨不淨故。大慧！是名須陀洹疑相。大慧！何者須陀洹戒取相？謂善見受生處苦相故，是故不取戒相。大慧！戒取者，謂諸凡夫持戒精進種種善行，求樂境界生諸天中，彼須陀洹不取是相而取自身內證迴向進趣勝處，離諸妄想修無漏戒分。大慧！是名須陀洹戒取相。大慧！須陀洹斷三結煩惱離貪瞋癡。}@
%{佛对大慧说：就是下、中、上。大慧！什么是须陀洹下？就是三界中七次轮回受生。大慧！什么是中？就是三生或五生就进入涅槃。大慧！什么是上？就是一生就进入涅槃。大慧！这三种须陀洹，有三种结，也是下、中、上。大慧！哪三种结？就是身见、疑、戒取。大慧！这三种结，上上胜进就能得到阿罗汉果。大慧！身见有两种。哪两种？一是俱生；二是虚妄分别而生，就像因缘分别法一样。大慧！就像依靠因缘法相虚妄分别而生实相，这些因缘法中既不是有也不是无，因为分别有无不是实相；愚痴凡夫执着种种法相，就像禽兽看到阳炎，就以为是水。大慧！这就是须陀洹分别身见。为什么呢？因为无智；从无始世以来虚妄执着法相。大慧！这种身见垢见人无我才能远离。大慧！什么是须陀洹俱生身见？就是自身、他身都见。这二四阴无色，色阴生的时候，依靠四大和四尘等，彼此因缘和合生色，而须陀洹已经知道，就能远离有无邪见，断除身见，断除身见以后，就不生贪心。大慧！这就是须陀洹身见的样子。大慧！什么是须陀洹疑相？就是得到证法，善见相以后，先断除身见和二见分别的心，所以对诸法中不生疑心，也不生心对其他尊者，以为尊相是清净还是不净。大慧！这就是须陀洹疑相。大慧！什么是须陀洹戒取相？就是善见受生处苦相，所以不取戒相。大慧！戒取，就是凡夫持戒精进，做种种善行，求乐境界，生到诸天中，须陀洹不取这种相，而是取自身内证，回向进趣胜处，远离妄想，修无漏戒分。大慧！这就是须陀洹戒取相。大慧！须陀洹断除三结烦恼，远离贪瞋痴。}%

@{大慧白佛言：世尊！世尊說眾多貪，須陀洹離何等貪？}@
%{大慧对佛说：世尊！世尊说了很多贪，须陀洹远离哪种贪？}%

@{佛告大慧：須陀洹遠離與諸女人和合，不為現在樂種未來苦因，遠離打摑嗚抱眄視。大慧！須陀洹不生如是貪心。何以故？以得三昧樂行故。大慧！須陀洹遠離如是等貪，非離涅槃貪。大慧！何者斯陀含果相？謂一往見色相現前生心，非虛妄分別想見，以善見禪修行相故，一往來世間，便斷苦盡入於涅槃，是名斯陀含。大慧！何者阿那含相？謂於過去現在未來色相中生有無心，以見使虛妄分別心，諸結不生，不來故名阿那含。大慧！何者阿羅漢相？謂不生分別思惟可思惟，三昧解脫力通煩惱苦等分別心故，名阿羅漢。}@
%{佛对大慧说：须陀洹远离和女人结合，不为现在快乐而种未来痛苦的因，远离打、掴、呜、抱、眄、视。大慧！须陀洹不生这种贪心。为什么呢？因为得到三昧快乐修行。大慧！须陀洹远离这种贪，不是远离涅槃的贪。大慧！什么是斯陀含果相？就是一往见色相现前生心，不是虚妄分别想见，因为善见禅修行相，一往来世间，就断除苦尽，进入涅槃，这就是斯陀含。大慧！什么是阿那含相？就是对过去、现在、未来色相中，生起有无心，以见使虚妄分别心，诸结不生，不来，所以叫阿那含。大慧！什么是阿罗汉相？就是不生分别思惟，可以思惟，三昧解脱力通达烦恼苦等分别心，所以叫阿罗汉。}%

@{大慧菩薩白佛言：世尊說三種阿羅漢，此說何等羅漢名阿羅漢？世尊為說得決定寂滅羅漢？為發菩提願善根忘善根羅漢？為化應化羅漢？}@
%{大慧菩萨对佛说：世尊说了三种阿罗汉，这里说的是哪种阿罗汉叫阿罗汉？世尊是说得到决定寂灭的阿罗汉？还是发菩提愿，善根忘善根的阿罗汉？还是化应化的阿罗汉？}%
@{佛告大慧：為說得決定寂滅聲聞羅漢，非餘羅漢。}@
%{佛陀对大慧说：我告诉你，决定寂灭的声闻罗汉，不是其他的罗汉。}%

@{大慧！餘羅漢者，謂曾修行菩薩行者，復有應化佛所化羅漢，本願善根方便力故，現諸佛土生大眾中莊嚴諸佛大會眾故。}@
%{大慧！其他的罗汉，指的是曾经修行过菩萨道的行者，还有应化佛所度化的罗汉，因为他们本愿的善根和方便力的缘故，会在各种佛土中出生，在人群中庄严佛的大法会。}%

@{大慧！分別去來說種種事，遠離證果能思惟所思惟可思惟故，以見自心為見所見說得果相。}@
%{大慧！分别去讲述各种事情，远离证果时能思惟的所思惟可思惟，因为这样才能以见自心为见所见，说出果相。}%

@{復次，大慧！若須陀洹生如是心，此是三結；我離三結者。}@
%{再说，大慧！如果须陀洹生起这样的心：这是三结；我远离了三结。}%

@{大慧！是名見三法墮於身見，彼若如是不離三結。}@
%{大慧！这就是看到三种法而堕入身见，如果他这样不远离三结。}%

@{大慧！是故須陀洹不生如是心。}@
%{大慧！所以须陀洹不会生起这样的心。}%

@{復次，大慧！若欲遠離禪無量無色界者，應當遠離自心見相遠離少相，寂滅定三摩跋提相故。}@
%{再说，大慧！如果想要远离禅定无量无色界，就应该远离自心所见的一切相，远离少相，远离寂灭定三摩跋提的相，因为这样才能远离。}%

@{大慧！若不如是，彼菩薩心見諸法，以惟心故。}@
%{大慧！如果不像这样，那个菩萨的心就会看到诸法，因为这是唯心的缘故。}%

@{爾時世尊重說偈言：}@
%{当时，世尊说偈言：}%

@{諸禪四無量，無色三摩提；}@
%{各种禅定，四无量心，无色三摩提；}%

@{少相寂滅定，一切心中無。}@
%{少相寂灭定，一切心中都没有。}%

@{逆流修無漏，及於一往來；}@
%{逆流修行无漏，到达一来往返；}%

@{往來及不還，羅漢心迷沒。}@
%{一来往返，以及不还，罗汉的心迷失了。}%

@{思可思能思，遠離見真諦；}@
%{思惟可思惟，能思惟，远离见真谛；}%

@{惟是虛妄心，能知得解脫。}@
%{只有虚妄的心，才能知道解脱。}%

@{復次，大慧！有二種智。}@
%{再说，大慧！有两种智慧。}%

@{何等為二？}@
%{哪两种呢？}%

@{一者、觀察智；}@
%{一是观察智；}%

@{二者、虛妄分別取相住智。}@
%{二是虚妄分别取相住智。}%

@{大慧！何者觀察智？}@
%{大慧！哪一个是观察智呢？}%

@{謂何等智觀察一切諸法體相，離於四法無法可得，是名觀察智。}@
%{就是哪种智慧观察一切诸法体相，离开四法，没有法可以得到，这就是观察智。}%

@{大慧！何者四法？}@
%{大慧！哪四法呢？}%

@{謂一、異、俱、不俱，是名四法。}@
%{就是一、异、俱、不俱，这就是四法。}%

@{大慧！若離四法一切法不可得。}@
%{大慧！如果离开四法，一切法都不可得。}%

@{大慧！若欲觀察一切法者，當依四法而觀諸法。}@
%{大慧！如果想要观察一切法，就应该依四法来观察诸法。}%

@{大慧！妄想分別取相住智者，所謂執著堅、熱、濕、動，虛妄分別四大相故，執著建立，因譬喻相故建立，非實法以為實。}@
%{大慧！妄想分别取相住智，就是执着坚、热、湿、动，虚妄分别四大相，执着建立，因为比喻相的缘故而建立，不是真实的法却认为是真实的。}%

@{大慧！是名虛妄分別執著取相住持智。}@
%{大慧！这就是虚妄分别执着取相住持智。}%

@{大慧！是名二種智相。}@
%{大慧！这就是两种智慧的相。}%

@{大慧！諸菩薩摩訶薩，畢竟知此二相進趣法無我相，善知真實智地行相，知已即得初地得百三昧，依三昧力見百佛見百菩薩，能知過去未來各百劫事，照百佛世界；照百佛世界已，善知諸地上上智相，以本願力故，能奮迅示現種種神通，於法雲地中依法雨授位，證如來內究竟法身智慧地，依十無量善根願轉，為教化眾生種種應化，自身示現種種光明，以得自身修行證智三昧樂故。}@
%{大慧！诸菩萨摩诃萨，最终知道这两种相，进趣法无我相，善于知道真实智地的行相，知道了以后，就得到初地，得到百三昧，依三昧力，能见百佛，见百菩萨，能知道过去未来各百劫的事情，能照亮百佛世界；照亮百佛世界以后，善于知道诸地上上智的相，因为本愿力的缘故，能奋迅地示现种种神通，在法云地中，依法雨授位，证得如来内究竟法身智慧地，依十无量善根愿力转，为了教化众生，种种应化，自身示现种种光明，因为得到自身修行证智三昧的快乐的缘故。}%

@{復次，大慧！菩薩摩訶薩應善知四大及四塵相。}@
%{再说，大慧！菩萨摩诃萨应该善于知道四大及四尘的相。}%

@{大慧！云何菩薩善知四大及四塵相？}@
%{大慧！菩萨怎样才能善于知道四大及四尘的相呢？}%

@{大慧！菩薩摩訶薩應如是修行，所言實者，謂無四大處，觀察四大本來不生。}@
%{大慧！菩萨摩诃萨应该这样修行，所言真实，就是说没有四大处，观察四大本来不生。}%

@{如是觀已復作是念：言觀察者，惟自心見虛妄覺知，以見外塵無有實物，惟是名字分別心見，所謂三界離於四大及四塵相。}@
%{这样观察以后，又作如是念：说观察，只是自心所见，虚妄的觉知，以见外尘，没有真实的物体，只是名字分别心所见，所谓三界离开四大及四尘的相。}%

@{見如是已離四種見，見清淨法離我我所，住於自相如實法中。}@
%{看到这样以后，离开四种见，看到清净法，离开我我所，住于自相如实法中。}%

@{大慧！住自相如實法中者，謂住建立諸法無生自相法中。}@
%{大慧！住于自相如实法中，就是住于建立诸法无生自相法中。}%

@{大慧！於四大中云何有四塵？}@
%{大慧！在四大中，怎样会有四尘呢？}%

@{大慧！謂妄想分別柔軟濕潤，生內外水大。}@
%{大慧！就是妄想分别柔软湿润，生起内外水大。}%

@{大慧！妄想分別煖增長力，生內外火大。}@
%{大慧！妄想分别温暖增长力量，生起内外火大。}%

@{大慧！妄想分別輕轉動相，生內外風大。}@
%{大慧！妄想分别轻转动相，生起内外风大。}%

@{大慧！妄想分別所有堅相，生內外地大。}@
%{大慧！妄想分别所有坚相，生起内外地大。}%

@{大慧！妄想分別內外共虛空生內外想，以執著虛妄內外邪見，五陰聚落四大及四塵生故。}@
%{大慧！妄想分别内外共同的虚空，生起内外想，因为执着虚妄的内外邪见，五阴聚落四大及四尘生起的缘故。}%

@{佛告大慧：識能執著種種境界樂求異道，取彼境界故。}@
%{佛陀对大慧说：识能执着种种境界，乐求异道，取彼境界的缘故。}%

@{大慧！四大有四，謂色、香、味、觸。}@
%{大慧！四大有四，就是色、香、味、触。}%

@{大慧！四大無因。}@
%{大慧！四大没有因。}%

@{何以故？}@
%{为什么呢？}%

@{謂地自體形相長短不生四大相故。}@
%{就是说，地本身的形相，长短，不生四大相的缘故。}%

@{大慧！依形相大小上下容貌而生諸法，不離形相大小長短而有法故。}@
%{大慧！依形相大小上下容貌而生诸法，不离开形相大小长短而有法的缘故。}%

@{是故，大慧！外道虛妄分別四大及四塵，非我法中如此分別。}@
%{所以，大慧！外道虚妄分别四大及四尘，不是我的法中这样分别的。}%

@{復次，大慧！我為汝說五陰體相。}@
%{再说，大慧！我为你讲五阴的体相。}%

@{大慧！何者五陰相？}@
%{大慧！哪是五阴相呢？}%

@{謂色、受、想、行、識。}@
%{就是色、受、想、行、识。}%

@{大慧！四陰無色相，謂受、想、行、識。}@
%{大慧！四阴无色相，就是受、想、行、识。}%

@{大慧！色依四大生，四大彼此不同相。}@
%{大慧！色依四大而生，四大彼此不同的相。}%

@{大慧！無色相法同如虛空，云何得成四種數相？}@
%{大慧！无色相法如同虚空，怎样能成为四种数相呢？}%

@{大慧！譬如虛空離於數相，而虛妄分別此是虛空。}@
%{大慧！譬如虚空离开数相，而虚妄分别这是虚空。}%

@{大慧！陰之數相離於諸相，離有無相離於四相；愚癡凡夫說諸數相，非謂聖人。}@
%{大慧！阴的数相离开诸相，离开有无相，离开四相；愚痴的凡夫说诸数相，不是说圣人。}%

@{大慧！我說諸相如幻種種形相離一二相依假名說，如夢鏡像不離所依。}@
%{大慧！我说诸相如同幻，种种形相离开一二相，依假名说，如同梦境，如同镜子中的影像，不离开所依。}%

@{大慧！如聖人智修行分別見五陰虛妄。}@
%{大慧！如同圣人智慧修行分别看到五阴是虚妄的。}%

@{大慧！是名五陰無五陰體相。}@
%{大慧！这就是五阴没有五阴的体相。}%

@{大慧！汝今應離如是虛妄分別之相，離如是已為諸菩薩說離諸法相寂靜之法，為遮外道諸見之相。}@
%{大慧！你现在应该远离这样虚妄分别的相，远离了以后，为诸菩萨说离开诸法相的寂静之法，为了遮止外道诸见的相。}%

@{大慧！說寂靜法得證清淨無我之相入遠行地，入遠行地已得無量三昧自在如意生身故，以得諸法如幻三昧故，以得自在神通力修行進趣故，隨一切眾生自在用如大地故。}@
%{大慧！说寂静法，得到证得清净无我的相，进入远行地，进入远行地以后，得到无量三昧，自在如意生身，因为得到诸法如幻三昧，因为得到自在神通力，修行进趣，随一切众生自在运用，如同大地。}%

@{大慧！譬如大地一切眾生隨意而用。}@
%{大慧！譬如大地，一切众生随心而用。}%

@{大慧！菩薩摩訶薩隨眾生用亦復如是。}@
%{大慧！菩萨摩诃萨随众生运用，也如同这样。}%
@{復次，大慧！外道說有四種涅槃。何等為四？一者、自體相涅槃；二者、種種相有無涅槃；三者、自覺體有無涅槃；四者、諸陰自相同相斷相續體涅槃。大慧！是名外道四種涅槃，非我所說。大慧！我所說者，見虛妄境界分別識滅名為涅槃。}@
%{接着，大慧！外道说有四种涅槃。这四种是什么呢？第一种，自相涅槃；第二种，种种相的有没有涅槃；第三种，自觉体的有没有涅槃；第四种，诸阴自相相同相断相续体的涅槃。大慧！这就是外道所说的四种涅槃，不是我所说的。大慧！我所说的，是见到虚妄境界，分别识灭了，这叫做涅槃。}%

@{大慧白佛言：世尊！世尊可不說八種識耶？}@
%{大慧对佛说：世尊！世尊您可不可以讲一下八种识呢？}%

@{佛告大慧：我說八種識。}@
%{佛告诉大慧：我讲八种识。}%

@{大慧言：若世尊說八種識者，何故但言意識轉滅，不言七識轉滅？}@
%{大慧说：如果世尊您讲八种识，为什么只说意识转灭，不说七识转灭呢？}%

@{佛告大慧：以依彼念觀有故，轉識滅七識亦滅。復次，大慧！意識執著取境界生，生已種種熏習增長阿梨耶識，共意識故，離我我所相，著虛妄空而生分別。大慧！彼二種識無差別相，以依阿梨耶識因觀自心見境，妄想執著生種種心，猶如束竹迭共為因，如大海波，以自心見境界風吹而有生滅。是故，大慧！意識轉滅七種識轉滅。}@
%{佛告诉大慧：因为依着那个念头观察有，转识灭了，七识也灭了。接着，大慧！意识执着取境界而生，生了以后，种种熏习增长阿赖耶识，和意识一起，离开我我所相，执着虚妄空而生分别。大慧！这两种识没有差别相，因为依着阿赖耶识，因观察自心见境，妄想执着生种种心，就像捆绑的竹子互相成为原因一样，就像大海的波浪，因为自心见境界，风吹而有生灭。所以，大慧！意识转灭，七种识也转灭。}%

@{爾時世尊重說偈言：}@
%{这时世尊就说偈子：}%

@{我不取涅槃，亦不捨作相；}@
%{我不取涅槃，也不舍作相；}%

@{轉滅虛妄心，故言得涅槃。}@
%{转灭虚妄心，所以说得到涅槃。}%

@{依彼因及念，意趣諸境界；}@
%{依着那个因和念头，意趣诸境界；}%

@{識與心作因，為識之所依。}@
%{识和心作因，成为识所依。}%

@{如水流枯竭，波浪則不起；}@
%{就像水流枯竭，波浪就不会起来；}%

@{如是意識滅，種種識不生。}@
%{就像这样意识灭了，种种识就不会生。}%

@{復次，大慧！我為汝說虛妄分別法體差別相，汝及諸菩薩摩訶薩，善分別知虛妄法體差別之相，離分別所分別法，善知自身內修行法，遠離外道能取可取境界，遠離種種虛妄分別因緣法體相，遠離已不復分別虛妄之相。大慧！何者虛妄分別法體差別之相？大慧！虛妄分別自體差別相有十二種。何等為十二？一者、言語分別；二者、可知分別；三者、相分別；四者、義分別；五者、實體分別；六者、因分別；七者、見分別；八者、建立分別；九者、生分別；十者、不生分別；十一者、和合分別；十二者、縛不縛分別。大慧！是名分別自體相差別法相。大慧！言語分別者，謂樂著種種言語美妙音聲。大慧！是名言語分別。大慧！可知分別者，謂作是思惟應有前法實事之相，聖人修行知，依彼法而生言語。如是分別，大慧！是名可知分別。大慧！相分別者，謂即彼可知境界中熱濕動堅種種相執以為實，如空陽焰諸禽獸見生於水想。大慧！是名相分別。大慧！義分別者，謂樂金銀等種種實境界。大慧！是名義分別。大慧！自體分別者，謂專念有法自體形相，此法如是如是不異，非正見見分別。大慧！是名自體分別。大慧！因分別者，謂何等何等因，何等何等緣，有無了別因相生了別相。大慧！是名因分別。大慧！見分別者，謂有無一異俱不俱，邪見外道執著分別。大慧！是名見分別。大慧！建立分別者，謂取我我所相說虛妄法。大慧！是名建立分別。大慧！生分別者，謂依眾緣有無法中生執著心。大慧！是名生分別。大慧！不生分別者，謂一切法本來不生，以本無故依因緣有而無因果。大慧！是名無生分別。大慧！和合分別者，謂何等何等法和合，如金縷共；何等何等法和合，如金縷和合。大慧！是名和合分別。大慧！縛不縛分別者，謂縛因執著如所縛。大慧！如人方便結繩作結，結已還解。大慧！是名縛不縛分別。大慧！是名虛妄分別法體差別之相，以此虛妄分別法體差別之相，一切凡夫執著有無故，執著法相種種因緣。是故，大慧！分別法體差別之相，見種種法執著為實，如依於幻見種種事，凡夫分別知異於幻有如是法。大慧！我於種種法中不異幻說，亦非不異。何以故？若幻異於種種法者，不應因幻而生種種；若幻即是種種法者，不應異見此是幻此是種種而見差別，是故我說不異非不異。是故，大慧！汝及諸菩薩摩訶薩，莫分別幻有實無實。}@
%{接着，大慧！我为你讲虚妄分别法体差别相，你以及诸菩萨摩诃萨，要善于分别，知道虚妄法体差别相，离开分别所分别的法，善于知道自身内修行法，远离外道能取可取境界，远离种种虚妄分别因缘法体相，远离已经不再分别虚妄的相。大慧！什么叫做虚妄分别法体差别相呢？大慧！虚妄分别自体差别相有十二种。这十二种是什么呢？第一种，言语分别；第二种，可知分别；第三种，相分别；第四种，义分别；第五种，实体分别；第六种，因分别；第七种，见分别；第八种，建立分别；第九种，生分别；第十种，不生分别；第十一种，和合分别；第十二种，缚不缚分别。大慧！这就是分别自体相差别法相。大慧！言语分别，就是说喜欢执着种种言语美妙的音声。大慧！这就是言语分别。大慧！可知分别，就是说作这样思惟，应该有前面法实事之相，圣人修行知道，依着那个法而生言语。这样分别，大慧！这就是可知分别。大慧！相分别，就是说就在那个可知境界中，热湿动坚种种相执着以为真实，就像空中的阳焰，诸禽兽看见生于水想一样。大慧！这就是相分别。大慧！义分别，就是说喜欢金银等种种真实境界。大慧！这就是义分别。大慧！自体分别，就是说专门念有法自体形相，这个法就是这样，这样，不异，不是正见见分别。大慧！这就是自体分别。大慧！因分别，就是说，什么什么因，什么什么缘，有无了别因相生了别相。大慧！这就是因分别。大慧！见分别，就是说，有无一异，俱不俱，邪见外道执着分别。大慧！这就是见分别。大慧！建立分别，就是说，取我我所相，说虚妄法。大慧！这就是建立分别。大慧！生分别，就是说，依着种种缘有无法中生执着心。大慧！这就是生分别。大慧！不生分别，就是说，一切法本来不生，因为本来没有，所以依着因缘有而没有因果。大慧！这就是无生分别。大慧！和合分别，就是说，什么什么法和合，就像金线在一起；什么什么法和合，就像金线和合。大慧！这就是和合分别。大慧！缚不缚分别，就是说，因为执着，就像被束缚一样。大慧！就像人方便地结绳作结，结了以后又解开。大慧！这就是缚不缚分别。大慧！这就是虚妄分别法体差别相，因为这个虚妄分别法体差别相，一切凡夫执着有无，所以执着法相种种因缘。所以，大慧！分别法体差别相，看见种种法执着为真实，就像依着幻觉看见种种事情，凡夫分别知道，异于幻觉，有这样法。大慧！我在种种法中，不异幻说，也不是不异。为什么呢？如果幻觉异于种种法，就不应该因为幻觉而生种种；如果幻觉就是种种法，就不应该异见，这个是幻觉，这个是种种，而看见差别，所以我说不异，也不不是不异。所以，大慧！你以及诸菩萨摩诃萨，不要分别幻觉有真实，没有真实。}%

@{爾時大慧菩薩摩訶薩復白佛言：世尊！惟願為說自身內證聖智修行相及一乘法，不由於他遊行一切諸佛國土通達佛法。}@

%{引导众生，所以分别说诸乘。}%

@{解脫有三種，及二法無我；}@
%{解脱有三種，以及二法无我；}%

@{不離二種障，遠離真解脫。}@
%{不离开两种障，远离真解脱。}%

@{譬如海浮木，當隨波浪轉；}@
%{譬如海上的浮木，随着波浪转动；}%

@{諸聲聞亦然，相風所漂蕩。}@
%{诸声闻也是一样，被相风所漂荡。}%

@{離諸隨煩惱，熏習煩惱縛；}@
%{离开随顺烦恼，熏习烦恼束缚；}%

@{味著三昧樂，安住無漏界。}@
%{贪着三昧的快乐，安住无漏界。}%

@{無有究竟趣，亦復不退還；}@
%{没有究竟的趣向，也不退还；}%

@{得諸三昧身，無量劫不覺。}@
%{得到诸三昧身，无量劫不觉。}%

@{譬如惛醉人，酒消然後悟；}@
%{譬如昏醉的人，酒消了以后才醒悟；}%

@{得佛無上體，是我真法身。}@
%{得到佛无上体，是我的真法身。}%
