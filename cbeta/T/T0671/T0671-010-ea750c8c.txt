@{入楞伽經卷第十元魏天竺三藏菩提留支譯}@
%{入楞伽經卷第十元魏天竺三藏菩提留支譯}%

@{總品第十八之二}@
%{總品第十八之二}%

@{依諸佛住持，諸願力清淨；}@
%{按照诸佛的住持，诸愿力清净；}%

@{受職及三昧，功德及十地。}@
%{接受职务和三昧，功德和十地。}%

@{虛空及兔角，及與石女兒；}@
%{虚空和兔子角，以及石头女儿；}%

@{分別法如是，無而說名字。}@
%{分别法如是，没有却说名字。}%

@{因熏種世間，非有非無處；}@
%{因为熏染而种下世间，既非有也非无处；}%

@{能見得解脫，解於法無我。}@
%{能够看见得到解脱，明白于法无我。}%

@{實體分別名，他體從因生；}@
%{实体分别名字，其他体从因缘而生；}%

@{我說是成就，諸經常說是。}@
%{我说这是成就，诸经经常说是。}%

@{字句名身等，於名身勝法；}@
%{字句、名字、身体等等，在名字、身体胜于法；}%

@{愚癡人分別，如象沒深泥。}@
%{愚痴的人分别，如同大象陷入泥潭。}%

@{天乘及梵乘，及於聲聞乘；}@
%{天乘和梵乘，以及声闻乘；}%

@{如來及緣覺，我說如是乘。}@
%{如来和缘觉，我说这是乘。}%

@{諸乘不可盡，有心如是生；}@
%{诸乘不可尽，有心如是生；}%

@{心轉滅亦無，無乘及乘者。}@
%{心转灭也无，无乘和乘者。}%

@{心分別及識，意及於意識；}@
%{心分别和识，意和意识；}%

@{阿梨耶三有，思惟心異名。}@
%{阿梨耶三有，思维心异名。}%

@{命及於煖識，阿梨耶命根；}@
%{命和暖识，阿梨耶命根；}%

@{意及於意識，是分別異名。}@
%{意和意识，是分别异名。}%

@{心住持於身，意常覺諸法；}@
%{心住持于身，意常觉诸法；}%

@{識自心境界，共於識分別。}@
%{识来自心境界，共同于识分别。}%

@{我說愛是母，無明以為父；}@
%{我说爱是母亲，无明以为父亲；}%

@{識覺諸境界，是故說名佛。}@
%{识觉诸境界，是故说名为佛。}%

@{諸使是怨家，眾和合是陰；}@
%{诸使是怨家，众和合是阴；}%

@{無於相續體，斷彼名無間。}@
%{无于相续体，断彼名为无间。}%

@{二無我煩惱，及二種無我；}@
%{二无我烦恼，以及两种无我；}%

@{不可思議變，無生死名佛。}@
%{不可思议变，无生死名为佛。}%

@{意相應法體，我法是內身；}@
%{意相应法体，我的法是内身；}%

@{若能如是見，彼不隨妄覺。}@
%{若能如是见，彼不随妄觉。}%

@{實無於諸法，如愚癡分別；}@
%{实无于诸法，如同愚痴分别；}%

@{依虛妄無法，云何得解脫？}@
%{依虚妄无法，云何得解脱？}%

@{生滅和合縛，見於有為法；}@
%{生灭和合缚，看见有为法；}%

@{增長於二見，不失因緣法。}@
%{增长于二见，不失因缘法。}%

@{芭蕉夢幻等，是世間如是；}@
%{芭蕉梦幻等等，是世间如是；}%

@{惟是一法實，涅槃離意識。}@
%{唯是一法实，涅槃离意识。}%

@{有貪及與瞋，及有癡無人；}@
%{有贪以及瞋，以及有痴无人；}%

@{從愛生諸陰，陰有亦如夢。}@
%{从爱生诸阴，阴有亦如梦。}%

@{何等夜證法，何等夜入滅，}@
%{何等夜证法，何等夜入灭，}%

@{於此二中間，我不說一字。}@
%{于此二中间，我不说一字。}%

@{內身證於法，我依如是說；}@
%{内身证于法，我依如是说；}%

@{彼佛及我身，無有說勝法。}@
%{彼佛及我身，无有说胜法。}%

@{實有神我物，五陰離彼相；}@
%{实有神我物，五阴离彼相；}%

@{陰體是實有，彼陰中無我。}@
%{阴体是实有，彼阴中无我。}%

@{各各見分別，隨煩惱及使；}@
%{各各见分别，随烦恼及使；}%

@{得世間自心，離苦得解脫。}@
%{得世间自心，离苦得解脱。}%

@{諸因及因緣，世間如是生；}@
%{诸因及因缘，世间如是生；}%

@{是四法相應，彼不住我教。}@
%{是四法相应，彼不住我教。}%

@{非有無生法，離有無不生；}@
%{非有無生法，离有無不生；}%

@{愚云何分別，從因及諸緣。}@
%{愚云何分别，从因及诸缘。}%

@{有無四句離，若能見世間；}@
%{有無四句离，若能见世间；}%

@{爾時轉心識，即得無我法。}@
%{尔时转心识，即得无我法。}%

@{諸法本不生，是故因緣生；}@
%{诸法本不生，是故因缘生；}%

@{諸緣即是果，果中生於有。}@
%{诸缘即是果，果中生于有。}%

@{果中生二種，果中應有二；}@
%{果中生二种，果中应有二；}%

@{而二中無果，果中不見物。}@
%{而二中无果，果中不见物。}%

@{離於觀可觀，若見有為法；}@
%{离于观可观，若见有为法；}%

@{離心惟是心，故我說惟心。}@
%{离心惟是心，故我說惟心。}%

@{量實體形相，離於緣實體；}@
%{量实体形相，离于缘实体；}%

@{究竟第一淨，我說如是量。}@
%{究竟第一净，我说如是量。}%

@{如假名為我，無實法可見；}@
%{如假名为我，无实法可见；}%

@{如是陰陰體，是假名非實。}@
%{如是阴阴体，是假名非实。}%

@{平等有四種，相因及於生；}@
%{平等有四种，相因以及生；}%

@{無我亦平等，四修行者法。}@
%{无我亦平等，四修行者法。}%

@{轉諸一切見，分別可分別；}@
%{转诸一切见，分别可分别；}%

@{不見及不生，故我說惟心。}@
%{不见及不生，故我說惟心。}%

@{無法亦非無，離於有無體；}@
%{无法亦非无，离于有無体；}%

@{真如離於心，故我說惟心。}@
%{真如离于心，故我說惟心。}%

@{真如空實際，涅槃及法界；}@
%{真如空实际，涅槃及法界；}%

@{意生身及心，故我說惟心。}@
%{意生身及心，故我說惟心。}%

@{分別依熏習，種種生種種；}@
%{分别依熏习，种种生种种；}%

@{眾生心見外，故我說惟心。}@
%{众生心见外，故我說惟心。}%

@{可見無外物，見心種種見；}@
%{可见无外物，见心种种见；}%

@{身資生住持，故我說惟心。}@
%{身资生住持，故我說惟心。}%

@{諸聲聞盡智，諸佛如來生；}@
%{诸声闻尽智，诸佛如来生；}%

@{一切辟支佛，無和合而生。}@
%{一切辟支佛，无和合而生。}%

@{無外諸色相，自心見外法；}@
%{无外诸色相，自心见外法；}%

@{覺知於自心，愚分別有為。}@
%{觉知于自心，愚分别有为。}%

@{愚人不知外，自心種種見；}@
%{愚人不知外，自心种种见；}%

@{譬喻遮愚人，著於四種法。}@
%{譬喻遮愚人，著于四种法。}%

@{無因無分別，譬喻五種論；}@
%{无因无分别，譬喻五种论；}%

@{自心體形相，能知是黠慧。}@
%{自心体形相，能知是黠慧。}%

@{依分別可別，此是分別相；}@
%{依分别可别，此是分别相；}%

@{依止於分別，分別於現生。}@
%{依止于分别，分别于现生。}%

@{一一別和合，是一種子因；}@
%{一一别和合，是一种子因；}%

@{客二法是二，故人心不生。}@
%{客二法是二，故人心不生。}%

@{分別心心法，住於二界中；}@
%{分别心心法，住于二界中；}%

@{現生於諸法，彼體是虛妄。}@
%{现生于诸法，彼体是虚妄。}%

@{因依現和合，故有十二入；}@
%{因依现和合，故有十二入；}%

@{依因觀和合，我不說是法。}@
%{依因观和合，我不说是法。}%

@{如鏡中見像，眼𪾼見毛輪；}@
%{如同镜中见像，眼𪾼见毛轮；}%

@{如是依熏心，愚癡人心見。}@
%{如是依熏心，愚痴人心见。}%

@{共分別可別，而生於分別；}@
%{共分别可别，而生于分别；}%

@{如外道分別，無如是外相。}@
%{如同外道分别，无如是外相。}%

@{如人不識繩，而取以為蛇；}@
%{如同人不识绳，而取以为蛇；}%

@{不識自心義，分別於外法。}@
%{不识自心义，分别于外法。}%

@{而乘於乘體，離於一二中；}@
%{而乘于乘体，离于一二中；}%

@{以分別於乘，是自心過失。}@
%{以分别于乘，是自心过失。}%

@{依何法何體，分別不能見；}@
%{依何法何体，分别不能见；}%

@{不得言彼無，諸法體如是。}@
%{不得言彼无，诸法体如是。}%

@{依有故言無，依無故言有；}@
%{依有故言无，依无故言有；}%

@{故不得言無，亦不得言有。}@
%{故不得言无，亦不得言有。}%

@{即分別分別，此非彼法體；}@
%{即分别分别，此非彼法体；}%

@{云何見無體，而生於分別？}@
%{云何见无体，而生于分别？}%

@{色體無色身，如瓶及疊等；}@
%{色体无色身，如同瓶和叠等等；}%

@{可見是無法，云何有分別？}@
%{可见是无法，云何有分别？}%

@{若分別是迷，有為法無始；}@
%{若分别是迷，有为法无始；}%

@{何法迷眾生？牟尼為我說。}@
%{何法迷众生？牟尼为我说。}%

@{諸法無法體，而說惟是心；}@
%{诸法无法体，而说惟是心；}%

@{不見於自心，而起於分別。}@
%{不见于自心，而起于分别。}%

@{若分別是無，如愚癡分別；}@
%{若分别是无，如同愚痴分别；}%

@{彼法無異體，而智不能覺。}@
%{彼法无异体，而智不能觉。}%

@{若聖有彼法，非凡妄分別；}@
%{若圣有彼法，非凡妄分别；}%

@{若聖妄有彼，聖愚癡無別。}@
%{若圣妄有彼，圣愚痴无别。}%

@{聖人無迷惑，以得心清淨；}@
%{圣人无迷惑，以得心清净；}%

@{愚人無信心，故分別分別。}@
%{愚人无信心，故分别分别。}%

@{如母為諸子，虛空將果來；}@
%{如同母亲为诸子，虚空将果来；}%

@{汝取果莫啼，兒取種種果。}@
%{汝取果莫啼，儿取种种果。}%

@{我於諸眾生，分別種種果；}@
%{我于诸众生，分别种种果；}%

@{令貪種種說，離有無朋黨。}@
%{令贪种种说，离有無朋党。}%

@{若本無法體，非因非從因；}@
%{若本无法体，非因非从因；}%

@{本不生始生，亦無其身體。}@
%{本不生始生，亦无其身体。}%

@{無身亦非生，離因緣無處；}@
%{无身亦非生，离因缘无处；}%

@{生滅諸法體，離因緣處無。}@
%{生灭诸法体，离因缘处无。}%

@{略觀察如是，有無非餘處；}@
%{略观察如是，有無非余处；}%

@{從因緣生法，智者莫分別。}@
%{从因缘生法，智者莫分别。}%

@{說一體二體，外道愚癡說；}@
%{说一体二体，外道愚痴说；}%

@{世間如幻夢，不從因緣生。}@
%{世间如幻梦，不从因缘生。}%

@{依言語境界，大乘無上法；}@
%{依言语境界，大乘无上法；}%

@{我依了義說，而愚癡不覺。}@
%{我依了义说，而愚痴不觉。}%

@{聲聞及外道，依嫉妬說法；}@
%{声闻及外道，依嫉妒说法；}%

@{於義不相應，以依妄覺說。}@
%{于义不相應，以依妄觉说。}%

@{相體及形相，名是四種法；}@
%{相体及形相，名是四种法；}%

@{觀於如是法，故生於分別。}@
%{观于如是法，故生于分别。}%

@{分別一二多，彼隨梵天縛；}@
%{分别一二多，彼随梵天缚；}%

@{日月及諸天，是見非我子。}@
%{日月及诸天，是见非我子。}%

@{聖人見正法，以如實修行；}@
%{圣人见正法，以如实修行；}%

@{能轉虛妄相，亦離於去來。}@
%{能转虚妄相，亦离于去来。}%

@{此是解脫印，我教諸佛子；}@
%{此是解脱印，我教诸佛子；}%

@{離於有無法，亦離去來相。}@
%{离于有無法，亦离去来相。}%

@{轉種種色識，若滅一切業；}@
%{转种种色识，若灭一切业；}%

@{不應常無常，無世間生法。}@
%{不应常无常，无世间生法。}%

@{於轉時若滅，色離於彼處；}@
%{于转时若灭，色离于彼处；}%

@{離於無過失，業住阿梨耶。}@
%{离于无过失，业住阿梨耶。}%

@{色是滅體相，識中有亦爾；}@
%{色是灭体相，识中有亦尔；}%

@{色識共和合，而不失諸業。}@
%{色识共和合，而不失诸业。}%

@{若共彼和合，眾生失諸業；}@
%{若共彼和合，众生失诸业；}%

@{若滅和合業，無縛無涅槃。}@
%{若灭和合业，无缚无涅槃。}%

@{若共於彼滅，生於世間中；}@
%{若共于彼灭，生于世间中；}%

@{色亦共和合，無差別應有。}@
%{色亦共和合，无差别应有。}%

@{有別亦無別，但是心分別；}@
%{有别亦无别，但是心分别；}%

@{諸法無滅體，離有無朋黨。}@
%{诸法无灭体，离有無朋党。}%

@{假名因緣法，迭共無差別；}@
%{假名因缘法，迭共无差别；}%

@{如色中無常，迭共生諸法。}@
%{如色中无常，迭共生诸法。}%

@{離於彼此相，分別不可知；}@
%{离于彼此相，分别不可知；}%

@{無有有何成？如色中無常。}@
%{无有何成？如色中无常。}%

@{若善見分別，即不起他力；}@
%{若善见分别，即不起他力；}%

@{見於他力法，亦不起分別。}@
%{见于他力法，亦不起分别。}%

@{若滅於分別，是滅於我法；}@
%{若灭于分别，是灭于我法；}%

@{於我法中作，亦謗於有無。}@
%{于我法中作，亦谤于有無。}%

@{是諸謗法人，於何時中有？}@
%{是诸谤法人，于何时中有？}%

@{是滅我法輪，不得共彼語。}@
%{是灭我法轮，不得共彼语。}%

@{智者不共語，不共比丘法；}@
%{智者不共语，不共比丘法；}%

@{已滅於分別，妄見離有無。}@
%{已灭于分别，妄见离有無。}%

@{見如毛輪幻，如夢乾闥婆；}@
%{见如毛轮幻，如同梦乾闥婆；}%

@{亦見如陽焰，時見於有無。}@
%{亦见如阳焰，时见于有無。}%

@{彼人不覺佛，若有人攝彼；}@
%{彼人不觉佛，若有人摄彼；}%

@{彼人墮二邊，亦壞於餘人。}@
%{彼人堕二边，亦坏于余人。}%

@{若知寂靜法，是實修行者；}@
%{若知寂静法，是实修行者；}%

@{離於有無法，應攝取彼人。}@
%{离于有無法，应摄取彼人。}%

@{如有處可出，金銀諸珍寶；}@
%{如同有处可出，金银诸珍宝；}%

@{無業作種種，而眾生受用。}@
%{无业作种种，而众生受用。}%

@{眾生真如性，不由於業有；}@
%{众生真如性，不由于业有；}%

@{不見故無業，亦非作業生。}@
%{不见故无业，亦非作业生。}%

@{諸法無法體，如聖人分別；}@
%{诸法无法体，如同圣人分别；}%

@{而有於諸法，如愚癡分別。}@
%{而有于诸法，如同愚痴分别。}%

@{若法無如是，如愚癡分別；}@
%{若法无如是，如同愚痴分别；}%

@{無有一切法，眾生亦無染。}@
%{无有一切法，众生亦无染。}%

@{諸法依心有，煩惱亦如是；}@
%{诸法依心有，烦恼亦如是；}%

@{生死諸世間，隨於諸根轉。}@
%{生死诸世间，随于诸根转。}%

@{無明愛和合，而生於諸身；}@
%{无明爱和合，而生于诸身；}%

@{餘人恒無法，如愚癡分別。}@
%{余人恒无法，如同愚痴分别。}%

@{若人法不生，行者不見根；}@
%{若人法不生，行者不见根；}%

@{若諸法是無，能作世間因。}@
%{若诸法是无，能作世间因。}%

@{愚人離於作，自然應解脫；}@
%{愚人离于作，自然应解脱；}%

@{愚聖無差別，有無云何成？}@

@{聖人無法體，以修三解脫；}@
%{圣人无法体，以修三解脱；}%

@{五陰及人法，有同有異相。}@
%{五阴及人法，有同有异相。}%

%{诸因缘及根，我为声闻说；}%

%{无因惟于心，妙事及诸地。}%

%{内身真如净，为诸佛子说；}%

@{於未來世有，謗於我法輪。}@
%{于未来世有，谤于我法轮。}%

%{身披于袈裟，说有無诸法；}%

%{无法因缘有，是圣人境界。}%

@{分別無法體，妄覺者分別；}@
%{分别无法体，妄觉者分别；}%

@{未來世有人，噉糠愚癡種。}@
%{未来世有人，噉糠愚痴种。}%

@{無因而邪見，破壞世間人；}@
%{无因而邪见，破坏世间人；}%

@{從微塵生世，而微塵無因。}@
%{从微尘生世，而微尘无因。}%

@{九種物是常，邪見如是說；}@
%{九种物是常，邪见如是说；}%

@{從物生於物，功德生功德。}@
%{从物生于物，功德生功德。}%

@{此法異於法，分別是體是；}@
%{此法异于法，分别是体是；}%

@{若本無始生，世間應有本。}@
%{若本无始生，世间应有本。}%

@{我說於世間，無有於本際；}@
%{我说于世间，无有于本际；}%

@{三界諸眾生，是本無始生。}@
%{三界诸众生，是本无始生。}%

@{狗駝驢無角，必應生無疑；}@
%{狗驼驴无角，必应生无疑；}%

@{眼本無始有，色及識亦爾。}@
%{眼本无始有，色及识亦尔。}%

@{席冠白疊等，泥團中應生；}@
%{席冠白叠等等，泥团中应生；}%

@{於疊中無瓶，蒲中亦無疊。}@
%{于叠中无瓶，蒲中亦无叠。}%

@{一於一中實，何故因不生？}@
%{一于一中实，何故因不生？}%

@{即命即是身，是本無始生。}@
%{即命即是身，是本无始生。}%

@{此是他說法，我說諸法異；}@
%{此是他说法，我说诸法异；}%

@{我領因緣法，然後遮他法。}@
%{我领因缘法，然后遮他法。}%

@{遮彼邪見者，後說於自法；}@
%{遮彼邪见者，后说于自法；}%

@{故領外道法，然後說正法。}@
%{故领外道法，然后说正法。}%

@{恐諸弟子迷，立於有無法；}@
%{恐诸弟子迷，立于有無法；}%

@{從勝人生世，迦毘羅惡意。}@
%{从胜人生世，迦毘罗恶意。}%

@{為諸弟子說，諸功德轉變；}@
%{为诸弟子说，诸功德转变；}%

@{非實非不實，非從緣即緣。}@
%{非实非不实，非从缘即缘。}%

@{以無諸因緣，無實法不生；}@
%{以无诸因缘，无实法不生；}%

@{離於有無法，離因亦離緣。}@
%{离于有無法，离因亦离缘。}%

@{離於生滅法，自法離可見；}@
%{离于生灭法，自法离可见；}%

@{世間如幻夢，離諸因緣法。}@
%{世间如幻梦，离诸因缘法。}%

@{立因緣者見，是故生分別；}@
%{立因缘者见，是故生分别；}%

@{如禽獸愛水，乾闥婆毛輪。}@
%{如同禽兽爱水，乾闥婆毛轮。}%

@{離於有無法，離因及於緣；}@
%{离于有無法，离因及于缘；}%

@{見三有無因，如是見淨心。}@
%{见三有無因，如是见净心。}%

@{何等人無事？但有於內心；}@
%{何等人無事？但有于内心；}%

@{遠離於心事，不得說惟心。}@
%{远离于心事，不得说惟心。}%

@{若觀於外事，眾生起於心；}@
%{若观于外事，众生起于心；}%

@{云何心無因？不得說惟心。}@
%{云何心无因？不得说惟心。}%

@{真如惟心有，何人無聖法？}@
%{真如惟心有，何人无圣法？}%

@{有及於非有，彼不解我法。}@
%{有及于非有，彼不解我法。}%

@{能取可取法，若心如是生；}@
%{能取可取法，若心如是生；}%

@{此是世間心，不應說惟心。}@
%{此是世间心，不应说惟心。}%

@{身資生住持，若如夢中生；}@
%{身资生住持，若如同梦中生；}%

@{應有二種心，而心無二相。}@
%{应有两种心，而心无二相。}%

@{如刀不自割，指亦不自指；}@
%{如同刀不自割，指亦不自指；}%

@{如心不自見，其事亦如是。}@
%{如同心不自见，其事亦如是。}%

@{非他非因緣，分別分別事；}@
%{非他非因缘，分别分别事；}%

@{五法及二心，寂靜無如是。}@
%{五法及二心，寂静无如是。}%

%{能生及于生，及二种法相；}%

%{我意无能生，说法无自相。}%

%{以无外分别，离心更无法。}%

%{以心无生因，而见于外义。}%
