@{佛說須真天子經卷第二}@
%{佛说须真天子经卷第二}%

@{西晉月氏三藏竺法護譯}@
%{西晋月氏三藏竺法护译}%

@{答法議品第二}@
%{答法议品第二}%

@{須真天子則語文殊師利童子言：如來為我發遣三十二事章句法品，惟願重為廣說令解。云何菩薩於此大乘意不忘信？}@
%{须真天子就对文殊师利童子说：如来为我派遣了三十二事章句法品，我只希望您再详细地讲解，让我明白。菩萨在这一大乘法中，如何才能不忘信呢？}%

@{文殊師利答言：心自審信不隨他教故。}@
%{文殊师利回答说：因为内心自己审察而相信，不随从别人的教导。}%

@{天子復問：云何菩薩所作堅強？}@
%{天子又问：菩萨所作如何才能坚强呢？}%

@{答言：降棄諸欲故。}@
%{回答说：因为舍弃了所有欲望。}%

@{天子復問：云何菩薩得最禑之福？}@
%{天子又问：菩萨如何才能得到最殊胜的福报呢？}%

@{答言：了知法界而不動故。}@
%{回答说：因为明了法界而不动摇。}%

@{天子復問：云何菩薩得無礙行？}@
%{天子又问：菩萨如何才能获得无障碍的修行呢？}%

@{答言：不為諸入之所惑故。}@
%{回答说：因为不被各种境界所迷惑。}%

@{天子復問：云何菩薩去離冥塵？}@
%{天子又问：菩萨如何才能远离冥尘呢？}%

@{答曰：知諸法界本皆淨故。}@
%{回答说：因为知道所有法界本来都是清净的。}%

@{天子復問：云何菩薩入眾勇辯離諸恐畏？}@
%{天子又问：菩萨如何才能进入大众中勇猛辩论，远离一切恐惧呢？}%

@{答言：選求諸法，不可得貌故。}@
%{回答说：因为选择寻求所有法，却找不到它的相貌。}%

@{天子復問：云何菩薩得所聞義，依而有護？}@
%{天子又问：菩萨如何才能得到所闻的义理，并且依此而守护呢？}%

@{答言：知諸法默然故。}@
%{回答说：因为知道所有法都是寂静无声的。}%

@{天子復問：云何菩薩得依法議？}@
%{天子又问：菩萨如何才能依法而议论呢？}%

@{答言：已知住法界故。}@
%{回答说：因为已经知道住于法界。}%

@{天子復問：云何菩薩而得順教？}@
%{天子又问：菩萨如何才能顺应教法呢？}%

@{答言：不隨諸音故。}@
%{回答说：因为不随从各种声音。}%

@{天子復問：云何菩薩得隨法教？}@
%{天子又问：菩萨如何才能随顺教法呢？}%

@{文殊答言：得諸解脫故。}@
%{文殊回答说：因为获得了各种解脱。}%

@{天子復問：云何菩薩得無過者？}@
%{天子又问：菩萨如何才能没有过失呢？}%

@{答言：於諸法心不動故。}@
%{回答说：因为对于所有法，内心都不动摇。}%

@{天子復問：云何菩薩降伏魔眾？}@
%{天子又问：菩萨如何才能降伏魔众呢？}%

@{答言：以空覺於愛欲而求道故。}@
%{回答说：因为以空性觉悟，在爱欲中寻求道。}%

@{天子復問：云何菩薩智不可逮？}@
%{天子又问：菩萨的智慧如何才能不可企及呢？}%

@{文殊答曰：得諸佛法悉受持故。}@
%{文殊回答说：因为得到了所有佛法，并且全部持守。}%

@{天子復問：云何菩薩不為世法所污？}@
%{天子又问：菩萨如何才能不被世俗法所污染呢？}%

@{答曰：隨世俗行無能污故。}@
%{回答说：因为随顺世俗的行为，没有任何东西能够污染他。}%

@{天子復問：云何菩薩得諸深行？}@
%{天子又问：菩萨如何才能获得各种深奥的行为呢？}%

@{答言：於空法不恐懼故。}@
%{回答说：因为对于空性法不恐惧。}%

@{天子復問：云何菩薩知巧便根？}@
%{天子又问：菩萨如何才能知道巧妙便捷的根源呢？}%

@{答曰：於六情悉見諸情之本故。}@
%{回答说：因为在六情中，全部看到了各种情欲的根源。}%

@{天子復問：云何菩薩得至脫門？}@
%{天子又问：菩萨如何才能得到通往解脱的大门呢？}%

@{答言：於一切門為說脫教故。}@
%{回答说：因为在所有门中，宣说解脱的教法。}%

@{天子復問：云何菩薩得奇特方便？}@
%{天子又问：菩萨如何才能获得奇特方便呢？}%

@{答言：於生死索泥洹求而見之，於泥洹見生死徑故。}@
%{回答说：因为在生死中寻求涅槃而见到它，在涅槃中见到通往生死的道路。}%

@{天子復問：云何菩薩所作得因緣方便？}@
%{天子又问：菩萨所作如何才能获得因缘方便呢？}%

@{答曰：住於無數，悉見一切諸法故。}@
%{回答说：因为住于无数法中，全部看到了所有法。}%

@{天子復問：云何菩薩得律方便？}@
%{天子又问：菩萨如何才能获得律仪方便呢？}%

@{答言：視一切法無所屬故。}@
%{回答说：因为看待所有法，没有任何所属。}%

@{天子復問：云何菩薩得善權方便？}@
%{天子又问：菩萨如何才能获得善巧方便呢？}%

@{答言：隨世所作，不離不著故。}@
%{回答说：因为随顺世俗所作，不离开也不执着。}%

@{天子復問：云何菩薩得吉祥願？}@
%{天子又问：菩萨如何才能获得吉祥的愿望呢？}%

@{答言：已逮道智故。}@
%{回答说：因为已经获得了道智。}%

@{天子復問：云何菩薩得忍辱力？}@
%{天子又问：菩萨如何才能获得忍辱力呢？}%

@{答言：從本已來至於泥洹，悉知諸法故。}@
%{回答说：因为从本来开始，直到涅槃，全部知道了所有法。}%

@{天子復問：云何菩薩得度彼岸？}@
%{天子又问：菩萨如何才能渡过彼岸呢？}%

@{答言：悉知一切異法故。}@
%{回答说：因为全部知道了所有不同的法。}%

@{天子復問：云何菩薩得饒益一切？}@
%{天子又问：菩萨如何才能利益一切众生呢？}%

@{答言：悉知無盡法界故。}@
%{回答说：因为全部知道了无穷无尽的法界。}%

@{天子復問：云何菩薩為眾所愛？}@
%{天子又问：菩萨如何才能被大众所爱呢？}%

@{答言：視諸佛剎無有色故。}@
%{回答说：因为看待所有佛国，没有颜色。}%

@{天子復問：云何菩薩得眾稱譽？}@
%{天子又问：菩萨如何才能得到大众的称赞呢？}%

@{答言：不於諸法言是我所、非我所故。}@
%{回答说：因为对于所有法，不说是我的，也不是我的。}%

@{天子復問：云何菩薩德不可盡？}@
%{天子又问：菩萨的德行如何才能无穷无尽呢？}%

@{答曰：等知諸法如虛空故。}@
%{回答说：因为平等地知道所有法，就像虚空一样。}%

@{天子復問：云何菩薩得堅其本要？}@
%{天子又问：菩萨如何才能坚固其根本要义呢？}%

@{答言：於法界無所破壞故。}@
%{回答说：因为在法界中，没有任何东西能够破坏他。}%

@{天子復問：云何菩薩得為豪尊？}@
%{天子又问：菩萨如何才能成为豪迈尊贵呢？}%

@{答言：無心悉知一切心故。}@
%{回答说：因为无心，却全部知道了所有的心。}%

@{天子復問：云何菩薩而得為師？}@
%{天子又问：菩萨如何才能成为老师呢？}%

@{答言：於無念法無所捨故。}@
%{回答说：因为在无念法中，没有任何舍弃。}%

@{天子復問：云何菩薩得曉知眾事？}@
%{天子又问：菩萨如何才能明了知道所有事情呢？}%

@{答言：知一切法，無依無怙、無來往故。是故，天子！菩薩得曉了眾事。}@
%{回答说：因为知道所有法，没有依靠，没有怙恃，没有来去。所以，天子！菩萨能够明了知道所有事情。}%
@{爾時文殊師利說是事時，八萬菩薩得無所從生法忍。}@
%{当时文殊师利讲完这些事情的时候，八万菩萨都获得了无所得法忍。}%
@{於是世尊讚文殊師利童子言：善哉，善哉！如所解說，分別法義，何快如此！}@
%{于是世尊赞叹文殊师利童子说：很好，很好！你所讲解的，分别法义，真是令人欢喜！}%

@{須真天子經法純淑品第三}@
%{须真天子经法纯淑品第三}%

@{於是須真天子復問文殊師利言：仁者！我所問法為純淑不？}@
%{于是须真天子又问文殊师利说：仁者！我所问的法是纯淑的吗？}%

@{答言：世之所有，欲而無厭，心悉捨離，審於法奧，則為法之純淑。}@
%{文殊师利回答说：世间所有的一切，都想要而不知足，心全部舍弃，深入法奥，这就是法的纯淑。}%
@{如吾所報卿屬所問，豈復純淑法耶？}@
%{我所回答你所问的，难道是纯淑法吗？}%
@{法乎，天子！無純無淑。}@
%{法啊，天子！没有纯也没有淑。}%
@{所以者何？無像貌故。}@
%{为什么呢？因为没有形象。}%
@{法乎，天子！不可得見。}@
%{法啊，天子！不可见。}%
@{所以者何？目之所視不極於微，用有著故。}@
%{为什么呢？因为眼睛所看到的不能穷尽微细，因为有执着。}%
@{法乎，天子！亦無作者。}@
%{法啊，天子！也没有造作者。}%
@{所以者何？用無起故。}@
%{为什么呢？因为没有生起。}%
@{法乎，天子！無有道徑。}@
%{法啊，天子！没有道径。}%
@{所以者何？無吾無我故。}@
%{为什么呢？因为没有我，也没有自我。}%
@{法乎，天子！悉皆平等。}@
%{法啊，天子！都平等。}%
@{所以者何？如虛空故。}@
%{为什么呢？因为像虚空一样。}%
@{法乎，天子！不可得等。}@
%{法啊，天子！不可等。}%
@{所以者何？用無侶故。}@
%{为什么呢？因为没有同伴。}%
@{法乎，天子！常住，無來無去，無語無言，無毀無譽，離於毀譽；}@
%{法啊，天子！常住，没有来也没有去，没有言语，没有毁也没有誉，远离毁誉。}%
@{無綺無飾，無醜無陋故。}@
%{没有绮丽装饰，没有丑陋。}%
@{法乎，天子！無穿無漏、無補無納。}@
%{法啊，天子！没有穿没有漏，没有补没有纳。}%
@{所以者何？過於魔行故。}@
%{为什么呢？因为超越魔行。}%
@{法乎，天子！無長無養。}@
%{法啊，天子！没有增长也没有养育。}%
@{所以者何？離於起滅故。}@
%{为什么呢？因为远离生灭。}%
@{法乎，天子！無處無住。}@
%{法啊，天子！没有处所，没有停留。}%
@{所以者何？樂於法界故。}@
%{为什么呢？因为安乐于法界。}%
@{法乎，天子！無所畏。}@
%{法啊，天子！无所畏惧。}%
@{所以者何？用不惑故。}@
%{为什么呢？因为没有迷惑。}%
@{法乎，天子！無所愛。}@
%{法啊，天子！无所爱。}%
@{所以者何？離於貢高故。}@
%{为什么呢？因为远离贡高。}%
@{法乎，天子！不貢高。}@
%{法啊，天子！不贡高。}%
@{所以者何？習寂然故。}@
%{为什么呢？因为习惯于寂静。}%
@{法乎，天子！習寂然。}@
%{法啊，天子！习惯于寂静。}%
@{所以者何？離諸念故。}@
%{为什么呢？因为远离一切念头。}%
@{法乎，天子！無所念。}@
%{法啊，天子！无所念。}%
@{所以者何？降諸異道故。}@
%{为什么呢？因为降伏一切异道。}%
@{法乎，天子！無有巢窟。}@
%{法啊，天子！没有巢窟。}%
@{所以者何？離婬怒癡故。}@
%{为什么呢？因为远离淫怒痴。}%
@{法乎，天子！空。}@
%{法啊，天子！空。}%
@{所以者何？從本以來淨故。}@
%{为什么呢？因为从本以来就清净。}%
@{法乎，天子！無想。}@
%{法啊，天子！无想。}%
@{所以者何？無聲名故。}@
%{为什么呢？因为没有声名。}%
@{法乎，天子！無願。}@
%{法啊，天子！无愿。}%
@{所以者何？不造立識故。}@
%{为什么呢？因为不造作意识。}%
@{法乎，天子！無造立。}@
%{法啊，天子！没有造作。}%
@{所以者何？無疆界故。}@
%{为什么呢？因为没有疆界。}%
@{法乎，天子！無所猗。}@
%{法啊，天子！无所依。}%
@{所以者何？用無雙故。}@
%{为什么呢？因为没有对。}%
@{法乎，天子！無動搖。}@
%{法啊，天子！没有动摇。}%
@{所以者何？用堅住故。}@
%{为什么呢？因为坚住。}%
@{法乎，天子！無我。}@
%{法啊，天子！没有我。}%
@{所以者何？用不滅故。}@
%{为什么呢？因为不灭。}%
@{法乎，天子！無人。}@
%{法啊，天子！没有人。}%
@{所以者何？從本已來無所生故。}@
%{为什么呢？因为从本以来就无所生。}%
@{法乎，天子！無常。}@
%{法啊，天子！无常。}%
@{所以者何？無所起故。}@
%{为什么呢？因为没有生起。}%
@{法乎，天子！無所起。}@
%{法啊，天子！没有生起。}%
@{所以者何？習無所生故。}@
%{为什么呢？因为习惯于无所生。}%
@{法乎，天子！無所生。}@
%{法啊，天子！无所生。}%
@{所以者何？無苦器想故。}@
%{为什么呢？因为没有苦器想。}%
@{法乎，天子！無有想。}@
%{法啊，天子！没有想。}%
@{所以者何？離諸相故。}@
%{为什么呢？因为远离一切相。}%
@{如是，天子！此則為法之純淑義也。}@
%{这样，天子！这就是法的纯淑义。}%

@{天子復問：文殊師利！法無所有亦無所要，云何仁者說純淑法議乎？}@
%{天子又问：文殊师利！法无所有，也没有所要，怎么仁者还会说纯淑法义呢？}%

@{答言：善哉，善哉！如卿所語，誠無有異。}@
%{文殊师利回答说：很好，很好！你所说的话，确实没有错。}%
@{無所有者，此則純淑法議。}@
%{无所有，这就是纯淑法义。}%
@{所以者何？無身口意所作，是則法之純淑也。}@
%{为什么呢？没有身口意所作，这就是法的纯淑。}%
@{所以然者？天子！法無巢窟故。}@
%{为什么呢？天子！法没有巢窟。}%
@{有巢窟者，身與意而異，則為非時之心施。}@
%{有巢窟，身和意就不同，就属于非时之心所作。}%

@{天子復問：云何得知非時之心？}@
%{天子又问：怎么知道非时之心呢？}%

@{答言：天子！有身為六衰相所繫而計有常，則知非時之心。}@
%{文殊师利回答说：天子！有身被六衰相所系，而认为有常，就知道是非时之心。}%
@{知法求名著音聲響而隨邪徑，則知非時之心。}@
%{知道法，求名声，著相，随邪径，就知道是非时之心。}%
@{知法及僧受道果證，則知非時之心。}@
%{知道法，以及僧人受道果证，就知道是非时之心。}%
@{知愛欲本邪相施與，則知非時之心。}@
%{知道爱欲本来是邪相施与，就知道是非时之心。}%
@{知戒而離寂靜，則知非時之心。}@
%{知道戒律，却远离寂静，就知道是非时之心。}%
@{畢三惡道得出為人，志在天福，則知非時之心。}@
%{从三恶道出来做人，志在天福，就知道是非时之心。}%
@{其意不調而欲布施，則知非時之心。}@
%{其心不调，却想布施，就知道是非时之心。}%
@{意無寂滅，則知非時之心。}@
%{心没有寂灭，就知道是非时之心。}%
@{意有猗怙而欲忍辱，則知非時之心。}@
%{心有依怙，却想忍辱，就知道是非时之心。}%
@{不淨其意而欲精進，則知非時之心。}@
%{心不净，却想精进，就知道是非时之心。}%
@{多念喜忘禪思不定，則知非時之心。}@
%{多念，欢喜忘掉禅思，心不定，就知道是非时之心。}%
@{自大貢高忽於智慧，則知非時之心。}@
%{自大贡高，忽视智慧，就知道是非时之心。}%
@{住於我所而欲行慈，則知非時之心。}@
%{执著于自我，却想行慈，就知道是非时之心。}%
@{志於猶豫而欲行哀，則知非時之心。}@
%{志于犹豫，却想行哀，就知道是非时之心。}%
@{行墮四證而欲行喜，則知非時之心。}@
%{行堕四证，却想行喜，就知道是非时之心。}%
@{住於有身而欲行護，則知非時之心。}@
%{执著于有身，却想行护，就知道是非时之心。}%
@{無身痛痒意而欲念法不應止，則知非時之心。}@
%{没有身痛痒的念头，却想念法不应止，就知道是非时之心。}%
@{知諸起滅不應斷，則知非時之心。}@
%{知道一切生灭，不应断，就知道是非时之心。}%
@{身意相猗不應神足，則知非時之心。}@
%{身意相依，不应神足，就知道是非时之心。}%
@{以六情猗於五根，則知非時之心。}@
%{以六情依于五根，就知道是非时之心。}%
@{以所見力依於五力，則知非時之心。}@
%{以所见力依于五力，就知道是非时之心。}%
@{知七法無覺意，則知非時之心。}@
%{知道七法没有觉意，就知道是非时之心。}%
@{念愛欲貪不應八直，則知非時之心。}@
%{念爱欲贪，不应八正，就知道是非时之心。}%
@{於苦智而有疑，則知非時之心。}@
%{对于苦智有疑，就知道是非时之心。}%
@{於習而有疑，則知非時之心。}@
%{对于习有疑，就知道是非时之心。}%
@{於滅盡而有疑，則知非時之心。}@
%{对于灭尽有疑，就知道是非时之心。}%
@{齎俗所有欲入於道，則知非時之心。}@
%{带着世俗所有，想要进入道，就知道是非时之心。}%
@{如是，天子！受持淨心用專著故，則知非時之心。}@
%{这样，天子！受持清净心，专著，就知道是非时之心。}%
@{天子復問：云何得知是時之心？}@
%{天子又问：怎么才能知道此时此刻的心呢？}%

@{答言：天子！心等如虛空，則知是時之心。}@
%{回答说：天子！心就像虚空一样，这样就能知道此时此刻的心了。}%

@{復問：云何心如虛空？}@
%{又问：怎么才能说心就像虚空呢？}%

@{答言：如虛空無心，心亦如是。如是心，心等如虛空。}@
%{回答说：就像虚空没有心一样，心也是如此。这样的心，心就如同虚空。}%

@{復問：誰當信心等如虛空？}@
%{又问：谁应该相信心就像虚空呢？}%

@{答言：天子！計有吾我人者則不信。}@
%{回答说：天子！认为有“我”的人就不会相信。}%

@{天子復問：計吾我人者何所是？}@
%{天子又问：认为有“我”的人指的是什么呢？}%

@{答言：心等如虛空，而欲增益過出其上。}@
%{回答说：心就像虚空一样，却想要超越它，在它之上再增加点什么。}%

@{復問：云何增益過出其上？}@
%{又问：怎么才能超越它，在它之上再增加点什么？}%

@{答言：計有常者便欲出其上，計無常者亦欲出其上；苦者亦欲出其上，樂者亦欲出其上；憂者亦欲出其上，無憂者亦欲出其上；計有身者亦欲出其上，計無身者亦欲出其上；空、無想、無願亦欲出其上；幻、夢、水中月、影、響一切諸法，其譬如是。如是等而復欲出其上。如法有所處便可增益，婬怒癡更相猗，復欲出其上。形此生死不可讚歎泥洹之事，復欲出其上。如是，天子！是為計吾我人者。}@
%{回答说：认为有常住不变的人就想要超越它，认为没有常住不变的人也想要超越它；认为有苦的人就想要超越它，认为有乐的人也想要超越它；认为有忧愁的人就想要超越它，认为没有忧愁的人也想要超越它；认为有身体的人就想要超越它，认为没有身体的人也想要超越它；认为有空、无想、无愿的人也想要超越它；幻觉、梦境、水中月、影子、回声，一切法都是如此。如此等等，还想要超越它。如果法有所依，就可以增加，淫欲、愤怒、愚痴互相依附，又想要超越它。形于此生死，不可赞叹涅槃之事，又想要超越它。如此，天子！这就是认为有“我”的人。}%

@{復問：云何得無瞋恚而不佷？}@
%{又问：怎么才能做到没有瞋恚，没有怨恨呢？}%

@{答言：從生死出，住於泥洹。還世間，滅諸愛欲而淨行。於滅不永滅，於起無所起，諸形音聲不以畏懼。如是滅為習者有所脫，脫習者為已度如是事。一切法不能舉其功、不能勝其德，是為無恚而不恨。}@
%{回答说：从生死中解脱出来，住于涅槃。回到世间，灭除一切爱欲，而以清净的行为。在灭除的时候，不永远灭除，在生起的时候，没有生起的地方，对一切形体、声音都不以恐惧。如此灭除，对于习气来说有所解脱，解脱习气，就等于已经度过了这些事。一切法都不能举出它的功德，不能胜过它的美德，这就是没有瞋恚，没有怨恨。}%

@{說是法時。三萬二千天子得法眼淨，五千比丘心得解脫，萬二千菩薩得忍辱力。}@
%{说这些法的时候，三万二千天子获得了法眼清净，五千比丘获得了解脱，一万二千菩萨获得了忍辱力。}%

@{須真天子經聲聞品第四}@
%{须真天子经 声闻品第四}%

@{於是須真天子謂諸大弟子：仁者所狐疑，可問文殊師利。}@
%{于是须真天子对诸大弟子说：仁者所疑惑的，可以去问文殊师利。}%

@{長老摩訶迦葉前問文殊師利言：菩薩云何行八惟務禪？}@
%{长老摩诃迦叶上前问文殊师利说：菩萨如何修行八唯务禅？}%

@{答言：菩薩於八惟務禪，本無、無所造立禪，無恚禪等禪，是菩薩禪。}@
%{回答说：菩萨在八唯务禅中，本无、无所造立禅，无恚禅等禅，这就是菩萨禅。}%

@{復問：云何作是說？}@
%{又问：怎么这么说呢？}%

@{答言：惟迦葉！身本無，無造立於三界者，便起愛欲。已了離欲，是故知身本無，於三界無所造，於欲無所想，已知空而立禪。如是，迦葉！而作此說：八惟務禪，本無、無造立禪，無恚禪等禪，則菩薩禪。於是迦葉默而無言。}@
%{回答说：唯迦叶！身本无，没有在三界中造作，就会生起爱欲。已经了知离欲，所以知道身本无，在三界中无所造作，对欲无所想，已经知道空而立禅。如此，迦叶！才说这些话：八唯务禅，本无、无所造立禅，无恚禅等禅，这就是菩萨禅。于是迦叶默然无语。}%

@{賢者舍利弗復問：云何菩薩得無礙慧？}@
%{贤者舍利弗又问：菩萨如何获得无碍慧呢？}%

@{答言：菩薩於諸礙而無恚恨，於諸罣礙而無制著，一切愛欲而皆見知而不捨離。所以者何？養護一切故，是故菩薩得無礙慧。}@
%{回答说：菩萨对于一切障碍都没有瞋恨，对于一切挂碍都没有执着，对于一切爱欲都了知，却不舍弃。为什么呢？因为要护持一切，所以菩萨获得无碍慧。}%

@{賢者摩訶目犍連復問：云何菩薩而得神足？}@
%{贤者摩诃目犍连又问：菩萨如何获得神足呢？}%

@{答言：惟目犍連！菩薩於無為而無所受，度脫一切而降盡之，不於有為有所受。所以者何？將護一切故，是故菩薩得大神足。}@
%{回答说：唯目犍连！菩萨在无为中无所受，度脱一切，降伏一切，不在有为中有所受。为什么呢？因为要护持一切，所以菩萨获得大神足。}%

@{長老須菩提復問：云何菩薩得知他法行？}@
%{长老须菩提又问：菩萨如何知道他法的修行呢？}%

@{答言：惟須菩提！菩薩於一切他異法悉了知之，心於道事而不厭，常樂三昧而無足，諸所作為而示現，是故菩薩得知他法行。}@
%{回答说：唯须菩提！菩萨对于一切不同的法都了知，心于道事而不厌倦，常乐三昧而没有止息，一切所作都示现出来，所以菩萨知道他法的修行。}%

@{賢者分耨文陀尼子復問：云何菩薩博採眾義說明慧法？}@
%{贤者分耨文陀尼子又问：菩萨如何广采众义，说明慧法呢？}%

@{答言：惟分耨！菩薩悉示一切諸根，隨所樂喜而說其德，無常、苦、空、非身之義，各令得其所。無數生死百千劫，持是法義而遍教授，無有滅盡。其智如是，是故菩薩博採眾義說明慧法。}@
%{回答说：唯分耨！菩萨对一切众生都示现，根据他们各自的喜好而宣说其法，无常、苦、空、非身之义，都令他们得到各自的利益。无数生死，百千万劫，持着这些法义而遍教授，没有灭尽。智慧如此，所以菩萨广采众义，说明慧法。}%

@{賢者離越復問：云何菩薩如常樂禪。}@
%{贤者离越又问：菩萨如何如常乐禅呢？}%

@{答言：惟離越！菩薩習三摩越悉知諸法，於諸亂意者而起大哀，令發無央數行，不禪無所樂，是故菩薩得禪。}@
%{回答说：唯离越！菩萨修习三摩地，悉知一切法，对于一切乱意的人，就生起大悲心，令他们发起无量行，不修禅就无所乐，所以菩萨得禅。}%

@{賢者憂波離復問：云何菩薩得持法藏？}@
%{贤者忧波离又问：菩萨如何获得持法藏呢？}%

@{答言：唯憂波離！菩薩悉知諸法奧藏，從本已來泥曰，離愛欲者，已應法藏教授一切，為示愛欲令覺知之。於愛欲中令起道意，是故菩薩得法隩藏。}@
%{回答说：唯忧波离！菩萨悉知一切法的奥藏，从本以来，泥洹，离爱欲的人，已经应法藏教授一切，为了示现爱欲，令他们觉知。在爱欲中令他们生起道心，所以菩萨获得法藏。}%
@{賢者阿那律復問：云何菩薩得天眼徹視？}@
%{贤者阿那律又问：菩萨怎么才能获得天眼彻视呢？}%

@{答言：唯阿那律！菩薩於十方諸色悉照見已，有色習者而為示現於一切法，悉示現無所著，令尋跡而得出，是為菩薩天眼徹視。}@
%{答曰：阿那律啊！菩萨已经能遍观十方世界的一切颜色，他以有色的修行方式示现于一切法，处处示现无执着，让众生循着他的踪迹而悟道，这就是菩萨的天眼彻视。}%

@{賢者薄鳩盧復問：云何菩薩得諸根寂定？}@
%{贤者薄鸠卢又问：菩萨怎么才能获得诸根寂定呢？}%

@{答言：菩薩於一切界視如佛界，於佛界視諸界無所有，是故菩薩得諸根寂定。}@
%{答曰：菩萨看待一切界就像看待佛界一样，看待佛界就像看待一切界不存在一样，所以菩萨才能获得诸根寂定。}%

@{賢者鴦掘魔復問：云何菩薩得利諸根？}@
%{贤者鸯掘魔又问：菩萨怎么才能获得利诸根呢？}%

@{答言：菩薩視諸逆惡等之如道，是故菩薩得利諸根。}@
%{答曰：菩萨看待一切逆境和恶事就像看待大道一样，所以菩萨才能获得利诸根。}%

@{賢者摩訶迦旃延復問：云何菩薩得分別知眾經方便？}@
%{贤者摩诃迦旃延又问：菩萨怎么才能获得分别知众经的方便呢？}%

@{答言：菩薩得四等無盡。何等為四？一者、義，二者、法，三者、次第，四者、報答。是為四。以一絕句，於百千劫廣為一切分別演教，而是教不近有為、不有所染，已淨無所却。而是教於諸法界不動轉，於一切受而為作受。是故菩薩得分別知眾經方便。}@
%{答曰：菩萨获得四等无尽，这四等无尽是什么呢？第一是义，第二是法，第三是次第，第四是报答，这四种就是四等无尽。菩萨用一句偈语，在百千万劫中广泛地为一切众生分别演说教法，这教法不近有为，不染着任何东西，已经清净，没有退却。这教法在诸法界中不动不转，在一切受中为一切众生作受。所以菩萨才能获得分别知众经的方便。}%

@{賢者摩訶拘絺復問：云何菩薩得義、法、次第、報答四事？}@
%{贤者摩诃拘絺又问：菩萨怎么才能获得义、法、次第、报答四事呢？}%

@{答言：菩薩於寂然法得此，已義等教授；以住於法，以法等教授；於所為常歡喜而無恨，以等次第教授；而響不可護持，以報答教授。是故菩薩得義、法、次第、報答。}@
%{答曰：菩萨在寂静的法中获得这种义等教授，以住于法，以法等教授，对所作所为常欢喜而无恨，以等次第教授，而响不可护持，以报答教授。所以菩萨才能获得义、法、次第、报答。}%

@{賢者羅雲復問：云何菩薩得淨其戒？}@
%{贤者罗云又问：菩萨怎么才能获得清净戒律呢？}%

@{答言：惟羅雲！菩薩以淨戒三昧，捨戒犯戒將養一切，是故菩薩而得淨戒。}@
%{答曰：罗云啊！菩萨以清净戒三昧，舍弃戒律和犯戒，将养一切众生，所以菩萨才能获得清净戒律。}%

@{賢者阿難復問：云何菩薩而得博聞？}@
%{贤者阿难又问：菩萨怎么才能获得博闻呢？}%

@{答言：菩薩一切諸佛所說樂欲聽聞，已聞則受其義，聞已皆持，所聞便以教授，是為菩薩而得博聞。}@
%{答曰：菩萨乐于听闻一切佛陀所说，闻之即领会其义，闻之即持之，闻之即以之教授他人，这就是菩萨获得博闻的原因。}%

@{於是諸大弟子歡喜默然。}@
%{于是诸大弟子欢喜地默然不语。}%

@{爾時須真天子謂諸大弟子言：屬文殊師利所說法，仁寧有是事乎？}@
%{这时须真天子对诸大弟子说：你们觉得文殊师利所说的这些法，是真的吗？}%

@{諸大弟子言：吾等尚不能了知一法，何況爾所法耶？}@
%{诸大弟子说：我们尚且不能了解一种法，更何况你所说的这些法呢？}%

@{天子言：仁者！若干種身，各各異類，其道義一也。}@
%{天子说：仁者！虽然有许多种身，各自不同，但他们的道义是一样的。}%

@{大弟子答言：譬如牛跡中水，諸餘弟子所知如是。若車轂隱地其處受水，吾等之類其譬如是。譬如大海其水廣長，無有邊幅深難得底，於聲聞、辟支佛中，菩薩為尊。}@
%{大弟子回答说：就像牛蹄印里的水一样，其他弟子的智慧就像这样。如果车轮毂藏在地里，那里会积水，我们这些弟子就像这样。就像大海一样，海水广阔无边，深不可测，在声闻、辟支佛中，菩萨是最尊贵的。}%

@{天子讚言：善哉，善哉！所說至誠而不貢高。}@
%{天子赞叹说：善哉，善哉！你所说的话真诚而不自夸。}%

@{文殊師利言：如是，天子！弟子所言而不貢高，稱譽菩薩，審諦實爾。}@
%{文殊师利说：是的，天子！弟子所说的话并不自夸，称赞菩萨，这是真实不虚的。}%

@{天子問文殊師利：仁者！云何作是說？}@
%{天子问文殊师利：仁者！你为什么这样说呢？}%

@{答言：如是，天子！聲聞、辟支佛為猗貢高？為離貢高？菩薩貢高，出彼輩上，合聚佛法，則是菩薩為行勇悍。}@
%{答曰：是的，天子！声闻、辟支佛是自夸呢？还是不夸耀呢？菩萨是夸耀，超越他们之上，汇聚佛法，这就是菩萨的勇悍之处。}%

@{天子復問：菩薩貢高，欲令他人稱譽耶？}@
%{天子又问：菩萨夸耀，是想让别人称赞他吗？}%

@{答言：欲將導一切故。如是，天子！}@
%{答曰：是为了引导一切众生。是的，天子！}%

@{天子復問：云何如是？}@
%{天子又问：怎么才能做到这样呢？}%

@{答言：天子！是故菩薩方便稱譽佛乘、毀弟子乘。於大眾中，自現身所行及法事。所以者何？欲令一切皆發道意，不欲使人起小道意。所以者何？焦燒佛種故。教一切人皆令遠離。所以者何？不欲令人貪樂故也。如是，天子！欲令菩薩發大乘、滅弟子乘故。}@
%{答曰：天子！所以菩萨方便地称赞佛乘，贬低弟子乘，在大众中，展现自己所行和所作的法事。为什么呢？是为了让一切众生都生起道心，不希望让人产生小道心。为什么呢？因为要烧毁佛种。教导一切人远离它。为什么呢？因为不希望让人贪图快乐。是的，天子！是为了让菩萨发大乘，灭除弟子乘。}%

@{天子復問：得無過耶？}@
%{天子又问：这样做有没有过错呢？}%

@{答言：天子！稱譽摩尼、琉璃、水精甚淨，無所沾污。寧復過乎？}@
%{答曰：天子！称赞摩尼、琉璃、水精非常清净，没有染污，难道有错吗？}%

@{天子報言：所說無過。}@
%{天子回答说：你所说没有过错。}%

@{答言：如是，天子！菩薩稱譽大乘！毀弟子乘，不增不減也。天子！譬如長者子稱譽轉輪聖王功德，毀呰國中諸貧乞者，豈有不可？}@
%{答曰：是的，天子！菩萨称赞大乘，贬低弟子乘，既不增加也不减少。天子！就像富贵人家的子弟称赞转轮圣王的功德，贬低贫民窟里的乞丐，难道有什么不可吗？}%

@{天子言：無不可也。文殊師利！自如所說耶？}@
%{天子说：没有不可。文殊师利！你真是这样说的吗？}%

@{答言：如是，天子！菩薩稱譽大乘而毀呰弟子乘者，而無所損。}@
%{答曰：是的，天子！菩萨称赞大乘，贬低弟子乘，并没有什么损失。}%

@{佛爾時讚歎文殊師利言：善哉，善哉！如是所說為甚快也。何以故？文殊師利！稱譽大乘、毀弟子乘，弟子則毀一切乘矣。所以然者？其大乘者，皆生一切乘故。}@
%{佛陀当时赞叹文殊师利说：善哉，善哉！你这样说真是太好了。为什么呢？文殊师利！称赞大乘，贬低弟子乘，弟子乘就会毁掉一切乘。为什么呢？因为大乘包含了一切乘。}%
