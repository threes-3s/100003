@{大寶積經卷第一百一十九}@
%{大宝积经卷第一百一十九}%

@{大唐三藏菩提流志奉詔譯}@
%{大唐三藏菩提流志奉诏译}%

@{勝鬘夫人會第四十八}@
%{胜鬘夫人会第四十八}%

@{如是我聞：}@
%{如是我闻：}%

@{一時佛在舍衛國祇樹給孤獨園。}@
%{一时佛在舍卫国祇树给孤独园。}%

@{時憍薩羅波斯匿王及末利夫人初證法已，共相謂言：我女勝鬘！慈晤聰愍多聞智慧，若見如來，於甚深法速能解了無諸疑惑。}@
%{时憍萨罗波斯匿王及末利夫人初证法已，共相谓言：我女胜鬘！慈晤聪愍多闻智慧，若见如来，于甚深法速能解了无诸疑惑。}%

@{我今應當令善諭者發其誠信。}@
%{我今应当令善喻者发其诚信。}%

@{作是議已，王及夫人即便作書，稱揚如來真實功德。}@
%{作是议已，王及夫人即便作书，称扬如来真实功德。}%

@{時遣一使名真提羅，奉持王書詣無鬪城授勝鬘夫人。}@
%{时遣一使名真提罗，奉持王书诣无鬪城授胜鬘夫人。}%

@{時勝鬘夫人發書尋繹，頂受忻慶生希有心，向真提羅而說偈言：}@
%{时胜鬘夫人发书寻绎，顶受忻庆生希有心，向真提罗而说偈言：}%

@{我聞如來聲，世間頗難遇，}@
%{我闻如来声，世间颇难遇，}%

@{斯言若真實，當賜汝衣服。}@
%{斯言若真实，当赐汝衣服。}%

@{若彼佛世尊，為利世間現，}@
%{若彼佛世尊，为利世间现，}%

@{必應見哀愍，令我覩真相。}@
%{必应见哀愍，令我睹真相。}%

@{言念須臾頃，佛於虛空中，}@
%{言念须臾顷，佛于虚空中，}%

@{現不思議身，普放大光明。}@
%{现不可思议身，普放大光明。}%

@{勝鬘及眷屬，皆悉來集會，}@
%{胜鬘及眷属，皆悉来集会，}%

@{合掌瞻仰禮，稱讚大導師：}@
%{合掌瞻仰礼，称赞大导师：}%

@{如來妙色身，世間無與等，}@
%{如来妙色身，世间无与等，}%

@{無比不思議，是故今敬禮。}@
%{无比不可思议，是故今敬礼。}%

@{如來色無盡，智慧亦復然，}@
%{如来色无尽，智慧亦复然，}%

@{一切法常住，是故我歸依。}@
%{一切法常住，是故我归依。}%

@{善調心過惡，及與身四種，}@
%{善调心过恶，及与身四种，}%

@{到不思議地，故我今敬禮。}@
%{到不可思议地，故我今敬礼。}%

@{知諸爾炎法，智身無罣礙，}@
%{知诸尔炎法，智身无罣碍，}%

@{於法無忘失，故我今敬禮。}@
%{于法无忘失，故我今敬礼。}%

@{稽首過稱量，稽首無倫等，}@
%{稽首过称量，稽首无伦等，}%

@{稽首法自在，稽首超思惟。}@
%{稽首法自在，稽首超思惟。}%

@{哀愍覆護我，令法種增長，}@
%{哀愍覆护我，令法种增长，}%

@{逮及最後身，常在如來前。}@
%{逮及最后身，常在如来前。}%

@{我所修福業，此世及餘生，}@
%{我所修福业，此世及余生，}%

@{由斯善根力，願佛恒攝受。}@
%{由斯善根力，愿佛恒摄受。}%

@{時勝鬘夫人說此偈已，及諸眷屬一切大眾，頂禮佛足。}@
%{时胜鬘夫人说此偈已，及诸眷属一切大众，顶礼佛足。}%

@{爾時世尊即為勝鬘而說偈言：}@
%{尔时世尊即为胜鬘而说偈言：}%

@{我昔為菩提，曾已開示汝，}@
%{我昔为菩提，曾已开示汝，}%

@{今復值遇我，及來世亦然。}@
%{今复值遇我，及来世亦然。}%

@{說此偈已，即於會中授勝鬘夫人阿耨多羅三藐三菩提記：汝今稱歎如來殊勝功德，以此善根，當於無量阿僧祇劫天人之中為自在王，諸所受用皆悉具足，所生之處常得遇我，現前稱歎如今無異。}@
%{说此偈已，即于会中授胜鬘夫人阿耨多罗三藐三菩提记：汝今称叹如来殊胜功德，以此善根，当于无量阿僧祇劫天人之中为自在王，诸所受用皆悉具足，所生之处常得遇我，现前称叹如今无异。}%

@{復當供養無量無數諸佛世尊，過二萬阿僧祇劫當得作佛，號曰普光如來、應、正等覺。}@
%{复当供养无量无数诸佛世尊，过二万阿僧祇劫当得作佛，号曰普光如来、应、正等觉。}%

@{彼佛國土無諸惡趣衰老病苦，亦無不善惡業道名。}@
%{彼佛国土无诸恶趣衰老病苦，亦无不善恶业道名。}%

@{其中眾生形色端嚴，具五妙境純受快樂，蔽於他化自在諸天。}@
%{其中众生形色端严，具五妙境纯受快乐，蔽于他化自在诸天。}%

@{彼諸眾生皆趣大乘，諸有如是學大乘者悉來生彼。}@
%{彼诸众生皆趣大乘，诸有如是学大乘者悉来生彼。}%

@{時勝鬘夫人得授記已，無量天人心懷踊躍，咸願往生彼佛世界。}@
%{时胜鬘夫人得授记已，无量天人心怀踊跃，咸愿往生彼佛世界。}%

@{是時世尊皆與授記當生彼國。}@
%{是时世尊皆与授记当生彼国。}%

@{時勝鬘夫人聞佛記已，於如來前合掌而立，發十弘誓，作如是言：世尊！我從今日乃至菩提，於諸受戒不起犯心。}@
%{时胜鬘夫人闻佛记已，于如来前合掌而立，发十弘誓，作如是言：世尊！我从今日乃至菩提，于诸受戒不起犯心。}%

@{世尊！我從今日乃至菩提，於諸師長不起慢心。}@
%{世尊！我从今日乃至菩提，于诸师长不起慢心。}%

@{世尊！我從今日乃至菩提，於諸眾生不起恚心。}@
%{世尊！我从今日乃至菩提，于诸众生不起恚心。}%

@{世尊！我從今日乃至菩提，於諸勝已及諸勝事不起妬心。}@
%{世尊！我从今日乃至菩提，于诸胜已及诸胜事不起妬心。}%

@{世尊！我從今日乃至菩提，雖有少食不起慳心。}@
%{世尊！我从今日乃至菩提，虽有少食不起慳心。}%

@{世尊！我從今日乃至菩提，不自為己受畜財物，凡有所受為濟貧苦有情之類。}@
%{世尊！我从今日乃至菩提，不自为己受畜财物，凡有所受为济贫苦有情之类。}%

@{世尊！我從今日乃至菩提，不求恩報行四攝事，無貪利心、無厭足心、無限礙心，攝受眾生。}@
%{世尊！我从今日乃至菩提，不求恩报行四摄事，无贪利心、无厌足心、无限碍心，摄受众生。}%

@{世尊！我從今日乃至菩提，見諸眾生無有依怙，幽繫疾惱種種危厄，終不捨離必願安隱，以善饒益令免眾苦。}@
%{世尊！我从今日乃至菩提，见诸众生无有依怙，幽繫疾恼种种危厄，终不捨离必愿安隐，以善饶益令免众苦。}%

@{世尊！我從今日乃至菩提，若見一切諸惡律儀，毀犯如來清淨禁戒，凡我所攝城邑聚落，應調伏者而調伏之，應攝受者而攝受之。}@
%{世尊！我从今日乃至菩提，若见一切诸恶律仪，毁犯如来清净禁戒，凡我所摄城邑聚落，应调伏者而调伏之，应摄受者而摄受之。}%

@{何以故？以調伏攝受故則正法久住，正法久住故天人充滿惡道減少，能令如來法輪常轉。}@
%{何以故？以调伏摄受故则正法久住，正法久住故天人充满恶道减少，能令如来法轮常转。}%

@{世尊！我從今日乃至菩提，攝受正法終不忘失。}@
%{世尊！我从今日乃至菩提，摄受正法终不忘失。}%

@{何以故？忘失正法則忘大乘，忘大乘者則忘波羅蜜，忘波羅蜜者則捨大乘。}@
%{何以故？忘失正法则忘大乘，忘大乘者则忘波罗蜜，忘波罗蜜者则舍大乘。}%

@{若諸菩薩有於大乘不決定者，攝受正法則不堅固，便不堪任超凡夫境，則為大失。}@
%{若诸菩萨有于大乘不决定者，摄受正法则不坚固，便不堪任超凡夫境，则为大失。}%

@{世尊！現在未來攝受正法諸菩薩等，具足無邊廣大利益。}@
%{世尊！现在未来摄受正法诸菩萨等，具足无边广大利益。}%

@{發斯弘誓，聖主世尊雖復證知，而諸有情善根微薄或起疑網，以十弘誓難成就故。}@
%{发斯弘誓，圣主世尊虽复证知，而诸有情善根微薄或起疑网，以十弘誓难成就故。}%

@{彼或長夜習不善法受諸苦惱，為欲利益如斯眾生，今於佛前發誠實誓。}@
%{彼或长夜习不善法受诸苦恼，为欲利益如斯众生，今于佛前发诚实誓。}%

@{世尊！我今發此十弘誓願，若實不虛，於大眾上當雨天花出天妙音。}@
%{世尊！我今发此十弘誓愿，若实不虚，于大众上当雨天花出天妙音。}%

@{勝鬘夫人於如來前作斯言已，時虛空中即雨天花出天妙音，歎言：善哉勝鬘夫人！如汝所說真實無異。}@
%{胜鬘夫人于如来前作斯言已，时虚空中即雨天花出天妙音，叹言：善哉胜鬘夫人！如汝所说真实无异。}%

@{爾時眾會既覩斯瑞，無諸疑惑生大歡喜，同聲唱言：願與勝鬘夫人所生之處同其願行。}@
%{尔时众会既睹斯瑞，无诸疑惑生大欢喜，同声唱言：愿与胜鬘夫人所生之处同其愿行。}%

@{時佛世尊悉記大眾如其所願。}@
%{时佛世尊悉记大众如其所愿。}%
@{爾時勝鬘夫人復於佛前發三弘願：以茲願力利益無邊諸有情類。}@
%{当时胜鬘夫人又在佛陀面前发下了三个弘愿，希望以此愿力利益无量无边的众生。}%
@{第一願者，以我善根，於一切生得正法智。}@
%{第一个愿是，愿我以自己的善根，在每一世都能获得正法智慧。}%
@{第二願者，若我所生得正智已，為諸眾生演說無倦。}@
%{第二个愿是，愿我一旦获得正法智慧，就为所有众生宣说佛法，永不疲倦。}%
@{第三願者，我為攝受護持正法，於所生身不惜軀命。}@
%{第三个愿是，愿我为了护持正法，在每一世都不惜生命。}%
@{爾時世尊聞斯願已，告勝鬘言：如一切色悉入空界，如是菩薩恒沙諸願悉入茲願，此三願者真實廣大。}@
%{佛陀听到胜鬘夫人的三个弘愿后，对她说：就像所有颜色都融入空界一样，菩萨无数的愿望也全部融入这三个弘愿之中，这三个弘愿真实而广大。}%

@{爾時勝鬘夫人復白佛言：世尊！今當承佛威神辯才之力，欲說大願，幸垂聽許。}@
%{当时胜鬘夫人又对佛陀说：世尊，我现在要依靠佛陀的威神辩才之力，想要宣说我的大愿，希望您能听我讲。}%

@{佛言：勝鬘！恣汝所說。}@
%{佛陀说：胜鬘，你尽管说吧。}%

@{勝鬘夫人言：菩薩所有恒沙諸願，一切皆入一大願中。}@
%{胜鬘夫人说：菩萨所有无数的愿望，都归于一个大愿之中。}%
@{一大願者，所謂攝受如來正法。}@
%{这个大愿就是，所谓摄受如来正法。}%
@{如是攝受正法，真實廣大。}@
%{这样摄受正法，真实而广大。}%

@{佛言：善哉勝鬘！汝久修習，智慧方便甚深微妙。}@
%{佛陀说：善哉善哉，胜鬘！你已经修行很久了，智慧和方便非常深奥微妙。}%
@{有能解了汝所說義，彼於長夜植諸善本。}@
%{如果有人能够理解你所说的话，那么他在漫长的岁月里就种下了许多善根。}%
@{如汝所說攝受正法，皆是過去未來現在諸佛已說今說當說。}@
%{你所说的摄受正法，都是过去、现在、未来诸佛已经说过、正在说、将要说的。}%
@{我得無上正等菩提，亦復常以種種相說攝受正法，如是稱揚攝受正法，所有功德無有邊際，如來智慧亦無邊際。}@
%{我获得了无上正等菩提，也常常以各种方式宣说摄受正法，这样赞扬摄受正法，其功德无有边际，如来的智慧也无有边际。}%
@{何以故？是攝受正法有大功德、有大利益。}@
%{为什么呢？因为摄受正法有很大的功德，有很大的利益。}%

@{時勝鬘夫人復白佛言：世尊！我當承佛威神之力，更復演說攝受正法廣大之義。}@
%{这时胜鬘夫人又对佛陀说：世尊，我要依靠佛陀的威神之力，进一步宣说摄受正法的广大意义。}%

@{佛言：聽汝所說。}@
%{佛陀说：听你说吧。}%

@{勝鬘夫人言：攝受正法廣大義者，為得無量一切佛法，乃至能攝八萬行蘊。}@
%{胜鬘夫人说：摄受正法的广大意义，是为了获得无量无边的佛法，乃至能摄受八万行蕴。}%
@{譬如劫初興諸色雲雨眾寶雨，如是攝受正法善根之雲，能雨無量福報之雨。}@
%{就像劫初升起各种颜色的云雨和众宝雨一样，这样摄受正法的善根之云，能降下无量福报之雨。}%
@{世尊！又如劫初大水之中，能生三千大千界藏及四百億種種類洲。}@
%{世尊，又就像劫初的大水中，能生出三千大千界藏和四百亿种类的洲一样。}%
@{如是攝受正法，出生大乘無量界藏，并諸菩薩神通之力、種種法門，一切世間及出世間安樂具足，一切天人所未曾有。}@
%{这样摄受正法，就能生出大乘无量界藏，以及诸菩萨的神通力量、各种法门，一切世间和出世间安乐具足，一切天人都从未有过。}%
@{又如大地荷四重擔。}@
%{又像大地承受着四种重担。}%
@{何等為四？一者大海，二者諸山，三者草木，四者眾生。}@
%{这四种重担是什么呢？一是海洋，二是高山，三是草木，四是众生。}%
@{如是攝受正法諸善男子及善女人，堪能荷負四種重任逾彼大地。}@
%{这样摄受正法的善男子善女人，能够承担这四种重任，超过大地。}%
@{何等為四？謂離善友無聞非法，諸有情類以人天善根而成熟之，求聲聞者授聲聞乘，求獨覺者授獨覺乘，求大乘者授以大乘。}@
%{这四种重任是什么呢？就是远离善友，不闻非法，使所有众生以人天善根成熟，求声闻乘者授以声闻乘，求独觉乘者授以独觉乘，求大乘者授以大乘。}%
@{是名攝受正法諸善男子及善女人，堪能荷負四種重任逾彼大地。}@
%{这就是说，摄受正法的善男子善女人，能够承担这四种重任，超过大地。}%
@{世尊！如是攝受正法善男子善女人等建立大地，堪能荷負四種重任，普為眾生作不請友，大悲利益哀愍有情為世法母。}@
%{世尊，这样摄受正法的善男子善女人，建立起大地，能够承担这四种重任，普度众生，成为不请之友，以大悲利益哀愍众生，成为世间的母亲。}%
@{又如大地是四種寶所生之處。}@
%{又像大地是四种宝所生的地方。}%
@{何等為四？一者無價，二者上價，三者中價，四者下價。}@
%{这四种宝是什么呢？一是无价宝，二是上等宝，三是中等宝，四是下等宝。}%
@{如是攝受正法善男子善女人建立大地，有情遇已獲四大寶，一切寶中最為殊勝。}@
%{这样摄受正法的善男子善女人，建立起大地，众生遇到他们，就能获得这四种宝，这四种宝是最殊胜的。}%
@{何等為四？謂諸有情遇斯善友，或有獲得人天善根，有證聲聞及辟支佛或無上乘善根功德。}@
%{这四种宝是什么呢？就是众生遇到这些善友，有的能获得人天善根，有的能证得声闻乘和辟支佛乘，或者能获得无上乘的善根功德。}%
@{是名攝受正法善男子善女人建立大地，有情遇已便能獲得四種大寶。}@
%{这就是说，摄受正法的善男子善女人，建立起大地，众生遇到他们，就能获得这四种大宝。}%
@{世尊！出大寶者，名為真實攝受正法。}@
%{世尊，能获得大宝的人，就叫做真正摄受正法。}%
@{世尊！言攝受正法者，謂無異正法、無異攝受正法，正法即是攝受正法。}@
%{世尊，说摄受正法，就是说正法和摄受正法没有区别，正法就是摄受正法。}%
@{世尊！無異波羅蜜、無異攝受正法，攝受正法即是波羅蜜多。}@
%{世尊，波罗蜜和摄受正法没有区别，摄受正法就是波罗蜜多。}%
@{何以故？攝受正法善男子善女人，應以施成熟者以施成熟，乃至捨身隨順彼意而成熟之，令彼有情安住正法，是名施波羅蜜。}@
%{为什么呢？因为摄受正法的善男子善女人，应该以布施来成熟那些需要布施的人，乃至舍身随顺他们的意愿来成熟他们，使他们安住正法，这就是布施波罗蜜。}%
@{應以戒成熟者，守護六根淨身語意乃至威儀，隨順彼意而成熟之，令彼有情安住正法，是名戒波羅蜜。}@
%{应该以持戒来成熟那些需要持戒的人，守护六根，清净身语意，乃至威仪，随顺他们的意愿来成熟他们，使他们安住正法，这就是持戒波罗蜜。}%
@{應以忍成熟者，若彼有情罵詈毀辱誹謗擾亂，以無恚心及利益心最上忍力，乃至顏色亦不變異，隨順彼意而成熟之，令彼有情安住正法，是名忍波羅蜜。}@
%{应该以忍辱来成熟那些需要忍辱的人，如果他们辱骂、毁谤、诽谤、扰乱，以无嗔心和利益心，以最上忍力，乃至脸色也不改变，随顺他们的意愿来成熟他们，使他们安住正法，这就是忍辱波罗蜜。}%
@{應以精進而成熟者，於彼有情不起懈怠下劣之心，起大樂欲最上精進，於四威儀隨順彼意而成熟之，令彼有情安住正法，是名精進波羅蜜。}@
%{应该以精进来成熟那些需要精进的人，对他们不起懈怠和低劣之心，以大乐欲，以最上精进，在四威仪中随顺他们的意愿来成熟他们，使他们安住正法，这就是精进波罗蜜。}%
@{應以靜慮而成熟者，於彼有情以無散亂成就正念，曾所作事終不忘失，隨順彼意而成熟之，令彼有情安住正法，是名靜慮波羅蜜。}@
%{应该以禅定来成熟那些需要禅定的人，对他们以无散乱成就正念，曾经做过的事情永远不会忘记，随顺他们的意愿来成熟他们，使他们安住正法，这就是禅定波罗蜜。}%
@{應以智慧而成熟者，彼諸有情為利益故問諸法義，以無倦心而為演說一切諸論、一切明處乃至種種工巧之處，令得究竟隨順彼意而成熟之，令彼有情安住正法，是名智慧波羅蜜。}@
%{应该以智慧来成熟那些需要智慧的人，他们为了利益而询问诸法义，以无倦心为他们演说一切论典、一切明处，乃至各种巧妙之处，使他们最终获得究竟，随顺他们的意愿来成熟他们，使他们安住正法，这就是智慧波罗蜜。}%
@{是故世尊！無異波羅蜜、無異攝受正法，攝受正法即是波羅蜜。}@
%{所以，世尊，波罗蜜和摄受正法没有区别，摄受正法就是波罗蜜。}%
@{時勝鬘夫人復白佛言：世尊！我今承佛威神辯才之力，復說大義。}@
%{胜鬘夫人又对佛说：“世尊！我现在承蒙佛的威神和辩才之力，再来说说大义。”}%

@{佛言：云何大義？}@
%{佛说：“什么叫做大义呢？”}%

@{世尊！攝受正法者、無異攝受正法。無異攝受正法者，攝受正法善男子善女人，則是攝受正法。何以故？若攝受正法善男子善女人，為正法故捨身命財，如是人等以捨身故，證生死後際遠離老病得不壞常，無有變易究竟寂靜不可思議如來法身。以捨命故，證生死後際永離於死得無邊常，成就不可思議諸善功德，安住一切佛法神變。以捨財故，證生死後際超過有情無盡無減果報圓滿，具不思議功德莊嚴，為諸有情尊重供養。世尊！捨身命財攝受正法善男子善女人等，為諸如來之所授記。世尊！若善男子善女人，正法欲滅，有諸比丘、比丘尼、優婆塞、優婆夷，互相朋黨起諸諍訟，以不諂曲不欺誑心，愛樂正法攝受正法入善朋中。入善朋者，必為諸佛之所授記。世尊！我見攝受正法有斯大力，如來以此為眼、為法根本，為引導法、為通達法。}@
%{“世尊！摄受正法的人，就是摄受正法。摄受正法的人，摄受正法中的善男子善女人，这就是摄受正法。为什么呢？如果摄受正法中的善男子善女人，为了正法而舍弃身命财产，这些人因为舍弃身命，证得生死后际远离老病，得到不坏常，没有变易，究竟寂静，不可思議如来法身。因为舍弃生命，证得生死后际永远脱离死亡，得到无边常，成就不可思議的善功德，安住在一切佛法神变。因为舍弃财产，证得生死后际超越有情，无尽无减，果报圆满，具有不可思议的功德庄严，为诸有情尊重供养。世尊！舍弃身命财产，摄受正法中的善男子善女人等，都是诸如来的授记。世尊！如果善男子善女人，正法将要灭亡，有比丘、比丘尼、优婆塞、优婆夷，互相结党，发生争斗，以不谄曲不欺诳的心，喜爱正法，摄受正法，进入善朋之中。进入善朋的人，一定会得到诸佛的授记。世尊！我看到摄受正法有如此大的力量，如来以此为眼，为法根本，为引导法，为通达法。”}%

@{爾時世尊聞勝鬘夫人所說攝受正法有大威力，歎言：如是如是。善哉勝鬘！如汝所說，攝受正法大威德力。如大力士微觸末摩，生大苦痛更增重病。如是勝鬘！假令少分攝受正法，令魔波旬痛切愁惱悲號歎息，亦復如是。勝鬘！我常不見餘一善法令魔愁惱，猶如少分攝受正法。勝鬘！譬如牛王形色端正，身量殊特蔽於諸牛。如是勝鬘！修大乘者設令少分攝受正法，即能蔽於聲聞獨覺一切善法。勝鬘！又如須彌山王高廣嚴麗蔽於眾山。如是勝鬘！初趣大乘以饒益心不顧身命攝受正法，便能超過顧其身命久住大乘一切善根。是故勝鬘！當以攝受正法開示教化一切有情。如是勝鬘！攝受正法獲大福利及大果報。勝鬘！我於無數阿僧祇劫稱讚如是攝受正法所有功德，不得邊際，是故攝受正法成就如是無量功德。}@
%{当时，世尊听到胜鬘夫人所说摄受正法有大威力，就感叹说：“如是如是。善哉胜鬘！正如你所说，摄受正法有巨大的威德力。就像大力士稍微触碰末摩，就会生出巨大的痛苦，更加加重病情。同样，胜鬘！如果稍微摄受一点正法，就会让魔波旬痛苦万分，愁恼悲伤，悲号叹息。胜鬘！我从来没见过其他任何善法能让魔如此愁恼，就像稍微摄受一点正法一样。胜鬘！譬如牛王形色端正，身量殊特，遮蔽其他所有的牛。同样，胜鬘！修大乘的人，即使稍微摄受一点正法，就能遮蔽声闻独觉一切善法。胜鬘！又如须弥山王高广严丽，遮蔽其他所有的山。同样，胜鬘！初趣大乘，以饶益心不顾身命，摄受正法，就能超过那些顾及自身生命，久住大乘的一切善根。所以，胜鬘！应该用摄受正法来开示教化一切有情。这样，胜鬘！摄受正法就能获得巨大的福利和果报。胜鬘！我在无量阿僧祇劫中赞叹摄受正法所有的功德，都找不到边际，所以摄受正法才能成就如此无量的功德。”}%

@{佛告勝鬘：汝今復應演我所說攝受正法，一切諸佛共所愛樂。}@
%{佛对胜鬘说：“你现在应该再演说我所说的摄受正法，一切诸佛都共同喜爱。”}%

@{勝鬘白言：善哉世尊！攝受正法者則名大乘。何以故？大乘者出生一切聲聞獨覺、世出世間所有善法。如阿耨達池出八大河，如是大乘出生一切聲聞獨覺、世出世間所有善法。世尊！又如一切種子草木叢林，皆依大地而得生長。如是一切聲聞獨覺、世出世間所有善法，皆依大乘而得生長。是故世尊！住於大乘攝受大乘，即住攝受聲聞獨覺、世出世間所有善法。如佛世尊所說六處，謂正法住、正法滅、別解脫、毘奈耶、正出家、受具足，為大乘故說此六處。所以者何？正法住者為大乘說，大乘住者即正法住。正法滅者為大乘說，大乘滅者即正法滅。別解脫、毘奈耶，此之二法義一名異。毘奈耶者即大乘學。所以者何？為佛出家而受具足，是故大乘戒蘊是毘奈耶，是正出家是受具足。世尊！阿羅漢者無有出家及受具足。何以故？阿羅漢不為如來出家受具足故。阿羅漢有怖畏想，歸依如來。何以故？阿羅漢於一切行住怖畏想，如人執劍欲來害己，是故阿羅漢不證出離究竟安樂。世尊！依不求依，如諸眾生無有歸依，彼彼恐怖，為安隱故求於歸依。世尊！如是阿羅漢有恐怖故歸依如來。是故阿羅漢及辟支佛，生法有餘、梵行未立、所作未辦、當有所斷未究竟故，去涅槃遠。何以故？唯有如來、應、正等覺證得涅槃，成就無量不可思議一切功德，所應斷者皆悉已斷究竟清淨，為諸有情之所瞻仰，超過二乘菩薩境界。阿羅漢等則不如是，言得涅槃佛之方便，是故阿羅漢等去涅槃遠。世尊說阿羅漢及辟支佛，觀察解脫四智究竟得蘇息者，皆是如來隨他意語不了義說。何以故？有二種死。何等為二？一者分段，二者變易。分段死者，謂相續有情。變易死者，謂阿羅漢及辟支佛、自在菩薩，隨意生身乃至菩提。二種死中，以分段死說阿羅漢及辟支佛，生於我生已盡之智。由能證得有餘果故，生於梵行已立之智，一切愚夫所不能作，七種學人未能成辦，相續煩惱究竟斷故，生於所作已辦之智。世尊說生不受後有智者，謂阿羅漢及辟支佛不能斷於一切煩惱，不了一切受生之智。何以故？是阿羅漢及辟支佛有餘煩惱不斷盡故，不能了知一切受生。煩惱有二，謂住地煩惱及起煩惱。住地有四。何等為四？謂見一處住地，欲愛住地，色愛住地，有愛住地。世尊！此四住地，能生一切遍起煩惱。起煩惱者，剎那剎那與心相應。世尊！無明住地，無始時來心不相應。世尊！四住地力能作遍起煩惱所依，比無明地，算數譬喻所不能及。世尊！如是無明住地，於有愛住地，其力最大。譬如魔王色力威德及眾眷屬，蔽於他化自在諸天。如是無明住地蔽四住地，過恒沙數煩惱所依，亦令四種煩惱久住。聲聞獨覺智不能斷，唯有如來智所能斷。世尊！如是如是，無明住地其力最大。世尊！如取為緣，有漏業因而生三有。如是無明住地為緣，無漏業因能生阿羅漢及辟支佛、大力菩薩隨意生身。此之三地隨意生身及無漏業，皆以無明住地為所依處。彼雖有緣，亦能為緣。世尊！是故三種隨意生身及無漏業，皆以無明住地為緣，同於有愛。世尊！有愛住地不與無明住地業同，無明住地異四住地。異四住地唯佛能斷。何以故？阿羅漢辟支佛斷四住地，於漏盡力不得自在，不能現證。何以故？世尊言漏盡之增語，是故阿羅漢辟支佛及最後有諸菩薩等，為無明地所覆蔽故，於彼彼法不知不見。以不知見於彼彼法，應斷不斷、應盡不盡。於彼彼法不斷不盡故，得有餘解脫，非一切解脫；得有餘清淨，非一切清淨；得有餘功德，非一切功德。世尊！以得有餘解脫非一切解脫，乃至有餘功德非一切功德故，知有餘苦、斷有餘集、證有餘滅、修有餘道。}@
%{胜鬘说：“善哉世尊！摄受正法的人，就叫做大乘。为什么呢？大乘能出生一切声闻独觉，世出世间所有的善法。就像阿耨达池涌出八大河一样，大乘能出生一切声闻独觉，世出世间所有的善法。世尊！又像一切种子、草木、丛林，都依靠大地而生长一样，一切声闻独觉，世出世间所有的善法，都依靠大乘而生长。所以，世尊！住在大乘，摄受大乘，就是住摄受声闻独觉，世出世间所有的善法。就像佛世尊所说的六处，即正法住、正法灭、别解脱、毘奈耶、正出家、受具足，都是为了大乘而说的。为什么呢？正法住，是为了大乘而说，大乘住，就是正法住。正法灭，是为了大乘而说，大乘灭，就是正法灭。别解脱、毘奈耶，这两个法义相同，名称不同。毘奈耶，就是大乘学。为什么呢？为了佛出家而受具足，所以大乘戒蕴就是毘奈耶，正出家就是受具足。世尊！阿罗汉没有出家和受具足。为什么呢？因为阿罗汉不是为了如来出家受具足。阿罗汉有怖畏想，归依如来。为什么呢？因为阿罗汉在一切行住中都有怖畏想，就像有人拿着剑要来伤害自己一样，所以阿罗汉不能证得出离，究竟安乐。世尊！依不求依，就像诸众生没有归依，彼彼恐怖，为了安隐而求归依。世尊！同样，阿罗汉有恐怖，所以归依如来。所以，阿罗汉和辟支佛，因为生法有余，梵行未立，所作未办，应该断的还没有究竟，所以远离涅槃。为什么呢？只有如来、应、正等觉证得涅槃，成就无量不可思議的一切功德，应该断的都彻底断尽，究竟清净，为诸有情之所瞻仰，超越二乘菩萨境界。阿罗汉等就不一样，他们说得到涅槃是佛的方便，所以阿罗汉等远离涅槃。世尊说阿罗汉和辟支佛，观察解脱四智，究竟得到苏息的人，都是如来随他意语，不了义说。为什么呢？有两种死。哪两种呢？一是分段，二是变易。分段死，是指相续有情。变易死，是指阿罗汉和辟支佛、自在菩萨，随心所欲地生身，直到菩提。在两种死中，用分段死来说阿罗汉和辟支佛，生于我生已尽之智。因为能证得有余果，所以生于梵行已立之智，一切愚夫所不能作，七种学人未能成办，相续烦恼彻底断尽，所以生于所作已办之智。世尊说生不受后有智的人，是指阿罗汉和辟支佛不能断一切烦恼，不了解一切受生之智。为什么呢？因为阿罗汉和辟支佛还有余烦恼没有彻底断尽，所以不能了知一切受生。烦恼有两种，一是住地烦恼，一是起烦恼。住地有四种。哪四种呢？一是见一处住地，二是欲爱住地，三是色爱住地，四是有爱住地。世尊！这四种住地，能生一切遍起烦恼。起烦恼，是刹那刹那与心相应。世尊！无明住地，从无始时来，与心不相應。世尊！四住地之力能作遍起烦恼所依，比无明地，算数譬喻所不能及。世尊！同样，无明住地，对于有爱住地，其力量最大。譬如魔王色力威德和众多眷属，遮蔽他化自在诸天。同样，无明住地遮蔽四住地，超过恒河沙数烦恼所依，也让四种烦恼久住。声闻独觉智不能断，只有如来智才能断。世尊！如是如是，无明住地其力量最大。世尊！就像取为缘，有漏业因而生三有。同样，无明住地为缘，无漏业因能生阿罗汉和辟支佛、大力菩萨随心所欲地生身。这三种随心所欲地生身和无漏业，都以无明住地为所依处。它虽然有缘，也能成为缘。世尊！所以，三种随心所欲地生身和无漏业，都以无明住地为缘，同于有爱。世尊！有爱住地不与无明住地业相同，无明住地不同于四住地。不同于四住地，只有佛才能断。为什么呢？阿罗汉辟支佛断四住地，在漏尽力上不能自在，不能现证。为什么呢？世尊说漏尽是增语，所以阿罗汉辟支佛和最后的诸菩萨等，因为被无明地所遮蔽，所以对彼彼法不知不见。因为不知不见彼彼法，应该断的没有断，应该尽的没有尽。在彼彼法上没有断没有尽，所以得到有余解脱，不是一切解脱；得到有余清净，不是一切清净；得到有余功德，不是一切功德。世尊！因为得到有余解脱，不是一切解脱，乃至得到有余功德，不是一切功德，所以知道有余苦，断有余集，证有余灭，修有余道。”}%
@{爾時勝鬘夫人復白佛言：世尊！若復知有餘苦、斷有餘集、證有餘滅、修有餘道，是名少分滅度，證少分涅槃向涅槃界。}@
%{当时胜鬘夫人又对佛陀说：世尊！如果只知道有余苦、断有余集、证有余灭、修有余道，这叫做少分灭度，证得少分涅槃，向涅槃界。}%
@{若知一切苦、斷一切集、證一切滅、修一切道，彼於無常敗壞世間得證常寂清涼涅槃。}@
%{如果知道一切苦、断一切集、证一切灭、修一切道，那么他就能在无常败坏的世间证得常寂清凉的涅槃。}%
@{世尊！彼於無護無依世間為護為依。}@
%{世尊！他就能在无护无依的世间成为护持和依靠。}%
@{何以故？於諸法中見高下者不證涅槃，智平等者、解脫等者、清淨等者乃證涅槃。}@
%{为什么呢？因为在诸法中看到高低的人不能证得涅槃，只有智慧平等、解脱平等、清净平等的人才能证得涅槃。}%
@{是故涅槃名等一味。}@
%{所以涅槃叫做等一味。}%
@{云何一味？謂解脫味。}@
%{什么叫做等一味呢？就是解脱的味道。}%
@{世尊！若無明地不斷不盡，不得涅槃一味等味。}@
%{世尊！如果无明地不断不尽，就不能得到涅槃一味等味。}%
@{何以故？無明住地不斷不盡故，過恒沙等一切過法，應斷不斷、應盡不盡。}@
%{为什么呢？因为无明住地不断不尽，所以超过恒河沙数的一切过法，应该断的不断，应该尽的无尽。}%
@{過恒沙等一切過法不斷不盡故，過恒沙等諸功德法不了不證。}@
%{超过恒河沙数的一切过法不断不尽，所以超过恒河沙数的功德法就不能了知和证得。}%
@{是故無明住地與於一切所應斷法諸隨煩惱為生處故，從於彼生，障心煩惱、障止煩惱、障觀煩惱、障靜慮煩惱，如是乃至障三摩鉢底加行智果證力無畏。}@
%{所以无明住地是所有应该断法和随烦恼的生处，从它生出来，就障蔽心烦惱、障蔽止烦惱、障蔽观烦惱、障蔽静虑烦惱，像这样一直到障蔽三摩鉢底加行智果证力无畏。}%
@{所有過恒沙等一切煩惱，如來菩提、佛金剛智之所能斷。}@
%{所有超过恒河沙数的一切烦恼，都是如来菩提、佛金刚智所能断的。}%
@{諸起煩惱，一切皆依無明住地，無明住地為因緣故。}@
%{所有生起的烦恼，都是依无明住地，无明住地是它的因缘。}%
@{世尊！此起煩惱，剎那剎那與心相應。}@
%{世尊！这些生起的烦恼，每一刹那都和心相对应。}%
@{世尊！無明住地從無始來心不相應。}@
%{世尊！无明住地从无始以来就和心不相对应。}%
@{世尊！若復過恒河沙如來菩提、佛金剛智所應斷法，一切皆是無明住地依持建立。}@
%{世尊！如果超过恒河沙数的如来菩提、佛金刚智所应该断的法，都是依无明住地建立起来的。}%
@{譬如一切種子叢林，皆依大地之所生長，若地壞者彼亦隨壞。}@
%{比如一切种子丛林，都是依大地而生长，如果大地坏了，它们也会随之坏掉。}%
@{如是過恒沙等如來菩提、佛金剛智所應斷法，一切皆依無明住地之所生長。}@
%{像这样，超过恒河沙数的如来菩提、佛金刚智所应该断的法，都是依无明住地而生长。}%
@{若彼無明住地斷者，過恒沙等如來菩提、佛金剛智所應斷法，皆亦隨斷。}@
%{如果无明住地断掉了，超过恒河沙数的如来菩提、佛金刚智所应该断的法，也都随之断掉。}%
@{如是過恒沙等所應斷法，一切煩惱及起煩惱皆已斷故，便能證得過恒沙等不可思議諸佛之法。}@
%{像这样，超过恒河沙数的应该断法，所有烦恼和生起的烦恼都断掉了，就能证得超过恒河沙数的不可思议诸佛之法。}%
@{於一切法而能證得無礙神通，得諸智見離一切過，得諸功德為大法王於法自在，證一切法自在之地，正師子吼，我生已盡、梵行已立、所作已辦、不受後有。}@
%{在一切法中都能证得无碍神通，得到诸智慧见，离开一切过失，得到诸功德，成为大法王，在法中自在，证得一切法自在之地，正狮子吼：我生已尽、梵行已立、所作已办、不受后有。}%
@{是故世尊以師子吼，依於了義一向記說。}@
%{所以世尊以狮子吼，依于了义一向记说。}%
@{世尊！不受後有智有二種。}@
%{世尊！不受后有智有两种。}%
@{何謂為二？一者謂諸如來以調御力摧伏四魔超諸世間，一切有情之所瞻仰；證不思議清淨法身，於所知地得法自在，最勝無上更無所作，不見更有所證之地；具足十力，登於最勝無畏之地，於一切法無礙觀察，正師子吼，不受後有。}@
%{哪两种呢？第一种，就是诸如来以调御力摧伏四魔，超越诸世间，一切有情都瞻仰他；证得不可思议的清净法身，在所知地得到法自在，最胜无上，再没有所作，不见再有所证之地；具足十力，登上最胜无畏之地，对一切法无碍观察，正狮子吼，不受后有。}%
@{二者謂阿羅漢及辟支佛，得度無量生死怖畏，受解脫樂，作如是念：我今已離生死怖畏，不受諸苦。}@
%{第二种，就是阿罗汉和辟支佛，得到度过无量生死怖畏，受解脱乐，作如是念：我今已离开生死怖畏，不受诸苦。}%
@{世尊！阿羅漢辟支佛如是觀察謂不受後有，不證第一蘇息涅槃。}@
%{世尊！阿罗汉辟支佛这样观察，说不受后有，不证得第一苏息涅槃。}%
@{彼等於未證地不遇法故，能自解了，我今證得有餘依地，決定當證阿耨多羅三藐三菩提。}@
%{他们因为在未证得之地没有遇到法，所以能自己解了：我今证得有余依地，决定当证得阿耨多罗三藐三菩提。}%
@{何以故？聲聞獨覺皆入大乘，而大乘者即是佛乘，是故三乘即是一乘，證一乘者得阿耨多羅三藐三菩提，阿耨多羅三藐三菩提者即是涅槃，言涅槃者即是如來清淨法身，證法身者即是一乘，無異如來、無異法身。}@
%{为什么呢？因为声闻和独觉都进入大乘，而大乘就是佛乘，所以三乘就是一乘，证得一乘者就得到阿耨多罗三藐三菩提，阿耨多罗三藐三菩提就是涅槃，说涅槃就是如来清净法身，证得法身就是一乘，没有区别于如来，没有区别于法身。}%
@{言如來者即是法身，證究竟法身者即究竟一乘，究竟一乘者即離相續。}@
%{说如来就是法身，证得究竟法身就是究竟一乘，究竟一乘就是离开相续。}%
@{何以故？世尊！如來住時無有限量等於後際，如來能以無限大悲無限誓願利益世間，作是說者是名善說。}@
%{为什么呢？世尊！如来住世时没有有限量，等同于后际，如来能以无限大悲无限誓愿利益世间，这样说就是善说。}%
@{若復說言，如來是常、是無盡法、一切世間究竟依者，亦名善說。}@
%{如果再说：如来是常、是无尽法、一切世间究竟依者，也叫做善说。}%
@{是故能於無護世間無依世間與等後際，作無盡歸依、常住歸依、究竟歸依者，謂如來、應、正等覺。}@
%{所以，能于无护世间无依世间和等同后际，作无尽归依、常住归依、究竟归依者，就是如来、应、正等觉。}%
@{法者是一乘道，僧者是三乘眾，此二歸依非究竟依，名少分依。}@
%{法就是一乘道，僧就是三乘众，这两种归依不是究竟依，叫做少分依。}%
@{何以故？說一乘道證究竟法身，於後更無說一乘道，三乘眾者有恐怖故；歸依如來求出修學，有所作故，向阿耨多羅三藐三菩提故。}@
%{为什么呢？因为说一乘道证得究竟法身，以后不再说一乘道，三乘众有恐怖，所以要归依如来寻求出离修学，有所作，向阿耨多罗三藐三菩提。}%
@{二依非究竟依，是有限依。}@
%{这两种归依不是究竟依，是有限依。}%
@{若諸有情如來調伏，歸依如來得法津潤，由信樂心歸依於法及比丘僧。}@
%{如果诸有情被如来调伏，归依如来得到法津润，由信乐心归依于法和比丘僧。}%
@{是二歸依，由法津潤信入歸依。}@
%{这两种归依，是由法津润信入归依。}%
@{如來者非法津潤信入歸依，言如來者是真實依。}@
%{如来不是法津润信入归依，说如来就是真实依。}%
@{此二歸依以真實義，即名究竟歸依如來。}@
%{这两种归依以真实义，就叫做究竟归依如来。}%
@{何以故？如來不異此二歸依，是故如來即三歸依。}@
%{为什么呢？因为如来不异于这两种归依，所以如来就是三归依。}%
@{何以故？說一乘道、如來最勝、具四無畏、正師子吼。}@
%{为什么呢？因为说一乘道、如来最胜、具足四无畏、正狮子吼。}%
@{若諸如來隨彼所欲，而以方便說於二乘即是大乘。}@
%{如果诸如来随他所欲，以方便说于二乘就是大乘。}%
@{以第一義無有二乘，二乘者同入一乘，一乘者即勝義乘。}@
%{以第一义没有二乘，二乘者同入一乘，一乘就是胜义乘。}%
@{世尊！聲聞獨覺初證聖諦，非以一智斷諸住地，亦非一智證四遍知諸功德等，亦非以法能善了知此四法義。}@
%{世尊！声闻独觉初证圣谛，不是以一种智慧断诸住地，也不是以一种智慧证得四遍知诸功德等，也不是以法能善了知这四法义。}%
@{世尊！於出世智，無有四智漸至漸緣。}@
%{世尊！在出世智中，没有四种智慧渐至渐缘。}%
@{世尊！出世間智無漸至法，如金剛喻。}@
%{世尊！出世间智没有渐至的法，就像金刚喻。}%
@{世尊！聲聞獨覺以於種種聖諦之智斷諸住地，無有出世第一義智。}@
%{世尊！声闻独觉以种种圣谛之智断诸住地，没有出世第一义智。}%
@{唯有如來、應、正遍知，非諸聲聞獨覺境界，以不思議空性之智，能破一切諸煩惱㲉。}@
%{只有如来、应、正遍知，不是声闻独觉的境界，以不可思议空性之智，能破一切烦恼㲉。}%
@{世尊！破煩惱㲉究竟之智，是名出世第一義智。}@
%{世尊！破烦恼㲉究竟的智慧，就是出世第一义智。}%
@{初聖諦智非究竟智，是於趣向阿耨多羅三藐三菩提智。}@
%{初圣谛智不是究竟智，是趣向阿耨多罗三藐三菩提的智慧。}%
@{世尊！真聖義者即非二乘。}@
%{世尊！真圣义者就不是二乘。}%
@{何以故？聲聞獨覺唯能成就少分功德，名之為聖。}@
%{为什么呢？因为声闻独觉只能成就少分功德，叫做圣。}%
@{世尊！言聖諦者，非諸聲聞獨覺之諦及彼功德。}@
%{世尊！说圣谛者，不是声闻独觉的谛和他们的功德。}%
@{而此諦者，唯有如來、應、正等覺初始了知，然後為彼無明㲉藏世間眾生開示演說，故名聖諦。}@
%{而这个谛，只有如来、应、正等觉最初了知，然后为那些无明㲉藏世间众生开示演说，所以叫做圣谛。}%
@{世尊！此聖諦者，甚深微妙、難見難了、不可分別、非思量境，一切世間所不能信。}@
%{世尊！这个圣谛，非常深妙，难见难了，不可分别，不是思量境，一切世间都不可信。}%
@{唯有如來、應、正等覺之所能知。}@
%{只有如来、应、正等觉才能知道。}%
@{何以故？此說甚深如來之藏。}@
%{为什么呢？因为这个说的是深妙的如来之藏。}%
@{如來藏者是佛境界，非諸聲聞獨覺所行。}@
%{如来藏是佛的境界，不是声闻独觉所行的。}%
@{於如來藏說聖諦義，此如來藏甚深微妙，所說聖諦亦復深妙，難見難了、不可分別、非思量境，一切世間所不能信，唯有如來、應、正等覺之所能知。}@
%{在如来藏中说圣谛义，这个如来藏非常深妙，所说的圣谛也深妙，难见难了，不可分别，不是思量境，一切世间都不可信，只有如来、应、正等觉才能知道。}%
@{若於無量煩惱所纏如來之藏不疑惑者，於出一切煩惱之藏如來法身亦無疑惑。}@
%{如果对无量烦恼所缠绕的如来之藏没有疑惑，对超出一切烦恼之藏的如来法身也无疑惑。}%
@{世尊！若有於此如來之藏及佛法身不可思議佛祕密境心得究竟，於彼所說二聖諦義能信能了、能生勝解。}@
%{世尊！如果有谁对这个如来之藏和佛法身不可思议的佛秘密境界得到究竟，对他说出的两种圣谛义能信能了，能生起胜解。}%
@{何等名為二聖諦義？所謂有作及以無作。}@
%{哪两种叫做两种圣谛义呢？就是有作和无作。}%
@{作聖諦者，是不圓滿四聖諦義。}@
%{作圣谛，是不圆满四圣谛义。}%
@{何以故？由他護故，而不能得知一切苦、斷一切集、證一切滅、修一切道，是故不知有為無為及於涅槃。}@
%{为什么呢？因为由他人护持，所以不能知道一切苦、断一切集、证一切灭、修一切道，所以不知道有为无为和涅槃。}%
@{世尊！無作諦者，是說圓滿四聖諦義。}@
%{世尊！无作谛，是说圆满四圣谛义。}%
@{何以故？能自護故，知一切苦、斷一切集、證一切滅、修一切道。}@
%{为什么呢？因为能自己护持，所以知道一切苦、断一切集、证一切灭、修一切道。}%
@{如是所說八聖諦義，如來但以四聖諦說。}@
%{像这样所說的八圣谛义，如来只以四圣谛说。}%
@{於此無作四聖諦義，唯有如來、應、正等覺作事究竟，非阿羅漢及辟支佛力所能及。}@
%{在这个无作四圣谛义中，只有如来、应、正等觉作事究竟，不是阿罗汉和辟支佛力所能及的。}%
@{何以故？非諸勝劣下中上法能證涅槃。}@
%{为什么呢？因为不是诸胜劣下中上法能证得涅槃。}%
@{云何如來於無作諦得事究竟？謂諸如來、應、正等覺，遍知諸苦，斷諸煩惱及超煩惱所攝苦集，能證一切意生身蘊所有苦滅，及修一切苦滅之道。}@
%{如来如何在无作谛中得事究竟呢？就是诸如来、应、正等觉，遍知诸苦，断诸烦恼和超越烦恼所摄的苦集，能证得一切意生身蕴所有苦灭，以及修一切苦灭之道。}%
@{世尊！非壞法故名為苦滅。}@
%{世尊！不是坏法，所以叫做苦灭。}%
@{何以故？言苦滅者，無始無作、無起無盡、常住不動、本性清淨、出煩惱㲉。}@
%{为什么呢？说苦灭，就是无始无作、无起无尽、常住不动、本性清净、超出烦恼㲉。}%
@{世尊！如來成就過於恒沙，具解脫智不思議法，說名法身。}@
%{世尊！如来成就超过恒河沙数，具足解脱智不可思议法，说叫做法身。}%
@{世尊！如是法身不離煩惱，名如來藏。}@
%{世尊！像这样的法身不离开烦恼，叫做如来藏。}%
@{世尊！如來藏者，即是如來空性之智。}@
%{世尊！如来藏，就是如来空性之智。}%
@{如來藏者，一切聲聞獨覺所未曾見亦未曾得，唯佛了知及能作證。}@
%{如来藏，一切声闻独觉都没有见过也没有得到，只有佛了知和能作证。}%
@{世尊！此如來藏空性之智復有二種。}@
%{世尊！这个如来藏空性之智还有两种。}%
@{何等為二？謂空如來藏，所謂離於不解脫智一切煩惱。}@
%{哪两种呢？叫做空如来藏，就是离开不解脱智的一切烦恼。}%
@{世尊！不空如來藏具過恒沙佛解脫智不思議法。}@
%{世尊！不空如来藏具足超过恒河沙数的佛解脱智不可思议法。}%
@{世尊！此二空智，諸大聲聞由信能入。}@
%{世尊！这两种空智，诸大聲聞由信就能进入。}%
@{世尊！如是一切聲聞獨覺空性之智，於四倒境攀緣而轉，是故一切聲聞獨覺所未曾見亦未曾證。}@
%{世尊！像这样，一切声闻独觉的空性之智，在四倒境攀缘而转，所以一切声闻独觉都没有见过也没有证得。}%
@{一切苦滅唯佛現證，壞諸煩惱修苦滅道。}@
%{一切苦灭只有佛能现证，破除诸烦恼，修苦灭道。}%
@{世尊！此四諦中，三諦無常，一諦是常。}@
%{世尊！这四谛中，三谛无常，一谛是常。}%
@{何以故？如是三諦入有為相，有為相者則是無常。}@
%{为什么呢？因为这三谛进入有为相，有为相就是无常。}%
@{言無常者是破壞法，破壞法者非諦非常非歸依處。}@
%{说无常，就是破壞法，破壞法不是谛，不是常，不是归依处。}%
@{是故三諦以第一義非諦非常非歸依處。}@
%{所以，三谛以第一义不是谛，不是常，不是归依处。}%
@{世尊！一苦滅諦離有為相，離有為相則性常住，性常住者非破壞法，非破壞者是諦是常、是歸依處。}@
%{世尊！一苦灭谛离开有为相，离开有为相，它的性就常住，性常住者不是破壞法，不是破壞法，就是谛，就是常，就是归依处。}%
@{世尊！是故苦滅聖諦以勝義故，是諦是常是歸依處。}@
%{世尊！所以，苦灭圣谛以胜义故，是谛，是常，是归依处。}%
@{世尊！此苦滅諦是不思議，過諸有情心識境界，亦非一切聲聞獨覺智所能及。}@
%{世尊！这个苦灭谛不可思议，超过诸有情心识境界，也不是一切声闻独觉智所能及的。}%
@{譬如生盲不見眾色，七日嬰兒不見日輪。}@
%{比如生来就瞎的人看不见各种颜色，七天的婴儿看不见太阳。}%
@{苦滅諦者亦復如是，非諸凡夫心識所緣，亦非一切聲聞獨覺智之境界。}@
%{苦灭谛也是这样，不是凡夫心识所缘，也不是一切声闻独觉智的境界。}%
@{凡夫識者謂二邊見，一切聲聞獨覺智者名為淨智。}@
%{凡夫识就是说二边见，一切声闻独觉智叫做净智。}%
@{言邊見者，於五取蘊執著為我，生異分別。}@
%{说边见，就是执着于五取蕴，认为是我，生起不同的分别。}%
@{邊見有二。}@
%{边见有两种。}%
@{何者為二？所謂常見及以斷見。}@
%{哪两种呢？就是常见和断见。}%
@{世尊！若復有見生死無常涅槃是常，非斷常見是名正見。}@
%{世尊！如果有人看到生死无常，涅槃是常，没有断常见，叫做正见。}%
@{何以故？諸計度者見身諸根受者思者現法滅壞，於有相續不能了知，盲無慧目起於斷見。}@
%{为什么呢？因为那些计度者看到身、诸根、受者、思者现法灭坏，对有相续不能了知，盲目无慧眼，就生起断见。}%
@{於心相續剎那滅壞，愚闇不了意識境界，起於常見。}@
%{对心相续刹那灭坏，愚暗不了知意识境界，就生起常见。}%
@{世尊！然彼彼義過諸分別及下劣見，由諸愚夫妄生異想顛倒執著，謂斷謂常。}@
%{世尊！然而，那些见解超过诸分别和下劣见，由那些愚夫妄生异想，颠倒执着，说断说常。}%
@{世尊！顛倒有情於五取蘊，無常常想、苦為樂想、無我我想、不淨淨想。}@
%{佛陀啊！那些颠倒众生，执着于五蕴，认为五蕴是常、是乐、是我、是净，而没有常想、乐想、我相、净相。}%
@{聲聞獨覺所有淨智，於如來境及佛法身所未曾見。}@
%{声闻、独觉虽然拥有清净智慧，但他们从未见过如来的境界和佛法身。}%
@{或有眾生信如來故，於如來所起於常想、樂想、我想及於淨想，非顛倒見即是正見。}@
%{或许有些众生因为信仰如来，而在如来身上生起常想、乐想、我相和净相，这并非颠倒见，而是正见。}%
@{何以故？如來法身是常波羅蜜、樂波羅蜜、我波羅蜜、淨波羅蜜。}@
%{为什么呢？因为如来的法身就是常波罗蜜、乐波罗蜜、我波罗蜜、净波罗蜜。}%
@{若諸有情作如是見是名正見，若正見者名真佛子，從佛口生、從正法生、從法化生、得佛法分。}@
%{如果众生能这样看待，这就是正见。拥有正见的人就是真正的佛子，他们从佛的口中出生，从正法中出生，从法中化生，得到佛法的份额。}%
@{世尊！言淨智者，則是一切聲聞獨覺智波羅蜜。}@
%{佛陀啊！所说的清净智慧，就是一切声闻、独觉的智慧波罗蜜。}%
@{此之淨智，於苦滅諦尚非境界，況苦滅諦。}@
%{这种清净智慧，连苦灭谛都不是它的境界，更不用说苦灭谛了。}%
@{是四入流智之所行。}@
%{它是四入流智的修行对象。}%
@{何以故？三乘初業不愚法者，能於彼義當證當了。}@
%{为什么呢？因为三乘初学者不愚痴于法，就能证悟并彻底理解这些道理。}%
@{世尊！為何義故說四入流？}@
%{佛陀啊！为什么说四入流呢？}%
@{世尊！此四入流是世間法。}@
%{佛陀啊！这四入流是世间法。}%
@{世尊！唯一入流，於諸入流為最為上，以第一義是為入流、是為歸依、是苦滅諦。}@
%{佛陀啊！唯一入流，在所有入流中是最为高尚的，以第一义为入流、为归依、为苦灭谛。}%

@{世尊！生死者依如來藏，以如來藏故說前際不可了知。}@
%{佛陀啊！生死依于如来藏，因为有如来藏，所以说前际不可知。}%
@{世尊！有如來藏故得有生死，是名善說。}@
%{佛陀啊！因为有如来藏，所以才有生死，这才是善说。}%
@{世尊！生死者，諸受根滅無間相續未受根起，名為生死。}@
%{佛陀啊！生死是指所有受根灭尽，无间相续，未受根生起，这叫做生死。}%
@{世尊！生死二法是如來藏，於世俗法名為生死。}@
%{佛陀啊！生死这两个法就是如来藏，在世俗法中被称为生死。}%
@{世尊！死者諸受根滅，生者諸受根起，如來藏者則不生不死、不昇不墜、離有為相。}@
%{佛陀啊！死亡是指所有受根灭尽，出生是指所有受根生起，如来藏则不生不死、不升不坠、远离有为相。}%
@{世尊！如來藏者常恒不壞，是故世尊！如來藏者與不離解脫智藏是依是持，是為建立。}@
%{佛陀啊！如来藏是永恒不坏的，所以佛陀啊！如来藏与不离解脱智藏是互相依持、互相建立。}%
@{亦與外離不解脫智諸有為法依持建立。}@
%{它也与远离解脱智的所有有为法互相依持、互相建立。}%
@{世尊！若無如來藏者，應無厭苦樂求涅槃。}@
%{佛陀啊！如果没有如来藏，就应该没有厌苦乐求涅槃。}%
@{何以故？於此六識及以所知如是七法，剎那不住不受眾苦，不堪厭離願求涅槃。}@
%{为什么呢？因为六识和所知这七法，刹那不住，承受着各种痛苦，无法厌离，也无法渴望涅槃。}%
@{如來藏者，無有前際無生無滅，法受諸苦。}@
%{如来藏没有前际，没有生灭，却承受着各种痛苦。}%
@{彼為厭苦，願求涅槃。}@
%{它厌恶痛苦，渴望涅槃。}%
@{世尊！如來藏者，非有我人眾生壽者。}@
%{佛陀啊！如来藏并非有我、有人、有众生、有寿命。}%
@{如來藏者，身見有情、顛倒有情、空見有情，非所行境。}@
%{如来藏并非身见众生、颠倒众生、空见众生，它不是修行对象。}%
@{世尊！如來藏者是法界藏，是法身藏、出世間藏、性清淨藏，此本性淨。}@
%{佛陀啊！如来藏是法界藏，是法身藏、出世间藏、性清净藏，它的本性是清净的。}%
@{如來藏者如我所解，縱為客塵煩惱所染，猶是不可思議如來境界。}@
%{如来藏，正如我所理解的，即使被客尘烦恼染污，仍然是不可思议的如来境界。}%
@{何以故？世尊！剎那剎那善不善心客塵煩惱所不能染。}@
%{为什么呢？佛陀啊！因为刹那刹那的善不善心，客尘烦恼无法染污。}%
@{何以故？煩惱不觸心，心不觸煩惱。}@
%{为什么呢？因为烦恼不触心，心也不触烦恼。}%
@{云何不觸法而能得染心？}@
%{如何不触法就能染污心呢？}%
@{世尊！由有煩惱有隨染心。}@
%{佛陀啊！因为有烦恼，所以有随烦恼染污的心。}%
@{隨煩惱染難解難了，唯佛世尊為眼、為智、為法根本、為尊、為導、為正法依，如實知見。}@
%{随烦恼染污，难以理解，难以证悟，只有佛陀才是眼、是智慧、是法的根本、是尊、是导、是正法的依靠，能如实知见。}%

@{爾時世尊歎勝鬘夫人言：善哉善哉，如汝所說。}@
%{这时，佛陀赞叹胜鬘夫人说：“善哉善哉，正如你所说。”}%
@{性清淨心隨煩惱染，難可了知。}@
%{性清净心被烦恼染污，难以了知。}%
@{復次勝鬘！有二種法難可了知。}@
%{胜鬘啊！还有两种法难以了知。}%
@{何等為二？謂性清淨心難可了知，彼心為煩惱染亦難了知。}@
%{哪两种呢？就是性清净心难以了知，它被烦恼染污也难以了知。}%
@{如此二法，汝及成就大法菩薩乃能聽受，諸餘聲聞由信能解。}@
%{这两种法，只有你以及成就大法菩萨才能听受，其他声闻只能凭借信仰来理解。}%
@{勝鬘！若我弟子增上信者，隨順法智，於此法中而得究竟。}@
%{胜鬘啊！如果我的弟子增长信仰，随顺法智，就能在这个法中获得究竟。}%
@{順法智者，觀根識境，觀察業報，觀羅漢眠觀心自在愛樂禪樂，觀聲聞獨覺聖神變通。}@
%{随顺法智，就是观察根、识、境，观察业报，观察罗汉入眠、观心自在、爱乐禅乐，观察声闻、独觉圣神变化。}%
@{由成就此五善巧觀，現在未來聲聞弟子因增上信隨順法智，善能解了性清淨心煩惱所染而得究竟。}@
%{因为成就了这五种善巧观，现在和未来的声闻弟子，因为增长信仰，随顺法智，就能善于理解性清净心被烦恼染污，从而获得究竟。}%
@{勝鬘！是究竟者為大乘因。}@
%{胜鬘啊！这个究竟就是大乘因。}%
@{汝今當知信如來者，於甚深法不生誹謗。}@
%{你现在应该知道，信仰如来的人，对甚深法不会生起诽谤。}%

@{爾時勝鬘夫人白佛言：世尊！復有餘義能多利益。}@
%{这时，胜鬘夫人对佛陀说：“佛陀啊！还有其他道理能带来更多利益。”}%
@{我當承佛威神之力，演說斯事。}@
%{我要凭借佛陀威神的力量，宣说这件事。}%

@{佛言：善哉。今恣汝說。}@
%{佛陀说：“善哉，现在就由你说吧。”}%

@{勝鬘夫人言：有三種善男子善女人，於甚深法離自毀傷，生多功德入大乘道。}@
%{胜鬘夫人说：“有三种善男子善女人，在甚深法中远离自毁伤，生起许多功德，进入大乘道。”}%
@{何等為三？若善男子善女人等，能自成就甚深法智，或有成就隨順法智，或有於此甚深法中不能解了，仰推如來，唯佛所知非我境界。}@
%{哪三种呢？如果善男子善女人能自己成就甚深法智，或者成就随顺法智，或者在这个甚深法中不能理解，就仰推如来，认为只有佛陀才知道，这不是我的境界。}%
@{除此三種善男子善女人已，諸餘有情於甚深法，隨己所取執著妄說，違背正法習諸外道。}@
%{除了这三种善男子善女人之外，其他众生在甚深法中，随自己所取执着妄说，违背正法，学习各种外道。}%
@{腐敗種子，設在餘方應往除滅。}@
%{腐败的种子，即使在其他地方，也应该去消灭。}%
@{彼腐敗者，一切天人應共摧伏。}@
%{那些腐败者，一切天人都应该共同摧伏。}%
@{勝鬘夫人說是語已，與諸眷屬頂禮佛足。}@
%{胜鬘夫人说完这些话，就和她的眷属一起，恭敬地顶礼佛的双足。}%

@{時佛世尊讚言：善哉。勝鬘！於甚深法方便守護，降伏怨敵善能通達。}@
%{这时，佛陀赞叹说：“很好！胜鬘！你能够巧妙地守护甚深法，降伏怨敌，你很善于通达这些道理。”}%

@{汝已親近百千俱胝諸佛如來，能說此義。}@
%{“你已经亲近过无数的佛陀，才能说出这些道理。”}%

@{爾時世尊放勝光明普照大眾，身昇虛空高七多羅量，以神通力足步虛空還舍衛城。}@
%{这时，佛陀放出无量光，照耀大众，然后身体升到虚空，高出地面七多罗量，用神通力在虚空中行走，回到舍卫城。}%

@{時勝鬘夫人與諸眷屬，瞻仰世尊目不暫捨，過眼境已歡喜踊躍，遞共稱歎如來功德，一心念佛還無鬪城，勸友稱王建立大乘。}@
%{这时，胜鬘夫人和她的眷属，都仰望着佛陀，眼睛一刻也不离开，看着佛陀的庄严相，心中欢喜踊跃，互相称赞佛陀的功德，一心念佛，回到无斗城，劝说朋友称王，建立大乘法。}%

@{城中女人七歲已上化以大乘，友稱大王亦以大乘化諸男子七歲已上，舉國人民無不學者。}@
%{城里的女人，七岁以上的都以大乘法来教化，她的朋友称王，也以大乘法来教化七岁以上的男子，整个国家的人民，没有不学习的。}%

@{爾時世尊入逝多林，告尊者阿難及念天帝。}@
%{这时，佛陀来到逝多林，对阿难尊者和念天帝说：“}%

@{時天帝釋與諸眷屬應念而至住於佛前。}@
%{这时，天帝释和他的眷属，应念而至，站在佛陀面前。”}%

@{爾時世尊告帝釋言：憍尸迦！汝當受持此經演說開示，為三十三天得安樂故。}@
%{佛陀对天帝释说：“憍尸迦！你应该持诵这部经，演说开示，让三十三天得到安乐。”}%

@{復告阿難：汝亦受持，為諸四眾分別演說。}@
%{“你也应该持诵这部经，分别为四众演说。”}%

@{時天帝釋白佛言：世尊！當何名斯經？云何奉持？}@
%{这时，天帝释对佛陀说：“世尊！这部经应该叫什么名字？应该怎样奉持呢？”}%

@{佛告天帝：此經成就無邊功德，一切聲聞獨覺力不能及，況餘有情。}@
%{佛陀对天帝释说：“这部经成就了无边的功德，一切声闻、独觉的力量都无法企及，更何况其他的众生呢？”}%

@{憍尸迦當知，此經甚深微妙大功德聚，今當為汝略說其名。諦聽諦聽，善思念之。}@
%{“憍尸迦！你应该知道，这部经非常深奥微妙，蕴含着无量的功德，现在我就简单地告诉你它的名字，你仔细听，仔细想。”}%

@{時天帝釋及尊者阿難白言：善哉世尊！唯然受教。}@
%{这时，天帝释和阿难尊者都恭敬地说：“很好！世尊！我们一定会遵照您的教诲。”}%

@{佛言：此經讚歎如來真實功德，應如是持；說不思議十種弘誓，應如是持；以一大願攝一切願，應如是持；}@
%{佛陀说：“这部经赞叹了如来的真实功德，应该这样持诵；它讲述了不可思议的十种弘誓，应该这样持诵；它用一个大愿摄受一切愿，应该这样持诵；}%

@{說不思議攝受正法，應如是持；說入一乘，應如是持；說無邊諦，應如是持；說如來藏，應如是持；}@
%{它讲述了不可思议的摄受正法，应该这样持诵；它讲述了进入一乘道，应该这样持诵；它讲述了无边的真理，应该这样持诵；它讲述了如来藏，应该这样持诵；}%

@{說佛法身，應如是持；說空性義隱覆真實，應如是持；說一諦義，應如是持；說常住不動寂靜一依，應如是持；}@
%{它讲述了佛的法身，应该这样持诵；它讲述了空性的意义，它隐藏着真实，应该这样持诵；它讲述了一个真理的意义，应该这样持诵；它讲述了永恒不变、寂静安住，应该这样持诵；}%

@{說顛倒真實，應如是持；說自性清淨心煩惱隱覆，應如是持；說如來真子，應如是持；說勝鬘夫人正師子吼，應如是持。}@
%{它讲述了颠倒的真实，应该这样持诵；它讲述了本来清净的心，被烦恼所遮蔽，应该这样持诵；它讲述了如来的真子，应该这样持诵；它讲述了胜鬘夫人的正法狮子吼，应该这样持诵。”}%

@{復次憍尸迦！此經所說斷一切疑，決定了義入一乘道。}@
%{“另外，憍尸迦！这部经所讲的，能够断除一切疑惑，决定了义，进入一乘道。”}%

@{憍尸迦！今以所說勝鬘夫人師子吼經付囑於汝，乃至法住，於十方界開示演說。}@
%{“憍尸迦！现在我把这部叫做胜鬘夫人狮子吼的经，付嘱给你，直到法住世，你就在十方世界，开示演说这部经。”}%

@{天帝釋言：善哉世尊！唯然受教。}@
%{天帝释说：“很好！世尊！我们一定会遵照您的教诲。”}%

@{時天帝釋、尊者阿難，及諸大會天、人、阿修羅、健闥婆等，聞佛所說，皆大歡喜，信受奉行。}@
%{这时，天帝释、阿难尊者，以及会众中的天人、阿修罗、健闼婆等等，听到佛陀所说，都非常欢喜，信受奉行。}%

@{大寶積經卷第一百一十九}@
