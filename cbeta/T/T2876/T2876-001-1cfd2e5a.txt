@{佛說天公經}@
%{佛祖说天公经}%

@{鎘銅灌作柱。}@
%{用镉铜铸成柱子。}%

@{象□作屋椽。}@
%{用象牙做屋椽。}%

@{虎珀作屋𣖷。}@
%{用虎珀做屋瓦。}%

@{真珠作羅網。}@
%{用珍珠做罗网。}%

@{星宿作鵲頭。}@
%{用星辰做屋檐上的雀头。}%

@{日月作繚牕。}@
%{用日月做窗棂。}%

@{可得萬萬九千年。}@
%{可以住上万万年九千年。}%

@{上有一小劫。}@
%{上面还有一小劫。}%

@{下有一由旬。}@
%{下面有一由旬。}%

@{此經雖小大有威神。}@
%{这经文虽然小，但威力无穷。}%

@{亦等法華。}@
%{也等同于法华经。}%

@{亦等涅槃。}@
%{也等同于涅槃经。}%

@{亦朦雜經。}@
%{也等同于杂经。}%

@{譬如東海。}@
%{比如东海。}%

@{喻如太山。}@
%{比喻为泰山。}%

@{入海取寶。}@
%{进入海里取宝。}%

@{隨其多少亦得重來。}@
%{无论多少都可以再来取。}%

@{入山取材。}@
%{进山取材。}%

@{任材長短亦得重來。}@
%{无论长短都可以再来取。}%

@{誰能抄此經。}@
%{谁能抄写这经文。}%

@{手中螺文成。}@
%{手中就会出现螺纹。}%

@{誰能看此經。}@
%{谁能读这经文。}%

@{眼中重光生。}@
%{眼中就会重现光明。}%

@{誰能誦此經。}@
%{谁能诵读这经文。}%

@{歷劫大聰明。}@
%{历经劫难都会保持大智慧。}%

@{觀世音造金。}@
%{观世音菩萨造金身。}%

@{大罪得除。}@
%{大罪可以消除。}%

@{小罪得滅。}@
%{小罪可以消灭。}%

@{欲向刀山。}@
%{想要去刀山。}%

@{刀山摧折。}@
%{刀山会自动摧毁。}%

@{欲向劍樹。}@
%{想要去剑树。}%

@{劍樹自缺。}@
%{剑树会自动断裂。}%

@{欲向鑪炭。}@
%{想要去火炉炭堆。}%

@{鑪炭自滅。}@
%{火炉炭堆会自动熄灭。}%

@{欲向鑵湯。}@
%{想要去沸腾的油锅。}%

@{鑵湯自冷。}@
%{沸腾的油锅会自动变冷。}%

@{倉倉靈音聲朗朗。}@
%{一声声灵音响彻云霄。}%

@{講堂中央瑠璃作地。}@
%{讲堂中央用琉璃铺地。}%

@{白銀為譬。}@
%{用白银作比喻。}%

@{前登金床後仚王凡手把金經。}@
%{前面登上金床，后面有王凡手持金经。}%

@{至佛說言。}@
%{佛祖说：}%

@{憶佛如憶食。}@
%{要像记得吃饭一样记得佛祖。}%

@{念佛如念命。}@
%{要像记得呼吸一样念佛。}%

@{一日誦五遍。}@
%{一天诵读五遍。}%

@{除罪一千五百萬劫。}@
%{就能消除一千五百劫的罪孽。}%

@{皆釋成佛道。}@
%{都能解脱成佛。}%

@{佛說天公經一卷}@
%{佛祖说天公经一卷完}%
