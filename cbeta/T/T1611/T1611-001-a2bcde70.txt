@{教化品第一}@

@{我今悉歸命，一切無上尊，}@
%{我现在一心归命，一切无上尊，}%

@{為開法王藏，廣利諸群生。}@
%{为了开启法王宝藏，广大利益一切众生。}%

@{諸佛勝妙法，謗以為非法，}@
%{诸佛胜妙法，被诽谤为非法，}%

@{愚癡無智慧，迷於邪正故。}@
%{愚痴没有智慧，迷于邪正，所以才这样。}%

@{具足智慧人，善分別邪正，}@
%{具有智慧的人，善于分辨邪正，}%

@{如是作論者，不違於正法。}@
%{这样造论的人，不违背正法。}%

@{順三乘菩提，對三界煩惱，}@
%{顺应三乘菩提，面对三界烦恼，}%

@{雖是弟子造，正取邪則捨。}@
%{虽然是弟子所造，正取邪则舍弃。}%

@{善說名句義，初中後功德，}@
%{善于解释名句的意义，阐述初、中、后三部分的功德，}%

@{智者聞是義，不取於餘法。}@
%{智者听到这些道理，就不会再执着于其他法。}%

@{如我知佛意，堅住深正義，}@
%{如同我了解佛意，坚守深奥的正理，}%

@{如實修行者，取同於佛語。}@
%{如实修行的人，所取的都与佛语一致。}%

@{雖無善巧言，但有真實義，}@
%{虽然没有巧妙的言辞，但有真实的意义，}%

@{彼法應受持，如取金捨石。}@
%{这种法应该坚持，就像取金舍石一样。}%

@{妙義如真金，巧語如瓦石，}@
%{妙义如同真金，巧语如同瓦石，}%

@{依名不依義，彼人無明盲。}@
%{只依名不依义，这个人就如同无明盲人一样。}%

@{依自罪業障，謗諸佛妙法，}@
%{依着自己的罪业障，诽谤诸佛妙法，}%

@{如是諸人等，則為諸佛呵。}@
%{这样的人，就会受到诸佛的呵斥。}%

@{或有取他心，謗諸佛妙法，}@
%{或者有人取他人之心，诽谤诸佛妙法，}%

@{如是諸人等，則為諸佛呵。}@
%{这样的人，也会受到诸佛的呵斥。}%

@{為種種供養，謗諸佛妙法，}@
%{为了种种供养，诽谤诸佛妙法，}%

@{如是諸人等，則為諸佛呵。}@
%{这样的人，同样会受到诸佛的呵斥。}%

@{愚癡及我慢，樂行於小法，}@
%{愚痴和我慢，喜欢修行小法，}%

@{謗法及法師，則為諸佛呵。}@
%{诽谤法和法师，也会受到诸佛的呵斥。}%

@{外現威儀相，不識如來教，}@
%{表面上显示威仪相，却不懂如来的教诲，}%

@{謗法及法師，則為諸佛呵。}@
%{诽谤法和法师，也会受到诸佛的呵斥。}%

@{為求名聞故，起種種異說，}@
%{为了求名闻，提出种种异说，}%

@{謗法及法師，則為諸佛呵。}@
%{诽谤法和法师，也会受到诸佛的呵斥。}%

@{說乖修多羅，言是真實義，}@
%{说违背修多罗，却说这是真实义，}%

@{謗法及法師，則為諸佛呵。}@
%{诽谤法和法师，也会受到诸佛的呵斥。}%

@{求利養攝眾，誑惑無智者，}@
%{为了求利养摄取大众，欺骗愚痴的人，}%

@{謗法及法師，則為諸佛呵。}@
%{诽谤法和法师，也会受到诸佛的呵斥。}%

@{佛觀如是等，極惡罪眾生，}@
%{佛看到这样的人，知道他们是极恶的罪业众生，}%

@{慈悲心自在，為說法除苦。}@
%{慈悲心自在，为他们说法，消除他们的痛苦。}%

@{深智大慈悲，能如是利益，}@
%{深智大慈悲，能够这样利益众生，}%

@{我說不求利，為正法久住。}@
%{我说这些不是为了求利，而是为了正法久住。}%

 
 

@{究竟一乘寶性論佛寶品第二}@
%{究竟一乘宝性论佛宝品第二}%

@{佛體無前際，及無中間際，}@
%{佛体没有前际，也没有中间际，}%

@{亦復無後際，寂靜自覺知。}@
%{也同样没有后际，寂静自觉知。}%

@{既自覺知已，為欲令他知，}@
%{既然自己觉知了，为了让别人也觉知，}%

@{是故為彼說，無畏常恒道。}@
%{所以就为他们说无畏常恒之道。}%

@{佛能執持彼，智慧慈悲刀，}@
%{佛能够执持着智慧慈悲的刀，}%

@{及妙金剛杵，割截諸苦芽，}@
%{以及妙金刚杵，割截一切苦的根芽，}%

@{摧碎諸見山，覆藏顛倒意，}@
%{摧毁一切错误见解的山，覆藏颠倒的心意，}%

@{及一切稠林，故我今敬禮。}@
%{以及一切稠密的烦恼森林，所以我现在敬礼。}%

 
 

@{究竟一乘寶性論法寶品第三}@
%{究竟一乘宝性论法宝品第三}%

@{非有亦非無，亦復非有無，}@
%{非有也非无，也同样非有无，}%

@{亦非即於彼，亦復不離彼，}@
%{也非就存在于那里，也同样不离开那里，}%

@{不可得思量，非聞慧境界，}@
%{不可思量，不是用耳朵听、用智慧思维的境界，}%

@{出離言語道，內心知清涼。}@
%{超越言语道，内心知道清凉。}%

@{彼真妙法日，清淨無塵垢，}@
%{那真妙法如太阳，清净无尘垢，}%

@{大智慧光明，普照諸世間，}@
%{大智慧光明，普照一切世间，}%

@{能破諸曀障，覺觀貪瞋癡，}@
%{能够破除一切障蔽，觉察贪嗔痴，}%

@{一切煩惱等，故我今敬禮。}@
%{一切烦恼等等，所以我现在敬礼。}%

 
 

@{究竟一乘寶性論僧寶品第四}@
%{究竟一乘宝性论僧宝品第四}%

@{正覺正知者，見一切眾生，}@
%{正觉正知者，看到一切众生，}%

@{清淨無有我，寂靜真實際。}@
%{清净没有自我，寂静真如实相。}%

@{以能知於彼，自性清淨心，}@
%{因为能够认识到那个，本性清净的心，}%

@{見煩惱無實，故離諸煩惱。}@
%{看到烦恼没有实质，所以就远离一切烦恼。}%

@{無障淨智者，如實見眾生，}@
%{无障碍的清净智慧者，如实看到众生，}%

@{自性清淨性，佛法僧境界。}@
%{本性清净的特性，就是佛法僧的境界。}%

@{無閡淨智者，見諸眾生性，}@
%{无阻碍的清净智慧者，看到一切众生的本性，}%

@{遍無量境界，故我今敬禮。}@
%{遍及无量境界，所以我现在敬礼。}%

 
 

@{問曰：依何等法有此三寶？答曰：偈言：}@
%{问曰：依何等法有此三宝？答曰：偈言：}%

@{真如有雜垢，及遠離諸垢，}@
%{真如如同有杂垢，以及远离一切垢，}%

@{佛無量功德，及佛所作業。}@
%{佛无量功德，以及佛所作的一切。}%

@{如是妙境界，是諸佛所知，}@
%{这样妙的境界，是诸佛所知道的，}%

@{依此妙法身，出生於三寶。}@
%{依着这个妙法身，才生出三宝。}%

 
 

@{究竟一乘寶性論一切眾生有如來藏品第五}@
%{究竟一乘宝性论一切众生有如来藏品第五}%

@{問曰：云何得知一切眾生有如來藏？答曰：偈言：}@
%{问曰：云何得知一切众生有如来藏？答曰：偈言：}%

@{一切眾生界，不離諸佛智，}@
%{一切众生的界，不离开诸佛的智慧，}%

@{以彼淨無垢，性體不二故。}@
%{因为那个清净无垢，本性体不二的缘故。}%

@{依一切諸佛，平等法性身，}@
%{依着一切诸佛，平等的法性身，}%

@{知一切眾生，皆有如來藏。}@
%{就知道一切众生，都有如来藏。}%

 
 

@{又復略說偈言：}@
%{又复略说偈言：}%

@{體及因果業，相應及以行，}@
%{体以及因果业，相应以及行为，}%

@{時差別遍處，不變無差別。}@
%{时间差别遍及一切地方，却是不变的无差别。}%

@{彼妙義次第，第一真法性，}@
%{那个妙义的次第，就是第一真法性，}%

@{我如是略說，汝今應善知。}@
%{我这样略说，你应该好好知道。}%

 
 

@{此偈明何義？偈曰：}@
%{此偈明何义？偈曰：}%
@{自性常不染，如寶空淨水，}@
%{本性永远不被污染，就像珍贵的空净水一样，}%
@{信法及般若，三昧大悲等，}@
%{对佛法和般若智慧、禅定、大悲心等等，}%
@{淨我樂常等，彼岸功德果，}@
%{清净自我、快乐永恒等等，彼岸的功德果报，}%
@{厭苦求涅槃，欲願等諸業，}@
%{厌恶苦难追求涅槃，欲望、愿望等等所有业力，}%
@{大海器寶水，無量不可盡，}@
%{就像大海里的宝藏水，无量无边无法穷尽，}%
@{如燈明觸色，性功德如是。}@
%{如同灯光照射颜色，本性功德也是如此。}%

@{見實者說言，凡夫聖人佛，}@
%{见到真相的人说：凡夫、圣人、佛，}%
@{眾生如來藏，真如無差別，}@
%{所有众生都有如来藏，真如法性没有差别，}%
@{有不淨雜淨，及以善淨等。}@
%{有不清净、杂染清净，以及善净等等。}%

@{如是次第說，眾生菩薩佛，}@
%{这样次第来说，众生、菩萨、佛，}%
@{如空遍一切，而空無分別，}@
%{就像虚空遍布一切，而虚空没有分别，}%
@{自性無垢心，亦遍無分別。}@
%{本性无垢的心，也遍布一切没有分别。}%

@{如虛空遍至，體細塵不染，}@
%{就像虚空无处不在，身体细微尘埃不染，}%
@{佛性遍眾生，諸煩惱不染。}@
%{佛性遍布所有众生，所有烦恼不染。}%

@{如一切世間，依虛空生滅，}@
%{就像一切世间，依托虚空生灭，}%
@{依於無漏界，有諸根生滅。}@
%{依托于无漏界，有各种根生灭。}%

@{火不燒虛空，若燒無是處，}@
%{火烧不了虚空，如果烧了就没有地方了，}%
@{如是老病死，不能燒佛性。}@
%{如同老病死，也烧不了佛性。}%

@{地依於水住，水復依於風，}@
%{大地依托于水存在，水又依托于风，}%
@{風依於虛空，空不依地等。}@
%{风依托于虚空，虚空不依托于大地等等。}%

@{如是陰界根，住煩惱業中，}@
%{如同阴界根，住在烦恼业中，}%
@{諸煩惱業等，依不善思惟。}@
%{所有烦恼业等等，依托于不善思惟。}%

@{不善思惟行，住清淨心中，}@
%{不善思惟的行为，住在清净心中，}%
@{自性清淨心，不住彼諸法。}@
%{本性清净的心，不执着于那些法。}%

@{陰入界如地，煩惱業如水，}@
%{阴入界如同大地，烦恼业如同水，}%
@{不正念如風，淨心界如空。}@
%{不正念如同风，清净心界如同虚空。}%

@{依性起邪念，念起煩惱業，}@
%{依托本性生起邪念，念起烦恼业，}%
@{依因煩惱業，能起陰界入。}@
%{依托烦恼业，就能生起阴界入。}%

@{依止於五陰，界入等諸法，}@
%{依托于五阴，界入等等诸法，}%
@{有諸根生滅，如世界成壞。}@
%{有各种根生灭，如同世界成住坏空。}%

@{淨心如虛空，無因復無緣，}@
%{清净心如同虚空，没有因也没有缘，}%
@{及無和合義，亦無生住滅。}@
%{没有和合的道理，也没有生住灭。}%

@{如虛空淨心，常明無轉變，}@
%{如同虚空清净心，永远光明没有改变，}%
@{為虛妄分別，客塵煩惱染。}@
%{因为虚妄分别，客尘烦恼染污。}%

@{菩薩摩訶薩，如實知佛性，}@
%{菩萨摩诃萨，如实知道佛性，}%
@{不生亦不滅，又無老病等。}@
%{不生也不灭，也没有老病等等。}%

@{菩薩如是知，得離於生死，}@
%{菩萨这样知道，就能脱离生死，}%
@{憐愍眾生故，示現有生滅。}@
%{怜悯众生，所以示现有生灭。}%

@{佛身不變異，以得無盡法，}@
%{佛身不变异，因为获得无尽法，}%
@{眾生所歸依，以無邊際故。}@
%{众生所归依，因为无边无际。}%

@{常住不二法，以離妄分別，}@
%{常住不二法，因为远离妄分别，}%
@{恒不執不作，清淨心力故。}@
%{恒常不执着不造作，因为清净心的力量。}%

@{法身及如來，聖諦與涅槃，}@
%{法身和如来，圣谛和涅槃，}%
@{功德不相離，如光不離日。}@
%{功德彼此不分离，如同光不离太阳。}%

@{究竟一乘寶性論為何義說品第七}@
%{究竟一乘宝性论，第七品，关于如来藏的意义}%

@{問曰：餘修多羅中皆說一切空，此中何故說有真如佛性？偈言：}@
%{问：其他经书中都讲一切皆空，为什么这里却说有真如佛性？诗句说：}%
@{處處經中說，內外一切空，}@

@{有為法如雲，及如夢幻等。}@

%{执着虚妄的法，诽谤真如的真实性，}%
@{計身有神我，為令如是等，}@
%{认为自身有神我，为了让这些人，}%
@{遠離五種過，故說有佛性。}@
%{远离五种错误，所以说有佛性。}%

@{究竟一乘寶性論身轉清淨成菩提品第八}@
%{究竟一乘宝性论，第八品，关于身转清净成菩提}%

@{淨得及遠離，自他利相應，}@
%{净得成就和远离，自利利他相辅相成，}%
@{依止深快大，如彼所為義。}@

@{初說佛菩提及得菩提方便。偈言：}@
%{首先，讲佛菩提和获得佛菩提的方便。诗句说：}%
@{向說佛法身，自性清淨體，}@

@{種種雜花樹，周匝常圍遶。}@
%{各种各样的花树，环绕在四周。}%
@{如月離羅睺，日無雲翳等，}@
%{就像月亮离开罗睺，太阳没有云翳一样，}%
@{無垢功德具，顯現即彼體。}@
%{无垢的功德具足，显现出来就是佛的法身。}%
@{蜂王美味蜜，堅實淨真金，}@

%{宝藏的大果树，无垢的真金像。}%
@{轉輪聖王身，妙寶如來像，}@
%{转轮圣王的身，妙宝如来的像，}%

%{其次，讲成就自利利他。诗句说：}%
@{無漏及遍至，不滅法與恒，}@
%{无漏和遍知，不灭的法和恒常，}%
@{清涼不變異，不退寂靜處。}@

@{諸佛如來身，如虛空無相，}@

@{為諸勝智者，作六根境界，}@
%{为了那些有胜妙智慧的人，化作六根的境界，}%
@{示現微妙色，出乎妙音聲，}@
%{显现微妙的颜色，发出美妙的声音，}%
@{令嗅佛戒香，與佛妙法味，}@
%{让人闻到佛戒的香气，品尝佛妙法的味道，}%
@{使覺三昧觸，令知深妙法，}@
%{使人觉悟三昧的触觉，让人知道深妙的法，}%
@{細思惟稠林，佛離虛空相。}@
%{仔细思考，佛远离了虚空的相。}%

@{次說第一義相應。偈言：}@
%{其次，讲第一义谛相应。诗句说：}%
@{如空不思議，常恒及清涼，}@
%{就像虚空一样不可思议，永远恒常清凉，}%
@{不變與寂靜，遍離諸分別，}@
%{不变异寂静，遍离一切分别，}%
@{一切處不著，離閡麁澁觸，}@
%{一切处都不执着，远离粗糙的触觉，}%
@{亦不可見取，佛淨心無垢，}@
%{也不可以见取，佛的清净心没有污垢，}%

@{次說佛法身。偈言：}@
%{其次，讲佛的法身。诗句说：}%
@{非初非中後，不破壞不二，}@
%{不是开始，不是中间，不是结束，不破坏不二，}%
@{遠離於三界，無垢無分別。}@

%{那个人将会在无量劫中，无法获得解脱。}%
@{若人令眾生，覺信如是法，}@

@{論曰：第一教化品，如向偈中已說應知。此論廣門有十一品，中則七品，略唯一品。初釋一品，具攝此論法義體相應知。偈言：}@
%{论说：第一教化品，正如刚才偈文中所说，应该知道。此论广门有十一品，中间则有七品，简略的只有一品。首先解释第一品，全面摄持此论法义的体相，应该知道。偈文说：}%
@{佛法及眾僧，性道功德業，}@
%{佛法、僧众、性、道、功德、业，}%
@{略說此論體，七種金剛句。}@
%{简单地说，这就是此论的体，七种金刚句。}%
@{此偈明何義？言金剛者，猶如金剛難可沮壞，所證之義亦復如是，故言金剛。}@
%{这偈文阐明了什么含义呢？ 这里说“金刚”是指像金刚一样难以摧毁，所证悟的义理也是如此，所以说“金刚”。}%
@{所言句者，以此論句能與證義為根本故。}@
%{这里说“句”是指，此论的语句能够作为证悟的义理的根本。}%
@{此明何義？內身證法無言之體，以聞思智難可證得，猶如金剛。}@
%{这阐明了什么含义呢？ 内心证悟法理，无言的体性，用闻思智慧难以证得，就像金刚一样。}%
@{名字章句以能詮彼理中證智，隨順正道能作根本，故名為句。}@
%{名字和语句，能够诠释证悟的法理，顺应正道，能够作为根本，所以称为“句”。}%
@{此復何義？有二義故。}@
%{这又阐明了什么含义呢？ 因为有两种含义。}%
@{何謂二義？一難證義、二者因義，是名為義。}@
%{哪两种含义呢？ 一是难以证悟的义理，二是因义，这就是所谓的“义”。}%
@{金剛字句應如是知。}@
%{金刚字句应该如此理解。}%
@{又何謂為義？何謂為字？義者，則有七種證義。}@
%{又什么是“义”？ 什么是“字”？ “义”是指有七种证悟的义理。}%
@{何謂七義？一者佛義、二者法義、三者僧義、四者眾生義、五者菩提義、六者功德義、七者業義，是名為義。}@
%{哪七种义理呢？ 一是佛义，二是法义，三是僧义，四是众生义，五是菩提义，六是功德义，七是业义，这就是所谓的“义”。}%
@{是故經言：又第一義諦者，所謂心緣尚不能知，何況名字章句故。}@
%{所以经文中说： 又“第一义谛”是指，心缘尚且不能知，何况是名字和语句呢？}%
@{所言字者，隨以何等名字、章句、言語、風聲、能表、能說、能明、能示，此七種義是名為字。}@
%{这里说“字”是指，无论用什么样的名字、语句、语言、风声，能够表达、能够说出、能够阐明、能够显示，这七种义理，这就是所谓的“字”。}%
@{是故經言：又世諦者，謂世間中所用之事，名字章句言語所說故。}@
%{所以经文中说： 又“世谛”是指，世间中所用之事，用名字、语句、语言来说明。}%
@{又此七種金剛句義，如諸經中廣說應知。應云何知？依佛義故，如來經中告阿難言：阿難！所言如來者，非可見法，是故眼識不能得見故。}@
%{又这七种金刚句的义理，正如诸经中广说，应该知道。应该如何知道呢？ 依据佛义，如来经中告诉阿难说： 阿难！ 这里说“如来”是指，不可见法，所以眼识不能见。}%
@{依法義故，如來經中告阿難言，阿難！所言法者，非可說事，以是故非耳識所聞故。}@
%{依据法义，如来经中告诉阿难说： 阿难！ 这里说“法”是指，不可说之事，所以耳识不能闻。}%
@{依僧義故，如來經中告阿難言：阿難！所言僧者，名為無為，是故不可身心供養禮拜讚歎故。}@
%{依据僧义，如来经中告诉阿难说： 阿难！ 这里说“僧”是指，名为无为，所以不可身心供养、礼拜、赞叹。}%
@{依眾生義故，如來經中告舍利弗言：舍利弗！言眾生者，乃是諸佛如來境界。}@
%{依据众生义，如来经中告诉舍利弗说： 舍利弗！ 这里说“众生”是指，乃是诸佛如来的境界。}%
@{一切聲聞辟支佛等，以正智慧不能觀察眾生之義，何況能證毛道凡夫。}@
%{一切声闻辟支佛等等，用正智慧也不能观察众生的义理，何况能证得毛道凡夫呢？}%
@{於此義中唯信如來。}@
%{在这个义理中，只能相信如来。}%
@{是故舍利弗！隨如來信此眾生義。}@
%{所以舍利弗！ 跟着如来相信这个众生义。}%
@{舍利弗！言眾生者，即是第一義諦。}@
%{舍利弗！ 这里说“众生”是指，就是第一义谛。}%
@{舍利弗言：第一義諦者，即是眾生界。}@
%{舍利弗说： 第一义谛是指，就是众生界。}%
@{舍利弗言：眾生界者，即是如來藏。}@
%{舍利弗说： 众生界是指，就是如来藏。}%
@{舍利弗言：如來藏者，即是法身故。}@
%{舍利弗说： 如来藏是指，就是法身。}%
@{依菩提義故，經中說言：世尊言阿耨多羅三藐三菩提者，名涅槃界。}@
%{依据菩提义，经文中说： 世尊说“阿耨多罗三藐三菩提”是指，名为涅槃界。}%
@{世尊言涅槃界者，即是法身故。}@
%{世尊说“涅槃界”是指，就是法身。}%
@{依功德義故，如來經中告舍利弗言：舍利弗！如來所說法身義者，過於恒沙不離不脫不思議佛法如來智慧功德。}@
%{依据功德义，如来经中告诉舍利弗说： 舍利弗！ 如来所说法身义是指，超过恒河沙数，不离不脱，不可思议的佛法，如来智慧功德。}%
@{舍利弗！如世間燈，明色及觸不離不脫；又如摩尼寶珠，明色形相不離不脫。}@
%{舍利弗！ 如世间灯，光明、颜色、触觉不离不脱； 又如摩尼宝珠，光明、颜色、形相不离不脱。}%
@{舍利弗！法身之義亦復如是，過於恒沙不離不脫不思議佛法如來智慧功德故。}@
%{舍利弗！ 法身的义理也是如此，超过恒河沙数，不离不脱，不可思议的佛法，如来智慧功德。}%
@{依業義故，如來經中告文殊師利言：文殊師利！如來不分別不分別，無分別而自然無分別，如所作業自然行故。}@
%{依据业义，如来经中告诉文殊师利说： 文殊师利！ 如来不分别，不分别，无分别而自然无分别，如同所造的业自然而然地运行。}%
@{如是等名略說七種金剛字句。總攝此論體相應知。}@
%{像这样，简单地说，就是七种金刚字句。总摄此论的体相，应该知道。}%
@{是故偈言：}@
%{所以偈文说：}%
@{七種相次第，總持自在王，}@
%{七种相次第，总持自在王，}%

@{菩薩修多羅，序分有三句，}@
%{菩萨修多罗，序分有三句，}%

@{餘殘四句者，在菩薩如來，}@
%{余残四句者，在菩萨如来，}%

@{智慧差別分，應當如是知。}@
%{智慧差别分，应当如是知。}%

@{此偈明何義？以是七種金剛字句總攝此論，一切佛法廣說其相。如《陀羅尼自在王經》序分中三句，餘四句在彼修多羅，菩薩如來法差別分應知。}@
%{这偈明何义？以这七种金刚字句总摄此论，一切佛法广说其相。如《陀罗尼自在王经》序分中三句，余四句在彼修多罗，菩萨如来法差别分应知。}%

@{云何序分有初三句？彼修多羅序分中言：婆伽婆平等證一切法，善轉法輪、善能教化調伏無量諸弟子眾。}@
%{云何序分有初三句？彼修多罗序分中言：婆伽婆平等证一切法，善转法轮、善能教化调伏无量诸弟子众。}%

@{如是三種根本字句，次第示現佛法僧寶，說彼三寶次第生起成就應知。}@
%{如是三种根本字句，次第示现佛法僧宝，说彼三宝次第生起成就应知。}%

@{餘四句者，說隨順三寶因、成就三寶因應知。}@
%{余四句者，说随顺三宝因、成就三宝因应知。}%

@{此明何義？以諸菩薩於八地中，十自在為首，具足得一切自在。}@
%{此明何义？以诸菩萨于八地中，十自在为首，具足得一切自在。}%

@{是故菩薩坐於道場勝妙之處，於一切法中皆得自在。}@
%{是故菩萨坐于道场胜妙之处，于一切法中皆得自在。}%

@{是故經言：婆伽婆平等證一切法故。}@
%{是故经言：婆伽婆平等证一切法故。}%

@{以諸菩薩住九地時，於一切法中得為無上最大法師，善知一切諸眾生心，到一切眾生根機第一彼岸，能斷一切眾生煩惱習氣，是故菩薩成大菩提。}@
%{以诸菩萨住九地时，于一切法中得为无上最大法师，善知一切诸众生心，到一切众生根机第一彼岸，能断一切众生烦恼习气，是故菩萨成大菩提。}%

@{是故經言：善轉法輪故。}@
%{是故经言：善转法轮故。}%

@{以諸菩薩於第十地中，得住無上法王位，後能於一切佛所作業自然而行，常不休息。}@

@{是故經言：善能教化調伏無量諸弟子眾故。}@
%{是故经言：善能教化调伏无量诸弟子众故。}%

@{彼善能教化調伏無量諸弟子眾，即彼經中次後示現。}@
%{彼善能教化调伏无量诸弟子众，即彼经中次后示现。}%

@{是故經言：與大比丘眾俱，如是乃至復有無量菩薩眾俱。}@
%{是故经言：与大比丘众俱，如是乃至复有無量菩萨众俱。}%

@{如是次第，善能教化聲聞位地及佛菩提、善能調伏一切煩惱，如是畢竟有無量功德。}@
%{如是次第，善能教化声闻位地及佛菩提、善能调伏一切烦恼，如是毕竟有無量功德。}%

@{又說聲聞菩薩諸功德已，次說諸佛如來不可思議三昧境界。}@
%{又说声闻菩萨诸功德已，次说诸佛如来不可思議三昧境界。}%

@{又說諸佛如來三昧境界已，次說無垢大寶莊嚴寶殿成就。}@
%{又说诸佛如来三昧境界已，次说无垢大宝莊严宝殿成就。}%

@{又說大寶莊嚴寶殿成就已，次說大眾雲集種種供養讚歎如來，雨種種衣、雨種種華、雨種種香，如是等示現佛寶不思議事應知。}@
%{又说大宝莊严宝殿成就已，次说大众云集种种供养赞叹如来，雨种种衣、雨种种华、雨种种香，如是等示现佛宝不可思议事应知。}%

@{又復次說妙法莊嚴法座。}@
%{又复次说妙法莊严法座。}%

@{又說妙法莊嚴法座已，次說法門名字及示現功德。}@
%{又说妙法莊严法座已，次说法门名字及示现功德。}%

@{此明法寶功德差別應如。}@
%{此明法宝功德差别应如。}%

@{又復次說諸菩薩摩訶薩迭共三昧行境界，示現種種功德。}@
%{又复次说诸菩萨摩诃萨迭共三昧行境界，示现种种功德。}%

@{此明僧寶功德差別應知。}@
%{此明僧宝功德差别应知。}%

@{又復次說如來放大光明，授諸菩薩摩訶薩太子法王位職。}@
%{又复次说如来放大光明，授诸菩萨摩诃萨太子法王位职。}%

@{又復次說與大無畏不怯弱辯才，又復讚歎諸佛如來第一功德，又復次說最上第一大乘之法，示現如實修行彼大乘故，於法中證果。}@
%{又复次说与大无畏不怯弱辩才，又复赞叹诸佛如来第一功德，又复次说最上第一大乘之法，示现如实修行彼大乘故，于法中证果。}%

@{即彼三寶無上功德次第差別，序分中義大都已竟，應如是知。}@
%{即彼三宝无上功德次第差别，序分中义大都已竟，应如是知。}%

@{已說自在王菩薩修多羅序分中三寶，次說佛性義。}@
%{已说自在王菩萨修多罗序分中三宝，次说佛性义。}%

@{有六十種法，清淨彼功德。}@
%{有六十种法，清净彼功德。}%

@{何以故？以有彼清淨無量功德性。}@
%{何以故？以有彼清净无量功德性。}%

@{為清淨彼性，修六十種法。}@
%{为清净彼性，修六十种法。}%

@{為此義故，《十地經》中數數說金以為譬喻，為清淨彼佛性義故。}@
%{为此义故，《十地经》中数数说金以为譬喻，为清净彼佛性义故。}%

@{又復即於此《陀羅尼自在王經》中說如來業已，次說不清淨大毘琉璃摩尼寶喻。}@
%{又复即于此《陀罗尼自在王经》中说如来业已，次说不清净大毘琉璃摩尼宝喻。}%

@{是故經言：善男子！譬如善巧摩尼寶師，善知清淨大摩尼寶，向大摩尼寶性山中取未清淨諸摩尼寶。}@
%{是故经言：善男子！譬如善巧摩尼宝师，善知清净大摩尼宝，向大摩尼宝性山中取未清净诸摩尼宝。}%

@{既取彼寶，以嚴灰洗。}@
%{既取彼宝，以严灰洗。}%

@{嚴灰洗已，然後復持黑頭髮衣以用揩磨。}@
%{严灰洗已，然后复持黑头发衣以用揩磨。}%

@{不以為足，勤未休息，次以辛味飲食汁洗。}@
%{不以为足，勤未休息，次以辛味饮食汁洗。}%

@{食汁洗已，然後復持衣纏裹木以用揩磨。}@
%{食汁洗已，然后复持衣缠裹木以用揩磨。}%

@{不以為足，勤未休息，次後復以大藥汁洗。}@
%{不以为足，勤未休息，次后复以大药汁洗。}%

@{藥汁洗已，次後復更持細軟衣以用揩磨。}@
%{药汁洗已，次后复更持细软衣以用揩磨。}%

@{以細軟衣用揩磨已，然後遠離銅鐵等鑛毘琉璃垢，方得說言大琉璃寶。}@
%{以细软衣用揩磨已，然后远离铜铁等鑛毘琉璃垢，方得说言大琉璃宝。}%

@{善男子！諸佛如來亦復如是，善知不淨諸眾生性，知已乃為說無常苦無我不淨，為驚怖彼樂世眾生，令厭世間入聲聞法中。}@
%{善男子！诸佛如来亦复如是，善知不净诸众生性，知已乃为说无常苦无我不净，为惊怖彼乐世众生，令厌世间入声闻法中。}%

@{而佛如來不以為足，勤未休息，次為說空無相無願，令彼眾生少解如來所說法輪。}@
%{而佛如来不以为足，勤未休息，次为说空无相无愿，令彼众生少解如来所说法轮。}%

@{而佛如來不以為足，勤未休息，次復為說不退法輪，次說清淨波羅蜜行，謂不見三事，令眾生入如來境界。}@
%{而佛如来不以为足，勤未休息，次复为说不退法轮，次说清净波罗蜜行，谓不见三事，令众生入如来境界。}%

@{如是依種種因、依種種性，入佛法中。}@
%{如是依种种因、依种种性，入佛法中。}%

@{入法中已，故名無上最大福田。}@
%{入法中已，故名无上最大福田。}%

@{又復依此自性清淨如來性故，經中偈言：}@
%{又复依此自性清净如来性故，经中偈言：}%
@{譬如石鑛中，真金不可見，}@
%{就像石头矿里，真金你看不见，}%

@{能清淨者見，見佛亦如是。}@
%{只有心清净的人才能看到，看佛也是这样。}%

@{向說佛性有六十種淨業功德。何謂六十？所謂四種菩薩莊嚴、八種菩薩光明、十六種菩薩摩訶薩大悲、三十二種諸菩薩業。}@
%{以前说过佛性有六十种清净功德，这六十种是什么呢？就是四种菩萨庄严，八种菩萨光明，十六种菩萨摩诃萨大悲，三十二种诸菩萨的功业。}%

@{已說佛性義，次說佛菩提，有十六種無上菩提大慈悲心。}@
%{已经说了佛性的含义，接下来说佛菩提，佛菩提有十六种无上菩提大慈悲心。}%

@{已說佛菩提，次說諸佛如來功德，所謂十力、四無所畏、十八不共法。}@
%{已经说了佛菩提，接下来说诸佛如来的功德，就是十力、四无所畏、十八不共法。}%

@{已說功德，次說如來三十二種無上大業。如是七種金剛句義，彼修多羅廣說體相，如是應知。問曰：此七種句有何次第？答曰：偈言：}@
%{已经说了功德，接下来说如来三十二种无上大业。这七种金刚句的含义，在那些经文中广说它的体相，你要这样知道。有人问：这七种句有什么顺序呢？回答说：偈文：}%

@{從佛次有法，次法復有僧，}@
%{从佛开始，其次有法，其次法又有僧，}%

@{僧次無礙性，從性次有智。}@
%{僧之后是无碍性，从无碍性之后是智慧。}%

@{十力等功德，為一切眾生，}@
%{十力等功德，是为了所有众生，}%

@{而作利益業，有如是次第。}@
%{而作利益事业，有这样的顺序。}%

@{已說一品具攝此論法義體相，次說七品具攝此論法義體相，解釋偈義應知。歸敬三寶者，此明何義？所有如來教化眾生，彼諸眾生歸依於佛，尊敬如來歸依於法，尊敬如來歸依於僧。依於三寶，說十二偈。初明佛寶，故說四偈。}@
%{已经说了第一品包含了这部论典的法义体相，接下来说七品包含了这部论典的法义体相，解释偈文的意思，你要知道。归敬三宝者，这说明什么意思呢？所有如来教化众生，那些众生归依于佛，尊敬如来归依于法，尊敬如来归依于僧。依于三宝，说十二偈。首先说明佛宝，所以说四偈。}%
