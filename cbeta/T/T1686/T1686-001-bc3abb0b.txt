%{嬉戏玩耍，享受快乐。如果在炎热的时候，}%

@{而作陰涼施，當感優曇鉢，}@
%{布施阴凉，就能感应到优昙钵，}%

@{尼俱菩提樹，復後得生天，}@
%{尼俱菩提树，将来能够生到天界，}%

@{恒受五欲樂。井泉及池沼，}@
%{长久享受五欲的快乐。井泉和池沼，}%

@{修飾令嚴淨，濟彼渴乏人，}@
%{修饰得庄严、干净，帮助那些口渴的人，}%

@{普皆令充足，後報得生天，}@
%{让他们都得到满足，将来能够生到天界，}%

@{或生於梵世，種種受快樂，}@
%{或者生到梵天世界，享受各种快乐，}%

@{復證寂滅果。若以鉢多羅，}@
%{并且证得涅槃寂灭的果位。如果用钵多罗，}%

@{布施於三寶，當生一切處，}@
%{布施给三宝，就能生到一切地方，}%

@{富貴而快樂，於彼世間中，}@
%{富贵快乐，在那个世间，}%

@{尊高德最上，復感諸眾生，}@
%{地位尊贵，德行高尚，并且感应到所有众生，}%

@{恒時而供養。智者若施刀，}@
%{长久供养他。智者如果布施刀，}%

@{當得生天上，智慧極聰利，}@
%{就能生到天界，智慧极其聪明，}%

@{永不值刀兵。若人施其針，}@
%{永远不会遇到刀兵。如果有人布施针，}%

@{智慧恒猛利，能斷諸煩惱，}@
%{智慧永远锐利，能够断除一切烦恼，}%

@{證彼寂靜道。佛說若有人，}@
%{证得寂静之道。佛陀说，如果有人，}%

@{塑畫於佛像，生於天界中，}@
%{塑造、绘画佛像，生到天界，}%

@{身體真金色，清淨光如日，}@
%{身体是真金色，清净的光芒如同太阳，}%

@{而諸天人眾，天男及天女，}@
%{天人大众，天男和天女，}%

@{恒時而歸命。若彼智慧人，}@
%{长久向他顶礼膜拜。如果那个智慧的人，}%

@{善能說法施，天上人間生，}@
%{善于说法，生到天界和人间，}%

@{智德力具足，恒受於快樂，}@
%{智慧、德行、力量都具足，长久享受快乐，}%

@{永離憂悲苦，於其後有身，}@
%{永远远离忧愁、悲伤、痛苦，在未来的生命中，}%

@{得證寂滅道。若彼諸有情，}@
%{能够证得涅槃寂灭之道。如果那些有情众生，}%

@{書寫妙法寶，當得宿命智，}@
%{书写妙法宝，就能获得宿命智，}%

@{富貴恒安樂，而滅一切罪，}@
%{富贵，长久安乐，并且消除一切罪业，}%

@{經彼俱胝劫，不墮於地獄，}@
%{经过无数劫，不会堕落到地狱，}%

@{及與鬼畜生。若彼有智者，}@
%{以及鬼畜道。如果有智者，}%

@{掃灑結壇場，旋繞散香花，}@
%{扫除、布置坛场，绕着坛场散布香花，}%

@{恒時而供養，後生於人間，}@
%{长久供养，将来能够生到人间，}%

@{及生於天上，恒受於富貴，}@
%{以及生到天界，长久享受富贵，}%

@{復證寂滅故。若以長明燈，}@
%{并且证得涅槃寂灭。如果用长明灯，}%

@{供養佛賢聖，生彼人天中，}@
%{供养佛陀和圣贤，生到人天之中，}%

@{三眼常清淨，慧眼與天眼，}@
%{三眼常清净，智慧眼、天眼，}%

@{一切悉通達。智者若施財，}@
%{一切都能通达。智者如果布施财物，}%

@{供養有德眾，沙門婆羅門，}@
%{供养有德行的沙门、婆罗门，}%

%{长久享用。另外，行布施的人，}%

%{布施给罪人，功德利益逐渐增多，}%

@{獲福成千倍。若施持戒人，}@

@{福獲千百倍，供養無心人，}@
