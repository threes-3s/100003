%{阿毘達磨識身足論卷第四}%

@{提婆設摩阿羅漢造}@
%{提婆設摩阿羅漢造}%

@{過去等因與善等，隨增幾緣幾因等，}@
%{过去等因与善等，随着增加一些缘一些因等，}%

@{十十五心非因等，結縛隨眠等相應。}@
%{十十五心非因等，结缚随眠等相应。}%

@{有六識身，謂眼識耳鼻舌身意識。}@
%{有六识身，谓眼识耳鼻舌身意识。}%

@{如是六識身，或過去或未來或現在。}@
%{如是六识身，或过去或未来或现在。}%

@{過去眼識，頗有過去為因，非未來為因、非現在為因耶？}@
%{过去眼识，是否有过去为因，而非未来为因、非现在为因呢？}%

@{頗有未來為因，非過去為因、非現在為因耶？}@
%{是否有未来为因，而非过去为因、非现在为因呢？}%

@{頗有現在為因，非過去為因、非未來為因耶？}@
%{是否有现在为因，而非过去为因、非未来为因呢？}%

@{頗有過去現在為因，非未來為因耶？}@
%{是否有过去现在为因，而非未来为因呢？}%

@{頗有未來現在為因，非過去為因耶？}@
%{是否有未来现在为因，而非过去为因呢？}%

@{頗有過去未來為因，非現在為因耶？}@
%{是否有过去未来为因，而非现在为因呢？}%

@{頗有過去未來現在為因耶？}@
%{是否有过去未来现在为因呢？}%

@{如過去眼識，未來現在眼識亦爾。}@
%{如过去眼识，未来现在眼识亦然。}%

@{如眼識，耳鼻舌身意識亦爾。}@
%{如眼识，耳鼻舌身意识亦然。}%

@{過去眼識，一切皆用過去為因，所餘諸句皆不可得。}@
%{过去眼识，一切皆用过去为因，其余诸句皆不可得。}%

@{未來眼識，或用過去、未來為因，非現在為因。}@
%{未来眼识，或用过去、未来为因，非现在为因。}%

@{何等未來因？謂此俱有相應等法。}@
%{何等未来因？谓此俱有相应等法。}%

@{何等過去因？謂過去法與此眼識或為同類或異熟等，非現在法與此眼識或為同類或異熟等。}@
%{何等过去因？谓过去法与此眼识或为同类或异熟等，非现在法与此眼识或为同类或异熟等。}%

@{或有過去未來現在為因。}@
%{或有过去未来现在为因。}%

@{何等未來因？謂此俱有相應等法。}@
%{何等未来因？谓此俱有相应等法。}%

@{何等過去現在因？謂過去現在法與此眼識或為同類或異熟等。}@
%{何等过去现在因？谓过去现在法与此眼识或为同类或异熟等。}%

@{現在眼識，一切皆用過去現在為因，所餘諸句皆不可得。}@
%{现在眼识，一切皆用过去现在为因，其余诸句皆不可得。}%

@{何等現在因？謂此俱有相應諸法。}@
%{何等现在因？谓此俱有相应诸法。}%

@{何等過去因？謂過去法與此眼識或為同類或異熟等。}@
%{何等过去因？谓过去法与此眼识或为同类或异熟等。}%

@{如眼識，耳鼻舌身識亦爾。}@
%{如眼识，耳鼻舌身识亦然。}%

@{過去意識，一切皆用過去為因，所餘諸句皆不可得。}@
%{过去意识，一切皆用过去为因，其余诸句皆不可得。}%

@{未來意識，或有未來為因，非過去現在為因，謂未證入正性離生補特伽羅，未來最初無漏意識。}@
%{未来意识，或有未来为因，非过去现在为因，谓未证入正性离生补特伽罗，未来最初无漏意识。}%

@{何等未來因？謂此俱有相應諸法。}@
%{何等未来因？谓此俱有相应诸法。}%

@{或有未來現在為因，非過去為因，謂苦法智忍現在前時，所有未來無漏意識。}@
%{或有未来现在为因，非过去为因，谓苦法智忍现在前时，所有未来无漏意识。}%

@{何等未來因？謂此俱有相應諸法。}@
%{何等未来因？谓此俱有相应诸法。}%

@{何等現在因？謂苦法智忍及彼俱有相應等法。}@
%{何等现在因？谓苦法智忍及彼俱有相应等法。}%

@{或有過去未來為因非現在為因。}@
%{或有过去未来为因非现在为因。}%

@{何等未來因？謂此俱有相應等法。}@
%{何等未来因？谓此俱有相应等法。}%

@{何等過去因？謂過去法與此意識或為同類或異熟等，非現在法與此意識或為同類或異熟等。}@
%{何等过去因？谓过去法与意识或为同类或异熟等，非现在法与意识或为同类或异熟等。}%

@{或有過去未來現在為因。}@
%{或有过去未来现在为因。}%

@{何等未來因？謂此俱有相應等法。}@
%{何等未来因？谓此俱有相应等法。}%

@{何等過去現在因？謂過去現在法與此意識或為同類或異熟等。}@
%{何等过去现在因？谓过去现在法与意识或为同类或异熟等。}%

@{現在意識，或有現在為因，非過去為因、非未來為因，謂苦法智忍現在前時，與彼俱有相應意識。}@
%{现在意识，或有现在为因，非过去为因、非未来为因，谓苦法智忍现在前时，与彼俱有相应意识。}%

@{何等現在因？謂此俱有相應諸法。}@
%{何等现在因？谓此俱有相应诸法。}%

@{或有過去現在為因，非未來為因。}@
%{或有过去现在为因，非未来为因。}%

@{何等現在因？謂此俱有相應諸法。}@
%{何等现在因？谓此俱有相应诸法。}%

@{何等過去因？謂過去法與此意識或為同類或異熟等。}@
%{何等过去因？谓过去法与意识或为同类或异熟等。}%

@{有六識身，謂眼識耳鼻舌身意識。}@
%{有六识身，谓眼识耳鼻舌身意识。}%

@{如是六識身，或善、或不善、或無記。}@
%{如是六识身，或善、或不善、或无记。}%

@{善眼識，頗有善為因，非不善為因、非無記為因耶？}@
%{善眼识，是否有善为因，而非不善为因、非无记为因呢？}%

@{頗有不善為因，非善為因、非無記為因耶。}@
%{是否有不善为因，非善为因、非无记为因呢？}%

@{頗有無記為因？非善為因、非不善為因耶？}@
%{是否有无记为因？非善为因、非不善为因呢？}%

@{頗有善無記為因，非不善為因耶。}@
%{是否有善无记为因，非不善为因呢？}%

@{頗有不善無記為因，非善為因耶？}@
%{是否有不善无记为因，非善为因呢？}%

@{頗有善不善為因，非無記為因耶？}@
%{是否有善不善为因，非无记为因呢？}%

@{頗有善不善無記為因耶？}@
%{是否有善不善无记为因呢？}%

@{如善眼識，不善無記眼識亦爾。}@
%{如善眼识，不善无记眼识亦然。}%

@{如眼識，耳鼻舌身意識亦爾。}@
%{如眼识，耳鼻舌身意识亦然。}%

@{善眼識，一切皆用善為因，所餘諸句皆不可得。}@
%{善眼识，一切皆用善为因，其余诸句皆不可得。}%

@{不善眼識，一切皆用不善無記為因，所餘諸句皆不可得。}@
%{不善眼识，一切皆用不善无记为因，其余诸句皆不可得。}%

@{無記眼識，或有無記為因、非善為因、非不善為因，謂威儀路工巧處眼識，及梵世所繫染污眼識。}@
%{无记眼识，或有无记为因、非善为因、非不善为因，谓威仪路工巧处眼识，及梵世所系染污眼识。}%

@{或有善無記為因、非不善為因，謂善異熟生眼識。}@
%{或有善无记为因、非不善为因，谓善异熟生眼识。}%

@{何等無記因？謂此俱有相應等法。}@
%{何等无记因？谓此俱有相应等法。}%

@{何等善因？謂彼善法能感於此眼識異熟。}@
%{何等善因？谓彼善法能感于此眼识异熟。}%

@{或有不善無記為因，非善為因，謂不善異熟生眼識。}@
%{或有不善无记为因，非善为因，谓不善异熟生眼识。}%

@{何等無記因？謂此俱有相應等法。}@
%{何等无记因？谓此俱有相应等法。}%

@{何等不善因？謂彼不善法能感於此眼識異熟。}@
%{何等不善因？谓彼不善法能感于此眼识异熟。}%

@{如眼識，耳鼻舌身識亦爾。}@
%{如眼识，耳鼻舌身识亦然。}%

@{此中差別者，鼻識舌識不應說有梵世所繫。}@
%{此中差别者，鼻识舌识不应说有梵世所系。}%

@{善意識，一切皆用善為因，所餘諸句皆不可得。}@
%{善意识，一切皆用善为因，其余诸句皆不可得。}%

@{不善意識，一切皆用不善無記為因，所餘諸句皆不可得。}@
%{不善意识，一切皆用不善无记为因，其余诸句皆不可得。}%

@{無記意識，或有無記為因、非善為因、非不善為因，謂威儀路工巧處意識，及色無色所繫染污意識。}@
%{无记意识，或有无记为因、非善为因、非不善为因，谓威仪路工巧处意识，及色无色所系染污意识。}%

@{或有善無記為因、非不善為因，謂善異熟生意識。}@
%{或有善无记为因、非不善为因，谓善异熟生意识。}%

@{何等無記因？謂此俱有相應等法。}@
%{何等无记因？谓此俱有相应等法。}%

@{何等善因？謂彼善法能感於此意識異熟。}@
%{何等善因？谓彼善法能感于此意识异熟。}%

@{或有不善無記為因、非善為因，謂欲所繫有身見、邊執見相應意識。}@
%{或有不善无记为因、非善为因，谓欲所系有身见、边执见相应意识。}%

@{何等無記因？謂此俱有相應等法。}@
%{何等无记因？谓此俱有相应等法。}%

@{何等不善因？謂欲所繫見苦所斷八隨眠等，及見集所斷遍行隨眠等。}@
%{何等不善因？谓欲所系见苦所断八随眠等，及见集所断遍行随眠等。}%
@{有六識身，謂眼識耳鼻舌身意識。如是六識身，或善或不善、或有覆無記或無覆無記。於善眼識，有幾隨眠之所隨增？於不善有覆無記無覆無記眼識，有幾隨眠之所隨增？如眼識，耳鼻舌身意識亦爾。}@
%{我们有六种意识，分别是眼识、耳识、鼻识、舌识、身识和意识。这六种意识，可以是善的，也可以是不善的，可以是有覆无记的，也可以是无覆无记的。对于善的眼识，有多少随眠会随之增长？对于不善的、有覆无记的、无覆无记的眼识，又有多少随眠会随之增长？眼识是这样，耳识、鼻识、舌识、身识和意识也是这样。}%

@{於善眼識，有欲纏色纏遍行隨眠及修所斷隨眠之所隨增。於不善眼識，有欲纏遍行隨眠及修所斷隨眠之所隨增。於有覆無記眼識，有色纏遍行隨眠及修所斷隨眠之所隨增。於無覆無記眼識，有欲纏色纏遍行隨眠及修所斷隨眠之所隨增。如眼識，耳鼻舌身識亦爾。此中差別者，鼻識舌識不應說有有覆無記。於善意識，有三界遍行隨眠及修所斷隨眠之所隨增。於不善意識，有欲纏一切隨眠之所隨增。於有覆無記意識，有色無色纏一切隨眠及欲纏見苦所斷一切隨眠見集所斷遍行隨眠之所隨增。於無覆無記意識，有三界遍行隨眠及修所斷隨眠之所隨增。}@
%{对于善的眼识，会有欲缠、色缠、遍行随眠和修所断随眠随之增长。对于不善的眼识，会有欲缠、遍行随眠和修所断随眠随之增长。对于有覆无记的眼识，会有色缠、遍行随眠和修所断随眠随之增长。对于无覆无记的眼识，会有欲缠、色缠、遍行随眠和修所断随眠随之增长。眼识是这样，耳识、鼻识、舌识、身识也是这样。这里需要注意的是，鼻识和舌识不应该说是有覆无记。对于善的意识，会有三界遍行随眠和修所断随眠随之增长。对于不善的意识，会有欲缠一切随眠随之增长。对于有覆无记的意识，会有色缠、无色缠一切随眠，以及欲缠、见苦所断一切随眠、见集所断遍行随眠随之增长。对于无覆无记的意识，会有三界遍行随眠和修所断随眠随之增长。}%

@{有六識身，謂眼識耳鼻舌身意識。如是六識身，或善或不善、或有覆無記或無覆無記。於善眼識，有幾隨眠當言為因、當言為緣？有幾隨眠，當言為緣而不為因？於不善有覆無記無覆無記眼識，有幾隨眠當言為因、當言為緣？有幾隨眠，當言為緣而不為因？如眼識，耳鼻舌身意識亦爾。}@
%{我们有六种意识，分别是眼识、耳识、鼻识、舌识、身识和意识。这六种意识，可以是善的，也可以是不善的，可以是有覆无记的，也可以是无覆无记的。对于善的眼识，有多少随眠可以说成是因，可以说成是缘？有多少随眠可以说成是缘，但不是因？对于不善的、有覆无记的、无覆无记的眼识，又有多少随眠可以说成是因，可以说成是缘？有多少随眠可以说成是缘，但不是因？眼识是这样，耳识、鼻识、舌识、身识和意识也是这样。}%
