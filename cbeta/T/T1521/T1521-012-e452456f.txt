@{十住毘婆沙論卷第十二}@
%{十住毘婆沙论卷第十二}%

@{聖者龍樹造}@
%{圣者龙树造}%

@{後秦龜茲國三藏鳩摩羅什譯}@
%{后秦龟兹国三藏鸠摩罗什译}%

@{讚偈品第二十四}@
%{赞偈品第二十四}%

@{已如是解四十不共法竟，應取是四十不共法相念佛，又應以諸偈讚佛，如現在前對面共語，如是則成念佛三昧。}@
%{已经这样理解了四十不共法，应该取这四十不共法相念佛，又应该用这些偈赞佛，就像现在面前对面说话一样，这样就能成就念佛三昧。}%

@{如偈說：}@
%{就像偈说：}%

@{聖主大精進，四十獨有法，}@
%{圣主大精进，四十独有法，}%

@{我今於佛前，敬心以稱讚：}@
%{我今在佛前，敬心称赞：}%

@{如意及飛行。其力無邊限。}@
%{如意及飞行，其力无边限。}%

@{於聖如意中，無有與等者。}@
%{在圣如意中，没有与等者。}%

@{聲聞中自在。他心智無量。}@
%{声闻中自在，他心智无量。}%

@{善能調伏心，隨意而應適。}@
%{善能调伏心，随意而应适。}%

@{其念如大海，湛然在安隱。}@
%{其念如大海，湛然在安隐。}%

@{世間無有法，而能擾亂者。}@
%{世间没有法，而能扰乱者。}%

@{諸佛所稱歎，金剛三昧寶，}@
%{诸佛所称叹，金刚三昧宝，}%

@{得之在胸中；如賢懷直心。}@
%{得之在胸中；如贤怀直心。}%

@{善知不定法。四無色定事，}@
%{善知不定法，四无色定事，}%

@{微細難分別，盡知無有餘。}@
%{微细难分别，尽知无有余。}%

@{眾生若已滅、今滅及當滅，}@
%{众生若已灭、今灭及当灭，}%

@{唯獨有世尊，智慧能通達。}@
%{唯独有世尊，智慧能通达。}%

@{善知不相應、非色法中事；}@
%{善知不相應，非色法中事；}%

@{一切諸世間，悉皆不能知。}@
%{一切诸世间，悉皆不能知。}%

@{世尊大威力，功德不可量。}@
%{世尊大威力，功德不可量。}%

@{智慧無邊際，皆無與等者。}@
%{智慧无边际，皆无与等者。}%

@{於四問答中，超絕無倫匹，}@
%{于四问答中，超绝无伦匹，}%

@{眾生諸問難，一切皆易答。}@
%{众生诸问难，一切皆易答。}%

@{若諸世間中，欲有害佛者，}@
%{若诸世间中，欲有害佛者，}%

@{是事皆不成，以成不殺法。}@
%{是事皆不成，以成不杀法。}%

@{若於三時中，諸有所說者，}@
%{若于三时中，诸有所说者，}%

@{言必不虛設，常有大果報。}@
%{言必不虚设，常有大果报。}%

@{凡有所說法，無非是希有。}@
%{凡有所说法，无非是希有。}%

@{義趣尚不謬，何況於言辭。}@
%{义趣尚不谬，何况于言辞。}%

@{於三聖弟子，上中下差別，}@
%{于三圣弟子，上中下差别，}%

@{四雙八輩等，第一大導師。}@
%{四双八辈等，第一大导师。}%

@{身口意業命，畢竟常清淨，}@
%{身口意业命，毕竟常清净，}%

@{是故於此中，不復須防護。}@
%{是故于此中，不复须防护。}%

@{自說一切智，心無有疑畏；}@
%{自说一切智，心无有疑畏；}%

@{若人來難我，恐有所不知。}@
%{若人来难我，恐有所不知。}%

@{自說漏盡相，盡到無漏邊，}@
%{自说漏尽相，尽到无漏边，}%

@{心無有疑畏，餘漏有不盡。}@
%{心无有疑畏，余漏有不尽。}%

@{自說障礙法，於中無疑難；}@
%{自说障碍法，于中无疑难；}%

@{雖有用此法，不能為障礙。}@
%{虽有用此法，不能为障碍。}%

@{所說八聖道，心無有疑畏；}@
%{所说八圣道，心无有疑畏；}%

@{有言是八道，不能至解脫。}@
%{有言是八道，不能至解脱。}%

@{如實知是因、是果及與非；}@
%{如实知是因、是果及与非；}%

@{故號一切智，名聞流無量。}@
%{故号一切智，名闻流无量。}%

@{三世所有業，是諸業定報，}@
%{三世所有业，是诸业定报，}%

@{及非定果報，種種皆悉知。}@
%{及非定果报，种种皆悉知。}%

@{諸禪三昧中，麁細深淺事，}@
%{诸禅三昧中，粗细深浅事，}%

@{皆悉能了知，禪中無等者。}@
%{皆悉能了知，禅中无等者。}%

@{先知眾生根，上中下差別。}@
%{先知众生根，上中下差别。}%

@{種種樂及性，隨宜而說法。}@
%{种种乐及性，随宜而说法。}%

@{行道得諸利，兼以化導人，}@
%{行道得诸利，兼以化导人，}%

@{是以弟子眾，如實得善利。}@
%{是以弟子众，如实得善利。}%

@{宿命知無量、天眼見無邊；}@
%{宿命知无量、天眼见无边；}%

@{一切人天中，無能知其限。}@
%{一切人天中，无能知其限。}%

@{住金剛三昧，滅煩惱及氣，}@
%{住金刚三昧，灭烦恼及气，}%

@{又知人漏盡，故名漏盡力。}@
%{又知人漏尽，故名漏尽力。}%

@{煩惱諸禪障、一切法障礙，}@
%{烦恼诸禅障、一切法障碍，}%

@{三礙得解脫，號無礙解脫。}@
%{三碍得解脱，号无碍解脱。}%

@{四十不共法，功德不可量，}@
%{四十不共法，功德不可量，}%

@{無能廣說者，我已略說竟。}@
%{无能广说者，我已略说竟。}%

@{世尊若一劫，稱說此佛法，}@
%{世尊若一劫，称说此佛法，}%

@{猶尚不可盡，況我無此智。}@
%{犹尚不可尽，况我无此智。}%

@{世尊大慈蔭，無量業善集，}@
%{世尊大慈荫，无量业善集，}%

@{四功德處故，得佛無量法。}@
%{四功德处故，得佛无量法。}%

@{世尊所稱說，四功德勝處，}@
%{世尊所称说，四功德胜处，}%

@{我今還以此，稱讚於如來。}@
%{我今还以此，称赞于如来。}%

@{三十二相具，相有百福德，}@
%{三十二相具，相有百福德，}%

@{八十種妙好，三界誰能有。}@
%{八十种妙好，三界谁能有。}%

@{三千大千界，眾生所有福，}@
%{三千大千界，众生所有福，}%

@{果報為百倍，相有如是德。}@
%{果报为百倍，相有如是德。}%

@{如此諸福德，并及其果報，}@
%{如此诸福德，并及其果报，}%

@{復以為百倍，成一白毫相。}@
%{复以为百倍，成一白毫相。}%

@{三十相一一，福德及果報，}@
%{三十相一一，福德及果报，}%

@{復以為千倍，成一肉髻相。}@
%{复以为千倍，成一肉髻相。}%

@{世尊諸功德，不可得度量，}@
%{世尊诸功德，不可得度量，}%

@{如人以尺寸，量空不可盡。}@
%{如人以尺寸，量空不可尽。}%

@{從初發大心，為度眾生故，}@
%{从初发大心，为度众生故，}%

@{堅心無量劫，是故成佛道。}@
%{坚心无量劫，是故成佛道。}%

@{精勤欲成滿，如此之大願，}@
%{精勤欲成满，如此之大愿，}%

@{無量劫數中，行諸難苦行。}@
%{无量劫数中，行诸难苦行。}%

@{如諸往古佛，說四功德處，}@
%{如诸往古佛，说四功德处，}%

@{無量劫乃成，今得安住中。}@
%{无量劫乃成，今得安住中。}%

@{本為護實諦，捨身及親愛，}@
%{本为护实谛，舍身及亲爱，}%

@{財寶諸富樂，是故得具足。}@
%{财宝诸富乐，是故得具足。}%

@{無量劫數中，見聞覺知法，}@
%{无量劫数中，见闻觉知法，}%

@{每先善思惟，而後為人說。}@
%{每先善思惟，而后为人说。}%

@{若於不見等，及於中有疑，}@
%{若于不见等，及于中有疑，}%

@{而能如實說，所益無有量。}@
%{而能如实说，所益无有量。}%

@{不說他匿事，嫌譏而拒逆，}@
%{不说他匿事，嫌讥而拒逆，}%

@{念常在安慧，順化令安隱。}@
%{念常在安慧，顺化令安隐。}%

@{第一真妙諦，涅槃實為最，}@
%{第一真妙谛，涅槃实为最，}%

@{餘者皆虛妄，世尊德具足。}@
%{余者皆虚妄，世尊德具足。}%

@{飲食臥具等，堂閣妙樓觀，}@
%{饮食卧具等，堂阁妙楼观，}%

@{名好象馬車，端嚴諸婇女；}@
%{名好象马车，端严诸婇女；}%

@{金銀珍寶等，聚落諸城邑，}@
%{金银珍宝等，聚落诸城邑，}%

@{國土及榮位，并以四天下；}@
%{国土及荣位，并以四天下；}%

@{愛子所親婦，支節及頭目，}@
%{爱子所亲妇，支节及头目，}%

@{割肉出骨髓，及以舉身施。}@
%{割肉出骨髓，及以举身施。}%

@{憐愍諸眾生，悉施無所惜，}@
%{怜愍诸众生，悉施无所惜，}%

@{為求出生死，不以求自樂。}@
%{为求出生死，不以求自乐。}%

@{虛空諸星宿，地上所有沙，}@
%{虚空诸星宿，地上所有沙，}%

@{世尊菩薩時，布施數過是。}@
%{世尊菩萨时，布施数过是。}%

@{終不以非法，求財而布施。}@
%{终不以非法，求财而布施。}%

@{無有不知施。無侵惱人施。}@
%{无有不知施。无侵恼人施。}%

@{不貪惜好物，而以惡者施。}@
%{不贪惜好物，而以恶者施。}%

@{無諂曲心施。無惜而強施。}@
%{无谄曲心施。无惜而强施。}%

@{無恚無疑心、無邪無輕笑、}@
%{无恚无疑心、无邪无轻笑、}%

@{無厭無不信、䫌面等布施。}@
%{无厌无不信、䫌面等布施。}%

@{無有分別心，此應彼不應；}@
%{无有分别心，此应彼不应；}%

@{但以悲心故，平等而行施。}@
%{但以悲心故，平等而行施。}%

@{不輕於眾生，以為非福田；}@
%{不轻于众生，以为非福田；}%

@{見聖心恭敬；破戒者憐愍。}@
%{见圣心恭敬；破戒者怜愍。}%

@{不自高其身、卑下於他人。}@
%{不自高其身、卑下于他人。}%

@{亦不為稱讚。不求報等施。}@
%{亦不为称赞。不求报等施。}%

@{無悔無憂愁、無惡賤心施。}@
%{无悔无忧愁、无恶贱心施。}%

@{無待急恨心。無法應當施。}@
%{无待急恨心。无法应当施。}%

@{無不敬心施。無棄著地施。}@
%{无不敬心施。无弃著地施。}%

@{無求惱者施。無垢競勝施。}@
%{无求恼者施。无垢竞胜施。}%

@{無戲弄求者。無不自手施。}@
%{无戏弄求者。无不自手施。}%

@{不輕於少物，以多自高施。}@
%{不轻于少物，以多自高施。}%

@{不以聲聞乘、辟支佛乘施。}@
%{不以声闻乘、辟支佛乘施。}%

@{不限一世施。無有非時施。}@
%{不限一世施。无有非时施。}%

@{世尊無數劫，行諸希有施，}@
%{世尊无数劫，行诸希有施，}%

@{皆為無上道，不為求自樂。}@
%{皆为无上道，不为求自乐。}%

@{於諸佛法中，出家行遠離，}@
%{于诸佛法中，出家行远离，}%

@{修習諸佛法，為諸人天說。}@
%{修习诸佛法，为诸人天说。}%

@{說如是施法，於諸施中上，}@
%{说如是施法，于诸施中上，}%

@{猶如日光明，星月中殊勝。}@
%{犹如日光明，星月中殊胜。}%

@{如是勝捨處，超越諸天人，}@
%{如是胜舍处，超越诸天人，}%

@{猶亦如世尊，一切世間上。}@
%{犹亦如世尊，一切世间上。}%

@{是故能具足，如是勝捨處，}@
%{是故能具足，如是胜舍处，}%

@{名聞無量劫，流布無窮已。}@
%{名闻无量劫，流布无穷已。}%

@{世尊無量劫，護持清淨戒，}@
%{世尊无量劫，护持清净戒，}%

@{開諸禪定門，為得深寂處。}@
%{开诸禅定门，为得深寂处。}%

@{先離於五相，後行八解脫，}@
%{先离于五相，后行八解脱，}%

@{入淨三三昧，亦住三解脫。}@
%{入净三三昧，亦住三解脱。}%

@{世尊善分別，六十五種禪，}@
%{世尊善分别，六十五种禅，}%

@{無有一禪定，先來不生者。}@
%{无有一禅定，先来不生者。}%

@{於此諸定中，亦不受其味。}@
%{于此诸定中，亦不受其味。}%

@{世尊因諸定，得三種神通，}@
%{世尊因诸定，得三种神通，}%

@{以此度眾生，是故一切勝。}@
%{以此度众生，是故一切胜。}%

@{世尊無量劫，等心弘慈化。}@
%{世尊无量劫，等心弘慈化。}%

@{阿僧祇眾生，令住於梵世。}@
%{阿僧祇众生，令住于梵世。}%

@{能以巧方便，善說禪定故。}@
%{能以巧方便，善说禅定故。}%

@{世尊菩薩時，常於無量世，}@
%{世尊菩萨时，常于无量世，}%

@{無貪煩惱纏，而往來世間。}@
%{无贪烦恼缠，而往来世间。}%

@{過去得值者，無量生天上。}@
%{过去得值者，无量生天上。}%

@{過去諸菩薩，所可行寂滅，}@
%{过去诸菩萨，所可行寂灭，}%

@{世尊菩薩時，亦等無有異。}@
%{世尊菩萨时，亦等无有异。}%

@{是故於寂滅，勝處悉充滿。}@
%{是故于寂灭，胜处悉充满。}%

@{世尊菩薩時，所有諸智慧，}@
%{世尊菩萨时，所有诸智慧，}%

@{以慧求菩提，今成是慧報。}@
%{以慧求菩提，今成是慧报。}%

@{一切所資食，如人依地生。}@
%{一切所资食，如人依地生。}%

@{世尊於世世，捨十闇惡道，}@
%{世尊于世世，舍十闇恶道，}%

@{常行十善道，斯由慧氣分。}@
%{常行十善道，斯由慧气分。}%

@{捨五欲五蓋，得種種禪定，}@
%{舍五欲五盖，得种种禅定，}%

@{無量劫數世，不從他人受，}@
%{无量劫数世，不从他人受，}%

@{善哉大聖尊，悉是慧勢力。}@
%{善哉大圣尊，悉是慧势力。}%

@{眾生因世尊，無量生六天，}@
%{众生因世尊，无量生六天，}%

@{亦令至梵世，斯皆由慧力。}@
%{亦令至梵世，斯皆由慧力。}%

@{世尊於生死，苦樂所迷悶，}@
%{世尊于生死，苦乐所迷闷，}%

@{不失菩提心，斯皆是慧力。}@
%{不失菩提心，斯皆是慧力。}%

@{世尊於生死，不樂而常在，}@
%{世尊于生死，不乐而常在，}%

@{樂涅槃不取，斯皆是慧力。}@
%{乐涅槃不取，斯皆是慧力。}%

@{安坐道場時，降魔及軍眾，}@
%{安坐道场时，降魔及军众，}%

@{度脫諸群生，斯皆是慧力。}@
%{度脱诸群生，斯皆是慧力。}%

@{本求菩提時，集無量助法，}@
%{本求菩提时，集无量助法，}%

@{聞者常迷悶，何況能受行，}@
%{闻者常迷闷，何况能受行，}%

@{世尊能堪忍，斯皆是慧力。}@
%{世尊能堪忍，斯皆是慧力。}%

@{經書諸技術，世世生自知，}@
%{经书诸技术，世世生自知，}%

@{亦能兼教人，斯皆是慧力。}@
%{亦能兼教人，斯皆是慧力。}%

@{親近無量佛，悉飲甘露教，}@
%{亲近无量佛，悉饮甘露教，}%

@{種種諮請問，亦隨而分別。}@
%{种种咨询请问，亦随而分别。}%

@{經法智慧中，未曾有悋惜，}@
%{经法智慧中，未曾有悋惜，}%

@{乃至僕僮奴，亦諮受善語，}@
%{乃至僕僮奴，亦咨询善语，}%

@{世尊以是故，慧勝處流布。}@
%{世尊以是故，慧胜处流布。}%

@{世尊於前世，求是菩提時，}@
%{世尊于前世，求是菩提时，}%

@{於一切眾生，行大慈悲心。}@
%{于一切众生，行大慈悲心。}%

@{以第一智慧，常出大勢力，}@
%{以第一智慧，常出大势力，}%

@{悉作無量種，希有諸難事。}@
%{悉作无量种，希有诸难事。}%

@{一切諸世間，盡共無量劫，}@
%{一切诸世间，尽共无量劫，}%

@{說之不可盡，亦非算數及，}@
%{说之不可尽，亦非算数及，}%

@{如是等諸事，超越於人天，}@
%{如是等诸事，超越于人天，}%

@{一切世間中，奇特無有比。}@
%{一切世间中，奇特无有比。}%

@{大業所獲果，具足一切智，}@
%{大业所获果，具足一切智，}%

@{能破生死王，安住法王處。}@
%{能破生死王，安住法王处。}%
@{助念佛三昧品第二十五}@
%{助念佛三昧品第二十五}%

@{菩薩應以此，四十不共法，}@
%{菩萨应该用这四十种不共法，}%

@{念諸佛法身，佛非色身故。}@
%{念诵诸佛的法身，因为佛不是色身。}%

@{是偈次第略解四十不共法六品中義，是故行者先念色身佛，次念法身佛。何以故？新發意菩薩應以三十二相、八十種好念佛，如先說。}@
%{这首偈颂是依次解释四十种不共法六种品类中的意义。所以修行者应该先念色身佛，再念法身佛。为什么呢？因为新发意菩萨应该用三十二相、八十种好相念佛，就像前面所说。}%

@{轉深入得中勢力，應以法身念佛。}@
%{修行深入到中等力量的时候，应该用法身念佛。}%

@{心轉深入得上勢力，應以實相念佛而不貪著。}@
%{修行深入到上等力量的时候，应该用实相念佛，而不贪着。}%

@{不染著色身，法身亦不著；}@
%{不染着色身，法身也不执着；}%

@{善知一切法，永寂如虛空。}@
%{善知一切法，永远寂静如同虚空。}%

@{是菩薩得上勢力，不以色身、法身深貪著佛。何以故？信樂空法故，知諸法如虛空。虛空者，無障礙故。障礙因緣者，諸須彌山、由乾陀等十寶山、鐵圍山、黑山、石山等，如是無量障礙因緣。何以故？是人未得天眼故，念他方世界佛，則有諸山障礙。}@
%{这个菩萨获得了上等力量，不以色身、法身深深贪着佛。为什么呢？因为他相信并乐于空法，知道诸法如同虚空。虚空是无障碍的。障碍因缘，比如须弥山、由乾陀等十宝山、铁围山、黑山、石山等等，有无数的障碍因缘。为什么呢？因为这个人没有得到天眼，念诵他方世界的佛，就会有这些山阻碍。}%

@{是故新發意菩薩，應以十號妙相念佛。如說：}@
%{所以新发意菩萨，应该用十种妙相念佛。就像经文所说：}%

@{新發意菩薩，以十號妙相；}@
%{新发意菩萨，用十种妙相；}%

@{念佛無毀失，猶如鏡中像。}@
%{念佛没有毁失，如同镜中的像。}%

@{十號妙相者，所謂如來、應供、正遍知、明行足、善逝、世間解、無上士、調御丈夫、天人師、佛、世尊。}@
%{十种妙相，就是：如来、应供、正遍知、明行足、善逝、世间解、无上士、调御丈夫、天人师、佛、世尊。}%

@{無毀失者，所觀事空如虛空，於法無所失。何以故？諸法本來無生寂滅故，如是一切諸法皆亦如是。是人以緣名號，增長禪法，則能緣相。是人爾時即於禪法得相，所謂身得殊異快樂，當知得成般舟三昧，三昧成故得見諸佛。}@
%{没有毁失，是指所观的事物如同虚空，在法上没有损失。为什么呢？因为诸法本来无生无灭，一切诸法都是如此。这个人用缘念佛号，增长禅定，就能缘到相。这个人当时就在禅定中得到相，也就是身体得到殊胜的快乐，就知道已经成就了般舟三昧，三昧成就了就能见到诸佛。}%

@{如鏡中像者，若菩薩成此三昧已，如淨明鏡自見面像，如清澄水中見其身相。}@
%{如同镜中的像，如果菩萨成就了这种三昧，就像净明的镜子自己照见自己的影像，就像清澈的水中看见自己的身体。}%

@{初時隨先所念佛，見其色像，見是像已後，若欲見他方諸佛，隨所念方得見諸佛，無所障礙。是故此人：}@
%{开始的时候，随着先前念诵的佛，看见他的色身，看见这个色身之后，如果想要看见他方世界的佛，随着念诵的方向就能看见诸佛，没有任何障碍。所以这个人：}%

@{雖未有神通，飛行到于彼，}@
%{虽然没有神通，飞到那里，}%

@{而能見諸佛，聞法無障礙。}@
%{却能看见诸佛，听闻佛法没有障碍。}%

@{是新發意菩薩，於諸須彌山等諸山無能為作障礙，亦未得神通、天眼、天耳，未能飛行從此國至彼國，以是三昧力故，住此國土得見他方諸佛世尊，聞所說法，常修習是三昧故，得見十方真實諸佛。}@
%{这个新发意菩萨，对须弥山等一切山都没有障碍，也没有得到神通、天眼、天耳，不能飞行从这个国土到那个国土，以这种三昧的力量，住在这个国土就能看见他方世界的佛世尊，听闻他们所说佛法，经常修习这种三昧，就能看见十方真实的诸佛。}%

@{問曰：}@
%{问：}%

@{如是定以何法能生？云何可得？}@
%{这种定以什么方法才能生起？怎么才能获得呢？}%

@{答曰：}@
%{答：}%

@{親近善知識，精進無懈退，}@
%{亲近善知识，精进不懈怠，}%

@{智慧甚堅牢，信力不妄動。}@
%{智慧非常坚固，信心不妄动。}%

@{以是四法能生是三昧。}@
%{用这四种方法就能生起这种三昧。}%

@{親近善知識者，能以是三昧教誨人者，名為善知識，應加恭敬，勤心親近。}@
%{亲近善知识，是指能够用这种三昧教诲别人的人，叫做善知识，应该恭敬他，勤心亲近。}%

@{莫有懈怠、廢退、捨離，則得聞是深三昧義。}@
%{不要懈怠、退步、舍弃，就能听到这种深奥的三昧的意义。}%

@{利智、通達智、不失智，名為堅牢。}@
%{利智、通达智、不失智，叫做坚固。}%

@{信根深固，若沙門、婆羅門，若天、魔、梵及餘世人無能傾動，名為信力不可動。}@
%{信心根基深固，无论是沙门、婆罗门，还是天、魔、梵和其他世间人，都不能动摇你的信心，叫做信心不可动摇。}%

@{如是四法能生三昧。}@
%{这四种方法就能生起三昧。}%

@{復次——}@
%{其次——}%

@{慚愧愛恭敬，供養說法者，}@
%{惭愧、爱、恭敬，供养说法人，}%

@{猶如諸世尊，能生是三昧。}@
%{如同诸佛世尊，就能生起这种三昧。}%

@{慚愧、愛、恭敬者，於說法者深生慚愧、恭恪、愛樂、供養如佛。}@
%{惭愧、爱、恭敬，是指对说法人深生惭愧、恭敬、喜爱、供养，如同供养佛。}%

@{如是四法能生是三昧。}@
%{这四种方法就能生起这种三昧。}%

@{復次，初四法者：一、於三月未甞睡眠，唯除便利、飲食、坐起。二、於三月乃至彈指，不生我心。三、於三月經行不息。四、於三月兼以法施，不求利養。是為四。}@
%{其次，最初四种方法是：一、在三个月内不睡觉，只除方便、饮食、坐起。二、在三个月内，即使弹指，也不生起“我”的心。三、在三个月内不停地经行。四、在三个月内，兼以布施法，不求利养。这四种。}%

@{復有四法：一、能見佛。二、安慰勸人聽是三昧。三、常不貪嫉行菩提心者。四、能集菩薩所行道法。是為四。}@
%{还有四种方法：一、能看见佛。二、安慰劝人听闻这种三昧。三、经常不贪嫉，行菩提心的人。四、能够集聚菩萨所行之道法。这四种。}%

@{復有四法：一、造作佛像，乃至畫像。二、當善書寫是三昧經，令信樂者得已誦讀。三、教增上慢人，令離增上慢法，使得阿耨多羅三藐三菩提。四、當護持諸佛正法。是為四。}@
%{还有四种方法：一、制作佛像，甚至画佛像。二、应该善于书写这种三昧经，让信乐者能够诵读。三、教导增上慢的人，让他们远离增上慢法，获得阿耨多罗三藐三菩提。四、应该护持诸佛正法。这四种。}%
@{復有四法：一、少語言。二、在家、出家不與共住。三、常繫心取所緣相。四、樂遠離，空閑靜處。是為四。}@
%{还有四种方法：第一，少说话。第二，在家、出家都不和别人住在一起。第三，经常把心系在所缘之相上。第四，喜欢远离人群，在清静的地方。这四种方法，就是这四种。}%

@{初五法者：一、無生忍法，厭離一切諸有為法，不樂一切諸所生處，不受一切諸外道法，惡厭一切世間諸欲乃至不念，何況身近。二、心常修習無量諸法，定在一處，於諸眾生無有瞋礙，心常隨順行四攝法。三、能成就慈、悲、喜、捨，不出他過。四、能多集佛所說法，如所說行。五、清淨身、口、意業及見。是為五。}@
%{这最初五种方法：第一，无生忍法，厌离一切有为法，不喜欢一切有生之处，不接受一切外道法，厌恶一切世间欲望，甚至不念，更何况近身了。第二，心经常修习无量诸法，安定在一处，对一切众生没有瞋恨，心经常随顺行四摄法。第三，能够成就慈、悲、喜、舍，不揭露别人的过失。第四，能够多集佛所说法，如同所说去修行。第五，清净身、口、意业以及见。这五种方法，就是这五种。}%

@{復有五法：一、樂如經所讚布施，無有慳心，樂說深法，無所悋惜，亦能自住。二、忍辱柔和，同住歡喜，惡口、罵詈、鞭、捶、縛等，但推業緣，不恚他人。三、常樂聽是三昧，讀誦通利，為人解說，令流布增廣，勤行修習。四、心無妬嫉，不自高身，不下他人，除眠睡蓋。五、於佛、法、僧寶信心清淨。於上、中、下坐，深心供奉，他有小恩，常憶不忘，常住真實語中。是為五。}@
%{还有五种方法：第一，喜欢如同经书所赞叹的布施，没有吝啬之心，喜欢说深法，无所吝惜，也能够自住。第二，忍辱柔和，与人同住欢喜，恶口、谩骂、鞭打、捆绑等等，都推说是业缘，不怨恨他人。第三，经常喜欢听这种三昧，读诵通利，为别人讲解，使它流传增广，勤奋修行。第四，心没有嫉妒，不自高身，不贬低他人，除掉睡眠的遮盖。第五，对佛、法、僧宝信心清净。在上位、中位、下位坐，深心供奉，他有小恩，经常不忘，常住真实语中。这五种方法，就是这五种。}%

%{此外——}%

@{出家諸菩薩，所學三昧法；}@
%{出家菩萨，所学的这种三昧法；}%

@{在家菩薩者，是法應當知。}@
%{在家菩萨，应该知道这种方法。}%

@{若在家菩薩，欲修習是三昧：一、當深以信心。二、不求業果報。三、當捨一切內、外物。四、歸命三寶。五、淨持五戒，無有毀缺。六、具足行十善道，亦令餘人住此法中。七、斷除婬欲。八、毀呰五欲。九、不嫉妬。十、於妻子中不生愛著。十一、心常願出家。十二、常受齊戒。十三、心樂住寺廟。十四、具足慚愧。十五、於淨戒比丘起恭敬心。十六、不慳悋法。十七、於說法者，深愛敬心。十八、於說法者，生父母、大師想。十九、於說法者以諸樂具，敬心供養。二十、知恩報恩。如是在家菩薩，住如是等功德者，則能學是三昧。}@
%{如果在家菩萨，想要修习这种三昧：第一，应该深深以信心。第二，不求业果报。第三，应该舍弃一切内、外物。第四，归命三宝。第五，清净持守五戒，没有毁坏。第六，具足地行十善道，也让其他人住在这个法中。第七，断除淫欲。第八，毁弃五欲。第九，不嫉妒。第十，对妻子中不生爱着。第十一，心经常愿意出家。第十二，经常受持齐戒。第十三，心喜欢住寺庙。第十四，具足惭愧。第十五，对持净戒的比丘起恭敬心。第十六，不吝啬法。第十七，对说法者，深爱敬心。第十八，对说法者，生父母、大师想。第十九，对说法者以各种乐具，恭敬心供养。第二十，知恩报恩。像这样的在家菩萨，住持如是等功德者，就能学习这种三昧。}%

@{出家菩薩修習是三昧法者，所謂：一、於戒無毀疵。二、持戒不雜污。三、持戒不濁。四、清淨戒。五、無損戒。六、不取戒。七、不依戒。八、不得戒。九、不退戒。十、持聖所讚戒。十一、持智所稱戒。十二、隨波羅提木叉戒。十三、具足威儀行處。十四、乃至微小罪心大怖畏。十五、淨身、口、意業。十六、淨命。十七、所有戒盡受持。十八、信樂甚深法。十九、於無所得法，心能忍，空、無相、無願法中心不驚。二十、勤發精進。二十一、念常在前。二十二、信心堅固。二十三、具足慚愧。二十四、不貪利養。二十五、無嫉妬。二十六、住頭陀功德。二十七、住細行法中。二十八、不樂說世間俗語。二十九、遠離聚語。三十、知報恩。三十一、知作恩，報恩者。三十二、於和上、阿闍梨所，生恭敬，忌難心。三十三、破除憍慢。三十四。降伏我心。三十五、善知識難遇故，勤心供給。三十六、所從聞是法處。若得經卷，若口誦處。於此人所，生父母想、善知識想、大師想、大慚愧愛敬想。三十七、常樂阿練若。三十八、不樂住城邑聚落。三十九、不貪著檀越善知識家。四十、不惜身命。四十一、心常念死。四十二、不存利養。四十三、於諸物中，心不染著。四十四、無所渴愛。四十五、守護正法。四十六、不著衣鉢。四十七、不畜遺餘。四十八、但欲乞食。四十九、次第乞食。五十、常知慚愧心常有悔。五十一、不畜金銀、珍寶、錢財，離諸不善悔。五十二、心無纏垢。五十三、常行慈心。五十四、除斷瞋恚。五十五、常行悲心。五十六、除斷愛著。五十七、常求利安一切世間。五十八、常憐愍一切眾生。五十九、常樂經行。六十、除却睡眠。}@
%{出家菩萨修习这种三昧法，就是：第一，对戒律没有毁坏。第二，持戒不杂污。第三，持戒不浊。第四，清净戒。第五，无损戒。第六，不取戒。第七，不依戒。第八，不得戒。第九，不退戒。第十，持圣人所赞叹的戒。第十一，持智慧所称赞的戒。第十二，随波罗提木叉戒。第十三，具足威仪行处。第十四，即使是微小的罪过，心中也感到很大恐惧。第十五，清净身、口、意业。第十六，清净命。第十七，所有戒律都尽力受持。第十八，信乐甚深法。第十九，对无所得法，心能忍，空、无相、无愿法心中不惊。第二十，勤奋发精进。第二十一，念常在前。第二十二，信心坚固。第二十三，具足惭愧。第二十四，不贪利养。第二十五，无嫉妒。第二十六，住头陀功德。第二十七，住在细行法中。第二十八，不喜欢说世间俗语。第二十九，远离聚语。第三十，知报恩。第三十一，知道作恩，报恩者。第三十二，对和上、阿闍梨，生恭敬，忌难心。第三十三，破除憍慢。第三十四，降伏我心。第三十五，善知识难遇，勤心供养。第三十六，从哪里听到这个法，无论是得到经卷，还是口头诵读，对这个人，生父母想、善知识想、大师想、大惭愧爱敬想。第三十七，经常喜欢阿练若。第三十八，不喜欢住在城邑聚落。第三十九，不贪着檀越善知识家。第四十，不惜身命。第四十一，心经常念死。第四十二，不存利养。第四十三，对一切事物，心中不染着。第四十四，无所渴爱。第四十五，守护正法。第四十六，不执着衣钵。第四十七，不蓄积遗留。第四十八，只想要乞食。第四十九，次第乞食。第五十，经常知道惭愧，心中经常有悔恨。第五十一，不畜金银、珍宝、钱财，远离一切不善悔恨。第五十二，心无缠垢。第五十三，经常行慈心。第五十四，除断瞋恚。第五十五，经常行悲心。第五十六，除断爱着。第五十七，经常求利安一切世间。第五十八，经常怜悯一切众生。第五十九，经常喜欢经行。第六十，除掉睡眠。}%
@{出家菩薩住如是等法中，應修習是三昧。}@
%{出家修行的人，要住在这些法中，应该修习这种三昧。}%

@{能生是般舟三昧餘助法亦應修習。何等是？一、緣佛恩常念在前。二、不令心散亂。三、繫心在前。四、守護根門。五、飲食知止足。六、初夜後夜常修三昧。七、離諸煩惱障。八、生諸禪定。九、禪中不受味。十、散壞色相。十一、得不淨相。十二、不貪五陰。十三、不著十八界。十四、不染十二入。十五、不恃族姓。十六、破憍慢。十七、於一切法，心常空寂。十八、於諸眾生，生親族想。十九、不取戒。二十、不分別定。二十一、應勤多學。二十二、以是多學，而不憍慢。二十三、於諸法無疑。二十四、不違諸佛。二十五、不逆法。二十六、不壞僧。二十七、常詣諸賢聖。二十八、遠離凡夫。二十九、樂出世間論。三十、修六和敬法。三十一、常修習五解脫處。三十二、除九瞋惱事。三十三、斷八懈怠法。三十四、修八精進。三十五、常觀九相。三十六、得大人八覺。三十七、具足諸禪定三昧。三十八、於此禪定無所貪無所得。三十九、聽法專心。四十、壞五陰相。四十一、不住事相。四十二、深怖畏生死。四十三、於五陰生怨賊想。四十四、於諸入中，生空聚想。四十五、於四大中生毒蛇想。四十六、於涅槃中生寂滅想安隱樂想。四十七、於五欲中生涎唾想，心樂出離。四十八、不違佛教。四十九、於一切眾生無所諍訟。五十、教化眾生令安住一切功德。}@
%{能生起这种般舟三昧，还有其他辅助的方法也应该修习。这些方法是什么呢？第一，要经常想着佛的恩德。第二，不要让心散乱。第三，要专注于当下。第四，要守护好六根。第五，吃饭要懂得知足。第六，初夜和后夜都要经常修习三昧。第七，要远离一切烦恼。第八，要生起各种禅定。第九，在禅定中不执着于任何感受。第十，要破除一切色相。第十一，要获得不净观。第十二，不要贪恋五蕴。第十三，不要执着于十八界。第十四，不要染着十二入。第十五，不要依靠出身。第十六，要破除傲慢。第十七，对一切法，心要常空寂。第十八，对一切众生，要生起亲族想。第十九，不要执着于戒律。第二十，不要分别定。第二十一，要勤奋学习。第二十二，因为学习很多，所以不要傲慢。第二十三，对一切法都没有疑惑。第二十四，不要违背佛的教诲。第二十五，不要逆反佛法。第二十六，不要破坏僧团。第二十七，要经常去拜访贤圣之人。第二十八，要远离凡夫。第二十九，要喜欢谈论出世间法。第三十，要修习六和敬法。第三十一，要经常修习五解脱处。第三十二，要消除九种瞋恼。第三十三，要断除八种懈怠。第三十四，要修习八种精进。第三十五，要经常观照九相。第三十六，要获得大人八觉。第三十七，要具足各种禅定三昧。第三十八，在禅定中，不要贪求任何东西，也不要执着于任何所得。第三十九，听法的时候要专心。第四十，要破除五蕴的相。第四十一，不要执着于事相。第四十二，要深深地恐惧生死。第四十三，对五蕴要生起怨贼想。第四十四，对一切入要生起空聚想。第四十五，对四大要生起毒蛇想。第四十六，对涅槃要生起寂灭想，安隐乐想。第四十七，对五欲要生起涎唾想，心中渴望出离。第四十八，不要违背佛的教诲。第四十九，对一切众生不要争执。第五十，要教化众生，让他们安住于一切功德。}%

@{菩薩行是般舟三昧，果報亦應知。}@

@{問曰：}@
%{问：}%

@{修習是三昧得何果報？}@
%{修习这种三昧，能得到什么果报呢？}%

@{答曰：}@
%{答：}%

@{於無上道得不退轉報。}@
%{在无上道中得到不退转的果报。}%

@{復次，如《經》所說果報。}@
%{另外，就像经书中所说，这种三昧的果报是：}%

@{佛語颰陀婆羅菩薩：譬如有人能摧碎三千世界地皆如微塵，又三千大千世界中所有草木花葉，一切諸物皆為微塵。颰陀婆羅！以一微塵為一佛世界，有爾所世界，皆滿中上妙珍寶以用布施。跋陀婆羅於意云何？是人以是布施因緣得福多不？甚多！世尊。佛言：颰陀婆羅！我今實語汝。若有善男子，得聞諸佛現前三昧，不驚、不畏其福無量。何況信受、持讀、諷誦、為人解說，何況定心修習，如一搆牛乳頃。颰陀婆羅！我說此人福德尚無有量，何況能得成是三昧者。}@
%{佛对颰陀婆羅菩萨说：就像有人能把三千个世界的地都摧毁成微尘一样，又像三千大千世界中所有的草木花叶，一切东西都变成微尘一样。颰陀婆羅！用一粒微尘来比喻一个佛的世界，有那么多世界，都充满了最上妙的珍宝，用来布施。跋陀婆羅，你觉得这个人因为布施而得到的福报多不多？非常多！世尊。佛说：颰陀婆羅！我现在告诉你，如果有人能听到佛现在讲的三昧，不惊慌、不害怕，他的福报就无量了。何况是相信、持诵、背诵、为别人讲解呢？何况是专心修习，就像挤牛奶那么一会儿呢？颰陀婆羅！我说这个人福德尚且无量，何况是能得到这种三昧呢？}%

@{佛又告颰陀婆羅：}@
%{佛又对颰陀婆羅说：}%

@{若有善男子、善女人，受持讀誦為他人說，若劫盡時，設墮此火，火即尋滅。}@
%{如果有人，无论是男是女，都能持诵、背诵，并且为别人讲解，即使在劫尽的时候，即使掉进火里，火也会立刻熄灭。}%

@{颰陀婆羅！持是三昧者，若有官事，若遇怨賊、師子、虎狼、惡獸、惡龍、諸毒虫等，若夜叉、羅剎、鳩槃荼、毘舍闍等，若人、非人等，若害身，若害命，若毀戒，無有是處。若讀誦為人說時，亦無衰惱，唯除業報必應受者。}@
%{颰陀婆羅！持诵这种三昧的人，即使有官司，即使遇到强盗、狮子、老虎、狼、恶兽、恶龙、各种毒虫，即使遇到夜叉、罗刹、鸠槃荼、毘舍闍，即使遇到人、非人，即使要伤害身体、要伤害生命、要毁坏戒律，都没有这种事情发生。如果持诵、背诵，并且为别人讲解，也不会衰败、烦恼，只是除掉业报中必须承受的。}%

@{復次，颰陀婆羅！菩薩受持讀誦是三昧時，若得眼、耳、鼻、舌、口、齒病、風、寒、冷、病，如是等種種餘病，以是病故，而失壽命，無有是處，唯除業報必應受者。}@
%{另外，颰陀婆羅！菩萨持诵、背诵这种三昧的时候，即使得了眼病、耳病、鼻病、舌病、口病、牙病、风寒、感冒、各种疾病，因为这些疾病而失去寿命，都不会发生，只是除掉业报中必须承受的。}%

@{復次，颰陀婆羅！若人受持讀誦是三昧者，諸天守護，諸龍、夜叉、摩睺羅伽、人、非人、四天王、帝釋、梵天王、諸佛世尊，皆共護念。}@
%{另外，颰陀婆羅！如果有人持诵、背诵这种三昧，诸天都会守护他，诸龙、夜叉、摩睺罗伽、人、非人、四天王、帝释天、梵天，以及诸佛世尊，都会共同护念他。}%

@{復次，是人皆為諸天所共愛念，乃至諸佛皆共愛念。}@
%{另外，这个人都会受到诸天的共同爱念，甚至诸佛都会共同爱念他。}%
@{復次，是人皆為諸天所共稱讚，乃至諸佛皆共稱讚。}@
%{再说，这个人会被所有天人赞叹，甚至所有佛陀都会赞叹他。}%

@{復次，諸天皆欲見是菩薩來至其所，乃至諸佛皆欲見是菩薩來至其所。}@
%{再说，所有天人都想见到这位菩萨来到他们那里，甚至所有佛陀也都想见到这位菩萨来到他们那里。}%

@{復次，是菩薩受持是三昧者，所未聞經，自然得聞。}@
%{再说，这位菩萨持守这种三昧，他以前没有听过的经文，自然就能听到了。}%

@{復次，是菩薩得是三昧者，乃至夢中皆得如是諸利益事。}@
%{再说，这位菩萨得到了这种三昧，甚至在梦中也能得到这些利益。}%

@{颰陀婆羅！菩薩若我一劫，若減一劫，說、受持、讀誦是三昧者，功德不可得盡。何況得成就者。}@
%{迦陀婆罗！菩萨如果修行一劫，或者少修一劫，说、持守、诵读这种三昧，它的功德是不可穷尽的。更何况是成就了这种三昧的人呢。}%

@{颰陀婆羅！如人於百歲中，身力輕健，其疾如風，是人百歲行不休息，常至東方、南、西、北方、四維上下。於汝意云何？是人所詣十方，有人能數知里數不？颰陀婆羅言：不可數也，唯除如來、舍利弗、阿惟越致，餘不能知。}@
%{迦陀婆罗！就像一个人在百岁中，身体轻健，行动迅速如风，这个人百岁中不停地行走，经常去东方、南方、西方、北方、上下四方。你认为怎么样？这个人所到之处，有人能数清路程吗？迦陀婆罗说：不可数清，只有佛陀、舍利弗、阿惟越致能知道，其他人不知道。}%

@{颰陀婆羅！若有善男子、善女人以是人所行處，滿中真金布施。若有人但聞是三昧，以四種隨喜迴向阿耨多羅三藐三菩提，常求多聞。如過去諸佛行菩薩道時，隨喜是三昧，我亦如是。如今現在菩薩，隨喜是三昧，我亦如是。如未來諸佛行菩薩道時，隨喜是三昧，我亦如是。如過去、未來、現在菩薩所行三昧，我亦隨喜，皆為得多聞，我亦如是，求多聞故，隨喜是三昧。}@
%{迦陀婆罗！如果有善男子、善女人，以这个人所行之处，布满真金来布施。如果有人只是听到这种三昧，以四种随喜回向无上正等正觉，经常寻求多闻。就像过去诸佛修行菩萨道时，随喜这种三昧，我也一样。现在菩萨随喜这种三昧，我也一样。将来诸佛修行菩萨道时，随喜这种三昧，我也一样。就像过去、未来、现在菩萨所修行的三昧，我也随喜，都是为了得到多闻，我也一样，因为追求多闻，所以随喜这种三昧。}%

@{颰陀婆羅！是隨喜福德，於上福德百分不及一，百千萬億分不及一，乃至算數譬喻所不能及，是三昧得如是無量無邊果報。}@
%{迦陀婆罗！这种随喜的福德，比上面的福德百分之一还少，比百千万亿分之一还少，甚至用算术和比喻都无法形容，这种三昧能得到如此无量无边的果报。}%

%{再说——}%

@{是三昧住處，少中多差別；}@
%{这种三昧的住处，有少、中、多三种差别。}%

@{如是種種相，皆當須論義。}@
%{这些种种相貌，都需要进行解释。}%

@{是三昧所住處，少相、中相、多相，如是等應分別如，是事應當解釋。}@
%{这种三昧的住处，有少相、中相、多相，这些都需要分别清楚，这件事应该解释清楚。}%

@{住處者，是三昧或於初禪可得，或第二禪、或第三禪、或第四禪可得，或初禪中間得勢力，能生是三昧。}@
%{住处，就是说这种三昧，或者在初禅可以得到，或者在第二禅、第三禅、第四禅可以得到，或者在初禅中间得到力量，就能生起这种三昧。}%

@{或少者，人勢力少故名為少，又少時住故名為少，又見少佛世界故名為少。}@
%{或者少，是因为人的力量少，所以叫做少，又因为住的时间少，所以叫做少，又因为看到佛的世界少，所以叫做少。}%

@{中、多亦如是。}@
%{中、多也是这样。}%

@{說是三昧，或說有覺有觀，或無覺有觀，或無覺無觀。}@
%{说这种三昧，或者说是有觉有观，或者说无觉有观，或者说无觉无观。}%

@{或喜相應，或樂相應，或不苦不樂相應。}@
%{或者与喜相应，或者与乐相应，或者与不苦不乐相应。}%

@{或有入出息，或無入出息。}@
%{或者有入出息，或者无入出息。}%

@{或定是善性。}@
%{或者定是善性。}%

@{或有漏、或無漏。}@
%{或者是有漏，或者无漏。}%

%{或者与欲界相系，或者与色界相系，或者与无色界相系，或者不属于欲界，或者不属于色界，或者不属于无色界。}%

@{是三昧是心數法、心相應、隨心行法、共心生法。}@
%{这种三昧是心数法、心相应、随心行法、共心生法。}%

%{不是色，不是现，是能缘。}%

@{非業、業相應、隨業行。}@
%{不是业，是业相应，随业行。}%

@{知見亦如是，不與七覺合。}@
%{知见也是这样，不与七觉合。}%

@{如是一切諸分別三昧義，皆應此中說。}@
%{像这样所有分别三昧的意义，都应该在这里说。}%

@{復次，修習是三昧得見諸佛。如說：}@
%{再说，修习这种三昧就能见到诸佛。就像经文中所说：}%

@{得見諸佛已，勤心而供養，}@
%{见到诸佛之后，要勤心供养他们，}%

@{善根得增長，能疾化眾生。}@
%{善根就会增长，就能迅速地教化众生。}%

@{供養，名心意清淨。}@
%{供养，就是指心意清净。}%

@{恭敬歡喜念佛有無量功德，以種種讚歎名口供養。}@
%{恭敬欢喜地念佛，有无量的功德，用各种赞叹的话语，叫做口供养。}%

@{敬禮華香等名身供養。}@
%{恭敬地礼拜、献花、献香等，叫做身供养。}%

@{是故福德轉更增長。如穀子在地雨潤生長，疾教化者，令眾生住三乘中，如是菩薩增長善根。}@
%{所以福德就会更加增长。就像谷子在地里，雨水滋润生长，迅速教化众生，使众生住于三乘之中，像这样的菩萨，善根就会增长。}%

@{以初二攝法，攝取諸眾生；}@
%{用前两种摄法，摄取众生；}%

@{後餘二攝法，未盡能信受。}@
%{后面的两种摄法，他们还不完全能接受。}%

@{初二者：布施、愛語。利益、同事名為後二。是菩薩在初地，不能具解故，但能信受。}@
%{前两种是：布施、爱语。利益、同事叫做后两种。这位菩萨在初地，不能完全理解，所以只能够接受。}%

@{爾時諸善根，迴向於佛道；}@
%{这时，所有的善根都回向佛道；}%

@{如彼成煉金，調熟則堪用。}@
%{就像炼金，调熟了才能使用。}%

@{智慧火所煉故，於菩薩所行事中，善根成熟則堪任用。}@
%{智慧之火炼化，在菩萨所做的事情中，善根成熟了，才能担当重任。}%

@{譬喻品第二十六}@
%{第二十六品 譬喻品}%

@{是菩薩應聞，地相得修果，}@
%{菩萨应该了解，地相、得、修、果，}%

@{為得諸地分，故勤行精進。}@
%{为了得到每一地的分位，所以勤奋精进修行。}%

@{相者，是相貌，因以得知。}@
%{相，就是相貌，通过它来认识。}%

@{得者，成就。以是法故，名成就是法。}@
%{得，就是成就。因为这个法，所以叫做成就法。}%

@{修，名得修、行修。}@
%{修，叫做得修、行修。}%

@{常念果者：從因有事成，名為果。}@
%{经常想到果位：从因缘有事情成就，就叫做果位。}%

@{是菩薩欲得十地行，應善聞相、得、修、果。}@
%{菩萨想要得到十地修行，应该了解地相、得、修、果。}%

@{聞者，從諸佛菩薩所聞，及勝己者。}@
%{闻，就是从诸佛菩萨那里听到，以及从比自己优秀的人那里听到。}%

@{為得諸地分者，為得是地分，故勤行精進。}@
%{为了得到每一地的分位，为了得到这个地的分位，所以勤奋精进修行。}%

@{此中初地相者，如先說：}@
%{这里讲初地的相，就像之前说的：}%

@{菩薩在初地，多所能堪受，}@
%{菩萨在初地，有很多能承受的，}%

@{不好於諍訟，其心多喜悅。}@
%{不喜欢争执，内心很喜悦。}%

@{常樂於清淨，悲心愍眾生，}@
%{经常喜欢清净，悲心怜悯众生，}%

@{無有瞋恚心，多行是七事。}@
%{没有嗔恨之心，经常做这七件事。}%

@{是故堪受、不諍、喜、悅、清淨、悲心、無瞋等七法，是初地相。}@
%{所以，堪受、不争、喜、悦、清净、悲心、无嗔这七法，是初地的相。}%

@{成就此堪受等七法，名為得。}@
%{成就了这堪受等七法，就叫做得。}%

@{復次，堪受等七法相，即是初地得。如偈說：}@
%{其次，堪受等七法相，就是初地的得。就像偈颂里说的：}%

@{若厚種善根，善行於諸行，}@
%{如果厚厚地种下善根，善于修行一切善行，}%

@{善集諸資生，善供養諸佛；}@
%{善于积累资生之物，善于供养诸佛；}%

@{善知識所護，具足於深心，}@
%{善知识护佑，具有圆满的深心，}%

@{悲心念眾生，信解無上法。}@
%{悲心念着众生，相信并理解无上法。}%

@{具此八法已，當自發願言；}@
%{具备了这八法之后，就应该自己发愿说：}%

@{我已得自度，當復度眾生。}@
%{我已经得到解脱，应该再度化众生。}%

@{為得十力故，入於必定聚；}@
%{为了得到十力，进入必定聚；}%

@{則生如來家，無有諸過咎。}@
%{那么就会出生在如来家，没有过失。}%

@{即轉世間道，入出世上道；}@
%{立刻转入世间道，进入出世间道；}%

@{以是得初地，此地名歡喜。}@
%{因为这样就得到了初地，这个地叫做欢喜地。}%

@{是故當知，為菩提故，所得決定心，名為初地得。}@
%{所以应该知道，为了菩提，所得到的决定心，叫做初地的得。}%

@{修，名從初發心乃至成諸佛現前三昧，於其中間，具說諸地功德。能生是諸功德，生已，修集增長，名為初地修。}@
%{修，就是从最初发心直到成佛现前三昧，在这期间，具足地说出每一地的功德。能够生起这些功德，生起之后，修集增长，叫做初地的修。}%

@{果者，先已處處說得若干福德，不迴向聲聞、辟支佛地。今當更說菩薩得初地果，能得菩薩數百定等。}@
%{果位，之前已经处处说得到很多福德，不回向声闻、辟支佛地。现在要再说菩萨得到初地果位，能够得到菩萨的数百度等。}%

@{初地分者；所有諸法合成初地，名為諸分。如麴米等合能成酒故，名酒因緣。所有諸法能成初地，名為初地分。}@
%{初地的分位，就是所有一切法合成初地，叫做诸分。就像用曲米等合起来就能酿成酒，叫做酒的因缘。所有一切法能够成就初地，叫做初地的分位。}%

@{所謂：}@
%{就是说：}%

@{信力轉增上，成就大悲心；}@
%{信力增长，成就大悲心；}%

@{慈愍眾生類，修善心無惓；}@
%{慈悲怜悯一切众生，修善心没有疲倦；}%

@{喜樂於妙法，常近善知識；}@
%{喜乐于妙法，经常亲近善知识；}%

@{慚愧及恭敬，柔軟和其心；}@
%{惭愧和恭敬，心柔软和顺；}%

@{樂觀法無著，一心求多聞；}@
%{喜欢观察法，没有执着，一心求多闻；}%

@{不貪於利養，離奸欺諂誑；}@
%{不贪图利养，远离奸诈欺骗；}%

@{不污諸佛家，不毀戒欺佛；}@
%{不污辱佛家，不毁戒欺骗佛；}%

@{深樂薩婆若，不動如大山；}@
%{深深地喜欢佛陀的智慧，不动如大山；}%

@{常樂修習行，轉上之妙法；}@
%{经常喜欢修习修行，转向上等的妙法；}%

@{樂出世間法，不樂世間法；}@
%{喜欢出世间法，不喜欢世间法；}%

@{即治歡喜地，難治而能治；}@
%{这就是欢喜地，难以治理却能够治理；}%

@{是故常一心，勤行此諸法；}@
%{所以经常一心，勤奋地修习这些法；}%

@{菩薩能成就，如是上妙法；}@
%{菩萨能够成就，像这样上等的妙法；}%

@{是則為安住，菩薩初地中。}@
%{这样就叫做安住，菩萨初地之中。}%

@{問曰：}@
%{问：}%

@{菩薩何用聞是初地相等為？}@
%{菩萨为什么要听闻初地的相等呢？}%

@{答曰：}@
%{答：}%

@{是菩薩初地相等法中，應善知方便，是故應聞。}@
%{菩萨在初地的相等法中，应该了解方便法门，所以应该听闻。}%

@{問曰：}@
%{问：}%

@{菩薩但應於此法中善知方便？更於餘法中善知方便？}@
%{菩萨只应该在这个法中了解方便法门吗？还要在其他法中了解方便法门吗？}%

@{答曰：}@
%{答：}%

@{是諸法中應善知方便，亦於餘法善知方便。}@
%{在一切法中都应该了解方便法门，也应该在其他法中了解方便法门。}%

@{問曰：}@
%{问：}%

@{若爾者，可略說。}@
%{如果是这样，可以简单地说一下。}%

@{答曰：}@
%{答：}%

@{有法能助地，有法違於地，}@
%{有一些法能够帮助地，有一些法违背地，}%

@{有法能生地，有法能壞地，}@
%{有一些法能够生起地，有一些法能够破坏地，}%

@{有諸地相果，有諸地中得，}@
%{有一些地的相和果位，有一些地在其中得到，}%

@{諸地清淨分，從地至一地，}@
%{每一地的清净分位，从一个地到另一个地，}%

@{住地轉增益，無能令退者，}@
%{住在每一地，不断地增长利益，没有能够使人退转的，}%

@{從菩薩淨地，至無量佛地，}@
%{从菩萨的净地，到无量佛地，}%

@{於此諸事中，應善知方便，}@
%{在这些事情中，应该了解方便法门，}%

@{請問諸善人，除破於憍慢。}@
%{请问诸位善人，去除掉我慢。}%

@{助初地法者：所謂信、戒、聞、捨、精進、念、慧等，如是等及餘諸法隨順初地者，是名助法。}@
%{帮助初地的法，就是说信、戒、闻、舍、精进、念、慧等等，像这些以及其他一切顺应初地的法，叫做助法。}%

@{相違法者：不信、破戒、少聞、慳、貪、懈怠、亂念、無慧等，及餘不隨順，不能助初地者是。}@
%{违背初地的法，就是不信、破戒、少闻、悭贪、懈怠、乱念、无慧等等，以及其他不顺应，不能帮助初地的法。}%
@{滅地法者：能令此地退失、障礙、不現，如劫盡時，萬物都滅。何者是？}@
%{灭地法，就是能让人退失、受阻、无法显现，就像劫难结束的时候，所有东西都灭绝了。这指的是什么？}%

@{所謂能偷奪菩提心法是，先已說。}@
%{所谓能偷夺菩提心法的，之前已经说过了。}%

@{生地法者：能生、能成初地。所謂不偷奪菩提心法是，先已說。}@
%{生地法，就是能生、能成就初地。所谓不偷夺菩提心法的，之前已经说过了。}%

@{地相、得、果、地分上已說。}@
%{地相、得、果、地分，之前都已经说过了。}%

@{清淨法者：用是法，能淨初地。所謂如先說}@
%{清净法，就是用这个方法，能净化初地。所谓如之前所说}%

@{初地中七法。}@
%{初地中的七法。}%

@{菩薩在初地，多所能堪受，}@
%{菩萨在初地，能承受很多东西，}%

@{不好於諍訟，其心多喜悅。}@
%{不喜欢争论，内心充满喜悦。}%

@{常樂於清淨，悲心愍眾生，}@
%{经常乐于清净，悲心怜悯众生，}%

@{無有瞋恚心，多行是七事。}@
%{没有瞋恚之心，经常做这七件事。}%

@{如是七法，能淨治初地。}@
%{这七法，能净化初地。}%

@{從一地至一地者：如從初地至二地，從二地至三地，餘亦如是。}@
%{从一个地到另一个地，就像从初地到二地，从二地到三地，其他的也是这样。}%

@{從初地至二地，得不諂曲等十心故。}@
%{从初地到二地，是因为获得了不谄曲等十种心。}%

@{從二地至三地，得信樂等十心故。}@
%{从二地到三地，是因为获得了信乐等十种心。}%

@{得如是等種種心、種種法故，能從一地至一地。}@
%{获得了这样种种心、种种法，就能从一个地到另一个地。}%

@{住地轉增益者：如初地中檀波羅蜜多。}@
%{住在地位上不断增益，就像初地中的檀波罗蜜多。}%

@{第二地中尸波羅蜜多，又信等諸法，轉得勢力。}@
%{第二地中的尸波罗蜜多，还有信等诸法，不断增强力量。}%

@{第三地中多聞多，又布施、持戒、信等轉得勢力。}@
%{第三地中多闻多，还有布施、持戒、信等不断增强力量。}%

@{餘地中亦如是。}@
%{其他的地也是这样。}%

@{無能令退者：住是地中，若沙門、婆羅門，若天、魔、梵及餘世間無能轉者。}@
%{没有人能够让人退失：住在这个地中，无论是沙门、婆罗门，还是天、魔、梵，以及其他世间的人，都没有办法让人退失。}%

@{何以故？}@
%{为什么呢？}%

@{得大功德力故，深入法性底故，大信解故。}@
%{因为获得了大功德力，深入法性底蕴，有了大信解。}%

@{從菩薩淨地至無量佛地者：若菩薩具足清淨一切地已，則得佛地。}@
%{从菩萨净地到无量佛地：如果菩萨具足清净一切地，就能获得佛地。}%

@{於此諸事中，皆應善知方便。}@
%{在这些事情中，都要善于知道方便法门。}%

@{請問諸善人者：成就正法故，名為善人。}@
%{请问诸善人：成就正法，才叫做善人。}%

@{正法者，略說：一、信，二、精進，三、念，四、定，五、慧，六、身、口、意律儀，七、無貪、無恚、無癡。}@
%{正法，简单来说：一、信，二、精进，三、念，四、定，五、慧，六、身、口、意律仪，七、无贪、无恚、无痴。}%

@{除捨於憍慢者：自謂我於勝人中勝，名為大慢。}@
%{除了要舍弃憍慢：自认为我在胜人中是最胜的，叫做大慢。}%

@{於與己等中勝，而心自高，名為憍慢。}@
%{在与自己等同的人中胜过，而内心自高，叫做憍慢。}%

@{大不如他，言小不如，名為小慢。}@
%{大大不如别人，说自己比别人差一点，叫做小慢。}%

@{問曰：}@
%{问曰：}%

@{汝說：於是諸法中應善知方便，得是方便，何用為？}@
%{你说：在这些法中应该善于知道方便，获得了这种方便，有什么用呢？}%

@{答曰：}@
%{答曰：}%

@{菩薩若善知，諸地中相得，}@
%{菩萨如果善于知道，诸地中的相得，}%

@{不得成佛道，終不轉初地。}@
%{得不到成佛之道，最终不会从初地转到其他地。}%

@{相名：助諸地等七法。}@
%{相名：帮助诸地等的七法。}%

@{得名：相違法有八種，滅等八法不應行。}@
%{得名：相违法有八种，灭等八法不应该做。}%

@{若菩薩善知是法，不得佛道，終不退轉。}@
%{如果菩萨善于知道这些法，得不到佛道，最终不会退转。}%
