@{西天譯經三藏朝散大夫試光祿卿明教大師臣法賢奉詔譯}@
%{西天译经三藏朝散大夫试光禄卿明教大师臣法贤奉诏译}%

@{行品第一}@
%{行品第一}%

@{爾時世尊，為令四眾各得歡喜，說是般若波羅蜜經使獲利樂。即說伽陀曰：}@
%{当时世尊，为了让四众各得欢喜，就说这部般若波罗蜜经，使他们获得利益快乐。于是就说了伽陀说：}%

@{所有菩薩為世間，滅除蓋障煩惱垢，}@
%{所有菩萨为了世间，灭除盖障烦恼垢，}%

@{發淨信心住寂靜，當行智度彼岸行。}@
%{发净信心住寂静，应当行智度彼岸行。}%

@{諸江河流閻浮提，華果藥草皆得潤，}@
%{诸江河流阎浮提，花果药草皆得润，}%

@{龍王主住無熱池，彼龍威力流江河。}@
%{龙王主住无热池，彼龙威力流江河。}%

@{亦如佛子聲聞等，說法教他方便說，}@
%{亦如佛子声闻等，说法教他方便说，}%

@{樂最聖行求果報，此諸如來勝威德。}@
%{乐最圣行求果报，此诸如来胜威德。}%

@{云何佛說此法眼，令諸弟子如佛學，}@
%{云何佛说此法眼，令诸弟子如佛学，}%

@{自證教他及方便，此亦佛力非自力。}@
%{自证教他及方便，此亦佛力非自力。}%

@{最上般若不可知，非心可知非菩提。}@
%{最上般若不可知，非心可知非菩提。}%

@{如是聞已不驚怖，彼菩薩行知佛智。}@
%{如是闻已不惊怖，彼菩萨行知佛智。}%

@{色受想行識皆無，不著纖塵無處所，}@
%{色受想行识皆无，不著纤尘无处所，}%

@{彼若不住一切法，行無受想得菩提。}@
%{彼若不住一切法，行无受想得菩提。}%

@{菩薩若求出家智，照見五蘊無實相，}@
%{菩萨若求出家智，照见五蕴无实相，}%

@{知此不求於寂靜，彼是菩薩之行智。}@
%{知此不求于寂静，彼是菩萨之行智。}%

@{復次云何智所得？照見一切法皆空，}@
%{复次云何智所得？照见一切法皆空，}%

@{不著不驚照見時，自覺覺他諸菩薩。}@
%{不著不惊照见时，自觉觉他诸菩萨。}%

@{色受想行及識蘊，是蘊見行而不知，}@
%{色受想行及识蕴，是蕴见行而不知，}%

@{菩薩照見蘊皆空，行無相化不著句。}@
%{菩萨照见蕴皆空，行无相化不著句。}%

@{無色受想行識等，不行是名無相行，}@
%{无色受想行识等，不行是名无相行，}%

@{若行不得最上智，無相寂靜三摩地。}@
%{若行不得最上智，无相寂静三摩地。}%

@{若菩薩行自寂靜，過去諸佛咸授記，}@
%{若菩萨行自寂静，过去诸佛咸授记，}%

@{身苦樂等皆不及，由知因果法本性。}@
%{身苦乐等皆不及，由知因果法本性。}%

@{若行於法不可得，行如是行乃佛智，}@
%{若行于法不可得，行如是行乃佛智，}%

@{行無所行了知已，是行最上般若行。}@
%{行无所行了知已，是行最上般若行。}%

@{彼無所有不可得，愚癡著相謂有無，}@
%{彼无所有不可得，愚痴著相谓有無，}%

@{有無二法皆非實，出此了知乃菩薩。}@
%{有無二法皆非實，出此了知乃菩薩。}%

@{菩薩若知諸幻化，色受想行識亦然，}@
%{菩萨若知诸幻化，色受想行识亦然，}%

@{寂靜行離種種相，此名最上般若行。}@
%{寂静行离种种相，此名最上般若行。}%

@{善友方便令知覺，使聞佛母不驚怖，}@
%{善友方便令知觉，使闻佛母不惊怖，}%

@{惡友同行及化他，坏器盛水非堅牢。}@
%{恶友同行及化他，坏器盛水非坚牢。}%

@{云何得名為菩薩？一切樂行皆無著，}@
%{云何得名为菩萨？一切乐行皆无著，}%

@{求佛菩提無所著，是故得名為菩薩。}@
%{求佛菩提无所著，是故得名为菩萨。}%

@{云何得名摩訶薩？得第一義眾生中，}@
%{云何得名摩诃萨？得第一义众生中，}%

@{斷眾生界諸邪見，是故得名摩訶薩。}@
%{断众生界诸邪见，是故得名摩诃萨。}%

@{大施大慧大威德，佛乘最上而得乘，}@
%{大施大慧大威德，佛乘最上而得乘，}%

@{發菩提心度眾生，是故得名摩訶薩。}@
%{发菩提心度众生，是故得名摩诃萨。}%

@{幻化四足俱胝數，多人眾前悉截首，}@
%{幻化四足俱胝数，多人众前悉截首，}%

@{一切世界皆幻化，菩薩知已得無怖。}@
%{一切世界皆幻化，菩萨知已得无怖。}%

@{色受想行識纏縛，知不實已不求解，}@
%{色受想行识缠缚，知不实已不求解，}%

@{行菩提心無所著，此名最上諸菩薩。}@
%{行菩提心无所著，此名最上诸菩萨。}%

@{云何得名為菩薩？乘大乘行度眾生，}@
%{云何得名为菩萨？乘大乘行度众生，}%

@{大乘體相如虛空，菩薩由得安隱樂。}@
%{大乘体相如虚空，菩萨由得安隐乐。}%

@{大乘之乘不可得，乘涅盤往諸方所，}@
%{大乘之乘不可得，乘涅槃往诸方所，}%

@{行已不見如火滅，是故名為入涅盤。}@
%{行已不见如火灭，是故名为入涅槃。}%

@{菩薩所行不可得，初後現在三清淨，}@
%{菩萨所行不可得，初后现在三清净，}%

@{清淨無畏無戲論，是行最上般若行。}@
%{清净无畏无戏论，是行最上般若行。}%

@{大智菩薩行行時，發大慈悲為眾生，}@
%{大智菩萨行行时，发大慈悲为众生，}%

@{為已不起眾生相，是行最上般若行。}@
%{为已不起众生相，是行最上般若行。}%

@{菩薩起念為眾生，修諸苦行有苦相，}@
%{菩萨起念为众生，修诸苦行有苦相，}%

@{是有我相眾生相，此非最上般若行。}@
%{是有我相众生相，此非最上般若行。}%

@{知自及諸眾生等，乃至諸法亦復然，}@
%{知自及诸众生等，乃至诸法亦复然，}%

@{生滅無二無分別，是行最上般若行。}@
%{生灭无二无分别，是行最上般若行。}%

@{乃至所說世界等，名離一切生滅法，}@
%{乃至所说世界等，名离一切生灭法，}%

@{最上無比甘露智，是故得名為般若。}@
%{最上无比甘露智，是故得名为般若。}%

@{菩薩如是所行行，了知方便無所求，}@
%{菩萨如是所行行，了知方便无所求，}%

@{知此法本性非實，是行最上般若行。}@
%{知此法本性非实，是行最上般若行。}%

@{若不住色亦無受，亦不住想亦無行，}@
%{若不住色亦无受，亦不住想亦无行，}%

@{復不住識住正法，是名最上般若行。}@
%{复不住识住正法，是名最上般若行。}%

@{佛母寶德藏般若伽陀帝釋品第二}@
%{佛母宝德藏般若伽陀帝释品第二}%

@{歡喜地攝布施波羅蜜伽陀}@
%{欢喜地摄布施波罗蜜伽陀}%

@{常與無常苦樂等，我及無我悉皆空，}@
%{常与无常苦乐等，我及无我悉皆空，}%

@{不住有為及無為，住無相行佛亦然。}@
%{不住有为及无为，住无相行佛亦然。}%

@{若求聲聞緣覺等，乃至佛果亦復然，}@
%{若求声闻缘觉等，乃至佛果亦复然，}%

@{不住此忍不可得，如渡大河不見岸。}@
%{不住此忍不可得，如渡大河不见岸。}%

@{若聞此法彼定得，成等正覺證涅盤，}@
%{若闻此法彼定得，成等正觉证涅槃，}%

@{見於一切如自身，是大智者如來說。}@
%{见于一切如自身，是大智者如来说。}%
@{佛子當住四補特伽羅，是行大智行：一真實善法；二不退心；三應供離垢無煩惱無求；四善友同等。}@
%{佛弟子应该常住于这四种补特伽罗，这是行持大智慧的行为：第一是真实的善法；第二是不退转心；第三是应供离垢，没有烦恼，没有所求；第四是善友，彼此平等。}%

@{大智菩薩如是行，不學聲聞及緣覺，}@
%{大智慧的菩萨这样行持，不学习声闻和缘觉的果位，}%

@{樂學如來一切智，是學非學名為學。}@
%{乐于学习如来的无上智慧，这种学习虽然不是真正的学习，却可以被称为学习。}%

@{學不受色不增減，亦復不學種種法，}@
%{学习不执着于色法，不增不减，也不学习各种法，}%

@{攝受樂學一切智，若此功德出離者。}@
%{只摄受乐于学习如来的无上智慧，如果能成就这种功德，就可以脱离轮回。}%

@{色非有智非無智，受想行識亦復爾，}@
%{色法并非有智慧，也并非无智慧，受、想、行、识也是如此，}%

@{色性自性如虛空，平等無二無分別。}@
%{色法的本质，其自性如同虚空，平等无二，没有分别。}%

@{妄想本性無彼岸，眾生之界亦復然，}@
%{妄想的本性没有彼岸，众生的世界也是如此，}%

@{虛空自性亦同然，智慧世間解亦爾。}@
%{虚空的本质也是如此，智慧的世间解也是如此。}%

@{智慧無色佛所說，離一切想到彼岸，}@
%{智慧无色，是佛陀所说，远离一切念头，到达彼岸，}%

@{若人得離諸想已，是人語意住真如。}@
%{如果有人能够远离一切念头，这个人言语和意念就住于真如。}%

@{彼人住世恒沙劫，不聞佛說眾生聲，}@
%{这个人住世恒沙劫，不听到佛陀说众生的声音，}%

@{眾生不生本清淨，是行最上般若行。}@
%{因为众生本来就清净，不生不灭，这是行持最上般若的行为。}%

@{佛說種種之語言，皆具最上般若義，}@
%{佛陀所说种种语言，都包含着最上般若的意义，}%

@{過去佛為我受記，於未來世證菩提。}@
%{过去佛为我授记，将在未来世证得菩提。}%

@{佛母寶德藏般若伽陀持無量功德建塔品第三}@
%{佛母宝德藏般若伽陀持无量功德建塔品第三}%

@{無垢地攝持戒波羅蜜伽陀}@
%{无垢地摄持戒波罗蜜伽陀}%

@{若人常受持般若，所作上應諸佛行，}@
%{如果有人常持般若，他的所作所为，都符合诸佛的行为，}%

@{刀劍毒藥水火等，乃至諸魔不能為。}@
%{刀剑、毒药、水火等等，乃至诸魔都不能伤害他。}%

@{若人於佛滅度後，建七寶塔以供養，}@
%{如果有人在佛陀灭度之后，建造七宝塔来供养，}%

@{如是圓滿千俱胝，佛剎恒沙等佛塔。}@
%{像这样圆满千俱胝佛刹恒沙等佛塔，}%

@{眾生無邊千俱胝，以妙香華塗香等，}@
%{无边无际的众生，用妙香花、涂香等等，}%

@{供養三世無邊劫，所有功德之數量，}@
%{供养三世无边劫，所有功德的数量，}%

@{不及書寫於佛母，諸佛由此而得生，}@
%{都比不上书写佛母宝德藏般若伽陀，诸佛由此而得生，}%

@{若受持讀誦供養，功德倍勝於佛塔。}@
%{如果受持、读诵、供养，功德比供养佛塔还要胜妙。}%

@{大明般若諸佛母，能除苦惱徧世界，}@
%{大明般若诸佛母，能够消除苦恼，遍及世界，}%

@{所有三世十方佛，學此明得無上師。}@
%{所有三世十方佛，学习这种明智，就能获得无上师。}%

@{行般若行利有情，使學大智證菩提，}@
%{行持般若，利益有情，使他们学习大智慧，证得菩提，}%

@{有為無為諸快樂，一切樂從般若生。}@
%{有为无为的一切快乐，都从般若而生。}%

@{譬如大地植諸種，得和合生種種色，}@
%{譬如大地种植各种种子，就能和合而生各种颜色，}%

@{五波羅蜜及菩提，皆從般若所生出。}@
%{五波罗蜜和菩提，都是从般若而生出来。}%

@{又如輪王出行時，七寶四兵為導從，}@
%{又如轮王出行的时候，七宝和四兵作为导从，}%

@{若依佛母最上行，一切功德法集聚。}@
%{如果依佛母最上行，一切功德法都会集聚。}%

@{佛母寶德藏般若伽陀功德品第四}@
%{佛母宝德藏般若伽陀功德品第四}%

@{發光地攝忍辱波羅蜜伽陀}@
%{发光地摄忍辱波罗蜜伽陀}%

@{帝釋有疑問佛曰：恒河沙數等佛剎，}@
%{帝释天有疑问，佛陀说：恒河沙数等佛刹，}%

@{佛界圓滿如芥子，能受佛剎般若力，}@
%{佛界圆满如同芥子，能够承受佛刹的般若力量，}%

@{如是了知般若已，此界云何不供養？}@
%{既然已经这样了解般若，这个世界为什么不供养呢？}%

@{譬如人王人所重，住般若者合亦爾。}@
%{譬如人王是人所重视的，住持般若的人也应该如此。}%

@{佛界般若摩尼寶，具一切德價無比，}@
%{佛界的般若摩尼宝，具足一切功德，无与伦比，}%

@{經函安處經有無，供養悉獲寶功德。}@
%{经函安处，经书有没有，供养都能获得宝贵的功德。}%

@{佛滅供養於舍利，不及供養於般若，}@
%{佛陀灭度后，供养舍利，不如供养般若，}%

@{若樂受持供養者，是人速得證解脫。}@
%{如果乐于受持、供养，这个人就能快速证得解脱。}%

@{首行布施波羅蜜，次戒忍進及禪定，}@
%{首先是行持布施波罗蜜，然后是戒、忍、进和禅定，}%

@{受持善法不可壞，彼一一生一切法。}@
%{受持善法不可破坏，这在每一世，每一生的所有法中都是如此。}%

@{如閻浮提種種樹，百千俱胝無數色，}@
%{如同阎浮提各种各样的树木，百千俱胝，无量颜色，}%

@{雖一一樹影皆別，無量影同一名攝。}@
%{虽然每一棵树的影子都不同，但是无量的影子，都归于同一个名称。}%

@{五波羅蜜五名異，般若波羅復一名，}@
%{五波罗蜜有五个不同的名称，般若波罗蜜又有一个名称，}%

@{一切迴施為菩提，一味同歸菩提名。}@
%{一切回施都是为了菩提，最终都归于菩提的名称。}%

@{佛母寶德藏般若伽陀福量品第五}@
%{佛母宝德藏般若伽陀福量品第五}%

@{焰慧地攝精進波羅蜜伽陀}@
%{焰慧地摄精进波罗蜜伽陀}%

@{彼色受想行識等，菩薩觀照悉無常，}@
%{色、受、想、行、识等等，菩萨观照，都无常，}%

@{各各現行而不知，非法非生智者見。}@
%{各自显现，却不知无常，这是没有智慧的人所见。}%

@{無色無受想行識，是法無得復無生，}@
%{没有色法，没有受、想、行、识，这是法无所得，也无生，}%

@{了知一切法皆空，是名最上般若行。}@
%{了解一切法都是空，这就是最上般若的行为。}%

@{如化恒沙等佛剎，諸眾生證羅漢果，}@
%{如同化现恒沙等佛刹，所有众生都证得罗汉果，}%

@{若能書寫此般若，令他受持功德勝。}@
%{如果能够书写这种般若，让别人受持，功德非常殊胜。}%

@{如佛修行云何學？信重般若諸法空，}@
%{佛陀修行如何学习呢？就是信重般若，诸法空，}%

@{速證聲聞及緣覺，乃至無上正覺尊。}@
%{快速证得声闻和缘觉，乃至无上正等觉。}%

@{世間無種不生樹，枝葉華果悉無有，}@
%{世间没有种子不生树，枝叶、花果，也都无有，}%

@{無佛誰指菩提心，亦無釋梵聲聞果。}@
%{没有佛陀，谁来指点菩提心，也没有释梵声闻的果位。}%
