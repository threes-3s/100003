@{如人有德力最勝，善解一切幻化法，}@
%{如人有德力最胜，善解一切幻化法，}%

@{乃至器仗及工巧，而能一向為世間。}@
%{乃至器仗及工巧，而能一向为世间。}%

@{彼人父母妻及子，遊行遠路多冤中，}@
%{彼人父母妻及子，游行远路多冤中，}%

@{是人勇猛眾所知，安樂還家無怖畏。}@
%{是人勇猛众所知，安乐还家无怖畏。}%

@{大智菩薩為眾生，安住第一三摩地，}@
%{大智菩萨为众生，安住第一三摩地，}%

@{降伏四魔離二乘，亦復不求佛菩提。}@
%{降伏四魔离二乘，亦复不求佛菩提。}%

@{譬如虛空無所有，風水大地皆依住，}@
%{譬如虚空无所有，风水大地皆依住，}%

@{世間眾生得快樂，虛空無意住非住。}@
%{世间众生得快乐，虚空无意住非住。}%

@{菩薩住空亦如是，現於世間種種相，}@
%{菩萨住空亦如是，现于世间种种相，}%

@{以眾生智及願力，非彼寂靜非空故。}@
%{以众生智及愿力，非彼寂静非空故。}%

@{若菩薩行大智時，住空寂靜三摩地，}@
%{若菩萨行大智时，住空寂静三摩地，}%

@{此中不見一切相，亦復不見彼非相。}@
%{此中不见一切相，亦复不见彼非相。}%

@{菩薩行此解脫門，非求寂靜非行相，}@
%{菩萨行此解脱门，非求寂静非行相，}%

@{如鳥飛空而往來，非住虛空非住地。}@
%{如鸟飞空而往来，非住虚空非住地。}%

@{亦如有人習射法，習之不住經多歲，}@
%{亦如有人习射法，习之不住经多岁，}%

@{射法久習得盡妙，一一箭發無不中。}@
%{射法久习得尽妙，一一箭发无不中。}%

@{最上般若行亦爾，修習智慧及方便，}@
%{最上般若行亦尔，修习智慧及方便，}%

@{直至眾善悉圓滿，方獲最上神通力。}@
%{直至众善悉圆满，方获最上神通力。}%

@{若苾芻證神通力，現神變化住虛空，}@
%{若苾芻证神通力，现神变化住虚空，}%

@{行住坐臥四威儀，經俱胝劫不退倦。}@
%{行住坐卧四威仪，经俱胝劫不退倦。}%

@{住空菩薩亦如是，修無相行到彼岸，}@
%{住空菩萨亦如是，修无相行到彼岸，}%

@{行種種行現世間，經俱胝劫不退倦。}@
%{行种种行现世间，经俱胝劫不退倦。}%

@{如人經險遇大風，二手持蓋心專注，}@
%{如人经险遇大风，二手持盖心专注，}%

@{是人怖險不能行，直至無風乃前進。}@
%{是人怖险不能行，直至无风乃前进。}%

@{大智菩薩住大悲，智慧方便為二手，}@
%{大智菩萨住大悲，智慧方便为二手，}%

@{執空無相願法蓋，見法不住於寂靜。}@
%{执空无相愿法盖，见法不住于寂静。}%

@{如人求寶往寶洲，獲寶安隱而還家，}@
%{如人求宝往宝洲，获宝安隐而还家，}%

@{是人心足而快樂，豈有眷屬心苦惱。}@
%{是人心足而快乐，岂有眷属心苦恼。}%

@{詣空寶洲亦如是，獲得根力禪定寶，}@
%{詣空宝洲亦如是，获得根力禅定宝，}%

@{菩薩不住歡喜心，令諸眾生離苦惱。}@
%{菩萨不住欢喜心，令诸众生离苦恼。}%

@{商人為利悉所經，聚落國城諸里巷，}@
%{商人为利悉所经，聚落国城诸里巷，}%

@{雖達寶所亦非住，大智善道而復還。}@
%{虽达宝所亦非住，大智善道而复还。}%

@{大智菩薩悉了知，聲聞緣覺解脫智，}@
%{大智菩萨悉了知，声闻缘觉解脱智，}%

@{乃至佛智亦非住，何況行彼有為道。}@
%{乃至佛智亦非住，何况行彼有为道。}%

@{大智菩薩為世間，住空無相願三昧，}@
%{大智菩萨为世间，住空无相愿三昧，}%

@{若得寂靜無所著，乃可得知於無為。}@
%{若得寂静无所著，乃可得知于无为。}%

@{譬如人生人未識，稱其名故眾乃知，}@
%{譬如人生人未识，称其名故众乃知，}%

@{菩薩若行解脫門，於解脫門眾知識。}@
%{菩萨若行解脱门，于解脱门众知识。}%

@{菩薩聞彼甚深法，而於諸根悉照明，}@
%{菩萨闻彼甚深法，而于诸根悉照明，}%

@{住空無相無願法，無退無思無授記。}@
%{住空无相无愿法，无退无思无授记。}%

@{觀於三界如夢幻，不求聲聞緣覺地，}@
%{观于三界如梦幻，不求声闻缘觉地，}%

@{如佛說法為世間，名不退地應授記。}@
%{如佛说法为世间，名不退地应授记。}%

@{知諸眾生墮三塗，發願剎那滅惡道，}@
%{知诸众生堕三涂，发愿刹那灭恶道，}%

@{以真實力滅火蘊，名不退地應授記。}@
%{以真实力灭火蕴，名不退地应授记。}%

@{諸惡宿曜及鬼神，作種種疫惱世間，}@
%{诸恶宿曜及鬼神，作种种疫恼世间，}%

@{真實願力悉滅除，無我能作應授記。}@
%{真实愿力悉灭除，无我能作应授记。}%
@{佛母寶德藏般若伽陀魔業品第二十一}@
%{佛母宝德藏般若伽陀魔业品第二十一}%

@{我得授記非能所，是實願力得增長，}@
%{我得到授记，这并非是我自己的能力，而是实实在在的愿力在增长，}%

@{若見授記及能所，是名執著及少智。}@
%{如果执着于得到授记以及自己的能力，这就是执着和愚痴的表现。}%

@{菩薩有執魔即知，現親友相來嬈惱，}@
%{菩萨如果执着于魔，就会知道，魔会以亲友的相貌出现，来迷惑和烦恼你，}%

@{或作父母七代人，言汝名此佛可證。}@
%{或者化身为你的父母，七代祖先，对你说，你将来可以成佛。}%

@{魔所現作無數相，皆云：愍汝作利樂。}@
%{魔会以无数种相貌出现，都说：我可怜你，是来帮助你的。}%

@{菩薩聞已有所忻，是名少智魔所著。}@
%{菩萨听了之后，心里很高兴，这就是愚痴，被魔所迷惑。}%

@{或住城隍及聚落，山林曠野寂靜處，}@
%{魔可能会出现在城隍庙、村庄，或者山林旷野的寂静之处，}%

@{自稱己德毀他人，應知少智為魔作。}@
%{自夸自己的功德，贬低别人，要知道，这是愚痴，是魔在作祟。}%

@{雖住城隍聚落中，不求聲聞緣覺證，}@
%{即使住在城隍庙、村庄，也不追求声闻、缘觉的果位，}%

@{此心為度眾生故，我說是名為菩薩。}@
%{而是为了度化众生，这才叫做菩萨。}%

@{五百由旬山險深，共諸惡獸多年住，}@
%{在五百由旬的山里，那里环境险恶，和各种野兽一起生活了很多年，}%

@{若見逼迫著我慢，若無分別知菩薩。}@
%{如果遇到逼迫，就生起我慢，那就不能分辨什么是真正的菩萨。}%

@{菩薩住彼為世間，得力解脫三摩地，}@
%{菩萨住在那里，是为了度化世间，获得解脱的禅定，}%

@{彼著山野寂靜行，此亦知彼魔所作。}@
%{如果执着于山野的寂静生活，这也是魔的作为。}%

@{雖住城隍及山野，樂佛菩提離二乘，}@
%{即使住在城隍庙、山野，也乐于佛的菩提，远离二乘，}%

@{修如是行利世間，一念如秤名菩薩。}@
%{修持这样的行为，利益世间，心念如同秤一样，这就是菩萨。}%

@{佛母寶德藏般若伽陀善友品第二十二}@
%{佛母宝德藏般若伽陀善友品第二十二}%

@{有大智者依師學，速疾得證無上覺，}@
%{有智慧的人，依靠老师学习，就能很快证得无上觉悟，}%

@{亦如良醫除眾患，學從善友心無疑。}@
%{就像良医能消除各种病患一样，学习善友的教诲，心无疑惑。}%

@{菩薩行佛菩提行，依彼善友波羅蜜，}@
%{菩萨修持佛的菩提行，依靠善友的波罗蜜，}%

@{此最上地能調伏，為二種事證菩提。}@
%{这是最上乘的修行方法，能调伏身心，成就两种事，证得菩提。}%

@{過去未來十方佛，行此正道無異路，}@
%{过去、未来、十方诸佛，都走这条正道，没有其他道路，}%

@{行佛菩提最上行，說波羅蜜如電光。}@
%{修持佛的菩提，这是最上乘的修行，说波罗蜜就像闪电一样，非常快速。}%

@{如彼般若空無相，知諸法相亦如是，}@
%{就像般若空无相，要了解一切法相也是如此，}%

@{了知一切法皆空，此即名行佛般若。}@
%{明白一切法都是空的，这就是修持佛的般若。}%

@{繫著色欲及飲食，常在輪迴不休息，}@
%{执着于色欲和饮食，就会永远轮回，永无止息，}%

@{此愚迷人所見倒，於不實法生實想。}@
%{这是愚痴的人，颠倒的见解，对虚假的事物产生真实的念头。}%

@{譬如得食疑有毒，以虛妄見而不食，}@
%{就像得到食物，怀疑有毒，因为虚妄的见解而不吃，}%

@{愚人妄心生我想，以我想故有生死。}@
%{愚人妄心生起我执，因为我执，所以有生死。}%

@{亦如恒說諸煩惱，於諸煩惱不著相，}@
%{就像一直说各种烦恼，但对烦恼不执着，}%

@{煩惱清淨俱無有，如是菩薩知般若。}@
%{烦恼清净，全部都没有，菩萨就是这样了解般若。}%

@{如閻浮提諸眾生，皆發無上菩提心，}@
%{就像阎浮提的众生，都发起了无上菩提心，}%

@{多千俱胝劫布施，迴施一切證菩提。}@
%{经过无数劫的布施，回向一切众生，才能证得菩提。}%

@{若復有人於一日，奉行最上般若行，}@
%{如果有人在一天之中，奉行最上乘的般若修行，}%

@{千俱胝施不及一，行般若功無為故。}@
%{胜过千俱胝劫的布施，因为修持般若的功德，是无为的。}%

@{菩薩大悲行般若，度眾生故不起想，}@
%{菩萨以大悲心修持般若，为了度化众生，不生起任何念头，}%

@{恒行乞食於國城，是得一切名大智。}@
%{经常在城里乞食，这就是获得一切智慧。}%

@{菩薩欲度於人天，乃至三塗極苦眾，}@
%{菩萨想要度化人天众生，乃至三恶道极苦的众生，}%

@{皆令速到於彼岸，晝夜勤行於般若。}@
%{让所有众生都能快速到达彼岸，昼夜精进地修持般若。}%

@{如人欲求無價寶，必過大海諸險難，}@
%{就像人想要寻找无价之宝，必须经过大海的各种险阻，}%

@{無心忽爾而獲得，憂惱皆除喜無量。}@
%{无心无意间就得到了，忧愁烦恼都消除，喜悦无量。}%

@{求菩提寶亦如是，勤行般若諸功德，}@
%{求取菩提宝也是一样，勤奋地修持般若的功德，}%

@{得無取捨無上寶，菩薩速證於菩提。}@
%{就能获得无取舍的无上宝，菩萨就能快速证得菩提。}%
