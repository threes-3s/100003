@{姚秦罽賓三藏曇摩耶舍共曇摩崛多等譯}@
%{姚秦罽宾三藏昙摩耶舍共昙摩崛多等译}%

@{非問分緣品第五}@
%{非问分缘品第五}%

@{善緣方便善緣解。有緣方便聖忍非智，有緣方便聖智非忍，有緣方便受問答，因俱生法，若因此有此、若無因無此、若此生有此生、若此滅有此滅。}@
%{善缘方便善缘解。有缘方便圣忍非智，有缘方便圣智非忍，有缘方便受问答，因俱生法，若因此有此，若无因无此，若此生有此生，若此灭有此灭。}%

@{若無明緣行乃至生緣老死，憂悲苦惱苦聚成就，如是純苦具足。}@
%{若无明缘行乃至生缘老死，忧悲苦恼苦聚成就，如是纯苦具足。}%

@{無明滅則行滅，乃至生滅則老死憂悲苦惱聚滅。}@
%{无明灭则行灭，乃至生灭则老死忧悲苦恼聚灭。}%

@{如是苦聚滅，是緣方便成就。}@
%{如是苦聚灭，是缘方便成就。}%

@{若彼於過去緣疑惑：我過去有、我非過去有？何姓過去有、何因過去有？}@
%{若彼于过去缘疑惑：我过去有，我非过去有？何姓过去有，何因过去有？}%

@{若於未來緣疑惑：我未來有，乃至何因未來有？}@
%{若于未来缘疑惑：我未来有，乃至何因未来有？}%

@{若彼因緣疑惑：我云何有、我云何非有？何因有、何生處？此眾生從何來、去至何處？}@
%{若彼因缘疑惑：我云何有，我云何非有？何因有，何生处？此众生从何来，去至何处？}%

@{若於佛疑惑：是佛世尊、非佛世尊？世尊善說法、世尊不善說法？世尊聲聞眾善趣、世尊聲聞眾不善趣？}@
%{若于佛疑惑：是佛世尊，非佛世尊？世尊善说法，世尊不善说法？世尊声闻众善趣，世尊声闻众不善趣？}%

@{行常行、無常行？苦行、非苦行？我法、非我法？寂靜涅槃、非寂靜涅槃？}@
%{行常行，无常行？苦行，非苦行？我法，非我法？寂静涅槃，非寂静涅槃？}%

@{有與、無與？有施、無施？有祀、無祀？有善惡業果報、無善惡業果報？}@
%{有与，无与？有施，无施？有祀，无祀？有善恶业果报，无善恶业果报？}%

@{有今世、無今世？有後世、無後世？有父母、無父母？有天、無天？}@
%{有今世，无今世？有后世，无后世？有父母，无父母？有天，无天？}%

@{眾生有化生、眾生無化生？世有沙門婆羅門正趣正至若今世後世自證知說、世無沙門婆羅門正趣正至若今世後世自證知說？}@
%{众生有化生，众生无化生？世有沙门婆罗门正趣正至若今世后世自证知说，世无沙门婆罗门正趣正至若今世后世自证知说？}%

@{若於法疑惑心不決定，猶豫二心疑心不了，無量疑不盡，非解脫。}@
%{若于法疑惑心不决定，犹豫二心疑心不了，无量疑不尽，非解脱。}%

@{彼時無有。}@
%{彼时无有。}%

@{若有沙門婆羅門異緣，實我世常，此實餘虛妄，乃至如去不如去。}@
%{若有沙门婆罗门异缘，实我世常，此实余虚妄，乃至如去不如去。}%

@{彼時亦無有。}@
%{彼时亦无有。}%

@{何況聖緣方便成就，終無此煩惱垢。}@
%{何况圣缘方便成就，终无此烦恼垢。}%

@{云何緣？}@
%{云何缘？}%

@{如佛告諸比丘：我當說緣、緣生法。}@
%{如佛告诸比丘：我当说缘，缘生法。}%

@{云何緣？無明緣行。}@
%{云何缘？无明缘行。}%

@{若諸佛出世若不出世，法住法界。}@
%{若诸佛出世若不出世，法住法界。}%

@{住彼法界，如來正覺正解已，演說開示分別顯現，說無明緣行乃至生緣老死。}@
%{住彼法界，如来正觉正解已，演说开示分别显现，说无明缘行乃至生缘老死。}%

@{若如此法，如爾非不如爾、不異不異物、常法實法法住法定，如是緣是名緣。}@
%{若如此法，如尔非不如尔，不异不异物，常法实法法住法定，如是缘是名缘。}%

@{云何緣生法？老死無常，有為緣生，盡法、變異法、離欲法、滅法。}@
%{云何缘生法？老死无常，有为缘生，尽法，变异法，离欲法，灭法。}%

@{乃至無明無常，有為緣生法，盡法、變異法、離欲法、滅法，是名緣生法。}@
%{乃至无明无常，有为缘生法，尽法，变异法，离欲法，灭法，是名缘生法。}%

@{云何緣方便？}@
%{云何缘方便？}%

@{若彼緣、若此緣生法，若見解射方便，是名緣方便。}@
%{若彼缘，若此缘生法，若见解射方便，是名缘方便。}%

@{比丘！齊幾名善緣方便？}@
%{比丘！齐几名善缘方便？}%

@{彼緣此緣生法，如實知、如實見，齊是名善緣方便。}@
%{彼缘此缘生法，如实知，如实见，齐是名善缘方便。}%

@{云何無明？癡不善根，是名無明。}@
%{云何无明？痴不善根，是名无明。}%

@{云何無明緣行？無明緣福行、非福行、不動行。}@
%{云何无明缘行？无明缘福行，非福行，不动行。}%

@{云何非福行？不善身行、不善口行、不善意行。}@
%{云何非福行？不善身行，不善口行，不善意行。}%

@{云何不善身行？}@
%{云何不善身行？}%

@{若人無慧無明不斷，行殺盜婬及餘不善身行，是名不善身行。}@
%{若人无慧无明不断，行杀盗婬及余不善身行，是名不善身行。}%

@{云何不善口行？}@
%{云何不善口行？}%

@{若人無慧無明未斷，行妄語、兩舌、惡口、綺語及餘不善口行，是名不善口行。}@
%{若人无慧无明未断，行妄语，两舌，恶口，绮语及余不善口行，是名不善口行。}%

@{云何不善意行？}@
%{云何不善意行？}%

@{若人無慧無明未斷，起貪欲、瞋恚、邪見，是名不善意行。}@
%{若人无慧无明未断，起贪欲，瞋恚，邪见，是名不善意行。}%

@{此身口意不善行，名非福行，無明緣現世行。}@
%{此身口意不善行，名非福行，无明缘现世行。}%

@{云何福行？身善行、口善行、意善行。}@
%{云何福行？身善行，口善行，意善行。}%

@{云何身善行？}@
%{云何身善行？}%

@{若人無慧無明未斷，不殺盜婬，及餘身善行，是名身善行。}@
%{若人无慧无明未断，不杀盗婬，及余身善行，是名身善行。}%

@{云何口善行？}@
%{云何口善行？}%

@{若人無慧無明未斷，不妄語、兩舌、惡口、綺語，及餘口善行，是名口善行。}@
%{若人无慧无明未断，不妄语，两舌，恶口，绮语，及余口善行，是名口善行。}%

@{云何意善行？}@
%{云何意善行？}%

@{若人無慧無明未斷，無貪無恚正見，是名意善行。}@
%{若人无慧无明未断，无贪无恚正见，是名意善行。}%

@{此身口意善行，是名福行無明緣現世行。}@
%{此身口意善行，是名福行无明缘现世行。}%

@{復次若人無慧無明未斷，離欲惡不善法有覺有觀離生喜樂成就初禪行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟。}@
%{复次若人无慧无明未断，离欲恶不善法有觉有观离生喜乐成就初禅行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟。}%

@{如是身口意善行，是名福行無明緣現世行。}@
%{如是身口意善行，是名福行无明缘现世行。}%

@{復次若人無慧無明未斷，滅覺觀內信心無覺無觀定生喜樂成就二禪行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，灭觉观内信心无觉无观定生喜乐成就二禅行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意業善行，是名福行無明緣現世行。}@
%{如是身口意业善行，是名福行无明缘现世行。}%

@{復次若人無慧無明未斷，離喜捨行念知身受樂如諸聖人解捨念樂行成就三禪行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，离喜舍行念知身受乐如诸圣人解舍念乐行成就三禅行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名福行無明緣現世行。}@
%{如是身口意善行，是名福行无明缘现世行。}%

@{復次若人無慧無明未斷，斷苦樂先滅憂喜不苦不樂捨念淨成就四禪行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，断苦乐先灭忧喜不苦不乐舍念净成就四禅行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名福行無明緣現世行。}@
%{如是身口意善行，是名福行无明缘现世行。}%

@{是名福行。}@
%{是名福行。}%

@{云何不動行？}@
%{云何不动行？}%

@{若人無慧無明未斷，離一切色想、滅瞋恚想、不思惟若干想，成就無邊空處行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{若人无慧无明未断，离一切色想，灭瞋恚想，不思惟若干想，成就无边空处行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名不動行無明緣現世行。}@
%{如是身口意善行，是名不动行无明缘现世行。}%

@{復次若人無慧無明未斷，離一切空處成就無邊識處行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，离一切空处成就无边识处行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名不動行無明緣現世行。}@
%{如是身口意善行，是名不动行无明缘现世行。}%

@{復次若人無慧無明未斷，離一切識處成就無所有處行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，离一切识处成就无所有处行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名不動行無明緣現世行。}@
%{如是身口意善行，是名不动行无明缘现世行。}%

@{復次若人無慧無明未斷，離一切無所有處成就非有想非無想處行，彼身業無教戒法入攝意識所知、口業無教戒法入攝意識所知、意業由意生受想思觸思惟，}@
%{复次若人无慧无明未断，离一切无所有处成就非有想非无想处行，彼身业无教戒法入摄意识所知，口业无教戒法入摄意识所知，意业由意生受想思触思惟，}%

@{如是身口意善行，是名不動行無明緣現世行。}@
%{如是身口意善行，是名不动行无明缘现世行。}%

@{是名不動行。}@
%{是名不动行。}%

@{復次若人無慧無明未斷，作不善身口意行，作不善行故身壞命終墮地獄畜生餓鬼，以因緒緣故墮地獄畜生餓鬼受五陰身。}@
%{复次若人无慧无明未断，作不善身口意行，作不善行故身坏命终堕地狱畜生饿鬼，以因绪缘故堕地狱畜生饿鬼受五阴身。}%

@{如是緣現世行受未來行，是名無明緣未來行。}@
%{如是缘现世行受未来行，是名无明缘未来行。}%

@{復次若人無慧無明未斷，作有漏身善行當受欲界生，作有漏口善行意善行當受欲界生，作善行已身壞命終若生人中欲界天上，以因緒緣故人中欲界天上受五陰身。}@
%{复次若人无慧无明未断，作有漏身善行当受欲界生，作有漏口善行意善行当受欲界生，作善行已身坏命终若生人中欲界天上，以因绪缘故人中欲界天上受五阴身。}%

@{如是緣現世行受未來行，是名無明緣未來行。}@
%{如是缘现世行受未来行，是名无明缘未来行。}%

@{復次若人無慧無明未斷，作有漏身善行當受色界生、作有漏口善行意善行當受色界生，作善行已身壞命終生色界天上；以因緒緣故，色界天上受五陰身。}@
%{复次若人无慧无明未断，作有漏身善行当受色界生，作有漏口善行意善行当受色界生，作善行已身坏命终生色界天上；以因绪缘故，色界天上受五阴身。}%

@{如是緣現世行受未來行，是名無明緣未來行。}@
%{如是缘现世行受未来行，是名无明缘未来行。}%

@{復次若人無慧無明未斷，作有漏身口意善行當受無色界生，作善行已身壞命終生無色界天上，以因緒緣故無色界天上受四陰身。}@
%{复次若人无慧无明未断，作有漏身口意善行当受无色界生，作善行已身坏命终生无色界天上，以因绪缘故无色界天上受四阴身。}%

@{如是緣現世行受未來行，是名無明緣未來行。}@
%{如是缘现世行受未来行，是名无明缘未来行。}%

@{如佛說：阿難！行有緣。}@
%{如佛说：阿难！行有缘。}%

@{如是阿難問已有答。}@
%{如是阿难问已有答。}%

@{行何緣？無明緣行。}@
%{行何缘？无明缘行。}%

@{此是答。}@
%{此是答。}%

@{阿難！若無無明者有行不？}@
%{阿难！若无无明者有行不？}%

@{世尊！無也。}@
%{世尊！无也。}%

@{阿難！以因緒緣故行，若無明緣行，如向所說。}@
%{阿难！以因绪缘故行，若无明缘行，如向所说。}%

@{以是故說。}@
%{以是故说。}%
@{云何行緣識？}@
%{什么是行为缘起的意识呢？}%

@{有緣共欲思生共欲識，如是緣現在行生現在識，名行緣現在識。}@
%{如果有共同的欲望，就会产生共同的欲望的意识，这样，因为现在的行为而产生的现在的意识，就叫做行为缘起的现在的意识。}%

@{有共瞋恚、有共愚癡，無共欲、無共瞋恚、無共愚癡，善不善有緣無記思，有生無記識，如是緣現在行生現在識，是名行緣現在識。}@
%{如果有共同的瞋恚，如果有共同的愚痴，没有共同的欲望，没有共同的瞋恚，没有共同的愚痴，善的、不善的，或者无记的，都有缘无记的思维，也会产生无记的意识，这样，因为现在的行为而产生的现在的意识，就叫做行为缘起的现在的意识。}%

@{緣眼緣色生識，彼眼行色行若緣識，如是緣現在行生現在識，是為現在行緣現在識。}@
%{因为眼识缘色而产生意识，眼识的行为和色识的行为，如果与意识有缘，这样，因为现在的行为而产生的现在的意识，就叫做现在的行为缘起的现在的意识。}%

@{耳鼻舌身緣意緣法生識，彼意行法行若緣識。如是緣現在行生現在識，是名現在行緣現在識。}@
%{因为耳识、鼻识、舌识、身识、意识缘法而产生意识，意识的行为和法识的行为，如果与意识有缘，这样，因为现在的行为而产生的现在的意识，就叫做现在的行为缘起的现在的意识。}%

@{復次若人無慧無明未斷，起不善身行不善口行不善意行，作不善行已身壞命終墮地獄畜生餓鬼，以因緒緣故生地獄畜生餓鬼初識，以業因緒集緣生眼識，乃至意識及後了識。如是緣現在行生未來識，是為行緣未來識。}@
%{再者，如果一个人没有智慧，无明没有断除，起了不善的身行、不善的口行、不善的意行，做了不善的行为之后，身体坏掉了，命终了，堕入地狱、畜生、饿鬼道，因为因缘的缘故，在地狱、畜生、饿鬼道中产生最初的意识，因为业因缘的集聚而产生眼识，乃至意识和后来的了别识。这样，因为现在的行为而产生的未来的意识，就叫做行为缘起的未来的意识。}%

@{復次若人無慧無明未斷，作有漏身善行當受欲界生、作有漏口意善行當受欲界生，作善行已身壞命終，若生人中若生欲界天上，以因緒緣故若生人中若生欲界六天初識，以業因緒集緣生眼識，乃至意識及後了識。如是緣現在行生未來識，是為行緣識。}@
%{再者，如果一个人没有智慧，无明没有断除，做了有漏的身善行，应该受欲界生，做了有漏的口意善行，应该受欲界生，做了善行之后，身体坏掉了，命终了，如果生为人中，或者生在欲界天，因为因缘的缘故，如果生为人中，或者生在欲界六天，产生最初的意识，因为业因缘的集聚而产生眼识，乃至意识和后来的了别识。这样，因为现在的行为而产生的未来的意识，就叫做行为缘起的意识。}%

@{復次若人無慧無明未斷，作有漏身善行當受欲界生、作有漏口意善行當受欲界生，作善行已身壞命終若生人中生欲界天上，由因緒緣故，若人中欲界天上有善思、共彼思識。如是緣未來行受未來識，是名行緣未來識。}@
%{再者，如果一个人没有智慧，无明没有断除，做了有漏的身善行，应该受欲界生，做了有漏的口意善行，应该受欲界生，做了善行之后，身体坏掉了，命终了，如果生为人中，生在欲界天，因为因缘的缘故，在人中、欲界天中会有善的思维，并与之相应的意识。这样，因为未来的行为而产生的未来的意识，就叫做行为缘起的未来的意识。}%

%{再者，如果一个人没有智慧，无明没有断除，做了有漏的身善行，应该受色界生，做了有漏的口意善行，应该受色界生，做了善行之后，身体坏掉了，命终了，生在色界天，因为因缘的缘故，在色界天中会有思维，并与之相应的意识。这样，因为未来的行为而产生的未来的意识，就叫做行为缘起的未来的意识。}%

@{復次若人無慧無明未斷，作有漏身善行當受無色界生、作有漏口意善行當受無色界生，作善行已身壞命終生無色界天上，以因緒緣故無色界天上有不動思共彼思識。如是緣未來行生未來識，是名行緣未來識。}@
%{再者，如果一个人没有智慧，无明没有断除，做了有漏的身善行，应该受无色界生，做了有漏的口意善行，应该受无色界生，做了善行之后，身体坏掉了，命终了，生在无色界天，因为因缘的缘故，在无色界天中会有不动思维，并与之相应的意识。这样，因为未来的行为而产生的未来的意识，就叫做行为缘起的未来的意识。}%

@{復次若最後行未知而滅、若無間行滅已識續餘道生，彼行彼緣無間緣。}@
%{再者，如果最后的行为没有灭尽，或者无间行为灭尽了，意识延续到余道而生，那个行为，那个缘，就是无间缘。}%

@{若因識續餘道生，彼行緣彼識因緣。}@
%{如果因为意识延续到余道而生，那个行为，那个缘，就是意识的因缘。}%

@{若思行彼識續餘道生，彼行緣彼識境界緣。}@
%{如果因为思维行为，意识延续到余道而生，那个行为，那个缘，就是意识的境界缘。}%

@{若彼行識續餘道生，彼行緣彼識依緣。}@
%{如果因为那个行为，意识延续到余道而生，那个行为，那个缘，就是意识的依缘。}%

@{若報行識續餘道生，彼行緣彼識報緣。}@
%{如果因为报应行为，意识延续到余道而生，那个行为，那个缘，就是意识的报缘。}%

@{若起行識續餘道生，彼行緣彼識起緣。}@
%{如果因为起因行为，意识延续到余道而生，那个行为，那个缘，就是意识的起缘。}%

@{若行相應識續餘道生，彼行緣彼識異緣。}@
%{如果因为行为相应的意识，延续到余道而生，那个行为，那个缘，就是意识的异缘。}%

@{若行增上識續餘道生，彼行緣彼識增上緣。}@
%{如果因为行为增上的意识，延续到余道而生，那个行为，那个缘，就是意识的增上缘。}%

@{此最後識滅，初識續餘道生，最後識滅已，初識即生，無有中間。}@
%{这个最后的意识灭尽了，最初的意识延续到余道而生，最后的意识灭尽了，最初的意识就产生了，没有中间。}%

@{喻如影移日續、日移影續，影之與日無有中間。}@
%{就像影子随着太阳移动，太阳随着影子移动，影子和太阳之间没有中间。}%

@{如是最後識滅，初識續餘道生，後識滅已，即受初識，無有中間。}@
%{这样，最后的意识灭尽了，最初的意识延续到余道而生，后面的意识灭尽了，就马上受用最初的意识，没有中间。}%

@{若最初識、若最後識相應法，不至後識。}@
%{无论是最初的意识，还是最后的意识，与之相应的法，都不会到达后面的意识。}%

@{喻如眼識滅已生耳識，耳識滅已生眼識。}@
%{就像眼识灭尽了，就产生耳识，耳识灭尽了，就产生眼识。}%

@{眼識相應法不至耳識，耳識相應法不至眼識。}@
%{眼识相应的法，不会到达耳识，耳识相应的法，不会到达眼识。}%

@{如是最後識、最後識相應法不至初識，初識相應法不至後識。}@
%{这样，最后的意识，最后的意识相应的法，不会到达最初的意识，最初的意识相应的法，不会到达后面的意识。}%

@{後識滅已即生初識，謂此時過，謂此滅彼生、謂此終彼始，非彼命彼身、非異命異身、非常非斷、非去非來非變、非無因非無作、非此作此受非異作異受，知有去來、知有生死、知有業相續、知有說法、知有緣，無有從此至彼者、無有從彼至此者。}@
%{后面的意识灭尽了，就马上产生最初的意识，就是说，时间过去了，就是说，这个灭了，那个生了，就是说，这个结束了，那个开始了，不是那个生命，那个身体，不是不同的生命，不同的身体，不是常，不是断，不是去，不是来，不是变，不是无因，不是无作，不是这个作这个受，不是不同的作不同的受，知道有去有来，知道有生死，知道有业的相续，知道有说法，知道有缘，没有从这个到那个的，没有从那个到这个的。}%

@{何以故？業緣相續生。}@
%{为什么呢？因为业缘相续而生。}%

%{就像佛陀所说：阿难！意识是有缘的。}%

@{如是阿難問已有答：識有何緣？行緣。}@
%{这样，阿难就问了，佛陀已经回答了：意识有什么缘呢？行为缘。}%

%{这就是答案。}%

%{阿难问：如果没有行为，会有意识吗？}%

@{世尊！無也。}@
%{世尊说：没有。}%

%{所以，阿难！这个因缘的意识，也就是行为缘起的意识，就像刚才所说的一样。}%

@{云何識緣名色？}@
%{怎么理解意识缘起名色呢？}%

@{有緣共欲識生，有欲身業生、有欲口業生、有欲意業生。}@
%{有缘起，共同产生欲识，有欲识产生身业，有欲识产生口业，有欲识产生意业。}%

@{共有欲身業口業，是謂色。}@
%{欲识产生的身业和口业，这就是色。}%

@{共有欲意業，由意生受、想、思、觸、思惟，謂名。}@
%{欲识产生的意业，由意生起受、想、思、触、思惟，这就是名。}%

%{这样，现在意识产生现在的名色，这就是意识缘起现在的名色。}%

%{无记识产生的身业和口业，这就是色。}%

@{無記意業由意生受、想、思、觸、思惟，謂名。}@
%{无记识产生的意业，由意生起受、想、思、触、思惟，这就是名。}%

%{世尊，没有。}%
@{阿難！以因緒緣愛。}@
%{阿难，以因缘爱。}%
@{阿難！受緣愛，如向所說。}@
%{阿难，受缘爱，就像先前所说的一样。}%
@{以是故說。}@
%{所以这样说。}%
@{云何愛緣取？愛未斷，愛欲取、見取、戒取、我取，是名愛緣取。}@
%{什么是爱缘取呢？爱没有断，爱欲取、见取、戒取、我取，这就是爱缘取。}%

@{云何欲取？除欲界愛初觸，若餘欲界愛廣，是名欲取。}@
%{什么是欲取呢？除了欲界爱最初接触，如果其余欲界爱广泛存在，这就是欲取。}%

@{云何見取？除戒取，若餘見取。}@
%{什么是见取呢？除了戒取，如果其余见取存在，这就是见取。}%

@{云何戒取？戒道是名戒取。}@
%{什么是戒取呢？戒道，这就是戒取。}%

@{云何我取？除色無色界愛初觸，若餘色無色界愛廣，是名我取。}@
%{什么是我取呢？除了色无色界爱最初接触，如果其余色无色界爱广泛存在，这就是我取。}%

@{云何欲界取？除欲界愛初觸見取、戒取，若餘欲界煩惱，是名欲取。}@
%{什么是欲界取呢？除了欲界爱最初接触、见取、戒取，如果其余欲界烦恼存在，这就是欲界取。}%

@{云何見取？六十二見及邪見，是名見取。}@
%{什么是见取呢？六十二见以及邪见，这就是见取。}%

@{云何戒取？戒淨、道淨、二俱淨、解脫、無依、盡苦邊，若於彼堪忍欲愛。戒謂身口戒。道謂邪緣求吉，養髮、入水、事火、事日月、牛行鹿行狗行默行、求力行、求大人行，種種苦行及餘求邪吉，是名道。若彼戒此道，求覓、重求覓、究竟求覓，齊是謂淨、謂解脫、謂戒淨、謂我解脫、謂聖、謂阿羅漢、謂般涅槃。若於彼欲重欲究竟欲堪忍，是名戒取。}@
%{什么是戒取呢？戒净、道净、二俱净、解脱、无依、尽苦边，如果能忍受这些，这就是戒取。戒指身口戒。道指邪缘求吉，养头发、入水、事火、事日月、牛行鹿行狗行默行、求力行、求大人行，各种苦行以及其他邪恶的求吉，这就是道。如果能守持这些戒道，并不断寻求、反复寻求、最终寻求，这些都称为净、称为解脱、称为戒净、称为我解脱、称为圣、称为阿罗汉、称为般涅槃。如果能忍受这些欲、重欲、究竟欲，这就是戒取。}%

@{云何我取？除色無色界初觸愛戒取、見取，若餘色無色界煩惱，是名我取。}@
%{什么是我取呢？除了色无色界最初接触爱、戒取、见取，如果其余色无色界烦恼存在，这就是我取。}%

@{如佛說：阿難！取有何緣？如是阿難問已有答：取由何緣？愛緣取。此是答。阿難！若無一切愛者，有取不？世尊！無也。阿難！以是因緒緣取。阿難！愛緣取，如向所說。以是故說。}@
%{就像佛陀所说：阿难，取有什么缘由呢？阿难问了，佛陀已经回答了：取由什么缘由呢？爱缘取。这就是答案。阿难，如果没有任何爱，会有取吗？世尊，没有。阿难，因为这个因缘而有取。阿难，爱缘取，就像之前所说的那样。因此我说。}%

@{云何取緣有？欲取、見取、戒取、我取緣未斷，若作欲行、色行、無色行，有報身口意業，是名取緣有。}@
%{什么是取缘有呢？欲取、见取、戒取、我取缘没有断，如果做欲行、色行、无色行，有报身口意业，这就是取缘有。}%

@{復次取緣三有，欲有、色有、無色有。}@
%{再者，取缘有三有，欲有、色有、无色有。}%

@{云何欲有？二種欲有，或欲有即業有、或欲有即生有。}@
%{什么是欲有呢？有两种欲有，一种是欲有即业有，另一种是欲有即生有。}%

@{云何欲有即業有？欲行未竟未知未斷，若作欲行有報身口意業，是名欲有即業有。}@
%{什么是欲有即业有呢？欲行还没有完成，不知道还没有断，如果做欲行，有报身口意业，这就是欲有即业有。}%

@{云何欲有即生有？若作業成就已，欲界天上受五種我分身色受想行識，是名欲有即生有。此謂受有、此謂報有、此謂復有。如是欲行業有、如是欲行生有，是名欲有。}@
%{什么是欲有即生有呢？如果作业已经成就，欲界天上的五种我分身色受想行识，这就是欲有即生有。这称为受有，这称为报有，这称为复有。就像这样，欲行业有、欲行生有，这就是欲有。}%

@{云何色有？二種色有，或色有即業有、或色有即生有。}@
%{什么是色有呢？有两种色有，一种是色有即业有，另一种是色有即生有。}%

@{云何色有即業有？色行未竟未知未斷，若作色行有報身口意業，是名色有即業有。}@
%{什么是色有即业有呢？色行还没有完成，不知道还没有断，如果做色行，有报身口意业，这就是色有即业有。}%

@{云何色有即生有？若作業成就已，色界天上若受五種我分色受想行識，是名色有即生有。此謂受有、此謂報有、此謂復有。如是色行業有、如是色行生有，是名色有。}@
%{什么是色有即生有呢？如果作业已经成就，色界天上的五种我分色受想行识，这就是色有即生有。这称为受有，这称为报有，这称为复有。就像这样，色行业有、色行生有，这就是色有。}%

@{云何無色有？二種無色有，或無色有即業有、或無色有即生有。}@
%{什么是无色有呢？有两种无色有，一种是无色有即业有，另一种是无色有即生有。}%

@{云何無色有即業有？無色行未竟未知未斷，若作無色行有報身口意業，名無色有即業有。}@
%{什么是无色有即业有呢？无色行还没有完成，不知道还没有断，如果做无色行，有报身口意业，这就是无色有即业有。}%

@{云何無色有即生有？若作業成就已，無色界天上受四種我分身受想行識，是名無色有即生有。此謂受有、此謂報有、此謂復有。如是無色行業有、如是無色行生有，是名無色有。}@
%{什么是无色有即生有呢？如果作业已经成就，无色界天上的四种我分身受想行识，这就是无色有即生有。这称为受有，这称为报有，这称为复有。就像这样，无色行业有、无色行生有，这就是无色有。}%

@{如佛說：阿難！有有緣。如是阿難問已有答：有何緣？取緣有。此是答。阿難！若無一切取者，有有不？世尊！無也。阿難！以因緒緣。阿難！取緣有，如向所說。以是故說。}@
%{就像佛陀所说：阿难，有有什么缘由呢？阿难问了，佛陀已经回答了：有什么缘由呢？取缘有。这就是答案。阿难，如果没有任何取，会有有吗？世尊，没有。阿难，因为这个因缘而有有。阿难，取缘有，就像之前所说的那样。因此我说。}%

@{云何有緣生？若諸眾生，眾中生重生、住胎出胎、得生陰、具諸入眾和合，是名生。}@
%{什么是有缘生呢？如果一切众生，在众中再生、住胎出胎、得到生阴、具备各种入众的和合，这就是生。}%

@{如佛說：阿難！生有緣。阿難問已有答：生何緣？有緣生。此是答。阿難！若無一切有者，有生不？世尊！無也。阿難！以因緒緣有緣生，如向所說。以是故說。}@
%{就像佛陀所说：阿难，生有什么缘由呢？阿难问了，佛陀已经回答了：生什么缘由呢？有缘生。这就是答案。阿难，如果没有任何有，会有生吗？世尊，没有。阿难，因为这个因缘而有缘生，就像之前所说的那样。因此我说。}%

@{云何生緣老死憂悲苦惱大苦聚？}@
%{什么是生缘老死忧悲苦恼大苦聚呢？}%

@{云何老？謂眾生衰老戰掉諸根熟念減行故，是名老。}@
%{什么是老呢？是指众生衰老，战掉诸根，熟念减少，行为衰退，这就是老。}%

@{云何死？若諸眾生終沒死盡，時過陰壞捨身，此陰變異眾別離，是名死。}@
%{什么是死呢？是指一切众生最终死亡，时间过去，阴坏，舍弃身体，这个阴变化，众生分离，这就是死。}%

@{云何憂？眾生觸若干苦法，若憂、重憂、究竟憂，內熱內心憂，是名憂。}@
%{什么是忧呢？是指众生接触到很多苦法，如果忧、重忧、究竟忧，内心烦躁不安，这就是忧。}%

@{云何悲？憂纏逼迫、憂煎具足、憂惱心亂，窮歎啼哭追憶並語，或自撲亂語，是名悲。}@
%{什么是悲呢？是指忧愁缠绕逼迫，忧愁煎熬，忧愁烦恼心乱，穷尽地叹息、哭泣、追忆、言语，或者自打自己、胡言乱语，这就是悲。}%

@{云何苦？若身覺苦，眼觸苦受乃至身觸苦受，是名苦。}@
%{什么是苦呢？是指身体感觉到苦，眼睛接触到苦受，乃至身体接触到苦受，这就是苦。}%

@{云何惱？若心覺苦，意觸苦受，是名惱。}@
%{什么是恼呢？是指内心感觉到苦，意念接触到苦受，这就是恼。}%

@{云何大苦聚？若眾苦、若罵辱苦、若心不定，是名大苦聚。}@
%{什么是大苦聚呢？是指各种苦，以及被辱骂的苦，以及心不定的苦，这就是大苦聚。}%

@{如佛說：阿難！老死憂悲苦惱大苦聚有緣。阿難問已有答：老死憂悲苦惱大苦聚有何緣？生緣老死憂悲苦惱大苦聚。此是答。阿難！若無生，有老死憂悲苦惱大苦聚不？世尊！無也。阿難！以因緒緣故，老死憂悲苦惱大苦聚。若生緣老死憂悲苦惱大苦聚，如上說。}@
%{就像佛陀所说：阿难，老死忧悲苦恼大苦聚有什么缘由呢？阿难问了，佛陀已经回答了：老死忧悲苦恼大苦聚有什么缘由呢？生缘老死忧悲苦恼大苦聚。这就是答案。阿难，如果没有生，会有老死忧悲苦恼大苦聚吗？世尊，没有。阿难，因为这个因缘，而有老死忧悲苦恼大苦聚。如果生缘老死忧悲苦恼大苦聚，就像上面所说的那样。}%
@{云何如是純苦聚集？}@
%{什么是纯粹的痛苦聚集呢？}%

@{謂七苦法，老、死、憂、悲、苦、惱、大苦聚，是名純苦陰。}@
%{就是指七种痛苦：衰老、死亡、忧愁、悲伤、痛苦、烦恼、大苦聚集，这就是纯粹的苦阴。}%

@{復次十一苦法，無明、行、識、名色、六入、觸、受、愛、取、有、生，是名純苦陰。}@
%{或者说，十一種痛苦：无明、行、识、名色、六入、触、受、爱、取、有、生，这就是纯粹的苦阴。}%

@{復次亦十八苦法，無明、行、識、名色、六入、觸、受、愛、取、有、生、老、死、憂、悲、苦、惱、大苦聚，是名純苦陰。}@
%{或者说，十八种痛苦：无明、行、识、名色、六入、触、受、爱、取、有、生、老、死、忧愁、悲伤、痛苦、烦恼、大苦聚集，这就是纯粹的苦阴。}%

@{如是純苦陰，有集和合生俱生。}@
%{这种纯粹的苦阴，是由于集聚而产生，是和合而生，生生不息。}%

@{生已俱生已，出俱出。}@
%{当它已经生起，它就和生一起生起，当它消失，它就和生一起消失。}%

@{出已俱出已得成就，是謂純苦陰集。}@
%{当它消失，和生一起消失，并最终成就消失，这就是纯粹的苦阴的集聚。}%

@{云何無明滅則行滅？}@
%{什么是无明灭则行灭呢？}%

@{若無明生則行生、若無明滅則行滅，是謂無明滅則行滅。}@
%{就是说，如果无明生起，那么行就会生起；如果无明灭尽，那么行就会灭尽，这就是无明灭则行灭。}%

@{乃至若有生則有老死、若生滅則老死滅，是謂生滅則老死滅。}@
%{以此类推，如果有生，那么就会有老死；如果生灭尽，那么老死就会灭尽，这就是生灭则老死灭。}%

@{云何純苦陰滅？}@
%{什么是纯粹的苦阴的灭尽呢？}%

@{純苦陰者，謂七苦法，老、死、憂、悲、苦、惱、大苦聚，是名純苦陰。}@
%{纯粹的苦阴是指七种痛苦：衰老、死亡、忧愁、悲伤、痛苦、烦恼、大苦聚集，这就是纯粹的苦阴。}%

@{復次十一苦，是名純苦陰。}@
%{或者说，十一種痛苦，这就是纯粹的苦阴。}%

@{復次十八苦法，無明乃至大苦聚，是名純苦陰。}@
%{或者说，十八种痛苦：无明乃至大苦聚集，这就是纯粹的苦阴。}%

@{如是純苦陰，盡、變異、寂靜、滅、沒，名純苦陰滅。}@
%{这种纯粹的苦阴，最终会消失、改变、寂静、灭尽、消失，这就是纯粹的苦阴的灭尽。}%
