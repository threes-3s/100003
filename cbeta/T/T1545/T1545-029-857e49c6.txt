@{阿毘達磨大毘婆沙論卷第二十九五百大阿羅漢等造}@
%{阿毘达磨大毘婆沙论卷第二十九，五百大阿罗汉等造}%

@{三藏法師玄奘奉詔譯}@
%{三藏法师玄奘奉诏译}%

@{雜蘊第一中補特伽羅納息第三之七}@
%{杂蕴第一中补特伽罗纳息第三之七}%

@{如世尊說。有三界。謂斷界離界。滅界乃至廣說。}@
%{就像世尊所说，有三界，分别是断界、离界、灭界，乃至广说。}%

@{問何故作此論答為廣分別契經義故。}@
%{问：为什么作此论？答：为了广泛分别契经的义理。}%

@{謂契經說。具壽阿難往詣尊者名上座所。}@
%{契经说，具寿阿难前往尊者名上座处。}%

@{問何故阿難往詣彼所。}@
%{问：为什么阿难前往那里？}%

@{答尊者阿難。是樂法者。是正法將攝受聖教。御聖教船。恒巡四眾。教授教誡。}@
%{答：尊者阿难，是乐法者，是正法将摄受圣教，御圣教船，恒巡四众，教授教诫。}%

@{數數觀察諸苾芻等。勿有懈怠耽著戲論。}@
%{数数观察诸苾刍等，勿有懈怠耽著戏论。}%

@{或於境界顛倒思惟。令彼一生空過顛墜。故往彼所。}@
%{或者在境界上颠倒思惟，令彼一生空过颠坠。所以前往那里。}%

@{復次阿難作如是念。彼名上座恒樂寂靜居阿練若。勇猛精勤證何妙德。}@
%{其次，阿难作如是念：彼名上座恒乐寂静居阿练若，勇猛精勤证何妙德。}%

@{我應往問。若能為我說所證德。我當合掌隨喜讚歎。}@
%{我应该前往询问，如果能为我说所证德，我当合掌随喜赞叹。}%

@{若不爾者。方便慇懃示其加行。令速證得。}@
%{若不尔者，方便慇懃示其加行，令速证得。}%

@{勿彼多時居阿練若空無所獲。故往彼所。}@
%{勿彼多时居阿练若空无所获。所以前往那里。}%

@{如彼經說。具壽阿難到已施設同分言論。非不同分。}@
%{如彼经说，具寿阿难到达已，设施同分言论，非不同分。}%

@{問何等名為同分言論。}@
%{问：何等名为同分言论？}%

@{答若居阿練若者。問以阿練若法。若持毘奈耶者。問毘奈耶。若誦素怛纜者。問素怛纜若學阿毘達磨者。問阿毘達磨。}@
%{答：若居阿练若者，问以阿练若法；若持毘奈耶者，问毘奈耶；若诵素怛缆者，问素怛缆；若学阿毘达磨者，问阿毘达磨。}%

@{是名同分言論。}@
%{是名为同分言论。}%

@{與此相違名不同分言論。}@
%{与此相违，名为不同分言论。}%

@{謂居阿練若者。問以三藏。持毘奈耶者。問阿練若及餘二藏。}@
%{谓：居阿练若者，问以三藏；持毘奈耶者，问阿练若及余二藏。}%

@{誦素怛纜者。問阿練若及餘二藏。學阿毘達磨者。問阿練若及餘二藏。}@
%{诵素怛缆者，问阿练若及余二藏；学阿毘达磨者，问阿练若及余二藏。}%

@{或更問餘事。皆名不同分言論。}@
%{或者更问余事，皆名为不同分言论。}%

@{尊者阿難所以唯作同分言論。若作不同分言論者。彼不解故。便不能答。}@
%{尊者阿难所以唯作同分言论。若作不同分言论者，彼不解故，便不能答。}%

@{既不能答。心便羞恥。以羞恥故鬪諍違拒。}@
%{既不能答，心便羞耻，以羞耻故鬪诤违拒。}%

@{不欲令彼起如是過。是故唯作同分言論。}@
%{不欲令彼起如是过，是故唯作同分言论。}%

@{謂但問彼阿練若法。}@
%{谓：但问彼阿练若法。}%

@{如彼經說。爾時阿難問名上座。若有苾芻。居阿練若。或居樹下。或居靜室。或在塚間。應數思惟何等行法。}@
%{如彼经说，尔时阿难问名上座：若有苾刍，居阿练若，或者居树下，或者居静室，或者在塚间，应数思惟何等行法。}%

@{時名上座白阿難言。若有苾芻。居阿練若。或居樹下。或居靜室。或在塚間。應數思惟二種行法。}@
%{时名上座白阿难言：若有苾刍，居阿练若，或者居树下，或者居静室，或者在塚间，应数思惟二种行法。}%

@{謂奢摩他毘鉢舍那所以者何。}@
%{谓：奢摩他毘鉢舍那，所以者何？}%

@{若奢摩他熏修心者。依毘鉢舍那而得解脫。若毘鉢舍那熏修心者。依奢摩他而得解脫。}@
%{若奢摩他熏修心者，依毘鉢舍那而得解脱；若毘鉢舍那熏修心者，依奢摩他而得解脱。}%

@{若奢摩他毘鉢舍那熏修心者依三種界而得解脫。}@
%{若奢摩他毘鉢舍那熏修心者，依三种界而得解脱。}%

@{云何三界。所謂斷界離界滅界。}@
%{云何三界？所谓断界、离界、灭界。}%

@{問依對法義於一心中有奢摩他毘鉢舍那云何建立如是二種行者差別。}@
%{问：依对法义，于一心中有奢摩他毘鉢舍那，云何建立如是二种行者差別？}%

@{答由加行故二種差別。謂加行時或多修習奢摩他資糧。或多修習毘鉢舍那資糧}@
%{答：由加行故二种差別，谓加行时，或者多修习奢摩他资粮，或者多修习毘鉢舍那资粮。}%

@{多修習奢摩他資糧者。謂加行時恒樂獨處閑居寂靜怖畏憒鬧見諠雜過恒居靜室入聖道時。}@
%{多修习奢摩他资粮者，谓加行时恒乐独处闲居寂静，怖畏憒闹见諠杂过，恒居静室，入圣道时。}%

@{名奢摩他行者。}@
%{名为奢摩他行者。}%

@{多修習毘鉢舍那資糧者謂加行時恒樂讀誦思惟三藏。}@
%{多修习毘鉢舍那资粮者，谓加行时恒乐读诵思惟三藏。}%

@{於一切法自相共相數數觀察入聖道時。名毘鉢舍那行者。}@
%{于一切法自相共相数数观察，入圣道时，名为毘鉢舍那行者。}%

@{復次或有繫心一緣不分別法相。或有分別法相不繫心一緣。}@
%{其次，或者有繫心一缘不分别法相，或者有分别法相不繫心一缘。}%

@{若繫心一緣不分別法相者入聖道時。名奢摩他行者。}@
%{若繫心一缘不分别法相者，入圣道时，名为奢摩他行者。}%

@{若分別法相不繫心一緣者入聖道時。名毘鉢舍那行者。}@
%{若分别法相不繫心一缘者，入圣道时，名为毘鉢舍那行者。}%

@{復次若利根者。名毘鉢舍那行者若鈍根者。名奢摩他行者。}@
%{其次，若利根者，名为毘鉢舍那行者；若钝根者，名为奢摩他行者。}%

@{如利根鈍根如是。因力緣力。內分力外分力。內正思惟力外聞他音力。應知亦爾。}@
%{如利根钝根如是，因力缘力，内分力外分力，内正思惟力外闻他音力，应知亦尔。}%

@{問斷離滅界體是無為無因無果。}@
%{问：断离灭界体是无为无因无果。}%

@{云何乃說若奢摩他毘鉢舍那熏修心者依三種界而得解脫。}@
%{云何乃说若奢摩他毘鉢舍那熏修心者，依三种界而得解脱？}%

@{答彼契經於緣涅槃勝解以界聲說。謂修行者雖加行時精進勇猛修習止觀二種資糧。}@
%{答：彼契经，于缘涅槃胜解，以界声说，谓修行者，虽加行时精进勇猛修习止观二种资粮。}%

@{若於涅槃不起勝解決定趣證。畢竟不能斷諸煩惱心得解脫。}@
%{若于涅槃不起胜解，决定趣证，毕竟不能断诸烦恼，心得解脱。}%

@{故緣涅槃勝解名界。依此界故心得解脫。}@
%{故缘涅槃胜解名界，依此界故心得解脱。}%

@{如彼經說。爾時阿難問名上座。何等斷故名為斷界。何等離故名為離界。何等滅故名為滅界。}@
%{如彼经说，尔时阿难问名上座：何等断故名为断界？何等离故名为离界？何等灭故名为灭界？}%

@{名上座言。一切行斷故名斷界。一切行離故名離界。一切行滅故名滅界。}@
%{名上座言：一切行断故名断界，一切行离故名离界，一切行灭故名灭界。}%

@{尊者阿難聞已合掌隨喜讚歎辭退。}@
%{尊者阿难闻已，合掌随喜赞叹，辞退。}%

@{復詣竹林道場以此事問五百苾芻。}@
%{复诣竹林道场，以此事问五百苾刍。}%

@{彼復皆如名上座答。}@
%{彼复皆如名上座答。}%

@{問彼諸苾芻云何而答。}@
%{问：彼诸苾刍云何而答？}%

@{有作是說。從少至老次第而答。如法集時少者先問。}@
%{有作是说：从少至老次第而答，如法集时，少者先问。}%

@{有餘師說。從老至少次第而答。如行施物自老至少。}@
%{有余师说：从老至少次第而答，如行施物，自老至少。}%

@{復有說者。一苾芻答餘皆隨喜。}@
%{复有说者，一苾刍答，余皆随喜。}%

@{脇尊者言。先作白已後次行籌。受籌名答。}@
%{脇尊者言：先作白已，后次行筹，受筹名答。}%

@{如彼經說。爾時阿難聞已合掌隨喜讚歎。}@
%{如彼经说，尔时阿难闻已，合掌随喜赞叹。}%

@{辭詣佛所到已頂禮世尊雙足。却住一面。以此句義問佛世尊。}@
%{辞诣佛所，到达已，顶礼世尊双足，却住一面，以此句义问佛世尊。}%

@{佛還如彼上座等答。}@
%{佛还如彼上座等答。}%

@{問尊者阿難忍可上座五百苾芻所說義不。設爾何失。}@
%{问：尊者阿难忍可上座五百苾刍所说义否？设尔何失？}%

@{若忍可者。何故復以問佛世尊。若不忍可何故合掌隨喜讚歎。}@
%{若忍可者，何故复以问佛世尊？若不忍可，何故合掌随喜赞叹？}%

@{答阿難忍可彼所說義。}@
%{答：阿难忍可彼所说义。}%

@{問何故復以問佛世尊。}@
%{问：何故复以问佛世尊？}%

@{答如佛世尊知而故問。}@
%{答：如佛世尊知而故问。}%

@{尊者阿難亦復如是。所以者何。阿難欲顯善說法中同見同欲文義決定。}@
%{尊者阿难亦复如是，所以者何？阿难欲显善说法中同见同欲文义决定。}%

@{如大師說。徒眾亦然。如親教說。弟子亦然。如軌範說。受學亦然。}@
%{如大师说，徒众亦然；如亲教说，弟子亦然；如轨范说，受学亦然。}%

@{如是文義微妙決定。依之修學乃至能證阿羅漢果。}@
%{如是文义微妙决定，依之修学，乃至能证阿罗汉果。}%

@{非如外道所說文義師徒眾等展轉相違。依之修學空無所證。}@
%{非如外道所说文义，师徒众等展转相违，依之修学，空无所证。}%

@{復次阿難欲以佛妙言印印所說義。故重問佛。}@
%{其次，阿难欲以佛妙言印印所说义，故重问佛。}%

@{若不以佛妙言印之則所說義猶可傾動。當來四眾不敬信故。}@
%{若不以佛妙言印之，则所说义犹可倾动，当来四众不敬信故。}%

@{如世文符。若無王印則所行處人不敬受。此亦如是。故重問佛。}@
%{如世文符，若无王印，则所行处人不敬受，此亦如是，故重问佛。}%

@{如彼經說佛問阿難汝知上座五百苾芻有何功德。}@
%{如彼经说，佛问阿难：汝知上座五百苾刍有何功德？}%

@{阿難白佛。彼名上座五百苾芻皆阿羅漢諸漏已盡已捨重擔盡諸有結。}@
%{阿难白佛：彼名上座五百苾刍，皆阿罗汉，诸漏已尽，已舍重担，尽诸有结。}%

@{逮得己利善辦聖旨心善解脫。}@
%{逮得己利，善办圣旨，心善解脱。}%

@{佛告阿難。如汝所說。}@
%{佛告阿难：如汝所说。}%

@{問何故世尊問彼功德。}@
%{问：何故世尊问彼功德？}%

@{答為欲開發少欲喜足所覆真實功德寶藏。}@
%{答：为欲开发少欲喜足所覆真实功德宝藏。}%

@{令諸世間知已敬養得勝果故。}@
%{令诸世间知已，敬养得胜果故。}%

@{如世伏藏雖多珍寶沙土覆之不得顯現。}@
%{如世伏藏，虽多珍宝，沙土覆之，不得显现。}%

@{若有開發令無量人採取受用得世富樂。}@
%{若有开发，令无量人采取受用，得世富乐。}%

@{此亦如是。故佛問之。}@
%{此亦如是，故佛问之。}%

@{復次開覺施主勝思願故。謂有施主。恒以衣服等四種供具。施彼上座及五百苾芻。}@
%{其次，开觉施主胜思愿故，谓有施主，恒以衣服等四种供具，施彼上座及五百苾刍。}%

@{而不知彼有勝功德。欲令知已歡喜踊躍起勝思願。}@
%{而不知彼有胜功德，欲令知已，欢喜踊跃起胜思愿。}%

@{我等得遇如是福田已種善種。定於來世受大快樂。是故世尊問彼功德。}@
%{我等得遇如是福田，已种善种，定于来世受大快乐，是故世尊问彼功德。}%

@{復次為止世間誹謗事故。謂彼上座在母胎中經六十年。}@
%{其次，为止世间诽谤事故，谓彼上座在母胎中经六十年。}%

@{既出胎已形容衰老無有威德。故初生已立上座名。}@
%{既出胎已，形容衰老，无有威德，故初生已立上座名。}%

@{後雖出家而被嗤笑。少年強盛晝夜精勤尚難得果況此衰老氣力羸劣能得果耶。}@
%{后虽出家，而被嗤笑，少年强盛，昼夜精勤尚难得果，况此衰老，气力羸劣，能得果耶？}%

@{又彼上座所度五百新學苾芻先隨天授。眾人毀曰。如是老叟貪著名利度五百人。}@
%{又彼上座所度五百新学苾刍，先随天授，众人毁曰：如是老叟贪著名利度五百人。}%

@{為充自身驅役供侍。不能教誡令從邪法。}@
%{为充自身驱役供侍，不能教诫令从邪法。}%

@{五百苾芻先受邪化後雖歸正得無學果。而有謗言。}@
%{五百苾刍先受邪化，后虽归正，得无学果，而有谤言。}%

@{此愚人輩先貪利養捨佛從邪。雖後還來而無所得。}@
%{此愚人辈，先贪利养，舍佛从邪，虽后还来，而无所得。}%

@{為止如是諸誹謗故問彼功德。}@
%{为止如是诸诽谤故，问彼功德。}%

@{令世共知捨誹謗罪勤修敬養。於當來世生天解脫。}@
%{令世共知，舍诽谤罪，勤修敬养，于当来世生天解脱。}%

@{彼經雖說斷等三界。而不廣辯三界差別。}@
%{彼经虽说道等三界，而不广辩三界差別。}%

@{彼是此論所依根本。彼不說者今欲說之。故作斯論。}@
%{彼是此论所依根本，彼不说者，今欲说之，故作斯论。}%
@{云何斷界。答除愛結餘結斷名斷界。}@
%{什么是断界呢？就是除了爱结以外，其他结断，叫做断界。}%
@{云何離界答愛結斷名離界。}@
%{什么是离界呢？就是爱结断，叫做离界。}%
@{云何滅界。答諸餘順結法斷名滅界。}@
%{什么是灭界呢？就是其他的顺结法断，叫做灭界。}%
@{此中先約阿毘達磨。依世俗理說三界別。近對治道有差別故。}@
%{这里先引用阿毗达磨，根据世俗道理来说三界分别，因为近对治道有差别。}%
@{餘結斷者。餘八結斷。順結法者。謂除九結餘有漏法。是名一種三界差別。}@
%{余结断是指其他的八结断；顺结法是指除了九结以外的其他有漏法。这就是一种三界分别。}%
@{復有說者。若八結及此相應并生等斷名斷界。}@
%{还有的说，如果八结以及与它相应并生等断，叫做断界。}%
@{若愛結及此相應并生等斷名離界。}@
%{如果爱结以及与它相应并生等断，叫做离界。}%
@{若諸餘順結法及此相應并生等斷名滅界。}@
%{如果其他的顺结法以及与它相应并生等断，叫做灭界。}%
@{即有漏善及諸有為無覆無記名順結法。}@
%{就是有漏善以及其他的有为无覆无记，叫做顺结法。}%
@{復有說者。若無明結斷名斷界。若愛結斷名離界。若諸餘結斷名滅界。}@
%{还有的说，如果无明结断，叫做断界；如果爱结断，叫做离界；如果其他的结断，叫做灭界。}%
@{復有說者。或有諸法能縛非能染。彼斷名斷界。}@
%{还有的说，有些法能缚，但不能染，它们断叫做断界。}%
@{或有諸法能縛亦能染。彼斷名離界。}@
%{有些法能缚，也能染，它们断叫做离界。}%
@{或有諸法非能縛非能染。而是所縛是所染。彼斷名滅界。}@
%{有些法不能缚，也不能染，但是被缚被染，它们断叫做灭界。}%
@{復有說者。或有諸法是能繫非能染。彼斷名斷界。}@
%{还有的说，有些法是能系，但不能染，它们断叫做断界。}%
@{或有諸法是能繫是能染。彼斷名離界。}@
%{有些法是能系，也能染，它们断叫做离界。}%
@{或有諸法非能繫非能染。而是所繫所染。彼斷名滅界。}@
%{有些法不能系，也不能染，但是被系被染，它们断叫做灭界。}%
@{有餘師說。唯諸隨眠有自性斷。}@
%{有的老师说，只有诸随眠有自性断。}%
@{問契經所說當云何通。}@
%{问：契经所说应该怎么解释呢？}%
@{如說。一切行斷故名斷界。一切行離故名離界。一切行滅故名滅界。}@
%{就像所说：一切行断，叫做断界；一切行离，叫做离界；一切行灭，叫做灭界。}%
@{品類足說復云何通。}@
%{品类足论又应该怎么解释呢？}%
@{如說。云何所斷法。答一切有漏法。云何遍知法。答一切有漏法。}@
%{就像所说：什么是断法？答：一切有漏法。什么是遍知法？答：一切有漏法。}%

%{他们这样回答：如果诸随眠缘八结起，它们断叫做断界。}%

%{如果诸随眠缘爱结起，它们断叫做离界。}%
@{若諸隨眠緣餘法起。彼斷名滅界。}@
%{如果诸随眠缘其他法起，它们断叫做灭界。}%

%{评说：他们不应该这样说，因为一切有漏法，先被系缚，离系缚时，都可以断。}%
@{有作是說。唯愛隨眠有自性斷。}@
%{有的说，只有爱随眠有自性断。}%
@{問若爾。前說契經及論當云何通彼作是答。}@
%{问：如果是这样，前面所说契经和论典应该怎么解释呢？他们这样回答：}%
@{若愛隨眠緣八結起。彼斷名斷界。}@
%{如果爱随眠缘八结起，它们断叫做断界。}%
@{若愛隨眠緣愛結起。彼斷名離界。}@
%{如果爱随眠缘爱结起，它们断叫做离界。}%

%{对境离系，叫做离界。}%

%{舍弃诸重担，叫做灭界。}%

%{脇尊者说：没有系缚，系缚息灭，叫做断界。}%

%{没有染污，污染污秽息灭，叫做离界。}%

%{没有彼果，彼果息灭，叫做灭界。}%

%{尊者设摩达多说：诸烦恼断，叫做断界。}%

%{没有贪，治贪，叫做离界。}%

%{果相续灭，叫做灭界。}%

%{尊者左取这样说：相续断，叫做断界。}%
@{於緣離繫名離界。}@
%{对缘离系，叫做离界。}%
@{離執受故名滅界。}@
%{离执受，叫做灭界。}%
@{復有說者。過去煩惱斷故名斷界。現在煩惱斷故名離界。未來煩惱斷故名滅界。}@
%{还有的说：过去烦恼断，叫做断界；现在烦恼断，叫做离界；未来烦恼断，叫做灭界。}%
@{如煩惱斷蘊斷亦爾。}@
%{就像烦恼断，蕴断也是这样。}%
@{復有說者。苦受斷故名斷界。樂受斷故名離界。不苦不樂受斷故名滅界。}@
%{还有的说：苦受断，叫做断界；乐受断，叫做离界；不苦不乐受断，叫做灭界。}%
@{如三受斷順三受法斷亦爾。}@
%{就像三受断，顺三受法断也是这样。}%
@{復有說者。若苦苦斷名斷界。若壞苦斷名離界。若行苦斷名滅界。}@
%{还有的说：如果苦苦断，叫做断界；如果坏苦断，叫做离界；如果行苦断，叫做灭界。}%
@{復有說者。若欲界斷名斷界。若色界斷名離界。若無色界斷名滅界。}@
%{还有的说：如果欲界断，叫做断界；如果色界断，叫做离界；如果无色界断，叫做灭界。}%

%{像这样等等的说法，都是根据世俗，随便选择一个门类来辩论三界分别，都不是胜义。}%
@{諸斷界是離界耶。答如是。}@
%{诸断界是离界吗？答：是的。}%
@{設離界是斷界耶。答如是。}@
%{假如离界是断界吗？答：是的。}%
@{諸斷界是滅界耶。答如是。}@
%{诸断界是灭界吗？答：是的。}%
@{設滅界是斷界耶。答如是。}@
%{假如灭界是断界吗？答：是的。}%
@{諸離界是滅界耶。答如是。}@
%{诸离界是灭界吗？答：是的。}%
@{設滅界是離界耶。答如是。}@
%{假如灭界是离界吗？答：是的。}%
@{問何故復作此論。}@
%{问：为什么又要作这个论呢？}%
@{答前約阿毘達磨依世俗理。就近對治辯三界別。今隨契經顯此三界體無差別。}@
%{答：前面引用阿毗达磨，根据世俗道理，就近对治来辩论三界分别。现在根据契经来显示，这三界体无差别。}%

%{就是说，有漏法，每一次断的时候，都能得到一次断。}%

%{这每一次断，根据差别意义来说，就是三界。}%
@{故此三界義雖有別而體無異。}@
%{所以，这三界义虽然有分别，但是体无差异。}%

@{如世尊說。有三想。謂斷想離想滅想。}@
%{就像世尊所说，有三想，就是断想、离想、灭想。}%

%{等等广说。}%
@{問何故作此論。}@
%{问：为什么作这个论呢？}%
@{答為廣分別契經義故。}@

%{就是说，契经说有三種想，就是断、离、灭。}%

%{它就是这个论的根本，它没有说的，现在想说它。}%
@{故作此論。}@
%{所以作这个论。}%

@{云何愛。云何敬。如是等章。及解章義。既領會已。次應廣釋。問何故作此論。答為廣分別契經義故。謂契經說。若有修習慚愧圓滿。應知愛敬亦得圓滿。契經雖作是說。而不分別云何愛云何敬。契經是此論所依根本。彼所不分別者今應盡分別之。復次為欲訶毀非善士法令棄捨故。為欲讚歎諸善士法令修習故。為欲顯示五濁增時廣大有情甚難得故。此中非善士法者。謂有一類。愛則妨敬。敬則妨愛。愛妨敬者。如有父母於子寵極。子於父母有愛無敬。師於弟子應知亦然。此等名為愛則妨敬敬妨愛者。如有父母於子嚴酷子於父母有敬無愛。師於弟子應知亦然。此等名為敬則妨愛。如是俱名非善士法。善士法者。謂有一類愛則加敬敬則加愛。愛敬俱行名善士法。若有此法增上圓滿。應知即是廣大有情。如是有情甚為難得。世若無佛此類難遇。設令有者是大菩薩。諸大菩薩愛敬必俱。為顯此事及前所說三種因緣故作斯論。}@
%{什么是爱？什么是敬？诸如此类的章节，以及解释章节的意思，既然已经领会了，接下来应该广泛解释。问：为什么作此论？答：为了广泛分别契经的意思。契经说：如果有人修习惭愧圆满，应该知道爱敬也能够圆满。契经虽然作此说，但是不分别什么是爱什么是敬。契经是此论所依的根本，契经中没有分别的，现在应该尽力分别它。其次，为了想要呵斥毁坏不善士法，令其舍弃；为了想要赞叹诸善士法，令其修习；为了想要显示五浊增盛的时候，广大有情非常难得到。此中不善士法，指的是有一类人，爱就妨碍敬，敬就妨碍爱。爱妨碍敬，比如父母对孩子宠爱至极，孩子对父母有爱没有敬，师对弟子也应该知道是同样的道理。这些人被称为爱则妨碍敬，敬妨碍爱。比如父母对孩子严酷，孩子对父母有敬没有爱，师对弟子也应该知道是同样的道理。这些人被称为敬则妨碍爱。像这样，两者都称为不善士法。善士法，指的是有一类人，爱就增加敬，敬就增加爱，爱敬同时存在，被称为善士法。如果拥有这种法，增上圆满，应该知道这就是广大有情。像这样的有情非常难得。世间如果没有佛，这类人很难遇到。即使有，也是大菩萨。诸大菩萨爱敬必定同时存在。为了显示此事，以及前面所说的三种因缘，所以作此论。}%

@{云何愛。答諸愛等愛憙等憙樂等樂是謂愛。此本論師於異文義得善巧故。以種種文顯示此愛而體無別。問愛以何為自性。答愛有二種。一染污謂貪。二不染污謂信問諸貪皆愛耶。答應作順前句謂貪皆愛。有愛非貪此即是信問諸信皆愛耶。有作是說諸信皆愛。有愛非信謂染污愛。應作是說信有二種。一者於境唯信不求。二者於境亦信亦求。是故此中應作四句有是信非愛。謂信不求有是愛非信。謂染污愛。有亦信亦愛。謂信亦求。有非信非愛。謂除前相。云何敬。答謂有敬有敬性。有自在有自在性。於自在者有畏怖轉是謂敬。此本論師於異文義得善巧故。以種種文顯示此敬。而體無別。問敬以何為自性答敬以慚為自性。}@
%{什么是爱？答：诸爱、爱喜、喜乐、快乐，这些都是爱。此本论师在异文义上得到善巧，所以用各种文字来显示这种爱，而本质上没有区别。问：爱以什么为自性？答：爱有两种：一、染污，叫做贪；二、不染污，叫做信。问：所有的贪都是爱吗？答：应该顺着前一句说，贪都是爱。有爱不是贪，这就是信。问：所有的信都是爱吗？有些人说所有的信都是爱。有爱不是信，指的是染污的爱。应该这样说：信有两种：一、对境只信不求；二、对境既信也求。所以，这里应该作四句：有信不是爱，指的是信不求；有爱不是信，指的是染污的爱；有既信也爱，指的是信也求；有非信非爱，指的是除了前三种情况以外。什么是敬？答：指的是有敬，有敬性；有自在，有自在性。对自在者有畏怖，转变为敬。此本论师在异文义上得到善巧，所以用各种文字来显示敬，而本质上没有区别。问：敬以什么为自性？答：敬以惭为自性。}%

@{云何愛敬。乃至廣說。問何故復作此論。答前雖別說愛敬自性。而未總說於一境轉。今欲顯示愛敬二種於一境轉故作斯論。云何愛敬。答如有一類於佛法僧親教軌範及餘隨一有智尊重同梵行者。愛樂心悅恭敬而住。若於是處有愛及敬。是謂愛敬。此中一類者。謂異生或聖者。異生於佛愛樂心悅恭敬住者彼作是念。佛威力故我等解脫災橫王役種種苦事。及得世間諸資生具。聖者於佛愛樂心悅恭敬住者彼作是念。佛威力故我等永捨諸惡趣因。斷二十種薩迦耶見。得正決定見四聖諦。於無邊際生死輪迴諸苦事中。已作分限。復次彼二於佛愛樂心悅恭敬住者。俱作是念。佛威力故我等出家受具足戒得苾芻性。及餘利益安樂資糧。是故尊者鄔陀夷言。世尊於我有大恩德。謂拔我無量苦。與我無量樂。滅我無量惡。生我無量善。復次彼二於佛俱作是念。世尊開發我等慧眼故應愛敬。是故尊者舍利子言。若佛世尊不出于世。我等一切盲生盲死。復次彼二於佛俱作是念。佛為法王最初開示無上正法。令諸有情無倒了達。雜染清淨繫縛解脫。流轉還滅生死涅槃。餘無此能。故應愛敬。復次彼二於佛俱作是念。世尊最初出無明㲉宣說正法。亦令無量無邊有情出無明㲉。餘無此能。故應愛敬。復次彼二於佛俱作是念。無始時來七依勝定隱蔽不現。佛出世間無倒開示令無量眾依之趣入大涅槃宮。餘無此能。故應愛敬。復次彼二於佛俱作是念。佛威力故能令無量無邊有情修諸善法。謂從不淨觀乃至無生智。餘無此力。故應愛敬。復次彼二於佛俱作是念。佛威力故令諸有情種諸善根成熟解脫。餘無此力。故應愛敬。復次彼二於佛俱作是念。佛威力故念住正斷神足根力覺支道支三十七種菩提分等功德寶藏出現世間。利益安樂無邊有情故應愛敬。彼二於法愛樂心悅恭敬住者俱作是念。我依此法解脫一切身心苦惱究竟安樂。彼二於僧愛樂心悅恭敬住者俱作是念。僧威力故我於正法毘奈耶中淨信出家。受具足戒得苾芻性。能正受持百一羯磨。無所毀犯安樂而住。由此速證究竟涅槃。彼二俱於親教軌範及餘隨一有事尊重同梵行者。愛樂心悅恭敬住者。謂作是念。此諸師友為我伴侶。令我於法勤修正行速得成辦。於前所說三寶師友殊勝境中。具起愛敬。餘則不定。應作四句。謂或有境起愛非敬。如父母於子師於弟子等。或復有境起敬非愛。如於有德非已師長。或復有境起愛及敬。如有一類子於父母弟子於師等。或有境不起愛敬。謂除前相。問如是愛敬於何處有。答三界五趣雖皆容有。而此中說殊勝愛敬唯在欲界人趣非餘。唯佛法中有此愛敬。}@
%{什么是爱敬？乃至广泛解释。问：为什么又要作此论？答：前面虽然分别说了爱敬的自性，但是没有总说在一种境地上转变。现在想要显示爱敬两种在一种境地上转变，所以作此论。什么是爱敬？答：比如有一类人，对佛、法、僧、亲教、轨范，以及其他任何有智慧、值得尊敬、与梵行者相同的修行者，心怀爱乐、心悦诚服、恭敬地住着。如果在这个地方有爱和敬，这就是爱敬。这里的“一类人”，指的是凡夫或圣者。凡夫对佛心怀爱乐、心悦诚服、恭敬地住着，他们这样想：因为佛的威力，我们解脱了灾横、王役、各种痛苦之事，以及得到了世间各种资生具。圣者对佛心怀爱乐、心悦诚服、恭敬地住着，他们这样想：因为佛的威力，我们永远舍弃了诸恶趣的因，断除了二十种萨迦耶见，得到了正决定见四圣谛，在无边无际的生死轮回诸苦事中，已经作了分限。其次，这两类人对佛心怀爱乐、心悦诚服、恭敬地住着，都这样想：因为佛的威力，我们出家受具足戒，得到了苾刍性，以及其他利益、安乐、资粮。所以，尊者邬陀夷说：世尊对我有大恩德，拔除了我无量苦，给了我无量乐，灭掉了我无量恶，生起了我无量善。其次，这两类人对佛都这样想：世尊开发了我们慧眼，所以应该爱敬。所以，尊者舍利子说：如果佛世尊不出于世，我们一切盲生盲死。其次，这两类人对佛都这样想：佛是法王，最初开示无上正法，令诸有情无倒地了达，杂染清净、系缚解脱、流转还灭、生死涅槃，其他没有人能做到这一点，所以应该爱敬。其次，这两类人对佛都这样想：世尊最初出无明，宣说正法，也令无量无边有情出无明，其他没有人能做到这一点，所以应该爱敬。其次，这两类人对佛都这样想：无始时来，七依胜定隐蔽不现，佛出世间，无倒地开示，令无量众依之趣入大涅槃宫，其他没有人能做到这一点，所以应该爱敬。其次，这两类人对佛都这样想：因为佛的威力，能够令无量无边有情修诸善法，从不净观乃至无生智，其他没有人有这种力量，所以应该爱敬。其次，这两类人对佛都这样想：因为佛的威力，令诸有情种诸善根成熟解脱，其他没有人有这种力量，所以应该爱敬。其次，这两类人对佛都这样想：因为佛的威力，念住、正断、神足、根力、觉支、道支、三十七种菩提分等等功德宝藏出现世间，利益安乐无边有情，所以应该爱敬。这两类人对法心怀爱乐、心悦诚服、恭敬地住着，都这样想：我依靠此法解脱一切身心苦恼，究竟安乐。这两类人对僧心怀爱乐、心悦诚服、恭敬地住着，都这样想：因为僧的威力，我在正法毘奈耶中净信出家，受具足戒，得到了苾刍性，能够正受持百一羯磨，无所毁犯，安乐地住着，由此速证究竟涅槃。这两类人对亲教、轨范，以及其他任何值得尊敬、与梵行者相同的修行者，心怀爱乐、心悦诚服、恭敬地住着，都这样想：这些师友是我的伴侣，令我在法中勤修正行，快速得到成就。在前面所说的三宝、师友、殊胜境地中，具起爱敬，其他则不定。应该作四句：有的境地起爱不敬，比如父母对孩子，师对弟子等等；有的境地起敬不爱，比如对有德行，不是自己师长的人；有的境地起爱和敬，比如有一类人，子对父母，弟子对师等等；有的境地不起爱敬，指的是除了前三种情况以外。问：这样的爱敬在何处存在？答：三界五趣虽然都容许有，但是这里说殊胜的爱敬只存在于欲界人趣，不在其他地方。只有佛法中存在这种爱敬。}%

%{能够供养恭敬，就不会退转，接近涅槃}%

@{契經雖作是說。而不廣分別。云何供養。云何恭敬。彼是此論所依根本。彼不說者今應說之。復次為欲訶毀非善士法令棄捨故。為欲讚歎諸善士法令修習故。為欲顯示五濁增時廣大有情甚難得故。此中非善士法者。謂有一類若供養則妨恭敬。若恭敬則妨供養。云何供養則妨恭敬。如在家者或有男女雖具勢力能以種種資生珍饌供養父母。而恃此力心生輕慢。如出家者。或有弟子福德多聞。雖於其師能設種種財法供養。而恃此事遂於師所不生恭敬。此等供養則妨恭敬。云何恭敬則妨供養。如有一類懼他威力雖恭敬之而不供養。如是俱名非善士法。善士法者。謂有一類若供養彼則加恭敬。若恭敬彼則加供養。二事俱行。名善士法。若有此法增上圓滿。應知即是廣大有情。如是有情甚為難得世若無佛。此類難遇。設令有者是大菩薩諸大菩薩此二必俱。為顯此事及前所說三種因緣故作斯論。}@
%{契经虽然这样说，但没有广泛地分别，怎么讲供养，怎么讲恭敬。这些是此论所依据的根本。契经没有说，现在就应该说。其次，为了想要呵斥毁坏非善士法，令其舍弃；为了想要赞叹诸善士法，令其修习；为了想要显示五浊增时广大有情非常难得到。这里非善士法是指，有一类人，如果供养他，就会妨碍恭敬；如果恭敬他，就会妨碍供养。怎么讲供养就会妨碍恭敬呢？比如在家的人，有的男女虽然具有势力，能用各种资生珍馔供养父母，却仗着这种力量，心中生起轻慢。比如出家的人，有的弟子福德多闻，虽然对他的师父能设置种种财法供养，却仗着这件事，就不对师父生起恭敬。这种供养就会妨碍恭敬。怎么讲恭敬就会妨碍供养呢？比如有一类人，害怕别人的威力，虽然恭敬他，却并不供养。这样的人，都叫做非善士法。善士法是指，有一类人，如果供养他，就会更加恭敬；如果恭敬他，就会更加供养。这两件事都能做到，叫做善士法。如果有这种法增上圆满，应该知道这就是广大有情。这种有情非常难得到，世间如果没有佛，这种人很难遇到。即使有，也是大菩萨，诸大菩萨，这两者必然俱足。为了显明这件事，以及前面所说的三种因缘，所以作此论。}%

@{云何供養。答此有二種。一財供養。二法供養。財供養者。問財供養以何為自性。有作是說。為饒益故捨諸財物。即所捨財是此自性。有餘師說。能供養者。身語二業是此自性。或有說者。即能發彼心心所法是此自性。復有說者。受者受已諸根大種及餘造色皆得增長是此自性。評曰。應作是說。若所捨財。若能捨者身語二業。若能發彼心心所法。若受者受已諸根大種造色增長皆此自性。如是財供養。總用五蘊以為自性。已說自性。所以今當說。問何故名財供養。財供養是何義。答能為緣義。是供養義。此以財為初故名財供養。若為饒益故捨諸財物。受者受已身心增益如是名施亦名供養。若為饒益故捨諸財物。受者受已身心損減。如是名施不名供養。若為損害故捨匪宜物。受者受已或由神通或由呪藥或由福力身心增盛。此雖非施而名供養若為損害故捨匪宜物。受者受已身心損減。此不名施亦非供養。問此財供養在何處有。答唯在欲界。非色無色唯在四趣。非捺落迦。問此財供養誰設誰受。有作是說。傍生趣設。唯傍生趣受。乃至天趣設。唯天趣受。有餘師說。傍生趣設唯傍生受。鬼趣設二趣受。人趣設三趣受。天趣設四趣受。下不及上上及下故。評曰。應作是說。四趣皆能展轉供養問施設論說。天欲食時取空寶器。以衣覆上而置座前。經須臾頃隨其福力麁妙飲食自然盈滿。既爾如何受他供養。答雖不受他飲食供養。而有受餘香花資具。法供養者。問法供養以何為自性。有作是說。以說法者語為自性。有餘師說。以語所起名為自性。或有說者。以能發語心心所法為其自性。復有說者。受者聞已生未曾有善巧覺慧以為自性。評曰。應作是說。若說法者語。若能發語心心所法。若受者聞已生未曾有善巧覺慧皆此自性。如是法供養。總用五蘊以為自性。已說自性。所以今當說。問何故名法供養。法供養是何義。答能為緣義是供養義。此以法為初故名法供養。若為饒益故為他說法。他聞法已生未曾有善巧覺慧。如是名施亦名供養。若為饒益故為他說法。他聞法已不生未曾有善巧覺慧。如是名施不名供養。若為損害故說譏刺他法。他聞是已住正憶念歡喜忍受不數其過。生未曾有善巧覺慧。此雖非施而名供養。若為損害故說譏刺他法。他聞是已發恚恨心。不生未曾有善巧覺慧。此不名施亦非供養。問此法供養在何處有。答此法供養在欲色界。非無色界。五趣皆有。地獄有者如慈授子生地獄中。謂是浴室見諸苦具。便說頌言甞聞世間受苦樂非我非他之所作。受諸苦樂皆緣身身若滅無誰復受時彼地獄無量眾生聞此頌已脫地獄苦。從彼命終生天受樂。傍生有者。如迦賓折羅鳥自修梵行為他說法。鬼趣有者。如發愛鬼母為諸鬼子說是頌言。}@
%{怎么讲供养？答有两种，一是财供养，二是法供养。财供养是指，问财供养以何为自性？有人说，为了饶益，舍弃财物，所舍弃的财物就是它的自性。有的师说，能供养者，身语二业是它的自性。或者有人说，能发彼心心所法是它的自性。又有人说，受者受了之后，诸根大种以及其余造色都能够增长，是它的自性。评曰：应该这样说，无论是所舍弃的财物，还是能舍弃者的身语二业，无论是能发彼心心所法，还是受者受了之后诸根大种造色增长，都是它的自性。这样，财供养总用五蕴作为自性。已经说了自性，所以现在应该说，问为什么叫做财供养？财供养是什么含义？答能为缘义，是供养义。因为以财为初，所以叫做财供养。如果为了饶益，舍弃财物，受者受了之后，身心增益，这样叫做施，也叫做供养。如果为了饶益，舍弃财物，受者受了之后，身心损减，这样叫做施，不叫做供养。如果为了损害，舍弃不合适的物，受者受了之后，或者由于神通，或者由于咒药，或者由于福力，身心增盛，虽然这不是施，但叫做供养。如果为了损害，舍弃不合适的物，受者受了之后，身心损减，这就不叫做施，也不叫做供养。问此财供养在哪里有？答只在欲界，不在色界无色界，只在四趣，不在捺落迦。问此财供养谁设置谁受？有人说，傍生趣设置，唯傍生趣受，乃至天趣设置，唯天趣受。有的师说，傍生趣设置，唯傍生受；鬼趣设置，二趣受；人趣设置，三趣受；天趣设置，四趣受。下不及上，上及下，所以这样。评曰：应该这样说，四趣都能互相供养。问设施论说，天欲食时，取空宝器，用衣覆盖，放在座位前，经过须臾顷，随着其福力，粗妙饮食自然盈满。既然如此，如何受他供养？答虽然不受他饮食供养，但有受其余香花资具。法供养是指，问法供养以何为自性？有人说，以说法者的语为自性。有的师说，以语所起名为自性。或者有人说，以能发语心心所法为其自性。又有人说，受者闻了之后，生起未曾有善巧觉慧，以为自性。评曰：应该这样说，无论是说法者的语，还是能发语心心所法，无论是受者闻了之后生起未曾有善巧觉慧，都是它的自性。这样，法供养总用五蕴作为自性。已经说了自性，所以现在应该说。问为什么叫做法供养？法供养是什么含义？答能为缘义，是供养义。因为以法为初，所以叫做法供养。如果为了饶益，为他说法，他闻法之后，生起未曾有善巧觉慧，这样叫做施，也叫做供养。如果为了饶益，为他说法，他闻法之后，不生起未曾有善巧觉慧，这样叫做施，不叫做供养。如果为了损害，说讥刺他法，他闻了之后，住正忆念，欢喜忍受，不数其过，生起未曾有善巧觉慧，虽然这不是施，但叫做供养。如果为了损害，说讥刺他法，他闻了之后，发恚恨心，不生起未曾有善巧觉慧，这就不叫做施，也不叫做供养。问此法供养在哪里有？答此法供养在欲色界，不在无色界，五趣都有。地狱有，比如慈授子生地狱中，说是在浴室，看见诸苦具，就说颂言：尝闻世间受苦乐，非我非他之所作，受诸苦乐皆缘身，身若灭，无谁复受时，彼地狱无量众生闻此颂已，脱地狱苦，从彼命终生天受乐。傍生有，比如迦宾折罗鸟，自修梵行为他说法。鬼趣有，比如发爱鬼母，为诸鬼子说这颂言：}%

@{人趣有者。如今現見。天趣有者。欲界天中如補處慈尊為諸天說法。色界天中如手天子來白佛言。如此世尊四眾圍遶為說正法聞已奉行。我聞法已還無熱天。為彼諸天說法亦爾。}@
%{人道有，现在就有人在，天道也存在。欲界天中，就像补处菩萨，为诸天说法。色界天中，就像手天子，来向佛陀禀告，像这样，世尊带领四众弟子围绕着自己，宣讲正法，听闻之后就奉行。我听闻了佛法，就再也不感到烦热，为那些诸天说法也是这样。}%
@{問此法供養誰設誰受。有作是說。地獄趣設唯地獄趣受。乃至唯天趣設唯天趣受。}@
%{那么，这个法供养谁设置，谁受用呢？有些人说，地狱道设置，只有地狱道受用。一直到天道设置，只有天道受用。}%
@{有餘師說。地獄趣設唯地獄趣受。傍生趣設二趣受。鬼趣設三趣受。人趣設四趣受。天趣設五趣受。}@
%{还有的老师说，地狱道设置，只有地狱道受用。畜生道设置，两道受用。鬼道设置，三道受用。人道设置，四道受用。天道设置，五道受用。}%
@{評曰。應作是說。五趣皆能展轉供養。}@

@{云何恭敬。}@
%{什么是恭敬呢？}%
@{答諸有恭敬有恭敬性。有自在有自在性。於自在者有怖畏轉。是謂恭敬。}@
%{回答说，凡是有恭敬的，就有恭敬的本质。凡是有自在的，就有自在的本质。对自在者，有敬畏心，这就是恭敬。}%
@{此本論師於異文義得善巧故。以種種文顯示恭敬。而體無別恭敬亦以慚為自性。}@
%{这位论师对不同的文义都很精通，所以用各种文字来显示恭敬。但实际上，恭敬的本质没有区别，都是以惭愧为本性的。}%
