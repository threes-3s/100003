@{有七隨眠。謂欲貪隨眠瞋恚隨眠有貪隨眠慢隨眠無明隨眠見隨眠疑隨眠。}@
%{有七种随眠，分别是：欲贪随眠、瞋恚随眠、有贪随眠、慢随眠、无明随眠、见随眠、疑随眠。}%
@{問此七隨眠以何為自性。}@
%{请问这七种随眠，其本质是什么呢？}%
@{答以九十八事為自性。}@
%{回答说，以九十八件事作为其本质。}%
@{謂欲貪瞋恚隨眠各欲界五部為十事。}@
%{也就是说，欲贪随眠、瞋恚随眠，各自在欲界五部中，各有十件事。}%
@{有貪隨眠色無色界各五部為十事。}@
%{有贪随眠，在色界和无色界各五部中，也各有十件事。}%
@{慢無明隨眠各三界五部為三十事。}@
%{慢随眠和无明随眠，各自在三界五部中，各有三十件事。}%
@{見隨眠三界各十二為三十六事。}@
%{见随眠，在三界中，各自有十二件事，共计三十六件事。}%
@{疑隨眠三界各四部為十二事。}@
%{疑随眠，在三界中，各自有四件事，共计十二件事。}%
@{由此七隨眠以九十八事為自性。}@
%{由此可知，这七种随眠以九十八件事作为其本质。}%
@{已說自性所以今當說。}@
%{已经说了其本质，现在就来说说为什么叫它随眠。}%
@{問何故名隨眠。}@
%{请问，为什么叫做随眠呢？}%
@{隨眠是何義}@
%{随眠是什么意思呢？}%
@{答微細義隨增義隨縛義是隨眠義。}@
%{回答说，随眠的意思是指微细义、随增义、随缚义。}%
@{微細義是隨眠義者。}@
%{微细义是随眠义，指的是：}%
@{欲貪等七行相微細。}@
%{欲贪等七种随眠，其行相非常微细。}%
@{如七極微成一細色。}@
%{就像七个极微组成一个细色一样。}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
@{欲貪等七普於一切微細有漏皆悉隨增。}@
%{欲贪等七种随眠，普遍地依附于一切微细的有漏法，不断地随增。}%
@{乃至一極微。}@
%{即使是一极微，}%
@{或一剎那頃欲貪等七皆隨增故。}@
%{或是一刹那间，欲贪等七种随眠都会随之增长。}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
@{如空行影水行隨故。}@
%{就像空行鸟的影子被水行鱼追逐一样。}%
@{空行謂鳥。}@
%{空行指的是鸟。}%
@{水行謂魚。}@
%{水行指的是鱼。}%
@{鳥以翅力欲度大海。}@
%{鸟用翅膀的力量想要渡过大海。}%
@{水中有魚善取其相而作是念。}@
%{水里的鱼，善于观察鸟的形相，心想：}%
@{無有飛鳥能過大海。}@
%{没有飞鸟能够渡过大海。}%
@{唯除勇迅妙翅鳥王。}@
%{只有勇猛迅疾、拥有妙翅的鸟王。}%
@{即逐其影鳥乏墮水魚便吞之。}@
%{鱼便追逐鸟的影子，等到鸟力竭而落入水中，鱼就一口吞下它。}%
@{如是隨眠於一切位恒現起得非理作意。}@
%{就像这样，随眠在一切位中，恒常现起，使人获得非理的作意。}%
@{若現前時即受等流或異熟果}@
%{如果现前的时候，就会受生等流果或异熟果。}%
@{復次微細義是隨眠義者。}@
%{再次，微细义是随眠义，指的是：}%
@{依自性說}@
%{根据其本质来说，}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
@{依作用說。}@
%{根据其作用来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
@{依彼得說。}@
%{根据其所获得的来说。}%
@{復次微細義是隨眠義者。}@
%{再次，微细义是随眠义，指的是：}%
@{依自性說。}@
%{根据其本质来说，}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
@{依相續說。}@
%{根据其相续来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
@{依習氣堅牢說。}@
%{根据其习气坚牢来说。}%
@{復次微細義是隨眠義者。}@
%{再次，微细义是随眠义，指的是：}%
@{依過去隨眠說。}@
%{根据过去的随眠来说，}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
%{依現在隨眠說。}%
%{根据现在的随眠来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
%{依未來隨眠說。}%
%{根据未来的随眠来说。}%
@{復次微細義是隨眠義者。}@
%{再次，微细义是随眠义，指的是：}%
@{依行相說。}@
%{根据其行相来说，}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
%{依所緣縛說。}%
%{根据其所缘缚来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
%{依相應縛說。}%
%{根据其相应缚来说。}%
@{復次微細義隨增義是隨眠義者。}@
%{再次，微细义、随增义是随眠义，指的是：}%
@{依相應隨眠說。}@
%{根据其相应随眠来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
%{依不相應隨眠說。}%
%{根据其不相应随眠来说。}%
@{問隨眠皆與心等相應。}@
%{请问，随眠都与心等相应，}%
@{如何言依不相應說。}@
%{怎么可以说根据不相应来说呢？}%
@{答此中於得立隨眠名得隨眠故。}@
%{回答说，这里是指获得随眠的名称，所以才说获得随眠。}%
@{說名隨眠。}@
%{所以叫做随眠。}%
@{外國諸師作如是說。}@
%{外國的老师们是这样说的：}%
@{由四種義故名隨眠。}@
%{由于四种意思，所以叫做随眠。}%
@{謂微細義隨入義隨增義隨縛義。}@
%{这四种意思是指：微细义、随入义、随增义、随缚义。}%
@{是隨眠義。}@
%{这是随眠的意思。}%
@{微細義是隨眠義者。}@
%{微细义是随眠义，指的是：}%
@{謂欲貪等自性行相俱極微細。}@
%{欲贪等随眠的本质和行相，都非常微细。}%
@{隨入義是隨眠義者。}@
%{随入义是随眠义，指的是：}%
%{謂欲貪等隨入相續無不周遍。}%
%{欲贪等随眠，依附于相续，无处不遍。}%
@{如油在麻膩在團中無不周遍。}@
%{就像油在麻中，腻在团中，无处不遍一样。}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
%{謂欲貪等於相續中展轉隨增。}%
%{欲贪等随眠，在相续中不断地增长。}%
@{如孩乳母。}@
%{就像孩子依附于乳母一样。}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
@{如空行影水行隨逐。}@
%{就像空行鸟的影子被水行鱼追逐一样。}%
@{復次微細義是隨眠義者。}@
%{再次，微细义是随眠义，指的是：}%
@{依自性說。}@
%{根据其本质来说，}%
@{隨入義是隨眠義者。}@
%{随入义是随眠义，指的是：}%
@{依相應說。}@
%{根据其相应来说，}%
@{隨增義是隨眠義者。}@
%{随增义是随眠义，指的是：}%
@{依行相說。}@
%{根据其行相来说，}%
@{隨縛義是隨眠義者。}@
%{随缚义是随眠义，指的是：}%
@{依彼得說}@
%{根据其所获得的来说。}%
@{應以三事知諸隨眠。}@
%{应该用三件事来了解诸随眠。}%
@{一以自性故。}@
%{第一，根据其本质来说，}%
@{二以果故。}@
%{第二，根据其果报来说，}%
@{三以補特伽羅故。}@
%{第三，根据补特伽罗来说。}%
@{以自性故者。}@
%{根据其本质来说，}%
@{欲貪隨眠如食興蕖瞋恚隨眠如食辛辣。}@
%{欲贪随眠就像吃兴蕖一样，瞋恚随眠就像吃辛辣一样。}%
@{有愛隨眠如乳母衣。}@
%{有爱随眠就像乳母的衣服一样。}%
@{慢隨眠如憍傲人。}@
%{慢随眠就像骄傲的人一样。}%
@{無明隨眠如盲瞽者。}@
%{无明随眠就像盲人一样。}%
@{見隨眠如失道者。}@
%{见随眠就像迷失道路的人一样。}%
@{疑隨眠如臨岐路。}@
%{疑随眠就像站在岔路口的人一样。}%
@{以果故者。}@
%{根据其果报来说，}%
@{欲貪隨眠若習若修若多所作。}@
%{如果经常习练、修行、多做欲贪随眠的业，}%
@{當生鴿雀鴛鴦等中。}@
%{就会生为鸽子、雀鸟、鸳鸯等。}%
@{瞋恚隨眠若習若修若多所作當生蜂蝎毒蛇等中。}@
%{如果经常习练、修行、多做瞋恚随眠的业，就会生为蜂、蝎、毒蛇等。}%
@{有貪隨眠若習若修若多所作當生色無色界。}@
%{如果经常习练、修行、多做有贪随眠的业，就会生为色界和无色界。}%
@{慢隨眠若習若修若多所作當生卑賤種族。}@
%{如果经常习练、修行、多做慢随眠的业，就会生为卑贱的种族。}%
@{無明隨眠若習若修若多所作當生愚盲種族。}@
%{如果经常习练、修行、多做无明随眠的业，就会生为愚昧的种族。}%
@{見隨眠若習若修若多所作當生外道種族。}@
%{如果经常习练、修行、多做见随眠的业，就会生为外道的种族。}%
@{疑隨眠若習若修若多所作當生邊鄙種族。}@
%{如果经常习练、修行、多做疑随眠的业，就会生为边鄙的种族。}%
@{以補特伽羅故者。}@
%{根据补特伽罗来说，}%
@{欲貪隨眠如難陀等。}@
%{欲贪随眠就像难陀等一样。}%
@{瞋恚隨眠如氣噓指鬘等。}@
%{瞋恚随眠就像气嘘指鬘等一样。}%
@{有貪隨眠如遏璽多阿邏荼嗢達洛迦等。}@
%{有贪随眠就像遏玺多阿邏荼嗢達洛迦等一样。}%
@{慢隨眠如傲士等。}@
%{慢随眠就像傲慢的人一样。}%
@{無明隨眠如鄔盧頻螺婆迦葉波等。}@
%{无明随眠就像鄔盧頻螺婆迦葉波等一样。}%
@{見隨眠如善星等。}@
%{见随眠就像善星等一样。}%
@{疑隨眠如摩洛迦子等。}@
%{疑随眠就像摩洛迦子等一样。}%
@{問嫉慳何故不立隨眠。}@
%{请问，嫉妒和悭吝为什么不立为随眠呢？}%
@{答彼二無有隨眠相故。}@
%{回答说，这两个没有随眠的相。}%
@{復次隨眠微細彼二麁動。}@
%{再次，随眠微细，这两个则比较粗重。}%
@{復次隨眠輕微彼二尤重。}@
%{再次，随眠比较轻微，这两个则更加沉重。}%
@{復次隨眠猛利彼二數行。}@
%{再次，随眠比较猛烈，这两个则比较缓慢。}%
@{復次隨眠是根本煩惱彼二是煩惱等流。}@
%{再次，随眠是根本烦恼，这两个是烦恼的等流。}%
@{謂嫉是瞋恚等流。}@
%{也就是说，嫉妒是瞋恚的等流。}%
@{慳是欲貪等流。}@
%{悭吝是欲贪的等流。}%
@{復次隨眠習氣堅固如於此地燒擔山木火滅雖久其地猶熱。}@
%{再次，随眠的习气坚固，就像在这个地方烧担山木，火虽然灭了很久，但是土地依然发热一样。}%
@{彼二習氣不堅固。}@
%{这两个的习气不坚固。}%
@{如於此地燒草樺皮。}@
%{就像在这个地方烧草桦皮一样，}%
@{火纔滅已其地便冷。}@
%{火一灭，土地就变凉了。}%
@{復次隨眠難伏。}@
%{再次，随眠很难降伏。}%
@{彼二易伏。}@
%{这两个则很容易降伏。}%
@{是故彼二不立隨眠。}@
%{所以，这两个不立为随眠。}%
@{餘纏及垢准二應說。}@
%{其他的缠缚和垢染，也应该按照这两个来解释。}%
@{有九結。謂愛結恚結慢結無明結見結取結疑結嫉結慳結。}@
%{有九种烦恼结，分别是：爱结、恚结、慢结、无明结、见结、取结、疑结、嫉结、悭结。}%
@{問此九結以何為自性。}@
%{问这九种烦恼结以什么为本性呢？}%
@{答以百事為自性。}@
%{答：以一百件事为本性。}%
@{謂愛慢無明結。}@
%{就是说，爱、慢、无明这三种结。}%
@{各三界五部為四十五事。}@
%{分别以三界五部为四十五件事。}%
@{恚結唯欲界五部為五事。}@
%{恚结只在欲界五部为五件事。}%
@{見結有十八事。}@
%{见结有十八件事。}%
@{謂有身見邊執見各三界見苦所斷為六事。}@
%{就是说：有身见、边执见，分别以三界见苦所断为六件事。}%
@{邪見三界各四部為十二事。}@
%{邪见以三界各四部为十二件事。}%
@{取結有十八事。}@
%{取结有十八件事。}%
@{謂見取三界各四部為十二事。}@
%{就是说：见取以三界各四部为十二件事。}%
@{戒禁取三界各見苦道所斷為六事。}@
%{戒禁取以三界各见苦道所断为六件事。}%
@{疑結三界各四部為十二事。}@
%{疑结以三界各四部为十二件事。}%
@{嫉慳結各欲界修所斷為二事。}@
%{嫉结、悭结分别以欲界修所断为两件事。}%
@{由此九結以百事為自性。}@
%{由此可知，九种烦恼结以一百件事为本性。}%
@{已說自性所以今當說。}@
%{已经说了它们的本性，现在就来说说。}%
@{問何故名結。}@
%{问：为什么叫它们为结呢？}%
@{結是何義。}@
%{结是什么意思呢？}%
@{答繫縛義合苦義雜毒義是結義。}@
%{答：结就是束缚的意思，包含着痛苦的意思，也包含着混合着毒素的意思。}%
@{所餘廣釋如三結處已釋。}@
%{其他的详细解释，如《三结品》中已经解释过了。}%
@{諸結總義一一自性今當廣說。}@
%{现在就详细解释一下各种结的总义和各自的本性。}%
@{云何愛結。}@
%{什么是爱结呢？}%
@{謂三界貪。}@
%{就是说：三界贪。}%
@{然三界貪於九結中總立愛結。}@
%{然而，三界贪在九种烦恼结中总称为爱结。}%
@{七隨眠中立二隨眠。}@
%{在七种随眠中，立为两种随眠。}%
@{謂欲界貪名欲貪隨眠。}@
%{就是说：欲界贪，名为欲贪随眠。}%
@{色無色界貪名有貪隨眠。}@
%{色界、无色界贪，名为有贪随眠。}%
@{於餘經中立為三愛。}@
%{在其他的经文中，立为三爱。}%
@{謂欲愛色愛無色愛。}@
%{就是说：欲爱、色爱、无色爱。}%
@{問此三何別。}@
%{问：这三者有什么区别呢？}%
@{答世尊所化根有三品。}@
%{答：世尊所化的众生根器有三种。}%
@{為利根者說一愛結。}@
%{对利根者，就说一种爱结。}%
@{為中根者說二隨眠。}@
%{对中根者，就说两种随眠。}%
@{為鈍根者說三界愛。}@
%{对钝根者，就说三界爱。}%
@{復次世尊所化修有三種。}@
%{其次，世尊所化的修行也有三种。}%
@{為初習業者說一愛結。}@
%{对初学者，就说一种爱结。}%
@{為已熟修者說二隨眠。}@
%{对已经熟练修行者，就说两种随眠。}%
@{為超作意者說三界愛。}@
%{对超越作意者，就说三界爱。}%
@{復次世尊所化樂有三種。}@
%{其次，世尊所化的快乐也有三种。}%

@{云何恚結。}@
%{什么是恚结呢？}%

%{就是说：对有情众生想要伤害他们。}%
@{問若於非情欲為損害亦應是恚。}@

%{为什么不说呢？}%

%{答：因为从多数情况来说。}%

%{就是说，这种恚结大多是对有情众生想要伤害他们。}%

%{对无情很少。}%
@{是故不說。}@
%{所以不说。}%

%{其次，从严重性来说。}%

%{就是说，对有情众生想要伤害他们，罪过非常严重。}%
@{非於非情。}@
%{对无情则不然。}%
@{是故不說復次從本說故。}@
%{所以不说。其次，从根本来说。}%

%{就是说，这种恚结一定要对有情众生想要伤害他们。}%

%{然后才会对无情也产生恚结。}%
@{是故不說。}@
%{所以不说。}%

@{阿素洛宮有端正女勝彼諸天。}@
%{阿修罗宫里，有一个端正的女子，比天界里的那些天女都漂亮。}%
@{天自慳味嫉他美女。}@
%{天界的天人自己吝啬美味，嫉妒阿修罗的美女。}%
@{非天慳女嫉他美味。}@
%{非天人吝啬美女，嫉妒天人的美味。}%
@{天為美女往非天處。}@
%{天人为了美女，就到非天界去。}%
@{非天為味復往天宮。}@
%{非天人为了美味，就回到天宫去。}%
@{是故諸天與阿素洛由嫉慳結數興戰鬪。}@
%{所以，天界的天人和阿修罗，由于嫉妒和吝啬的结，经常发生战争。}%
@{爾時天帝與阿素洛適鬪戰已。}@
%{当时，天帝和阿修罗正好打完仗。}%
@{心猶恐怖來詣佛所。}@
%{内心还感到恐惧，就来到佛陀那里。}%
@{作如是問。}@
%{问佛陀说。}%
@{由何結故人天及龍阿素洛等屢興戰鬪。}@
%{由于什么结，所以人天、龙、阿修罗等等经常发生战争呢？}%
@{彼意問言由何結故天與非天數興戰鬪。}@
%{他的意思是想问，由于什么结，所以天界的天人和非天人经常发生战争呢？}%
@{故佛告曰。}@
%{所以，佛陀说。}%
@{由嫉與慳。}@
%{由于嫉妒和吝啬。}%
@{佛意告言。}@
%{佛陀的意思是说。}%
@{汝等天眾及阿素洛由嫉慳結數興戰鬪。}@
%{你们这些天人和阿修罗，由于嫉妒和吝啬的结，经常发生战争。}%
@{故嫉與慳是汝等患亦是重擔傷害汝等應速捨離。}@
%{所以，嫉妒和吝啬是你们的病患，也是沉重的负担，伤害你们，应该尽快舍弃。}%
@{問六煩惱垢何故非結。}@
%{问：六种烦恼垢，为什么不是结呢？}%
%{答相麁動故。}%
%{答：因为它们的相粗糙、动荡。}%
@{若相微細繫縛堅牢可立為結。}@
%{如果相细微、束缚牢固，就可以立为结。}%
@{垢相麁動繫義不堅故不立結。}@
%{烦恼垢的相粗糙、动荡，束缚的意思不牢固，所以不立为结。}%
@{有九十八隨眠。謂欲界繫三十六隨眠。色無色界繫各三十一隨眠。此即以九十八事為自性。隨眠名義如前已釋。問何故說此九十八隨眠耶。}@
%{有九十八種煩惱，叫做隨眠。其中欲界有三十六種隨眠，色界和無色界各三十一種隨眠。這九十八種煩惱就是指我們本性的煩惱。關於隨眠的名稱和含義，前面已經解釋過了。有人問：為什麼要說這九十八種隨眠呢？}%

@{答是作論者意欲爾故。謂本論師隨欲作論不違法相故不應責。}@
%{回答：因為寫這部論著的人想這樣說。意思是說，論著的作者想怎麼說就怎麼說，只要不違背佛法，就不應該責怪他。}%

@{復次為止著文沙門意故。謂有沙門執著文字離經所說終不敢言。彼作是說誰有智慧過於佛者。}@
%{另外，也是為了阻止那些只執著文字的沙門。有些沙門只執著於佛經上的文字，離開了佛經所說的道理，就什麼也不敢說。他們這樣說，難道誰的智慧比佛陀還高嗎？}%

@{佛唯說有七種隨眠如何強增為九十八。}@
%{佛陀只說了七種隨眠，怎麼能強行增加到九十八種呢？}%

@{為遮彼意廣七隨眠為九十八。謂依行相界部別故。}@
%{為了反駁他們，我們把七種隨眠擴展為九十八種。這是根據隨眠的行為、所繫界和部別來分的。}%

@{七隨眠中欲貪隨眠部別故為五。瞋恚隨眠亦爾。有貪隨眠界別故為二。部別故為五。界部別故為十。}@
%{七種隨眠中，欲貪隨眠根據部別分為五種；瞋恚隨眠也是一樣。有貪隨眠根據所繫界分為兩種，根據部別分為五種，根據所繫界和部別分為十種。}%

@{慢隨眠界別故為三。部別故為五。界部別故為十五。無明隨眠亦爾。}@
%{慢隨眠根據所繫界分為三種，根據部別分為五種，根據所繫界和部別分為十五種。無明隨眠也是一樣。}%

@{見隨眠界別故為三。行相別故為五。部別故為十二。行相界部別故為三十六。}@
%{見隨眠根據所繫界分為三種，根據行為分為五種，根據部別分為十二種，根據行為、所繫界和部別分為三十六種。}%

@{疑隨眠界別故為三。部別故為四。界部別故為十二。}@
%{疑隨眠根據所繫界分為三種，根據部別分為四種，根據所繫界和部別分為十二種。}%

@{是故七隨眠依行相界部別故為九十八隨眠。廣略雖異而體無差別。}@
%{所以，七種隨眠根據行為、所繫界和部別，就變成了九十八種隨眠。雖然說法上有所不同，但本質上沒有差別。}%

@{三結乃至九十八隨眠幾不善幾無記。}@
%{三結到九十八種隨眠，哪些是不善的，哪些是無記的呢？}%

@{問何故作此論。}@
%{有人問：為什麼要寫這部論著呢？}%

@{答為止他宗顯正義故謂或有說。一切煩惱皆是不善。由不巧便所攝持故。}@
%{回答：為了阻止其他宗派的錯誤觀點，顯明佛法的正理。有些人說，一切煩惱都是不善的，因為它們是由不巧便所攝持的。}%

@{如譬喻者。為遮彼意顯諸煩惱。有是不善有是無記。}@
%{就像用比喻來解釋一樣，為了反駁他們的觀點，我們要說明，煩惱有的是不善的，有的是無記的。}%

@{若諸煩惱由不巧便所攝持故是不善者。此不巧便應非不善非不巧便所攝持故。}@
%{如果說所有煩惱都是因為不巧便所攝持，所以是不善的，那麼這個不巧便就應該不是不善的，它也不應該是被不巧便所攝持的。}%

@{不巧便者即是無知。所攝持者是相應義。自體不與自體相應故不巧便應非不善。}@
%{不巧便就是無知，所攝持是指相應的意思。因為自身不會與自身相應，所以不巧便就不應該是不善的。}%

@{復有欲令欲界煩惱皆是不善。色無色界一切煩惱皆是無記。}@
%{還有些人想讓欲界的煩惱都是不善的，而色界和無色界的煩惱都是無記的。}%

@{為遮彼意顯示欲界有身見邊執見及彼相應無明亦是無記。}@
%{為了反駁他們，我們要說明，欲界中，身見、邊執見以及它們相應的無明，也是無記的。}%

@{復有說者。欲現門義故作斯論。謂前已說何故於此先立章耶。}@
%{還有些人說，為了要顯現佛法的門義，才寫了這部論著。他們說：前面已經說了，為什麼要先立章呢？}%

@{為欲顯示諸門義故。若不立章門義無由得顯。如彩畫者不能彩畫虛空。}@
%{為了要顯現佛法的門義。如果沒有先立章，佛法的門義就沒有辦法顯現。就像畫家不能在虛空中作畫一樣。}%

@{既立章已應顯門義。答三結中一無記謂有身見。}@
%{既然已經立章了，就應該要顯現佛法的門義。回答：三結中，有一種是無記的，就是身見。}%

@{問何故有身見是無記耶。}@
%{有人問：為什麼身見是無記的呢？}%

@{答若法是無慚無愧自性與無慚無愧相應。是無慚無愧等起等流果者是不善。}@
%{回答：如果一種法是無慚無愧的本性，並且與無慚無愧相應，而且是無慚無愧的等起和等流果，那麼它就是不善的。}%

@{有身見非無慚無愧自性不與無慚無愧相應。非無慚無愧等起等流果。故是無記。}@
%{身見不是無慚無愧的本性，它不與無慚無愧相應，也不是無慚無愧的等起和等流果，所以它是無記的。}%

@{復次此有身見非一向壞意樂。故非不善。無慚無愧不相應故非一向壞意樂。}@
%{另外，身見不是完全破壞意樂的，所以它不是不善的。因為它不與無慚無愧相應，所以它不是完全破壞意樂的。}%

@{復次此見不違施戒修故。謂執我者作如是說。由布施故我當富樂。由持戒故我當生天由修定故我當解脫。}@
%{另外，身見不違背布施、持戒和修行的功德。執著我的人會這樣說：因為布施，所以我會富貴快樂；因為持戒，所以我會生天；因為修定，所以我會解脫。}%

@{故是無記。復次此有身見唯迷自體。不逼惱他故是無記。}@
%{所以它是無記的。另外，身見只迷惑自身，不逼迫傷害他人，所以它是無記的。}%

@{謂執我者眼見色時言我見色色是我所。廣說乃至意了法時言我了法法是我所。}@
%{執著我的人，當他看到顏色時，就會說：我看到顏色，顏色是我的；廣泛地說，當他用意識了知法的時候，就會說：我了知法，法是我的。}%

@{雖於自體有此倒執。而不惱他故是無記。復次此有身見無異熟果故是無記。}@
%{雖然他對自身有這種錯誤的執著，但他不傷害他人，所以它是無記的。另外，身見沒有異熟果，所以它是無記的。}%

@{尊者世友作如是說。此有身見不能發起麁身語業。故是無記。}@
%{尊者世友說：身見不能引起粗重的身、語、意業，所以它是無記的。}%

@{問不善煩惱亦有不能起麁身語業者。應是無記。}@
%{有人問：不善的煩惱也有不能引起粗重的身、語、意業的，那也應該算作無記的。}%

@{答貪瞋癡慢若增盛時必能發起麁身語業。此有身見設增盛時亦不能起麁身語業。故是無記}@
%{回答：貪、瞋、癡、慢如果增盛的時候，一定會引起粗重的身、語、意業。但是身見即使增盛，也不能引起粗重的身、語、意業，所以它是無記的。}%

@{復次此有身見不令有情墮諸惡趣。故是無記。}@
%{另外，身見不會讓有情眾生墮入惡趣，所以它是無記的。}%

@{問不善煩惱亦有不令墮惡趣者。應是無記。}@
%{有人問：不善的煩惱也有不讓有情眾生墮入惡趣的，那也應該算作無記的。}%

@{答不善煩惱若增盛時必令有情墮諸惡趣。此有身見設增盛時亦終不令墮諸惡趣。故是無記}@
%{回答：不善的煩惱如果增盛的時候，一定會讓有情眾生墮入惡趣。但是身見即使增盛，也不會讓有情眾生墮入惡趣，所以它是無記的。}%

@{復次此見不能感非愛果。故是無記。問此見既令後有相續。後有即是非愛果攝。如何不能感非愛果。}@
%{另外，身見不能感召非愛果，所以它是無記的。有人問：既然身見讓後有相續，而後有就是屬於非愛果的範圍，怎麼說它不能感召非愛果呢？}%

@{如契經說。苾芻當知。我終不讚起後有者。所以者何。若起後有一剎那者則為增苦。苦者即是非愛果攝。}@
%{就像佛經中說：比丘應該知道，我從來不讚嘆後有。為什麼呢？因為如果有後有，即使只是一剎那，也會增加苦惱。苦惱就是屬於非愛果的範圍。}%

@{答此中所說非愛果者是苦苦類。契經所說非愛果者通三苦類。此有身見令有相續。非苦苦類故不相違。}@
%{回答：這裡所說的非愛果，是指苦苦類。佛經中所說的非愛果，包括三苦類。身見讓後有相續，但它不是苦苦類，所以不矛盾。}%

@{復次此有身見雖起後有。為苦苦本說為增苦。而不與彼為異熟因。故是無記。}@
%{另外，雖然身見讓後有相續，但它是作為苦苦的根本，也就是增加苦惱，它不是後有的異熟因，所以它是無記的。}%

@{大德說曰。此有身見是顛倒執是不安隱是愚癡類。故是不善。}@
%{大德說：身見是顛倒執著，是不安穩的，是愚癡類，所以是不善的。}%

@{若有身見非不善者。更有何法可名不善。如世尊說。乃至愚癡皆是不善。}@
%{如果身見不是不善的，那麼還有什麼法可以叫做不善呢？就像世尊說：直到愚癡都是不善的。}%

@{彼說不應理非異熟因故。若有身見皆不善者色無色界應有苦苦。}@
%{他們說：身見不合理，它不是異熟因。如果身見都是不善的，那麼色界和無色界就應該有苦苦。}%

@{然世尊說乃至愚癡皆不善者。非巧便故說為不善。不言能感不愛果故二應分別。}@
%{但是世尊說，直到愚癡都是不善的，是因為不巧便，所以說它是
不善的。世尊沒有說它能感召非愛果，所以要區分兩種情況。}%

@{謂戒禁取疑結或不善或無記。問應分別者義何謂耶。}@
%{也就是說，戒禁取和疑結，有的是不善的，有的是無記的。有人問：應分別的意思是什麼呢？}%

@{答應分析故名應分別。謂後二結。一分是不善一分是無記。故應分別。}@
%{回答：因為要分析，所以叫做應分別。意思是說，後面的兩個結，一部分是不善的，一部分是無記的，所以要區分開來。}%

@{分別論者作如是言。所問二結應分別記非一向等。由此故言二應分別。謂彼二結欲界是不善。色無色界是無記。}@
%{分別論者說：所問的兩個結，應分別記，它們的性質並不完全相同。因此，才說兩個結要區分開來。意思是說，那兩個結，在欲界是不善的，在色界和無色界是無記的。}%

@{問何故色無色界煩惱是無記耶。}@
%{有人問：為什麼色界和無色界的煩惱是無記的呢？}%

@{答若法是無慚無愧自性與無慚無愧相應。是無慚無愧等起等流果者是不善。色無色界煩惱不爾。故是無記。}@
%{回答：如果一種法是無慚無愧的本性，並且與無慚無愧相應，而且是無慚無愧的等起和等流果，那麼它就是不善的。色界和無色界的煩惱不是這樣，所以它是無記的。}%

@{復次色無色界煩惱非一向壞意樂。故非不善。無慚無愧不相應故非一向壞意樂。}@
%{另外，色界和無色界的煩惱不是完全破壞意樂的，所以它不是不善的。因為它不與無慚無愧相應，所以它不是完全破壞意樂的。}%

@{復次色無色界煩惱無異熟果。故是無記。問因論生論。何故色無色界煩惱無異熟果耶。}@
%{另外，色界和無色界的煩惱沒有異熟果，所以它是無記的。有人問：根據因論和生論，為什麼色界和無色界的煩惱沒有異熟果呢？}%

@{答四支五支定所伏故。如毒蛇等呪術所伏不能為害。此亦如是。}@
%{回答：因為它們被四支定和五支定所伏，就像毒蛇被法術所伏，不能傷害人一樣。}%

@{復次上界無彼異熟器故。若彼煩惱有異熟者應是苦受。苦受必是欲界所繫。不應上界煩惱異熟是欲界繫。故彼煩惱定無異熟。}@
%{另外，上界沒有那些異熟果的器皿。如果那些煩惱有異熟果，就應該是有苦受。苦受一定是由欲界所繫，不應該上界的煩惱異熟果是由欲界所繫。所以那些煩惱一定沒有異熟果。}%

@{復次彼邪見等非極顛倒。於分相似處所起故。不惱他故但是無記。}@
%{另外，那些邪見等等，不是極端的顛倒，它們是在部分相似的地方產生的，它們不傷害他人，所以只是無記的。}%

@{謂彼邪見謗言無苦。然上二界有相似樂。上界見取執彼諸蘊以為第一。然彼亦有相似第一。}@
%{意思是說，那些邪見誹謗說沒有苦。但是，上兩個界有相似的快樂。上界見取執著那些蘊，認為它們是第一。但是，它們也有相似的第一。}%

@{彼戒禁取執彼諸蘊以為能淨。然彼亦有相似能淨。謂色界道能淨欲界。無色界道能淨色界。故彼煩惱定非不善。}@
%{那些戒禁取執著那些蘊，認為它們能淨化。但是，它們也有相似的淨化。意思是說，色界的道能淨化欲界，無色界的道能淨化色界。所以那些煩惱一定不是不善的。}%

@{尊者世友作如是說。上界煩惱不能發起麁身語業。故是無記。}@
%{尊者世友說：上界的煩惱不能引起粗重的身、語、意業，所以它是無記的。}%

@{問不善煩惱亦有不能起麁身語業者應是無記。}@
%{有人問：不善的煩惱也有不能引起粗重的身、語、意業的，那也應該算作無記的。}%

@{答不善煩惱若增盛時必能發起麁身語業。上界煩惱設增盛時亦不能起麁身語業。故是無記。}@
%{回答：不善的煩惱如果增盛的時候，一定會引起粗重的身、語、意業。但是上界的煩惱即使增盛，也不能引起粗重的身、語、意業，所以它是無記的。}%

@{復次上界煩惱不令有情墮諸惡趣。故是無記。問不善煩惱亦有不令墮惡趣者應是無記。}@
%{另外，上界的煩惱不會讓有情眾生墮入惡趣，所以它是無記的。有人問：不善的煩惱也有不讓有情眾生墮入惡趣的，那也應該算作無記的。}%

@{答不善煩惱若增盛時必令有情墮諸惡趣。上界煩惱設增盛時亦終不令墮諸惡趣。故是無記。}@
%{回答：不善的煩惱如果增盛的時候，一定會讓有情眾生墮入惡趣。但是上界的煩惱即使增盛，也不會讓有情眾生墮入惡趣，所以它是無記的。}%

@{復次彼惑不能感非愛果。故是無記。問彼惑既令後有相續。後有即是非愛果攝。如何不能感非愛果。}@
%{另外，那些惑不能感召非愛果，所以它是無記的。有人問：既然那些惑讓後有相續，而後有就是屬於非愛果的範圍，怎麼說它不能感召非愛果呢？}%

@{如契經說。苾芻當知。我終不讚起後有者。所以者何。若起後有一剎那者則為增苦。苦者即是非愛果攝。}@
%{就像佛經中說：比丘應該知道，我從來不讚嘆後有。為什麼呢？因為如果有後有，即使只是一剎那，也會增加苦惱。苦惱就是屬於非愛果的範圍。}%

@{答此中所說非愛果者是苦苦類。契經所說非愛果者通三苦類。上界煩惱令有相續。非苦苦類故不相違。}@
%{回答：這裡所說的非愛果，是指苦苦類。佛經中所說的非愛果，包括三苦類。上界的煩惱讓後有相續，但它不是苦苦類，所以不矛盾。}%

@{大德說曰。上界煩惱若是無記更有何法可名不善。如世尊說。若諸煩惱能發起業皆是不善。}@
%{大德說：如果上界的煩惱是無記的，那麼還有什麼法可以叫做不善呢？就像世尊說：如果那些煩惱能引起業，那麼它們都是不善的。}%

@{彼說非理。若彼煩惱是不善者。色無色界應有苦苦。然世尊說若諸煩惱能發起業皆不善者。依惡業說故不相違。}@
%{他們說：這是沒有道理的。如果那些煩惱是不善的，那麼色界和無色界就應該有苦苦。但是世尊說，如果那些煩惱能引起業，那麼它們都是不善的，這是根據惡業來說的，所以不矛盾。}%

@{三不善根唯不善。以彼自性是不善。復與一切不善法為因。為本為道路為由序為能作為生為緣為有為集為等起故。}@
%{三不善根，只有不善。因為它們的本性是不善的，而且它們是所有不善法的原因，是根本、道路、由緒、能作、生、緣、有、集、等起。}%

@{三漏中一無記謂有漏。由上所說諸因緣故。色無色界一切煩惱皆是無記。}@
%{三漏中，有一種是無記的，就是有漏。因為上面說過的那些因緣，所以色界和無色界的煩惱都是無記的。}%

@{二應分別。謂欲漏或不善或無記。無慚無愧及彼相應是不善。}@
%{要區分兩種情況。意思是說，欲漏有的是不善的，有的是無記的。無慚無愧以及它們相應的煩惱，是不善的。}%

@{無慚無愧者顯彼自性唯是不善。及彼相應者。顯欲漏中三十四事及三少分亦是不善。}@
%{無慚無愧，顯明它們的本性就是不善的。它們相應的煩惱，也顯明欲漏中的三十四種，以及三少分，也是不善的。}%

@{三少分者。謂彼相應惛沈睡眠掉舉少分。餘是無記。謂欲漏中有身見邊執見及三少分皆是無記。}@
%{三少分是指它們相應的惛沈、睡眠、掉舉少分。其他的是無記的。意思是說，欲漏中的身見、邊執見，以及三少分，都是無記的。}%

@{三少分者。謂有身見邊執見相應惛沈睡眠掉舉少分。如是五法無慚無愧不相應故皆非不善。}@
%{三少分是指身見、邊執見相應的惛沈、睡眠、掉舉少分。這五種法，不與無慚無愧相應，所以都不是不善的。}%

@{問無慚不與無慚相應。無愧不與無愧相應。豈是無記。}@
%{有人問：無慚不與無慚相應，無愧不與無愧相應，難道是無記的嗎？}%

@{答無慚雖不與無慚相應。而與無愧相應。無愧雖不與無愧相應。而與無慚相應。俱不相應者方是無記故。}@
%{回答：無慚雖然不與無慚相應，但它與無愧相應。無愧雖然不與無愧相應，但它與無慚相應。只有同時不與無慚無愧相應，才是無記的。}%

@{無明漏或不善或無記。無慚無愧相應是不善。謂欲界見集滅道及修所斷無明唯是不善。}@
%{無明漏，有的是不善的，有的是無記的。無慚無愧相應的無明，是不善的。意思是說，欲界的見、集、滅、道，以及修所斷的無明，只有不善的。}%

@{見苦所斷三見疑慢貪瞋相應及不共無明亦是不善。餘是無記。}@
%{見苦所斷的三見，疑、慢、貪、瞋相應的無明，以及不共無明，也是不善的。其他的是無記的。}%

@{謂欲界二見相應及色無色界一切無明。無慚無愧不相應故皆是無記。}@
%{意思是說，欲界的二見相應的無明，以及色界和無色界的所有無明，都不與無慚無愧相應，所以都是無記的。}%

@{問何故十纏中唯說與無慚無愧相應耶。}@
%{有人問：為什麼十纏中，只說與無慚無愧相應呢？}%

@{答是作論者意欲爾故。謂作論者隨欲造論不違法相故不應責。}@
%{回答：因為寫這部論著的人想這樣說。意思是說，論著的作者想怎麼說就怎麼說，只要不違背佛法，就不應該責怪他。}%

@{復次此二唯是不善遍與一切不善心相應。是故偏說。忿覆嫉慳雖唯是不善。而不遍與一切不善心相應。}@
%{另外，這兩種煩惱，只有不善的，而且遍與所有不善心相應。所以，論著中只說了這兩種。忿、覆、嫉、慳，雖然只有不善的，但它們不遍與所有不善心相應。}%

@{惛沈掉舉雖遍與一切不善心相應。而非唯是不善。睡眠惡作二義俱無。是故不說隨眠與垢准此應知。}@
%{惛沈、掉舉，雖然遍與所有不善心相應，但它們不是只有不善的。睡眠、惡作，這兩種煩惱，既沒有不善的，也沒有無記的。所以，論著中沒有說隨眠和垢相應。根據這個道理，我們應該知道：}%

@{無慚無愧二義俱有與不善義多少量等。如函蓋相稱故偏說相應。四瀑流中一無記。謂有瀑流義如前說。}@
%{無慚無愧，既有善的，也有不善的，它們與不善的程度和數量相當，就像函蓋和東西相稱一樣。所以，論著中只說了相應的。四瀑流中，有一種是無記的，就是有瀑流。關於有瀑流的含義，前面已經說過了。}%

@{三應分別。謂欲瀑流或不善或無記無慚無愧及彼相應是不善。無慚無愧者顯彼自性唯是不善。}@
%{要區分兩種情況。意思是說，欲瀑流有的是不善的，有的是無記的。無慚無愧以及它們相應的煩惱，是不善的。無慚無愧，顯明它們的本性就是不善的。}%

@{及彼相應者顯欲瀑流中二十四事及三少分。亦是不善。三少分者。謂彼相應惛沈睡眠掉舉少分。}@
%{它們相應的煩惱，也顯明欲瀑流中的二十四種，以及三少分，也是不善的。三少分是指它們相應的惛沈、睡眠、掉舉少分。}%

@{餘是無記。謂欲瀑流中與有身見邊執見相應惛沈睡眠掉舉少分無慚無愧不相應故皆是無記。}@
%{其他的是無記的。意思是說，欲瀑流中，與身見、邊執見相應的惛沈、睡眠、掉舉少分，不與無慚無愧相應，所以都是無記的。}%

@{見瀑流或不善或無記。欲界三見是不善。謂邪見見取戒禁取無慚無愧相應故。}@
%{見瀑流，有的是不善的，有的是無記的。欲界的邪見、見取、戒禁取，是不善的，因為它們與無慚無愧相應。}%

@{欲界二見色無色界五見是無記。無慚無愧不相應故。無明瀑流或不善或無記。無慚無愧相應是不善。}@
%{欲界的二見，色界和無色界的五見，是無記的，因為它們不與無慚無愧相應。無明瀑流，有的是不善的，有的是無記的。無慚無愧相應的無明，是不善的。}%

@{謂欲界見集滅道及修所斷無明唯是不善。見苦所斷三見疑慢貪瞋相應及不共無明亦是不善。}@
%{意思是說，欲界的見、集、滅、道，以及修所斷的無明，只有不善的。見苦所斷的三見，疑、慢、貪、瞋相應的無明，以及不共無明，也是不善的。}%

@{餘是無記。謂欲界二見相應及色無色界一切無明。無慚無愧不相應故皆是無記。}@
%{其他的是無記的。意思是說，欲界的二見相應的無明，以及色界和無色界的所有無明，都不與無慚無愧相應，所以都是無記的。}%

@{如四瀑流四軛亦爾。瀑流與軛名體等故。四取中一無記謂我語取。義如前說。}@
%{就像四瀑流一樣，四軛也是如此。因為瀑流和軛的名稱和本質相同。四取中，有一種是無記的，就是我語取。關於我語取的含義，前面已經說過了。}%

@{三應分別。謂欲取或不善或無記。無慚無愧及彼相應是不善。無慚無愧者顯彼自性唯是不善。}@
%{要區分兩種情況。意思是說，欲取有的是不善的，有的是無記的。無慚無愧以及它們相應的煩惱，是不善的。無慚無愧，顯明它們的本性就是不善的。}%

@{及彼相應者顯欲取中二十八事及四少分亦是不善。四少分者謂彼相應惛沈睡眠掉舉無明少分。}@
%{它們相應的煩惱，也顯明欲取中的二十八種，以及四少分，也是不善的。四少分是指它們相應的惛沈、睡眠、掉舉、無明少分。}%

@{餘是無記。謂欲取中與有身見邊執見相應惛沈睡眠掉舉無明少分。無慚無愧不相應故皆是無記。}@
%{其他的是無記的。意思是說，欲取中，與身見、邊執見相應的惛沈、睡眠、掉舉、無明少分，不與無慚無愧相應，所以都是無記的。}%

@{見取或不善或無記。欲界二見是不善謂邪見見取。欲界二見色無色界四見是無記。}@
%{見取，有的是不善的，有的是無記的。欲界的邪見、見取，是不善的。欲界的二見，色界和無色界的四見，是無記的。}%

@{欲界二見者謂有身見邊執見。色無色界四見者。謂五見中除戒禁取。戒禁取或不善或無記。欲界是不善。無慚無愧相應故。}@
%{欲界的二見是指身見、邊執見。色界和無色界的四見是指五見中除了戒禁取以外的。戒禁取，有的是不善的，有的是無記的。欲界的戒禁取是不善的，因為它與無慚無愧相應。}%

@{色無色界是無記。無慚無愧不相應故。四身繫中二不善。謂貪欲瞋恚。二應分別。謂戒禁取此實執身繫欲界是不善。無慚無愧相應故。}@
%{色界和無色界的戒禁取是無記的，因為它不與無慚無愧相應。四身繫中，有兩種是不善的，就是貪欲、瞋恚。要區分兩種情況。意思是說，戒禁取，這種實執身繫，在欲界是不善的，因為它與無慚無愧相應。}%

@{色無色界是無記。無慚無愧不相應故。五蓋唯不善。皆與無慚無愧相應故。五結中三不善。謂瞋嫉慳結。}@
%{色界和無色界的戒禁取是無記的，因為它不與無慚無愧相應。五蓋只有不善的，因為它們都與無慚無愧相應。五結中，有三個是不善的，就是瞋、嫉、慳結。}%

@{二應分別。謂貪慢結或不善或無記。欲界是不善。無慚無愧相應故。色無色界是無記。無慚無愧不相應故。}@
%{要區分兩種情況。意思是說，貪、慢結，有的是不善的，有的是無記的。欲界的貪、慢結是不善的，因為它與無慚無愧相應。色界和無色界的貪、慢結是無記的，因為它不與無慚無愧相應。}%

@{五順下分結中。二不善。謂貪欲瞋恚一無記謂有身見。二應分別。謂戒禁取疑結或不善或無記。欲界是不善。色無色界是無記。}@
%{五順下分結中，有兩個是不善}%
