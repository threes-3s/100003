%{而有些众生对于所知境完全愚痴。}%
@{愚少分者為說六界。}@
%{对于那些只愚痴了一部分的众生，就给他们讲六界。}%
@{愚一切者為說十八界。}@
%{而对于那些完全愚痴的众生，就给他们讲十八界。}%
@{復次世尊所化有利根者。}@
%{再者，佛陀所教化的众生，有根性好的，}%
@{有鈍根者。}@
%{也有根性不好的。}%
@{為利根者說六界。}@
%{对于根性好的众生，就给他们讲六界。}%
@{為鈍根者說十八界。}@
%{而对于根性不好的众生，就给他们讲十八界。}%
@{復次世尊所化有開智者有說智者。}@
%{再者，佛陀所教化的众生，有已经开悟的，}%
%{有說智者。}%
%{也有正在学习智慧的。}%
@{為開智者說六界。}@
%{对于已经开悟的众生，就给他们讲六界。}%
@{為說智者。}@
%{而对于正在学习智慧的众生，}%
%{說十八界。}%
%{就给他们讲十八界。}%
@{復次世尊所化有樂略者有樂廣者。}@
%{再者，佛陀所教化的众生，有喜欢简略的，}%
%{有樂廣者。}%
%{也有喜欢详细的。}%
@{為樂略者說六界。}@
%{对于喜欢简略的众生，就给他们讲六界。}%
@{為樂廣者說十八界。}@
%{而对于喜欢详细的众生，就给他们讲十八界。}%
@{復次於十八界為略現門故說六界謂十八界中有是色有非色。}@
%{再者，为了简略地展现十八界，所以才说六界。因为十八界中既有色，也有无色。}%
@{若說前五界。}@
%{如果说前五界，}%
@{當知已說諸是色界。}@
%{就相当于已经说过了所有的色界。}%
@{若說識界。}@
%{如果说识界，}%
@{當知已說諸非色界。}@
%{就相当于已经说过了所有的无色界。}%
@{復次十八界中有有見有無見。}@
%{再者，十八界中既有有见，也有无见。}%
@{若說空界。}@
%{如果说空界，}%
%{當知已說諸有見者若說餘五界。}%
%{就相当于已经说过了所有的有见者；如果说其余五界，}%
%{當知已說諸無見者。}%
%{就相当于已经说过了所有的无见者。}%
@{復次十八界中有有對有無對。}@
%{再者，十八界中既有有对，也有无对。}%
@{若說前五界。}@
%{如果说前五界，}%
%{當知已說諸有對者。}%
%{就相当于已经说过了所有的有对者。}%
@{若說識界。}@
%{如果说识界，}%
%{當知已說諸無對者。}%
%{就相当于已经说过了所有的无对者。}%
@{復次十八界中有相應有不相應。}@
%{再者，十八界中既有相应，也有不相应。}%
@{若說識界。}@
%{如果说识界，}%
%{當知已說諸相應者。}%
%{就相当于已经说过了所有的相应者。}%
@{若說餘五界。}@
%{如果说其余五界，}%
%{當知已說不相應者。}%
%{就相当于已经说过了所有的不相应者。}%
@{如相應不相應。}@
%{就像相应和不相应一样，}%
@{如是有所依無所依。}@
%{还有有所依和无所谓，}%
@{有所緣無所緣。}@
%{有所缘和无所谓，}%
@{有行相無行相。}@
%{有行相和无行相，}%
@{有警覺無警覺。}@
%{有警觉和无警觉。}%
@{應知亦爾。}@
%{也都应该这样理解。}%
@{復次由此六界能生能養能長有情色無色身故復施設。}@
%{再者，这六界能够生养和增长有情色身和无色身，所以才设立了这六界。}%
@{能生者。}@
%{能够生者，}%
%{謂識界。}%
%{指的是识界。}%
@{能養者謂地水火風界。}@
%{能够养者，指的是地界、水界、火界和风界。}%
@{能長者謂空界。}@
%{能够增长者，指的是空界。}%
@{復次由此六界能引能持能增有情色無色身故復施設。}@
%{再者，这六界能够引导、保持和增长有情色身和无色身，所以才设立了这六界。}%
@{能引者謂識界。}@
%{能够引导者，指的是识界。}%
@{能持者謂地水火風界。}@
%{能够保持者，指的是地界、水界、火界和风界。}%
@{能增者謂空界。}@
%{能够增长者，指的是空界。}%
@{復次由此六界是根本有情事。}@
%{再者，这六界是根本有情事，}%
@{是遍行有情事。}@
%{是遍行有情事，}%
@{是無始有情事。}@
%{是无始有情事，}%
@{是無分別有分別有情事故復施設。}@
%{是无分别有分别有情事，所以才设立了这六界。}%
@{根本有情事者。}@
%{根本有情事，}%
@{謂欲色界受生有情。}@
%{指的是欲界和色界中受生的有情。}%
@{從結生心乃至死有。}@
%{从结生心到死亡，一直都有。}%
@{無此六界無勢用時。}@
%{如果没有这六界，就没有作用。}%
@{遍行有情事者。}@
%{遍行有情事，}%
@{謂欲色界一切有情從結生心乃至死有。}@
%{指的是欲界和色界中所有有情，从结生心到死亡，一直都有。}%
@{無此六界不增上時。}@
%{如果没有这六界，就不会增上。}%
@{無始有情事者。}@
%{无始有情事，}%
%{謂不可知本際已來諸有情類。}%
%{指的是从不可知本际开始，所有有情类。}%
@{從結生心乃至死有。}@
%{从结生心到死亡，一直都有。}%
@{無此六界不作用時無分別有分別有情事者。}@
%{如果没有这六界，就没有作用。无分别有分别有情事，}%
@{謂有有情未可分別是男是女。}@
%{指的是有些有情，还不能分别出是男是女。}%
@{如羯剌藍頞部曇閉尸揵南位。}@
%{比如羯剌蓝頞部昙闭尸揵南位。}%
@{如是六界亦有勢用。}@
%{这六界对他们也有作用。}%
@{或有有情已可分別是男是女鉢羅奢佉等位。}@
%{或者有些有情，已经可以分别出是男是女，比如鉢羅奢佉等位。}%
@{如是六界亦有勢用。}@
%{这六界对他们也有作用。}%
@{尊者妙音作如是說。}@
%{尊者妙音如是说：}%
@{由此六界得入母胎勢用增上故復施設。}@
%{因为这六界能够使有情进入母胎，并使他们的作用增上，所以才设立了这六界。}%
@{由如是等種種因緣故。}@
%{由于种种这样的因缘，}%
@{佛世尊於十八界略出少分。}@
%{佛陀世尊在十八界的基础上，略微地设立了六界。}%
@{施設如是六界差別。}@
%{这就是六界的差别。}%



@{問地界云何。}@
%{请问地界是什么呢？}%
@{答堅性。}@
%{答：地界就是坚性。}%
@{雖此地界總是堅性。}@
%{虽然地界总是坚性，}%
@{而此堅性差別無邊。}@
%{但是坚性却有无边的差别。}%
@{謂內外分堅性各異。}@
%{也就是说，内外分中的坚性各不相同。}%
@{內分中堅性者。}@
%{内分中的坚性，}%
@{謂髮毛爪齒塵垢皮肉。}@
%{指的是头发、毛发、指甲、牙齿、尘垢、皮肤、肉。}%
@{筋骨脈心脾腎肝肺胃肚腸糞。}@
%{筋骨、脉络、心脏、脾脏、肾脏、肝脏、肺脏、胃、肚子、肠、粪便。}%
@{生藏熟藏手足支節。}@
%{生藏、熟藏、手脚、支节。}%
@{如是等中所有堅性。}@
%{等等这些中所包含的坚性，}%
@{此諸堅性有勝有劣。}@
%{这些坚性有的胜过有的，有的不如有的。}%
@{謂足堅性勝手堅性。}@
%{比如，脚的坚性就胜过手的坚性。}%
@{若諸有情少時手行手皮血肉即便壞盡。}@
%{如果有些众生在年幼的时候，用手走路，那么他们的手皮、血肉就会很快坏掉。}%
@{若以足行盡眾同分足皮血肉都無損壞。}@
%{而如果用脚走路，即使走完所有同分，脚皮、血肉也不会受到任何损害。}%
@{由此故知內分堅性有勝有劣。}@
%{由此可见，内分中的坚性，有的胜过有的，有的不如有的。}%
@{外分中堅性者。}@
%{外分中的坚性，}%
@{謂地山礫石塼瓦草木。}@
%{指的是地、山、砾石、砖瓦、草木。}%
@{螺蜯䗾蛤銅鐵金銀白鑞鉛錫末尼真珠珊瑚琥珀珂貝璧玉。}@
%{螺、蜯、䗾、蛤、铜、铁、金、银、白鑞、铅、锡、末尼、珍珠、珊瑚、琥珀、珂贝、璧玉。}%
@{帝青大青末羅羯多。}@
%{帝青、大青、末罗羯多。}%
@{杵藏石藏颯頗胝迦。}@
%{杵藏、石藏、颯頗胝迦。}%
@{及紅頗黎吠琉璃等所有堅性。}@
%{以及红颇黎吠、琉璃等等所包含的坚性。}%
@{此內外分種種堅性。}@
%{这些内分和外分中的种种坚性，}%
@{以相同故略為一聚總名地界。}@
%{由于它们具有共同点，所以就简略地将其归为一类，总称为地界。}%
@{問水界云何。答濕性。}@
%{问水界是什么？ 答：湿性。}%
@{雖此水界總是濕性。而此濕性差別無邊。}@
%{虽然水界总体上都是湿性，但这种湿性种类繁多，无边无际。}%
@{謂內外分濕性各異。內分中濕性者。謂淚污涕唾肪膏髓腦涎膽痰癊膿血尿等所有濕性。}@
%{也就是说，内外湿性不同。内分中的湿性，指的是眼泪、鼻涕、唾液、脂肪、膏油、骨髓、脑浆、涎液、胆汁、痰液、脓液、血、尿等等这些湿性。}%
@{外分中濕性者。謂江河池沼泉井溝渠四大海等所有濕性。}@
%{外分中的湿性，指的是江河、池沼、泉井、沟渠、四大海等等这些湿性。}%
@{此內外分種種濕性。以相同故略為一聚總名水界。}@
%{这些内外不同的各种湿性，因为它们都具有相同点，所以就简单地归为一类，统称为水界。}%

@{問火界云何。答煖性。}@
%{问火界是什么？ 答：暖性。}%
@{雖此火界總是煖性。而此煖性差別無邊。}@
%{虽然火界总体上都是暖性，但这种暖性种类繁多，无边无际。}%
@{謂內外分煖性各異。內分中煖性者。謂身中熱等熱遍熱。}@
%{也就是说，内外暖性不同。内分中的暖性，指的是身体内部的热度，这种热度遍布全身。}%
@{由此所飲所食所噉。皆易消熟令身安隱。}@
%{因此，我们所喝的、所吃的、所嚼的，都容易消化吸收，让身体安稳舒适。}%
@{此若增時便成熱病。}@
%{如果这种热度增加，就会变成发热病。}%
@{外分中煖性者。謂炬燈燭陶竈爐等火聚炎焰。燒諸城。村山林曠野。及諸藥草日輪未尼天龍宮殿所出光焰。并地獄等諸火煖性。}@
%{外分中的暖性，指的是火炬、灯烛、陶灶、炉子等等这些火堆的火焰。它们可以烧毁城池、村庄、山林、旷野，以及各种药草。太阳、月亮、天龙宫殿所发出的光焰，以及地狱等等这些火，都是暖性。}%
@{應作是說。內火煖性熱於外火。}@
%{应该这样说：内火暖性比外火更热。}%
@{所以者何。若以飲食置釜鑊中。下然熾火經一日夜。猶不能令形色變易。如在腹中經須臾頃。}@
%{为什么这么说呢？ 比如把食物放在锅里，下面用旺火烧一整天一夜，都还不能让食物的颜色改变。而如果食物在肚子里面，只要经过很短的时间，就会被消化吸收。}%
@{此內外分種種煖性以相同故。略為一聚總名火界。}@
%{这些内外不同的各种暖性，因为它们都具有相同点，所以就简单地归为一类，统称为火界。}%

@{問風界云何。答輕等動性。}@
%{问风界是什么？ 答：轻等动性。}%
@{雖此風界總是動性。而此動性差別無邊。}@
%{虽然风界总体上都是动性，但这种动性种类繁多，无边无际。}%
@{謂內外分動性各異。內分中動性者。謂有上行風有下行風。有住脇風有住腹風。有住背風有如鍼風。有如刀風。}@
%{也就是说，内外动性不同。内分中的动性，指的是有上升的风、有下降的风，有停留在肋部的风、有停留在腹部的风，有停留在背部的风，有像针一样的风，有像刀一样的风。}%
@{有蓽茇羅風有婆呾瑟恥羅風。有婆呾寠拉摩風。有入息風有出息風。有隨身分支節行風所有動性。}@
%{还有蓽茇罗风、婆呾瑟耻罗风、婆呾寠拉摩风，有吸入的风、有呼出的风，有随着身体分支节肢而运动的风，等等这些动性。}%
@{外分中動性者。謂有四方風。或有塵風。或無塵風。或毘濕縛風。或吠嵐婆風。或小風。或大風。或風輪風等所有動性。}@
%{外分中的动性，指的是有四面八方的风，或者有尘土的风，或者没有尘土的风，或者毘濕縛风，或者吠嵐婆风，或者小风，或者大风，或者风轮风，等等这些动性。}%
@{此內外分種種動性以相同故。略為一聚總名風界。}@
%{这些内外不同的各种动性，因为它们都具有相同点，所以就简单地归为一类，统称为风界。}%

@{問空界云何。答如契經說。有眼穴空。有耳穴空。有鼻穴空。有面門空。有咽喉空。有心中空有心邊空。有通飲食處空。有貯飲食處空。有棄飲食處空。有諸支節毛孔等空。是名空界。}@
%{问空界是什么？ 答：就像经书中所说，有眼睛的空洞、有耳朵的空洞、有鼻子的空洞、有面门的空洞、有咽喉的空洞、有心里的空洞、有心边的空洞，有通往饮食的地方的空洞、有储存饮食的地方的空洞、有排泄饮食的地方的空洞，还有各种四肢关节、毛孔等等这些空洞，这些都叫做空界。}%
@{阿毘達磨作如是說。云何空界。謂隣礙色。礙謂積聚即牆壁等有色。近此名隣礙色。如牆壁間空。叢林間空。樹葉間空。窓牖間空。往來處空。指間等空。是名空界。}@
%{《阿毘達磨》中是这样说的：什么是空界呢？ 空界指的是“邻碍色”。“碍”指的是聚集在一起，比如墙壁等等这些有色法。靠近这些有色法的空间，就叫做“邻碍色”。比如墙壁之间的空隙、丛林之间的空隙、树叶之间的空隙、窗户之间的空隙、来往的地方的空隙、手指之间的空隙等等这些，都叫做空界。}%
@{有作是說此文應言云何空界謂隣難除色。然色有二種。一者易除謂有情數。二者難除謂無情數。}@
%{也有人说，这段文字应该说“什么是空界？ 空界指的是邻难除色”。 因为色有两种：一种是容易除去的，指的是有情众生；一种是难以除去的，指的是无情众生。}%
@{此空界色多近非情牆壁樹等。而施設故名隣難除色。}@
%{空界中的色法，大多是靠近无情众生的，比如墙壁、树木等等。 因此，就叫做“邻难除色”。}%
@{舊對法者及此國師。俱說空界處處皆有。謂骨肉筋脈皮血身分。晝夜明闇形顯等處皆有此色。}@
%{以前研究对法论的人，以及现在的国师，都认为空界无处不在。 比如骨头、肉、筋、脉、皮、血、身体的各个部分，白天、黑夜、光明、黑暗、形体显现等等这些地方，都有这种色法。}%
@{問緣空界色眼識生不。}@
%{问：缘空界色，眼识会生起吗？}%
@{有說緣此眼識不生。謂空界色雖眼識境而此眼識畢竟不生。}@
%{有人说：缘空界色，眼识不会生起。 因为虽然空界色是眼识的境，但眼识终究不会生起。}%
@{復有說者。緣空界色眼識亦生。}@
%{也有人说：缘空界色，眼识也会生起。}%
@{問若爾何故見不明了。}@
%{问：如果是这样，为什么我们看到的景象不明了？}%
@{答此空界色晝為明所覆。夜為闇所覆。故眼雖見而不明了。}@
%{答：因为空界色白天被光明所覆盖，夜晚被黑暗所覆盖。 所以虽然眼睛看到了，但景象却不明了。}%

@{問虛空空界有何差別。答虛空非色空界是色。}@
%{问：虚空和空界有什么区别？ 答：虚空不是色法，空界是色法。}%
@{虛空無見空界有見。虛空無對空界有對。虛空無漏空界有漏。虛空無為空界有為。}@
%{虚空无见，空界有见。 虚空无对，空界有对。 虚空无漏，空界有漏。 虚空无为，空界有为。}%
@{問若此虛空是無為者。契經所說當云何通。如契經說。世尊以手摩捫虛空告苾芻眾。豈佛以手摩捫無為而告弟子。}@
%{问：如果虚空是无为法，那经书中所说的话该如何理解？ 比如经书中说，世尊用手摩挲虚空，告诉比丘大众。 佛陀难道会用手摩挲无为法，然后告诉弟子吗？}%
@{答彼於空界說虛空聲。非謂虛空手可摩捫。}@
%{答：佛陀在空界中说“虚空”这个名词，并不是说虚空可以用手去摩挲。}%
@{餘經亦說。佛告苾芻。若有畫師或彼弟子持諸彩色來作是言。我能彩畫虛空。作種種文像有是事不。苾芻白佛無有是事。}@
%{其他经书中也说：佛陀告诉比丘们，如果有一个画师，或者他的弟子，拿着各种颜料来说，我能用颜色画虚空，画各种各样的图案，有这样的事吗？ 比丘们回答佛陀说，没有这样的事。}%
@{彼亦於空界說虛空聲}@
%{佛陀在空界中说“虚空”这个名词。}%
@{又伽他說。}@
%{伽他中也说：}%

@{獸歸林藪鳥歸虛空聖歸涅槃}@
%{兽归山林，鸟归虚空，圣人归涅槃。}%
@{法歸分別}@
%{法归分别。}%

@{彼亦於空界。說虛空聲。復有頌言。}@
%{佛陀在空界中说“虚空”这个名词。 又有颂词说：}%

@{虛空無鳥跡外道無沙門}@
%{虚空无鸟迹，外道无沙门。}%
@{愚夫樂戲論如來則無有}@
%{愚人喜欢玩弄理论，如来则无有。}%

@{彼亦於空界說虛空聲。有餘經說。鳥步虛空跡難可現亦不可尋。}@
%{佛陀在空界中说“虚空”这个名词。 其他经书中也说，鸟在虚空行走留下的痕迹，难以显现，也难以寻找。}%
@{彼亦於空界說虛空聲。有處問虛空而答以空界。}@
%{佛陀在空界中说“虚空”这个名词。 有的地方问“虚空”，就回答“空界”。}%
@{如品類足作如是言。云何虛空謂有虛空無障無礙色於中行周遍增長。}@
%{比如《品类足》中是这样说的：什么是虚空？ 虚空就是有虚空，无障碍、无阻碍的色法，在其中行走，周遍增长。}%
@{問何故問虛空而答以空界。}@
%{问：为什么问“虚空”，就回答“空界”？}%
@{答虛空微細難可顯說。空界相麁易可開示。以麁顯細故作是說。}@
%{答：因为虚空非常细微，难以描述。 空界比较粗糙，容易解释。 用粗糙的来显现细微的，所以就用这种方式来解释。}%
@{問以何緣故知有虛空。尊者世友作如是說。以佛說故知有虛空。謂契經中。佛處處說虛空虛空。故知實有。}@
%{问什么原因知道有虚空呢？世友尊者这样说：因为佛陀所说，所以知道有虚空。意思是说，在契经中，佛陀处处都说虚空虚空，所以知道确实有虚空。}%
@{問為但信教知有虛空。為此虛空亦現量得。}@
%{问是仅仅相信佛经知道有虚空，还是说这个虚空也能用现量获得呢？}%
@{答亦現量得。若無虛空一切有物應無容處。既有容受諸有物處知有虛空。}@
%{答：也能用现量获得。如果没有虚空，所有有形之物就应该没有容身之处。既然有容纳所有有形之物的地方，就知道有虚空。}%
@{復作是說。以有往來聚集處故知有虛空。若無彼因彼亦不有。言彼因者即是虛空。虛空是彼容受因故。}@
%{再说，因为有来来往往聚集的地方，所以知道有虚空。如果没有那个原因，那个地方也不会存在。说的那个原因就是虚空。虚空是那个地方容纳的原因。}%
@{復作是說。容有礙物知有虛空。若無虛空彼無容處。}@
%{再说，因为容纳有障碍之物，所以知道有虚空。如果没有虚空，就没有容纳这些障碍之物的地方。}%
@{復作是說若無虛空應一切處皆有障礙。既現見有無障礙處故。知虛空決定實有。無障礙相是虛空故。}@
%{再说，如果没有虚空，应该所有地方都有障碍。既然现在看到有无障碍的地方，所以知道虚空确实存在。无障碍的相就是虚空。}%
@{大德說曰。虛空不可知非所知事故。所知事者色非色性虛空與彼俱不相應。所知事者謂此彼性虛空與彼俱不相應。此虛空名。但是世間分別假立。}@
%{大德说：虚空不可知，因为它不是可以知的事。可以知的事是色，非色，性，虚空与它们都不相应。可以知的事就是指这些东西，虚空的性质与它们都不相应。这个虚空的名字，仅仅是世间分别假立的。}%
@{評曰應作是說。實有虛空非彼不知即謂非有由前教理實有虛空。}@
%{评曰：应该这样说，确实有虚空，不是说不知道它就是没有，而是根据之前的教理，确实有虚空。}%

@{問若爾虛空有何作用。}@
%{问：既然如此，虚空有什么作用呢？}%
@{答虛空無為無有作用。然此能與種種空界作近增上緣。彼種種空界能與種種大種作近增上緣。彼種種大種能與有對造色等作近增上緣。彼有對造色能與心心所法作近增上緣。}@
%{答：虚空无为，没有作用。但是它能与各种空界作为近因增上缘。各种空界能与各种大种作为近因增上缘。各种大种能与有对造色等作为近因增上缘。有对造色能与心心所法作为近因增上缘。}%
@{若無虛空。如是展轉因果次第皆不成立。勿有此失。是故虛空體相實有不應撥無。}@
%{如果没有虚空，那么这样循环的因果次第就都无法成立。不可有这种错误。所以虚空的体相确实存在，不应该说它不存在。}%

@{問識界云何。}@
%{问：识界是什么？}%
@{答五識身及有漏意識。}@
%{答：五识身和有漏意识。}%
@{問何故無漏識不立識界耶。}@
%{问：为什么无漏识不立识界呢？}%
@{答與識界相不相應故。若法能長養諸有。攝益諸有。任持諸有者。立六界中。}@
%{答：因为它与识界不相相应。如果一个法能够滋养众生，摄益众生，任持众生，就立于六界之中。}%
@{無漏意識。能損減諸有。散壞諸有。破滅諸有。是故不立在六界中。}@
%{无漏意识，能够损减众生，散坏众生，破灭众生，所以不立于六界之中。}%
@{復次若法能令諸有相續生老病死流轉不絕者立六界中。無漏意識與此相違。是故不立在六界中。}@
%{再说，如果一个法能够使众生不断地生老病死，流转不息，就立于六界之中。无漏意识与这个相违背，所以不立于六界之中。}%
@{復次若法是趣苦集行亦是趣有世間生老病死集行者立六界中。無漏意識與此相違。是故不立在六界中。}@
%{再说，如果一个法是趋向于苦集行，也是趋向于有世间生老病死集行，就立于六界之中。无漏意识与这个相违背，所以不立于六界之中。}%
@{復次若法是有身見事。顛倒事愛事。隨眠事與貪瞋癡為安足處。有垢有毒有穢有刺。有過有濁墮在諸有苦集諦攝者立六界中。無漏意識與此相違。是故不立在六界中。}@
%{再说，如果一个法是有身见事，颠倒事，爱事，随眠事，以贪嗔痴为安足之处，有垢，有毒，有秽，有刺，有过，有浊，堕落在诸有苦集谛中，就立于六界之中。无漏意识与这个相违背，所以不立于六界之中。}%
@{尊者世友作是問言。此六界中何故不攝無漏意識。即自答言。如是六界從諸漏生。無漏意識不從漏生。}@
%{世友尊者这样问：这六界中为什么不摄受无漏意识？他马上自己回答：这六界是从诸漏产生的，无漏意识不是从漏产生的。}%
@{復作是說。如是六界能生諸漏。無漏意識不生諸漏。}@
%{再说，这六界能生诸漏，无漏意识不生诸漏。}%
@{復作是說。如是六界是我執緣。無漏意識非我執緣。}@
%{再说，这六界是我执缘，无漏意识非我执缘。}%
@{復作是說。如是六界是有情依。無漏意識非有情依。}@
%{再说，这六界是有情依，无漏意识非有情依。}%
@{復作是說。如是六界是異熟依。無漏意識非異熟依。}@
%{再说，这六界是异熟依，无漏意识非异熟依。}%
@{復作是說如是六界是入胎緣。無漏意識非入胎緣。}@
%{再说，这六界是入胎缘，无漏意识非入胎缘。}%
@{復作是說。如是六界無始來有。無漏意識非無始有。}@
%{再说，这六界无始以来就存在，无漏意识不是无始有的。}%
@{大德說曰。如是六界是自體分。無漏意識非。自體分。}@
%{大德说：这六界是自体分，无漏意识不是自体分。}%
@{脇尊者言。如是六界是生死依。無漏意識非生死依。由如是等種種因緣無漏意識不立識界。}@
%{胁尊者说：这六界是生死依，无漏意识非生死依。由于这些种种原因，无漏意识不立识界。}%

@{問蘊取蘊界有何差別。}@
%{问：蕴、取蕴、界有什么区别？}%
@{答名即差別。謂名為蘊名為取蘊名為界故。}@
%{答：名字就是区别。因为叫做蕴，叫做取蕴，叫做界，所以有区别。}%
@{復次於有為法施設蘊於有漏法施設取蘊。於有情數法施設界。}@
%{再说，对于有为法安立蕴，对于有漏法安立取蕴，对于有情数法安立界。}%
@{復次蘊有流轉還滅作用。取蘊唯有流轉作用。界有結生入胎作用如是名為蘊取蘊界三種差別。}@
%{再说，蕴有流转还灭的作用，取蕴只有流转的作用，界有结生入胎的作用。这样叫做蕴、取蕴、界三种区别。}%

@{有二法。謂有色無色法。}@
%{有两种法，就是有色法和无色法。}%
@{問何故作此論。}@
%{问：为什么作此论呢？}%
@{答為欲遮遣補特伽羅。及為顯示智殊勝故。為欲遮遣補特伽羅者。謂顯唯有有色無色法畢竟無實補特伽羅故。}@
%{答：为了想要遮遣补特伽罗，以及为了显示智的殊胜。想要遮遣补特伽罗，就是说显明只有有色法和无色法，毕竟没有真实的补特伽罗。}%
@{及為顯示智殊勝者。謂有聰慧殊勝智者。由此二法通達一切。此二遍攝一切法故。由此二緣故作斯論。}@
%{以及为了显示智的殊胜，就是说有聪明智慧殊胜的智者，由此两种法通达一切，这两种法遍摄一切法，所以由此两种原因作此论。}%
@{問有色法云何。}@
%{问：有色法是什么？}%
@{答謂十處一處少分。十處者。謂眼耳鼻舌身色聲香味觸處。一處少分者。謂法處少分。}@
%{答：就是十处和一处少分。十处是指眼、耳、鼻、舌、身、色、声、香、味、触处。一处少分是指法处少分。}%
@{問無色法云何。}@
%{问：无色法是什么？}%
%{答謂一處一處少分。一處者謂意處。一處少分者。謂法處少分。}%
%{答：就是一处和一处少分。一处是指意处，一处少分是指法处少分。}%
@{問此中何等名有色法無色法耶。}@
%{问：这里面什么叫做有色法，什么叫做无色法呢？}%
@{答若法有色名體是色者名有色法。若法有非色名體非色者名無色法。}@
%{答：如果一个法有色名，它的体是色，就叫做有色法。如果一个法有非色名，它的体是非色，就叫做无色法。}%
@{或有法雖有色名而體非色。如契經說寂靜解脫超有色法。至無色法。應知此中有色法者即有色定。}@
%{或者一个法虽然有色名，但是它的体是非色。比如契经说：寂静解脱超越有色法，到达无色法。应该知道，这里面的有色法就是指有色定。}%
@{又契經說。身證色定具足而住。又如有言。我今正受如是色受。又如佛說。我以如是色經典句付屬汝等應正受持。如是等處雖有色名而體非色。}@
%{又契经说：身证色定，具足而住。又有人说：我现在正在受如是色受。又比如佛陀说：我以如是色经典句付嘱你们，应该正受持。这些地方虽然有色名，但是体是非色。}%
@{若有色名體是色者名有色法。或色體有色用。或色用有色體。或體與相互相有故立有色名。}@
%{如果一个法有色名，它的体是色，就叫做有色法。或者色体有色用，或者色用有色体，或者体与用相互存在，所以安立色名。}%
@{復次若法體是四大種。或是四大種所造者名有色法。若法體非四大種。或非四大種所造者名無色法。}@
%{再说，如果一个法的体是四大种，或者是由四大种所造，就叫做有色法。如果一个法的体不是四大种，或者不是由四大种所造，就叫做无色法。}%
@{復次若法大種為因及體是所造色者名有色法。若法非大種為因及體非所造色者名無色法。}@
%{再说，如果一个法以大种为因，它的体是所造色，就叫做有色法。如果一个法不是以大种为因，它的体不是所造色，就叫做无色法。}%
@{復次若法可種植可增長者名有色法。若法不可種植及不可增長者名無色法。}@
%{再说，如果一个法可以种植可以增长，就叫做有色法。如果一个法不可种植，也不可以增长，就叫做无色法。}%
@{尊者世友作如是說。有色相者名有色法。何等名有色相。謂有漸次積集相者名有色相。}@
%{世友尊者这样说：有色相者叫做有色法。什么叫做色相呢？就是说，有逐渐积累的相，叫做色相。}%
@{復作是說。若有漸次散壞相者名有色相。}@
%{再说，如果有一个逐渐散坏的相，就叫做色相。}%
@{復作是說。若有形質可取相者名有色相。}@
%{再说，如果有一个形质可以取的相，就叫做色相。}%
@{復作是說。若有方所可取相者名有色相。}@
%{再说，如果有一个方所可以取的相，就叫做色相。}%
@{復作是說。若有大小所取相者名有色相。}@
%{再说，如果有一个大小可以取的相，就叫做色相。}%
@{復作是說。若有障礙可取相者名有色相。}@
%{再说，如果有一个障碍可以取的相，就叫做色相。}%
@{復作是說。若有怨害可取相者名有色相。}@
%{再说，如果有一个怨害可以取的相，就叫做色相。}%
@{復作是說。若有損害可取相者名有色相。}@
%{再说，如果有一个损害可以取的相，就叫做色相。}%
@{復作是說。若增益可取相者名有色相。}@
%{再说，如果有一个增益可以取的相，就叫做色相。}%
@{復次若有三種色相可得名有色相。謂或有色有見有對。或復有色無見有對。或復有色無見無對。}@
%{再说，如果三种色相可以得到，就叫做色相。就是说，或者有色，有见，有对；或者有色，无见，有对；或者有色，无见，无对。}%
@{復作是說。若有牽來引去相者名有色相。}@
%{再说，如果有一个牵来引去的相，就叫做色相。}%
@{復作是說。有變礙相名有色相。問若有變礙相名有色相者。過去未來極微無表既無變礙應無色相。若無色相體應非色。}@
%{再说，有变碍相叫做色相。问：如果有变碍相叫做色相的话，过去未来极微无表既然没有变碍，应该没有色相。如果没有色相体，应该是非色。}%
@{答彼亦是色得色相故。謂過去色雖今無變礙而曾有變礙。未來色雖今無變礙而當有變礙極微一一雖無變礙而多積集即有變礙。無表自體雖無變礙而彼所依有變礙故亦名變礙。}@
%{答：它也是色，所以得到色相。就是说，过去色虽然现在没有变碍，但是曾经有变碍。未来色虽然现在没有变碍，但是将来会有变碍。极微一一虽然没有变碍，但是很多聚集起来就有了变碍。无表自体虽然没有变碍，但是它所依有变碍，所以也叫做变碍。}%
@{所依者何謂四大種。所依有變礙故無表亦可說有變礙。如樹動時影亦隨動。}@
%{所依指的是什么？就是四大种。所依有变碍，所以无表也可以说有变碍。就像树动的时候，影子也跟着动。}%
@{復作是說。若有容受障礙相者名有色相。復作是說。若有大種為因相者名有色相。}@
%{再说，如果有一个容受障碍的相，就叫做色相。再说，如果以大种为因的相，就叫做色相。}%
@{復作是說。無一切色同一色相。所以者何。眼處色相異乃至法處所攝色相異。}@
%{再说，所有色都没有相同的色相。为什么呢？因为眼处色相不同，乃至法处所摄色相也不同。}%
@{大德說曰。若有能壞有對色相是有色相。與前所說色相相違名無色相。若法有此無色相者。名無色法。}@
%{大德说：如果有能坏有对的色相，就是有色相。与前面所说的色相相违背，叫做无色相。如果一个法有这个无色相，就叫做无色法。}%
@{問墮法處色。何不攝在十色處中。}@
%{问堕法处色，为什么不把它归入十色处呢？}%
@{答若色可以剎那極微而分析者立十色處。}@
%{答：如果色可以以刹那极微来分析，就可以建立十色处。}%
@{墮法處色雖有剎那可分析義。而無極微可分析義故不攝在十色處中。}@
%{堕法处色虽然有刹那可以分析的意思，但没有极微可以分析的意思，所以不把它归入十色处。}%
@{復次若色可作五識所依及所緣者立十色處。}@
