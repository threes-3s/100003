%{一切天人，有大散乱，初如亲友，}%

%{后作怨家，如火焚烧，如刀割切。}%

@{人間、天上，得失平等，心意狂迷，}@
%{人间、天上，得失平等，心意狂迷，}%

%{如同怨家。天人不知，智者远离，}%

@{散亂毒樹，說有三枝，為老、病、死；}@
%{散乱毒树，说有三枝，为老、病、死；}%

%{恒复相续，智者离妄，纯善相应，}%

%{老等三枝，而不为害。如过远离，}%

%{散乱不生，恶怖解脱，得最上乐。}%

@{若散亂樂，不久有怖；若解脫樂，}@
%{若散乱乐，不久有怖；若解脱乐，}%

%{乐定无尽。百千万亿，般都摩人，}%

@{樂住名利，妄亂所賺。妄亂之上，}@
%{乐住名利，妄乱所赚。妄乱之上，}%
@{有四種過；若離妄亂，破世間怨。}@
%{有四种过；若离妄乱，破世间怨。}%

@{彼妄亂行：多疑、多怖，多難、多苦。}@
%{彼妄乱行：多疑、多怖，多难、多苦。}%

@{展轉輪迴，無有窮盡；四過離一，}@
%{辗转轮回，无有穷尽；四过离一，}%
@{獲樂無邊。汝等天人，一心散亂，}@
%{获乐无边。汝等天人，一心散乱，}%

@{障覆一切，無漏善法，云何而得，}@
%{障覆一切，无漏善法，云何而得，}%
@{寂靜安樂？若樂自利，修善思惟，}@
%{寂静安乐？若乐自利，修善思惟，}%

@{煩惱苦等，於後不生。若諸天眾，}@
%{烦恼苦等，于后不生。若诸天众，}%

%{所著欲乐，虚妄不实，一切不久。}%

@{從幻化生，彼不知見，幻化自性，}@
%{从幻化生，彼不知见，幻化自性，}%

%{而不决定；远离安乐，恒生众苦。}%

@{實言告汝，若人恒善，世所敬愛，}@
%{实言告汝，若人恒善，世所敬爱，}%

%{贤圣围遶，后生天上，受天妙乐；}%

@{若復行施，得一切人，心生敬喜，}@
%{若复行施，得一切人，心生敬喜，}%

%{后得富贵，施果如是。若行忍辱，}%

@{後生天上，無所怖畏，世人愛重，}@
%{后生天上，无所怖畏，世人爱重，}%

%{名称远闻，面貌端严，具足朋友，}%
@{受天妙樂。彼等有情，由如一切，}@
%{受天妙乐。彼等有情，由如一切，}%

@{世間父母，能破罪暗。譬如日光，}@
%{世间父母，能破罪暗。譬如日光，}%

%{住意最上，为慈悲宝，出生善根，}%
@{通達諸法；雖復在家，若住正見，}@
%{通达诸法；虽复在家，若住正见，}%

@{彼智解脫，輪迴枷鎖，罪友不生，}@
%{彼智解脱，轮回枷锁，罪友不生，}%
@{人獲安樂，聞未聞法，聞之不退，}@
%{人获安乐，闻未闻法，闻之不退，}%
@{超諸惡趣，得生天上。心善無垢，}@
%{超诸恶趣，得生天上。心善无垢，}%

@{諸過不生，離染清淨，善知報應，}@
%{诸过不生，离染清净，善知报应，}%
@{解微妙義，絕諸過罪。譬如虛空，}@
%{解微妙义，绝诸过罪。譬如虚空，}%

@{不住淤泥，得寂靜果，深信三寶，}@
%{不住淤泥，得寂静果，深信三宝，}%
@{獲無所畏，與諸天眾，尊敬如來，}@
%{获无所畏，与诸天众，尊敬如来，}%
@{信受佛教，蒙法救護。生我有果，}@
%{信受佛教，蒙法救护。生我有果，}%

@{生身有愛，淪溺無邊，此苦難窮。}@
%{生身有爱，沦溺无边，此苦难穷。}%

@{十六分苦，不及此一，少味多怖，}@
%{十六分苦，不及此一，少味多怖，}%
@{恒賺有情。喻乾闥城，智者寧信，}@
%{恒赚有情。喻乾闼城，智者宁信，}%

@{說殺眾生，至命終時，樂不隨行，}@
%{说杀众生，至命终时，乐不随行，}%
@{都無所得。種種眾生，有種種心，}@
%{都无所得。种种众生，有种种心，}%

@{種種業行，輪迴枷鎖，一步隨去。}@
%{种种业行，轮回枷锁，一步随去。}%

@{生復有死，愚癡迷惑，真實自性，}@
%{生复有死，愚痴迷惑，真实自性，}%
@{而不覺悟，無常大怖；隨彼眾生，}@
%{而不觉悟，无常大怖；随彼众生，}%

@{來而復去，散亂不知，境界天迷。}@
%{来而复去，散乱不知，境界天迷。}%

@{如逢蛇螫，墜墮天人，此苦無等，}@
%{如逢蛇螫，坠堕天人，此苦无等，}%
@{父母、妻子，男女眷屬，親友、知識，}@
%{父母、妻子，男女眷属，亲友、知识，}%

@{不可代受。味著境界，樂欲迷深，}@
%{不可代受。味著境界，乐欲迷深，}%
@{不覺死王，大苦速至。云何方便，}@
%{不觉死王，大苦速至。云何方便，}%

@{令離苦惱？若無愛欲，苦不能害，}@
%{令离苦恼？若无爱欲，苦不能害，}%
@{亦復不生，地獄、餓鬼，傍生之類。}@
%{亦复不生，地狱、饿鬼，傍生之类。}%

@{爾時，帝釋天主復說頌曰：}@
%{尔时，帝释天主复说颂曰：}%

@{人行邪道，欲見正真，如手探水，}@
%{人行邪道，欲见正真，如手探水，}%
@{而取其火；若復無因，何所有果？}@
%{而取其火；若复无因，何所有果？}%

@{散亂無德，利益何生？破壞天人，}@
%{散乱无德，利益何生？破坏天人，}%
@{此意若生，迷欲適悅，後得無常，}@
%{此意若生，迷欲适悦，后得无常，}%
@{心大熱惱。境界不迷，欲火不發，}@
%{心大热恼。境界不迷，欲火不发，}%

@{名稱不貪，安樂最上。若復一心，}@
%{名称不贪，安乐最上。若复一心，}%
@{貪愛世樂，十六分罪，不及此一。}@
%{贪爱世乐，十六分罪，不及此一。}%

@{若發大信，解脫輪迴，愛盡境亡，}@
%{若发大信，解脱轮回，爱尽境亡，}%
@{見無為樂，禪樂相應，無瞋、無喜；}@
%{见无为乐，禅乐相应，无瞋、无喜；}%

@{彼善若生，無復輪迴，究竟彼岸。}@
%{彼善若生，无复轮回，究竟彼岸。}%

@{若有欲心，而求快樂，被業繩牽，}@
%{若有欲心，而求快乐，被业绳牵，}%
@{墜墮地獄，得一切苦，誰人救解？}@
%{坠堕地狱，得一切苦，谁人救解？}%

@{往而復來，從業因生，無前、中、後，}@
%{往而复来，从业因生，无前、中、后，}%
@{非今、後世，作過無明，墮人惡趣。}@
%{非今、后世，作过无明，堕人恶趣。}%

@{是知自愚，行散亂毒，浮生頃刻；}@
%{是知自愚，行散乱毒，浮生顷刻；}%
@{迷愚不覺，後無常位，乃可知苦。}@
%{迷愚不觉，后无常位，乃可知苦。}%

@{若生有苦，怖彼無常，作意妙法，}@
%{若生有苦，怖彼无常，作意妙法，}%
@{彼救為實，真成安樂。如斯言說，}@
%{彼救为实，真成安乐。如斯言说，}%

@{實為利益，令心行善，速得安樂；}@
%{实为利益，令心行善，速得安乐；}%
@{樂中最上，決定無失。天若迷惑，}@
%{乐中最上，决定无失。天若迷惑，}%

@{不依彼法，而作亂意；生墮地獄，}@
%{不依彼法，而作乱意；生堕地狱，}%
@{無有出期。}@
%{无有出期。}%
@{彼有飛禽名種種鬘覺悟彼天，而說頌曰：}@
%{据说有一种鸟叫“种种鬘”，它觉悟了那个天界，就吟诵了这首诗：}%

@{種種業生，隨彼天人，知不依法，}@
%{各种各样的业力产生，随着那些天人，知道不依法，}%

@{彼後熱惱。世間所有，種種善果，}@
%{他们以后就会遭受热恼。世间所有，各种各样的善果，}%

@{園林、花鬘，階道、樓閣，適悅身心；}@
%{园林、花鬘，阶道、楼阁，都能让人身心愉悦；}%

@{如作善業，得見彼果。天人業因，}@
%{就像做善业，就能看到这些果报。天人的业因，}%

@{有上、中、下，天非作彼，無彼彼果。}@
%{有上、中、下，天人如果不做那些事，就不会得到相应的果报。}%

@{癡降天人，迷種種行，彼意若迷，}@
%{愚痴降伏了天人，他们迷失在各种行为中，如果他们的心迷失了，}%

@{不見大怖，失自利行；種種愛樂，}@
%{就看不到巨大的恐惧，就失去自利行为；各种各样的爱乐，}%

@{求種種果，迷心作業。天無戒行，}@
%{追求各种各样的果报，迷失了心，造了业。天人没有戒行，}%

@{如夜無燈，而覓光明，因與果等，}@
%{就像夜晚没有灯，却要寻找光明，因果相等，}%

@{真實可修；真如智果，得有樂分，}@
%{这是真实可以修行的；真如智慧的果报，可以得到快乐的部分，}%

@{離命無果，離燈無光，離戒無天。}@
%{没有生命就没有果报，没有灯就没有光，没有戒行就没有天界。}%

@{若離智慧，而無解脫；若離解脫，}@
%{如果离开智慧，就没有解脱；如果离开解脱，}%

@{而無安樂。百般求救，方脫愛身。}@
%{就没有安乐。要千方百计地求救，才能摆脱爱欲之身。}%

@{智者離我，無此業樂，一切有罪，}@
%{智慧的人远离自我，没有这种业力的快乐，一切有罪，}%

@{而復不生，得彼利益。一切聖人，}@
%{就不会再生，就能得到利益。一切圣人，}%

@{無垢所說。百千劫人，樂欲境界，}@
%{无垢地宣说。千百劫的人，沉迷于快乐和欲望的境界，}%

@{耽著不足，一心不捨，漸漸增長，}@
%{贪著不足，一心不舍，慢慢增长，}%

@{增長彼毒，須臾墜墮。智知德過，}@

%{这就是法师的相貌；不知道德行和过失，这就是迷失的行为相貌。}%

@{德處生德，過處亦然，知真過德，}@
