@{廬山蓮宗寶鑑念佛正信卷第五(六章)}@
%{庐山莲宗宝鉴念佛正信卷第五(六章)}%

@{念佛正信說}@
%{念佛正信说}%

@{夫唯心樂國普遍十方。}@
%{因为只有心念才能创造快乐的国土，这国土遍布十方世界。}%

@{自性彌陀圓融一智。}@
%{我们内心的本性就是弥陀，圆融无碍，智慧无穷。}%

@{妙應於色聲之境。}@
%{它奇妙地响应着各种色声的境界。}%

@{流光於心目之間。}@
%{它光辉流转于我们心中和眼前。}%

@{就中反妄歸真。}@
%{从这些现象中，我们能反观内心，回归本真。}%

@{直下背塵合覺。}@
%{直接背弃尘世，与觉悟合一。}%

@{昔我法藏發弘誓。}@
%{过去，我法藏菩萨发下了宏大的誓愿。}%

@{啟極樂之玄途。}@
%{开启了通往极乐世界的玄妙之路。}%

@{故佛世尊指西方示韋提之妙域。}@
%{所以，佛陀世尊指着西方，指明了极乐世界的奇妙境界。}%

@{是乃廣長舌覆而同贊。}@
%{这正是佛陀广长的舌头所覆盖的，所有佛经都赞叹的地方。}%

@{諸餘經盡而獨留。}@
%{其他经文都消失了，只有这一个留了下来。}%

@{蓋以利生之喜捨心增。}@
%{因为救度众生的喜舍之心越来越强烈。}%

@{應化之慈悲量大。}@
%{应化的慈悲之量越来越大。}%

@{教分九品乃別開方便之門。}@
%{教法分为九品，这是为了开辟方便之门。}%

@{觀明一心。}@
%{观照明心见性，这是直达本源的道路。}%

@{實徑直還源之路。}@
%{观照明心见性，这是直达本源的道路。}%

@{聖凡際會如久客歸於家鄉。}@
%{圣凡相遇，就像久客归家一样。}%

@{感應道交似稚子投於慈母。}@
%{感应道交，就像稚子投向慈母怀抱一样。}%

@{昧斯至理觸類皆迷。}@
%{不明白这些至理，接触到任何事物都会迷失。}%

@{信此圓談事無不達。}@
%{相信这些圆满的道理，世间万物都能明白。}%

@{況復慈光願攝佛力難思。}@
%{何况还有慈悲的光明和愿力摄持，佛的力量难以想象。}%

@{順水乘船不勞自力。}@
%{顺水行舟，不需要自己费力。}%

@{推門落臼豈有他哉。}@
%{推开门，掉落的臼，怎么会是别的呢？}%

@{有願必迎無機不被。}@
%{有愿必迎，没有一个人不被接引。}%

@{舟石可濟獄火頓消。}@
%{舟石可以渡过地狱之火，顷刻间就能消除。}%

@{菩薩聲聞生彼者無量無數。}@
%{菩萨和声闻乘的众生，往生极乐世界的，数不胜数。}%

@{前賢後聖得道者可撿可尋。}@
%{古今贤圣得道者，都可以找到。}%

@{䴍鵡頻伽尚有法音演唱。}@
%{鹦鹉和孔雀尚且能演唱佛法的声音。}%

@{蜎飛蝡動悉蒙教化恩慈。}@
%{飞禽走兽都蒙受佛法的教化和恩慈。}%

@{聖境非虛佛言不妄。}@
%{圣境不是虚幻的，佛陀的话语不是虚假的。}%

@{何乃愛河浪底沈溺而不憂。}@
%{为什么还在爱河的浪底沉溺，而不忧愁呢？}%

@{火宅焰中焚燒而不懼。}@
%{为什么还在火宅的火焰中焚烧，而不害怕呢？}%

@{密織癡網。}@
%{被无明编织的网紧紧地束缚着。}%

@{淺智之刃莫能揮。}@
%{浅薄的智慧之刃无法挥动。}%

@{深種疑根。}@
%{深深地种下了怀疑的根。}%

@{汎信之力焉能拔。}@
%{泛泛的信心之力怎么能拔除呢？}%

@{遂即甘心伏意幸禍樂災却誹清淨之邦。}@
%{于是就甘心屈服，希望灾难和痛苦，反而诽谤清净的国土。}%

@{貪戀煩惱之世。}@
%{贪恋烦恼的尘世。}%

@{焦蛾爛蠒自處餘殃。}@
%{像飞蛾扑火，自取灭亡，自找余殃。}%

@{籠鳥鼎魚翻稱快樂。}@
%{像笼中的鸟，鼎里的鱼，还自称快乐。}%

@{皆由善力微而業力勝。}@
%{都是因为善力微弱，业力强大。}%

@{信根少而罪根多。}@
%{因为信根薄弱，罪根深厚。}%

@{是以三界茫茫四生擾擾。}@
%{所以，三界无边无际，四生轮回，纷乱不堪。}%

@{盡貪生而兀兀孰解知歸。}@
%{大家都贪恋着生命，却不知道如何回归本源。}%

@{悉逐業以悠悠。}@
%{都随着业力漂泊，浑浑噩噩。}%

@{不求出要。}@
%{不寻求解脱之道。}%

@{過去生死却石難窮。}@
%{过去的生死轮回，犹如石头般难以穷尽。}%

@{未來輪迴芥城何盡。}@
%{未来的轮回，如同芥子城般，没有尽头。}%

@{匪夙生之有幸豈得遇於斯因。}@
%{如果不是前世有幸，怎么能够遇到这个因缘呢？}%

@{擊鼓開囹圄之門宜應速出。}@
%{敲响鼓声，打开牢笼的门，应该尽快出去。}%

@{逢舟濟沈淪之難詎可遲疑。}@
%{遇到船只，可以渡过沉沦之苦，怎么可以犹豫呢？}%

@{敬順金文善隨佛學。}@
%{恭敬地遵循佛经的教诲，善于跟随佛法学习。}%

@{不聞不解者可痛可傷焉。}@
%{不听不懂的人，真是令人痛心疾首啊！}%

@{矧茲五濁惡世四面火焚。}@
%{何况在这个五浊恶世，四面火焚。}%

@{惟佛一人力能救援既聞妙法宜植淨緣。}@
%{只有佛陀一人，才能救度我们，既然听到了妙法，就应该种下清净的因缘。}%

@{一念信誠萬德因種。}@
%{一念真诚，就能种下万德的因。}%

@{思齊先哲希悟真常。}@
%{希望效法先贤，希求悟证真常。}%

@{普皆如說奉行盡心頂禮信受。}@
%{大家都要按照佛陀所说去奉行，尽心顶礼，信受奉行。}%

@{斷疑生信}@
%{断除疑虑，生起信心}%

@{經云。}@
%{经书上说：}%

@{得為人難。}@
%{得到人身很难。}%

@{六根完具難。}@
%{六根完具很难。}%

@{生中國難。}@
%{生在中国很难。}%

@{值佛道難。}@
%{遇到佛法很难。}%

@{興信心難。}@
%{生起信心很难。}%

@{蓋嘗論信之難興者。}@
%{之所以说信心难生，是因为有疑虑。}%

@{緣有其疑也。}@
%{之所以说信心难生，是因为有疑虑。}%

@{不斷其疑何由生信。}@
%{不消除疑虑，怎么能够生起信心呢？}%

@{故金剛經云。}@
%{所以金刚经说：}%

@{正信希有。}@
%{正信难得。}%

@{法華經云。}@
%{法华经说：}%

@{以信得入。}@
%{凭着信心才能进入佛法。}%

@{王龍舒云。}@
%{王龙舒说：}%

@{佛為人醫王能救一切病。}@
%{佛陀是人间的医王，能够救治一切疾病。}%

@{不能救命盡者佛能度一切人。}@
%{佛陀虽然不能救活所有的人，但能够度化一切众生。}%

@{不能度不信者。}@
%{佛陀不能度化不信的人。}%

@{蓋信者一念之真誠也。}@
%{因为信心是一念的真诚。}%

@{若人心念要去身則隨去。}@
%{如果心念想要离开身体，身体就会跟着离开。}%

@{心念欲住身則隨住。}@
%{如果心念想要住在身体里，身体就会跟着住下来。}%

@{是知此身隨心念動。}@
%{由此可知，这个身体随着心念而动。}%

@{有念欲去而身被牽繫者。}@
%{如果心念想要离开，而身体被牵绊着。}%

@{色身壞時惟一念而已。}@
%{色身坏的时候，只凭一念而已。}%

@{一念到處則無處不到。}@
%{一念能够到达任何地方，没有地方不能到达。}%

@{所以一念信心念佛。}@
%{所以，只要一念信心念佛。}%

@{求生淨土則必到淨土。}@
%{求生净土，就一定能够到达净土。}%

@{況吾佛世尊諸大菩薩。}@
%{何况佛陀世尊和诸大菩萨。}%

@{又有本誓願力接引往生乎。}@
%{还有本誓愿力接引我们往生呢？}%

@{或問之曰。}@
%{有人问：}%

@{今有一生持戒念佛。}@
%{现在有人一生持戒念佛。}%

@{臨終不得生淨土者何也。}@
%{临终却不能往生净土，这是为什么呢？}%

@{答曰。}@
%{回答说：}%

@{此蓋信力不深。}@
%{这是因为信心不够深厚。}%

@{行願之有虧也。}@
%{修行和发愿有所欠缺。}%

@{兼不曾發菩提之大心。}@
%{并且没有发菩提心。}%

@{又不曾斷十惡之邪行。}@
%{也没有断除十恶的邪行。}%

@{雖曰修行未嘗言行相應。}@
%{虽然说是修行，但言行不一致。}%

@{雖曰念佛未嘗淨念相繼。}@
%{虽然说是念佛，但没有清净的念佛相续。}%

@{既無真實之功。}@
%{既然没有真实的功德。}%

@{焉得淨土之報也。}@
%{怎么能够得到净土的果报呢？}%

@{淨名經云。}@
%{净名经说：}%

@{隨其心淨則佛土淨。}@
%{心清净，佛土就清净。}%

@{六祖壇經云。}@
%{六祖坛经说：}%

@{心中若無不善西方去此不遙。}@
%{心中没有不善，西方离这里并不遥远。}%

@{若懷不善之心念佛求生難到。}@
%{如果心中怀有不善，念佛求生就很难到达。}%

@{不斷十惡之心何佛而來迎。}@
%{没有断除十恶之心，哪有佛来迎接呢？}%

@{請大師故。}@
%{请大师说：}%

@{見世人不務淨心只是口念佛名。}@
%{看到世人，不注重内心清净，只是口念佛号。}%

@{於其心外求佛妄想執著。}@
%{在内心之外求佛，妄想执着。}%

@{而不自淨其心。}@
%{而不自觉地净化自己的心。}%

@{心造諸惡是謂自喪其佛。}@
%{内心造作各种恶业，就是自己丧失了佛性。}%

@{又何更別求佛哉。}@
%{又何必再另外求佛呢？}%

@{所以道迷即眾生悟即佛也。}@
%{所以，迷失了道就是众生，觉悟了道就是佛。}%

@{圓通法門云。}@
%{圆通法门说：}%

@{憶佛念佛現前當來必定見佛。}@
%{忆佛念佛，现在和将来一定会见到佛。}%

@{蓋憶念者是心念也。}@
%{因为忆念就是心念。}%

@{信心清淨。}@
%{信心清净。}%

@{信行願俱備也。}@
%{信行愿都具备了。}%

@{楞嚴云。}@
%{楞严经说：}%

@{一根既返源六根成解脫。}@
%{一根返本归源，六根就获得解脱。}%

@{十惡化為十善。}@
%{十恶转化为十善。}%

@{六識化六神通。}@
%{六识转化为六神通。}%

@{所以正信心中憶念佛名。}@
%{所以，正信的人心中忆念佛号。}%

@{淨念相繼自性彌陀現前。}@
%{清净的念佛相续，自性弥陀就现前。}%

@{此感彼應臨命終時。}@
%{这样一来，感应道交，临命终时。}%

@{安得不見佛而生淨土乎。}@
%{怎么能够不见佛而往生净土呢？}%

@{凡修淨業者。}@
%{凡是修行净土法门的人。}%

@{當信佛之言。}@
%{应该相信佛陀的话语。}%

@{而行佛之行。}@
%{并且按照佛陀的教诲去修行。}%

@{心口既不相違。}@
%{内心和口语一致。}%

@{因果決然不昧耳。}@
%{因果关系决然不会错乱。}%

@{若也聞而不信信而不行。}@
%{如果听到而不信，信而不行。}%

@{由畫餅之不充飢也。}@
%{就像画饼充饥一样。}%

@{心既無信則生疑謗。}@
%{内心没有信心，就会产生怀疑和诽谤。}%

@{既興疑謗則自昧其心。}@
%{一旦产生怀疑和诽谤，就会蒙蔽自己的心。}%

@{自昧其心則淨土遠之遠矣。}@
%{蒙蔽了自己的心，净土就离我们越来越远了。}%

@{故香山周居士。}@
%{所以，香山周居士。}%

@{述慈照蓮宗懺序云。}@
%{在慈照莲宗忏悔文中说：}%

@{導師者化物有方誨人存理。}@
%{导师能够教化众生，教导人们存心向善，遵循道理。}%

@{以深信為能入破諸闡提。}@
%{以深信为能够进入佛法，破除一切阐提。}%

@{專念佛為行門令心不亂。}@
%{专念佛号为修行的方法，使心不乱。}%

@{加菩提願心為根本度脫眾生。}@
%{加上菩提心和发愿心为根本，度脱众生。}%

@{備此三者乃極樂佛土上上品之可登。}@
%{具备这三种，就能登上极乐佛土的上上品。}%

@{毘盧果海流流而易到也。}@
%{就像在毘卢果海中，漂流就能轻易到达。}%

@{又云。}@
%{又说：}%

@{此事人人本具箇箇圓成。}@
%{这件事每个人本来就具足，个个圆满成就。}%

@{無信行願三字因茲淪墜。}@
%{没有信行愿这三个字，就会因此而沉沦。}%

@{嗚呼凡為人者。}@
%{唉，凡是做人的人。}%

@{不在福德不在尊貴。}@
%{不在于福德，不在于尊贵。}%

@{不在聰明不在相貌。}@
%{不在于聪明，不在于相貌。}%

@{但具信行願者。}@
%{只要具备信行愿。}%

@{乃生淨土之資糧也。}@
%{就是往生净土的资粮。}%

@{四料楝云。}@
%{四料简说：}%

@{從是西方過十萬億佛土有世界名曰極樂。}@
%{从这里往西，经过十万亿佛土，有一个世界，叫做极乐世界。}%

@{是約遠也。}@
%{这是说它很远。}%

@{若一念信心念佛即到西方。}@
%{如果一念信心念佛，就能够立刻到达西方。}%

@{是約近也。}@
%{这是说它很近。}%

@{故云亦近而亦遠在人之信願。}@
%{所以说，极乐世界既近又远，取决于人的信心和愿力。}%

@{信不隔絲毫疑而生死轉。}@
%{信心不夹杂一丝怀疑，就能转生死。}%

@{又云。}@
%{又说：}%

@{從是西方過迢迢十萬程。}@
%{从这里往西，经过遥远的十万程。}%

@{資糧若具足何愁去不成是也。}@
%{如果资粮具足，何愁去不了呢？}%

@{所以道信為萬善之首。}@
%{所以，道信是万善之首。}%

@{信為百行之宗。}@
%{信心是百行之宗。}%

@{華嚴以十信為成佛之始。}@
%{华严经以十信作为成佛的开始。}%

@{法華以正信為入道之門。}@
%{法华经以正信作为入道的门。}%

@{五根以信根為先。}@
%{五根以信根为先。}%

@{五力以信力為最。}@
%{五力以信力为最。}%

@{是故三世諸佛諸大菩薩歷代祖師。}@
%{所以，三世诸佛，诸大菩萨，历代祖师。}%

@{修諸功行具大願力入佛境界成就菩提。}@
%{修持各种功德，具足大愿力，进入佛的境界，成就菩提。}%

@{未有不從這箇信字而入也。}@
%{没有一个不是从这个信字而入的。}%

@{鸞法師遇留支觀經。}@
%{鸾法師遇到留支观经。}%

@{即焚仙經。}@
%{就焚烧了仙经。}%

@{而修淨業豈不是這信字也。}@
%{而修持净土法门，难道不是这个信字吗？}%

@{白侍郎行也彌陀。}@
%{白侍郎走路的时候，念弥陀佛。}%

@{坐也彌陀。}@
%{坐着的时候，念弥陀佛。}%

@{不出這信字也。}@
%{没有超出这个信字。}%

@{蘇學士佩帶彌陀畫像。}@
%{苏学士佩带弥陀佛的画像。}%

@{行坐隨身。}@
%{走路坐着，都随身带着。}%

@{言是西方公據者。}@
%{说这是往生西方的凭证。}%

@{蓋亦不出這信字也。}@
%{也都没有超出这个信字。}%

@{張善和一生殺牛。}@
%{张善和一生杀牛。}%

@{臨終自見地獄相現。}@
%{临终的时候，自己看到了地狱的景象。}%

@{遇僧教令念阿彌陀佛。}@
%{遇到僧人教他念阿弥陀佛。}%

@{念未十聲見佛接引往生淨土。}@
%{念了不到十声，就见到佛接引他往生净土。}%

@{即脫地獄。}@
%{立刻就脱离了地狱。}%

@{勸發信心}@
%{劝发信心}%

@{夫阿彌陀佛者。諸佛中尊四生之父。歸信者滅罪河沙。稱念者得福無量。凡欲念佛要起信心。若無信心空無所獲。是故肇法師云。是事如是者信之相也。是事不如是者不信之相也。夫信為入道之初宗。智為究竟之玄術。諸經首稱如是者信也。後曰奉行者智也。故彌陀經云。若有信者應當發願生彼國土。此是釋迦本師勸信處。又云。汝等當信是稱讚不可思議功德。此是六方諸佛勸信處又云。若人種善根疑則花不開。信心清淨者花開即見佛。此是往生論勸信處。又云。信者順之詞。信則所言之理順。順則師資之道成。經無豐約非信不傳。此是肇法師勸信處。又云。念佛法門不問道俗。不論男女不拘貴賤。修此法者唯要信心。此是大行和尚勸信處。}@
%{阿弥陀佛啊，是所有佛陀中最尊贵的，他是四种众生的父亲。归信他的人，罪业可以消除得像河沙一样多；称念他的人，可以获得无量的福报。凡是想要念佛的人，一定要先建立信心。如果没有信心，就什么都得不到。所以肇法师说：“事情本来如此，这就是信心的表现；事情本来不是如此，这就是没有信心的表现。” 信心是进入佛道的第一步，智慧是最终的玄妙方法。所有佛经开头都称“如是”的，这就是信心；后面说“奉行”的，这就是智慧。所以《弥陀经》说：“如果有信心的人，就应该发愿往生到那个国土。” 这是释迦牟尼佛劝人建立信心的地方。又说：“你们应该相信这些赞叹不可思議的功德。” 这是六方诸佛劝人建立信心的地方。又说：“如果有人种下善根，却有怀疑，那么花就不会开；信心清净的人，花开就能见到佛。” 这是《往生论》劝人建立信心的地方。又说：“信，是顺从的意思。相信，就是所讲的道理顺从，顺从，就是师徒之道成就。佛经没有丰富的内容，没有简洁的语言，没有信心就无法传播。” 这是肇法师劝人建立信心的地方。又说：“念佛法门，不分道俗，不论男女，不拘贵贱，修持这个法门的人，只要有信心就行了。” 这是大行和尚劝人建立信心的地方。}%

@{問曰。既言信者未知信何法門。答曰。信意者信憑經中佛說念佛定生淨土。信念佛定滅諸罪。信念佛定得佛護。信念佛定得佛證。信念佛臨終定得佛來迎接。信念佛不問眾生。同信之人皆得往生。信念佛往生定得不退地。信念佛生淨土。定不墮三惡道。所以勸信念佛。受此法持此念。則往生淨土必矣。故大行和尚遣念佛人。心唯信佛。佛則知之他心通故。口惟稱佛。佛則聞之天耳通故。身惟禮佛。佛則見之天眼通故。是以大行和尚以此念佛法。勸人生信心也。}@
%{有人问：“既然说要信，但不知道信什么法门呢？” 回答说：“信心，就是相信经中佛所说的念佛一定能往生净土，相信念佛一定能灭除罪业，相信念佛一定能得到佛的护佑，相信念佛一定能得到佛的证悟，相信念佛临终一定能得到佛来迎接，相信念佛不问众生，只要同信的人都能往生，相信念佛往生一定能得到不退地，相信念佛生到净土，一定不会堕落到三恶道。所以劝你相信念佛，受持这个法门，持着这个念，就能往生净土，这是必然的。” 所以大行和尚教导念佛的人，心里只信佛，佛就能知道，因为佛有他心通，口中只称佛，佛就能听到，因为佛有天耳通，身体只礼佛，佛就能看到，因为佛有天眼通。所以大行和尚用这个念佛法门，劝人建立信心。}%

@{又喻云。信心者猶如深栽果樹。根深故風吹不動。後著果實濟人飢渴。念佛之人亦復如是。要由深信得到西方。若無信心空無所獲。是故經言。十住菩薩一起信心念佛之後。縱遇惡緣寧捨身命不退信故。維摩云。深信堅固猶如金剛。法珍普照如雨甘露。念佛三昧從深信生。君看淨土恒沙佛。盡是當年正信人。}@
%{又比喻说：信心就像深深地栽种果树，根深，所以风吹不动，后来结出果实，可以救济人们的饥渴。念佛的人也是这样，要靠深信才能到达西方。如果没有信心，就什么都得不到。所以经上说，十住菩萨一起建立信心念佛之后，即使遇到恶缘，宁愿舍弃生命也不退信。维摩诘说：“深信坚固就像金刚，法宝普照就像雨露甘霖，念佛三昧是从深信中生出来的。你看净土恒沙佛，都是当年正信的人。”}%

@{夫四果之明不能測如來之奧。十地之聖不能窮極果之妙。矧茲凡夫生無慧目。陷三界有必固之妄。指淨土謬難轉之疑。疑之於身知五蘊之未滅。疑之於心知萬惑之未遣。疑之於行知六度之未濟。由漂沈未濟之行。期濟於已濟之域。在於其己猶惑之。況他人哉。此物情難信為道俗所共疑也。隋智者大師心興物冥。智將神會。乘時翼教異跡由著。因其所異考其所證。不知其三賢之人乎。十聖之人乎。其為應正等覺妙神化于難思乎。蓋亦思之不能知也。但大師念定總持昭晣行業期生淨土。間無容髮發大師之所有。存者其十疑之文乎。然其文遠取諸經。近取諸論。去就其意以取諸傳。無離經以備事。無飾辭以增巧。審辭達意。能極九品之所生。乃見大師之用心也。見大師之心者。其於淨土之有疑。所謂高天無雲群星列彩。經紀之所相殊。躔次之所相異。粲粲然無一而隱者也。予得其文載念能仁所稱淨土者。累焉諸佛證而成之者。或舌相以覆大千。或潮音而演梵說。余雖陋學不得以相輝映烏可無言哉。故於論首略陳一二以罄予所懷。雖辭膚理淺不足起大師之化。在其先佛證成之義且有歸矣。}@
%{四果的明智，不能测度如来的奥妙；十地的圣人，不能穷尽果位的奇妙。何况是凡夫，生来没有慧眼，陷入三界有生必死的妄念，指向净土，谬误难以转变，充满了怀疑。怀疑自己的身体，知道五蕴没有灭尽；怀疑自己的内心，知道万种烦恼没有遣除；怀疑自己的行为，知道六度没有圆满。因为漂泊不定，没有圆满的行为，想要依靠已经圆满的净土，自己都还迷惑，何况是别人呢？这真是人心难信，是道俗共同的疑虑。隋朝的智者大师，内心兴起，物我冥合，智慧将要神会，乘着时代的风气，翼护佛法，异迹由此著述。因为其异于常人，考察其所证悟的，不知道是三贤的人呢？还是十地的圣人呢？难道是应正等觉的妙神化，难以思量吗？大概也是思之不能知啊。但是大师念定总持，昭昭然行持善业，期盼往生净土，中间没有一丝一毫可以发扬大师的言行，留下的只有十疑的文字。然而这些文字，远取诸经，近取诸论，取舍其意，以取诸传，没有离开经论来补充说明，没有修饰辞藻来增加巧妙，审察词句，表达意思，能够极尽九品往生之处，才可见大师的用心。看见大师用心的人，对于净土的疑虑，就像高天无云，群星列彩，经纪之所相殊，运行次序之所相异，灿灿然没有一个隐藏的。我得到他的文章，记载着能仁所称赞的净土，是累积无数佛陀证悟而成就的，或者舌相覆盖大千世界，或者潮音演说梵语。我虽然学识浅陋，不能与之相辉映，怎么能无话可说呢？所以，在论述开头，略微陈述一二，以表达我的心怀。虽然语言肤浅，道理浅薄，不足以兴起大师的教化，但是对于先佛证成的义理，却是有所归宿了。}%

@{彌陀光明如大圓月遍照十方。水澄而清則月現全體。月非趣水而遽來。水動而濁則月無定光。月非捨水而遽去。在水則有清濁動靜。在月則無取捨去來。故華嚴解脫長者云。知一切佛猶如影像。自心如水。彼諸如來不來至此。我亦不往彼。我若欲見安樂世界阿彌陀佛隨意即見。是知眾生注念定見阿彌陀佛。此乃稱性實言非權教也。}@
%{弥陀光明如大圆月遍照十方。水澄清则月现全體，月并非趋向水而突然到来。水动而浑浊则月无定光，月并非舍弃水而突然离去。在水则有清浊动静，在月则无取舍去来。所以华严解脱长者说，知道一切佛犹如影像，自心如水。那些如来不来至此，我也没有前往彼处。我若想见安乐世界阿弥陀佛，随心即可见到。这就是说，众生专注念定，就能见到阿弥陀佛。这是称性实言，并非权教。}%

%{净土无欲非欲界，其国地居非色界，生有形相非无色界。一切众生未觉醒正觉，处在梦中，六道轮回不停歇。诸天虽然乐报，最终会衰败。修罗方嗔怒，战争互相胜负。旁生飞走，互相残食。鬼神幽阴，饥渴困逼。地狱长夜，痛苦号呼。能得生人趣，已经算是幸运了。然而生老病死，各种痛苦缠绕。唯有净邦，再无诸苦。}%

@{蓮苞託質無生苦也。寒暑不遷無老苦也。身無分段無病苦也。壽命無量無死苦也。無父母妻子無愛別離苦也。上善人聚無怨憎會苦也。花裓香食珍寶受用無求不得苦也。觀照空寂無蘊苦也。悲濟有情欲生則生。不住寂滅非二乘也。智照生死得不退轉非凡夫也。三界蕩然。}@
%{莲苞托质，无生苦。寒暑不迁，无老苦。身无分段，无病苦。寿命无量，无死苦。无父母妻子，无爱别离苦。上善人聚，无怨憎会苦。花裓香食珍宝受用，无求不得苦。观照空寂，无蕴苦。悲济有情，欲生则生。不住寂灭，非二乘。智照生死，得得不退转，非凡夫。三界荡然。}%

@{譬如四裔丘陵坑坎腐穢所積。溪壑阻絕孰為津梁迺有狂人迷路。于此惡獸魑魅惱害。雜居刀兵水火。或時傷暴風霜霹靂凌厲摧懾罔知城域可以庇覆。飲食衣服未或充足。甘受是苦不求安樂。}@
%{譬如四裔丘陵坑坎腐秽所积，溪壑阻绝，谁为津梁？乃有狂人迷路，在此恶兽魑魅恼害，杂居刀兵水火。有时伤暴风霜霹雳凌厉摧摄，罔知城域可以庇覆。饮食衣服未或充足，甘受是苦不求安乐。}%

@{有佛釋迦是大導師。指清淨土是安樂國。無量壽佛是淨土師。爾諸眾生若生彼土則無諸惱。不聞知者固可哀憐。亦有善士發三種心。不求生者尤可嗟惜。}@
%{有佛释迦是大导师，指清净土是安乐国。无量寿佛是净土师。尔诸众生若生彼土，则无诸恼。不闻知者固可哀怜。也有善士发三种心，不求生者尤可嗟惜。}%

@{一曰。吾當超佛越祖。淨土不足生也。二曰。處處皆淨土西方不必生也。三曰。極樂聖域。我輩凡夫不能生也。夫行海無量。普賢願見彌陀佛國雖空。維摩常修淨土。十方如來有廣舌之讚。十方菩薩有同往之心。試自忖量孰與諸聖。謂不足生者何其自欺哉。}@
%{一曰：我当超佛越祖，净土不足生也。二曰：处处皆净土，西方不必生也。三曰：极乐圣域，我辈凡夫不能生也。夫行海无量，普贤愿见弥陀佛国虽空。维摩常修净土。十方如来有广舌之赞。十方菩萨有同往之心。试自忖量，孰与诸圣？谓不足生者，何其自欺哉！}%

@{至如龍猛祖師也。楞伽經有預記之文。天親教宗也。無量論有求生之偈。慈恩通讚首稱十勝。智者析理明辯十疑。彼皆上哲精進往生。謂不必生者何其自慢哉。}@
%{至如龙猛祖师，楞伽经有预记之文。天亲教宗，无量论有求生之偈。慈恩通赞首称十胜。智者析理明辩十疑。彼皆上哲精进往生。谓不必生者，何其自慢哉！}%

@{火車可滅舟石不沈。現華報者莫甚於張馗。十念而超勝處。入地獄者莫速於雄俊。再甦而證妙因。世人愆尤未必若此。謂不能生者何其自棄哉。}@
%{火车可灭，舟石不沉。现华报者莫甚于张馗，十念而超胜处。入地狱者莫速于雄俊，再苏而证妙因。世人愆尤未必若此。谓不能生者，何其自弃哉！}%

@{般舟三昧經云。跌陀和菩薩問佛。未來眾生。云何得見十方諸佛。佛教令念阿彌陀佛。即見十方諸佛。寶積經云。他方眾生聞無量壽如來名。迺至能發一念淨信。歡喜愛樂所有善根回向願生彼國者。隨願皆生得不退轉。}@
%{般舟三昧经云：跌陀和菩萨问佛，未来众生，云何得见十方诸佛？佛教令念阿弥陀佛，即见十方诸佛。宝积经云：他方众生闻无量寿如来名，乃至于能发一念净信，欢喜爱乐所有善根回向愿生彼国者，随愿皆生得不退转。}%

@{此皆佛言也。不信佛言。何言可信。不生淨土。何土可生。自欺自慢自棄己靈流入輪迴是誰之咎。四十八願悉為度生。一十六觀同歸繫念。一念既信已投種於寶池。眾善相資定化生於金地。毋輒悔惰誤認疑城。即時蓮開得解脫道。唯心淨土自性彌陀。大光明中決無魔事。}@
%{此皆佛言也。不信佛言，何言可信？不生净土，何土可生？自欺自慢，自弃己灵，流入轮回，是谁之咎？四十八愿悉为度生。一十六观同归系念。一念既信，已投种于宝池。众善相资，定化生于金地。毋辄悔惰，误认疑城。即时莲开，得解脱道。唯心净土，自性弥陀。大光明中，决无魔事。}%

@{決疑諸者王敏仲侍郎之編也。開釋疑情徑超信地。其載要旨最為詳審。蓋安養國之鄉導也。若登彼岸舟固可忘。來者問津斯言毋忽。}@
%{决疑诸者，王敏仲侍郎之编也。开释疑情，径超信地。其载要旨最为详审。盖安养国之乡导也。若登彼岸，舟固可忘。来者问津，斯言毋忽。}%
