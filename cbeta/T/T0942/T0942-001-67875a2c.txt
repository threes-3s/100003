@{釋迦佛讚}@
%{釋迦佛讚}%

@{內廷清字經館行走唐古忒學教習闡福寺副達喇嘛薩穆丹達爾吉譯}@
%{內廷清字經館行走唐古忒學教習闡福寺副達喇嘛薩穆丹達爾吉譯}%

@{納摩沙喇伊麻尼鴉徂郭卡阿鴉}@
%{南无释迦牟尼佛}%

@{三時及釋冤，成道與龍乾，}@
%{三世一切冤亲，成道时与龙种，}%

@{悉戴上寶尊，妙莊足蓮下。}@
%{都戴着上宝尊，妙庄严足莲花下。}%

@{金光朗耀威德身，乾闥妙音難比語，}@
%{金光朗照威德身，乾闼婆妙音难以比喻，}%

@{億日光潔清明心，導引眾生釋迦尊。}@
%{亿日光洁清明心，引导众生释迦尊。}%

@{如清天列諸星宿，遍照海中悉澈明，}@
%{如清天列诸星宿，遍照海中悉澈明，}%

@{誠意海內眾德像，分明不共極顯然。}@
%{诚意海内众德像，分明不共极显然。}%

@{手指相扶心間持，如月映照少水生，}@
%{手指相扶心间持，如月映照少水生，}%

@{一心現前與尊身，微讚世尊祈悲憐。}@
%{一心现前与尊身，微赞世尊祈悲怜。}%

@{諸勇猛菩薩圍繞，眾羅漢聚會排前，}@
%{诸勇猛菩萨围绕，众罗汉聚会排前，}%

@{如星在天皆拱月，似群牛中之勝尊。}@
%{如星在天皆拱月，似群牛中之胜尊。}%

@{光現千輻網遶身，其輝遍滿虛空界，}@
%{光现千辐网绕身，其辉遍满虚空界，}%

@{種種神變施無礙，來如鵞王飛降臨。}@
%{种种神变施无碍，来如鹅王飞降临。}%

@{爾時因昔修善業，百千人天應成熟，}@
%{尔时因昔修善业，百千人天应成熟，}%

@{將覩尊面即解脫，諸縛百罪暴心捐。}@
%{将睹尊面即解脱，诸缚百罪暴心捐。}%

@{菩提樹下英勇坐，以慈慧兵鎮魔君，}@
%{菩提树下英勇坐，以慈慧兵镇魔君，}%

@{威力降伏皆已盡，猶如烈風掃黑雲。}@
%{威力降伏皆已尽，犹如烈风扫黑云。}%

@{不著甲冑器，獨與億萬魔，}@
%{不著甲胄器，独与亿万魔，}%

@{全勝無餘陣，除尊有誰能？}@
%{全胜无余阵，除尊有谁能？}%

@{世尊仁慈火，極煎欲天心，}@
%{世尊仁慈火，极煎欲天心，}%

@{慈海於有情，親疎無不均。}@
%{慈海于有情，亲疏无不均。}%

@{為利眾生故，絲毫無倦時，}@
%{为利众生故，丝毫不倦时，}%

@{眾生在尊前，誦德亦無倦。}@
%{众生在尊前，诵德亦无倦。}%

@{心深莫測如淵海，語善妙若天鼓音，}@
%{心深莫测如渊海，语善妙若天鼓音，}%

@{身勝現似須彌臺，見聞憶念悉不空。}@
%{身胜现似须弥台，见闻忆念悉不空。}%

@{所有世界諸眾生，一時共稟所疑義，}@
%{所有世界诸众生，一时共禀所疑义，}%

@{隨彼一一各等前，同時示現身及語，}@
%{随彼一一各等前，同时示现身及语，}%

@{即斷諸惑已盡矣。}@
%{即断诸惑已尽矣。}%

@{聖身語意之密理，菩薩緣覺及羅漢，}@
%{圣身语意之密理，菩萨缘觉及罗汉，}%

@{任爾觀察非智境，何況釋梵世間等。}@
%{任尔观察非智境，何况释梵世间等。}%

@{如金翅王行虛空，有時力弱即止回，}@
%{如金翅王行虚空，有时力弱即止回，}%

@{惟有虛空無盡止，言佛功德亦如是，}@
%{唯有虚空无尽止，言佛功德亦如是，}%

@{亦如大鵬奮翼飛，途間微雀勉隨行。}@
%{亦如大鹏奋翼飞，途间微雀勉随行。}%

@{十力自在所行境，彼勝道中雖欲入，}@
%{十力自在所行境，彼胜道中虽欲入，}%

@{見至理之智眼盲，菩提厭離心又窮，}@
%{见至理之智眼盲，菩提厌离心又穷，}%

@{大力煩惱冤常擾，復陷我執之深窟，}@
%{大力烦恼冤常扰，复陷我执之深窟，}%

@{如是垢辱可歎處，設被慈尊再棄置，}@
%{如是垢辱可叹处，设被慈尊再弃置，}%

@{大慈憐憫諸卑下，非歸依仁而誰持？}@
%{大慈怜悯诸卑下，非归依仁而谁持？}%

@{五濁惡世儳雜境，他尊棄時仁全持，}@
%{五浊恶世儳杂境，他尊弃时仁全持，}%

@{諸佛菩薩咸讚歎，直比白蓮今柰何。}@
%{诸佛菩萨咸赞叹，直比白莲今奈何。}%

@{世尊事業為調伏，雖無微舍不觀察，}@
%{世尊事业为调伏，虽无微舍不观察，}%

@{盡因惡緣自作罪，故於爾尊無過時。}@
%{尽因恶缘自作罪，故于尔尊无过时。}%

@{我願從今修菩提，乃至成道伏魔君，}@
%{我愿从今修菩提，乃至成道伏魔君，}%

@{生生世尊常隨護，承受甘露語無厭。}@
%{生生世尊常随护，承受甘露语无厌。}%

@{右為釋迦牟尼讚文，謂速得攝受因。持三藏弟子將𭘏𭄢磋祈請多聞大德慧賢名師在西藏雪山洞萵德工𭄻之神祠著作。}@
%{右为释迦牟尼赞文，谓速得摄受因。持三藏弟子将𭘏𭄢磋祈请多闻大德慧贤名师在西藏雪山洞萵德工𭄻之神祠著作。}%

@{如來體微妙云何，及於眷屬共壽量，}@
%{如来体微妙云何，及于眷属共寿量，}%

@{境界及於號云何，願我等皆亦復然。}@
%{境界及于号云何，愿我等皆亦复然。}%

@{讚祝世尊微善力，我等隨方所在處，}@
%{赞祝世尊微善力，我等随方所在处，}%

@{病魔貧爭盡消除，善祥增長祈皆賜。}@
%{病魔贫争尽消除，善祥增长祈皆赐。}%

@{一切諸佛興於世，聖教顯明如日光，}@
%{一切诸佛兴于世，圣教显明如日光，}%

@{持教相和如兄弟，願施正教恒吉祥。}@
%{持教相和如兄弟，愿施正教恒吉祥。}%

@{右十二句，誦者之願文，隨前讚誦之可也。}@
%{右十二句，诵者之愿文，随前赞诵之可也。}%
