@{定蘊第七中攝納息第三}@

@{攝可得相應成不得捨退}@
%{摄可得相应成不得舍退}%

@{頓漸滅依定此章願具說}@
%{顿渐灭依定此章愿具说}%

@{十想。謂：無常想。無常苦想。苦無我想。死想。不淨想。厭食想。一切世間不可樂想。斷想。離想。滅想。}@
%{十想，就是：无常想，无常苦想，苦无我想，死想，不净想，厌食想，一切世间不可乐想，断想，离想，灭想。}%

@{四靜慮。四無量。四無色。八解脫。八勝處。十遍處。八智。三三摩地。}@
%{四静虑，四无量，四无色，八解脱，八胜处，十遍处，八智，三三摩地。}%

@{無常想等，攝幾靜慮等耶？}@
%{无常想等等，摄几静虑等等呢？}%

@{答：無常想，攝四靜慮、四無色、四解脫。}@
%{答：无常想，摄四静虑、四无色、四解脱。}%

@{如無常想；無常苦想、苦無我想、死想、斷想、離想、滅想，亦爾。}@
%{如同无常想一样，无常苦想、苦无我想、死想、断想、离想、灭想，也一样。}%

@{不淨想，攝第三、第四靜慮，初二解脫。}@
%{不净想，摄第三、第四静虑，初二解脱。}%

@{如不淨想；厭食想，亦爾。}@
%{如同不净想一样，厌食想，也一样。}%

@{一切世間不可樂想，攝第三、第四靜慮。}@
%{一切世间不可乐想，摄第三、第四静虑。}%

@{初靜慮等，攝幾靜慮等耶？}@
%{初静虑等等，摄几静虑等等呢？}%

@{答：初靜慮，攝初靜慮、四無量、初二解脫、前四勝處、八智、三三摩地。}@
%{答：初静虑，摄初静虑、四无量、初二解脱、前四胜处、八智、三三摩地。}%

@{第二靜慮，攝第二靜慮、四無量、初二解脫、前四勝處、八智、三三摩地。}@
%{第二静虑，摄第二静虑、四无量、初二解脱、前四胜处、八智、三三摩地。}%

@{第三靜慮，攝第三靜慮、三無量、八智、三三摩地。}@
%{第三静虑，摄第三静虑、三无量、八智、三三摩地。}%

@{第四靜慮，攝第四靜慮、三無量、淨解脫、後四勝處、前八遍處、八智、三三摩地。}@
%{第四静虑，摄第四静虑、三无量、净解脱、后四胜处、前八遍处、八智、三三摩地。}%

@{慈無量等，攝幾無量等耶？}@
%{慈无量等等，摄几无量等等呢？}%

@{答：慈攝慈世俗智，乃至捨攝捨世俗智。}@
%{答：慈摄慈世俗智，乃至舍摄舍世俗智。}%

@{空無邊處等，攝幾無色等耶？}@
%{空无边处等等，摄几无色等等呢？}%

@{答：空無邊處，攝空無邊處，及彼解脫，彼遍處，六智，三三摩地。}@
%{答：空无边处，摄空无边处，及彼解脱，彼遍处，六智，三三摩地。}%

@{識無邊處，攝識無邊處，及彼解脫，彼遍處，六智，三三摩地。}@
%{识无边处，摄识无边处，及彼解脱，彼遍处，六智，三三摩地。}%

@{無所有處，攝無所有處，及彼解脫，六智，三三摩地。}@
%{无所有处，摄无所有处，及彼解脱，六智，三三摩地。}%

@{非想非非想處，攝非想非非想處，及彼解脫，滅想受解脫，世俗智。}@
%{非想非非想处，摄非想非非想处，及彼解脱，灭想受解脱，世俗智。}%

@{初解脫等，攝幾解脫等耶？}@
%{初解脱等等，摄几解脱等等呢？}%

@{答：初、第二、第三解脫，攝初、第二、第三解脫世俗智。}@
%{答：初、第二、第三解脱，摄初、第二、第三解脱世俗智。}%

@{空無邊處解脫，攝空無邊處解脫，及彼遍處，六智，三三摩地。}@
%{空无边处解脱，摄空无边处解脱，及彼遍处，六智，三三摩地。}%

@{識無邊處解脫，攝識無邊處解脫，及彼遍處，六智，三三摩地。}@
%{识无边处解脱，摄识无边处解脱，及彼遍处，六智，三三摩地。}%

@{無所有處解脫，攝無所有處解脫，六智，三三摩地。}@
%{无所有处解脱，摄无所有处解脱，六智，三三摩地。}%

@{非想非非想處解脫，攝非想非非想處解脫，世俗智。}@
%{非想非非想处解脱，摄非想非非想处解脱，世俗智。}%

@{滅想受解脫，攝滅想受解脫。}@
%{灭想受解脱，摄灭想受解脱。}%

@{初勝處等，攝幾勝處等耶？}@
%{初胜处等等，摄几胜处等等呢？}%

@{答：初勝處，攝初勝處世俗智，乃至第八勝處，攝第八勝處世俗智。}@
%{答：初胜处，摄初胜处世俗智，乃至第八胜处，摄第八胜处世俗智。}%

@{初遍處等，攝幾遍處等耶？}@
%{初遍处等等，摄几遍处等等呢？}%

@{答：初遍處，攝初遍處世俗智，乃至第十遍處，攝第十遍處世俗智。}@
%{答：初遍处，摄初遍处世俗智，乃至第十遍处，摄第十遍处世俗智。}%

@{法智等，攝幾智等耶？}@
%{法智等等，摄几智等等呢？}%

@{答：法智，攝法智，五智少分。}@
%{答：法智，摄法智，五智少分。}%

@{類智，攝類智，五智少分。}@
%{类智，摄类智，五智少分。}%

@{他心智，攝他心智，四智少分。}@
%{他心智，摄他心智，四智少分。}%

@{世俗智，攝世俗智，一智少分。}@
%{世俗智，摄世俗智，一智少分。}%

@{苦智，攝苦智，二智少分。}@
%{苦智，摄苦智，二智少分。}%

@{集智，攝集智，二智少分。}@
%{集智，摄集智，二智少分。}%

@{滅智，攝滅智，三智少分。}@
%{灭智，摄灭智，三智少分。}%

@{道智，攝道智，三智少分。}@
%{道智，摄道智，三智少分。}%

@{空三摩地等，攝幾三摩地耶？}@
%{空三摩地等等，摄几三摩地呢？}%

@{答：空攝空，無願攝無願，無相攝無相。}@
%{答：空摄空，无愿摄无愿，无相摄无相。}%

@{如攝；可得，亦爾。}@
%{如同摄一样，可得，也一样。}%

@{無常想等，與幾靜慮等相應耶？}@
%{无常想等等，与几静虑等等相应呢？}%

@{答：無常想，與四靜慮，四無色，四解脫，四智，一三摩地相應。}@
%{答：无常想，与四静虑，四无色，四解脱，四智，一三摩地相应。}%

@{如無常想；無常苦想、苦無我想、死想、斷想、離想、滅想，亦爾。}@
%{如同无常想一样，无常苦想、苦无我想、死想、断想、离想、灭想，也一样。}%
@{不淨想，與後二靜慮，初二解脫，世俗智相應。}@
%{不净想，和后两种静虑，以及前两种解脱，都对应世俗智。}%

@{如不淨想；厭食想，亦爾。}@
%{比如不净想，厌食想也一样。}%

@{一切世間不可樂想，與後二靜慮，世俗智相應。}@
%{所有世间不可乐想，也和后两种静虑，都对应世俗智。}%

@{初靜慮等，與幾靜慮等相應耶？}@
%{初静虑等等，和哪些静虑等等对应呢？}%

@{答：初靜慮，與初靜慮，四無量，初二解脫，前四勝處，八智，三三摩地相應。}@
%{答：初静虑，和初静虑，四无量，前两种解脱，前四胜处，八智，三种三摩地对应。}%

@{第二靜慮，與第二靜慮，四無量，初二解脫，前四勝處，八智，三三摩地相應。}@
%{第二静虑，和第二静虑，四无量，前两种解脱，前四胜处，八智，三种三摩地对应。}%

@{第三靜慮，與第三靜慮，三無量，八智，三三摩地相應。}@
%{第三静虑，和第三静虑，三种无量，八智，三种三摩地对应。}%

@{第四靜慮，與第四靜慮，三無量，淨解脫，後四勝處，前八遍處，八智，三三摩地相應。}@
%{第四静虑，和第四静虑，三种无量，净解脱，后四胜处，前八遍处，八智，三种三摩地对应。}%

@{慈無量等，與幾無量等相應耶？}@
%{慈无量等等，和哪些无量等等对应呢？}%

@{答：慈與慈世俗智相應，乃至捨與捨世俗智相應。}@
%{答：慈和慈的世俗智对应，乃至舍和舍的世俗智对应。}%

@{空無邊處等，與幾無色等相應耶？}@
%{空无边处等等，和哪些无色等等对应呢？}%

@{答：空無邊處，與空無邊處，及彼解脫，彼遍處，六智，三三摩地相應。}@
%{答：空无边处，和空无边处，以及它的解脱，那个遍处，六智，三种三摩地对应。}%

@{識無邊處，與識無邊處，及彼解脫，彼遍處，六智，三三摩地相應。}@
%{识无边处，和识无边处，以及它的解脱，那个遍处，六智，三种三摩地对应。}%

@{無所有處，與無所有處，及彼解脫，六智，三三摩地相應。}@
%{无所有处，和无所有处，以及它的解脱，六智，三种三摩地对应。}%

@{非想非非想處，與非想非非想處，及彼解脫，世俗智相應。}@
%{非想非非想处，和非想非非想处，以及它的解脱，都对应世俗智。}%

@{初解脫等，與幾解脫等相應耶？}@
%{初解脱等等，和哪些解脱等等对应呢？}%

@{答：初、第二、第三解脫，與初、第二、第三解脫，世俗智相應。}@
%{答：初、第二、第三解脱，和初、第二、第三解脱，都对应世俗智。}%

@{空無邊處解脫，與空無邊處解脫，及彼遍處，六智，三三摩地相應。}@
%{空无边处解脱，和空无边处解脱，以及那个遍处，六智，三种三摩地对应。}%

@{識無邊處解脫，與識無邊處解脫，及彼遍處，六智，三三摩地相應。}@
%{识无边处解脱，和识无边处解脱，以及那个遍处，六智，三种三摩地对应。}%

@{無所有處解脫，與無所有處解脫，六智，三三摩地相應。}@
%{无所有处解脱，和无所有处解脱，六智，三种三摩地对应。}%

@{非想非非想處解脫，與非想非非想處解脫，世俗智相應。}@
%{非想非非想处解脱，和非想非非想处解脱，都对应世俗智。}%

@{滅想受解脫，非相應。}@
%{灭想受解脱，不对应。}%

@{初勝處等，與幾勝處等相應耶？}@
%{初胜处等等，和哪些胜处等等对应呢？}%

@{答：初勝處，與初勝處世俗智相應，乃至第八勝處，與第八勝處世俗智相應。}@
%{答：初胜处，和初胜处的世俗智对应，乃至第八胜处，和第八胜处的世俗智对应。}%

@{初遍處等，與幾遍處等相應耶？}@
%{初遍处等等，和哪些遍处等等对应呢？}%

@{答：初遍處，與初遍處世俗智相應，乃至第十遍處，與第十遍處世俗智相應。}@
%{答：初遍处，和初遍处的世俗智对应，乃至第十遍处，和第十遍处的世俗智对应。}%

@{法智等，與幾三摩地相應耶？}@
%{法智等等，和哪些三摩地对应呢？}%

@{答：法智，與三三摩地，少分相應。}@
%{答：法智，和三种三摩地，少分对应。}%

@{如法智；類智，亦爾。}@
%{比如法智，类智也一样。}%

@{他心智，與一三摩地，少分相應。}@
%{他心智，和一种三摩地，少分对应。}%

@{如他心智；集、滅、道智，亦爾。}@
%{比如他心智，集、灭、道智也一样。}%

@{苦智，與二三摩地，少分相應。}@
%{苦智，和两种三摩地，少分对应。}%

@{若成就初靜慮等，彼於四靜慮等幾成就？幾不成就？}@
%{如果成就初静虑等等，那么他对于四静虑等等，成就多少？不成就多少？}%

@{答：若成就初靜慮，彼於四靜慮：或一、二、三、四。}@
%{答：如果成就初静虑，那么他对于四静虑，可能成就一种，两种，三种，或者四种。}%

@{一者謂：梵世愛未盡。}@
%{一种是指：梵世爱还没有断尽。}%

@{二者謂：梵世愛盡，上愛未盡。}@
%{两种是指：梵世爱断尽了，但是上爱还没有断尽。}%

@{三者謂：極光淨愛盡，上愛未盡。}@
%{三种是指：极光净爱断尽了，但是上爱还没有断尽。}%

@{四者謂：遍淨愛盡。}@
%{四种是指：遍净爱断尽了。}%

@{於四無量：或無，或三，或四。}@
%{对于四无量：可能没有，或者三种，或者四种。}%

@{無者謂：生無色界。}@
%{没有是指：生于无色界。}%

@{三者謂：生遍淨、廣果。}@
%{三种是指：生于遍净，广果。}%

@{四者謂：生欲界、梵世、極光淨。}@
%{四种是指：生于欲界，梵世，极光净。}%

@{於四無色：或無，或一、二、三、四。}@
%{对于四无色：可能没有，或者一种，两种，三种，或者四种。}%

@{無者謂：色愛未盡。}@
%{没有是指：色爱还没有断尽。}%

@{一者謂：色愛盡，上愛未盡。}@
%{一种是指：色爱断尽了，但是上爱还没有断尽。}%

@{二者謂：空無邊處愛盡，上愛未盡。}@
%{两种是指：空无边处爱断尽了，但是上爱还没有断尽。}%

@{三者謂：識無邊處愛盡，上愛未盡。}@
%{三种是指：识无边处爱断尽了，但是上爱还没有断尽。}%

@{四者謂：無所有處愛盡。}@
%{四种是指：无所有处爱断尽了。}%

@{於八解脫：或無，或一、二、三、四、五、六、七、八。}@
%{对于八解脱：可能没有，或者一种，两种，三种，四种，五种，六种，七种，或者八种。}%

@{無者謂：生遍淨。遍淨愛未盡。}@
%{没有是指：生于遍净，遍净爱还没有断尽。}%

@{一者謂：生遍淨。遍淨愛盡，上愛未盡。}@
%{一种是指：生于遍净，遍净爱断尽了，但是上爱还没有断尽。}%

@{若生廣果，即彼愛未盡。}@
%{如果生于广果，那么他的爱还没有断尽。}%
@{若生空無邊處，即彼愛未盡。}@
%{如果生到空无边处，那时的爱还没有断尽。}%

@{二者謂：生欲界、梵世、極光淨，遍淨愛未盡。}@
%{第二种情况是：生到欲界、梵世、极光净，遍净的爱还没有断尽。}%

@{若生遍淨、廣果，廣果愛盡，上愛未盡。}@
%{如果生到遍净、广果，广果的爱断尽了，但上爱还没有断尽。}%

@{若生空無邊處，即彼愛盡，上愛未盡。}@
%{如果生到空无边处，那时的爱断尽了，但上爱还没有断尽。}%

@{若生識無邊處，即彼愛未盡。}@
%{如果生到识无边处，那时的爱还没有断尽。}%

@{三者謂：生欲界、梵世、極光淨，遍淨愛盡，上愛未盡。}@
%{第三种情况是：生到欲界、梵世、极光净，遍净的爱断尽了，但上爱还没有断尽。}%

@{若生遍淨、廣果，空無邊處愛盡，上愛未盡。}@
%{如果生到遍净、广果，空无边处那时的爱断尽了，但上爱还没有断尽。}%

@{若生空、識無邊處，識無邊處愛盡，上愛未盡。}@
%{如果生到空、识无边处，识无边处那时的爱断尽了，但上爱还没有断尽。}%

@{若生無所有處，即彼愛未盡。}@
%{如果生到无所有处，那时的爱还没有断尽。}%

@{四者謂：生欲界、梵世、極光淨，廣果愛盡，上愛未盡。}@
%{第四种情况是：生到欲界、梵世、极光净，广果的爱断尽了，但上爱还没有断尽。}%

@{若生遍淨、廣果，識無邊處愛盡，上愛未盡。}@
%{如果生到遍净、广果，识无边处那时的爱断尽了，但上爱还没有断尽。}%

@{若生空、識無邊處、無所有處，無所有處愛盡。}@
%{如果生到空、识无边处、无所有处，无所有处那时的爱断尽了。}%

@{若生非想非非想處，不得滅盡定。}@

%{或者没有，或者有一、二、三、四种情况。}%

@{無者謂生非想非非想處。}@
%{没有的情况，指的是生于非想非非想处。}%

@{一者謂生廣果。}@
%{一种情况指的是：生于广果。}%

@{二者謂生遍淨。}@
%{两种情况指的是：生于遍净。}%

@{三者謂生極光淨。}@
%{三种情况指的是：生于极光净。}%
