@{齎持蜜美食，以用奉上我。}@
%{带来蜂蜜和美食，用来供养我。}%

@{我尋便受之，自然極美多，}@
%{我马上就接受了，自然而然地非常美味，而且很多，}%

@{以施比丘僧，可意甚飽滿。}@
%{用来供养比丘僧众，他们都感到满意，非常饱足。}%

@{我應時生已，獼猴所作行，}@
%{我应时投生以后，曾经作为猕猴所做的行为，}%

@{度脫無徑路，便得甘露句。}@
%{度脱了无路可走的困境，就得到了甘露法语。}%

@{如我本所願，輒得如其意，}@
%{就像我原本所发愿的那样，总是能够如愿以偿，}%

@{供養佛世尊，所求則具足。}@
%{供养佛世尊，所求的一切都能够满足。}%

@{唯仁每悉念，我所作功德，}@
%{仁者啊，请您时刻记住，我所做的功德，}%

@{悉獲其果實，可意安隱吉。}@
%{都获得了相应的果报，令人满意，安稳吉祥。}%

@{如是出蜜尊，在比丘僧中，}@
%{像这样出蜜尊者，在比丘僧众中，}%

@{於阿耨達池，自說本所作。}@
%{在阿耨达池边，自己讲述了以前所做的事情。}%

 
 

@{世尊品第三十(五十偈)}@
%{第三十品 世尊品 (五十偈)}%

@{一切勝普明，一切世間最，}@
%{一切胜过普明，是世间最尊贵的，}%

@{得除盡諸垢，降一切眾會。}@
%{能够去除一切污垢，降伏一切大众。}%

@{諸通慧普見，大人一切暢，}@
%{具备一切神通智慧，遍观一切，大人无所不能，}%

@{度諸怨恐懼，法船濟彼岸。}@
%{度化一切怨恨恐惧，法船渡过彼岸。}%

@{曉了眾所化，欣然愍世間，}@
%{明了众生的种种变化，欣然悲悯世间，}%

@{矜傷脫眾生，以義一切救。}@
%{怜悯众生，以正法救度一切。}%

@{除去一切人，悉解諸繫縛，}@
%{除去一切众生的烦恼，解脱一切束缚，}%

@{一切人中最，說法為眾眼。}@
%{在一切人中最尊贵，说法如众生的眼睛。}%

@{大人無極慧，大雄極名聞，}@
%{大人拥有无边的智慧，大雄拥有极高的名声，}%

@{大光無極法，以度於最法。}@
%{大光拥有无边的法力，用来度化最难度化的众生。}%

@{大力化無黠，開化大明慧，}@
%{大力能够化解一切愚痴，开化众生的大智慧，}%

@{歡勸大眾人，大醫多所兼。}@
%{欢喜劝诫大众，是世间的大医，能够治疗各种疾病。}%

@{世尊壞眾恐，無上除諸憂，}@
%{世尊能够消除众生的恐惧，无上地解除一切忧愁，}%

@{佛仁為度脫，大牢獄閉繫。}@
%{佛陀的仁慈，能够度化一切被关在牢狱中的人。}%

@{大龍大師子，無著大比丘，}@
%{大龙大师子，无著的大比丘，}%

@{大智慧世尊，救濟眾塵勞。}@
%{大智慧的世尊，救济众生的烦恼。}%

@{精進有大力，方便大堅彊，}@
%{精进拥有大力，方便无比坚强，}%

@{降伏眾天民，大道寂靜安。}@
%{降伏一切天人和凡人，大道寂静安稳。}%

@{佛大天中天，一切諸鬼神，}@
%{佛陀是天中之天，一切鬼神，}%

@{悉禮智慧足，佛出哀世間。}@
%{都礼拜佛陀的智慧，佛陀哀悯世间。}%

@{恒在大生死，壞決羂羅網，}@
%{永远存在于生死轮回中，破除一切束缚的罗网，}%

@{神通無極哀，度脫大牢獄。}@
%{神通无边，哀悯众生，度化一切牢狱。}%

@{大龍大天人，於眾會最先，}@
%{大龙，大天人，在一切大众中，最先，}%

@{廣施無極施，已逮弘寂跡。}@
%{广施无边的布施，已经到达了寂静的境界。}%

@{尊長士仙人，已度諸尊法，}@
%{尊贵的仙人，已经度化了所有尊贵的法，}%

@{成就大弟子，導師德極尊。}@
%{成就了伟大的弟子，导师的德行无比尊贵。}%

@{眾祐中最上，無上除愁憂，}@
%{众生中最尊贵，无上地解除一切忧愁，}%

@{諸所度脫勝，一切相好尊。}@
%{度化一切众生，一切相好，无比尊贵。}%

@{斷絕諸色欲，拔濟諸恩愛，}@
%{断绝一切色欲，拔济一切恩爱，}%

@{時遊在龍王，阿耨達大池。}@
%{当时游历在龙王，阿耨达大池。}%

@{一切所作辦，踊在虛空中，}@
%{一切所做的事情，都圆满完成，在虚空中跳跃，}%

@{弟子眾圍繞，寂然有五百。}@
%{弟子们围绕着他，寂静地有五百人。}%

@{愍傷有極哀，慈護一切人，}@
%{悲悯众生，无比哀伤，慈爱地保护一切人，}%

@{觀察比丘眾，便自說是言：}@
%{观察着比丘僧众，就自己讲述说：}%

@{明聽我所語，前世之所造，}@
%{请你们仔细听我说，我前世所造的业，}%

@{身始有所作，今所獲餘殃。}@
%{我身开始做了一些事情，现在所获得的恶报。}%

@{吾昔宿命時，作人名文羅，}@
%{我以前宿命的时候，做人，名字叫文罗，}%

@{誹謗無瑕穢，善妙辟支佛。}@
%{诽谤了无瑕疵，善妙的辟支佛。}%

@{眾人大來會，縛束善妙士，}@
%{很多人都来参加集会，捆绑了那个善妙的圣者，}%

@{著杻械閉繫，須出如死囚。}@
%{戴上脚镣手铐，关押起来，就像要被处死一样。}%

@{吾時見沙門，得縛束苦惱，}@
%{我当时看到那个沙门，被捆绑折磨，}%

@{其心發慈哀，身則為救解。}@
%{心里生起慈悲，就想要去救他。}%

@{用是罪殃故，墮地獄甚久，}@
%{因为这样做的罪业，堕落到地狱里很久，}%

@{後來生人間，常為人所謗。}@
%{后来投生到人间，经常被人诽谤。}%

@{用是有餘殃，於此最後世，}@
%{因为这样做的恶报，在这个最后一世，}%

@{須陀利異道，共議誣謗我。}@
%{须陀利外道，一起议论诽谤我。}%

@{曾為婆羅門，博聞持道術，}@
%{曾经做过婆罗门，博学多闻，精通道术，}%

@{有五百學志，講術藂樹間。}@
%{有五百个学生，在树林间学习道术。}%

@{時有大神力，五通比丘來，}@
%{当时有神通广大，五种神通的比丘来了，}%

@{我見道人至，誹謗揚其惡。}@
%{我看到道人来了，就诽谤他，宣扬他的恶行。}%

@{仙人深愛欲，自高處樹間，}@
%{仙人沉迷于欲望，从树枝高处，}%

@{諸摩納聞之，便共效我宣。}@
%{摩纳们听到后，就一起效仿我，宣扬他的恶行。}%

@{時一切學志，家家行乞匃，}@
%{当时所有的学生，都挨家挨户地去乞讨，}%

@{大眾中誹謗，仙人有垢欲。}@
%{在大众中诽谤，说仙人有污秽的欲望。}%

@{緣是所犯罪，須陀利女人，}@
%{因为犯了这样的罪业，须陀利女人，}%

@{佛五百弟子，悉共被誹謗。}@
%{佛陀的五百个弟子，都被一起诽谤。}%

@{佛為一切明，有虛妄之謗，}@
%{佛陀是所有一切的明灯，知道这是虚妄的诽谤，}%

@{知世吒弟子，是為沙門耶？}@
%{知道世吒的弟子，是真正的沙门吗？}%

@{犯是罪殃已，便墮惡道中，}@
%{犯了这样的罪业，就堕落到恶道中，}%

@{生在太山獄，勤苦甚酷毒。}@
%{投生到太山狱，非常辛苦，非常痛苦。}%

@{以此有餘殃，旃遮摩尼女，}@
%{因为这样做的恶报，旃遮摩尼女人，}%

@{在大眾會中，虛妄掩殺佛。}@
%{在大众集会中，虚妄地想要杀害佛陀。}%

@{曾為三兄弟，而共諍錢財，}@
%{曾经是三个兄弟，一起争夺钱财，}%

@{推撲墜深谷，石抬以殺之。}@
%{推搡着对方，跌落到深谷里，用石头砸死他。}%

@{以是所犯罪，墮太山地獄，}@
%{因为犯了这样的罪业，堕落到太山地狱，}%

@{燒炙在黑繩，毒痛甚酷苦。}@
%{被烧烤在黑绳上，毒痛无比，非常痛苦。}%

@{以此有餘殃，調達石所抬，}@
%{因为这样做的恶报，调达用石头抬，}%

@{於是石墮落，中傷佛足指。}@
%{结果石头掉下来，伤了佛陀的脚趾。}%

@{乘船入江海，俱欲渡深水，}@
%{乘船进入江海，一起想要渡过深水，}%

@{時共載船上，拔刀殺賈人。}@
%{当时一起在船上，拔刀杀害了商人。}%

@{用犯此罪故，身墮地獄中，}@
%{因为犯了这样的罪业，身体就堕落到地狱里，}%

@{以是餘殃故，鐵刺見佛前。}@
%{因为这样做的恶报，就用铁刺来刺佛陀。}%

@{曾在捕魚肆，生為漁者子，}@
%{曾经在捕鱼的地方，投生为渔夫的儿子，}%

@{有捕殺魚者，我爾時生心。}@
%{有捕杀鱼的人，我当时起了杀心。}%

@{從是所犯罪，墮太山地獄，}@
%{因为犯了这样的罪业，堕落到太山地狱，}%

@{燒炙在黑繩，勤苦甚毒痛。}@
%{被烧烤在黑绳上，非常辛苦，非常痛苦。}%

@{隨樓勒國王，傷殺釋子時，}@
%{随楼勒国王，伤害杀害释迦子的时候，}%

@{以是有餘殃，於今得頭痛。}@
%{因为这样做的恶报，现在就头痛。}%

@{惟衛世尊時，罵詈其弟子，}@
%{侍奉世尊的时候，辱骂他的弟子，}%

@{不應食粳米，當令噉生麥。}@
%{不应该吃白米饭，应该让他们吃生麦。}%

@{用是所犯罪，坐口出惡言，}@
%{因为犯了这样的罪业，因为说了一些恶毒的话，}%

@{墮於黑繩獄，受苦不可計。}@
%{堕落到黑绳狱里，所受的苦难无法计算。}%

@{以此有餘殃，怨結婆羅門，}@
%{因为这样做的恶报，怨恨的婆罗门，}%

@{請我終一時，三月中噉麥。}@
%{请我终生，三个月之内只吃麦子。}%

@{曾為治病醫，時療尊者子，}@
%{曾经做过治病的医生，当时治疗尊者的儿子，}%

@{合藥分倒錯，令疾轉增劇。}@
%{把药方弄错了，结果病情更加严重。}%

@{用犯此罪故，墮地獄甚苦，}@
%{因为犯了这样的罪业，堕落到地狱里，非常痛苦，}%

@{以此有餘殃，是故得下利。}@
%{因为这样做的恶报，所以就患了痢疾。}%

@{吾昔前世時，曾為手搏師，}@
%{我以前前世的时候，曾经做过摔跤师傅，}%

@{與力士相撲，害殺有佛子。}@
%{和大力士摔跤，害死了佛陀的儿子。}%

@{用犯此罪故，受苦難訾量，}@
%{因为犯了这样的罪业，所受的苦难无法形容，}%

@{以此餘殃故，脅肋為之痛。}@
%{因为这样做的恶报，所以肋骨就疼痛。}%

@{謂難提和羅，輕毀迦葉佛，}@
%{说难提和罗，轻蔑诽谤迦叶佛，}%

@{用見此沙門，言不得佛道。}@
%{因为看到了这个沙门，就说他得不到佛道。}%

 
 

@{佛五百弟子自說本起經}@
%{佛陀的五百个弟子，自己讲述了他们以前所做的事情。}%
