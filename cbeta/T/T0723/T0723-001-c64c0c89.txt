@{大勇菩薩撰}@
%{大勇菩薩撰}%

@{最勝無上尊，知見悉具足，}@
%{最尊贵的佛陀，您拥有圆满的智慧和见解，}%
@{是故稽首禮，及法應真僧。}@
%{我向您和您的弟子，所有证悟真理的僧伽，顶礼膜拜。}%
@{我今撰安住，知見具足說，}@
%{我今天要写一部经书，详细阐述关于业报的知识，}%
@{五趣所緣起，由淨不淨業。}@
%{六道轮回的起因，是由于清净和不净的业力。}%
@{普為諸世間，開示契經義，}@
%{为了利益所有众生，我将解释这部经书的意义，}%
@{隨智力所及，分別業果報。}@
%{根据每个人的智慧程度，分别讲解各种业力的果报。}%

@{佛以法自覺，諸天咸勸請，}@
%{佛陀通过正法觉悟了真理，所有的天人都来请求他宣讲，}%
@{即至波羅㮈，演暢真諦義。}@
%{于是佛陀来到波罗奈，宣讲了宇宙真理的意义。}%
@{謂苦及苦因，苦集究竟滅，}@
%{佛陀说，苦难的本质和苦难的根源，以及彻底消除苦难的方法，}%
@{八正悉具足，盡苦清淨道。}@
%{就是八正道，也就是通往解脱苦难的清净之路。}%

@{無上人中尊，說苦業果報，}@
%{最尊贵的佛陀，讲解了苦难的业力果报，}%
@{從是轉相生，煩惱及諸業。}@
%{从烦恼开始，不断地产生各种各样的业力。}%
@{種種相煩惱，無量諸業行，}@
%{各种各样的烦恼，导致了无数的业力行为，}%
@{次第略分別，隨順大仙說。}@
%{我将逐一简要地解释这些业力，并遵循佛陀的教诲。}%
@{契經所顯示，不違諸法相，}@
%{这部经书所阐述的道理，符合一切事物的真相，}%
@{真實決定義，慧者當受持。}@
%{是真实可靠的真理，有智慧的人应该认真学习和实践。}%

@{非自在所作，果報非無因，}@
%{业力的果报不是由命运决定的，也不是无缘无故的，}%
@{亦非自性起，亦不從時生。}@
%{也不是由人的本性所决定，也不是由时间所决定。}%
@{自在天無因，自性及與時，}@
%{命运、本性和时间，}%
@{以果有勝劣，當知彼非因。}@
%{都是因为果报有优劣之分，所以它们不是业力的原因。}%

@{無知生煩惱，從是起諸業，}@
%{无明导致烦恼，烦恼产生业力，}%
@{因業開眾趣，今當說差別。}@
%{业力决定了众生投生于六道轮回中的不同之处，现在我将解释这些差异。}%

@{造諸不善業，隨業入惡道，}@
%{造作恶业，就会随着业力堕入恶道，}%
@{彼諸罪眾生，閻王慈哀說：}@
%{阎王对这些犯了罪的众生慈悲地说：}%
@{生老病死苦，王法所拘執，}@
%{你们经历着生老病死等痛苦，被王法所约束，}%
@{汝見彼天使，何不生勝覺！}@
%{看到这些痛苦，为什么不觉醒呢？}%

@{惠施清淨戒，能調身口意，}@
%{布施和持戒，可以调伏身、口、意，}%
@{汝為何所求，而不發上願！}@
%{你想要得到什么，为什么不发菩提心呢？}%

@{不幸遇惡友，唯聞非法事，}@
%{不幸遇到恶友，只听到邪法，}%
@{增我貪恚癡，何由起淨業？}@
%{只会增长贪嗔痴，怎么能修行清净的业力呢？}%

@{汝曾不修善，但作諸惡行，}@
%{你从前不修善业，只做恶事，}%
@{不覺罪報至，今來入地獄。}@
%{没有意识到罪报会来临，现在就堕入地狱了。}%

@{爾時諸獄卒，即執罪眾生，}@
%{这时，狱卒们抓住了这些罪犯，}%
@{驅向地獄門，恐怖身毛竪。}@
%{把他们赶向地狱之门，吓得他们毛骨悚然。}%

@{等活若黑繩，眾合二叫呼，}@
%{等活地狱像黑绳一样，众合地狱像叫喊一样，}%
@{無擇大地獄，燒熱及大熱，}@
%{无间地狱像烈火一样，烧热地狱和极热地狱也是如此，}%
@{土海及糞池，鋒利劍葉林，}@
%{还有土海、粪池、锋利的剑叶林，}%
@{刀道劍枝樹，灰河鐵鑊獄，}@
%{刀道、剑枝树、灰河、铁镬地狱，}%
@{造諸惡業者，生此泥黎中。}@
%{造作恶业的人，都会堕入这些地狱之中。}%

@{今當說彼業，苦報差別相。}@
%{现在我要解释这些恶业，以及它们所带来的不同痛苦果报。}%

@{等活死復生，經歷億千劫，}@
%{等活地狱里，死后又会复活，经历无数劫，}%
@{今聞結怨憎，互相傷害故。}@
%{都是因为今生结怨憎恨，互相伤害所致。}%

@{陷人以非道，兩舌離親友，}@
%{陷害他人，用不正当手段离间亲友，}%
@{讒謗及妄語，死墮黑繩獄。}@
%{诽谤和说谎，就会堕入黑绳地狱。}%

@{屠捕及餘殺，死入眾合獄，}@
%{杀生，包括屠宰和其他的杀戮，会堕入众合地狱，}%
@{諸山所磨切，身碎血髓流。}@
%{被山石碾压，身体碎裂，血肉横流。}%

@{為政無慈惻，峻法多因緣，}@
%{当官不仁慈，制定严苛的法律，}%
@{廣設諸方便，種種加楚毒，}@
%{用各种手段折磨和毒害百姓，}%
@{亦入眾合獄，隨業受苦報。}@
%{也会堕入众合地狱，根据业力承受痛苦。}%

@{輪轉崩山芒，鐵石所磨擣，}@
%{在众合地狱里，被崩山芒和铁石碾压，}%
@{隨貪恚癡怖，聽訟違抂直，}@
%{因为贪嗔痴恐惧，在审判案件时不公正，}%
@{亦入眾合獄，鐵輪斷其身。}@
%{也会堕入众合地狱，被铁轮碾断身体。}%

@{自恃強力勢，嶮暴陵孤弱，}@
%{仗着强势欺凌弱小，}%
@{亦入眾合獄，黑象競來踐。}@
%{也会堕入众合地狱，被黑象践踏。}%

@{逼迫多人眾，令彼大呼泣，}@
%{逼迫很多人，让他们哭喊，}%
@{死墮叫呼獄，舉身常洞燃。}@
%{就会堕入叫喊地狱，全身永远燃烧。}%

@{斗秤欺誑人，心惡而口善，}@
%{用计谋欺骗他人，内心邪恶，嘴上却说好话，}%
@{言行無誠實，入大叫呼獄。}@
%{言行不诚实，就会堕入大叫喊地狱。}%

@{呼哉大呼獄，見者身毛竪，}@
%{大叫喊地狱，看到的人都会毛骨悚然，}%
@{於中受劇苦，寄付不還故。}@
%{在那里承受极大的痛苦，永远无法解脱。}%

@{非法言是法，見法言非法，}@
%{把邪法说成正法，把正法说成邪法，}%
@{邪見無因果，侮傲謗賢聖，}@
%{邪见不相信因果，傲慢地诽谤贤者和圣人，}%
@{如是諸人等，死入無擇獄。}@
%{这样的人，就会堕入无间地狱。}%

@{父母賢善人，沙門婆羅門，}@
%{如果父母是贤善之人，或者出家人和婆罗门，}%
@{犯忤令憂惱，死入熱地獄。}@
%{犯了忤逆之罪，让父母忧愁烦恼，就会堕入热地狱。}%

@{父母賢善人，沙門婆羅門，}@
%{如果父母是贤善之人，或者出家人和婆罗门，}%
@{惡心加苦痛，死入大熱獄。}@
%{怀着恶念，加害父母，使他们痛苦，就会堕入大热地狱。}%

@{出家修淨行，虧犯律儀戒，}@
%{出家修行清净的行为，却违反了戒律，}%
@{展轉相形毀，死入熱土獄。}@
%{就会堕入热土地狱，遭受折磨。}%

@{越禁捨正命，邪諂營穢生，}@
%{违反禁令，放弃正道，用不正当的手段谋取利益，}%
@{死入熱糞池，毒虫貫骨髓。}@
%{就会堕入热粪池，被毒虫啃噬骨髓。}%

@{田獵焚林澤，燒害陸眾生，}@
%{打猎、烧毁森林和沼泽，伤害陆地上的众生，}%
@{死入火劍獄，燒剝斷支節。}@
%{就会堕入火剑地狱，被烧灼，肢体断裂。}%

@{誘取諸眾生，詐親害其命，}@
%{诱骗众生，假装亲近，然后杀害他们，}%
@{烏鵄群餓狗，競來食其肉。}@
%{就会被乌鸦、鸱枭、饿狗抢着吃肉。}%

@{毀壞正法橋，導人非法行，}@
%{破坏正法，引导他人走邪路，}%
@{死經利刀道，截足斷肌骨。}@
%{就会堕入利刀地狱，被砍断手脚，割断筋骨。}%

@{長身百足虫，貌像端正女，}@
%{长身百足虫，外表像端正的女子，}%
@{纏身唼髓腦，由彼邪婬故。}@
%{缠绕在身上，啃食骨髓和脑浆，都是因为邪淫所致。}%

@{於他婦女身，摩觸深染著，}@
%{触碰其他女人的身体，沉溺其中，}%
@{驅上劍枝樹，往還貫身體。}@
%{就会被赶到剑枝树上，身体被穿刺。}%

@{種種設方便，殺諸水虫類，}@
%{用各种方法杀害水里的虫类，}%
@{死入沸灰河，舉身悉糜爛。}@
%{就会堕入沸腾的灰河，全身腐烂。}%

@{吞食熱鐵丸，融銅灌其口，}@
%{吞食滚烫的铁丸，用熔化的铜灌入嘴巴，}%
@{鐵釘釘其身，盜竊他財故。}@
%{用铁钉钉穿身体，都是因为偷盗他人财物所致。}%

@{增上十不善，神逝入地獄，}@
%{造作十种恶业，神识就会堕入地狱，}%
@{次罪墮畜生，餘則入餓鬼。}@
%{轻一些的罪过，会堕入畜生道，剩下的就堕入饿鬼道。}%

@{恚憎不善行，心常樂惡法，}@
%{心怀憎恨，经常喜欢恶法，}%
@{見他苦隨喜，死作閻羅卒。}@
%{看到别人痛苦就感到高兴，死后就会成为阎罗的狱卒。}%

@{已說諸業行，重者入地獄，}@
%{我已经解释了各种恶业，罪孽深重的会堕入地狱，}%
@{今當說畜生，餓鬼業果報。}@
%{现在我要讲畜生和饿鬼的业力果报。}%

@{身三口四過，及意三不善，}@
%{身三口四的过失，以及意三不善，}%
@{此業若非增，死墮畜生趣。}@
%{如果这些业力不严重，就会堕入畜生道。}%

@{多欲生鵝鴿，孔雀鴛鴦鳥；}@
%{贪欲多的人，会投生为鹅、鸽子、孔雀、鸳鸯等鸟类；}%
@{愚癡業所生，蛆蟻飛蛾等。}@
%{愚痴的人，会投生为蛆虫、蚂蚁、飞蛾等。}%

@{無智好打縛，報生象馬中，}@
%{没有智慧，喜欢打缚，就会投生为象、马等动物，}%
@{或復作牛羊，麞鹿諸野獸。}@
%{或者投生为牛、羊、鹿等野兽。}%

@{瞋恨作虺蛇，蜂蠍毒虫類，}@

@{憍慢自矜高，惡心密懷害，}@
%{傲慢自大，内心暗藏害人之心，}%
@{報生合羅婆(八脚獸)，}@
%{就会投生为蜘蛛，}%
@{及作虎師子。虛傲踈嶮報，}@
%{或者投生为老虎、狮子。虚伪傲慢的人，会投生为猪、狗、驴、狐狸、狼等。}%

%{吝啬，不乐意布施，}%
@{疾忌多憎惡，輕躁心不住，}@
%{嫉妒心强，经常憎恨别人，心浮气躁，}%
@{死墮猿猴中。強顏少羞恥，}@
%{就会堕入猿猴道。强颜欢笑，不讲礼义廉耻，}%
@{無節多言說，隨業獲果報，}@
%{说话没有节制，爱说闲话，就会得到相应的果报，}%
@{後受烏鳥身。邪貪無厭足，}@

@{急性多瞋怒，不依正憶念，}@

@{後作大力龍。能修大布施，}@
%{就会投生为大力龙。如果能修大布施，}%
@{高心𣣋蔑人，由斯業行生，}@

@{大力金翅鳥。劫盜賢善人，}@
%{劫掠贤善之人，}%

%{吝啬，贪得无厌，就会投生为下等的饿鬼，}%

@{或施還自毀，死墮食吐鬼，}@
%{或者布施了又后悔，就会堕入食吐鬼道，}%

%{伤害别人，抢夺财物，其他的罪过也会堕入饿鬼道，}%
