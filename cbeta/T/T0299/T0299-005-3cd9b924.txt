@{大方廣總持寶光明經卷第五}@
%{大方广总持宝光明经卷第五}%

@{西天中印度摩伽陀國那爛陀寺傳教大師三藏賜紫沙門臣法天奉詔譯}@
%{西天中印度摩伽陀国那烂陀寺传教大师三藏赐紫沙门臣法天奉诏译}%

@{又放光明種種嚴，種種幢幡及傘蓋，}@
%{又放光明种种严，种种幢幡及伞盖，}%
@{作諸妓樂及眾香，嚴持上妙雜花香，}@
%{作诸妓乐及众香，严持上妙杂花香，}%
@{如是種種供養佛，花香燒香及粖香，}@
%{如是种种供养佛，花香烧香及粖香，}%
@{幢幡妙蓋垂寶帳，能出種種妙莊嚴。}@
%{幢幡妙盖垂宝帐，能出种种妙庄严。}%

@{又放光明發淨心，手持眾寶而住立，}@
%{又放光明发净心，手持众宝而住立，}%
@{清淨大仙佛塔處，得此清淨妙光明。}@
%{清净大仙佛塔处，得此清净妙光明。}%

@{又放廣大光明雲，復雨種種妙塗香，}@
%{又放广大光明云，复雨种种妙涂香，}%
@{塗香水洒塔界地，是故得此光明雲。}@
%{涂香水洒塔界地，是故得此光明云。}%

@{又放光明種種嚴，裸者得衣而莊飾，}@
%{又放光明种种严，裸者得衣而庄饰，}%
@{種種瓔珞種種衣，是故出生莊嚴施。}@
%{种种璎珞种种衣，是故出生庄严施。}%

@{又放光明名眾味，施諸上味飢渴者，}@
%{又放光明名众味，施诸上味饥渴者，}%
@{種種上味之飲食，施已獲得大光明。}@
%{种种上味之饮食，施已获得大光明。}%

@{又放光明名勝義，庫藏珍寶施貧窮，}@
%{又放光明名胜义，库藏珍宝施贫穷，}%
@{及施三寶無有盡，是故檀行勝義成。}@
%{及施三宝无有尽，是故檀行胜义成。}%

@{又放光明眼清淨，盲者得視眾色相，}@
%{又放光明眼清净，盲者得视众色相，}%
@{以燈供佛及佛塔，獲得放光清淨眼。}@
%{以灯供佛及佛塔，获得放光清净眼。}%

@{又放光明耳清淨，聾者各得聞眾聲，}@
%{又放光明耳清净，聋者各得闻众声，}%
@{施佛音樂及佛塔，獲得清淨光明耳。}@
%{施佛音乐及佛塔，获得清净光明耳。}%

@{又放光明清淨鼻，鼻根不具聞妙香，}@
%{又放光明清净鼻，鼻根不具闻妙香，}%
@{施妙塗香佛及塔，獲得清淨光明鼻。}@
%{施妙涂香佛及塔，获得清净光明鼻。}%

@{又放光明清淨舌，得佛柔軟慈意語，}@
%{又放光明清净舌，得佛柔软慈意语，}%
@{遠離麁獷雜穢言，獲得光明舌相好。}@
%{远离粗犷杂秽言，获得光明舌相好。}%

@{又放光明身清淨，身根不具復圓滿，}@
%{又放光明身清净，身根不具复圆满，}%
@{捨身量等佛及塔，是故得此清淨身。}@
%{舍身量等佛及塔，是故得此清净身。}%

@{又放光明意清淨，一切妙意彼皆得，}@
%{又放光明意清净，一切妙意彼皆得，}%
@{因心意作三摩地，得此清淨意光明。}@
%{因心意作三摩地，得此清净意光明。}%

@{又放光明色清淨，思見仁王眾色相，}@
%{又放光明色清净，思见仁王众色相，}%
@{種種妙色靡不周，得獲莊嚴光明塔。}@
%{种种妙色靡不周，得获庄严光明塔。}%

@{又放光明聲清淨，非聲空聲悉盡知，}@
%{又放光明声清净，非声空声悉尽知，}%
@{生信由如於谷嚮，放此清淨聲光明。}@
%{生信由如于谷响，放此清净声光明。}%

@{又放清淨香光明，一切臭氣為妙香，}@
%{又放清净香光明，一切臭气为妙香，}%
@{妙香水灑佛塔廟，放光住此菩提樹。}@
%{妙香水洒佛塔庙，放光住此菩提树。}%

@{又放清淨味光明，有毒無毒變上味，}@
%{又放清净味光明，有毒无毒变上味，}%
@{供佛聲聞及父母，施諸上味得光明。}@
%{供佛声闻及父母，施诸上味得光明。}%

@{又放光明觸清淨，澁滑軟觸悉安樂，}@
%{又放光明触清净，涩滑软触悉安乐，}%
@{劍戟箭槍如雲雨，變為柔軟妙花鬘，}@
%{剑戟箭枪如云雨，变为柔软妙花鬘，}%
@{無數軟衣觸獲安，渡生往詣於佛所，}@
%{无数软衣触获安，渡生往诣于佛所，}%
@{妙花塗香清淨衣，得施鬘蓋放光明。}@
%{妙花涂香清净衣，得施鬘盖放光明。}%

@{又放清淨法光明，一切毛端難思法，}@
%{又放清净法光明，一切毛端难思法，}%
@{為諸世間而出現，得佛演說一切法。}@
%{为诸世间而出现，得佛演说一切法。}%
@{信法自性本不生，法身報身亦復爾，}@
%{信法自性本不生，法身报身亦复尔，}%
@{法常寂住等虛空，是故得獲清淨法。}@
%{法常寂住等虚空，是故得获清净法。}%

@{光明現前應是作，於大仙面一毛端，}@
%{光明现前应是作，于大仙面一毛端，}%
@{出生殑伽沙等光，一切各各隨業因。}@
%{出生恒河沙等光，一切各各随业因。}%

@{亦如現前一毛塵，出現殑伽微塵等，}@
%{亦如现前一毛尘，出现恒河微尘等，}%
@{如是一切毛皆盡，此三摩地大仙現。}@
%{如是一切毛皆尽，此三摩地大仙现。}%

@{若放光明如本行，是光所作過去同，}@
%{若放光明如本行，是光所作过去同，}%
@{彼若不現此光明，云何神變等大仙。}@
%{彼若不现此光明，云何神变等大仙。}%

@{彼得福德既同因，隨喜勸請亦如是，}@
%{彼得福德既同因，随喜劝请亦如是，}%
@{若有得見悉獲安，是光自他俱解了。}@
%{若有得见悉获安，是光自他俱解了。}%

@{若作福業善安和，相續不斷供養佛，}@
%{若作福业善安和，相续不断供养佛，}%
@{如來功德應忻求，覩此光明如是作。}@
%{如来功德应忻求，睹此光明如是作。}%

@{由如盲人不覩日，不分晝夜及世間，}@
%{由如盲人不睹日，不分昼夜及世间，}%
@{云何眼等能了色，各各法義而得解。}@
%{云何眼等能了色，各各法义而得解。}%

@{調御放光亦如是，分明示彼不自見，}@
%{调御放光亦如是，分明示彼不自见，}%
@{未離顛倒妄想言，畢竟不得廓徹意。}@
%{未离颠倒妄想言，毕竟不得廓彻意。}%

@{乘寶宮殿眾嚴持，種種資具妙香華，}@
%{乘宝宫殿众严持，种种资具妙香华，}%
@{大眾無有能知者，彼之疾病甚難袪，}@
%{大众无有能知者，彼之疾病甚难袪，}%

@{調御光明亦如是，此光自障不能知，}@
%{调御光明亦如是，此光自障不能知，}%
@{虛誑妄語未能除，畢竟不能心廣大，}@
%{虚诳妄语未能除，毕竟不能心广大，}%
@{覩此光明能了別，常得樂說而安樂。}@
%{睹此光明能了别，常得乐说而安乐。}%

@{彼身無疑妄想除，離我即大功德幢，}@
%{彼身无疑妄想除，离我即大功德幢，}%
@{變化主伴而莊嚴，依此無上三摩地。}@
%{变化主伴而庄严，依此无上三摩地。}%

@{無邊一切諸十方，顯示佛子并主伴，}@
%{无边一切诸十方，显示佛子并主伴，}%
@{種種三千微塵剎，光明嚴飾蓮花座。}@
%{种种三千微尘刹，光明严饰莲花座。}%

@{一切身同跏趺坐，變化示此三摩地，}@
%{一切身同跏趺坐，变化示此三摩地，}%
@{自他十剎微塵等，蓮花坐彼諸眷屬。}@
%{自他十刹微尘等，莲花坐彼诸眷属。}%

@{各各佛子眾圍遶，安住遍入三摩地，}@
%{各各佛子众围遶，安住遍入三摩地，}%
@{彼以大仙化導力，眾生從佛功德出。}@
%{彼以大仙化导力，众生从佛功德出。}%

@{互相圍遶大蓮華，同時合掌而恭敬，}@
%{互相围遶大莲花，同时合掌而恭敬，}%
@{此法是大調御師，若入寂靜三摩地。}@
%{此法是大调御师，若入寂静三摩地。}%

@{得諸弟子眾圍遶，由如眾星中朗月，}@
%{得诸弟子众围遶，由如众星中朗月，}%
@{如彼入於一方中，顯示佛子并眷屬。}@
%{如彼入于一方中，显示佛子并眷属。}%

@{一切方中此皆盡，變化示此三摩地，}@
%{一切方中此皆尽，变化示此三摩地，}%
@{彼方覆閉悉蠲除，最上三摩地行入。}@
%{彼方覆闭悉蠲除，最上三摩地行入。}%

@{或有從於三昧起，自身顯示於十方，}@
%{或有从于三昧起，自身显示于十方，}%
@{或有東方三昧中，西方不起能搖動；}@
%{或有东方三昧中，西方不起能摇动；}%
@{或有西方三昧中，東方現起大人相，}@
%{或有西方三昧中，东方现起大人相，}%
@{悉能入此十方中，異方復現諸三昧。}@
%{悉能入此十方中，异方复现诸三昧。}%

@{廣能現此功德智，大仙神變三摩地，}@
%{广能现此功德智，大仙神变三摩地，}%
@{若盡異方一切剎，無數如來皆往詣。}@
%{若尽异方一切刹，无数如来皆往诣。}%

@{一一足下而致敬，現住安樂三摩地，}@
%{一一足下而致敬，现住安乐三摩地，}%
@{或以三昧視眾生，西方盡剎而湛然。}@
%{或以三昧视众生，西方尽刹而湛然。}%

@{一切如來悉現前，現作無數諸供養，}@
%{一切如来悉现前，现作无数诸供养，}%
@{如是西方盡見已，復往阿僧祇佛剎，}@
%{如是西方尽见已，复往阿僧祇佛刹，}%
@{於彼足下而致敬，得住安樂三摩地。}@
%{于彼足下而致敬，得住安乐三摩地。}%

@{彼以等持而正見，盡於東方諸剎土，}@
%{彼以等持而正见，尽于东方诸刹土，}%
@{一切如來悉現前，又覩無數諸供養。}@
%{一切如来悉现前，又睹无数诸供养。}%

@{入此十方佛剎已，一一方盡無動搖，}@
%{入此十方佛刹已，一一方尽无动摇，}%
@{於中現起諸三昧，種種供養世間燈。}@
%{于中现起诸三昧，种种供养世间灯。}%

@{彼於眼塵三昧中，能現大色諸境界，}@
%{彼于眼尘三昧中，能现大色诸境界，}%
@{見此不可思議色，天上人間悉愕然。}@
%{见此不可思议色，天上人间悉愕然。}%

@{既覩色塵三昧已，作是思惟眼塵境，}@
%{既睹色尘三昧已，作是思惟眼尘境，}%
@{眼之自性非生滅，是故開闡無相空。}@
%{眼之自性非生灭，是故开阐无相空。}%

@{或有耳塵三昧中，能現大聲諸境界，}@
%{或有耳尘三昧中，能现大声诸境界，}%
@{入於一切語言音，天上人間悉愕然。}@
%{入于一切语言音，天上人间悉愕然。}%

@{既覩聲塵三昧已，審諦思惟耳塵境，}@
%{既睹声尘三昧已，审谛思惟耳尘境，}%
@{耳之自性非生滅，是故開闡無相空。}@
%{耳之自性非生灭，是故开阐无相空。}%

@{或有鼻塵三昧中，能現大香諸境界，}@
%{或有鼻尘三昧中，能现大香诸境界，}%
@{如是妙香悉遍入，天上人間悉愕然。}@
%{如是妙香悉遍入，天上人间悉愕然。}%

@{既覩香塵三昧已，復諦思惟鼻塵境，}@
%{既睹香尘三昧已，复谛思惟鼻尘境，}%
@{鼻之自性非生滅，是故開闡無相空。}@
%{鼻之自性非生灭，是故开阐无相空。}%

@{或有舌塵三昧中，能現大舌塵境界，}@
%{或有舌尘三昧中，能现大舌尘境界，}%
@{如是上味普周遍，天上人間悉愕然。}@
%{如是上味普周遍，天上人间悉愕然。}%

@{既得上味三昧已，審諦思惟舌塵境，}@
%{既得上味三昧已，审谛思惟舌尘境，}%
@{舌之自性非生滅，是故開闡無相空。}@
%{舌之自性非生灭，是故开阐无相空。}%

@{或有身塵三昧中，能現大觸塵境界，}@
%{或有身尘三昧中，能现大触尘境界，}%
@{世間觸塵普遍入，天上人間悉愕然。}@
%{世间触尘普遍入，天上人间悉愕然。}%

@{既覩觸塵三昧已，身塵遍起諦思惟，}@
%{既睹触尘三昧已，身尘遍起谛思惟，}%
@{身之自性非生滅，是故開闡無相空。}@
%{身之自性非生灭，是故开阐无相空。}%

@{或有心塵三昧中，能現大法塵境界，}@
%{或有心尘三昧中，能现大法尘境界，}%
@{既覩法塵三昧已，復諦思惟心塵境，}@
%{既睹法尘三昧已，复谛思惟心尘境，}%
@{心之自性非生滅，是故開闡無相空。}@
%{心之自性非生灭，是故开阐无相空。}%

@{彼以三昧嬰孩身，出現盛年壯色住，}@
%{彼以三昧婴孩身，出现盛年壮色住，}%
@{彼現盛年三昧已，復住衰老朽痼身，}@
%{彼现盛年三昧已，复住衰老朽痼身，}%
@{住此衰老身三昧，復發心作優婆塞。}@
%{住此衰老身三昧，复发心作优婆塞。}%

@{住此優婆塞三昧，改質現作比丘身，}@
%{住此优婆塞三昧，改质现作比丘身，}%
@{住此比丘身三昧，即得多聞比丘身。}@
%{住此比丘身三昧，即得多闻比丘身。}%

@{住此多聞身三昧，即得有學無學身，}@
%{住此多闻身三昧，即得有学无学身，}%
@{既住如是身三昧，即得住於如來身。}@
%{既住如是身三昧，即得住于如来身。}%

@{住是如來身三昧，得佛最上妙色身，}@
%{住是如来身三昧，得佛最上妙色身，}%
@{住此最上身三昧，化身現作天人形。}@
%{住此最上身三昧，化身现作天人形。}%

@{從此天形身三昧，出生眾多大龍身，}@
%{从此天形身三昧，出生众多大龙身，}%
@{從此大龍身三昧，出生諸大藥叉身。}@
%{从此大龙身三昧，出生诸大药叉身。}%

@{從此藥叉身三昧，出生一切步多身，}@
%{从此药叉身三昧，出生一切步多身，}%
@{從此步多身三昧，現前出生一毛端。}@
%{从此步多身三昧，现前出生一毛端。}%

@{從此一毛端三昧，出生一切眾毛端，}@
%{从此一毛端三昧，出生一切众毛端，}%
@{從此一切毛端中，出生一髮毛端塵。}@
%{从此一切毛端中，出生一发毛端尘。}%

@{從此一髮塵三昧，出現一切髮塵境，}@
%{从此一发尘三昧，出现一切发尘境，}%
@{一一髮塵三昧中，復能出現微塵數。}@
%{一一发尘三昧中，复能出现微尘数。}%

@{若見一塵三昧已，出生一切微塵盡，}@
%{若见一尘三昧已，出生一切微尘尽，}%
@{若見一一塵三昧，出生大海金剛際。}@
%{若见一一尘三昧，出生大海金刚际。}%

@{從此金剛際三昧，出生摩尼樹花果，}@
%{从此金刚际三昧，出生摩尼树花果，}%
@{從此摩尼樹三昧，出現如來大光明。}@
%{从此摩尼树三昧，出现如来大光明。}%

@{從此如來光三昧，出現一切江海水，}@
%{从此如来光三昧，出现一切江海水，}%
@{從此江海三昧中，復現大火塵境界。}@
%{从此江海三昧中，复现大火尘境界。}%

@{從此火塵三昧中，出生風塵之思念，}@
%{从此火尘三昧中，出生风尘之思念，}%
@{從此風塵之三昧，出生大地塵境界。}@
%{从此风尘之三昧，出生大地尘境界。}%

@{從此地塵三昧中，出生一切天宮殿，}@
%{从此地尘三昧中，出生一切天宫殿，}%
@{於此宮殿三昧中，思念等彼虛空界。}@
%{于此宫殿三昧中，思念等彼虚空界。}%

@{三昧解脫不思議，悉得無邊諸功德，}@
%{三昧解脱不可思议，悉得无边诸功德，}%
@{復能照耀無邊劫，一切如來盡難量。}@
%{复能照耀无边劫，一切如来尽难量。}%

@{一切如來同說此，世間業果難思議，}@
%{一切如来同说此，世间业果难思议，}%
@{龍化雲水佛入定，定中變化不思議。}@
%{龙化云水佛入定，定中变化不可思议。}%

@{見彼出生說少分，於中彼眾悉驚怪，}@
%{见彼出生说少分，于中彼众悉惊怪，}%
@{法師方便智令知，所說敏速能解了，}@
%{法师方便智令知，所说敏速能解了，}%
@{即得住於八解脫。聲聞得一即為多，}@
%{即得住于八解脱。声闻得一即为多，}%
@{或是得多由如一，觀彼虛空光熾焰。}@
%{或是得多由如一，观彼虚空光炽焰。}%

@{炎赫熾盛俱洞然。周匝俱為熾火城，}@
%{炎赫炽盛俱洞然。周匝俱为炽火城，}%
@{火城下徹於水輪，於輪坐臥而安住，}@
%{火城下彻于水轮，于轮坐卧而安住，}%
@{剎那身變不思議。彼眾無有大悲心，}@
%{刹那身变不可思议。彼众无有大悲心，}%
@{菩提行遠棄世間，身雖變化不思議，}@
%{菩提行远弃世间，身虽变化不可思议，}%
@{不能利他為世間。譬如日月遊虛空，}@
%{不能利他为世间。譬如日月游虚空，}%
@{悉能照見十方界，陂湖池沼及淵泉，}@
%{悉能照见十方界，陂湖池沼及渊泉，}%
@{方圓大小寶海河，悉現不思議色相。}@
%{方圆大小宝海河，悉现不可思议色相。}%

@{如日照見於十方，如智解脫諸三昧，}@
%{如日照见于十方，如智解脱诸三昧，}%
@{若見如來佛亦然。澄湛池邊四兵眾，}@
%{若见如来佛亦然。澄湛池边四兵众，}%
@{各各於中而現形，利劍弓刀箭甲冑，}@
%{各各于中而现形，利剑弓刀箭甲胄，}%
@{兜鍪覆膊絛鐵札，亦如眾色而莊嚴。}@
%{兜鍪覆膊絛铁札，亦如众色而庄严。}%

@{見彼如來光明網，不分憎愛離疑惑，}@
%{见彼如来光明网，不分憎爱离疑惑，}%
@{解脫功德三摩地。天中海內說彼名，}@
%{解脱功德三摩地。天中海内说彼名，}%
@{乃至眾生出生海，聞彼所說盡能知，}@
%{乃至众生出生海，闻彼所说尽能知，}%
@{自身語言皆歡喜。愛樂貪恚雜語言，}@
%{自身语言皆欢喜。爱乐贪恚杂语言，}%
@{一切隨類能了知，得妙總持法之力，}@
%{一切随类能了知，得妙总持法之力，}%
@{人間天上過有無。名為覩嚕婆惹娑，}@
%{人间天上过有無。名为睹噜婆惹娑，}%
@{婆羅門女稱讚彼，志願無有愛恚心。}@
%{婆罗门女称赞彼，志愿无有爱恚心。}%

@{為無嫉妬得超身，雖有辯才愛恚存，}@
%{为无嫉妬得超身，虽有辩才爱恚存，}%
@{不能成就大檀行。名稱遠聞於世間，}@
%{不能成就大檀行。名称远闻于世间，}%
@{無人不喜智功德，譬如明智作幻術，}@
%{无人不喜智功德，譬如明智作幻术，}%
@{能現種種無邊色，或於晝夜月一念。}@
%{能现种种无边色，或于昼夜月一念。}%

@{百年由如於風燭，云何幻化有愛恚，}@
%{百年由如于风烛，云何幻化有爱恚，}%
@{世間變化如幻夢，定知遠離於解脫。}@
%{世间变化如幻梦，定知远离于解脱。}%

@{如智善修人皆喜，天與修羅共鬪戰，}@
%{如智善修人皆喜，天与修罗共鬪战，}%
@{天得修羅自退散，以蓋覆上乘車輦，}@
%{天得修罗自退散，以盖覆上乘车辇，}%
@{兵甲自退而散走，修羅於彼生愛恚，}@
%{兵甲自退而散走，修罗于彼生爱恚，}%
@{自說身變不思議，富貴勇猛無能敵。}@
%{自说身变不可思议，富贵勇猛无能敌。}%

@{現身變化能如是，能持大龍及金剛，}@
%{现身变化能如是，能持大龙及金刚，}%
@{帝釋眷屬乘大象，此象頭現三十三，}@
%{帝释眷属乘大象，此象头现三十三，}%
@{一一復現於六牙，一一牙上現七池，}@
%{一一复现于六牙，一一牙上现七池，}%
@{池中現水皆盈滿，一一池現七蓮花，}@
%{池中现水皆盈满，一一池现七莲花，}%
@{蓮花種種而莊嚴，一一華中復現七，}@
%{莲花种种而庄严，一一华中复现七，}%
@{天女一切悉能知，能解歌舞眾妓樂。}@
%{天女一切悉能知，能解歌舞众妓乐。}%

@{帝釋天眾同娛樂，知彼自身復龍形，}@
%{帝释天众同娱乐，知彼自身复龙形，}%
@{得同一切而變化，種種造作等莊嚴。}@
%{得同一切而变化，种种造作等庄严。}%

@{或時現龍為最上，彼現有愛有恚癡，}@
%{或时现龙为最上，彼现有爱有恚癡，}%
@{隨自福業現神變，智力方便相應行。}@
%{随自福业现神变，智力方便相应行。}%

@{非三摩地不能作，由如羅睺身變現，}@
%{非三摩地不能作，由如罗睺身变现，}%
@{變為金剛縛足下，現身海中水至臍，}@
%{变为金刚缚足下，现身海中水至脐，}%
@{頭與妙高而同等，彼有愛恚貪瞋癡。}@
%{头与妙高而同等，彼有爱恚贪瞋癡。}%

@{羅睺神變力如是，魔王破壞世間燈，}@
%{罗睺神变力如是，魔王破坏世间灯，}%
@{若現神變無有邊，帝釋化現不思議。}@
%{若现神变无有边，帝释化现不可思议。}%

@{天與修羅共戰時，制伏修羅不能變，}@
%{天与修罗共战时，制伏修罗不能变，}%
@{彼現自身於帝釋，能知修羅勇猛力。}@
%{彼现自身于帝释，能知修罗勇猛力。}%

@{同時自往帝釋前，各各執持於金剛，}@
%{同时自往帝释前，各各执持于金刚，}%
@{修羅執持而戰慄，變現千眼能怖畏。}@
%{修罗执持而战慄，变现千眼能怖畏。}%

@{光明熾盛執金剛，身披甲冑有大力，}@
%{光明炽盛执金刚，身披甲胄有大力，}%
@{修羅既覩而退走，彼以薄福力不任。}@
%{修罗既睹而退走，彼以薄福力不任。}%

@{帝釋為天現神變，一切世間盡能救，}@
%{帝释为天现神变，一切世间尽能救，}%
@{說此神變福無盡，空中天鼓勸諸天。}@
%{说此神变福无尽，空中天鼓劝诸天。}%

@{說於彼眾業果報，知天耽著於欲樂，}@
%{说于彼众业果报，知天耽著于欲乐，}%
@{天鼓出聲而救度，觀身無常不久停。}@
%{天鼓出声而救度，观身无常不久停。}%

@{此等自性不能了，如幻如雲翳星月，}@
%{此等自性不能了，如幻如云翳星月，}%
@{萬物自性如夢覺。一切煩惱怨家因，}@
%{万物自性如梦觉。一切烦恼怨家因，}%
@{除得甘露離無常，其中若有樂耽著，}@
%{除得甘露离无常，其中若有乐耽著，}%
@{無常迅速如摩竭。一切眷屬悉遭苦，}@
%{无常迅速如摩竭。一切眷属悉遭苦，}%
@{聖者增長於一切，樂著欲樂如盲人，}@
%{圣者增长于一切，乐著欲乐如盲人，}%
@{若有聞法免無常。鼓為帝釋常誡勸，}@
%{若有闻法免无常。鼓为帝释常诫劝，}%
@{及諸天眾說妙法，恒時演暢妙語言，}@
%{及诸天众说妙法，恒时演畅妙语言，}%
@{無邊煩惱能廣說。彼能現此無色相，}@
%{无边烦恼能广说。彼能现此无色相，}%
@{天鼓出於大音聲，隨諸天意現眾色，}@
%{天鼓出于大音声，随诸天意现众色，}%
@{廣為無作諸眾生。天與修羅相持時，}@
%{广为无作诸众生。天与修罗相持时，}%
@{以天福力勝於彼，天鼓空中勸天人，}@
%{以天福力胜于彼，天鼓空中劝天人，}%
@{出眾音聲悉同彼。天鼓勸令生忻喜，}@
%{出众音声悉同彼。天鼓劝令生忻喜，}%
@{得生遠離兵怖畏，修羅王眾自退散，}@
%{得生远离兵怖畏，修罗王众自退散，}%
@{剎那恐懼盡消除。鼓施甘露經劫數，}@
%{刹那恐惧尽消除。鼓施甘露经劫数，}%
@{救度一切諸世間，彼離煩惱魔王眾，}@
%{救度一切诸世间，彼离烦恼魔王众，}%
@{不受煩惱說安樂。帝釋天女九十二，}@
%{不受烦恼说安乐。帝释天女九十二，}%
@{化身令他悉歡喜，一一幻身與彼同，}@
%{化身令他悉欢喜，一一幻身与彼同，}%
@{一切天女各同示，諸女同時貪慾樂，}@
%{一切天女各同示，诸女同时贪欲乐，}%
@{若住自性妙法中，而為開示演妙法。}@
%{若住自性妙法中，而为开示演妙法。}%

@{帝釋神變一剎那，帝釋有愛有恚癡，}@
%{帝释神变一刹那，帝释有爱有恚癡，}%
@{唯自娛樂諸眷屬，無人於世恒方便。}@
%{唯自娱乐诸眷属，无人于世恒方便。}%

@{世間變化離愛欲，魔王鬪諍住世間，}@
%{世间变化离爱欲，魔王鬪诤住世间，}%
@{攝伏一切眾生類，}@@{爾時三千大千世界，一切諸天及諸人民海會聖眾，同作是言：此是如來第二會轉此法輪，讚言善哉！善說世尊，希有，希有！我等過去未曾聞，說如是正法。若有眾生，聞此正法，甚為希有！即得不生貧窮下賤之家，心得平等，由如供養恭敬眾多如來。是諸眾生展轉相教，如佛塔廟。彼諸眾生，天上人間，魔王眷屬彼不能害。若諸人天，在家出家，是諸國王及諸大臣長者宰官，亦不能害。}@
%{當時，三千大千世界裡，所有天界和人間的眾生，都聚集在一起，齊聲說道：“這是佛陀第二次轉動法輪，真是太好了！佛陀講得真好，太稀有，太稀有了！我們以前從未聽過這樣的正法，如果有人能聽到這個正法，真是太稀有了！他們將不再出生於貧窮低賤的家庭，內心會變得平等，就像供養和恭敬許多佛陀一樣。這些眾生互相教導，就像佛塔和寺廟一樣。這些眾生，無論是在天界還是人間，魔王的眷屬都不能傷害他們。如果這些天人，無論是出家還是在家，是國王、大臣、長者、官員，也無法傷害他們。”}%

@{是時，尊者須菩提白佛言：世尊！希有如來云何得此，如是寶光明經正法行於世間？此眾經中寶，皆是如來威神力故，出現世間。復次是諸眾生得聞是法，亦是如來威神力故。}@
%{當時，須菩提尊者對佛陀說：“世尊！稀有的如來，是如何讓這部寶光明經的正法在世間流傳的？這部經是所有經典中的寶藏，都是因為如來的威神力，才顯現於世間。而且，這些眾生能夠聽到這個法，也是因為如來的威神力。”}%

@{爾時世尊復開金口，告尊者阿難言：汝當受持，阿難！如是正法，勤心精進，受持讀誦。}@
%{當時，佛陀再次開口，對阿難尊者說：“阿難，你應該好好持守這部正法，用心精進地持誦。”}%

@{是時，尊者阿難遶佛三匝，而作是言：世尊！我當受持如是正法，永不忘失。}@
%{當時，阿難尊者繞佛三圈，說道：“世尊！我一定會持守這部正法，永不忘記。”}%

@{佛說此經已，一切聖眾菩薩，普賢菩薩、妙吉祥童子，尊者舍利弗等，諸大聲聞，天、人、阿修羅、乾達嚩等，聞佛所說，皆大歡喜，頂禮而去。}@
%{佛陀說完這部經後，所有的聖眾菩薩，普賢菩薩、妙吉祥童子、舍利弗尊者等，所有大聲聞，天人、阿修羅、乾闥婆等，都聽到佛陀的教誨，非常歡喜，頂禮佛陀後就離開了。}%
