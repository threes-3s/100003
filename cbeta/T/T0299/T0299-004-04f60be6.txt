@{大方廣總持寶光明經卷第四}@
%{大方广总持宝光明经卷第四}%

@{西天中印度摩伽陀國那爛陀寺傳教大師三藏賜紫沙門臣法天奉詔譯}@
%{西天中印度摩揭陀国那烂陀寺传教大师三藏赐紫沙门臣法天奉诏译}%

@{若得發心淨覺意，彼說最上最殊勝，}@
%{如果能发心净觉意，那就是说最上最殊胜，}%

@{若得最上殊勝已，常行波羅蜜多行。}@
%{如果得到最上殊胜了，就应该常行波罗蜜多行。}%

@{若行波羅蜜多行，則能隨順此大乘，}@
%{如果行波罗蜜多行，就能随顺此大乘，}%

@{若能隨順此大乘，則能隨順供養佛。}@
%{如果能随顺此大乘，就能随顺供养佛。}%

@{若得隨順供養佛，則得智慧志堅固，}@
%{如果得到随顺供养佛，就能得到智慧志坚固，}%

@{若能智慧志堅固，常見彼佛不思議。}@
%{如果能智慧志坚固，就能常见彼佛不可思议。}%

@{若常見佛不思議，見佛無去無有住，}@
%{如果常见佛不可思议，就能见佛无去无有住，}%

@{若得見佛無去住，此法住世恒不滅。}@
%{如果得到见佛无去住，此法住世恒不灭。}%

@{若見此法恒不滅，遠離積集諸煩惱，}@
%{如果见此法恒不灭，就能远离积集诸烦恼，}%

@{若得遠離煩惱因，則能說法無邊際。}@
%{如果得到远离烦恼因，就能说法无边际。}%

@{若得說法無邊際，彼為興慈住世間，}@
%{如果得到说法无边际，那就是为兴慈住世间，}%

@{若為慈心住世間，則是大悲堅牢本。}@
%{如果为慈心住世间，那就是大悲坚牢本。}%

@{若得大悲堅牢本，即是為他法喜樂，}@
%{如果得到大悲坚牢本，那就是为他法喜乐，}%

@{若得喜樂法根本，彼捨造作種種罪。}@
%{如果得到喜乐法根本，就能舍造作种种罪。}%

@{若捨造作眾罪已，無我無人離眾非，}@
%{如果舍造作众罪已，无我无人离众非，}%

@{若得無我離眾非，彼常自利及利他。}@
%{如果得到无我离众非，就能常自利及利他。}%

@{若常自利及利他，永免輪迴諸苦惱，}@
%{如果常自利及利他，就能永免轮回诸苦恼，}%

@{若離輪迴諸苦惱，彼得大力最為勝。}@
%{如果离轮回诸苦恼，就能得到大力最为胜。}%

@{若得大力最殊勝，則得出生清淨智，}@
%{如果得到大力最殊胜，就能得到出生清净智，}%

@{若生清淨智慧已，彼入世間修行網。}@
%{如果生清净智慧已，就能入世间修行网。}%

@{若能入此修行網，成熟世間甚希有，}@
%{如果能入此修行网，成熟世间甚希有，}%

@{若得成熟於世間，積集世間妙智慧。}@
%{如果得到成熟于世间，就能积集世间妙智慧。}%

@{若得世間妙智慧，則能隨順於四攝，}@
%{如果得到世间妙智慧，就能随顺于四摄，}%

@{若能隨順於四攝，則於世間廣施設。}@
%{如果能随顺于四摄，就能于世间广施設。}%

@{若於世間廣施設，則住方便智慧力，}@
%{如果于世间广施設，就能住方便智慧力，}%

@{若得方便智慧力，則住勇猛無上道。}@
%{如果得到方便智慧力，就能住勇猛无上道。}%

@{若住勇猛無上道，永不見彼魔軍眾，}@
%{如果住勇猛无上道，就能永不见彼魔军众，}%

@{若得不見魔軍眾，則能遠離四魔道。}@
%{如果得到不见魔军众，就能远离四魔道。}%

@{若得遠離四魔道，則能到於不退地，}@
%{如果得到远离四魔道，就能到于不退地，}%

@{若得到於不退地，即得名為無生忍。}@
%{如果得到于不退地，就能得名为无生忍。}%

@{若得如是無生忍，得佛授記號世燈，}@
%{如果得到如是无生忍，就能得佛授记号世灯，}%

@{若得授記號世燈，則住一切如來前。}@
%{如果得到授记号世灯，就能住一切如来前。}%

@{若得住於如來前，得佛祕密變化智，}@
%{如果得到住于如来前，就能得佛秘密变化智，}%

@{若得如來變化智，一切如來皆授記。}@
%{如果得到如来变化智，一切如来皆授记。}%

@{若得諸佛授記已，一切功德眾莊嚴，}@
%{如果得到诸佛授记已，一切功德众庄严，}%

@{若得功德眾莊嚴，得妙福德清淨身。}@
%{如果得到功德众庄严，就能得妙福德清净身。}%

@{若得如是清淨身，由若金山光照耀，}@
%{如果得到如是清净身，就能由若金山光照耀，}%

@{若得如是金山光，即得具相三十二。}@
%{如果得到如是金山光，就能得到具相三十二。}%

@{若得具相三十二，即得相好莊嚴身，}@
%{如果得到具相三十二，就能得到相好庄严身，}%

@{若得相好莊嚴已，身放光明普皆照。}@
%{如果得到相好庄严已，就能身放光明普皆照。}%

@{若得光明普照已，不思議光眾嚴飾，}@
%{如果得到光明普照已，就能不可思议光众严饰，}%

@{若得如是眾嚴飾，光明足步蓮花生。}@
%{如果得到如是众严饰，就能光明足步莲花生。}%

@{若得光明蓮花行，得佛境界蓮花座，}@
%{如果得到光明莲花行，就能得佛境界莲花座，}%

@{若得如來花座已，則能照見十方界。}@
%{如果得到如来花座已，就能照见十方界。}%

@{若得照見十方界，教化眾生不空行，}@
%{如果得到照见十方界，就能教化众生不空行，}%

@{若得教化不空行，即得智辯無邊際。}@
%{如果得到教化不空行，就能得到智辩无边际。}%

@{若得智辯無邊際，說此不可思議法，}@
%{如果得到智辩无边际，就能说此不可思议法，}%

@{若得說法不思議，令彼無量眾忻慶。}@
%{如果得到说法不可思议，就能令彼无量众忻庆。}%

@{若得無量眾忻慶，積行不可思議剎，}@
%{如果得到无量众忻庆，就能积行不可思议刹，}%

@{若得如是廣大行，各各智力盡能知。}@
%{如果得到如是广大行，各各智力尽能知。}%

@{若得如是智慧力，世間隨類化群生，}@
%{如果得到如是智慧力，世间随类化群生，}%

@{若得隨化群生已，宿命住智彼皆得。}@
%{如果得到随化群生已，宿命住智彼皆得。}%

@{若得如是宿命智，三業清淨恒無間，}@
%{如果得到如是宿命智，三业清净恒无间，}%

@{若得三業恒無間，以自願力隨念至。}@
%{如果得到三业恒无间，就能以自愿力随念至。}%

@{若得願力隨念至，隨諸眾生現眾類，}@
%{如果得到愿力随念至，就能随诸众生现众类，}%

@{若得隨現眾生類，得不思議妙音聲。}@
%{如果得到随现众生类，就能得到不可思议妙音声。}%

@{若得如是妙音聲，能出種種眾語言，}@
%{如果得到如是妙音声，就能出种种众语言，}%

@{所出如是妙言辭，則能周遍諸世界。}@
%{所出如是妙言辞，就能周遍诸世界。}%

@{如是周遍諸世界，一剎那中盡能知，}@
%{如是周遍诸世界，一刹那中尽能知，}%

@{若得如是了眾心，不生不滅無退轉。}@
%{如果得到如是了众心，就能不生不灭无退转。}%

@{若離生滅無退轉，一切障惱不復生，}@
%{如果离生灭无退转，一切障恼不复生，}%

@{若得障惱不復生，法身功德智慧圓。}@
%{如果得到障恼不复生，法身功德智慧圆。}%

@{若得法身智慧圓，法行光明照世間，}@
%{如果得到法身智慧圆，法行光明照世间，}%

@{若得法光照世間，則得十地十種身。}@
%{如果得到法光照世间，就能得到十地十种身。}%

@{若得十地十種身，則得般若解脫道，}@
%{如果得到十地十种身，就能得到般若解脱道，}%

@{若得般若解脫道，灌頂莊嚴為最上。}@
%{如果得到般若解脱道，灌顶庄严为最上。}%

@{若得如是灌頂已，三摩地道悉皆成，}@
%{如果得到如是灌顶已，三摩地道悉皆成，}%

@{如是三摩地得已，各見十方一切佛。}@
%{如是三摩地得已，各见十方一切佛。}%

@{若見如是諸佛已，如是諸佛念灌頂，}@
%{如果见如是诸佛已，如是诸佛念灌顶，}%

@{作是思惟是事已，十方諸佛同灌頂。}@
%{作是思惟是事已，十方诸佛同灌顶。}%

@{諸佛同時灌頂已，各伸右手摩其頂，}@
%{诸佛同时灌顶已，各伸右手摩其顶，}%

@{既得諸佛摩頂已，則能變現等虛空。}@
%{既得诸佛摩顶已，就能变现等虚空。}%

@{如是變現等虛空，隨所住處能堅固，}@
%{如是变现等虚空，随所住处能坚固，}%

@{若得隨所能堅固，天上人間莫能測。}@
%{如果得到随所能坚固，天上人间莫能测。}%

@{既以人天無能測，如是所作離稱量，}@
%{既以人天无能测，如是所作离称量，}%

@{過諸稱量離語言，是為一切不空力。}@
%{过诸称量离语言，是为一切不空力。}%

@{以是一切不空力，聞名見身獲大辯，}@
%{以是一切不空力，闻名见身获大辩，}%

@{既以大辯力如是，能作世間大施主。}@
%{既以大辩力如是，能作世间大施主。}%

@{復名不空大丈夫，以住本性丈夫故，}@
%{复名不空大丈夫，以住本性丈夫故，}%

@{不捨大慈演妙法，於諸惡道化群生。}@
%{不舍大慈演妙法，于诸恶道化群生。}%

@{演說寶光最上乘，由如金剛寶雲聚，}@
%{演说宝光最上乘，由如金刚宝云聚，}%

@{彼之自性如海寶，不增不減無有損。}@
%{彼之自性如海宝，不增不减无有损。}%

@{無邊功德亦復然，有剎無佛不聞法，}@
%{无边功德亦复然，有刹无佛不闻法，}%

@{於彼現作佛菩提，見彼由如大法藏。}@
%{于彼现作佛菩提，见彼由如大法藏。}%

@{彼常說法離眾惑，十方世界無罣礙，}@
%{彼常说法离众惑，十方世界无罣礙，}%

@{亦如月光普照耀，教化眾生千方便。}@
%{亦如月光普照耀，教化众生千方便。}%

@{剎那剎那佛菩提，即見十方諸世界，}@
%{刹那刹那佛菩提，即见十方诸世界，}%

@{常轉法輪安世間，勇猛遍轉十方界。}@
%{常转法轮安世间，勇猛遍转十方界。}%

@{彼諸聲聞辟支地，無邊變化佛莊嚴，}@
%{彼诸声闻辟支地，无边变化佛庄严，}%

@{復經不可思議劫，化諸眾生親往詣。}@
%{复经不可思议劫，化诸众生亲往诣。}%

@{若男若女童男女，天人修羅龍王類，}@
%{若男若女童男女，天人修罗龙王类，}%

@{藥叉乃至摩護羅，以解脫智皆悉見。}@
%{药叉乃至摩护罗，以解脱智皆悉见。}%

@{若諸世間眾形類，隨眾語言悉皆同，}@
%{若诸世间众形类，随众语言悉皆同，}%

@{一切勇猛皆盡見，如是勇猛盡觀察。}@
%{一切勇猛皆尽见，如是勇猛尽观察。}%

@{海印三昧從口生，得是海印眾三昧，}@
%{海印三昧从口生，得是海印众三昧，}%

@{能嚴不可思議剎。嚴飾不思議剎已，}@
%{能严不可思议刹。严饰不可思议刹已，}%

@{供養十方諸如來，如是種種供養已，}@
%{供养十方诸如来，如是种种供养已，}%

@{復得光明眾嚴飾。若得光嚴不思議，}@
%{复得光明众严饰。若得光严不可思议，}%

@{則得無邊解脫智，若得無邊解脫智，}@
%{就能得到无边解脱智，如果得到无边解脱智，}%

@{得不思議身變化。若得如是身變化，}@
%{就能得到不可思议身变化。如果得到如是身变化，}%

@{口辯智辯亦如是，若得口智無礙辯，}@
%{口辩智辩亦如是，如果得到口智无碍辩，}%

@{布施變化不思議。若得布施不思議，}@
%{布施变化不可思议。如果得到布施不可思议，}%

@{持戒忍進亦復然，若得持戒忍進通，}@
%{持戒忍进亦复然，如果得到持戒忍进通，}%

@{禪定神變莫能測。若得禪定神通變，}@
%{禅定神变莫能测。如果得到禅定神通变，}%

@{出生方便神變智，得是方便神變智，}@
%{出生方便神变智，得到是方便神变智，}%

@{出生無邊諸功德。從佛口生三摩地，}@
%{出生无边诸功德。从佛口生三摩地，}%

@{三摩地入一微塵，一切微塵皆盡入，}@
%{三摩地入一微尘，一切微尘皆尽入，}%

@{一微塵中難思剎，一一微塵皆悉見。}@
%{一微尘中难思刹，一一微尘皆悉见。}%

@{如是佛剎微塵數，其中佛剎盡皆覩，}@
%{如是佛刹微尘数，其中佛刹尽皆睹，}%

@{種種微細清淨眾，無上貴重微妙剎，}@
%{种种微细清净众，无上貴重微妙刹，}%

@{如實遠離出興世，其中祕密妙昇騰，}@
%{如实远离出兴世，其中秘密妙昇腾，}%

@{有以除暗放光明，帝網重重復無盡。}@
%{有以除暗放光明，帝网重重复无尽。}%

@{由如見一大光明，如是一切微塵盡，}@
%{由如见一大光明，如是一切微尘尽，}%

@{此大仙行三摩地，即是無邊勝解脫。}@
%{此大仙行三摩地，即是无边胜解脱。}%

@{三摩地力供養佛，一切如來供養已，}@
%{三摩地力供养佛，一切如来供养已，}%

@{復於手中變千萬，作大丈夫興廣供。}@
%{复于手中变千万，作大丈夫兴广供。}%

@{乃至十方妙華鬘，塗香發越珍奇寶，}@
%{乃至十方妙华鬘，涂香发越珍奇宝，}%

@{一一手中親施與，菩提大樹詣佛所。}@
%{一一手中亲施与，菩提大树诣佛所。}%

@{價直千萬妙香衣，種種幢幡及傘蓋，}@
%{价直千万妙香衣，种种幢幡及伞盖，}%

@{閻浮檀金莊嚴具，彼自手中親施與。}@
%{阎浮檀金庄严具，彼自手中亲施与。}%

@{無邊一切所受用，清淨供養於大仙，}@
%{无边一切所受用，清净供养于大仙，}%

@{手中奉上皆捨施，住佛菩提大樹王。}@
%{手中奉上皆舍施，住佛菩提大树王。}%

@{乃至一切眾妓樂，擊鼓出眾妙音聲，}@
%{乃至一切众妓乐，击鼓出众妙音声，}%

@{種種雅樂琴箜篌，一一手中親撫擊。}@
%{种种雅乐琴箜篌，一一手中亲抚击。}%

@{種種讚唄妙伽陀，種種稱讚真實德，}@
%{种种赞呗妙伽陀，种种称赞真实德，}%

@{無邊一切十方界，一一皆自手中作。}@
%{无边一切十方界，一一皆自手中作。}%

@{或以光明興佛事，皆從右旋而動作，}@
%{或以光明兴佛事，皆从右旋而动作，}%

@{香水普灑諸佛剎，供養一切世間燈。}@
%{香水普洒诸佛刹，供养一切世间灯。}%

@{光明嚴麗適意香，蓮華瓔珞不思議，}@
%{光明严丽适意香，莲花瓔珞不可思议，}%

@{蓮華妙色無央數，一一自作供諸佛。}@
%{莲花妙色无央数，一一自作供诸佛。}%

@{放大光明嚴飾華，種種微妙華雲海，}@
%{放大光明严饰华，种种微妙华云海，}%

@{如是妙華普周遍，廣大供養作佛事。}@
%{如是妙华普周遍，广大供养作佛事。}%

@{放大光明嚴飾香，種種微妙香雲海，}@
%{放大光明严饰香，种种微妙香云海，}%

@{如是妙香普周遍，廣大供養作佛事。}@
%{如是妙香普周遍，广大供养作佛事。}%

@{放大光明塗香獻，種種塗香妙雲海，}@
%{放大光明涂香献，种种涂香妙云海，}%

@{如是塗香普周遍，廣大供養作佛事。}@
