@{佛本行經卷第二(一名佛本行讚傳)}@
%{佛本行经卷第二(一名佛本行赞传) }%

@{宋涼州沙門釋寶雲譯}@
%{宋凉州沙门释宝云译}%

@{與眾婇女遊居品第八}@
%{与众婇女游居品第八}%

@{種種嚴飾，猶如天宮；春秋冬夏，}@
%{各种严饰，犹如天宫；春秋冬夏，}%
@{四時各異。應節修治，遊觀園池；}@
%{四季各异。应节修治，游观园池；}%
@{亦如天帝，施安樹林。太子因遊，}@
%{亦如天帝，施安树林。太子因游，}%
@{至園池觀；婇女圍繞，如月處星。}@
%{至园池观；婇女围绕，如月处星。}%

@{於是眾女，晝夜作樂；謿調戲笑，}@
%{于是众女，昼夜作乐；謿调戏笑，}%
@{過數年已。或娛樂之，更造新術；}@
%{过数年已。或娱乐之，更造新术；}%
@{或現己身，或時書頌。或圖廟畫，}@
%{或现己身，或时书颂。或图庙画，}%
@{或有刻鏤；或有以泥，為若干像。}@
%{或有刻镂；或有以泥，为若干像。}%

@{或有結花，以為敷飾；或㽵面目，}@
%{或有结花，以为敷饰；或㽵面目，}%
@{或有塗香。或以鏡照，或櫛梳頭；}@
%{或有涂香。或以镜照，或栉梳头；}%
@{或黛黑眉，或丹口脣。或復有女，}@
%{或黛黑眉，或丹口唇。或复有女，}%
@{華相打擲；或戲笑者，或悲歎泣。}@
%{华相打掷；或戏笑者，或悲叹泣。}%

@{或口詠歌，可聽可樂；猶如華中，}@
%{或口咏歌，可听可乐；犹如华中，}%
@{眾蜜蜂鳴。眾女求浴，太子聽從；}@
%{众蜜蜂鸣。众女求浴，太子听从；}%
@{無憂樹間，丹華之池。圓光文飾，}@
%{无忧树间，丹华之池。圆光文饰，}%
@{照樹金色；猶金在火，耀耀林樹。}@
%{照树金色；犹金在火，耀耀林树。}%

@{諸樹傾屈，散供眾花；眾鳥相和，}@
%{诸树倾屈，散供众花；众鸟相和，}%
@{悲哀而鳴。女笑鳥鳴，震聲遠聞；}@
%{悲哀而鸣。女笑鸟鸣，震声远闻；}%
@{五音俱作，感動人情。太子入池，}@
%{五音俱作，感动人情。太子入池，}%
@{水至其腰；諸女圍繞，明耀浴池。}@
%{水至其腰；诸女围绕，明耀浴池。}%

@{猶如明珠，繞寶山王；妙相顯赫，}@
%{犹如明珠，绕宝山王；妙相显赫，}%
@{其好巍巍。眾女水中，種種戲笑；}@
%{其好巍巍。众女水中，种种戏笑；}%
@{或相湮沒，或水相灑。或有弄華，}@
%{或相湮没，或水相洒。或有弄华，}%
@{以華相擲；或入水底，良久乃出。}@
%{以华相掷；或入水底，良久乃出。}%

@{或於水中，現其眾華；或復於水，}@
%{或于水中，现其众华；或复于水，}%
@{但現其手。眾女池中，光耀眾華；}@
%{但现其手。众女池中，光耀众华；}%
@{令眾藕花，失其精光。或有攀緣，}@
%{令众藕花，失其精光。或有攀缘，}%
@{太子手臂；猶如雜花，纏著金柱。}@
%{太子手臂；犹如杂花，缠着金柱。}%

@{女粧塗香，水洗皆墮；栴檀木櫁，}@
%{女妆涂香，水洗皆堕；栴檀木櫁，}%
@{水成香池。如是戲笑，難可計數；}@
%{水成香池。如是戏笑，难可计数；}%
@{六萬婇女，圍繞其側。太子於中，}@
%{六万婇女，围绕其侧。太子于中，}%
@{如天帝釋；於天浴池，與天女俱。}@
%{如天帝释；于天浴池，与天女俱。}%

@{於是皆乘，金銀寶船；遊戲池中，}@
%{于是皆乘，金银宝船；游戏池中，}%
@{如天乘雲。太子復乘，七寶之船；}@
%{如天乘云。太子复乘，七宝之船；}%
@{妃在其側，俱共入池。色身金照，}@
%{妃在其侧，俱共入池。色身金照，}%
@{光各一丈；如日乘船，莫不驚愕。}@
%{光各一丈；如日乘船，莫不惊愕。}%

@{謂是日出，眾華開張；明重光照，}@
%{谓是日出，众华开张；明重光照，}%
@{喻日天子。太子出池，諸女更嚴；}@
%{喻日天子。太子出池，诸女更严；}%
@{作眾伎樂，行甘蔗漿。諸女飲已，}@
%{作众伎乐，行甘蔗浆。诸女饮已，}%
@{跳踉舞戲；時日便冥，向月燭明。}@
%{跳踉舞戏；时日便冥，向月烛明。}%

@{欲惑太子，意終不傾；欲使其貪，}@
%{欲惑太子，意终不倾；欲使其贪，}%
@{意終不著。慧燈甚明，終無能滅；}@
%{意终不著。慧灯甚明，终无能灭；}%
@{猶明珠燈，不損飛蛾。斯須至冥，}@
%{犹明珠灯，不损飞蛾。斯须至冥，}%
@{眾女睡眠；太子妃寐，夢覩憂變。}@
%{众女睡眠；太子妃寐，梦睹忧变。}%

@{太子出家，捨宮婇女；逃入山澤，}@
%{太子出家，舍宫婇女；逃入山泽，}%
@{妃獨逐走。從後求哀：莫相捐棄，}@
%{妃独逐走。从后求哀：莫相捐弃，}%
@{澡手見授，今棄付誰？惟今自察，}@
%{澡手见授，今弃付谁？惟今自察，}%
@{無量過失；願微重榮，莫生相捨。}@
%{无量过失；愿微重荣，莫生相舍。}%

@{獨入山澤，勤修眾德；何恡於妾，}@
%{独入山泽，勤修众德；何恡于妾，}%
@{致大酷惡。古賢出學，亦復好妃；}@
%{致大酷恶。古贤出学，亦复好妃；}%
@{唯垂愍傷，願聽侍從。追呼不息，}@
%{唯垂愍伤，愿听侍从。追呼不息，}%
@{太子入林；心意發狂，樹樹行求。}@
%{太子入林；心意发狂，树树行求。}%

@{㘁向樹曰：汝獨無憂，我獨懷惱；}@
%{㘁向树曰：汝独无忧，我独怀恼；}%
@{示我太子。仰見樹上，有赤觜鳥；}@
%{示我太子。仰见树上，有赤觜鸟；}%
@{向鳥歎苦：我失所怙。汝聲似之，}@
%{向鸟叹苦：我失所怙。汝声似之，}%
@{留聲與汝；願以鳴聲，除我心惱。}@
%{留声与汝；愿以鸣声，除我心恼。}%

@{又見樹曰：汝何不慈？吾厄夫怙，}@
%{又见树曰：汝何不慈？吾厄夫怙，}%
@{迷行犯觸。賢夫所棄，宜見愍傷；}@
%{迷行犯触。贤夫所弃，宜见愍伤；}%
@{如何見笑，華盡開敷。見雙頭鵒，}@
%{如何见笑，华尽开敷。见双头鵒，}%
@{相將俱飛；益增憂苦，流泣且言：}@
%{相将俱飞；益增忧苦，流泣且言：}%

@{唯願示我，懷重慊者；令我覩之，}@
%{唯愿示我，怀重慊者；令我睹之，}%
@{失之我亡。樹華散我，更耗我心；}@
%{失之我亡。树华散我，更耗我心；}%
@{唯汝仁鳥，莫嫉快我。當散我心，}@
%{唯汝仁鸟，莫嫉快我。当散我心，}%
@{好喜伎樂；未曾相犯，何為必爾。}@
%{好喜伎乐；未曾相犯，何为必尔。}%

@{為風所動，樹枝傾曲；猶如以掌，}@
%{为风所动，树枝倾曲；犹如以掌，}%
@{擊打於我。鳥獸角視，爾不遣夫；}@
%{击打于我。鸟兽角视，尔不遣夫；}%
@{水聲如罵，我不任治。太子不還，}@
%{水声如骂，我不任治。太子不还，}%
@{便悲歎曰：留目紺蓮，留超與應。}@
%{便悲叹曰：留目紺莲，留超与应。}%

@{留笑華敷，留顏金華；留髮辟兵，}@
%{留笑华敷，留颜金华；留发辟兵，}%
@{我見遺忽？口言未止，忽見太子，}@
%{我见遗忽？口言未止，忽见太子，}%
@{於林樹間，便前搏曰：何為相棄？}@
%{于林树间，便上前搏曰：何为相弃？}%

@{語頃驚覺；抱持太子，慚懼戰𤴨。}@
%{语顷惊觉；抱持太子，惭惧战𤴨。}%

@{太子問曰：何為如是？便說夢事。}@
%{太子问曰：何为如是？便说梦事。}%

@{太子報曰：此非為汝？吾無所去，}@
%{太子报曰：此非为汝？吾无所去，}%
@{誰往誰返？無有往者，亦無所至。}@
%{谁往谁返？无有往者，亦无所得。}%

@{汝諦覺是，色如聚沫，覺意如泡，}@
%{汝谛觉是，色如聚沫，觉意如泡，}%
@{相如光炎，行如芭蕉，識法如幻，}@
%{相如光炎，行如芭蕉，识法如幻，}%
@{諸根無力；形體相因，猶如華合。}@
%{诸根无力；形体相因，犹如华合。}%

@{覺世無常，譬如野馬；吾我無有，}@
%{觉世无常，譬如野马；吾我无有，}%
@{亦無堅要。合會有離，汝當諦覺。}@
%{亦无坚要。合会有离，汝当谛觉。}%

@{太子自思：是吾出應。}@
%{太子自思：是吾出应。}%
@{佛本行經現憂懼品第九}@
%{佛本行经现忧惧品第九}%

@{王愍太子愁，勸令行遊觀；}@
%{国王怜悯太子忧愁，劝他出去游玩观赏；}%

@{始出宮城門，霍然日出雲。}@
%{刚走出皇宫大门，突然太阳从云层中升起。}%

@{駕乘七寶車，眾德相自嚴；}@
%{太子乘坐七宝车，各种美德都显现出来；}%

@{所將從貴重，如月與眾星。}@
%{跟随他出行的人都是贵重之人，如同月亮和众星。}%

@{功德充滿備，形容甚殊妙；}@
%{功德圆满具备，容貌非常殊妙；}%

@{都勅國邑里，并除老病死。}@
%{命令全国城乡，都要消除老病死。}%

@{窮凍困厄者，莫令現道側；}@
%{贫穷困苦的人，不要让他们出现在路边；}%

@{各各盡力嚴，若干幢旛蓋。}@
%{每个人都尽力装饰，许多幢幡和盖子。}%

@{樓閣諸婦女，猶如天宮城；}@
%{楼阁上的妇女，如同天宫一样；}%

@{嚴飾甚靡麗，莫不懷歡慶。}@
%{装饰得非常华丽，没有不怀着喜庆心情的。}%

@{萬民皆歌詠，聲嚮震一國；}@
%{万民都歌唱，声音震动整个国家；}%

@{猶秋水歸海，諍競欲觀見。}@
%{如同秋水归海，大家都争先恐后地想要观看。}%

@{莊嚴易服飾，未竟便起走；}@
%{装饰还没完成，就起身离开了；}%

@{或未及莊嚴，聞聲便馳往。}@
%{或者还没来得及装饰，听到声音就赶去。}%

@{於眾閣欄楯，側塞不相容；}@
%{在众多的阁楼栏杆上，人挤得水泄不通；}%

@{或頭身自懸，猶如眾華垂。}@
%{有些人头身悬挂在空中，如同众多的花朵垂下来。}%

@{或有傾屈禮，各懷敬歎曰：}@
%{有些人恭敬地鞠躬，心中都敬佩地说：}%

@{當為世導師。散眾花香瓔。}@
%{他将来一定会成为世间的导师。散布着花香和璎珞。}%

@{見以皆愕觀，展轉相謂言：}@
%{看到他的人都惊愕地观望，互相转告说：}%

@{此當為何神？或云從天降，}@
%{这究竟是什么神？有人说他是从天上降下来的，}%

@{或云是天帝，魔王或梵王，}@
%{有人说他是天帝，魔王或者梵王，}%

@{懷疑歡踊躍，歌歎若干種。}@
%{大家带着怀疑，欢欣鼓舞，唱着各种赞歌。}%

@{諸天見太子，容飾導從出；}@
%{诸天看到太子，装饰和随从出城；}%

@{猶如天帝釋，出遊觀之時。}@
%{如同天帝释，出游观赏的时候。}%

@{於是淨居天，欲興降瑞應；}@
%{于是净居天，想要兴起降下瑞应；}%

@{如前佛見瑞，勸意出家學。}@
%{如同以前佛陀见到瑞相，劝他出家修行。}%

@{天卒化病人，喘臥在道側；}@
%{天卒化作病人，喘息着躺在路边；}%

@{色惡眼睛黃，體氣口燋乾。}@
%{脸色难看，眼睛发黄，身体虚弱，口干舌燥。}%

@{身腫腹膖脹，惡露諸不淨；}@
%{身体肿胀，腹部鼓胀，污秽之物流出；}%

@{宛轉而自塗，菩薩舉目見。}@
%{他翻滚着在地上，菩萨抬头看见了。}%

@{問：是為何物？醜惡難可視。}@
%{问：这是什么东西？丑陋不堪，难以观看。}%

@{御者尋對曰：食飲不時節，}@
%{御者随即回答说：饮食不规律，}%

@{四大錯不順，是名為病人。}@
%{四大不调和，这就是叫做病人。}%

@{菩薩報之曰：視何不分減？}@
%{菩萨回答他说：看什么东西都不能减少？}%

@{御者復對曰：是不可分代，}@
%{御者又回答说：这不可代替，}%

@{都世無能免，疾病之危厄。}@
%{世间上没有人能够免除，疾病的危险。}%

@{四百四種病，大患如世間；}@
%{四百四种疾病，是世间的大患；}%

@{尊亦未免離，處大變難患。}@
%{您也不能免除，处于重大变化的痛苦。}%

@{太子即停駕，慘然懷憂歎；}@
%{太子立刻停下车，悲伤地忧愁叹息；}%

@{聞病心驚痛，如象被毒箭。}@
%{听到病痛，内心惊恐疼痛，如同大象被毒箭射中。}%

@{見病觸其情，尋勅御迴車；}@
%{看到病人，触动了他的感情，随即命令御者掉头回去；}%

@{心懼懷悚然，如牛畏雷雹；}@
%{内心恐惧，心中忐忑不安，如同牛害怕雷雹；}%

@{聞雷聲怖愕，驚懅體不安。}@
%{听到雷声惊恐，害怕不安，身体不适。}%

@{後時復更出，天化作老人；}@
%{后来又再次出去，天化作老人；}%

@{頭如絲雪霧，皮緩肌體皺，}@
%{头发如同丝雪般稀疏，皮肤松弛，身体起皱，}%

@{戰如水中枝，身僂如張弓。}@
%{身体摇摇晃晃，如同水中的树枝，背驼得如同张开的弓。}%

@{太子見即問：是名為何人？}@
%{太子看到就问：这叫什么人？}%

@{生便如是耶？為有變者乎？}@
%{他天生就是这样吗？还是发生了变化？}%

@{御者因對曰：始從身受胎，}@
%{御者于是回答说：开始从身体受胎，}%

@{微起如泡沫；緣起五體見，}@
%{微弱地如同泡沫；缘起五体出现，}%

@{分合成六情，然後乃出生。}@
%{分合成为六情，然后才出生。}%

@{小飲母乳活，次長乃食穀；}@
%{小时候喝母乳长大，长大后就吃谷物；}%

@{轉緣地而行，初語如鸚鵡，}@
%{开始在地上行走，最初说话像鹦鹉一样，}%

@{爾乃立行走，體貌形容成；}@
%{然后站起来行走，体貌容貌形成；}%

@{諸根轉成熟，以故名曰老。}@
%{各种器官逐渐成熟，因此叫做老。}%

@{是名天使召，顯教悟眾生，}@
%{这就是天神的召集，明显地教导众生，}%

@{形衰失歡慘，如花被毒霜。}@
%{身体衰败，失去欢乐，悲惨，如同花朵被毒霜覆盖。}%

@{面如月遭蝕，心猶日雲霧；}@
%{面容如同月亮被蚀，内心如同太阳被云雾遮蔽；}%

@{壯進力枯竭，如夏沙淋水。}@
%{壮年时的力量枯竭，如同夏天的沙子被水淋湿。}%

@{竊人志思才，無形如來賊；}@
%{偷窃人的志向和才智，无形地如同来的贼；}%

@{眾人往自觀，如蜂集蓮花。}@
%{人们都去观看，就像蜜蜂聚集在荷花上。}%

@{厥所由至趣，眾人悉隨從；}@
%{太子所到之处，人们都跟随他；}%
@{譬如人諸根，隨心走周旋。}@
%{就像人的六根，随着心意来回转动。}%
@{因展轉相謂，歎羨其功德；}@
%{人们互相议论，赞叹他的功德；}%
@{觀此言是人，人中之妙寶。}@
%{说这个人，是人间的珍宝。}%

@{諦視其眼目，面之妙姿容；}@
%{仔细观察他的眼睛，他面容的美丽；}%
@{譬如聚粟金，中有帝紺寶。}@
%{就像一堆金粟米，其中藏着珍贵的青金石。}%
@{光明所纏繞，德相積聚成；}@
%{光明环绕着他，德行相好聚集在一起；}%
@{姿貌甚和合，眾好悉具足。}@
%{容貌非常和谐，各种美好的相貌都具备。}%

@{彌一切人意，眼睛俱相隨；}@
%{他超越了一切人的意念，人们的眼睛都跟着他；}%
@{數數諦熟視，而無有厭足。}@
%{一遍又一遍地仔细观察，却始终没有厌倦。}%
@{猶如遇暴雪，寒凍甚猛切；}@
%{就像遇到暴风雪，寒风刺骨；}%
@{眾人爭競前，猶如得火湯。}@
%{人们争先恐后地跑上前，就像得到了热汤一样。}%

@{諸貴姓女人，各馳出其舍；}@
%{那些贵族的妇女，都从家里跑出来；}%
@{猶如盛雲中，晃晃出電光。}@
%{就像盛大的云层中，闪耀着闪电的光芒。}%
@{譬如無憂樹，枝葉花繁茂；}@
%{就像无忧树，枝繁叶茂，花朵繁盛；}%
@{風之所吹屈，傾曲禮菩薩。}@
%{被风吹弯了，向菩萨致敬。}%

@{抱上嬰孩兒，口皆放母乳；}@
%{抱着怀里的婴儿，都张开嘴巴，流着口水；}%
@{熟視觀菩薩，忘不還求乳。}@
%{仔细观察菩萨，忘记了还要吃奶。}%
@{舉城中人民，皆共競嘆譽；}@
%{全城的人民，都争先恐后地赞美他；}%
@{各各言是好，是妙是好相。}@
%{都说他好，他妙，他相好。}%

@{時有一人言：若卒有人言，}@
%{这时有一个人说：如果有人说，}%
@{如甞石蜜餅，是美是不美？}@
%{就像吃石蜜饼，是好吃还是不好吃？}%
@{如彼之形貌，若偏歎一事；}@
%{就像他的相貌，如果只赞叹某一方面；}%
@{不可偏說一，眾德善積聚。}@
%{不应该只说一方面，他的各种美德都积累在一起。}%

@{審諦之表識，眾善盡顯露；}@
%{他真谛的标志，各种善行都显露出来；}%
@{以嚴飾其身，充滿人心目。}@
%{用这些美德庄严自己的身体，充满了人们的心目。}%
@{眾花飾其容，軟香感人情；}@
%{各种花朵装饰着他的容颜，芬芳的香气打动人心；}%
@{猶如陽春節，顯然而晃昱。}@
%{就像春光明媚的季节，显现出耀眼的光芒。}%

@{夫宿行之報，如何無慈心；}@
%{过去生中行善的报应，怎么可能会没有慈悲心；}%
@{以是天形乞，不為普地王。}@
%{以这种天人的形象乞讨，不像普天下的帝王。}%
@{識能識知是，行從他人乞；}@
%{能认识到这一点，行为却要向别人乞讨；}%
@{誰能施是人，計無有稱者。}@
%{谁能施舍他，真是想不出合适的人选。}%

@{爾時其國王，厥號為瓶沙；}@
%{当时那个国王，名叫瓶沙；}%
@{時處高觀上，遙見太子行。}@
%{他正站在高高的观台上，远远地看见太子走来。}%
@{即問左右曰：彼行者是誰？}@
%{就问左右的人说：那个走来的人是谁？}%
@{容貌甚鮮明，而服縵色衣。}@
%{容貌非常鲜明，却穿着素色的衣服。}%

@{傍臣即啟王，廣陳其種姓；}@
%{旁边的臣子就告诉国王，详细地说明他的出身；}%
@{王勅傍臣曰：察其行所趣？}@
%{国王命令臣子说：观察他要去哪里？}%
@{於城外食訖，上槃塔名山；}@
%{他在城外吃完饭，就上了槃塔山；}%
@{光影照然明，如日臨山崗。}@
%{光芒照耀，就像太阳照耀着山岗。}%

@{於時王瓶沙，單與將從俱；}@
%{于是瓶沙王，单独带着侍卫一起；}%
@{王至槃塔山，服飾形容殊。}@
%{国王来到槃塔山，换了衣服，装扮得与众不同。}%
@{侍從執寶蓋，祥雅師子步；}@
%{侍卫们举着宝盖，国王优雅地迈着狮子的步伐；}%
@{王至下寶車，步步而登山。}@
%{国王来到宝车下，一步一步地登上山去。}%

@{見太子獨坐，諸根寂然滅；}@
%{看见太子独自坐着，六根都寂静了；}%
@{譬如盛滿月，霍然處雲上。}@
%{就像圆满的月亮，突然出现在云层之上。}%
@{如諸法色像，忽然而化現；}@
%{就像所有的法相，突然显现出来；}%
@{意甚懷愕然，顧謂傍臣曰：}@
%{国王心中非常惊讶，回头对臣子说：}%

@{其有是形貌，姿容可愛喜；}@
%{他拥有如此的相貌，容颜令人喜爱；}%
@{今是必可保，能成大善德。}@
%{他一定能成就大善德。}%
@{今觀其眾善，宛軟而柔弱；}@
%{现在观察他的各种善行，柔和而柔软；}%
@{略視其要相，唯佛應有是。}@
%{稍微观察一下他的主要相貌，只有佛才应该拥有这样的相貌。}%

@{除去憍慢意，執謙而為禮；}@
%{国王放下傲慢的念头，恭敬地向太子行礼；}%
@{王因其時宜，而問訊菩薩。}@
%{国王趁机，向菩萨问候。}%
@{王以清淨意，前坐青石上；}@
%{国王怀着清净的心，走到前面，坐在一块青石上；}%
@{即便啟菩薩，因是說偈言：}@
%{就向菩萨问道，并用偈语说道：}%

@{太子之先族，出於日天子；}@
%{太子你的祖先，来自太阳天；}%
@{年時在壯幼，形容甚照曜。}@
%{你正值壮年，容貌非常光耀。}%
@{不審其緣故，乃興發此意；}@
%{不知道是什么原因，你才产生了这样的念头；}%
@{乞匃以自濟，不甘世王位。}@
%{为了解脱自己，你不愿意做世间的国王。}%

@{姿媚甚怡照，已積善快然；}@
%{你容貌非常美丽，已经积累了很多善行，令人欢喜；}%
@{猶如閻浮樹，眾花茂盛好。}@
%{就像阎浮树，花朵繁盛美丽。}%
@{服著此袈裟，喻如以草裹；}@
%{你穿着这件袈裟，就像用草包裹着；}%
@{猶樹花畏雪，不敢顯其輝。}@
%{就像树上的花朵害怕雪，不敢显露自己的光辉。}%

@{太子宜服著，天上之名寶；}@
%{太子你应该穿着天上的珍宝；}%
@{今此縵色衣，殊不與相宜。}@
%{现在这件素色的衣服，实在是不适合你。}%
@{若有潔淨物，有微穢可訶；}@
%{如果有什么洁净的东西，有一点污秽，就会被人批评；}%
@{顯露而悉現，無方可以除。}@
%{显露出来，就会全部显现，没有办法可以消除。}%

@{臂𦟛甚長好，猶如紫金柱；}@
%{你的手臂非常修长，就像紫金柱一样；}%
@{宜飾以七寶，應執持妙弓。}@
%{应该用七宝装饰，应该拿着神妙的弓箭。}%
@{如是之手臂，但宜以惠施；}@
%{这样美好的手臂，应该用来施舍；}%
@{不宜以是手，從人而乞取。}@
%{不应该用这双手，向别人乞讨。}%

@{若其執謙敬，不肯習父位；}@
%{如果你执着于谦虚恭敬，不愿意继承父王的王位；}%
@{今我盡愛敬，相請臨鄙國。}@
%{我现在真诚地敬爱你，邀请你到我的国家。}%
@{盡所有榮祿，享此摩竭國；}@
%{享用我所有的荣华富贵，享受这个摩竭国；}%
@{若欲普土地，鄙當躬相佐。}@
%{如果你想普度众生，我将亲自辅佐你。}%

@{如仁之德相，并應攝天世；}@
%{你的仁德，应该摄受天人和世间的人；}%
@{但以手執持，天世豈足勞。}@
%{只要你用手轻轻一挥，天人和世间的人，又有什么值得劳神呢？}%
@{如今天帝釋，起中為人臣；}@
%{就像现在的帝释天，也曾经做人的臣子；}%
@{何況此地上，如吾等諸王。}@
%{何况是人间，像我们这些国王。}%

@{我不厭善法，亦不患守義；}@
%{我并不厌恶善法，也不担心守持正道；}%
@{非時捨出家，唯此疑我情。}@
%{只是你没有选择合适的时间出家，这让我感到疑惑。}%
@{初如少壯年，已過志衰弱；}@
%{刚开始的时候，你正值壮年，志气强盛；}%
@{諸根以調良，易迴如馬勒。}@
%{六根已经调伏得很好，很容易像驯服马匹一样。}%

@{施慧戒自守，眾行之面目；}@
%{布施智慧，守持戒律，这是你善行的表现；}%
@{發現先世行，逮遇前善本。}@
%{显现出你过去生中的善行，你遇到了过去的善根。}%
@{頃年轉長大，爾乃可奉法；}@
%{现在你已经长大，才能奉行佛法；}%
@{年高意便調，不隨逐諸欲。}@
%{年纪大了，心意就容易调伏，不会再追逐欲望。}%

@{以是故不可，困極其形體；}@
%{因此，不应该过度地困苦自己的身体；}%
@{今所逮福慶，順理可享食。}@
%{现在你得到的福报，应该顺理成章地享受。}%
@{六情之可欲，如應決充益；}@
%{六根的欲望，应该满足之后，才能寻求解脱法门。}%
@{然後乃出求，甘露解脫法。}@
%{然后才能出去寻求解脱的甘露法。}%

@{姿容之光明，超逾日之精；}@
%{你的容貌的光明，超越了太阳的光芒；}%
@{仁德已過出，世人諸天人。}@
%{你的仁德已经超过了世间的人和天人。}%
@{自古未曾聞，亦所不曾見；}@
%{从古至今，从来没有听说过，也没有见过；}%
@{如是之形貌，覩者皆愕然。}@
%{像你这样的相貌，看到的人都会感到惊讶。}%

@{如今見太子，舉動之風姿；}@
%{现在我看见太子，举止的风度；}%
@{又察其志本，甚猛銳堅強。}@
%{又观察你的志向，非常坚定强盛。}%
@{猶如深淵底，群魚於中遊；}@
%{就像深海的底部，一群鱼在里面游动；}%
@{於上雖不現，察外動可知。}@
%{虽然在上面看不见，但从外表的行动可以知道。}%

@{今見諸楷式，在體而隱胗；}@
%{现在我看到你的各种特征，都隐含在你的身体里；}%
@{決定照然明，指示聖王位。}@
%{一定能照耀光明，指引你成为圣王。}%
@{如是之妙善，不加薄德人；}@
%{如此美妙的善行，不会施舍给薄德的人；}%
@{吉祥善名號，不歸不肖子。}@
%{吉祥的善名，不会属于不肖子。}%

@{假令心懷疑，還恥於門族；}@
%{即使你内心有疑虑，也会感到羞愧，愧对你的家族；}%
@{以服沙門形，如何當還捨。}@
%{穿着沙门的衣服，怎么能舍弃呢？}%
@{古世諸帝代，天王未以來；}@
%{古代的帝王，没有一个在壮年的时候，就放弃王位；}%
@{壯年欲國位，垂老皆捨家。}@
%{壮年的时候都想得到王位，到老了才舍弃家园。}%

@{瓶沙說是事，廣牽引比故；}@
%{瓶沙国王说了这些话，广泛地引经据典；}%
@{又有餘辛酸，寫體謙敬辭。}@
%{还带着一些辛酸，用谦卑恭敬的语言表达。}%
@{視覺菩薩意，不動如太山；}@
%{看着菩萨，他的心意像泰山一样稳固；}%
@{執慈而默視，清心聽報誨。}@
%{带着慈悲之心，默默地看着，清净地听着国王的教诲。}%
