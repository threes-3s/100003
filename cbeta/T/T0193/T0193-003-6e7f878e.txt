@{宋涼州沙門釋寶雲譯}@
%{宋涼州沙門釋寶雲譯}%

@{意如重慶雲，欲時降甘露；}@
%{如同重慶的云彩，想要及时降下甘露；}%

@{以深奧流聲，清淨梵天音。}@
%{用深奥的流淌的声音，清净的梵天之音。}%

@{又復以八種，柔軟和調嚮；}@
%{又用八种，柔软和谐的调子；}%

@{以妙辭之前，降測瓶沙王。}@
%{用美妙的语言，降下测度瓶沙王。}%

@{具照知王意，正直且清淨；}@
%{充分照知王的心意，正直且清净；}%

@{慈愛恭敬意，充盈於心懷。}@
%{慈爱恭敬的心意，充满于心怀。}%

@{今王雖興於，衰末濁亂世；}@
%{如今您虽然兴起于，衰败末世的混乱时代；}%

@{以善自將御，乃踰上世王。}@
%{用善巧的方式治理，已经超越了上世的国王。}%

@{見王諸淨意，內意清淨徹；}@
%{看到您诸多的清净心意，内心的清净贯彻始终；}%

@{猶如晝陰霧，花開知日出。}@
%{犹如白昼的阴霾，花开就知道太阳出来了。}%

@{雖施恩於愚人，恩德終不居之；}@
%{虽然施恩于愚人，恩德最终不会留存；}%

@{施少善於賢人，其恩好日日厚。}@
%{施予贤人少许善行，其恩情日日加深。}%

@{今視王甚奇特，不迷惑於王位；}@
%{如今看来您非常奇特，不迷惑于王位；}%

@{憍自恣所覆蓋，土地主皆迷惑。}@
%{骄慢自恣所覆盖，土地主都迷惑。}%

@{智慧人若得財，以不要財為要；}@
%{智慧的人如果得到财富，以不要财富为要；}%

@{身不要如炎花，慧者從身取要。}@
%{不要像炎花一样，慧者从自身取要。}%

@{眾生昇天，甚得大利；諸土地主，}@
%{众生升天，非常得大利；诸土地主，}%

@{以正治法。正法之王，順理而治；}@
%{用正治的法。正法的国王，顺理而治；}%

@{一切人民，皆從真正。若有財寶，}@
%{一切人民，都从正道。如果有财宝，}%

@{先審取要；若復別離，後無恨心。}@
%{先审察取要；如果又分离，后来没有恨心。}%

@{猶如從酪，以取酥去；漿若翻棄，}@
%{犹如从酪，取出酥去；浆液如果翻弃，}%

@{亦無恨心。以相迎接，用上賓禮；}@
%{也没有恨心。以相迎接，用上宾礼；}%

@{開恩厚意，以善友義。今我當以，}@
%{开恩厚意，以善友的义理。如今我应该以，}%

@{友恩相報；意欲相諫，開意善聽。}@
%{友谊恩情相报；想要劝谏您，开诚布公地听。}%

@{一切眾生，命如朝露；我今一切，}@
%{一切众生，生命如同朝露；我如今一切，}%

@{都就後世。猶如盛火，得酥益熾；}@
%{都归于后世。犹如盛火，得到酥油更加炽盛；}%

@{及燒草木，終無厭足。心之憎愛，}@
%{及燃烧草木，最终没有厌足。心里的憎爱，}%

@{由愚癡出；皆復迷惑，狂醉高藥。}@
%{由愚痴而出；都迷惑，狂醉于高药。}%

@{老耄之病，死亡之火；強燒五道，}@
%{老迈的疾病，死亡之火；强烧五道，}%

@{沈無漏脫。我今已覺，盛火之力；}@
%{沉沦无漏脱。我如今已经觉悟，盛火的力量；}%

@{今欲方便，免此大患。是故捨離，}@
%{如今想要方便，免除这种大患。因此舍弃，}%

@{親族知識；愛欲如毒，云何不捨？}@
%{亲族朋友；爱欲如同毒药，怎么能够不舍？}%

@{吾已不畏，是諸毒虺；雹及火熾，}@
%{我已经不畏惧，这些毒蛇；冰雹和火炽，}%

@{疾暴惡風。亦復不畏，拔刀劇賊；}@
%{疾风暴风。也同样不畏惧，拔刀的强盗；}%

@{但畏恩愛，數數生死。迷於欲者，}@
%{只畏惧恩爱，不断地生死。迷于欲望的人，}%

@{未曾厭足；諸天亦爾，況世間人。}@
%{从未厌足；诸天也是如此，何况世间人。}%

@{一切世間，欲求無厭；如火所熾，}@
%{一切世间，欲望无厌；如同火燃烧，}%

@{亦復厭極。普得土地，齊海以內；}@
%{也同样厌倦至极。普遍得到土地，齐海之内；}%

@{又貪求度，欲得彼岸。一切求索，}@
%{又贪求度，想要得到彼岸。一切求索，}%

@{無有厭足；猶如眾流，歸于大海。}@
%{没有厌足；如同众流，归于大海。}%

@{雨寶七日，乃至於膝；勝伏四方，}@
%{雨宝七天，乃至到达膝盖；战胜四方，}%

@{上至天上。天上壽命，長七劫半；}@
%{上至天上。天上的寿命，长七劫半；}%

@{頂生聖王，欲無厭足。更有聖王，}@
%{顶生的圣王，欲望无厌足。更有圣王，}%

@{食天福祿；時天帝退，畏阿須倫。}@
%{享用天上的福禄；当时天帝退位，畏惧阿修罗。}%

@{憍慢遂盛，仙人擔輩；邪住無厭，}@
%{骄慢于是盛行，仙人担负着；邪恶住处无厌，}%

@{從天退墮。有滿唯王，往至天上；}@
%{从天上退堕。有满唯王，前往天上；}%

@{得天婇女，形專將來。有犯神仙，}@
%{得到天上的美女，形体专一地将来。有犯神仙，}%

@{金寶精舍；懷貪滅亡，尋化成灰。}@
%{金宝精舍；怀着贪婪而灭亡，寻而化成灰。}%

@{有重擔王，將兵上天；復從天上，}@
%{有重担王，将兵上天；又从天上，}%

@{將天女還。緣是天女，自致死亡；}@
%{将天女回来。缘由是天女，自己导致死亡；}%

@{如是眾生，無厭而沒。名聞弊惡，}@
%{如此众生，无厌而亡。名声败坏，}%

@{弓強箭駛；捨棄王位，入林學仙。}@
%{弓强箭疾；舍弃王位，进入森林学仙。}%

@{心所不應，其身行之；殺害他人，}@
%{心所不应，其身却行之；杀害他人，}%

@{身亦滅亡。有女名賢，諸王競諍；}@
%{自身也灭亡。有女子名叫贤，诸王争斗；}%

@{興師率眾，對陣而戰。以愛欲故，}@
%{兴兵率领大众，对阵而战。因为爱欲，}%

@{興怨致諍；當捨愛欲，如棄怨敵。}@
%{兴起怨恨而争斗；应当舍弃爱欲，如同抛弃怨敌。}%

@{諸王種族，挾慊嫉害；殺諸剎利，}@
%{诸王种族，怀着嫉妒伤害；杀害诸刹利，}%

@{二百一十。往古列士，懷恚為害；}@
%{二百一十。往古的勇士，怀着怨恨而害人；}%

@{宜捨惡心，如虺脫皮。言鴈言鶴，}@
%{应当舍弃恶念，如同蛇蜕皮。说雁说鹤，}%

@{緣是致諍；因相殺傷，乃至億數。}@
%{缘由是争斗；因为互相杀害，乃至亿数。}%

@{愚癡故鬪，始起於微；宜捨愚癡，}@
%{愚痴所以争斗，开始于微小；应当舍弃愚痴，}%

@{棄自冥病。往古二王，鬪諍澤香；}@
%{抛弃自身冥顽的疾病。往古二王，争斗泽香；}%

@{因懷貢高，遂相殺傷。以鐵棒擊，}@
%{因为怀着傲慢，于是互相杀害。用铁棒击打，}%

@{破碎其頭；宜捨憍慢，如雲之蓋。}@
%{破碎其头；应当舍弃骄慢，如同云彩的遮盖。}%

@{又如列士，甚徤勦疾；以貪愛故，}@
%{又如同勇士，非常勇敢地攻打疾病；因为贪爱，}%

@{奪他婦女。以愛著故，乃致沒亡；}@
%{夺取别人的妻子。因为爱着，所以导致死亡；}%

@{如畏死故，宜捨愛著。二阿須倫，}@
%{如同畏惧死亡，应当舍弃爱着。二阿修罗，}%

@{坐女色亡；有貪女色，聽音沒命。}@
%{因为女色而亡；有贪图女色，听声音而丧命。}%

@{騰翔空中，賓頭王子；是二嗜味，}@
%{腾翔空中，宾头王子；这两个嗜好，}%

@{俱致喪命。昔伊象王，以鼻貪香；}@
%{都导致丧命。昔日伊象王，用鼻子贪图香气；}%

@{犯吉祥天，以致喪滅。昔殷頭王，}@
%{犯了吉祥天，导致灭亡。昔日殷头王，}%

@{身樂細滑；貪著無厭，頭裂而亡。}@
%{身体喜欢细滑；贪着无厌，头裂而亡。}%

@{此等以欲，恣極六情；如海受流，}@
%{这些人用欲望，放纵六情；如同大海接受河流，}%

@{終以無厭。摩竭魚口，猶尚可滿；}@
%{最终以无厌。摩竭鱼的嘴巴，尚且可以填满；}%

@{六情受欲，難可充滿。如是大貪，}@
%{六情受欲望，难以填满。如此大的贪婪，}%

@{及餘難計；六情未足，遭大艱難。}@
%{以及其他难以计数；六情不满足，遭遇大困难。}%

@{可如王言，先恣六情；誰有厭足，}@
%{可以如同您所说，先放纵六情；谁能够厌足，}%

@{宜審思此。王以敬意，相請以國；}@
%{应当审思此事。您以敬意，邀请我以国家；}%

@{諦視王位，亦無純樂。純事令人，}@
%{仔细观察王位，也没有纯粹的快乐。纯粹侍奉人，}%

@{受極快樂；緣樂之後，受大艱難。}@
%{得到极大的快乐；缘于快乐之后，承受大困难。}%

@{其厚煖衣，宜冬猛寒；至於夏暑，}@
%{那些厚实的暖衣，适合冬天寒冷；至于夏天炎热，}%

@{更反為患。飢者得食，以為飽樂；}@
%{反而成为祸患。饥饿的人得到食物，以为饱足快乐；}%

@{有強逼者，必成大苦。如好藕池，}@
%{有强迫的人，必然成为大苦。如同美好的藕池，}%

@{中有宿虫；猶花樹林，師子滿中。}@
%{其中有宿虫；如同花树林，狮子充满其中。}%

@{金寶之舍，熾火充盈；王位猶此，}@
%{金宝的住宅，炽火充满；王位如同此，}%

@{用相請為。如魚吞餌，不覺強鈎；}@
%{用相邀请。如同鱼吞食饵，不觉强钩；}%

@{猶如以蜜，塗利劍刃。王者之位，}@
%{如同用蜜，涂抹利剑的刀刃。王者的地位，}%

@{如七寶械；觀雖悅目，身心甚苦。}@
%{如同七宝的刑具；看起来虽然悦目，身心却非常苦。}%

@{王者容飾，被服如天；乘國土車，}@
%{王者的装饰，衣服如同天上的神仙；乘坐国土的车，}%

@{雜糅牢固。王者重擔，重於大山；}@
%{混合牢固。王者的重担，重于大山；}%

@{苦如馬鬪，觀者悅之。水火風雹，}@
%{苦如同马斗，观看的人却感到快乐。水火风雹，}%

@{疾疫飢匱；偷盜劫賊，強陵敵國。}@
%{疾疫饥荒；偷盗强盗，欺凌敌国。}%

@{邊境屯守，侵奪損耗；是諸艱難，}@
%{边境屯守，侵夺损耗；这些困难，}%

@{獨切王心。晝夜懷憂，寢不安床；}@
%{独切王的心。昼夜怀着忧愁，寝不安床；}%

@{思設方便，為國除患。心普懷疑，}@
%{思考设置方便，为国家除患。心普遍怀疑，}%

@{不信臣民；如至毒鄉，從人得食。}@
%{不相信臣民；如同到达毒乡，从人那里得到食物。}%

@{假令王者，領無數城；其身所處，}@
%{假如王者，领有无数城池；其自身所处，}%

@{限居其一。寢一宮室，坐御一座；}@
%{限制居住其一。寝一宫室，坐御一座；}%

@{榮樂無幾，憂勞甚廣。衣蓋一形，}@
%{荣乐不多，忧劳甚广。衣服遮盖一种形体，}%

@{食充一軀；出行遊觀，限駕一乘。}@
%{食物充实一个躯体；出行游览，限制驾驭一乘。}%

@{王所飲食，蓋少少耳；其餘勞動，}@
%{您所饮食，不过少少而已；其余劳作，}%

@{以恣憍奢。王唯以一，自在為樂；}@
%{用来放纵骄奢。您只以一种，自在为乐；}%

@{是樂亦復，雜諸苦毒。猶如處在，}@
%{这种快乐也同样，夹杂着诸多的苦毒。如同处于，}%

@{劍所作車；處事不明，輙還傷己。}@
%{剑所制作的车；处理事情不明，很容易就会伤害自己。}%

@{譬如好舍，花飾其表；含毒虺蛇，}@
%{譬如美好的房屋，用花装饰其外表；含有毒蛇，}%

@{充滿其中。觀其悅目，觸甚毒害；}@
%{充满其中。看起来悦目，触碰却非常有害；}%

@{吾以是故，不甘王位。以是之故，}@
%{我因此，不甘心王位。因此，}%

@{不宜受是；生死難保，猶如幻化。}@
%{不适合接受这个；生死难以保全，如同幻化。}%

@{焉知宿對，馳騁何趣？吾以是故，}@
%{焉知宿命，奔驰向何处？我因此，}%

@{不受王諫。王云捨家，而非其時；}@
%{不接受您的劝谏。您说舍弃家，并非时候；}%

@{今且諦聽，吾當報王。恣意飲食，}@
%{如今请您仔细听，我将回答您。放纵饮食，}%

@{不懷顧慮；死命追人，若干方便。}@
%{不怀顾虑；死命追赶人，有许多方便。}%

@{甘蔗之種，號曰白淨；王當覺此，}@
%{甘蔗的种子，号称白净；您应当觉悟到，}%

@{是我父王。吾欲脫苦，故棄王位；}@
%{是我父亲。我想要摆脱痛苦，所以舍弃王位；}%

@{欲設方便，建立善法。無畏滅度，}@
%{想要设置方便，建立善法。无畏灭度，}%

@{永求第一；常與生老，病死長別。}@
%{永远追求第一；永远与生老，病死长别。}%

@{欲求甘露，可保之處；是故不甘，}@
%{想要寻求甘露，可以保全的地方；因此不甘心，}%

@{與諸欲會。猶如野畜，渴行求水；}@
%{与诸欲会合。如同野兽，口渴行走寻找水；}%

@{疲勞之渴，迷惑而走。卒遇獵者，}@
%{疲劳的口渴，迷惑而行走。终于遇到猎人，}%

@{截道逐射；不愍其渴，必得殺之。}@
%{截断道路追赶射杀；不怜悯它的口渴，必然得到杀害。}%

@{世間亦爾，飢渴情故；迷惑萬端，}@
%{世间也是如此，饥渴的缘故；迷惑万端，}%

@{不計死迮。快意自娛，暢情飲食；}@
%{不计死生。快意自娱，畅情饮食；}%

@{不慮成事，當有壞敗。老來屈己，}@
%{不考虑成事，必然有坏败。年老时屈己，}%

@{如強張弓；疾病傷人，深於馳箭。}@
%{如同强拉弓；疾病伤害人，深入于疾驰的箭。}%

@{死來逼命，如獵放圍；愚意所迷，}@
%{死亡逼近生命，如同猎人放围；愚蠢的心意所迷惑，}%

@{如何待時？若覺若眠，若晝若夜；}@
%{如何等待时机？无论觉悟还是睡眠，无论白天还是黑夜；}%

@{若水若陸，各各人人。命逝不還，}@
%{无论水里还是陆地，每个人都是如此。生命逝去不复返，}%

@{迅速甚駃；猶如水性，入摩竭口。}@
%{迅速如同駃駃；如同水性，进入摩竭鱼的嘴巴。}%

@{法之燈光，莊美之明；加精進意，}@
%{法的灯光，庄严美好的光明；加上精进的心意，}%

@{增益酥油。奉行善者，懷歡喜逝；}@
%{增益酥油。奉行善法的人，怀着欢喜而去；}%

@{命終資糧，已悉備足。如進好華，}@
