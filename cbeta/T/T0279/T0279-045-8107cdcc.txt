@{爾時，世尊為心王菩薩而說頌曰：}@
%{当时，释迦牟尼佛为了心王菩萨而宣说了以下偈颂：}%

@{不可言說不可說，充滿一切不可說，}@
%{不可言说不可说，充满一切不可说，}%

@{不可言說諸劫中，說不可說不可盡。}@
%{不可言说诸劫中，说不可说不可尽。}%

@{不可言說諸佛剎，皆悉碎末為微塵，}@
%{不可言说诸佛刹，都悉碎末为微尘，}%

@{一塵中剎不可說，如一一切皆如是。}@
%{一尘中刹不可说，如同一切皆如是。}%

@{此不可說諸佛剎，一念碎塵不可說，}@
%{此不可说诸佛刹，一念碎尘不可说，}%

@{念念所碎悉亦然，盡不可說劫恒爾。}@
%{念念所碎悉亦然，尽不可说劫恒尔。}%

@{此塵有剎不可說，此剎為塵說更難，}@
%{此尘有刹不可说，此刹为尘说更难，}%

@{以不可說算數法，不可說劫如是數。}@
%{以不可说算数法，不可说劫如是数。}%

@{以此諸塵數諸劫，一塵十萬不可說，}@
%{以此诸尘数诸劫，一尘十万不可说，}%

@{爾劫稱讚一普賢，無能盡其功德量。}@
%{尔劫称赞一普贤，无人能尽其功德量。}%

@{於一微細毛端處，有不可說諸普賢，}@
%{于一微细毛端处，有不可说诸普贤，}%

@{一切毛端悉亦爾，如是乃至遍法界。}@
%{一切毛端悉亦尔，如是乃至遍法界。}%

@{一毛端處所有剎，其數無量不可說，}@
%{一毛端处所有刹，其数无量不可说，}%

@{盡虛空量諸毛端，一一處剎悉如是。}@
%{尽虚空量诸毛端，一一处刹悉如是。}%

@{彼毛端處諸國土，無量種類差別住，}@
%{彼毛端处诸国土，无量种类差别住，}%

@{有不可說異類剎，有不可說同類剎。}@
%{有不可说异类刹，有不可说同类刹。}%

@{不可言說毛端處，皆有淨剎不可說，}@
%{不可言说毛端处，皆有净刹不可说，}%

@{種種莊嚴不可說，種種奇妙不可說。}@
%{种种庄严不可说，种种奇妙不可说。}%

@{於彼一一毛端處，演不可說諸佛名，}@
%{于彼一一毛端处，演不可说诸佛名，}%

@{一一名有諸如來，皆不可說不可說。}@
%{一一名有诸如来，皆不可说不可说。}%

@{一一諸佛於身上，現不可說諸毛孔，}@
%{一一诸佛于身上，现不可说诸毛孔，}%

@{於彼一一毛孔中，現眾色相不可說。}@
%{于彼一一毛孔中，现众色相不可说。}%

@{不可言說諸毛孔，咸放光明不可說，}@
%{不可言说诸毛孔，咸放光明不可说，}%

@{於彼一一光明中，悉現蓮華不可說。}@
%{于彼一一光明中，悉现莲花不可说。}%

@{於彼一一蓮華內，悉有眾葉不可說，}@
%{于彼一一莲花内，悉有众叶不可说，}%

@{不可說華眾葉中，各現色相不可說。}@
%{不可说华众叶中，各现色相不可说。}%

@{彼不可說諸色內，復現眾葉不可說，}@
%{彼不可说诸色内，复现众叶不可说，}%

@{葉中光明不可說，光中色相不可說。}@
%{叶中光明不可说，光中色相不可说。}%

@{此不可說色相中，一一現光不可說，}@
%{此不可说色相中，一一现光不可说，}%

@{光中現月不可說，月復現月不可說。}@

@{於彼一一光明內，復現於日不可說。}@
%{于彼一一光明内，复现于日不可说。}%

@{於不可說諸日中，一一現色不可說，}@
%{于不可说诸日中，一一现色不可说，}%

%{于彼一一诸色内，又现光明不可说。}%

%{于彼一一光明内，现不可说狮子座，}%

@{一一嚴具不可說，一一光明不可說。}@
%{一一严具不可说，一一光明不可说。}%

@{光中妙色不可說，色中淨光不可說，}@
%{光中妙色不可说，色中净光不可说，}%

@{於彼一一淨光內，復現種種妙光明。}@
%{于彼一一净光内，复现种种妙光明。}%

@{此光復現種種光，不可言說不可說，}@
%{此光复现种种光，不可言说不可说，}%

@{如是種種光明內，各現妙寶如須彌。}@
%{如是种种光明内，各现妙宝如须弥。}%

@{一一光中所現寶，不可言說不可說，}@

%{种种开演不可说，哀愍世间不可说。}%

@{不可言說一切劫，讚不可說諸功德，}@
%{不可言说一切劫，赞不可说诸功德，}%

@{不可說劫猶可盡，不可說德不可盡。}@
%{不可说劫犹可尽，不可说德不可尽。}%

@{不可言說諸如來，不可言說諸舌根，}@
%{不可言说诸如来，不可言说诸舌根，}%

@{歎佛不可言說德，不可說劫無能盡。}@
%{叹佛不可言说德，不可说劫无能尽。}%

@{十方所有諸眾生，一切同時成正覺，}@
%{十方所有诸众生，一切同时成正觉，}%

@{於中一佛普能現，不可言說一切身。}@
%{于中一佛普能现，不可言说一切身。}%

@{此不可說中一身，示現於頭不可說；}@
%{此不可说中一身，示现于头不可说；}%

@{此不可說中一頭，示現於舌不可說；}@
%{此不可说中一头，示现于舌不可说；}%

@{此不可說中一舌，示現於聲不可說；}@
%{此不可说中一舌，示现于声不可说；}%

@{此不可說中一聲，經於劫住不可說。}@
%{此不可说中一声，经于劫住不可说。}%

@{如一如是一切佛，如一如是一切身，}@
%{如一如是一切佛，如一如是一切身，}%

@{如一如是一切頭，如一如是一切舌，}@
%{如一如是一切头，如一如是一切舌，}%

@{如一如是一切聲，不可說劫恒讚佛，}@
%{如一如是一切声，不可说劫恒赞佛，}%

@{不可說劫猶可盡，歎佛功德無能盡。}@
%{不可说劫犹可尽，叹佛功德无能尽。}%

@{一微塵中能悉有，不可言說蓮華界，}@
%{一微尘中能悉有，不可言说莲花界，}%

@{一一蓮華世界中，賢首如來不可說。}@
%{一一莲花世界中，贤首如来不可说。}%

@{乃至法界悉周遍，其中所有諸微塵，}@
%{乃至法界悉周遍，其中所有诸微尘，}%

@{世界若成若住壞，其數無量不可說。}@
%{世界若成若住坏，其数无量不可说。}%

@{一微塵處無邊際，無量諸剎普來入，}@
%{一微尘处无边际，无量诸刹普来入，}%

@{十方差別不可說，剎海分布不可說。}@
%{十方差别不可说，刹海分布不可说。}%

@{一一剎中有如來，壽命劫數不可說，}@
%{一一刹中有如来，寿命劫数不可说，}%

@{諸佛所行不可說，甚深妙法不可說，}@
%{诸佛所行不可说，甚深妙法不可说，}%

@{神通大力不可說，無障礙智不可說，}@
%{神通大力不可说，无障碍智不可说，}%

@{入於毛孔不可說，毛孔因緣不可說，}@
%{入于毛孔不可说，毛孔因缘不可说，}%

@{成就十力不可說，覺悟菩提不可說，}@
%{成就十力不可说，觉悟菩提不可说，}%

@{入淨法界不可說，獲深智藏不可說。}@
%{入净法界不可说，获深智藏不可说。}%

@{種種數量不可說，如其一切悉了知；}@
%{种种数量不可说，如其一切悉了知；}%

@{種種形量不可說，於此靡不皆通達。}@
%{种种形量不可说，于此靡不皆通达。}%

@{種種三昧不可說，悉能經劫於中住，}@
%{种种三昧不可说，悉能经劫于中住，}%

@{於不可說諸佛所，所行清淨不可說。}@
%{于不可说诸佛所，所行清净不可说。}%

@{得不可說無礙心，往詣十方不可說，}@
%{得不可说无碍心，往诣十方不可说，}%

@{神力示現不可說，所行無際不可說，}@
%{神力示现不可说，所行无际不可说，}%

@{往詣眾剎不可說，了達諸佛不可說，}@
%{往诣众刹不可说，了达诸佛不可说，}%

@{精進勇猛不可說，智慧通達不可說。}@
%{精进勇猛不可说，智慧通达不可说。}%

@{於法非行非不行，入諸境界不可說，}@
%{于法非行非不行，入诸境界不可说，}%

@{不可稱說諸大劫，恒遊十方不可說。}@
%{不可称说诸大劫，恒游十方不可说。}%

@{方便智慧不可說，真實智慧不可說，}@
%{方便智慧不可说，真实智慧不可说，}%

@{神通智慧不可說，念念示現不可說。}@
%{神通智慧不可说，念念示现不可说。}%

@{於不可說諸佛法，一一了知不可說，}@
%{于不可说诸佛法，一一了知不可说，}%

@{能於一時證菩提，或種種時而證入。}@
%{能于一时证菩提，或种种时而证入。}%

@{毛端佛剎不可說，塵中佛剎不可說，}@
%{毛端佛刹不可说，尘中佛刹不可说，}%

@{如是佛剎皆往詣，見諸如來不可說。}@
%{如是佛刹皆往诣，见诸如来不可说。}%

@{通達一實不可說，善入佛種不可說，}@
%{通达一实不可说，善入佛种不可说，}%

@{諸佛國土不可說，悉能往詣成菩提。}@
%{诸佛国土不可说，悉能往诣成菩提。}%

@{國土眾生及諸佛，體性差別不可說，}@
%{国土众生及诸佛，体性差别不可说，}%

@{如是三世無有邊，菩薩一切皆明見。}@
