@{歸命十方佛，法身及報化，}@
%{向十方所有的佛，包括他们的法身、报身和化身，顶礼膜拜。}%
@{願共諸眾生，速成法界性。}@
%{愿与所有众生，一起快速成就法界的本性。}%

@{輪迴三惡道，法界理凝然，}@
%{轮回的三恶道，法界的道理凝然不动。}%
@{本來常清淨，諸相不能遷；}@
%{本来就常清净，一切相貌不能改变。}%
@{寂靜如虛空，處處悉周遍，}@
%{寂静如同虚空，无处不在，周遍一切。}%
@{體皆離彼此，非深復非淺。}@
%{法界本身没有彼此，既不深也不浅。}%

@{乳未轉變時，酥醍醐不見，}@
%{牛奶还没有变成酥油的时候，酥油和醍醐就看不见。}%
@{煩惱未伏除，法界無由顯；}@
%{烦恼没有被降伏的时候，法界就无法显现。}%
@{如酥處乳中，酥本妙光瑩，}@
%{就像酥油存在牛奶中，酥油本身的光芒奇妙而明亮。}%
@{法界煩惱覆，圓滿體清淨。}@
%{法界被烦恼遮蔽，其圆满的本性是清净的。}%

@{如燈被障礙，非能照餘物，}@
%{就像灯被障碍物遮挡，就无法照亮其他物体。}%
@{無明恒覆心，法界非明了；}@
%{无明一直遮蔽着我们的心，法界就无法明了。}%
@{如燈離障礙，處處物能照，}@
%{就像灯离开障碍物，就能照亮一切。}%
@{煩惱破壞時，真如恒顯現。}@
%{烦恼被破除的时候，真如法性就能恒常显现。}%

@{初中及最後，二障不能擾，}@
%{从开始到最后，两种障碍都不能扰乱它。}%
@{如淨瑠璃珠，恒時光照曜；}@
%{就像纯净的琉璃珠，恒常散发光芒。}%
@{光明物所障，被障明非見，}@
%{光明被物体遮挡，被遮挡的光明就看不见。}%
@{法界煩惱覆，真如理難顯。}@
%{法界被烦恼遮蔽，真如的道理就难以显现。}%

@{圓寂體光潔，輪迴不能染，}@
%{圆寂的法界光明洁净，轮回不能染污它。}%
@{勤求趣法界，輪迴非能揜。}@
%{勤奋追求法界，轮回就无法遮蔽它。}%

@{如米糠纏裹，穀體米非無，}@
%{就像米糠包裹着米粒，米粒本身并没有消失。}%
@{煩惱覆真如，真如煩惱有；}@
%{烦恼遮蔽着真如，真如也包含着烦恼。}%
@{如穀去其糠，米體自然見，}@
%{就像去掉米糠，米粒自然就显现了。}%
@{若離煩惱糠，法界理方顯。}@
%{如果离开烦恼的“糠”，法界的道理才能显现。}%

@{妄執有世間，芭蕉終無實，}@
%{错误执着于有世间，就像芭蕉，最终没有果实。}%
@{法界非世間，亦非虛妄見。}@
%{法界不是世间，也不是虚妄的见解。}%

@{如人飲甘露，熱惱悉皆除，}@
%{就像人喝了甘露，所有的热恼都消除了一样。}%
@{若證法界性，煩惱熱皆棄。}@
%{如果证得法界的本性，所有的烦恼热都会舍弃。}%

@{滅除煩惱焰，法界甘露現，}@
%{灭除烦恼的火焰，法界的甘露就显现了。}%
@{一切有情中，高下皆平等。}@
%{在所有有情众生中，高低贵贱都是平等的。}%

@{體實果不生，執種果非有，}@
%{法界本身是果实，不生不灭，执着于种子就是执着于果实，也是错误的。}%
@{智慧出生時，有為非法界。}@
%{智慧出生的时候，有为法就不是法界。}%

@{法界本無處，究竟方可證，}@
%{法界本来没有地方，只有最终才能证得。}%
@{清淨恒光潔，日月皆瑩淨。}@
%{清净恒常光明洁净，就像日月一样明亮。}%

@{法界無垢染，如龍夜雨塵，}@
%{法界没有污垢染污，就像龙在夜里下雨，尘埃很快就会消失。}%
@{況似羅睺面，光明恒燦然。}@
%{更像罗睺星的面容，光芒恒常灿烂。}%

@{譬如火浣布，處火能離染，}@
%{就像火浣布，放在火里就能去除污垢。}%
@{垢去布猶存，光明轉瑩淨。}@
%{污垢去除，布依然存在，光明更加明亮。}%

@{貪愛令心染，虛妄有輪迴，}@
%{贪爱使心染污，虚妄的轮回就产生了。}%
@{亦如火浣布，真空妄非有。}@
%{就像火浣布，真空也是虚妄的，不存在的。}%

@{三毒生死本，智慧火能燒，}@
%{三毒是生死轮回的根源，智慧的火能烧毁它们。}%
@{法界體常有，朗然恒照曜。}@
%{法界的本性恒常存在，光明朗然照耀。}%

@{煩惱染稱垢，世尊恒所宣，}@
%{烦恼染污被称为垢，世尊恒常宣说。}%
@{垢滅真如顯，如汲地中泉。}@
%{垢灭了，真如法性就显现了，就像从地里汲取泉水一样。}%

@{法界體無垢，根隨能覆藏，}@
%{法界的本性没有污垢，根尘能遮蔽它。}%
@{若除煩惱盡，瑩淨叵難量。}@
%{如果彻底去除烦恼，法界的光明洁净，难以衡量。}%

@{法界本無我，二形及女男，}@
%{法界本来没有自我，没有两种形体，没有男女之分。}%
@{體無虛妄執，何處更思惟？}@
%{法界没有虚妄的执着，还有什么地方需要思惟呢？}%

@{法界離憎愛，根塵境本無，}@
%{法界远离憎爱，根尘境界本来就没有。}%
@{虛妄執為因，差別從此生。}@
%{虚妄的执着是因，差别由此产生。}%

@{真空非苦惱，貪愛苦惱因，}@
%{真空不是苦恼，贪爱才是苦恼的根源。}%
@{耽染由妄想，三界乃輪迴。}@
%{执着于妄想，就会沉迷于三界，从而陷入轮回。}%

@{懷孕在於腹，嬰子未言見，}@
%{怀孕在肚子里面，婴儿还没有出生就看不见。}%
@{二障覆真如，法界不能證。}@
%{两种障碍遮蔽着真如，法界就无法证得。}%

@{種種生疑慮，見慢及恚癡，}@
%{种种产生疑虑，比如我慢、嗔恚、愚痴。}%
@{妄計有真實，真實計非有。}@
%{错误地认为有真实，真实却认为不存在。}%

@{兔角體非有，妄執令真實，}@
%{兔角本身不存在，妄执却使它变得真实。}%
@{法界離妄執，妄執真非有。}@
%{法界远离妄执，妄执的真实性不存在。}%

@{如色必破壞，微塵猶可知，}@
%{就像色法一定会破灭，即使微尘也能感知到。}%
@{法界非破壞，三時不能得。}@
%{法界不会破灭，在过去、现在、未来都无法获得。}%

@{有生還有滅，榮辱亦皆隨，}@
%{有生就有灭，荣辱也随之而来。}%
@{法界非生滅，云何言所知？}@
%{法界不生不灭，怎么可以说知道它呢？}%

@{兔角本非有，三世猶可思，}@
%{兔角本来不存在，在过去、现在、未来都可以思惟。}%
@{真空非兔角，思慮不能知。}@
%{真空不是兔角，思惟无法知道它。}%

@{真空稱善逝，色相悉皆亡，}@
%{真空被称为善逝，色相都消失了。}%
@{應化隨緣有，修因離執非。}@
%{应化随缘而存在，修行要远离执着。}%

@{圓通如日月，水現影皆同，}@
%{圆通如同日月，水中的倒影也一样。}%
@{色聲雙泯絕，差別云何有？}@
%{色声都消失了，还有什么差别呢？}%

%{过去、现在、未来都可以思惟，生缘的时候就决定了。}%

%{如果悟道了己身法，己身又哪里存在呢？}%

%{就像水处于热的地方，在热的地方觉悟到它不是热。}%

%{在寒冷的地方也是一样的道理，圆通都是如此。}%

%{心恒常被烦恼遮蔽，迷惑不解。}%

%{如果离开烦恼的缠绕，觉悟了，就发现它本来就不存在。}%

%{就像眼睛观察各种颜色，离开障碍物就能照耀。}%
@{真空理亦然，照曜離生滅。}@
%{真空的道理也是一样，照耀着一切，远离生灭。}%

%{耳朵识听到声音，远离妄念和分别。}%

%{法界的本性也是一样，分别和妄念都是虚妄的，不存在的。}%

%{鼻子能闻到各种香味，妄执的本性不存在。}%
@{色相二俱亡，真空亦如是。}@
%{色相都消失了，真空也是如此。}%

@{舌根自性空，味界恒遠離，}@
%{舌根的本性是空，味界恒常远离。}%
@{識空體亦然，法界理如是。}@
%{识空的本性也是如此，法界的道理也是如此。}%

@{身根自性淨，冷煖觸非有，}@
%{身根的本性是清净的，冷暖触觉不存在。}%
@{法界理亦然，觸處常遠離。}@
%{法界的道理也是一样，触处恒常远离。}%

@{意緣法稱最，自性恒遠離，}@
%{意根缘于法，被称为最胜义，其本性恒常远离。}%
@{諸法性本空，圓通理如是。}@
%{一切法的本性都是空，圆通的道理也是如此。}%

@{見聞及覺知，相應法亦空，}@
%{眼耳鼻舌身意六根，以及相应的法，都是空的。}%
@{了絕諸妄想，見聞理亦非。}@
%{了断一切妄想，见闻的道理也就不存在了。}%

@{根塵起妄執，清淨體源無，}@
%{根尘引起妄执，清净的本性本来就没有。}%
@{迷執有根塵，根塵理非有。}@
%{执迷于根尘，根尘的道理就不存在了。}%

%{世间法和出世间法，空性的本质没有区别。}%
@{我法由迷起，遍計自輪迴。}@
%{我法是由迷惑而产生的，遍计执着，从而陷入轮回。}%

%{法界的道理是清净的，贪嗔痴本来就没有。}%
@{迷悟從心起，三毒法假名。}@
%{迷悟是从心而起的，三毒法是假名。}%

%{执迷于自我，而受到束缚，了达假名的智慧。}%

%{菩提不是近也不是远，三世的道理不存在。}%

%{智慧生起，烦恼染污就消灭了，不要执着于妄念。}%

@{菩提非妄執，正證亦知非。}@
%{菩提不是妄执，正证也了知它不存在。}%

%{水和牛奶在同一个地方，鹅喝牛奶不会混杂。}%
@{生空煩惱離，二障亦非雜。}@
%{生空和烦恼分离，两种障碍也不会混杂。}%

@{妄執我非無，了達本非有，}@
%{妄执自我，认为它存在，了达之后就发现它本来就不存在。}%
@{涅盤清淨理，二我俱非立。}@
%{涅槃的清净道理，两个自我都不成立。}%

%{菩提难以妄执，真空不生不灭。}%

%{了达空的本性，两种相也不存在。}%

%{守护菩提的种子，菩提从此生起。}%

%{就像在黑夜，光明看不见。}%

@{月初光雖有，漸漸而增長，}@
%{月初的光芒虽然存在，但逐渐增长。}%

@{十五月圓滿，處處光皎潔，}@
%{十五月圆满，处处光明皎洁。}%
@{解脫顯法身，法身理無缺。}@
%{解脱显现法身，法身的道理没有缺漏。}%

%{染污的意念相对应，缠缚一起生灭。}%

%{解脱一切障碍，三世都悟到它不存在。}%

%{初大僧祇满，三檀普遍地修行。}%

%{断除分别的障碍，欢喜的智慧难以比拟。}%

%{身口意三业错误地犯了，要防范轻重不同的过失。}%
@{尸羅圓滿戒，離垢獨標名。}@
%{持戒圆满，远离污垢，独树一帜。}%

%{两种障碍恒常染污，用空慧的利刃去除。}%
@{發光能照曜，破滅漸無餘。}@
%{发出光芒照耀，逐渐破灭，没有剩余。}%

@{遠離根隨染，漸增焰慧威，}@
%{远离根尘的染污，逐渐增长火焰般的智慧威德。}%
@{菩提稱最勝，燒照轉光輝。}@
%{菩提被称为最胜，燃烧照耀，光辉转盛。}%

@{合令無所礙，難勝事恒時。}@
%{合起来令其无所障碍，难以胜过的事情恒常存在。}%

@{甚深稱最勝，般若現於前。}@
%{甚深被称为最胜，般若智慧显现在眼前。}%

%{世俗二乘的修行，久修道业已经明了。}%
@{相無功用滿，最後稱遠行。}@

%{庄严都普遍存在，功德实在难以思惟。}%

@{十力兼無畏，三身四智圓，}@
%{十力兼具无畏，三身四智圆满。}%
@{六通恒自在，應物化機緣。}@
%{六通恒常自在，应物变化，机缘成熟。}%

@{照曜如圓月，恒時焰熾燃，}@
%{照耀如同圆月，恒常火焰炽燃。}%
@{十方無不遍，燦爛轉光鮮。}@
%{十方无处不遍，灿烂的光芒更加鲜明。}%

@{永絕緣生染，恒時處涅盤，}@
%{永远断绝缘生染污，恒常处于涅槃。}%
@{菩提稱最勝，化益物情歡。}@
%{菩提被称为最胜，利益化度众生，令其欢喜。}%

@{智用深如海，隨機現應身，}@
%{智慧运用如同大海般深广，随缘显现应化身。}%
@{水清來月影，處處度迷津。}@
%{水清澈，映照出月亮的影子，处处度化迷途的众生。}%

@{況似頗胝寶，隨緣現影同，}@
%{更像颇胝宝，随缘显现影子，都一样。}%
@{物情根有感，周普事無窮。}@
%{众生根有感，周遍一切，功德无穷。}%

@{餓鬼恒飢渴，不能見水泉，}@
%{饿鬼恒常饥渴，不能看见水泉。}%
@{眾生無少信，宿業自縈纏。}@
%{众生没有丝毫信心，宿业自己缠绕着。}%

@{化現身諸相，光明皆燦然，}@
%{化现身各种相貌，光明都灿烂。}%
@{佛雖恒在世，不覩宿無緣。}@
%{佛虽然恒常在世间，但由于宿业无缘，就看不到。}%

@{曉了塵沙界，根隨染久無，}@
%{明白了尘沙界，根尘的染污已经没有了。}%
@{二空殊勝智，妙用化童愚。}@
%{两种空性的殊胜智慧，妙用化度愚痴的人。}%

@{清淨絕諸垢，自他受用身，}@
%{清净，远离一切污垢，自他都能受用这个身。}%
@{恒居色究竟，利益五乘人。}@
%{恒常住于色究竟，利益五乘的众生。}%

@{救護眾生苦，俱胝壽命長，}@
%{救护众生的苦难，拥有无数的寿命。}%
@{二嚴無有盡，功德叵難量。}@
%{两种庄严无有穷尽，功德难以衡量。}%

@{佛演一乘法，隨機悟淺深，}@
%{佛演说一乘法，随缘悟道，浅深不一。}%
@{蓮花無垢染，玉本絕瑕侵。}@
%{莲花没有污垢染污，玉本来就没有任何瑕疵。}%

@{少分而稱讚，廣宣理趣玄，}@
%{略微称赞一下，广泛宣扬法理的奥妙。}%
@{願將諸功德，普利施人天。}@
%{愿将这些功德，普利施于人天。}%
