@{(一二九四)}@
%{(一二九四) }%

@{如是我聞：}@
%{我这样听说：}%

@{一時，佛住舍衛國祇樹給孤獨園。}@
%{有一次，佛住在舍卫国的祇树给孤独园。}%

@{時，有天子容色絕妙，於後夜時來詣佛所，稽首佛足，退坐一面。其身光明遍照祇樹給孤獨園。}@
%{这时，有一个相貌极其英俊的天子，在后半夜来到佛陀那里，向佛陀的双脚顶礼，然后退到一边坐下。他的身体放出光芒，照亮了整个祇树给孤独园。}%

@{時，彼天子說偈問佛：}@
%{这时，这位天子用偈颂问佛陀：}%

@{大力自在樂，所求無不得，}@
%{拥有强大的力量，自由自在，快乐无忧，想要什么就能得到什么，}%

@{何復勝於彼，一切所欲備？}@
%{还有什么能比这更胜一筹呢？一切愿望都能满足。}%

@{爾時，世尊說偈答言：}@
%{这时，世尊用偈颂回答说：}%

@{大力自在樂，彼則無所求，}@
%{拥有强大的力量，自由自在，快乐无忧，其实他们什么都不需要追求，}%

@{若有求欲者，是苦非為樂，}@
%{如果还有追求欲望的人，那是痛苦，不是快乐，}%

@{於求已過去，是則樂於彼。}@
%{对于已经过去的追求，才是真正的快乐。}%

@{時，彼天子復說偈言：}@
%{这时，这位天子又用偈颂说：}%

@{久見婆羅門，逮得般涅槃，}@
%{我久已看到婆罗门，他们已经证得涅槃，}%

@{一切怖已過，永超世恩愛。}@
%{一切恐惧都已消除，永远超越了世俗的恩爱。}%

@{於是，天子聞佛所說，歡喜隨喜，稽首佛足，即沒不現。}@
%{于是，天子听到佛陀所说，非常欢喜，顶礼佛陀的双脚，便消失不见了。}%

%{佛陀对天子说：“这不好，不善。”}%

%{这时，这位天子就用偈颂说：}%

@{人生子為樂，世間有子歡，}@

%{这时，世尊用偈颂回答说：}%

@{當知恒無常，純空陰非子，}@
%{要知道，世间万物都是无常的，都是虚幻的，空无一物的，哪有什么儿子呢？}%

@{生子常得苦，愚者說言樂。}@
%{生孩子总是伴随着痛苦，只有愚蠢的人才会说生孩子是快乐。}%

@{是故我說言，生子非為善，}@
%{所以我说，生孩子不是善事，}%

@{非善為善像，念像不可念。}@
%{不善的事，却像善事一样，执着于这种假象，是不可取的。}%

@{實苦貌似樂，放逸所踐蹈。}@
%{实际上是痛苦，却像快乐一样，被人们放纵地践踏着。}%

@{時，彼天子復說偈言：}@
%{这时，这位天子又用偈颂说：}%

@{何戒何威儀？何得何為業？}@

@{慧者云何住？云何往生天？}@
%{智慧的人怎样住世？怎样往生天界？}%

@{爾時，世尊說偈答言：}@
%{这时，世尊就用偈颂来回答：}%
@{遠離於殺生，持戒自防樂，}@
%{远离杀生，持戒自防快乐，}%
@{害心不加生，是則生天路。}@
%{不怀害心，这就是通往天界的道路。}%
@{遠離不與取，與取心欣樂，}@
%{远离不与取，不与取的心欣乐，}%
@{斷除賊盜心，是則生天路。}@
%{断除偷盗之心，这就是通往天界的道路。}%
@{不行他所受，遠離於邪婬，}@
%{不侵犯他人的配偶，远离邪淫，}%
@{自受知止足，是則生天路。}@
%{知足常乐，这就是通往天界的道路。}%
@{自為己及他，為財及戲笑，}@
%{为了自己或他人，为了钱财或玩乐，}%
@{妄語而不為，是則生天路。}@
%{不妄语，这就是通往天界的道路。}%
@{斷除於兩舌，不離他親友，}@
%{断除两舌，不离弃亲朋好友，}%
@{常念和彼此，是則生天路。}@
%{经常念念不忘和睦相处，这就是通往天界的道路。}%
@{遠離不愛言，軟語不傷人，}@
%{远离不爱语，用柔软的语言不伤害别人，}%
@{常說淳美言，是則生天路。}@
%{经常说淳美的话语，这就是通往天界的道路。}%
@{不為不誠說，無義不饒益，}@
%{不说不诚实的话，不说没有意义、没有利益的话，}%
@{常順於法言，是則生天路。}@
%{经常说符合法义的话，这就是通往天界的道路。}%
@{聚落若空地，見利言我有，}@
%{就像看到空地一样，看到利益就说“这是我的”，}%
@{不行此貪想，是則生天路。}@
%{不存这种贪婪的念头，这就是通往天界的道路。}%

@{等受堅固持，是則生天路。}@
%{平等地受持，坚定不移，这就是通往天界的道路。}%

%{由于母亲的饮食等等，滋养着胎儿的身体。}%

%{尼乾大师的弟子们，虚妄地自吹自擂，}%

@{精勤棄闇冥，守護於遠離，}@

%{如果所有颜色，在这里和那里，}%

@{若諸所有色，於此及與彼，}@
%{如果所有颜色，在这里和那里，}%

@{或復虛空中，各別光照耀。}@
%{或者在虚空中，各自发出光芒。}%

@{當知彼一切，不離魔魔縛，}@
%{要知道，这一切都离不开魔的束缚，}%

@{猶如垂鈎餌，鈎釣於遊魚。}@
%{就像垂下鱼钩，钓着游鱼一样。}%

@{時，諸天子復作是念：奇哉！沙門瞿曇！神力大德，能見天魔波旬，而我等不見，我等當復各各說偈讚歎沙门瞿曇。即說偈言：}@
%{这时，那些天子又想着：奇怪啊！沙门瞿昙！神力大德，能看到天魔波旬，而我们却看不到，我们应该各自说一首偈子赞叹沙门瞿昙。就说了一首偈子：}%

@{斷除於一切，有身愛貪想，}@
%{断除一切对身体的贪爱和妄想，}%

@{令此善護者，除一切妄語。}@
%{让这个善护者，去除一切妄语。}%

@{若欲斷欲愛，應供養大師。}@
%{如果想要断除对欲爱的执着，应该供养大师。}%

@{斷除三有愛，破壞於妄語。}@
%{断除对三界的执着，破坏妄语。}%

@{已斷於見貪，應供養大師。}@
%{已经断除了对见贪的执着，应该供养大师。}%

@{王舍城第一，名毘富羅山。}@
%{王舍城第一高峰，名叫毘富羅山。}%

@{雪山諸山最，金翅鳥中名。}@
%{雪山众山之最，金翅鸟中最有名。}%

@{八方及上下，一切眾生界，}@
%{八方上下，一切众生界，}%

@{於諸天人中，等正覺最上。}@
%{在所有天人和人中，等正觉最尊贵。}%

@{時，諸天子說偈讚佛已，聞佛所說，歡喜隨喜，稽首佛足，即沒不現。}@
%{这时，那些天子说完赞叹佛的偈子，听到佛所说的话，欢喜随喜，稽首佛足，就消失不见了。}%

@{時，尊者阿那律陀於摩竭提國人間遊行，到畢陵伽鬼子母住處宿。時，尊者阿那律陀夜後分時，端身正坐，誦憂陀那、波羅延那、見真諦、諸上座所說偈、比丘尼所說偈、尸路偈、義品、牟尼偈、修多羅，悉皆廣誦。}@
%{当时，尊者阿那律陀在摩揭陀国人间游行，来到毕陵伽鬼子母的住处住宿。晚上，尊者阿那律陀端正地坐着，诵读忧陀那、波罗延那、见真谛、诸上座所说偈、比丘尼所说偈、尸路偈、义品、牟尼偈、修多罗，都广泛地诵读。}%

@{爾時，畢陵伽鬼子夜啼，畢陵伽鬼子母為其子說偈呵止言：}@
%{这时，毕陵伽鬼子夜里哭闹，毕陵伽鬼子母就对自己的孩子说了一首偈子来制止他：}%

@{畢陵伽鬼子，汝今莫得啼，}@
%{毕陵伽鬼子，你快别哭了，}%
@{當聽彼比丘，誦習法句偈。}@
%{应该听那个比丘诵读法句偈。}%

@{若知法句者，能自護持戒，}@
%{如果知道法句的人，能够自己持戒，}%
@{遠離於殺生，實言不妄語，}@
%{远离杀生，诚实不妄语，}%
@{能自捨非義，解脫鬼神道。}@
%{能够舍弃不义的行为，就能解脱鬼神道。}%

@{畢陵伽鬼子母說是偈時，畢陵伽鬼子啼聲即止。}@
%{毕陵伽鬼子母说完这首偈子后，毕陵伽鬼子就停止了哭泣。}%

@{如是我聞：}@
%{我这样听说的：}%

@{一時，佛在摩竭提國人間遊行，與大眾俱，到富那婆藪鬼子母住處宿。}@
%{有一次，佛陀在摩揭陀国人间游行，和大众一起，来到富那婆藪鬼子母的住处住宿。}%

@{爾時，世尊為諸比丘說四聖諦相應法，所謂苦聖諦、苦集聖諦、苦滅聖諦、苦滅道跡聖諦。}@
%{这时，佛陀为诸比丘讲解四圣谛相应法，即苦圣谛、苦集圣谛、苦灭圣谛、苦灭道迹圣谛。}%
@{爾時，富那婆藪鬼母，兒富那婆藪及鬼女欝多羅，二鬼小兒夜啼。}@
%{当时，富那婆藪鬼母，儿子富那婆藪和鬼女欝多羅，两个小鬼孩子夜里哭闹。}%
@{時，富那婆藪鬼母教其男女故，而說偈言：}@
%{这时，富那婆藪鬼母教导她的孩子，就吟诵了一首偈语：}%

@{汝富那婆藪，欝多羅莫啼，}@
%{你富那婆藪，欝多羅不要哭，}%
@{令我得聽聞，如來所說法。}@
%{让我能够听到，如来所说之法。}%
@{非父母能令，其子解脫苦，}@
%{不是父母能够让，他们的孩子解脱苦难，}%
@{聞如來說法，其苦得解脫。}@
%{而是听到如来所说之法，才能解脱苦难。}%

@{世人隨愛欲，為眾苦所迫，}@
%{世人追随欲望，被各种痛苦所逼迫，}%
@{如來為說法，令破壞生死。}@
%{如来为他们说法，令他们破除生死轮回。}%
@{我今欲聞法，汝等當默然，}@
%{我现在想要听法，你们就安静下来，}%
@{時富那婆藪，鬼女欝多羅。}@
%{这时富那婆藪，鬼女欝多羅。}%

@{悉受其母語，默然而靜聽，}@
%{都听从了母亲的话，安静地听着，}%
@{語母言善哉，我亦樂聞法。}@
%{并对母亲说，很好，我也很乐意听法。}%
@{此正覺世尊，於摩竭勝山，}@
%{这正觉世尊，在摩竭胜山，}%
@{為諸眾生類，演說脫苦法。}@
%{为一切众生，宣说解脱苦难的法。}%

@{說苦及苦因，苦滅滅苦道，}@
%{说苦和苦的根源，说苦的灭除和灭除苦的道路，}%
@{從此四聖諦，安隱趣涅槃。}@
%{从这四圣谛，安稳地走向涅槃。}%
@{母今但善聽，世尊所說法。}@
%{母亲现在就好好听，世尊所说之法。}%

@{時，富那婆藪鬼母即說偈言：}@
%{这时，富那婆藪鬼母就吟诵了一首偈语：}%

@{奇哉智慧子，善能隨我心，}@
%{真是奇哉，智慧之子，能够随顺我的心意，}%
@{汝富那婆藪，善歎佛導師。}@
%{你富那婆藪，能够赞叹佛陀这位导师。}%
@{汝富那婆藪，及汝欝多羅，}@
%{你富那婆藪，和你的欝多羅，}%
@{當生隨喜心，我已見聖諦。}@
%{应该生起随喜之心，我已经见到了圣谛。}%

@{時，富那婆藪鬼母說是偈時，鬼子男女隨喜默然。}@
%{这时，富那婆藪鬼母吟诵完这首偈语，鬼子男女都随喜地默然。}%

@{(一三二三)}@

@{如是我聞：}@
%{如是我所闻：}%

@{一時，佛在摩竭提國人間遊行，與諸大眾至摩尼遮羅鬼住處夜宿。}@
%{有一次，佛陀在摩竭提国人间游行，和许多大众一起，来到摩尼遮羅鬼住处过夜。}%

@{爾時，摩尼遮羅鬼會諸鬼神，集在一處。}@
%{当时，摩尼遮羅鬼召集了所有的鬼神，聚集在一起。}%

@{時，有一女人，持香花鬘飾、飲食，至彼摩尼遮羅鬼神住處。}@
%{这时，有一个女人，拿着香花、花鬘、装饰品和食物，来到摩尼遮羅鬼神住处。}%
@{彼女人遙見世尊在摩尼遮羅鬼神住處坐，見已，作是念：我今現見摩尼遮羅鬼神。}@
%{这个女人远远地看见世尊坐在摩尼遮羅鬼神住处，看见了之后，心里想：我现在看见了摩尼遮羅鬼神。}%
@{即說偈言：}@
%{就吟诵了一首偈语：}%

@{善哉摩尼遮，住摩伽陀國，}@
%{真是太好了，摩尼遮羅鬼，住在摩伽陀国，}%
@{摩伽陀國人，所求悉如願。}@
%{摩伽陀国的人，所求都能如愿。}%
@{云何於此世，常得安樂住？}@
%{怎样才能在这个世间，永远安乐地生活？}%
@{後世復云何，而得生天樂？}@
%{来世又怎样，才能生到天界享受快乐？}%

@{爾時，世尊說偈答言：}@
%{这时，世尊吟诵一首偈语回答说：}%

@{莫放逸慢恣，用摩尼鬼為，}@
%{不要放逸懈怠，以摩尼遮羅鬼为依靠，}%
@{若自修所作，能得生天樂。}@
%{如果自己修行，就能生到天界享受快乐。}%

@{時，彼女人作是念：此非摩尼遮羅鬼，是沙門瞿曇。}@
%{这时，这个女人心里想：这不是摩尼遮羅鬼，而是沙门瞿曇。}%
@{如是知已，即以香花鬘飾供養世尊，稽首禮足，退坐一面。}@
%{知道了之后，就用香花、花鬘、装饰品供养世尊，恭敬地顶礼世尊的双足，退到一边坐下。}%
@{而說偈言：}@
%{并吟诵了一首偈语：}%

@{何道趣安樂？當修何等行，}@
%{什么道路可以通向安乐？应该修持什么样的行为，}%
@{此世常安隱，後世生天樂？}@
%{才能在这个世间永远安稳，来世生到天界享受快乐？}%

@{爾時，世尊說偈答言：}@
%{这时，世尊吟诵一首偈语回答说：}%

@{布施善調心，樂執護諸根，}@
%{布施，善调心，欢喜持守六根，}%
@{正見修賢行，親近於沙門。}@
%{正见，修持善行，亲近沙门。}%
@{以正命自活，他世生天樂，}@
%{以正命自活，来世生到天界享受快乐，}%
@{何用三十三，諸天之苦網。}@
%{何必去三十三天，那充满痛苦的天界网罗。}%

@{但當一其心，斷除於愛欲，}@
%{只要一心一意，断除对欲望的执着，}%
@{我當說離垢，甘露法善聽。}@
%{我就会宣说远离垢染的，甘露般美妙的法，你好好听。}%

@{時，彼女人聞世尊說法，示、教、照、喜。}@
%{这时，这个女人听到世尊说法，心里生起示、教、照、喜四种功德。}%
@{如佛常法，謂布施、持戒，生天之福，欲味欲患煩惱，清淨、出要、遠離，功德福利，次第演說清淨佛法。}@
%{正如佛陀一贯的做法，说布施、持戒，可以生到天界享受福报，说欲望的滋味、欲望的痛苦、烦恼的根源，说清净、出离、远离，说功德、利益，依次宣说清净的佛法。}%
@{譬如鮮淨白㲲，易染其色。}@
%{就像洁白无瑕的棉花，很容易染上颜色。}%
@{時，彼女人亦復如是。}@
%{这时，这个女人也像这样。}%
@{即於坐上，於四聖諦得平等觀苦、集、滅、道。}@
%{就在座位上，对四圣谛，能够平等地观照苦、集、灭、道。}%

@{時，彼女人見法、得法、知法、入法，度諸疑惑，不由於他，於正法、律得無所畏。}@
%{这时，这个女人见到法、得到法、知道法、进入法，解除了所有的疑惑，不用依靠他人，对正法、律法，不再有任何恐惧。}%
@{即從座起，整衣服，合掌白佛：已度，世尊！已度，善逝！我從今日，盡壽命，歸佛、歸法、歸比丘僧。}@
%{就从座位上站起来，整理衣服，合掌向佛陀说：我已经解脱了，世尊！我已经解脱了，善逝！我从今天起，直到生命结束，皈依佛、皈依法、皈依比丘僧。}%

@{時，彼女人聞佛所說，歡喜隨喜，禮佛而去。}@
%{这时，这个女人听到佛陀所说，欢喜随喜，礼拜佛陀之后就离开了。}%

@{(一三二四)}@

@{如是我聞：}@
%{如是我所闻：}%

@{一時，佛在摩竭提國人間遊行，到針毛鬼住處夜宿。}@
%{有一次，佛陀在摩竭提国人间游行，来到针毛鬼住处过夜。}%
@{爾時，針毛鬼會諸鬼神，集在一處。}@
%{当时，针毛鬼召集了所有的鬼神，聚集在一起。}%

@{時，有炎鬼見世尊在針毛鬼住處夜宿。}@
%{这时，有一个炎鬼看见世尊在针毛鬼住处过夜。}%
@{見已，往詣針毛鬼所，語針毛鬼言：聚落主！汝今大得善利，今如來、應、等正覺於汝室宿。}@
%{看见了之后，就走到针毛鬼那里，对针毛鬼说：聚落主！你现在真是太幸运了，现在如来、应、等正觉，在你家里过夜。}%

@{針毛鬼言：今當試看，為是如來、為非？}@
%{针毛鬼说：我现在就要试一试，看看是不是如来，还是不是如来？}%

@{時，針毛鬼與諸鬼神集會已，還歸自舍，束身衝佛。}@
%{这时，针毛鬼和所有的鬼神聚集在一起，就回到自己的住处，裹紧身体冲向佛陀。}%
@{爾時，世尊却身避之。}@
%{这时，世尊退身避开他。}%
@{如是再三，束身衝佛，佛亦再三却身避之。}@
%{就这样，针毛鬼一再裹紧身体冲向佛陀，佛陀也一再退身避开他。}%
@{爾時，針毛鬼言：沙門怖耶？}@
%{当时，针毛鬼说：“沙门害怕吗？”}%

@{佛言：聚落主！我不怖也，但汝觸惡。}@
%{佛说：“聚落主！我不害怕，只是你触犯了恶事。”}%

@{針毛鬼言：今有所問，當為我說，能令我喜者善，不能令我喜者，當壞汝心，裂汝胸，令汝熱血從其面出，捉汝兩手擲恒水彼岸。}@
%{针毛鬼说：“我现在有个问题要问你，你如果能让我高兴，就算了，如果不能让我高兴，我就打碎你的心，撕裂你的胸膛，让你热血从脸上流出来，抓住你的两只手把你扔到恒河彼岸。”}%

@{佛告針毛鬼：聚落主！我不見諸天、魔、梵、沙門、婆羅門、天神、世人能壞如來、應、等正覺心者，能裂其胸者，能令熱血從面出者，執其兩臂擲著恒水彼岸者。汝今但問，當為汝說。令汝歡喜。}@
%{佛对针毛鬼说：“聚落主！我从未见过任何天、魔、梵、沙门、婆罗门、天神、世人能打碎如来、应、等正觉的心，能撕裂他们的胸膛，能让他们热血从脸上流出来，能抓住他们的手臂把他们扔到恒河彼岸。你现在就问吧，我会告诉你，让你高兴。”}%

@{時，針毛鬼說偈問佛：}@
%{当时，针毛鬼便用偈子问佛：}%

@{一切貪恚心，以何為其因，}@
%{“一切贪嗔之心，以什么为其因？”}%

@{不樂身毛竪，恐怖從何起？}@
%{“为什么不喜欢身体毛发竖立，恐惧从何而来？”}%

@{意念諸覺想，為從何所起，}@
%{“念头和觉想，是从哪里产生的？”}%

@{猶如新生兒，依倚於乳母。}@
%{“就像新生儿，依赖于乳母一样。”}%

@{爾時，世尊說偈答言：}@
%{当时，世尊便用偈子回答说：}%

@{愛生自身長，如尼拘律樹，}@
%{“爱生自身，不断增长，就像尼拘律树一样。”}%

@{展轉相拘引，如籐綿叢林，}@
%{“互相牵引，就像藤蔓丛林一样。”}%

@{若知彼所因，當令鬼覺悟，}@
%{“如果你知道这些的根源，就能让鬼魂觉悟。”}%

@{度生死海流，不復重增有。}@
%{“渡过生死之海，不再重新增加存在。”}%

@{爾時，針毛鬼聞世尊說偈，心得歡喜，向佛悔過，受持三歸。}@
%{当时，针毛鬼听到世尊的偈子，内心感到欢喜，向佛忏悔，受持三归。”}%

@{佛說此經已，針毛鬼聞佛所說，歡喜奉行。}@
%{佛说完了这部经，针毛鬼听到佛所说，欢喜地奉行。”}%
