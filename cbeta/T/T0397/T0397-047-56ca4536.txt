@{爾時，於彼諸宮殿中所居大眾，男夫婦女、眷屬大小，悉作一朋，復作是言：大王！今可將諸眷屬速詣彼所，我等今當守護此諸城邑宮殿。}@

@{心平於一切，是故禮佛足。}@
%{心平气和地对待一切，所以礼拜佛的双足。}%

@{願護此一法，勿使魔更來，}@
%{希望您护持这一法，不要让魔再来了，}%

@{不令彼得勝，作此惡惱害。}@
%{不要让它得胜，做这些恶事来伤害众生。}%

@{一切惡瞋怒，及以最極惡，}@
%{所有的恶、瞋怒，以及最极端的恶，}%

@{實語願說呪，降伏諸魔軍。}@
%{希望您说出真实的咒语，降伏所有的魔军。}%

@{如人身無病，不求諸醫師，}@
%{就像人身没有病，就不需要医生一样，}%

@{如是阿修羅，無事不想佛。}@
%{像我们这些阿修罗，没事就想佛。}%

@{我今甚歡喜，來詣於佛所，}@
%{我现在非常高兴，来拜见佛，}%

@{願說清淨法，令到菩提道。}@
%{希望您说清净法，让我到达菩提道。}%

@{顯示諸賢聖，寂定如虛空，}@
%{展现出贤圣们，寂静不动，像虚空一样，}%

@{遠離於諸惡，及離我我所。}@
%{远离一切恶事，以及远离自我和我的执着。}%

@{速知於實際，能除諸煩惱，}@
%{快速地了解到真实，能消除所有的烦恼，}%

@{速斷一切縛，得知最上道。}@
%{快速地断除所有的束缚，了解到最高的道。}%

@{爾時，波羅陀阿修羅王及與眷屬，頂禮佛足右遶三匝，持眾雜色種種寶樹，狀如波吒羅七寶之樹置於佛後。種種寶葉華果、金縷真珠瓔珞、天生寶鬘、天瓔珞衣服、指印環玔、寶蓋幢幡、手瓔珞、脚瓔珞、臂瓔珞，寶莊嚴具於佛頭上空中垂下，乃至種種歌舞作樂而供養佛。一心合掌，說偈讚曰：}@
%{这时，波罗陀阿修罗王和他的眷属，顶礼佛的双足，绕着佛走了三圈，拿着各种颜色的宝树，形状像波吒罗七宝树，放在佛的后面。各种各样的宝叶、花果、金线、珍珠、璎珞、天生宝鬘、天上的璎珞衣服、指印、环玔、宝盖、幢幡、手璎珞、脚璎珞、臂璎珞，宝饰装饰在佛的头上，从空中垂下来，直到各种各样的歌舞演奏来供养佛。一心合掌，诵念赞颂的偈语说：}%

@{眾生常為彼，煩惱火所燒，}@
%{众生经常被烦恼的火烧着，}%

@{求樂恒不得，不遇施樂者。}@
%{寻求快乐总是得不到，没有遇到能给予快乐的人。}%

@{一切住惡道，惟佛眾生藥，}@
%{所有的人都住在恶道，只有佛是众生的良药，}%

@{安住解脫道，能救一切苦。}@
%{安住在解脱道上，能救一切苦。}%

@{惟佛商人主，三有中更無，}@
%{只有佛是商人中的主人，三界中再也没有了，}%

@{愍進智慧水，眾德滿如海。}@
%{您怜悯众生，用智慧之水，所有功德像大海一样圆满。}%

@{為盲失道者，安住於正路，}@
%{您为那些失道的人，安住在正道上，}%

@{導引諸眾生，住勝涅槃道。}@
%{引导所有众生，住在最好的涅槃道上。}%

@{凡夫飢無厭，唯佛能充飽；}@
%{凡夫总是感到饥饿，没有满足的时候，只有佛能让他们饱足；}%

@{常為煩惱溺，唯佛能救拔。}@
%{经常被烦恼淹没，只有佛能救拔他们。}%

@{遭溺之病苦，苦逼失其念，}@
%{遇到溺水而痛苦，苦难逼迫，失去了理智，}%

@{如此苦苦者，導師為親救。}@
%{像这样痛苦的人，导师像亲人一样来救他们。}%

@{惡心眾生等，龍鬼諸羅剎，}@
%{那些心肠恶毒的众生，龙、鬼、罗刹，}%

@{見佛得正念，安住大悲心。}@
%{见到佛，就得到了正念，安住在伟大的悲心之中。}%

@{唯佛於三界，能作救護者，}@
%{只有佛在三界中，能够作为救护者，}%

@{我等已孤獨，悉為眾苦溺。}@
%{我们已经孤独，都被各种苦难淹没。}%

@{我等皆一心，一切樂法住，}@
%{我们都一心一意，住在一切快乐的法中，}%

@{願說最上義，令我得菩提。}@
%{希望您说最高的法义，让我们得到菩提。}%

@{以得於佛眼，最上諸佛智，}@
%{得到佛眼，得到最高佛智，}%

@{速伏諸魔怨，願降正法雨。}@
%{快速地降伏所有的魔怨，希望您降下正法的雨露。}%


@{爾時，跋持毘盧遮那阿修羅王將諸眷屬，頂禮佛足右遶三匝，於佛右左積閻浮檀金；復以種種寶、種種華、種種末香、寶蓋、幢幡，及以金縷、真珠、瓔珞、歌舞作樂，合掌供養，說偈讚曰：}@
%{当时，跋持毘卢遮那阿修罗王带着他的眷属，顶礼佛陀的双足，绕佛三圈，在佛陀的左右堆放了阎浮檀金；又用各种宝物、各种鲜花、各种香末、宝盖、幢幡，以及金线、珍珠、璎珞、歌舞奏乐，合掌供养，说偈赞曰：}%

@{福田福德水，福種福德牙，}@
%{福田是福德之水，福种是福德之牙，}%

@{福樹福德枝，福條福德葉，}@
%{福树是福德之枝，福条是福德之叶，}%

@{福華福味果，最上福味漿。}@
%{福花是福德之果，最上等的福味是福浆。}%

@{福色福陰影，福德子成就，}@
%{福色是福德之阴影，福德之子成就，}%

@{佛福善堅固，福勇能伏他，}@
%{佛陀的福德是善的坚固，福德的勇气能够降伏他人，}%

@{福色勇健士，福德不動山，}@
%{福色是勇健的勇士，福德是岿然不动的山，}%

@{福華覆蓋土，福藥所依身。}@
%{福花覆盖大地，福药是依身之物。}%

@{福山最上頂，放福智慧水，}@
%{福山是最高的山顶，放出福德的智慧之水，}%

@{福德海甚深，福德眾生依，}@
%{福德之海深不可测，福德是众生依靠的根本，}%

@{妙福眾寶磺，福味如巨海。}@
%{妙福是众宝的泉源，福味如浩瀚的大海。}%

@{福器盛甘露，福願皆盛滿，}@
%{福器盛满了甘露，福愿都得到了满足，}%

@{福商恒將向，無漏寶國土。}@
%{福商永远朝着无漏的宝国土前进。}%

@{我等眾福盡，是故歸依佛，}@
%{我们所有的福德都来自于佛陀，所以我们皈依佛陀，}%

@{福德賢寶瓶，願愍施我等。}@
%{愿佛陀慈悲怜悯，施予我们福德。}%

@{爾時，復有阿修羅仙，名一切菩提鬘，具有大福大威德大智慧大苦行，以菩提心而作莊嚴，得五神通離諸塵垢，安住成就一切眾生，常化一切阿修羅等堪受彼供，是諸阿修羅無上導師，與九萬五千具足五通阿修羅仙，前後圍遶來詣佛所頂禮佛足，以真金瓶盛八功德水，置於佛前并獻寶杖；復與九萬五千眷屬各各執持異種寶蓋，以奉世尊右遶三匝，合掌向佛，說偈讚曰：}@
%{当时，又有一位阿修罗仙人，名叫一切菩提鬘，拥有大福德、大威德、大智慧、大苦行，以菩提心而庄严自身，获得了五神通，远离了尘垢，安住成就一切众生，经常化度一切阿修罗等，使他们能够接受供养，是所有阿修罗的无上导师，带着九万五千位具足五神通的阿修罗仙人，前后围绕着来到佛陀面前，顶礼佛陀的双足，用真金瓶盛着八功德水，放在佛陀面前，并献上宝杖；又让九万五千位眷属，每人手持不同的宝盖，奉献给世尊，绕佛三圈，合掌向佛陀，说偈赞曰：}%

@{忍辱如大地，忍水常盈滿，}@
%{忍辱像大地一样宽广，忍水如同河流一样充满，}%

@{安住於淨忍，寧心無所失。}@
%{安住于清净的忍辱，宁静心，无所失落。}%

@{煩惱渴愛盡，安處於信財，}@
%{烦恼、渴爱都尽了，安处在信财之中，}%

@{佛住慈悲心，置眾菩提道。}@
%{佛陀住着慈悲心，安置众生在菩提道上。}%

@{說法猶如水，若聞如是法，}@
%{说法如同水一样，如果听到这样的法，}%

@{愛樂菩提心，能成第一義。}@
%{就会爱乐菩提心，能够成就第一义。}%

@{大悲願調伏，此諸下劣行，}@
%{大悲愿调伏这些下劣的行为，}%

@{願受我修羅，所獻眾寶物。}@
%{愿接受我修罗所献的宝物。}%

@{三界所有供，棄捨不求染，}@
%{三界所有供养，都放弃了，不求染着，}%

@{唯佛無煩惱，堪受世供養。}@
%{只有佛陀没有烦恼，才堪受世间的供养。}%

@{佛以大菩提，嚴三有眾生，}@
%{佛陀以大菩提，庄严三界众生，}%

@{願說第一義，為得菩提故。}@
%{愿说第一义，为了得到菩提。}%

@{我於拘留孫，曾聞第一義，}@
%{我在拘留孙佛那里，曾经听闻第一义，}%

@{拘那含牟尼，迦葉佛亦然。}@
%{拘那含牟尼佛、迦叶佛也是如此。}%

@{我以自願力，應現阿修羅，}@
%{我以自愿之力，应该现身阿修罗，}%

@{為化修羅故，修行菩提道。}@
%{为了化度修罗，修行菩提道。}%

@{聞法獲德藏，復能轉示他，}@
%{听到法，获得了宝藏，又能转示他人，}%

@{降魔惡徒黨，熾然正法朋。}@
%{降伏魔恶的徒党，成为炽热的正法朋侣。}%

@{於此濁惡世，難有功德人，}@
%{在这个浊恶的世间，难得有功德之人，}%

@{現佛境界事，是佛妙神力。}@
%{显现佛的境界之事，是佛陀的妙法神力。}%

@{大集經卷第四十七}@
