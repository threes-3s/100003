@{高齊天竺三藏那連提耶舍譯}@
%{高齐天竺三藏那连提耶舍译}%

@{月藏分第十四月幢神呪品第一}@
%{月藏分第十四月幢神咒品第一}%

@{如是我聞：}@
%{如是我闻：}%

@{一時，佛在佉羅帝山牟尼諸仙所依住處，與大比丘眾有學無學六百萬人，於諸煩惱堅牢纏縛悉得解脫，唯勤方便求斷習氣，及諸菩薩摩訶薩眾，無量無邊不可算數不可稱計，悉得忍力化諸龍眾。}@
%{一时，佛在佉罗帝山牟尼诸仙所依住处，与大比丘众有学无学六百万人，于诸烦恼坚牢缠缚悉得解脱，唯勤方便求断习气，及诸菩萨摩诃萨众，无量无边不可算数不可称计，悉得忍力化诸龙众。}%

@{說日藏經已，即時西方現大華雲，所謂優波羅華、波頭摩華、拘牟陀華、芬陀利華、阿提目多華、瞻波迦華、婆利師迦華，如是華雲悉皆來現。其華雲中現一半月，廣十由旬，於半月中復現真金重閣講堂，莊嚴微妙。其堂光明過百千萬億日月光明，其光悉照佉羅帝山。}@
%{说日藏经已，即时西方现大华云，所谓优波罗华、波头摩华、拘牟陀华、芬陀利华、阿提目多华、瞻波迦华、婆利师迦华，如是华云悉皆来现。其华云中现一半月，广十由旬，于半月中复现真金重阁讲堂，庄严微妙。其堂光明过百千万亿日月光明，其光悉照佉罗帝山。}%

@{復現種種奇異華雲，所謂優波羅華，乃至婆利師迦華，其華光艶，照牟尼仙所依住處，七寶五柱重閣講堂，甚奇微妙，隱蔽日月光不能現。於其堂中復現半月，於半月中有於千葉青色蓮華，其華臺上復有世尊端坐說法。其光普照一切大眾，一一頭上皆現半月微妙天鬘；復雨種種寶、種種華、種種香。}@
%{复现种种奇异华云，所谓优波罗华，乃至婆利师迦华，其华光艳，照牟尼仙所依住处，七宝五柱重阁讲堂，甚奇微妙，隐蔽日月光不能现。于其堂中复现半月，于半月中有于千葉青色莲花，其华台上复有世尊端坐说法。其光普照一切大众，一一头上皆现半月微妙天鬘；复雨种种宝、种种华、种种香。}%

@{爾時，慧命大目揵連，見如是等神通變化生希有心，又知大眾心有所疑，從坐而起，偏袒右肩右膝著地，合掌向佛，而說偈言：}@
%{尔时，慧命大目揵连，见如是等神通变化生希有心，又知大众心有所疑，从坐而起，偏袒右肩右膝着地，合掌向佛，而说偈言：}%

@{惟佛悉除諸煩惱，於盲冥中能覺悟，}@
%{唯佛悉除诸烦恼，于盲冥中能觉悟，}%

@{為群生故閉惡趣，令諸眾生住善道。}@
%{为群生故闭恶趣，令诸众生住善道。}%

@{惟佛降伏諸億魔，令諸外道失光顯，}@
%{唯佛降伏诸亿魔，令诸外道失光显，}%

@{調伏眾生住檀尸，枯涸眾生煩惱海。}@
%{调伏众生住檀尸，枯涸众生烦恼海。}%

@{八功德水令洗浴，以覺分寶濟眾生，}@
%{八功德水令洗浴，以觉分宝济众生，}%

@{無量億眾入涅槃，能轉無上法輪寶。}@
%{无量亿众入涅槃，能转无上法轮宝。}%

@{一切所有諸龍眾，為瞋所使行諸惡，}@
%{一切所有诸龙众，为瞋所使行诸恶，}%

@{渴愛所逼無有慈。惟佛能益令歸信。}@
%{渴爱所逼无有慈。唯佛能益令归信。}%

@{四天下龍皆來集，一心歸於佛法僧，}@
%{四天下龙皆来集，一心归于佛法僧，}%

@{盡諸業障及煩惱，皆護正法而安住。}@
%{尽诸业障及烦恼，皆护正法而安住。}%

@{於此復現妙華雲，中有半月光照曜，}@
%{于此复现妙华云，中有半月光照曜，}%

@{一切現此半月鬘，今當欲有何佛事？}@
%{一切现此半月鬘，今当欲有何佛事？}%

@{積此諸華如大山，雨諸香華及眾寶，}@
%{积此诸华如大山，雨诸香华及众宝，}%

@{大眾覩瑞心有疑，當欲雨於何法雨？}@
%{大众睹瑞心有疑，当欲雨于何法雨？}%

@{此處微妙最第一，如此大眾皆依住，}@
%{此处微妙最第一，如此大众皆依住，}%

@{於諸過佛修供養，是人師子如是來。}@
%{于诸过佛修供养，是人师子如是来。}%

@{爾時，佛告大目揵連：西方有世界名月勝，佛號日月光。有月藏童真菩薩摩訶薩，將諸眷屬八十億那由他百千菩薩摩訶薩，欲來向此，為欲見我禮拜供養，與大眾集說欲隨喜，又欲付囑諸天、龍、夜叉、乾闥婆等正法眼故。}@
%{尔时，佛告大目揵连：西方有世界名月胜，佛号日月光。有月藏童真菩萨摩诃萨，将诸眷属八十亿那由他百千菩萨摩诃萨，欲来向此，为欲见我礼拜供养，与大众集说欲随喜，又欲付嘱诸天、龙、夜叉、乾闼婆等正法眼故。}%

@{應時月藏菩薩摩訶薩，與其眷屬八十億那由他百千菩薩摩訶薩，從彼世界來至佛所，頭面禮足，右遶三匝，住立佛前，皆悉合掌一音說偈：}@
%{应时月藏菩萨摩诃萨，与其眷属八十亿那由他百千菩萨摩诃萨，从彼世界来至佛所，头面礼足，右遶三匝，住立佛前，皆悉合掌一音说偈：}%

@{吉祥無數劫，所修為眾生，}@
%{吉祥无数劫，所修为众生，}%

@{吉祥見眾生，生死苦所逼。}@
%{吉祥见众生，生死苦所逼。}%

@{吉祥以檀施，大仙作饒益，}@
%{吉祥以檀施，大仙作饶益，}%

@{吉祥能行施，超越於人天。}@
%{吉祥能行施，超越于人天。}%

@{吉祥護淨戒，眾生不能動，}@
%{吉祥护净戒，众生不能动，}%

@{吉祥令怒者，住於慈善心。}@
%{吉祥令怒者，住于慈善心。}%

@{吉祥發勇進，度脫懈怠者，}@
%{吉祥发勇进，度脱懈怠者，}%

@{吉祥離惡道，安置於善趣。}@
%{吉祥离恶道，安置于善趣。}%

@{吉祥善修忍，容恕怒惡心，}@
%{吉祥善修忍，容恕怒恶心，}%

@{吉祥希有事，是故悉歸依。}@
%{吉祥希有事，是故悉归依。}%

@{吉祥修諸禪，諸天生希有，}@
%{吉祥修诸禅，诸天生希有，}%

@{吉祥悉枯竭，眾生諸苦海。}@
%{吉祥悉枯竭，众生诸苦海。}%

@{吉祥熏修智，追迴惡道輪，}@
%{吉祥熏修智，追回恶道轮，}%

@{吉祥上菩提，到最難到處。}@
%{吉祥上菩提，到最难到处。}%

@{吉祥降魔眾，建立正法幢，}@
%{吉祥降魔众，建立正法幢，}%

@{吉祥所轉者，最是正法輪。}@
%{吉祥所转者，最是正法轮。}%

@{吉祥愍異空，降伏諸外道，}@
%{吉祥愍异空，降伏诸外道，}%

@{吉祥降法雨，充足渴世間。}@
%{吉祥降法雨，充足渴世间。}%

@{吉祥作明證，天人乾闥婆，}@
%{吉祥作明证，天人乾闼婆，}%

@{吉祥滿世界，我為最上師。}@
%{吉祥满世界，我为最上师。}%

@{吉祥安四果，應受世供養，}@
%{吉祥安四果，应受世供养，}%

@{吉祥令億眾，安住於涅槃。}@
%{吉祥令亿众，安住于涅槃。}%

@{吉祥久住時，法眼所建立，}@
%{吉祥久住时，法眼所建立，}%

@{吉祥久住世，世間無與等。}@
%{吉祥久住世，世间无与等。}%

@{吉祥菩提果，眾所獲大利，}@
%{吉祥菩提果，众所获大利，}%

@{吉祥為眾生，宣說無上法。}@
%{吉祥为众生，宣说无上法。}%

@{吉祥以法水，洗浴諸眾生，}@
%{吉祥以法水，洗浴诸众生，}%

@{吉祥善能度，是諸天人等。}@
%{吉祥善能度，是诸天人等。}%

@{吉祥能顯現，無垢真妙法，}@
%{吉祥能显现，无垢真妙法，}%

@{吉祥除眾生，所有諸煩惱。}@
%{吉祥除众生，所有诸烦恼。}%

@{吉祥諸僧眾，世間最第一，}@
%{吉祥诸僧众，世间最第一，}%

@{吉祥善生世，能益於人天。}@
%{吉祥善生世，能益于人天。}%

@{吉祥令四眾，明淨善光顯，}@
%{吉祥令四众，明净善光显，}%

@{吉祥令四眾，全護戒律儀。}@
%{吉祥令四众，全护戒律仪。}%

@{吉祥行檀捨，持戒及精進，}@
%{吉祥行檀舍，持戒及精进，}%

@{吉祥修忍禪，及以妙般若。}@
%{吉祥修忍禅，及以妙般若。}%

@{吉祥大梵王，娑婆世界主，}@
%{吉祥大梵王，娑婆世界主，}%

@{吉祥大魔王，諸欲自在王，}@
%{吉祥大魔王，诸欲自在王，}%

@{吉祥憍尸迦，輔佐諸眷屬，}@
%{吉祥憍尸迦，辅佐诸眷属，}%

@{吉祥諸天眾，及諸宮殿等；}@
%{吉祥诸天众，及诸宫殿等；}%

@{吉祥毘沙門，及諸夜叉眾。}@
%{吉祥毘沙门，及诸夜叉众。}%

@{吉祥提頭賴，眷屬乾闥婆，}@
%{吉祥提头赖，眷属乾闼婆，}%

@{吉祥毘樓勒，并與鳩槃茶，}@
%{吉祥毘楼勒，并与鸠槃茶，}%

@{吉祥毘樓博，及諸龍軍眾。}@
%{吉祥毘楼博，及诸龙军众。}%

@{吉祥日月天，星辰及諸宿，}@
%{吉祥日月天，星辰及诸宿，}%

@{吉祥大自在，兒及造界主，}@
%{吉祥大自在，儿及造界主，}%

@{吉祥風火神，及以地神等。}@
%{吉祥风火神，及以地神等。}%

@{吉祥諸龍眾，及以阿修羅，}@
%{吉祥诸龙众，及以阿修罗，}%

@{吉祥諸羅剎，及與緊那羅，}@
%{吉祥诸罗刹，及与紧那罗，}%

@{吉祥迦樓羅，摩睺羅伽等。}@
%{吉祥迦楼罗，摩睺罗伽等。}%

@{吉祥雨甘雨，行雨大神王，}@
%{吉祥雨甘雨，行雨大神王，}%

@{吉祥護持國，一切人中王。}@
%{吉祥护持国，一切人中王。}%

@{吉祥婆羅門，剎利毘舍陀，}@
%{吉祥婆罗门，剎利毘舍陀，}%

@{吉祥所供養，最勝尊導師，}@
%{吉祥所供养，最胜尊导师，}%

@{吉祥願聞者，真正無過法。}@
%{吉祥愿闻者，真正无过法。}%

@{吉祥諸現未，供養三寶者，}@
%{吉祥诸现未，供养三宝者，}%

@{吉祥一切眾，滅除諸煩惱，}@
%{吉祥一切众，灭除诸烦恼，}%

@{吉祥諸眾生，同住於正法。}@
%{吉祥诸众生，同住于正法。}%

@{吉祥檀尸羅，精進之彼岸，}@
%{吉祥檀尸罗，精进之彼岸，}%

@{吉祥禪那度，忍辱波羅蜜，}@
%{吉祥禅那度，忍辱波罗蜜，}%

@{吉祥諸一切，到彼智岸者。}@
%{吉祥诸一切，到彼智岸者。}%

@{吉祥令諸病，一切皆除愈，}@
%{吉祥令诸病，一切皆除愈，}%

@{吉祥皆休息，一切濁惡世，}@
%{吉祥皆休息，一切浊恶世，}%

@{吉祥諸眾生，願令悉解脫，}@
%{吉祥诸众生，愿令悉解脱，}%

@{吉祥令一切，悉得諸無漏。}@
%{吉祥令一切，悉得诸无漏。}%

@{吉祥於大地，種子所生者，}@
%{吉祥于大地，种子所生者，}%

@{吉祥諸禾稼，藥草樹林果，}@
%{吉祥诸禾稼，药草树林果，}%

@{吉祥彼一切，依時悉成熟。}@
%{吉祥彼一切，依时悉成熟。}%

@{吉祥勝地精，一切處充滿，}@
%{吉祥胜地精，一切处充满，}%

@{吉祥人精氣，一切皆安住，}@
%{吉祥人精气，一切皆安住，}%

@{吉祥法精氣，充滿一切眾。}@
%{吉祥法精气，充满一切众。}%

@{吉祥皆休息，一切諸罪惡，}@
%{吉祥皆休息，一切诸罪恶，}%

@{吉祥令眾生，而得勝菩提，}@
%{吉祥令众生，而得胜菩提，}%

@{吉祥於諸法，自在到彼岸。}@
%{吉祥于诸法，自在到彼岸。}%

@{吉祥正法雨，普潤諸眾生，}@
%{吉祥正法雨，普润诸众生，}%

@{吉祥諸眾生，悉度於三有，}@
%{吉祥诸众生，悉度于三有，}%

@{吉祥令一切，悉證大涅槃。}@

@{大德婆伽婆！我今欲說吉祥章句大力神呪，如是神呪過去諸仙之所宣說，建立守護善能增長吉祥之事，能除一切罪垢惡見，入諸善根增長大悲。以此呪句，悉能資益一切眾生，乃至一切麞鹿鳥獸所得聞者，令如是等心得安隱，離濁惡世一切諸障，眾生障法障如是等障皆悉休息，一切善根隨所觸法令得入心，念慧堅固，名稱形色人所喜樂，勇健無畏於十善業道，堅固安住檀波羅蜜乃至般若波羅蜜。四念處乃至十八不共法堅固安住，大慈大悲、大方便力、一切種智乃至究竟無上涅槃堅固安住，除其造作五無間業、誹謗正法、毀呰賢聖、斷常二見，唯除如是諸罪人等。是吉祥句，常為先聖建立加護，如此呪句，亦復能令諸天信受得入十善業道，亦令得入檀波羅蜜乃至般若波羅蜜，四念處乃至十八不共法，大慈大悲、大方便力，乃至一切種智無上涅槃，復能令彼諸魔眷屬悉得歸信。諸神、龍王、夜叉、羅剎、阿修羅、乾闥婆、緊那羅、伽樓羅、摩睺羅伽、餓鬼、毘舍闍、剎利、婆羅門、毘舍、首陀，令入十善業道、檀波羅蜜乃至般若波羅蜜，大慈大悲、大方便力，四念處乃至十八不共法，一切智一切種智無上涅槃，惟除五逆、誹謗正法、毀呰賢聖。}@

@{唯佛一人於四流，能度眾生三有海，}@
%{唯有佛一人在四流中，能度众生三有海。}%
@{謝過世尊實語者，以我因緣後來故。}@
%{感谢世尊实语者，因为我因缘后来才到这里。}%

@{惟佛開諸正法藏，以七聖財濟眾生，}@
%{唯有佛打开正法藏，用七圣财济度众生。}%
@{謝過大勝法施主，以我因緣後來故。}@
%{感谢大胜法施主，因为我因缘后来才到这里。}%

@{惟佛能與眾生眼，於無明闇拔盲瞽，}@
%{唯有佛能给众生眼睛，在无明黑暗中拔除盲瞽。}%
@{我本端坐一三昧，阿頗那禪心安住，}@
%{我本来端坐一三昧，阿颇那禅心安住。}%
@{我不見佛現神變，以我因緣後來故。}@
%{我并没有看见佛现神变，因为我因缘后来才到这里。}%

@{佛言：善哉，善哉！}@
%{佛说：善哉，善哉！}%
@{善男子！汝大精進，能於七日入深禪定，如是妙定是丈夫住處，是如來住處、無上住處。}@
%{善男子！你大精进，能够在七日中进入深禅定，如此妙定是丈夫的住处，是如来的住处、无上的住处。}%

@{善男子！汝及眷屬七日安住阿頗那迦禪，以是義故今悉成就無量億那由他百千諸天、龍王、夜叉、阿修羅、緊那羅、人、非人等，於阿耨多羅三藐三菩提，除其業障、眾生障、法障、禪障、煩惱障、覺分障，悉滅無餘。}@
%{善男子！你和眷属七日安住阿颇那迦禅，因为这个缘故，现在都成就了无量亿那由他百千诸天、龙王、夜叉、阿修罗、紧那罗、人、非人等等，在阿耨多罗三藐三菩提，除去他们的业障、众生障、法障、禅障、烦恼障、觉分障，都灭尽无余。}%
@{彼諸眾生，有得不忘菩提心三昧者，於無上道不退轉者。}@
%{那些众生，有的得到了不忘菩提心三昧，在无上道不退转。}%
@{復有眾生，於一切佛法得大明忍，彼等眾生以此善根，不久於阿耨多羅三藐三菩提而成正覺。}@
%{又有一些众生，在一切佛法中得到了大明忍，他们这些众生，凭借着这个善根，不久就会在阿耨多罗三藐三菩提而成正觉。}%
@{如是，善男子！汝以七日入禪定故，一時能滅眾生大苦，令得成就大福德聚。}@
%{是这样的，善男子！你因为七天进入禅定，所以就能一下子消除众生的大苦，让他们得到成就大福德聚。}%

@{善男子！若有眾生，惟依讀誦欲求阿耨多羅三藐三菩提者，是人多憙著於世俗，以世俗故尚不得調己心煩惱，何能調伏他人煩惱。}@
%{善男子！如果有众生，只依靠读诵想要求阿耨多罗三藐三菩提，这个人大多喜欢执着于世俗，因为世俗的原因，连自己心里的烦恼都调伏不了，怎么能调伏别人的烦恼呢？}%

@{善男子、善女人樂著讀誦求菩提者，便有嫉妬求名利富貴，高心自是輕慢毀他，以自高故，尚不能得欲界善根，何況能得色無色界一切善根！}@
%{善男子、善女人喜欢读诵求菩提，就会有嫉妒心，追求名利富贵，自以为是，轻慢毁谤他人，因为自高自大的缘故，连欲界的善根都得不到，何况能得到色界无色界一切善根呢！}%

@{又不能得聲聞菩提，何況能得辟支佛道及無上菩提！何以故？第一義諦阿耨多羅三藐三菩提，不與聲聞辟支佛共，是故非以世俗能得阿耨多羅三藐三菩提最勝善根大福德聚。}@
%{又不能得到声闻菩提，何况能得到辟支佛道和无上菩提呢！为什么呢？因为第一义谛阿耨多罗三藐三菩提，不与声闻辟支佛共同，所以不是用世俗的方法就能得到阿耨多罗三藐三菩提最胜善根大福德聚的。}%

@{善男子！譬如星火不能枯竭甚深大海。如是，如是！善男子！不以世俗能竭自身煩惱大海，何能竭他眾生煩惱？}@
%{善男子！譬如星星之火不能枯竭深不见底的大海。这样，这样！善男子！不用世俗的方法不能竭尽自己心中的烦恼大海，怎么能竭尽其他众生的烦恼呢？}%

@{善男子！譬如一人口所吹風，不能損壞世界大地。如是，善男子！不以世俗能得成就大慈大悲。}@
%{善男子！譬如一个人用嘴吹的风，不能损害世界大地。这样，善男子！不用世俗的方法不能得到成就大慈大悲。}%

@{善男子！譬如藕絲不能秤動須彌山王。如是，善男子！不以世俗能自滿於阿耨多羅三藐三菩提智，何能令他得第一義？}@
%{善男子！譬如藕丝不能称动须弥山王。这样，善男子！不用世俗的方法不能自己圆满阿耨多罗三藐三菩提智，怎么能使他人得到第一义呢？}%

@{如是，善男子！能成就阿耨多羅三藐三菩提智者，非世俗也。何者是第一義？所謂修造一切福事。若修福者，亦當數數熏修身心；若修身者，則能修心；能修心者，則能修慧；若能修身心修慧，如是之人則能速滿六波羅蜜，能以四事攝諸眾生，成熟於阿耨多羅三藐三菩提。}@
%{是这样的，善男子！能成就阿耨多罗三藐三菩提智的人，不是依靠世俗的。什么才是第一义呢？所谓修造一切福事。如果修福，也应该经常熏修身心；如果修身，就能修心；能修心，就能修慧；如果能修身心修慧，这样的人就能很快圆满六波罗蜜，能用四事摄受一切众生，成熟于阿耨多罗三藐三菩提。}%

@{成等正覺，不以世俗也。}@
%{成等正觉，不是依靠世俗的。}%

@{於世俗中，復有眾生計斷常二見者，非第一義；復有眾生，於世俗中我見邊見，亦非第一義；復有眾生，於其世俗，求現世樂及後世樂，亦非第一義。}@
%{在世俗中，还有众生执着于断常二见，这不是第一义；还有众生，在世俗中执着于我见边见，也不是第一义；还有众生，在世俗中，追求现世快乐和后世快乐，也不是第一义。}%

@{而我不見更有一法，能盡業障乃至煩惱障，一日一夜令無量億那由他百千眾生，悉得敬信佛法僧寶，成熟安住無上大乘。}@
%{而我所见没有其他方法，能尽除业障乃至烦恼障，在一昼夜之间使无量亿那由他百千众生，都恭敬信仰佛法僧宝，成熟安住无上大乘。}%

@{若有禪士，雖復持戒不能具足，禪法不周未得三昧，是人於禪若坐若行，初夜後夜，得與禪定相應而住，則能除斷無量業障，能令多億那由他百千眾生悉得歸信，成熟菩提種種善根福德之聚，況具持戒，得真法三昧諸陀羅尼忍，得四梵住宴坐寂定，於七日中，所得福德，不可思議不可為喻，何況能除眾生障煩惱障等盡滅無餘，乃至成熟無量眾生，向阿耨多羅三藐三菩提，亦能積集無量福聚，又能滿足六波羅蜜。}@
%{如果有禅士，虽然持戒不能圆满，禅法不周全，没有得到三昧，这个人无论坐禅还是行走，初夜后夜，能与禅定相应而住，就能消除无量业障，能使多亿那由他百千众生都归信佛法，成熟菩提种种善根福德之聚，何况具足持戒，得到真法三昧诸陀罗尼忍，得到四梵住宴坐寂定，在七天中，所得福德，不可思议不可比喻，何况能消除众生障烦恼障等尽除无余，乃至成熟无量众生，向阿耨多罗三藐三菩提，也能积集无量福聚，又能圆满六波罗蜜。}%

@{何以故？是修禪者，若行若坐除諸障法令心清淨，於一切行捨攀緣想，是檀波羅蜜。於捨攀緣想，常不休息，是尸波羅蜜。於諸境界不生瘡疣，是羼提波羅蜜。不捨於離，是毘梨耶波羅蜜。於諸事中心不放縱，是禪波羅蜜。諸法體性無生樂忍，是般若波羅蜜。}@
%{为什么呢？因为修禅的人，无论行走还是坐禅，都能除掉一切障碍法，使心清净，在一切行为中舍弃攀缘的念头，这是布施波罗蜜。在舍弃攀缘的念头中，常不休息，这是持戒波罗蜜。在一切境界中不生疮疣，这是忍辱波罗蜜。不舍弃离欲，这是精进波罗蜜。在一切事情中，心不放纵，这是禅定波罗蜜。一切法体性无生乐忍，这是般若波罗蜜。}%

@{復次，若於境界不起擾濁，是檀波羅蜜。若於境界無有瘡疣，是尸波羅蜜。若於境界不能染污，是羼提波羅蜜。若於境界無有動轉，是毘梨耶波羅蜜。若於境界無有計念，是禪波羅蜜。若於境界一向清淨行，是般若波羅蜜。}@
%{再次，如果在境界中不起扰浊，这是布施波罗蜜。如果在境界中没有疮疣，这是持戒波罗蜜。如果在境界中不能染污，这是忍辱波罗蜜。如果在境界中没有动转，这是精进波罗蜜。如果在境界中没有计念，这是禅定波罗蜜。如果在境界中一向清净行，这是般若波罗蜜。}%

@{復次，於諸陰捨，是檀波羅蜜。於諸陰不計念，是尸波羅蜜。於諸陰求無我想，是羼提波羅蜜。於諸陰起怨家想，是毘梨耶波羅蜜。於諸陰不令熾然，是禪波羅蜜。於諸陰畢竟棄捨，是般若波羅蜜。}@
%{再次，对一切阴舍弃，这是布施波罗蜜。对一切阴不计念，这是持戒波罗蜜。对一切阴追求无我，这是忍辱波罗蜜。对一切阴起怨家想，这是精进波罗蜜。对一切阴不使炽盛，这是禅定波罗蜜。对一切阴最终舍弃，这是般若波罗蜜。}%

@{復次，於諸界捨，是檀波羅蜜。於諸界不擾濁，是尸波羅蜜。於諸界捨因緣，是羼提波羅蜜。於諸界數數棄捨，是毘梨耶波羅蜜。於諸界不起發，是禪波羅蜜。於諸界如幻想，是般若波羅蜜。}@
%{再次，对一切界舍弃，这是布施波罗蜜。对一切界不扰浊，这是持戒波罗蜜。对一切界舍弃因缘，这是忍辱波罗蜜。对一切界数数舍弃，这是精进波罗蜜。对一切界不起发，这是禅定波罗蜜。对一切界如同幻象，这是般若波罗蜜。}%
@{復次，菩薩於諸眾生起於慈心，是檀波羅蜜。於諸眾生心無憎愛，是尸波羅蜜。於諸眾生起於悲想，是羼提波羅蜜。於諸眾生起救濟想，是毘梨耶波羅蜜。於諸眾生以喜攝想，是禪波羅蜜。於諸眾生不作彼此吾我等想，是般若波羅蜜。}@
%{其次，菩萨对所有众生心怀慈悲，这就是布施波罗蜜。对所有众生没有憎恨和喜爱，这就是持戒波罗蜜。对所有众生心怀悲悯，这就是忍辱波罗蜜。对所有众生心怀救济，这就是精进波罗蜜。对所有众生心怀喜悦，这就是禅定波罗蜜。对所有众生不执着彼此和自我，这就是般若波罗蜜。}%

@{復次，菩薩於諸眾生，以法施之不生二想，是檀波羅蜜。於諸眾生柔和愛語，是尸波羅蜜。於諸眾生不起諸惡，是羼提波羅蜜。於諸眾生愛語不退，是毘梨耶波羅蜜。於諸眾生利益憐愍，是禪波羅蜜。於諸眾生同行其法，是般若波羅蜜。}@
%{其次，菩萨对所有众生，以法布施而不生二种想法，这就是布施波罗蜜。对所有众生以柔和的爱语，这就是持戒波罗蜜。对所有众生不起诸恶，这就是忍辱波罗蜜。对所有众生爱语不退，这就是精进波罗蜜。对所有众生利益怜悯，这就是禅定波罗蜜。对所有众生同行其法，这就是般若波罗蜜。}%

@{復次，菩薩安置眾生於諸善處，是檀波羅蜜。於一切法而不依倚，是尸波羅蜜。於一切法以一道入，是羼提波羅蜜。於一切法及一切難無擾濁想，是毘梨耶波羅蜜。於一切法而不分別，是禪波羅蜜。能以一字入一切法為眾生說，是般若波羅蜜。}@
%{其次，菩萨安置众生在善处，这就是布施波罗蜜。对一切法不依附，这就是持戒波罗蜜。对一切法以一条道进入，这就是忍辱波罗蜜。对一切法和一切困难没有烦恼和执着，这就是精进波罗蜜。对一切法不分别，这就是禅定波罗蜜。能以一个字进入一切法为众生讲说，这就是般若波罗蜜。}%

@{善男子！如是菩薩摩訶薩，以此第一義甚深法要，能滿六波羅蜜，非世俗也。}@
%{善男子！像这样的菩萨摩诃萨，以这个第一义甚深法要，能够圆满六波罗蜜，这并非世俗法。}%

@{如是，如是！善男子！諸菩薩摩訶薩等，第一義諦善巧方便，皆以此法自為為他，勤修令滿六波羅蜜，速於阿耨多羅三藐三菩提而成正覺。如是十方現在諸餘世界所有菩薩摩訶薩等，第一義諦善巧方便，一切皆悉以此道法，速於阿耨多羅三藐三菩提而成正覺。當來十方無量阿僧祇諸佛世界諸菩薩摩訶薩等，皆悉勤修如是甚深第一義諦善巧方便，修六波羅蜜，能於阿耨多羅三藐三菩提而成正覺，非世俗也。}@
%{这样，这样！善男子！诸菩萨摩诃萨等，以第一义谛巧妙方便，都以此法自利利他，精进修持，令圆满六波罗蜜，迅速成就无上正等正觉。像这样十方世界现在所有菩萨摩诃萨等，以第一义谛巧妙方便，都悉以此道法，迅速成就无上正等正觉。未来十方无量阿僧祇诸佛世界诸菩萨摩诃萨等，都悉精进修持这样甚深第一义谛巧妙方便，修持六波罗蜜，能够成就无上正等正觉，这并非世俗法。}%

@{彼諸菩薩摩訶薩，為法眼久住紹三寶種使不斷絕，勤求修學，為一切眾生，執大明炬而作照明，令其止息煩惱道苦道，與其慧眼。令度一切三有流轉，安置無上菩提之道。彼等於聖法默然，初夜後夜捨得相應，善能出生三昧正受，與三解脫門相應而住。}@
%{那些菩萨摩诃萨，为了法眼久住，传承三宝种子使不断绝，精进寻求学习，为一切众生，执持大明炬而照亮，令其止息烦恼道苦道，给予他们慧眼。令度一切三界轮回，安置在无上菩提之道。他们对圣法默然，初夜后夜舍得相应，善能出生三昧正受，与三解脱门相应而住。}%

@{善男子！汝今七日於阿頗那迦禪而入定故，成熟無量諸眾生故，是故勸汝及諸善男女等，若現在世未來世末世，於我法中初夜後夜常與於捨相應而住，以正法眼而作照明，紹隆三寶使不斷故，為成熟眾生故，勤修如是第一義諦，為滿六波羅蜜勤修而住。}@
%{善男子！你现在七天在阿婆那迦禅而入定，成熟无量诸众生，所以劝告你以及诸善男女等，无论现在世未来世末世，在我法中，初夜后夜常与舍得相应而住，以正法眼而照亮，传承三宝使不断，为了成熟众生，精进修持这样第一义谛，为了圆满六波罗蜜精进修持而住。}%

@{佛告一切諸天、人眾、龍、神、夜叉：應當養育供給是人，衣服、飲食、臥具、湯藥隨其所須盡給與之，亦當守護除其災橫，離諸凶衰殃惡疾病悉令除滅。何以故？與禪相應者是我真子，從佛口生從法化生。若有施主、天、龍、夜叉，能於現世及未來世與捨相應，以第一義諦為滿六波羅蜜故，為除眾生諸煩惱道及苦道故，法眼久住紹三寶種使不斷故，汝等施主、天、龍、夜叉皆應護養，并與衣服飲食臥具病瘦湯藥，隨其所須盡給與之，亦應勸請及以讚歎，以彼施主天、龍、夜叉持我正法，為欲令我法眼久住，紹三寶種使不斷故，如是等輩是我真子，從佛口生從法化生故。比丘、比丘尼、優婆塞、優婆夷，及餘清信士若善男子、善女人，以第一義乃至求於阿耨多羅三藐三菩提者，及養護者，我以彼等寄付於汝，彌勒為首及以賢劫諸菩薩等，當以四事攝受勸化，授其禁戒。復令住於四無量心、四禪四無色定、大方便力、大慈大悲，乃至十八不共法，當復授與無上道記。}@
%{佛告诉一切诸天、人众、龙、神、夜叉：应当养育供养这个人，衣服、饮食、卧具、汤药，随其所需尽给予之，也应当守护，除去他的灾横，远离一切凶衰殃恶疾病，悉令除灭。为什么呢？与禅相应者是我真子，从佛口生，从法化生。如果有施主、天、龙、夜叉，能够在现世及未来世与舍得相应，以第一义谛为圆满六波罗蜜，为了除众生诸烦恼道及苦道，法眼久住，传承三宝种子使不断，你们施主、天、龙、夜叉都应当护养，并给予衣服饮食卧具病瘦汤药，随其所需尽给予之，也应当劝请以及赞叹，因为那些施主天、龙、夜叉持我正法，为了令我法眼久住，传承三宝种子使不断，像这样等辈是我真子，从佛口生，从法化生。比丘、比丘尼、优婆塞、优婆夷，以及其他清信士，无论是善男子、善女人，以第一义谛，乃至求于无上正等正觉者，以及养护者，我将他们托付于你，弥勒为首，以及贤劫诸菩萨等，应当以四事摄受劝化，授予他们禁戒。再令住于四无量心、四禅四无色定、大方便力、大慈大悲，乃至十八不共法，应当再授予无上道记。}%

@{爾時，彌勒菩薩摩訶薩以為上首，及與賢劫諸菩薩等白佛言：世尊！如是，如是。大德婆伽婆！我當護念彼諸眾生，乃至與其授於阿耨多羅三藐三菩提記。若現在世，及未來世，乃至法住，是諸施主作大明者，亦當與授無上道記。天、龍、夜叉、揵闥婆等，於阿蘭若處靜默修行求第一義者，信樂受持供養供給，衣服臥具乃至湯藥，隨所須者。}@
%{尔时，弥勒菩萨摩诃萨以为首，以及贤劫诸菩萨等白佛言：世尊！如是，如是。大德婆伽婆！我当护念那些众生，乃至给予他们授予无上正等正觉记。无论是现在世，以及未来世，乃至法住，这些施主作大明者，也应当给予他们授予无上道记。天、龙、夜叉、紧那罗等，在阿兰若处静默修行，求第一义者，信乐受持供养供给，衣服卧具乃至汤药，随其所需者。}%
@{爾時，世尊欲重明此義，而說偈言：}@
%{当时，佛陀想要再次阐明这个道理，就用偈颂来说：}%

@{於此世間一日出，無量億華悉開敷，}@
%{在这个世间，太阳升起一天，无数亿朵花都开放，}%

@{如佛一人出世間，眾生所希福華現。}@
%{如同佛陀一人出世，众生所希求的福德之花就展现出来。}%

@{若欲速得十勝力，及度堅固誑煩惱，}@
%{如果你想要快速获得十种胜利的力量，以及克服坚固的欺骗烦恼，}%

@{復欲速得最勝定，靜默獨住阿蘭若。}@
%{并且想要快速获得最殊胜的禅定，就应该静默独自住在寂静处。}%

@{欲得人天信敬受，及除心之煩惱渴，}@
%{如果你想要得到人天众生的恭敬和信赖，以及消除心中的烦恼渴爱，}%

@{欲斷於心苦重擔，安心聖道奢摩他。}@
%{想要断除心中痛苦的重担，就应该安住于圣道奢摩他禅。}%

@{若欲排却諸惡難，及諸功德自莊嚴，}@
%{如果你想要排除一切恶难，以及让一切功德自然庄严，}%

@{於諸苦海欲自度，應當安心妙菩提。}@
%{想要在一切苦海中自度，就应该安住于妙菩提心。}%

@{若欲得彼七法財，及欲得於方便忍，}@
%{如果你想要获得那七种法财，以及想要获得方便忍，}%

@{欲為眾生說妙法，常當樂住阿蘭若，}@
%{想要为众生宣说妙法，就应该经常欢喜地住在寂静处。}%

@{六根常與三昧合，應當寂住阿蘭若，}@
%{六根应该经常与三昧相合，就应该寂静地住在寂静处。}%

@{少欲頭陀善知足，此人能入賢聖道。}@
%{少欲知足，善于头陀行，这个人就能进入贤圣之道。}%

@{若能速捨五欲樂，得五力故滅煩惱；}@
%{如果你能够快速舍弃五欲之乐，得到五种力量就能消灭烦恼。}%

@{若於五道度眾生，自捨過惡住三昧；}@
%{如果你能够在五道中度化众生，自己舍弃过失安住于三昧。}%

@{若欲得於四無量，及得無礙四辯才，}@
%{如果你想要获得四无量心，以及获得无碍的四种辩才，}%

@{欲得四禪彼岸處，是人應修第一義。}@
%{想要获得四禅的彼岸之处，这个人就应该修习第一义谛。}%

@{若欲速知於三有，欲知諸法苦無常，}@
%{如果你想要快速了解三界，想要了解诸法苦无常，}%

@{及知諸行性相空，應當樂住阿蘭若。}@
%{以及了解诸行性相空，就应该欢喜地住在寂静处。}%

@{若欲速知二種法，毘婆舍那奢摩他，}@
%{如果你想要快速了解两种法，就是毗婆舍那和奢摩他，}%

@{及欲速知有為過，要當住於菩提心。}@
%{以及想要快速了解有为法的过患，就应该安住于菩提心。}%

@{獨住閑靜不放逸，便能疾捨於世諦，}@
%{独自住在寂静处不放逸，就能快速舍弃世俗谛，}%

@{以精進求第一義，能速捨離諸惡道。}@
%{以精进求证第一义谛，就能快速舍弃一切恶道。}%

@{若欲枯竭膿血海，及欲枯竭煩惱海；}@
%{如果你想要枯竭脓血之海，以及想要枯竭烦恼之海，}%

@{若欲速竭三有海，常與聖種心相應；}@
%{如果你想要快速枯竭三界之海，就应该经常与圣种心相应。}%

@{若欲成熟眾生海，若欲滿諸大願海；}@
%{如果你想要成熟众生之海，如果你想要圆满一切大愿之海，}%

@{若欲得知生死際，如救頭燃處閑靜。}@
%{如果你想要了解生死之界，就像救头燃火一样安住于寂静处。}%

@{欲知本生及居處，久遠微細所從來，}@
%{想要了解本生和住处，以及久远微细的来处，}%

@{以諸方便樂閑靜，攝心於彼得三昧。}@
%{就应该以一切方便欢喜地住在寂静处，摄心于彼获得三昧。}%

@{若欲遊戲禪定海，若欲覺悟神通海；}@
%{如果你想要游玩禅定之海，如果你想要觉悟神通之海，}%

@{若欲度於渴愛海，若欲得於天中最；}@
%{如果你想要度过渴爱之海，如果你想要获得天中最殊胜的果位，}%

@{若欲得飲正法海，若欲見於莊嚴土；}@
%{如果你想要得到饮用正法之海，如果你想要见到庄严的佛土，}%

@{若欲見於諸佛海，欲問甚深諸義海，}@
%{如果你想要见到诸佛之海，想要询问甚深的诸义之海，}%

@{欲得如是勝功德，及欲速得勝菩提，}@
%{想要获得如此殊胜的功德，以及想要快速获得胜菩提，}%

@{當離眾惱住蘭若，以此得道亦不難。}@
%{就应该远离一切烦恼住在寂静处，以此获得道果也并不困难。}%

@{若人百億諸佛所，於多歲數常供養；}@
%{如果一个人在百亿诸佛处，供养很多年，}%

@{若能七日在蘭若，攝根得定福多彼。}@
%{如果能够在寂静处住七天，摄心获得禅定，福德比那还要多。}%

@{若人讀誦千億法，及解妙義如佛說；}@
%{如果一个人读诵千亿法，并且理解妙义如同佛陀所说，}%

@{若於七日住蘭若，三昧福聚轉多彼。}@
%{如果能够在寂静处住七天，三昧的福德积累比那还要多。}%

@{若人多歲營僧事，更不造作餘種業；}@
%{如果一个人多年为僧众服务，不再造作其他的恶业，}%

@{若人七日住蘭若，其人福聚多於彼。}@
%{如果能够在寂静处住七天，他的福德积累比那还要多。}%

@{為眾說法解深義，於多年歲無餘業，}@
%{为大众宣说佛法解释深义，多年没有其他的恶业，}%

@{若能七日心住寂，其福德聚不可數。}@
%{如果能够在寂静处住七天，心住寂静，他的福德积累不可计数。}%

@{若人營造多佛塔，伽藍田業給施僧；}@
%{如果一个人建造很多佛塔，伽蓝田地布施给僧众，}%

@{若能七日在蘭若，其福轉多勝於彼。}@
%{如果能够在寂静处住七天，他的福德比那还要多。}%

@{閑靜無為佛境界，於彼能得淨菩提，}@
%{寂静无为是佛的境界，在那里能够获得清净的菩提，}%

@{若人謗彼住禪者，是名毀謗諸如來。}@
%{如果有人诽谤那些住在禅定的人，就是诽谤诸佛如来。}%

@{若人破塔多百千，及以焚燒百千寺；}@
%{如果一个人毁坏很多佛塔，以及焚烧很多寺院，}%

@{若有毀謗住禪者，其罪甚多過於彼。}@
%{如果有人诽谤住在禅定的人，他的罪过比那还要严重。}%

@{若有供養住禪者，飲食衣服及湯藥，}@
%{如果有人供养住在禅定的人，饮食衣服以及汤药，}%

@{是人消滅無量罪，亦不墮於三惡道。}@
%{这个人就能消灭无量罪业，也不会堕落到三恶道。}%

@{是故我今普告汝，欲成佛道常在禪。}@
%{所以我现在普遍告诫你们，想要成佛之道，就应该常住在禅定中。}%

@{若不能住阿蘭若，應當供養於彼人；}@
%{如果你不能住在寂静处，就应该供养那些住在寂静处的人。}%

@{若能住禪不放逸，則能速滿於六度。}@
%{如果你能够住在禅定中不放逸，就能快速圆满六度。}%

@{欲求大明菩提道，以此方便疾能到，}@
%{如果你想要寻求光明的大菩提道，就应该用这个方便就能快速到达。}%

@{欲求菩提住寂靜，當捨一切諸緣業，}@
%{如果你想要寻求菩提，就应该住在寂静处，舍弃一切缘起之业，}%

@{及離煩惱捨諸樂，則能速到檀彼岸。}@
%{以及远离烦恼，舍弃一切快乐，就能快速到达彼岸。}%

@{若捨境界陰界入，及捨貪瞋愚癡過，}@
%{如果你舍弃境界，舍弃阴界入，以及舍弃贪瞋痴的过失，}%

@{棄諸煩惱修善業，以此能到檀彼岸。}@
%{舍弃一切烦恼，修习善业，就能到达彼岸。}%

@{當以慈悲念眾生，息諸分別不自是，}@
%{就应该以慈悲心念众生，熄灭一切分别，不执着自我，}%

@{常能憐愍諸眾生，則滿尸羅波羅蜜。}@
%{经常能够怜悯一切众生，就能圆满尸罗波罗蜜。}%

@{勤捨罪業修諸禪，亦當捨諸陰界入，}@
%{精勤舍弃罪业，修习一切禅定，也应该舍弃一切阴界入，}%

@{愛諸方便常求禪，除障到於精進岸。}@
%{喜爱一切方便，经常寻求禅定，消除障碍到达精进岸。}%

@{境界不動不味著，為捨因緣修悲喜，}@
%{境界不动，不贪着，为了舍弃因缘，修习悲喜，}%

@{一道清淨不移動，以此得滿忍辱度。}@
%{一道清净，不移动，以此就能圆满忍辱度。}%

@{於境界中不念慮，離疾不樂得於喜，}@
%{在境界中不思虑，远离疾苦，不贪求快乐，就能获得喜乐，}%

@{諸法離掉無分別，不染不愁是為捨。}@
%{一切法远离执着，没有分别，不染着，不忧愁，这就是舍度。}%

@{陰界如幻無起作，相續修行不斷絕，}@
%{阴界如同幻境，没有生起和造作，不断地修行，}%

@{善修了知如是法，以此得滿般若度。}@
%{善于修习，了解如是法，以此就能圆满般若度。}%

@{故我今告一切眾，若有欲除諸罪業，}@
%{所以我现在告诫一切众生，如果有想要消除一切罪业，}%

@{求忍三昧陀羅尼，當知如是住寂靜。}@
%{想要获得忍辱三昧陀罗尼，就应该知道如是住在寂静处。}%

@{若欲超越聲聞乘，及欲超越緣覺乘，}@
%{如果你想要超越声闻乘，以及想要超越缘觉乘，}%

@{又欲疾得勝佛地，應當速住阿蘭若。}@
%{并且想要快速获得胜佛地，就应该快速住在寂静处。}%

@{若心攝住阿蘭若，以此即是供諸佛，}@
%{如果你的心摄住在寂静处，以此就等于供养诸佛，}%

@{於是能捨一切罪，是則能滿於六度。}@
%{这样就能舍弃一切罪业，就能圆满六度。}%

@{當得作佛三有最，能轉清淨正法輪，}@
%{你就能成为三界中最殊胜的佛陀，能够转动清净的正法轮，}%

@{枯竭眾生諸惡趣，度脫眾生三有海。}@
%{枯竭众生一切恶趣，度脱众生三界之海。}%

@{當捨惡見諸緣事，常發最勝菩提心，}@
%{就应该舍弃一切错误的见解，一切缘起之事，经常发起最殊胜的菩提心，}%

@{應當速向蘭若處，於彼當成如是德。}@
%{就应该快速前往寂静处，在那里成就如是功德。}%

@{爾時，世尊說此經時，諸會大眾聞是甚深第一義禪，有於過去善修集者九萬二千人，得無生法忍；七十億那由他百千眾生，得種種三昧諸陀羅尼及無生忍；八萬一千人得授阿耨多羅三藐三菩提記；如恒河沙等眾生未發無上菩提心者，悉皆發心，於阿耨多羅三藐三菩提，得不退轉。}@
%{当时，佛陀说这部经的时候，所有聚集在一起的众生听到这个甚深的第一义禅，其中有九万二千人，他们在过去善于修集善根，获得了无生法忍；七十亿那由他百千众生，获得了各种三昧陀罗尼以及无生忍；八万一千人获得了阿耨多罗三藐三菩提的记别；如同恒河沙数的众生，还没有发起无上菩提心，都发起了菩提心，在阿耨多罗三藐三菩提中，获得了不退转。}%
