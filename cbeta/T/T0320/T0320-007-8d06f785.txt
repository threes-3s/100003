@{父子合集經卷第七}@
%{父子合集经卷第七}%

@{西天譯經三藏朝散大夫試鴻臚卿宣梵大師賜紫沙門臣日稱等奉詔譯}@
%{西天译经三藏朝散大夫试鸿胪卿宣梵大师赐紫沙门臣日称等奉诏译}%

@{緊那羅王授記品第十二}@
%{紧那罗王授记品第十二}%

@{爾時會中，有八俱胝緊那羅王，其上首者名曰大樹，見諸阿修羅王乃至諸藥叉眾，咸於佛所廣作供養，復聞如來與授記別，心生驚疑，歎未曾有，謂：若佛說一切世間諸蘊、界、處，但有假名，無有少分令眾生見，若佛菩薩亦不可見。云何世尊於大眾中與諸天龍次第授記，復令化度無量有情，從一佛剎至一佛剎，奉覲如來經河沙劫，廣修諸行得成佛道，具諸相好，國土莊嚴，壽命劫數正像法等，乃至緣畢，入般涅盤？既言諸法皆悉空寂，何能建立授記之事？}@
%{当时会上，有八俱胝紧那罗王，其上首者名叫大树，看见诸阿修罗王乃至诸药叉众，都于佛所广作供养，又听到如来与授记别，心中生起惊疑，感叹从未有过，说：如果佛说一切世间诸蕴、界、处，只有假名，没有少分令众生见，如果佛菩萨也不可见。那么世尊在大众中与诸天龙次第授记，又令化度无量有情，从一佛剎至一佛剎，奉觐如来经河沙劫，广修诸行得成佛道，具诸相好，国土庄严，寿命劫数正像法等，乃至缘毕，入般涅槃？既然说诸法皆悉空寂，怎么能建立授记之事？}%

@{時大樹緊那羅王生此疑心未能決了，合掌向佛，以偈問曰：}@
%{这时大树紧那罗王生起此疑心未能决了，合掌向佛，以偈问曰：}%

@{我聞如來作是說，於佛妙慧心有疑。}@
%{我听到如来作是说，对佛妙慧心中有疑。}%

@{既言空已復授記，此義甚深不能了。}@
%{既然说空了又授记，此义甚深不能了。}%

@{又云法界本空寂，而說如月現於水。}@
%{又说法界本空寂，而说如月现于水。}%

@{既言空寂復現形，此義甚深不能了。}@
%{既然说空寂又现形，此义甚深不能了。}%

@{云何諸法本無生，復說樂求菩提者？}@
%{云何诸法本无生，又说乐求菩提者？}%

@{既言無生何有求，此義甚深不能了。}@
%{既然说无生何有求，此义甚深不能了。}%

@{云何諸法本無盡，復說如來有滅度？}@
%{云何诸法本无尽，又说如来有灭度？}%

@{我今請問牟尼尊，此義甚深不能了。}@
%{我今请问牟尼尊，此义甚深不能了。}%

@{云何諸法皆如幻，復說此歿得生天？}@
%{云何诸法皆如幻，又说此歿得生天？}%

@{既言如幻何有生，此義甚深不能了。}@
%{既然说如幻何有生，此义甚深不能了。}%

@{云何諸法無所依，復說依止善知識？}@
%{云何诸法无所依，又说依止善知识？}%

@{既言無依何假人，此義甚深不能了。}@
%{既然说无依何假人，此义甚深不能了。}%

@{云何世間無作者，復說有能斷諸惑？}@
%{云何世间无作者，又说有能断诸惑？}%

@{既言無作斷無由，此義甚深不能了。}@
%{既然说无作断无由，此义甚深不能了。}%

@{云何諸法自性空，復說觀空得解脫？}@
%{云何诸法自性空，又说观空得解脱？}%

@{既言空已復何觀，此義甚深不能了。}@
%{既然说空了又何观，此义甚深不能了。}%

@{云何諸法剎那滅，復說諸法常不滅？}@
%{云何诸法剎那灭，又说诸法常不灭？}%

@{既言遷變或凝常，此義甚深不能了。}@
%{既然说迁变或凝常，此义甚深不能了。}%

@{云何福業非積聚，而說修習成菩提？}@
%{云何福业非积聚，而说修习成菩提？}%

@{既言無集豈能成，此義甚深不能了。}@
%{既然说无集岂能成，此义甚深不能了。}%

@{云何諸法無所說，復說毀謗墮惡道？}@
%{云何诸法无所说，又说毁谤堕恶道？}%

@{既言無說謗何生，此義甚深不能了。}@
%{既然说无说谤何生，此义甚深不能了。}%

@{無能勝者所宣說，種種差別難曉了，}@
%{无能胜者所宣说，种种差别难晓了，}%

@{佛眼清淨照世間，普施群生甘露味。}@
%{佛眼清净照世间，普施群生甘露味。}%

@{餘無有能為我等，決擇如上所疑事，}@
%{余无有能为我等，决择如上所疑事，}%

@{唯有如來能斷除，故我頂禮一切智。}@
%{唯有如来能断除，故我顶礼一切智。}%

@{爾時世尊，為彼大樹緊那羅王釋諸疑故，以偈答曰：}@
%{尔时世尊，为彼大树紧那罗王释诸疑故，以偈答曰：}%

@{緊那羅王汝所問，既空依何而授記？}@
%{紧那罗王汝所问，既然空了依何而授记？}%

@{由彼達法性本空，故得如來與授記。}@
%{由彼达法性本空，故得如来与授记。}%

@{以何因緣作是說？諸法若有實性者，}@
%{以何因缘作是说？诸法若有实性者，}%

@{則應不減亦不增，於一切時無改轉。}@
%{则应不减亦不增，于一切时无改转。}%

@{譬如圓鑑挂空中，假明緣故現色像，}@
%{譬如圆鉴挂空中，假明缘故现色像，}%

@{緊那羅王汝當知，自性清淨本如是，}@
%{紧那罗王汝当知，自性清净本如是，}%

@{法性無垢常湛然，亦無動亂及變異。}@
%{法性无垢常湛然，亦无动乱及变异。}%

@{汝觀供養福業因，於何法中有其相？}@
%{汝观供养福业因，于何法中有其相？}%

@{應知法界本寂靜，智者觀法不可得。}@
%{应知法界本寂静，智者观法不可得。}%

@{諸有取相凡夫人，於此疑惑不能了。}@
%{诸有取相凡夫人，于此疑惑不能了。}%

@{汝問諸法本無生，復說發心求佛道？}@
%{汝问诸法本无生，又说发心求佛道？}%

@{緊那羅王汝當知，十力妙智難思議。}@
%{紧那罗王汝当知，十力妙智难思议。}%

@{凡愚沒溺三界中，於諸欲境生耽著，}@
%{凡愚没溺三界中，于诸欲境生耽著，}%

@{由是常生顛倒心，故受生死諸苦惱。}@
%{由是常生颠倒心，故受生死诸苦恼。}%

@{從昔未聞於正法，設得聞已而不解，}@
%{从昔未闻于正法，设得闻已而不解，}%

@{使令安住實智中，漸能求趣菩提果。}@
%{使令安住实智中，渐能求趣菩提果。}%

@{汝問諸法本無滅，何故如來有滅度？}@
%{汝问诸法本无灭，何故如来有灭度？}%

@{為破常見諸眾生，無有少法為常者。}@
%{为破常见诸众生，无有少法为常者。}%

@{汝問諸法猶如幻，見生天者故懷疑？}@
%{汝问诸法犹如幻，见生天者故怀疑？}%

@{為化凡愚諸眾生，恃己所有生憍慢。}@
%{为化凡愚诸众生，恃己所有生憍慢。}%

@{汝問諸法無所依，而見依止善知識？}@
%{汝问诸法无所依，而见依止善知识？}%

@{若非依止說法師，畢竟世間無出離。}@
%{若非依止说法师，毕竟世间无出离。}%

@{汝問諸法無主宰，而見世間有作者？}@
%{汝问诸法无主宰，而见世间有作者？}%

@{當觀車由眾分成，有於運載作用事。}@
%{当观车由众分成，有于运载作用事。}%

@{為化著我諸眾生，及彼執著我所者，}@
%{为化著我诸众生，及彼执著我所者，}%

@{我為彼說無堪任，是故但唯假名想。}@
%{我为彼说无堪任，是故但唯假名想。}%

@{汝問諸法自性空，復說觀空得解脫？}@
%{汝问诸法自性空，又说观空得解脱？}%

@{由彼分別執著心，故於空義不能了。}@
%{由彼分别执著心，故于空义不能了。}%

@{汝問諸行剎那滅，復云諸法性不滅？}@
%{汝问诸行剎那灭，又云诸法性不灭？}%

@{為破著欲諸眾生，於非淨處生淨想。}@
%{为破著欲诸众生，于非净处生净想。}%

@{譬如遠矚於陽焰，渴者趣之而求水，}@
%{譬如远瞩于阳焰，渴者趣之而求水，}%

@{皆從妄想分別生，當知水體不可得。}@
%{皆从妄想分别生，当知水体不可得。}%

@{陽焰之處本無水，自性淨中本無染，}@
%{阳焰之处本无水，自性净中本无染，}%

@{由彼愚夫顛倒心，則為彼愛所纏縛。}@
%{由彼愚夫颠倒心，则为彼爱所缠缚。}%

@{汝問諸法無所說，亦云謗者墮惡道？}@
%{汝问诸法无所说，亦云谤者堕恶道？}%

@{愚夫聞已懷驚疑，智者聞之除障礙。}@
%{愚夫闻已怀惊疑，智者闻之除障碍。}%

@{世間諸法本空寂，無智妄計執為我，}@
%{世间诸法本空寂，无智妄计执为我，}%

@{彼等若聞諸法空，則生斷滅怖畏想。}@
%{彼等若闻诸法空，则生断灭怖畏想。}%

@{若人毀謗於空法，皆由執著我人相，}@
%{若人毁谤于空法，皆由执著我人相，}%

@{猶如繫縛於虛空，增長愚癡墮惡趣。}@
%{犹如系缚于虚空，增长愚癡堕恶趣。}%

@{或說善業往天中，或生人間受快樂，}@
%{或说善业往天中，或生人间受快乐，}%

@{作者非實業不亡，如夢中境唯心造。}@
%{作者非实业不亡，如梦中境唯心造。}%

@{當知諸趣皆如夢，由迷真故不覺知，}@
%{当知诸趣皆如梦，由迷真故不覺知，}%

@{夢中何有於去來，愚夫妄謂真實境，}@
%{梦中何有于去来，愚夫妄谓真实境，}%

@{我雖說有所作業，十方推求無作者。}@
%{我虽說有所作业，十方推求无作者。}%

@{譬如狂風吹樹枝，由相觸故生其火。}@
%{譬如狂风吹树枝，由相触故生其火。}%

@{彼風及木無思念，謂言我能出於火。}@
%{彼风及木无思念，谓言我能出於火。}%

@{兩相因故火發生，本無作者亦如此。}@
%{两相因故火发生，本无作者亦如此。}%

@{汝問福業無積聚，云何能得菩提果？}@
%{汝问福业无积聚，云何能得菩提果？}%

@{如世間有百歲人，雖經時分年無聚。}@
%{如世间有百岁人，虽经时分年无聚。}%

@{汝問諸法不可盡，云何復說業可盡？}@
%{汝问诸法不可尽，云何复说业可尽？}%

@{觀空理者說無窮，隨世俗中則有盡。}@
%{观空理者说无穷，随世俗中则有尽。}%

@{我雖演說真實際，由顛倒故於外求，}@
%{我虽演说真实际，由颠倒故于外求，}%

@{眾生業惑障緣深，不能了此勝義法。}@
%{众生业惑障缘深，不能了此胜义法。}%

@{緊那羅王汝當知，於此勝義善安住，}@
%{紧那罗王汝当知，于此胜义善安住，}%

@{一切諸相皆一相，善通達者謂無相。}@
%{一切诸相皆一相，善通达者谓无相。}%

@{一切菩薩之所行，一切諸法皆無作，}@
%{一切菩萨之所行，一切诸法皆无作，}%

@{所謂阿字總持門，由此入解一切法。}@
%{所谓阿字总持门，由此入解一切法。}%

@{緊那羅王汝當知，一切諸法本寂靜，}@
%{紧那罗王汝当知，一切诸法本寂静，}%

@{此說無相總持門，由阿字門而入解。}@
%{此说无相总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法無分別，}@
%{紧那罗王汝当知，一切诸法无分别，}%

@{此說平等總持門，由阿字門而入解。}@
%{此说平等总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法無自性，}@
%{紧那罗王汝当知，一切诸法无自性，}%

@{此說清淨總持門，由阿字門而入解。}@
%{此说清净总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法無有盡，}@
%{紧那罗王汝当知，一切诸法无有尽，}%

@{此說離障總持門，由阿字門而入解。}@
%{此说离障总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法非思議，}@
%{紧那罗王汝当知，一切诸法非思議，}%

@{此說實相總持門，由阿字門而入解。}@
%{此说实相总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法無所趣，}@
%{紧那罗王汝当知，一切诸法无所趣，}%

@{此說解脫總持門，由阿字門而入解。}@
%{此说解脱总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法本無動，}@
%{紧那罗王汝当知，一切诸法本无动，}%

@{此說證淨總持門，由阿字門而入解。}@
%{此说证净总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法假名說，}@
%{紧那罗王汝当知，一切诸法假名说，}%

@{此說真實總持門，由阿字門而入解。}@
%{此说真实总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法不可得，}@
%{紧那罗王汝当知，一切诸法不可得，}%

@{此說離相總持門，由阿字門而入解。}@
%{此说离相总持门，由阿字门而入解。}%

@{緊那羅王汝當知，一切諸法離思念，}@
%{紧那罗王汝当知，一切诸法离思念，}%

@{此說靜慮總持門，由阿字門而入解。}@
%{此说静虑总持门，由阿字门而入解。}%

@{此法非見非對治，求彼二相無所有，}@
%{此法非见非对治，求彼二相无所有，}%

@{了知無相亦無名，則能具足菩提道。}@
%{了知无相亦无名，则能具足菩提道。}%

@{若法可見可對治，是法非實非平等，}@
%{若法可见可对治，是法非实非平等，}%

@{彼法自性離言詮，譬如虛空無相似。}@
%{彼法自性离言诠，譬如虚空无相似。}%

@{法性無有種種相，亦非一相及異相，}@
%{法性无有种种相，亦非一相及异相，}%

@{非有冷相及熱相，如鏡中像不可得；}@
%{非有冷相及热相，如镜中像不可得；}%

@{非有曲相及直相，亦復無有明暗相，}@
%{非有曲相及直相，亦复无有明暗相，}%

@{遠離男相及女相，此名入解真實相。}@
%{远离男相及女相，此名入解真实相。}%

@{非諂非誑非動亂，無有舒卷微細相，}@
%{非谄非诳非动乱，无有舒卷微细相，}%

@{亦無忿怒忻喜相，非怯弱相起盡相，}@
%{亦无忿怒忻喜相，非怯弱相起尽相，}%

@{非有入相及出相，亦無進相及退相，}@
%{非有入相及出相，亦无进相及退相，}%

@{非其寤相及寐相，復無去來現在相，}@
%{非其寤相及寐相，复无去来现在相，}%

@{非是眼相及瞻視，亦非盲瞽與瘴瞖，}@
%{非是眼相及瞻视，亦非盲瞽与瘴瞖，}%

@{非其調順非剛強，非有覆藏及顯露，}@
%{非其调顺非刚强，非有覆藏及显露，}%

@{非是動轉及止息，離諸戲論常湛然。}@
%{非是动转及止息，离诸戏论常湛然。}%

@{智者當作如是觀，此名了知佛境界，}@
%{智者当作如是观，此名了知佛境界，}%

@{為欲調伏世間故，於離言法強分別，}@
%{为欲调伏世间故，于离言法强分别，}%

@{於此勝義善知了，則能通達一切法。}@
%{于此胜义善知了，则能通达一切法。}%
@{爾時大樹緊那羅王，聞佛世尊釋諸疑已，及聞演說總持字門，深生入解，心大歡喜得未曾有。}@
%{当时大树紧那罗王，听到佛世尊解除了他的疑惑，并且听到佛世尊讲解总持字门，他深切地理解了，心里非常欢喜，从未有过这样的喜悦。}%

@{以神通力化八俱胝殊勝樓閣，巧妙奇絕，或住山峯，或依林木，或化現於蓮華之上，是諸重閣皆以七寶合成花蓋而懸其上，一一皆有眾寶幢幡、寶幔花鬘次第嚴飾。}@
%{他运用神通力，化现出八俱胝个殊胜的楼阁，这些楼阁巧妙绝伦，有的建在山峰上，有的依附在树林中，有的化现于莲花之上，这些重阁都用七宝合成华盖悬挂在上面，每个楼阁上都有各种宝幢、宝幡、宝幔、花鬘，依次装饰着。}%

@{是八俱胝緊那羅王，各各陞彼樓閣之上，於虛空中右繞三匝，復以上妙水陸之花而散佛上。}@
%{这八俱胝个紧那罗王，各自登上那些楼阁，在虚空中右绕三匝，然后用最美好的水陆之花，洒落在佛陀身上。}%

@{散已即從重閣而下，至於佛所，復繞三匝，瞻仰尊顏，目不暫捨，一心合掌，住立一面，思念如來過去現在積集無量最勝功德。}@
%{洒完花后，他们就从重阁上下来，来到佛陀面前，再次绕佛三匝，仰望着佛陀的容颜，眼睛一刻也不离开，一心合掌，站在一边，思念着如来过去现在所积集的无量最胜功德。}%

@{佛知是已，口放淨光。}@
%{佛陀知道他们的心意，就口放净光。}%

@{時尊者馬勝比丘覩是相已，頭面禮佛，以偈問曰：}@
%{这时，尊者马胜比丘看见了这一幕，就头面礼拜佛陀，用偈颂问道：}%

@{如來今者放光明，淨妙難思甚希有，}@
%{如来现在放出光明，洁净美好，难以思量，非常稀有，}%

@{時眾諦視於慈容，猶若披雲瞻滿月。}@
%{当时大众都凝视着佛陀慈祥的面容，就像披着云彩仰望满月一样。}%

@{何人今於世尊所，樂聞最上微妙法，}@
%{现在是谁在世尊面前，喜欢听闻最上微妙法，}%

@{以真實智善了知，放此光明垂印可？}@
%{用真实的智慧，善于了解，放出这种光明，印证什么呢？}%

@{誰人初發菩提心，感佛加持現斯瑞？}@
%{是谁最初发起了菩提心，感动了佛陀加持，显现出这种瑞相呢？}%

@{唯願說此光因緣，制諸外道令生信。}@
%{唯愿您说一说这种光明的因缘，让那些外道生起信心。}%

@{是時諸來大集會，咸各歡喜發淨心，}@
%{当时，所有来参加集会的人，都欢喜地发起了清净心，}%

@{冀聞清淨甘露音，聞已奉行於佛教。}@
%{希望听到清净甘露般的音声，听到后就奉行佛法。}%

@{爾時世尊為尊者馬勝比丘說伽陀曰：}@
%{这时，世尊为尊者马胜比丘说了一段伽陀：}%

@{具壽馬勝問斯義，當一心聽無異緣，}@
%{具寿的马胜，你问这个道理，就应该一心听，没有别的缘由，}%

@{我今為彼緊那羅，授記未來成佛事。}@
%{我现在为那些紧那罗，授记他们未来成佛的事情。}%

@{彼所請問極難解，皆為利樂諸世間，}@
%{他们所问的问题非常难解，都是为了利益世间，}%

@{我今為斷彼疑心，令於佛法善安住。}@
%{我现在为了消除他们的疑惑，让他们安住在佛法中。}%

@{此大樹緊那羅王，與八俱胝徒屬眾，}@
%{这个大树紧那罗王，以及八俱胝个属下，}%

@{由興供養淨福因，命終當得生天上。}@
%{由于他们兴起供养，积累了清净的福报，命终后，应该会生到天上。}%

@{展轉天中受妙樂，滿足九十俱胝歲，}@
%{他们在天空中转生，享受美妙的快乐，满足九十俱胝岁，}%

@{獲得如意五神通，親近承事河沙佛。}@
%{获得如意五神通，亲近承事无数如河沙般的佛陀。}%

@{經歷那由他剎土，常說勝義第一法，}@
%{他们经历无数个佛土，常常宣说胜义第一法，}%

@{化度那由他有情，得成無上菩提道。}@
%{度化无数个有情众生，获得无上菩提道。}%

@{號曰無量光如來，劫名國土皆無異，}@
%{他们被称为无量光如来，劫名和国土都与佛陀相同，}%

@{人中師子十力尊，具足無邊大智慧。}@
%{是人中狮子，具有十力，具足无边的智慧。}%

@{彼國所有諸菩薩，修習善根悉圓滿，}@
%{他们的国土里，所有的菩萨，修习善根都圆满了，}%

@{亦無求趣二乘人，皆是一生居補處。}@
%{也没有追求二乘果位的，都是一生居补处。}%

@{如是彼諸菩薩眾，各懷悲願利群生，}@
%{这些菩萨们，各自怀着悲愿，利益众生，}%

@{能與世間作照明，後當次第得作佛。}@
%{能够为世间作光明，将来会次第地成佛。}%

@{彼諸國土勝莊嚴，離諸穢惡無煩惱，}@
%{他们的国土非常庄严，远离一切污秽，没有烦恼，}%

@{猶如覩史多天宮，受用清淨知止足。}@
%{就像看到兜率天宫一样，享受清净，知道满足。}%

@{一切過患八無暇，乃至名字未嘗聞，}@
%{一切过患，八无暇，乃至名字都没有听说过，}%

@{有情安隱住其中，常飡法味禪悅樂。}@
%{有情众生安稳地住在这里，常常吃法味，禅悦快乐。}%

@{大哉善逝天中天，與授緊那羅佛記，}@
%{伟大的善逝，天中之天，给紧那罗授记成佛，}%

@{時眾聞已心泰然，稽首致禮無上士。}@
%{当时大众听到后，内心安然，稽首礼拜无上士。}%

@{父子合集經寶瓔珞天授記品第十三}@
%{父子合集经宝璎珞天授记品第十三}%

@{爾時會中有八俱胝寶瓔珞天子，見諸阿修羅乃至緊那羅等於佛會所作大供養，及聞如來與彼授記，心甚歡喜得未曾有。}@
%{当时，集会中有八俱胝宝璎珞天子，看见阿修罗乃至紧那罗等，在佛会中作大供养，并且听到如来给他们授记，心里非常欢喜，从未有过这样的喜悦。}%

@{時諸天子為供佛故，發勇猛心，繞迦毘羅城周匝六十踰繕那量，於虛空中雨曼陀羅花，繽紛而下，積至于膝，作供養已，繞佛三匝，住在一面合掌向佛，以偈讚曰：}@
%{这些天子为了供养佛陀，发起了勇猛心，绕着迦毗罗城，周匝六十踰缮那量，在虚空中降下曼陀罗花，花瓣纷纷落下，积到膝盖，作完供养后，绕佛三匝，站在一边，合掌向佛，用偈颂赞叹道：}%

@{具足大悲十力尊，能救一切眾生者，}@
%{具足大悲，具有十力，能够救度一切众生，}%

@{諸根寂靜放光明，故我敬禮聖中聖。}@
%{您的六根寂静，放出光明，所以，我敬礼您，圣中之圣。}%

@{諸天及龍人非人，於佛法中生愛樂，}@
%{诸天、龙、人、非人，在佛法中生起爱乐，}%

@{最上牟尼大導師，是故我今歸命禮。}@
%{您是最上的牟尼，伟大的导师，所以，我今天归命礼拜您。}%

@{如來出現於世間，譬若眾星中滿月，}@
%{如来出现在世间，就像众星中的满月一样，}%

@{無邊福智莊嚴身，眾生觀者心無厭。}@
%{您的福德智慧，庄严您的身躯，众生观者，心中没有厌倦。}%

@{具足種種清淨法，六通三明四無畏，}@
%{您具足各种清净法，六通、三明、四无畏，}%

@{十八不共法難量，是故我今歸命禮。}@
%{十八不共法，难以计数，所以，我今天归命礼拜您。}%

@{三十二相皆圓滿，八十隨好妙嚴身，}@
%{您拥有三十二相，全部圆满，八十种随好，妙严您的身躯，}%

@{巍巍猶若天帝幢，是故我今歸命禮。}@
%{巍巍然，就像天帝的幢幡一样，所以，我今天归命礼拜您。}%

@{成就最勝三摩地，能斷輪迴諸結縛，}@
%{您成就了最胜的三摩地，能够断除轮回的各种束缚，}%

@{降伏無數億魔軍，湛然安住心無動。}@
%{降伏了无数亿的魔军，湛然安住，心不动摇。}%

@{如來智慧力堅固，常說四諦真實法，}@
%{如来的智慧力量坚固，常常宣说四谛真实法，}%

@{廣能化利諸人天，咸令超越於彼岸。}@
%{广泛地教化利益人天，使他们都超越到彼岸。}%

@{人中最勝釋師子，破諸邪見及異論，}@
%{您是人中之王，释迦狮子，破除一切邪见和异论，}%

@{願我當得如世尊，說法使離諸癡暗。}@
%{愿我能够像世尊一样，宣说佛法，使众生远离愚痴黑暗。}%

@{超出世間成正覺，常興四種無量心，}@
%{超越世间，成就正觉，常常兴起四种无量心，}%

@{憐愍攝受諸眾生，同證菩提無上道。}@
%{怜悯摄受一切众生，同证无上菩提道。}%
@{爾時世尊知諸天子深心所念，即於口中放大光明。}@
%{当时佛陀知道这些天子的内心所想，就从口中放射出光芒。}%

@{時尊者馬勝比丘覩是相已，合掌向佛，以偈問曰：}@
%{这时，马胜比丘看到这个景象，双手合十向佛陀问道：}%

@{牟尼現是希有相，時眾矚已心忻悅，}@
%{释迦牟尼佛展现出如此稀有的景象，当时所有的人都目不转睛地看着，心中充满了喜悦。}%

@{各各諦奉大慈顏，願說放光之所以。}@
%{每个人都恭敬地仰望着佛陀慈悲的面容，希望佛陀能解释放光的原因。}%

@{如來今放此淨光，必興廣大利益事，}@
%{如来现在放出这道净光，一定是为了兴起广大利益的事情。}%

@{我等頂禮無上尊，唯冀洪音為開示。}@
%{我们顶礼无上的尊者，只希望您能用洪亮的声音为我们开示。}%

@{爾時世尊為馬勝比丘說伽陀曰：}@
%{这时，佛陀为马胜比丘讲了以下的偈颂：}%

@{善哉馬勝問斯義，今當為汝分別說，}@
%{马胜问得好啊！我现在就为你详细解释。}%

@{我為利樂諸人天，與瓔珞天授佛記。}@
%{我为了利益所有的人间和天人，授予了瓔珞天佛的记号。}%

@{如來若不因請說，無由記別弟子事，}@
%{如来如果没有人请求，就不会特意去记录弟子的事情。}%

@{答汝問故息群疑，彼離疑則心安住。}@
%{我回答你的问题，是为了消除大家的疑惑，他们消除疑惑后，内心就能安住。}%

@{寶瓔珞天於我所，以淨意樂伸供養，}@
%{宝瓔珞天对我怀着清净的意愿，乐意供养我。}%

@{從此展轉生諸天，百千天子常圍繞。}@
%{从那时起，他转生到各个天界，身边常有上千的天子围绕着他。}%

@{復持天上妙香花，供養無量俱胝佛，}@
%{他还拿着天上的妙香花，供养无数的佛陀。}%

@{於彼佛所發淨心，愛樂修行菩薩道。}@
%{他对那些佛陀生起清净的内心，爱乐地修行菩萨道。}%

@{如是精進供養已，復於未來作佛事，}@
%{这样精进地供养之后，他将在未来成为佛陀。}%

@{奉覲那由他世尊，常說伽陀讚佛德。}@
%{他将觐见无数的佛陀，经常用偈颂赞叹佛陀的功德。}%

@{如理廣修清淨行，得那羅延堅固身，}@
%{他按照道理广修清净的行为，获得了那罗延坚固的身躯。}%

@{於其歡喜劫中生，具足圓成一切智。}@
%{他将在欢乐劫中出生，圆满成就一切智。}%

@{號曰華幢勝如來，有大名稱無與等，}@
%{他的名字叫“华幢胜如来”，有着无与伦比的伟大名号。}%

@{并餘天子八俱胝，同一劫中皆作佛。}@
%{还有其他八俱胝的天子，都在同一劫中成佛。}%

@{彼佛土中無地獄，亦無鬼趣與傍生，}@
%{他们的佛土中没有地狱，也没有鬼道和畜生道。}%

@{及阿修羅憍慢心，離於八難諸危苦。}@
%{也没有阿修罗的傲慢之心，远离了八难和一切苦难。}%

@{彼諸天子成佛剎，悉如三十三天中，}@
%{这些天子成佛的国土，都像三十三天一样。}%

@{國土富樂遍莊嚴，壽量長遠皆相似。}@
%{国土富饶快乐，遍地庄严，寿命长远，都差不多。}%

@{彼土既無惡趣名，況復有作惡業者，}@
%{那个国土没有恶趣的名字，更何况有造恶业的人呢？}%

@{眾生各各隨法行，皆住無諍清淨行。}@
%{众生都按照法行，都住着无争的清净行。}%

@{所化眾生無有邊，過彼恒河沙數量，}@
%{他们所教化的众生无边无际，超过恒河沙的数量。}%

@{彼佛出現世間時，常說勝義第一法。}@
%{当那些佛陀出现世间的时候，他们经常宣说胜义第一法。}%

@{如是化緣歸寂已，廣布舍利起塔廟，}@
%{这样，他们化缘归于寂灭之后，广泛地布舍利，建造塔庙。}%

%{每一颗舍利都胜妙难思，其中都包含着佛陀的身相。}%

@{咸各稱讚佛功德，住虛空中現神變，}@
%{他们都赞叹佛陀的功德，住在虚空之中，展现神通变化。}%

@{教化無量諸眾生，皆發無上菩提意。}@
%{他们教化无数的众生，都让众生发起了无上菩提心。}%

@{十力牟尼方便力，為彼天子授道記，}@

@{時諸大眾聞佛言，住正思惟心寂靜。}@
%{当时，所有的听众都听到佛陀的话语，住着正思惟，内心寂静。}%
